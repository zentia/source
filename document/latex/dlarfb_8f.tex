{\bfseries{DLARFB}} 

 \begin{DoxyParagraph}{Purpose\+:}
\begin{DoxyVerb}!>
!> DLARFB applies a real block reflector H or its transpose H**T to a
!> real m by n matrix C, from either the left or the right.
!> \end{DoxyVerb}
 
\end{DoxyParagraph}

\begin{DoxyParams}[1]{参数}
\mbox{\texttt{ in}}  & {\em SIDE} & \begin{DoxyVerb}!>          SIDE is CHARACTER*1
!>          = 'L': apply H or H**T from the Left
!>          = 'R': apply H or H**T from the Right
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em TRANS} & \begin{DoxyVerb}!>          TRANS is CHARACTER*1
!>          = 'N': apply H (No transpose)
!>          = 'T': apply H**T (Transpose)
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em DIRECT} & \begin{DoxyVerb}!>          DIRECT is CHARACTER*1
!>          Indicates how H is formed from a product of elementary
!>          reflectors
!>          = 'F': H = H(1) H(2) . . . H(k) (Forward)
!>          = 'B': H = H(k) . . . H(2) H(1) (Backward)
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em STOREV} & \begin{DoxyVerb}!>          STOREV is CHARACTER*1
!>          Indicates how the vectors which define the elementary
!>          reflectors are stored:
!>          = 'C': Columnwise
!>          = 'R': Rowwise
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em M} & \begin{DoxyVerb}!>          M is INTEGER
!>          The number of rows of the matrix C.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em N} & \begin{DoxyVerb}!>          N is INTEGER
!>          The number of columns of the matrix C.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em K} & \begin{DoxyVerb}!>          K is INTEGER
!>          The order of the matrix T (= the number of elementary
!>          reflectors whose product defines the block reflector).
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em V} & \begin{DoxyVerb}!>          V is DOUBLE PRECISION array, dimension
!>                                (LDV,K) if STOREV = 'C'
!>                                (LDV,M) if STOREV = 'R' and SIDE = 'L'
!>                                (LDV,N) if STOREV = 'R' and SIDE = 'R'
!>          The matrix V. See Further Details.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em LDV} & \begin{DoxyVerb}!>          LDV is INTEGER
!>          The leading dimension of the array V.
!>          If STOREV = 'C' and SIDE = 'L', LDV >= max(1,M);
!>          if STOREV = 'C' and SIDE = 'R', LDV >= max(1,N);
!>          if STOREV = 'R', LDV >= K.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em T} & \begin{DoxyVerb}!>          T is DOUBLE PRECISION array, dimension (LDT,K)
!>          The triangular k by k matrix T in the representation of the
!>          block reflector.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em LDT} & \begin{DoxyVerb}!>          LDT is INTEGER
!>          The leading dimension of the array T. LDT >= K.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in,out}}  & {\em C} & \begin{DoxyVerb}!>          C is DOUBLE PRECISION array, dimension (LDC,N)
!>          On entry, the m by n matrix C.
!>          On exit, C is overwritten by H*C or H**T*C or C*H or C*H**T.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em LDC} & \begin{DoxyVerb}!>          LDC is INTEGER
!>          The leading dimension of the array C. LDC >= max(1,M).
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ out}}  & {\em WORK} & \begin{DoxyVerb}!>          WORK is DOUBLE PRECISION array, dimension (LDWORK,K)
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em LDWORK} & \begin{DoxyVerb}!>          LDWORK is INTEGER
!>          The leading dimension of the array WORK.
!>          If SIDE = 'L', LDWORK >= max(1,N);
!>          if SIDE = 'R', LDWORK >= max(1,M).
!> \end{DoxyVerb}

\begin{DoxyParagraph}{Further Details\+:}
\begin{DoxyVerb}!>
!>  The shape of the matrix V and the storage of the vectors which define
!>  the H(i) is best illustrated by the following example with n = 5 and
!>  k = 3. The elements equal to 1 are not stored; the corresponding
!>  array elements are modified but restored on exit. The rest of the
!>  array is not used.
!>
!>  DIRECT = 'F' and STOREV = 'C':         DIRECT = 'F' and STOREV = 'R':
!>
!>               V = (  1       )                 V = (  1 v1 v1 v1 v1 )
!>                   ( v1  1    )                     (     1 v2 v2 v2 )
!>                   ( v1 v2  1 )                     (        1 v3 v3 )
!>                   ( v1 v2 v3 )
!>                   ( v1 v2 v3 )
!>
!>  DIRECT = 'B' and STOREV = 'C':         DIRECT = 'B' and STOREV = 'R':
!>
!>               V = ( v1 v2 v3 )                 V = ( v1 v1  1       )
!>                   ( v1 v2 v3 )                     ( v2 v2 v2  1    )
!>                   (  1 v2 v3 )                     ( v3 v3 v3 v3  1 )
!>                   (     1 v3 )
!>                   (        1 )
!> \end{DoxyVerb}
