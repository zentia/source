{\bfseries{SLARF}} 

 \begin{DoxyParagraph}{Purpose\+:}
\begin{DoxyVerb}!>
!> SLARF applies a real elementary reflector H to a real m by n matrix
!> C, from either the left or the right. H is represented in the form
!>
!>       H = I - tau * v * v**T
!>
!> where tau is a real scalar and v is a real vector.
!>
!> If tau = 0, then H is taken to be the unit matrix.
!> \end{DoxyVerb}
 
\end{DoxyParagraph}

\begin{DoxyParams}[1]{参数}
\mbox{\texttt{ in}}  & {\em SIDE} & \begin{DoxyVerb}!>          SIDE is CHARACTER*1
!>          = 'L': form  H * C
!>          = 'R': form  C * H
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em M} & \begin{DoxyVerb}!>          M is INTEGER
!>          The number of rows of the matrix C.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em N} & \begin{DoxyVerb}!>          N is INTEGER
!>          The number of columns of the matrix C.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em V} & \begin{DoxyVerb}!>          V is REAL array, dimension
!>                     (1 + (M-1)*abs(INCV)) if SIDE = 'L'
!>                  or (1 + (N-1)*abs(INCV)) if SIDE = 'R'
!>          The vector v in the representation of H. V is not used if
!>          TAU = 0.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em INCV} & \begin{DoxyVerb}!>          INCV is INTEGER
!>          The increment between elements of v. INCV <> 0.
!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ in}}  & {\em TAU} & \begin{DoxyVerb}!>          TAU is REAL

\\
\hline

!>          On entry, the m by n matrix C.

!>          or C * H if SIDE = 'R'.
!> \end{DoxyVerb}

\hline
\mbox{\texttt{ in}}  & {\em LDC} & \begin{DoxyVerb}!>          LDC is INTEGER

!> \end{DoxyVerb}
\\
\hline
\mbox{\texttt{ out}}  & {\em WORK} & \begin{DoxyVerb}!>          WORK is REAL array, dimension

!> \end{DoxyVerb}
