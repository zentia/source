43/*
44 * If NO_DEFAULT_MAPPING is not defined (and if FF_MAPPING_STRING is equal
45 * to ""), the FastFlow library pins each spawned thread to a core context
46 * according to a simple policy: thread 0 to core 0, thread 1
47 * to core 1, and so on restarting from the beginning if there are more
48 * threads than core contexts (which is usually not a good idea if
49 * performance matters).
50 * Depending on the OS numbering of core contexts, core 0 and core 1 may be
51 * "far away" one each other, e.g., they could be on two distinct CPUs,
52 * thus not sharing any level of cache.
53 * Therefore, to control the mapping of the thread when the OS numbering
54 * is a bit mess (e.g. Power8, AMD machines, some Intel Xeon), you have
55 * two options:
56 * 1. to use the Mammut library, which provides a layer for discovering
57 * core contexts
58 * 2. to set the FF_MAPPING_STRING in this file.
59 *
60 * For case 2, the simplest option is to run the Bash script
61 * 'mapping_string.sh', which returns a suitable string that can be
62 * copy-paste in the FF_MAPPING_STRING preprocessor variable.
63 * Note that, if you wish (and trust it) the script can modify the
64 * config.hpp file for you. The script also sets the FF_NUM_CORES and
65 * FF_NUM_REAL_CORES variables.
66 * Example:
67 * > ./mapping_string.sh
68 * > FF_MAPPING_STRING="0,2,1,3"
69 * > FF_NUM_CORES=4
70 * > FF_NUM_REAL_CORES=2
71 * > Do you want that I change the ./config.hpp file for you? (y/N) y
72 * > This is the new FF_MAPPING_STRING variable in the ./config.hpp file:
73 * > #if !defined MAPPING_STRING
74 * > #define FF_MAPPING_STRING "0,2,1,3"
75 * > #else
76 * > ...
77 *
78 */
79/*
80 * NOTE: if FF_MAPPING_STRING is "" (default), FastFlow executes a linear
81 * mapping of threads.
82 */
83#if !defined MAPPING_STRING
84#define FF_MAPPING_STRING ""
85#else
86#define FF_MAPPING_STRING MAPPING_STRING
87#endif
