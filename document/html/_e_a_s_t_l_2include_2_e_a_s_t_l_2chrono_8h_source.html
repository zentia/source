565 // Internal::GetTicks
567 inline uint64_t GetTicks()
568 {
569 #if defined EA_PLATFORM_MICROSOFT
570 auto queryFrequency = []
571 {
572 LARGE_INTEGER frequency;
573 QueryPerformanceFrequency(&frequency);
574 return double(1000000000.0L / (long double)frequency.QuadPart); // nanoseconds per tick
575 };
576
577 auto queryCounter = []
578 {
579 LARGE_INTEGER counter;
580 QueryPerformanceCounter(&counter);
581 return counter.QuadPart;
582 };
583
584 EA_DISABLE_VC_WARNING(4640) // warning C4640: construction of local static object is not thread-safe (VS2013)
585 static auto frequency = queryFrequency(); // cache cpu frequency on first call
586 EA_RESTORE_VC_WARNING()
587 return uint64_t(frequency * (double)queryCounter());
588 #elif defined EA_PLATFORM_SONY
589 static_assert(false, "Implementing GetTicks() requires first party support");
590 return 0;
591 #elif defined(EA_PLATFORM_APPLE)
592 auto queryTimeInfo = []
593 {
594 mach_timebase_info_data_t info;
595 mach_timebase_info(&info);
596 return info;
597 };
598
599 static auto timeInfo = queryTimeInfo();
600 uint64_t t = mach_absolute_time();
601 t *= timeInfo.numer;
602 t /= timeInfo.denom;
603 return t;
604 #elif defined(EA_PLATFORM_POSIX) // Posix means Linux, Unix, and Macintosh OSX, among others (including Linux-based mobile platforms).
605 #if (defined(CLOCK_REALTIME) || defined(CLOCK_MONOTONIC))
606 timespec ts;
607 int result = clock_gettime(CLOCK_MONOTONIC, &ts);
608
609 if (result == -1 && errno == EINVAL)
610 result = clock_gettime(CLOCK_REALTIME, &ts);
611
612 const uint64_t nNanoseconds = (uint64_t)ts.tv_nsec + ((uint64_t)ts.tv_sec * UINT64_C(1000000000));
613 return nNanoseconds;
614 #else
615 struct timeval tv;
616 gettimeofday(&tv, NULL);
617 const uint64_t nMicroseconds = (uint64_t)tv.tv_usec + ((uint64_t)tv.tv_sec * 1000000);
618 return nMicroseconds;
619 #endif
620 #else
621 #error "chrono not implemented for platform"
622 #endif
623 }
