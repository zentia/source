33 static ResultType read(const R& _r, const InputVarType& _var) noexcept {
34 static_assert(
35 internal::no_duplicate_field_names<rfl::Tuple<FieldTypes...>>(),
36 "Externally tagged variants cannot have duplicate field "

39 const auto to_result = [&](const auto _obj) -> ResultType {
40 auto field_variant = std::optional<Result<FieldVariantType>>();
41 const auto reader =
42 FieldVariantReader<R, W, ProcessorsType, FieldTypes...>(
43 &_r, &field_variant);
44 auto err = _r.read_object(reader, _obj);
45 if (err) {
46 return error(*err);
47 }
48 if (!field_variant) {
49 return error(
50 "Could not parse: Expected the object to have "
51 "exactly one field, but found more than one.");
52 }
53 return std::move(*field_variant);
54 };
55
56 return _r.to_object(_var).and_then(to_result);
57 }
