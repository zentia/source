143ssize_t readn(int fd, char *ptr, size_t n) {
144 size_t nleft = n;
145 ssize_t nread;
146
147 while (nleft > 0) {
148 if((nread = read(fd, ptr, nleft)) < 0) {
149 if (nleft == n) return -1; /* error, return -1 */
150 else break; /* error, return amount read so far */
151 } else if (nread == 0) break; /* EOF */
152 nleft -= nread;
153 ptr += nread;
154 }
155 return(n - nleft); /* return >= 0 */
156}

158ssize_t readvn(int fd, struct iovec *v, int count){
159 ssize_t rread;
160 for (int cur = 0;;) {
161 rread = readv(fd, v+cur, count-cur);
162 if (rread <= 0) return rread; // error or closed connection
163 while (cur < count && rread >= (ssize_t)v[cur].iov_len)
164 rread -= v[cur++].iov_len;
165 if (cur == count) return 1; // success!!
166 v[cur].iov_base = (char *)v[cur].iov_base + rread;
167 v[cur].iov_len -= rread;
168 }
169 return -1;
170}

172ssize_t writen(int fd, const char *ptr, size_t n) {
173 size_t nleft = n;

201static inline ssize_t recvnnb(int fd, char *buf, size_t size) {
202 size_t left = size;
203 int r;
204 while(left>0) {
205 if ((r=recv(fd ,buf,left,MSG_DONTWAIT)) == -1) {
206 if (errno == EINTR) continue;
207 if (left == size) return -1;
208 break;
209 }
210
211 if (r == 0) return 0; // EOF
212 left -= r;
213 buf += r;
214 }
215 return (size-left);
216}
