181 class thread_hp_storage {
182 public:
183 thread_hp_storage(guard *arr, size_t nSize) noexcept
184 : free_head_(arr), array_(arr), capacity_(nSize)
185# ifdef CDS_ENABLE_HPSTAT
186 , alloc_guard_count_(0)
187 , free_guard_count_(0)
188# endif
189 {
190 // Initialize guards
191 new( arr ) guard[nSize];
192
193 for ( guard *pEnd = arr + nSize - 1; arr < pEnd; ++arr )
194 arr->next_ = arr + 1;
195 arr->next_ = nullptr;
196 }
197
198 thread_hp_storage() = delete;
199
200 thread_hp_storage(thread_hp_storage const &) = delete;
201
202 thread_hp_storage(thread_hp_storage &&) = delete;
203
204 size_t capacity() const noexcept {
205 return capacity_;
206 }
207
208 bool full() const noexcept {
209 return free_head_ == nullptr;
210 }
211
212 guard *alloc() {
213# ifdef CDS_DISABLE_SMR_EXCEPTION
214 assert(!full());
215# else
216 if ( full() )
217 CDS_THROW_EXCEPTION(not_enough_hazard_ptr());
218# endif
219 guard *g = free_head_;
220 free_head_ = g->next_;
221 CDS_HPSTAT(++alloc_guard_count_);
222 return g;
223 }
224
225 void free(guard *g) noexcept {
226 assert(g >= array_ && g < array_ + capacity());
227
228 if ( g ) {
229 g->clear();
230 g->next_ = free_head_;
231 free_head_ = g;
232 CDS_HPSTAT(++free_guard_count_);
233 }
234 }
235
236 template<size_t Capacity>
237 size_t alloc(guard_array<Capacity> &arr) {
238 size_t i;
239 guard *g = free_head_;
240 for ( i = 0; i < Capacity && g; ++i ) {
241 arr.reset(i, g);
242 g = g->next_;
243 }
244
245# ifdef CDS_DISABLE_SMR_EXCEPTION
246 assert(i == Capacity);
247# else
248 if ( i != Capacity )
249 CDS_THROW_EXCEPTION(not_enough_hazard_ptr());
250# endif
251 free_head_ = g;
252 CDS_HPSTAT(alloc_guard_count_ += Capacity);
253 return i;
254 }
255
256 template<size_t Capacity>
257 void free(guard_array<Capacity> &arr) noexcept {
258 guard *gList = free_head_;
259 for ( size_t i = 0; i < Capacity; ++i ) {
260 guard *g = arr[i];
261 if ( g ) {
262 g->clear();
263 g->next_ = gList;
264 gList = g;
265 CDS_HPSTAT(++free_guard_count_);
266 }
267 }
268 free_head_ = gList;
269 }
270
271 // cppcheck-suppress functionConst
272 void clear() {
273 for ( guard *cur = array_, *last = array_ + capacity(); cur < last; ++cur )
274 cur->clear();
275 }
276
277 guard &operator[](size_t idx) {
278 assert(idx < capacity());
279
280 return array_[idx];
281 }
282
283 static size_t calc_array_size(size_t capacity) {
284 return sizeof(guard) * capacity;
285 }
286
287 guard *begin() const {
288 return array_;
289 }
290
291 guard *end() const {
292 return &array_[capacity_];
293 }
294
295 private:
296 guard *free_head_;
297 guard *const array_;
298 size_t const capacity_;
299# ifdef CDS_ENABLE_HPSTAT
300 public:
301 size_t alloc_guard_count_;
302 size_t free_guard_count_;
303# endif
304 };

324 size_t capacity() const noexcept {

327
328 size_t size() const noexcept {
329 return current_.load(atomics::memory_order_relaxed) - retired_;
330 }
331
332 bool push(retired_ptr &&p) noexcept {
333 retired_ptr *cur = current_.load(atomics::memory_order_relaxed);
334 *cur = p;
335 CDS_HPSTAT(++retire_call_count_);
336 current_.store(cur + 1, atomics::memory_order_relaxed);
337 return cur + 1 < last_;
338 }

342 }
343

349 current_.store(first() + nSize, atomics::memory_order_relaxed);

354 }

358 }
