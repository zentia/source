24int _mi_strnicmp(const char* s, const char* t, size_t n) {
25 if (n == 0) return 0;
26 for (; *s != 0 && *t != 0 && n > 0; s++, t++, n--) {
27 if (_mi_toupper(*s) != _mi_toupper(*t)) break;
28 }
29 return (n == 0 ? 0 : *s - *t);
30}

32void _mi_strlcpy(char* dest, const char* src, size_t dest_size) {
33 if (dest==NULL || src==NULL || dest_size == 0) return;
34 // copy until end of src, or when dest is (almost) full
35 while (*src != 0 && dest_size > 1) {

95static void mi_outc(char c, char** out, char* end) {
96 char* p = *out;
97 if (p >= end) return;
98 *p = c;
99 *out = p + 1;
100}

102static void mi_outs(const char* s, char** out, char* end) {
103 if (s == NULL) return;
104 char* p = *out;
105 while (*s != 0 && p < end) {
106 *p++ = *s++;

108 *out = p;

111static void mi_out_fill(char fill, size_t len, char** out, char* end) {
112 char* p = *out;

119static void mi_out_alignright(char fill, char* start, size_t len, size_t extra, char* end) {
120 if (len == 0 || extra == 0) return;
121 if (start + len + extra >= end) return;
122 // move `len` characters to the right (in reverse since it can overlap)
123 for (size_t i = 1; i <= len; i++) {
124 start[len + extra - i] = start[len - i];
125 }
126 // and fill the start
127 for (size_t i = 0; i < extra; i++) {
128 start[i] = fill;
129 }
130}

133static void mi_out_num(uintmax_t x, size_t base, char prefix, char** out, char* end)
134{
135 if (x == 0 || base == 0 || base > 16) {
136 if (prefix != 0) { mi_outc(prefix, out, end); }
137 mi_outc('0',out,end);
138 }
139 else {
140 // output digits in reverse
141 char* start = *out;
142 while (x > 0) {
143 char digit = (char)(x % base);
144 mi_outc((digit <= 9 ? '0' + digit : 'A' + digit - 10),out,end);
145 x = x / base;
146 }
147 if (prefix != 0) {
148 mi_outc(prefix, out, end);
149 }
150 size_t len = *out - start;
151 // and reverse in-place
152 for (size_t i = 0; i < (len / 2); i++) {
153 char c = start[len - i - 1];
154 start[len - i - 1] = start[i];
155 start[i] = c;
156 }
157 }
158}

163int _mi_vsnprintf(char* buf, size_t bufsize, const char* fmt, va_list args) {
164 if (buf == NULL || bufsize == 0 || fmt == NULL) return 0;
165 buf[bufsize - 1] = 0;
166 char* const end = buf + (bufsize - 1);
167 const char* in = fmt;
168 char* out = buf;
169 while (true) {
170 if (out >= end) break;
171 char c;
172 MI_NEXTC();
173 if (c != '%') {
174 if (c == '\\') {
175 MI_NEXTC();
176 switch (c) {
177 case 'e': mi_outc('\x1B', &out, end); break;
178 case 't': mi_outc('\t', &out, end); break;
179 case 'n': mi_outc('\n', &out, end); break;
180 case 'r': mi_outc('\r', &out, end); break;
181 case '\\': mi_outc('\\', &out, end); break;
182 default: /* ignore */ break;
183 }
184 }
185 else if ((c >= ' ' && c <= '~') || c=='\n' || c=='\r' || c=='\t' || c=='\x1b') { // output visible ascii or standard control only
186 mi_outc(c, &out, end);
187 }
188 }
189 else {
190 MI_NEXTC();
191 char fill = ' ';
192 size_t width = 0;
193 char numtype = 'd';
194 char numplus = 0;
195 bool alignright = true;
196 if (c == '+' || c == ' ') { numplus = c; MI_NEXTC(); }
197 if (c == '-') { alignright = false; MI_NEXTC(); }
198 if (c == '0') { fill = '0'; MI_NEXTC(); }
199 if (c >= '1' && c <= '9') {
200 width = (c - '0'); MI_NEXTC();
201 while (c >= '0' && c <= '9') {
202 width = (10 * width) + (c - '0'); MI_NEXTC();
203 }
204 if (c == 0) break; // extra check due to while
205 }
206 if (c == 'z' || c == 't' || c == 'L') { numtype = c; MI_NEXTC(); }
207 else if (c == 'l') {
208 numtype = c; MI_NEXTC();
209 if (c == 'l') { numtype = 'L'; MI_NEXTC(); }
210 }
211
212 char* start = out;
213 if (c == '%') {
214 mi_outc('%', &out, end);
215 }
216 else if (c == 's') {
217 // string
218 const char* s = va_arg(args, const char*);
219 mi_outs(s, &out, end);
220 }
221 else if (c == 'p' || c == 'x' || c == 'u') {
222 // unsigned
223 uintmax_t x = 0;
224 if (c == 'x' || c == 'u') {
225 if (numtype == 'z') x = va_arg(args, size_t);
226 else if (numtype == 't') x = va_arg(args, uintptr_t); // unsigned ptrdiff_t
227 else if (numtype == 'L') x = va_arg(args, unsigned long long);
228 else if (numtype == 'l') x = va_arg(args, unsigned long);
229 else x = va_arg(args, unsigned int);
230 }
231 else if (c == 'p') {
232 x = va_arg(args, uintptr_t);
233 mi_outs("0x", &out, end);
234 start = out;
235 width = (width >= 2 ? width - 2 : 0);
236 }
237 if (width == 0 && (c == 'x' || c == 'p')) {
238 if (c == 'p') { width = 2 * (x <= UINT32_MAX ? 4 : ((x >> 16) <= UINT32_MAX ? 6 : sizeof(void*))); }
239 if (width == 0) { width = 2; }
240 fill = '0';
241 }
242 mi_out_num(x, (c == 'x' || c == 'p' ? 16 : 10), numplus, &out, end);
243 }
244 else if (c == 'i' || c == 'd') {
245 // signed
246 intmax_t x = 0;
247 if (numtype == 'z') x = va_arg(args, intptr_t );
248 else if (numtype == 't') x = va_arg(args, ptrdiff_t);
249 else if (numtype == 'L') x = va_arg(args, long long);
250 else if (numtype == 'l') x = va_arg(args, long);
251 else x = va_arg(args, int);
252 char pre = 0;
253 if (x < 0) {
254 pre = '-';
255 if (x > INTMAX_MIN) { x = -x; }
256 }
257 else if (numplus != 0) {
258 pre = numplus;
259 }
260 mi_out_num((uintmax_t)x, 10, pre, &out, end);
261 }
262 else if (c >= ' ' && c <= '~') {
263 // unknown format
264 mi_outc('%', &out, end);
265 mi_outc(c, &out, end);
266 }
267
268 // fill & align
269 mi_assert_internal(out <= end);
270 mi_assert_internal(out >= start);
271 const size_t len = out - start;
272 if (len < width) {
273 mi_out_fill(fill, width - len, &out, end);
274 if (alignright && out <= end) {
275 mi_out_alignright(fill, start, len, width - len, end);
276 }
277 }
278 }
279 }
280 mi_assert_internal(out <= end);
281 *out = 0;
282 return (int)(out - buf);
283}

301static size_t mi_ctz_generic32(uint32_t x) {
302 // de Bruijn multiplication, see <http://keithandkatie.com/keith/papers/debruijn.html>
303 static const uint8_t debruijn[32] = {
304 0, 1, 28, 2, 29, 14, 24, 3, 30, 22, 20, 15, 25, 17, 4, 8,
305 31, 27, 13, 23, 21, 19, 16, 7, 26, 12, 18, 6, 11, 5, 10, 9
306 };
307 if (x==0) return 32;
308 return debruijn[(uint32_t)((x & -(int32_t)x) * (uint32_t)(0x077CB531U)) >> 27];
309}

311static size_t mi_clz_generic32(uint32_t x) {
312 // de Bruijn multiplication, see <http://keithandkatie.com/keith/papers/debruijn.html>
313 static const uint8_t debruijn[32] = {
314 31, 22, 30, 21, 18, 10, 29, 2, 20, 17, 15, 13, 9, 6, 28, 1,
315 23, 19, 11, 3, 16, 14, 7, 24, 12, 4, 8, 25, 5, 26, 27, 0
316 };
317 if (x==0) return 32;
318 x |= x >> 1;
319 x |= x >> 2;
320 x |= x >> 4;
321 x |= x >> 8;
322 x |= x >> 16;
323 return debruijn[(uint32_t)(x * (uint32_t)(0x07C4ACDDU)) >> 27];
324}

364// sum of all the bytes in `x` if it is guaranteed that the sum < 256!
365static size_t mi_byte_sum32(uint32_t x) {
366 // perform `x * 0x01010101`: the highest byte contains the sum of all bytes.
367 x += (x << 8);
368 x += (x << 16);
369 return (size_t)(x >> 24);
370}

372static size_t mi_popcount_generic32(uint32_t x) {
373 // first count each 2-bit group `a`, where: a==0b00 -> 00, a==0b01 -> 01, a==0b10 -> 01, a==0b11 -> 10
374 // in other words, `a - (a>>1)`; to do this in parallel, we need to mask to prevent spilling a bit pair
375 // into the lower bit-pair:
376 x = x - ((x >> 1) & mi_mask_even_bits32);
377 // add the 2-bit pair results
378 x = (x & mi_mask_even_pairs32) + ((x >> 2) & mi_mask_even_pairs32);
379 // add the 4-bit nibble results
380 x = (x + (x >> 4)) & mi_mask_even_nibbles32;
381 // each byte now has a count of its bits, we can sum them now:
382 return mi_byte_sum32(x);
383}
384
385mi_decl_noinline size_t _mi_popcount_generic(size_t x) {
386 return mi_popcount_generic32(x);
387}

402static size_t mi_popcount_generic64(uint64_t x) {

404 x = (x & mi_mask_even_pairs64) + ((x >> 2) & mi_mask_even_pairs64);
405 x = (x + (x >> 4)) & mi_mask_even_nibbles64;
406 return mi_byte_sum64(x);
407}
