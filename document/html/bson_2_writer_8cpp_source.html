71void Writer::end_array(OutputArrayType* _arr) const noexcept {
72 const auto handle = [&](const auto _parent) {
73 using Type = std::remove_cvref_t<decltype(_parent)>;
74 if constexpr (std::is_same<Type, IsArray>()) {
75 bson_array_builder_append_array_builder_end(_parent.ptr_, _arr->val_);
76 } else if constexpr (std::is_same<Type, IsObject>()) {
77 bson_append_array_builder_end(_parent.ptr_, _arr->val_);
78 } else if constexpr (std::is_same<Type, IsRoot>()) {
79 bson_array_builder_build(_arr->val_, doc_);
80 } else {
81 static_assert(rfl::always_false_v<Type>, "Unsupported type.");
82 }
83 };
84 std::visit(handle, _arr->parent_);
85}
