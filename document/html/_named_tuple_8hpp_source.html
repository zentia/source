426 template <int _i = 0, int _idx = -1>
427 constexpr static int find_extra_fields() {
428 if constexpr (_i == size()) {
429 return _idx;
430 } else {
431 using FieldType = internal::nth_element_t<_i, FieldTypes...>;
432 constexpr bool is_extra_fields =
433 internal::is_extra_fields_v<typename FieldType::Type>;
434 static_assert(_idx == -1 || !is_extra_fields,
435 "There can only be one rfl::ExtraFields in any struct or "
436 "named tuple.");
437 if constexpr (is_extra_fields) {
438 return find_extra_fields<_i + 1, _i>();
439 } else {
440 return find_extra_fields<_i + 1, _idx>();
441 }
442 }
