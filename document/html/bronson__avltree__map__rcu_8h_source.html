247 template <typename K, typename... Args>
248 bool emplace( K&& key, Args&&... args )
249 {
250 struct scoped_ptr
251 {
252 mapped_type * pVal;
253 scoped_ptr( mapped_type * p ): pVal( p ) {}
254 ~scoped_ptr() { if ( pVal ) cxx_allocator().Delete( pVal ); }
255 void release() { pVal = nullptr; }
256 };
257
258 scoped_ptr p( cxx_allocator().MoveNew( std::forward<Args>( args )... ));
259 if ( base_class::insert( std::forward<K>( key ), p.pVal )) {
260 p.release();
261 return true;
262 }
263 return false;
264 }

294 template <typename K, typename Func>
295 std::pair<bool, bool> update( K const& key, Func func, bool bAllowInsert = true )
296 {
297 int result = base_class::do_update( key, key_comparator(),
298 [&func]( node_type * pNode ) -> mapped_type*
299 {
300 mapped_type * pVal = pNode->m_pValue.load( memory_model::memory_order_relaxed );
301 if ( !pVal ) {
302 pVal = cxx_allocator().New();
303 func( true, pNode->m_key, *pVal );
304 }
305 else
306 func( false, pNode->m_key, *pVal );
307 return pVal;
308 },
309 (bAllowInsert ? update_flags::allow_insert : 0) | update_flags::allow_update
310 );
311 return std::make_pair( result != 0, (result & update_flags::result_inserted) != 0 );
312 }
