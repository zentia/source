80 int prepare_wraparound() {
81 const int last = static_cast<int>(nodes_list.size())-1;
82
83 bool isa2a_first = get_node(0)->isAll2All();
84 bool isa2a_last = get_lastnode()->isAll2All();
85 bool isfarm_last = get_lastnode()->isFarm();
86 // possible cases: [captured by]
87 //
88 // the last stage is a standard node (the first stage can also be a multi-input node) [last_single_standard]
89 // the first stage is a standard node (the last stage can also be a multi-output node) [first_single_standard]
90 //
91 //
92 // the last stage is a multi-output node:
93 // - it's a farm with a multi-output collector [last_single_multioutput]
94 // - it's a farm without collector and workers are standard nodes [last_multi_standard]
95 // - it's a farm without collector with multi-output workers [last_multi_multioutput]
96 // - it's a all2all with standard nodes [last_multi_standard]
97 // - it's a all2all with multi-output nodes [last_multi_multioutput]
98 // - it's a singolo nodo (comp or pipeline) with the last stage multi-output [last_single_multioutput]
99 //
100 //
101 // the first stage is multi-input
102 // - it's a farm (the emettitore is multi-input by default) [first_single_multiinput]
103 // - it's a single multi-input node [first_single_multiinput]
104 // - it's all2all with standard nodes [first_multi_standard]
105 // - its' all2all with multi-input nodes [first_multi_multiinput]
106 //
107
108 bool last_isfarm_nocollector = get_lastnode()->isFarm() && !isfarm_withcollector(get_lastnode());
109 bool last_isfarm_withcollector = get_lastnode()->isFarm() && !last_isfarm_nocollector;
110
111 bool first_single_standard = (!nodes_list[0]->isMultiInput());
112 // the farm is considered single_multiinput
113 bool first_single_multiinput = (nodes_list[0]->isMultiInput() && !isa2a_first);
114 bool first_multi_standard = [&]() {
115 if (!isa2a_first) return false;
116 const svector<ff_node*>& w1=isa2a_getfirstset(get_node(0));
117 assert(w1.size()>0);
118 if (w1[0]->isMultiInput()) return false; // NOTE: we suppose homogeneous first set
119 return true;
120 }();
121 bool first_multi_multiinput = [&]() {
122 if (!isa2a_first) return false;
123 const svector<ff_node*>& w1=isa2a_getfirstset(get_node(0));
124 assert(w1.size()>0);
125 if (w1[0]->isMultiInput()) return true; // NOTE: we suppose homogeneous first set
126 return false;
127 } ();
128
129 bool last_single_standard = (!nodes_list[last]->isMultiOutput());
130 bool last_single_multioutput = ((nodes_list[last]->isMultiOutput() && !isa2a_last && !isfarm_last) ||
131 (last_isfarm_withcollector && nodes_list[last]->isMultiOutput()));
132 bool last_multi_standard = [&]() {
133 if (last_isfarm_nocollector) {
134 return !isfarm_multimultioutput(get_lastnode());
135 } if (isa2a_last) {
136 const svector<ff_node*>& w1=isa2a_getsecondset(nodes_list[last]);
137 assert(w1.size()>0);
138 if (!w1[0]->isMultiOutput()) return true; // NOTE: we suppose homogeneous second set
139 }
140 return false;
141 } ();
142 bool last_multi_multioutput = [&]() {
143 if (last_isfarm_nocollector) {
144 return isfarm_multimultioutput(get_lastnode());
145 } if (isa2a_last) {
146 const svector<ff_node*>& w1=isa2a_getsecondset(nodes_list[last]);
147 assert(w1.size()>0);
148 if (w1[0]->isMultiOutput()) return true; // NOTE: we suppose homogeneous second set
149 }
150 return false;
151 } ();
152
153 // first stage: standard node
154 if (first_single_standard) {
155 if (create_input_buffer(out_buffer_entries, false)<0) return -1;
156 if (last_single_standard) {
157 if (set_output_buffer(get_in_buffer())<0) return -1;
158
159 // blocking stuff ------
160 pthread_mutex_t *m = NULL;
161 pthread_cond_t *c = NULL;
162 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
163 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
164 nodes_list[last]->set_output_blocking(m,c);
165 // ---------------------
166 } else {
167 if (last_single_multioutput) {
168 ff_node *t = new ff_buffernode(last, get_in_buffer(), get_in_buffer());
169 internalSupportNodes.push_back(t);
170 assert(t);
171 nodes_list[last]->set_output_feedback(t);
172
173 // blocking stuff ------
174 pthread_mutex_t *m = NULL;
175 pthread_cond_t *c = NULL;
176 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
177 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
178 // NOTE: if the node0 is multi-input, then the init_input_blocking
179 // method already calls set_output_blocking
180 t->set_output_blocking(m,c);
181 // ---------------------
182
183 } else {
184 error("PIPE, cannot connect stage %d with node %d\n", last, 0);
185 return -1;
186 }
187 }
188 }
189 // first stage: multi-input
190 if (first_single_multiinput) {
191
192 if (last_single_standard) {
193 if (nodes_list[last]->create_output_buffer(out_buffer_entries,false)<0) return -1;
194 nodes_list[0]->set_input_feedback(nodes_list[last]);
195
196 // blocking stuff ......
197 pthread_mutex_t *m = NULL;
198 pthread_cond_t *c = NULL;
199 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
200 // multi-input nodes execute set_output_blocking
201 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
202 // ---------------------
203 } else {
204 if (last_single_multioutput) {
205 ff_node *t = new ff_buffernode(out_buffer_entries,false, last);
206 assert(t);
207 internalSupportNodes.push_back(t);
208 nodes_list[last]->set_output_feedback(t);
209 nodes_list[0]->set_input_feedback(t);
210
211 // blocking stuff ......
212 pthread_mutex_t *m = NULL;
213 pthread_cond_t *c = NULL;
214 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
215 // multi-input nodes execute set_output_blocking
216 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
217 // ---------------------
218 } else {
219 if (last_multi_standard) {
220 if (nodes_list[last]->create_output_buffer(out_buffer_entries, false) <0)
221 return -1;
222 svector<ff_node*> w(MAX_NUM_THREADS);
223 nodes_list[last]->get_out_nodes(w);
224 assert(w.size());
225 for(size_t j=0;j<w.size();++j)
226 nodes_list[0]->set_input_feedback(w[j]);
227
228 // blocking stuff ......
229 pthread_mutex_t *m = NULL;
230 pthread_cond_t *c = NULL;
231 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
232 // multi-input nodes execute set_output_blocking
233 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
234 // ---------------------
235 } else {
236 if (last_multi_multioutput) {
237 svector<ff_node*> w;
238 ff_node *lastbb = get_lastnode();
239 if (lastbb->isAll2All())
240 w = isa2a_getsecondset(lastbb);
241 else {
242 assert(lastbb->isFarm());
243 w = isfarm_getworkers(lastbb);
244 }
245 assert(w.size());
246
247 for(size_t j=0;j<w.size();++j) {
248 ff_node* t = new ff_buffernode(out_buffer_entries,false, j);
249 assert(t);
250 internalSupportNodes.push_back(t);
251 nodes_list[0]->set_input_feedback(t);
252 w[j]->set_output_feedback(t);
253 }
254
255 // blocking stuff ......
256 pthread_mutex_t *m = NULL;
257 pthread_cond_t *c = NULL;
258 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
259 // multi-input nodes execute set_output_blocking
260 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
261 // ---------------------
262 } else {
263 error("PIPE, wrap_around invalid stage\n");
264 return -1;
265 }
266 }
267 }
268 }
269 }
270 // first stage: multi standard
271 if (first_multi_standard) { // all-to-all
272 assert(get_node(0)->isAll2All());
273 ff_node* a2a = nodes_list[0];
274 const svector<ff_node*>& firstSet=isa2a_getfirstset(a2a);
275 assert(firstSet.size()>0);
276
277 if (last_single_standard) {
278 error("PIPE, wrap_around, cannot connect last with first stage\n");
279 return -1;
280 }
281 if (last_single_multioutput) {
282 for(size_t j=0;j<firstSet.size();++j) {
283 ff_node* t = new ff_buffernode(out_buffer_entries,false, j);
284 assert(t);
285 internalSupportNodes.push_back(t);
286 firstSet[j]->set_input(t);
287 nodes_list[last]->set_output_feedback(t);
288
289 // blocking stuff ......
290 pthread_mutex_t *m = NULL;
291 pthread_cond_t *c = NULL;
292 if (!firstSet[j]->init_input_blocking(m,c)) return -1;
293 t->set_output_blocking(m,c);
294 // ---------------------
295 }
296 // blocking stuff ......
297 pthread_mutex_t *m = NULL;
298 pthread_cond_t *c = NULL;
299 nodes_list[last]->init_output_blocking(m,c);
300 // ---------------------
301 }
302 if (last_multi_standard) {
303 svector<ff_node*> w(MAX_NUM_THREADS);
304 nodes_list[last]->get_out_nodes(w);
305 assert(w.size());
306 assert(w.size() == firstSet.size());
307
308 if (a2a->create_input_buffer(in_buffer_entries, false)<0) return -1;
309 for(size_t j=0;j<w.size();++j)
310 w[j]->set_output_buffer(firstSet[j]->get_in_buffer());
311
312 // blocking stuff ......
313 pthread_mutex_t *m = NULL;
314 pthread_cond_t *c = NULL;
315 if (!a2a->init_output_blocking(m,c)) return -1;
316 for(size_t j=0;j<w.size();++j) {
317 if (!firstSet[j]->init_input_blocking(m,c)) return -1;
318 w[j]->set_output_blocking(m,c);
319 }
320 // ---------------------
321 }
322 if (last_multi_multioutput) {
323 error("PIPE, wrap_around, cannot connect last stage with first stage\n");
324 return -1;
325 }
326 }
327 // first stage: multi multi-input
328 if (first_multi_multiinput) {
329 assert(get_node(0)->isAll2All());
330 const svector<ff_node*>& firstSet=isa2a_getfirstset(nodes_list[0]);
331 assert(firstSet.size()>0);
332
333 if (last_single_standard) {
334 error("PIPE, wrap_around, cannot connect last with first stage\n");
335 return -1;
336 }
337 if (last_single_multioutput) {
338 for(size_t j=0;j<firstSet.size();++j) {
339 ff_node* t = new ff_buffernode(out_buffer_entries,false, j);
340 assert(t);
341 internalSupportNodes.push_back(t);
342 firstSet[j]->set_input_feedback(t);
343 nodes_list[last]->set_output_feedback(t);
344 }
345 }
346 if (last_multi_standard) {
347 svector<ff_node*> w(MAX_NUM_THREADS);
348 nodes_list[last]->get_out_nodes(w);
349 assert(w.size());
350 assert(w.size() == firstSet.size());
351
352 for(size_t i=0;i<firstSet.size(); ++i) {
353 ff_node* t = new ff_buffernode(out_buffer_entries,false, i);
354 assert(t);
355 internalSupportNodes.push_back(t);
356 firstSet[i]->set_input_feedback(t);
357 w[i]->set_output(t);
358 }
359 }
360 if (last_multi_multioutput) {
361 svector<ff_node*> w;
362 ff_node *lastbb = get_lastnode();
363 if (lastbb->isAll2All())
364 w = isa2a_getsecondset(lastbb);
365 else {
366 assert(lastbb->isFarm());
367 w = isfarm_getworkers(lastbb);
368 }
369 assert(w.size());
370
371 // here we have to create all connections
372 for(size_t i=0;i<firstSet.size(); ++i) {
373 for(size_t j=0;j<w.size();++j) {
374 ff_node* t = new ff_buffernode(out_buffer_entries,false, j);
375 assert(t);
376 internalSupportNodes.push_back(t);
377 firstSet[i]->set_input_feedback(t);
378 w[j]->set_output_feedback(t);
379 }
380 }
381 }
382 // blocking stuff ......
383 pthread_mutex_t *m = NULL;
384 pthread_cond_t *c = NULL;
385 if (!nodes_list[last]->init_output_blocking(m,c)) return -1;
386 if (!nodes_list[0]->init_input_blocking(m,c)) return -1;
387 // ---------------------
388 }
389 return 0;
390 }
391 inline int prepare() {
392
393 if (wraparound) {
394 if (nodes_list.size()<2) {
395 error("PIPE, too few pipeline nodes\n");
396 return -1;
397 }
398 if (prepare_wraparound()<0) {
399 error("PIPE, prepare_wraparound failed\n");
400 return -1;
401 }
402 }
403
404 const int nstages=static_cast<int>(nodes_list.size());
405
406 // possible cases: [captured by]
407 //
408 // the current stage is a standard node (the previous stage can also be multi-output) [curr_single_standard]
409 //
410 // the current stage is multi-input
411 // - it's a farm (the emettitore is multi-input by default) [curr_single_multiinput]
412 // - it's a single multi-input node [curr_single_multiinput]
413 // - it's all2all with standard nodes [curr_multi_standard]
414 // - it's all2all with multi-input nodes [curr_multi_multiinput]
415 //
416 // the previous stage is a standard node [prev_single_standard]
417 //
418 // the previous stage is a multi-output node:
419 // - it's a farm with a multi-output collector [prev_single_multioutput]
420 // - it's a farm without collector and workers are standard node [prev_multi_standard]
421 // - it's a farm without collector with multi-output workers [prev_multi_multioutput]
422 // - it's a all2all with standard nodes [prev_multi_standard]
423 // - it's a all2all with multi-output nodes [prev_multi_multioutput]
424 // - it's a single node (comp or pipeline) with the last stage multi-output [prev_single_multioutput]
425 //
426 //
427 for(int i=1;i<nstages;++i) {
428 const bool isa2a_curr = get_node(i)->isAll2All();
429 const bool curr_single_standard = (!nodes_list[i]->isMultiInput());
430 // the farm is considered single_multiinput
431 const bool curr_single_multiinput = (!isa2a_curr && nodes_list[i]->isMultiInput());
432 const bool curr_multi_multiinput = [&]() {
433 if (!isa2a_curr) return false;
434 const svector<ff_node*>& w1=isa2a_getfirstset(get_node(i));
435 assert(w1.size()>0);
436 for(size_t k=0;k<w1.size();++k) {
437 svector<ff_node*> w2(1);
438 w1[k]->get_in_nodes(w2);
439 for(size_t j=0;j<w2.size();++j)
440 if (w2[j]->isMultiInput()) return true;
441 }
442 return false;
443 } ();
444 const bool curr_multi_standard = [&]() {
445 if (!isa2a_curr) return false;
446 return !curr_multi_multiinput;
447 }();
448 const bool isa2a_prev = get_node_last(i-1)->isAll2All();
449 const bool isfarm_prev = get_node_last(i-1)->isFarm();
450 const bool prev_isfarm_nocollector = (isfarm_prev && !get_node_last(i-1)->isOFarm() && !(isfarm_withcollector(get_node_last(i-1))));
451 const bool prev_isfarm_withcollector = isfarm_withcollector(get_node_last(i-1));
452
453
454 const bool prev_single_standard = (!get_node_last(i-1)->isMultiOutput());
455 const bool prev_single_multioutput = ((get_node_last(i-1)->isMultiOutput() && !isa2a_prev && !isfarm_prev) ||
456 (prev_isfarm_withcollector && get_node_last(i-1)->isMultiOutput()));
457 const bool prev_multi_standard = [&]() {
458 if (prev_isfarm_nocollector) {
459 svector<ff_node*> w1;
460 nodes_list[i-1]->get_out_nodes(w1);
461 if (!w1[0]->isMultiOutput()) return true; // NOTE: we suppose homogeneous workers
462 } if (isa2a_prev) {
463 const svector<ff_node*>& w1=isa2a_getsecondset(get_node_last(i-1));
464 assert(w1.size()>0);
465 if (!w1[0]->isMultiOutput()) return true; // NOTE: we suppose homogeneous workers
466 }
467 return false;
468 } ();
469 const bool prev_multi_multioutput = [&]() {
470 if (prev_isfarm_nocollector) {
471 svector<ff_node*> w1;
472 nodes_list[i-1]->get_out_nodes(w1);
473 if (w1[0]->isMultiOutput()) return true; // NOTE: we suppose homogeneous workers
474 } if (isa2a_prev) {
475 const svector<ff_node*>& w1=isa2a_getsecondset(get_node_last(i-1));
476 assert(w1.size()>0);
477 if (w1[0]->isMultiOutput()) return true; // NOTE: we suppose homogeneous workers
478 }
479 return false;
480 } ();
481
482 pthread_mutex_t *m = NULL;
483 pthread_cond_t *c = NULL;
484 if (curr_single_standard) {
485 bool skip_set_output_blocking = false;
486 if (nodes_list[i]->create_input_buffer(in_buffer_entries, fixedsizeIN)<0) return -1;
487 if (prev_single_standard) {
488 if (nodes_list[i-1]->set_output_buffer(nodes_list[i]->get_in_buffer())<0) return -1;
489 } else {
490 skip_set_output_blocking = true;
491 // WARNING: here we add as output node of the previous stage the
492 // current node and not a buffer-node.
493 if (prev_multi_standard || prev_multi_multioutput) {
494 svector<ff_node*> w(1);
495 nodes_list[i-1]->get_out_nodes(w);
496 if (w.size()>1) {
497 error("PIPE, cannot connect stage %d with stage %d\n", i-1, i);
498 return -1;
499 }
500 nodes_list[i-1]->set_output(nodes_list[i]);
501 } else {
502 assert(prev_single_multioutput);
503 nodes_list[i-1]->set_output(nodes_list[i]);
504 }
505 }
506 // blocking stuff --------------------------------------------
507 if (!nodes_list[i]->init_input_blocking(m,c)) {
508 error("PIPE, init input blocking mode for node %d\n", i);
509 return -1;
510 }
511 if (!skip_set_output_blocking) // we do not want to overwrite previous setting
512 nodes_list[i-1]->set_output_blocking(m,c);
513 if (!nodes_list[i-1]->init_output_blocking(m,c,false)) {
514 error("PIPE, init output blocking mode for node %d\n", i-1);
515 return -1;
516 }
517 // ------------------------------------------------------------
518 }
519 if (curr_single_multiinput) {
520 if (prev_single_standard) {
521 if (nodes_list[i-1]->create_output_buffer(in_buffer_entries, fixedsizeOUT)<0) return -1;
522 if (nodes_list[i]->set_input(nodes_list[i-1])<0) return -1;
523
524 // blocking stuff --------------------------------------------
525 if (!nodes_list[i]->init_input_blocking(m,c)) {
526 error("PIPE, init input blocking mode for node %d\n", i);
527 return -1;
528 }
529 // since the curr node is multi-input, the following op has been executed
530 // by the previous one.
531 //nodes_list[i-1]->set_output_blocking(m,c);
532 if (!nodes_list[i-1]->init_output_blocking(m,c)) {
533 error("PIPE, init output blocking mode for node %d\n", i-1);
534 return -1;
535 }
536 // ------------------------------------------------------------
537 } else {
538 if (prev_single_multioutput) {
539 ff_node* t = new ff_buffernode(in_buffer_entries,fixedsizeIN|fixedsizeOUT, i);
540 internalSupportNodes.push_back(t);
541 nodes_list[i-1]->set_output(t);
542 nodes_list[i]->set_input(t);
543 }
544 if (prev_multi_standard) {
545 if (nodes_list[i-1]->create_output_buffer(out_buffer_entries, fixedsizeOUT)<0) return -1;
546 svector<ff_node*> w(MAX_NUM_THREADS);
547 nodes_list[i-1]->get_out_nodes(w);
548 assert(w.size());
549 nodes_list[i]->set_input(w);
550
551 //if (w.size() == 0) nodes_list[i]->set_input(nodes_list[i-1]);
552 }
553 if (prev_multi_multioutput) {
554 svector<ff_node*> w(MAX_NUM_THREADS);
555 nodes_list[i-1]->get_out_nodes(w);
556 assert(w.size());
557 for(size_t j=0;j<w.size();++j) {
558 ff_node* t = new ff_buffernode(in_buffer_entries,fixedsizeIN|fixedsizeOUT, j);
559 internalSupportNodes.push_back(t);
560 w[j]->set_output(t);
561 nodes_list[i]->set_input(t);
562 }
563 }
564 // blocking stuff --------------------------------------------
565 if (!nodes_list[i]->init_input_blocking(m,c)) {
566 error("PIPE, init input blocking mode for node %d\n", i);
567 return -1;
568 }
569 svector<ff_node*> w(MAX_NUM_THREADS);
570 nodes_list[i-1]->get_out_nodes(w);
571 assert(w.size());
572 for(size_t j=0;j<w.size();++j) {
573 // we set p_cons_* for each buffernode
574 w[j]->set_output_blocking(m,c);
575 }
576 if (!nodes_list[i-1]->init_output_blocking(m,c)) {
577 error("PIPE, init output blocking mode for node %d\n", i);
578 return -1;
579 }
580 // ------------------------------------------------------------
581 }
582 }
583 if (curr_multi_standard) {
584 ff_node* a2a = nodes_list[i];
585 const svector<ff_node*>& W1 = isa2a_getfirstset(a2a);
586 assert(W1.size()>0);
587
588 if (prev_single_standard) {
589 if (W1.size()>1) {
590 error("PIPE, cannot connect stage %d with stage %d because of different cardinality\n", i-1, i);
591 return -1;
592 }
593 if (a2a->create_input_buffer(in_buffer_entries, fixedsizeIN)<0) return -1;
594 svector<ff_node*> w(1);
595 nodes_list[i]->get_in_nodes(w);
596 assert(w.size() == 1);
597 if (nodes_list[i-1]->set_output_buffer(w[0]->get_in_buffer())<0) return -1;
598
599 // blocking stuff --------------------------------------------
600 if (!nodes_list[i]->init_input_blocking(m,c)) {
601 error("PIPE, init input blocking mode for node %d\n", i);
602 return -1;
603 }
604 nodes_list[i-1]->set_output_blocking(m,c);
605 if (!nodes_list[i-1]->init_output_blocking(m,c)) {
606 error("PIPE, init output blocking mode for node %d\n", i-1);
607 return -1;
608 }
609 // -----------------------------------------------------------
610 } else {
611 if (prev_single_multioutput) {
612 if (a2a->create_input_buffer(in_buffer_entries, fixedsizeIN)<0) return -1;
613 svector<ff_node*> w(MAX_NUM_THREADS);
614 nodes_list[i]->get_in_nodes(w);
615 if (nodes_list[i-1]->set_output(w)<0) return -1;
616
617 // blocking stuff --------------------------------------------
618 if (!a2a->init_input_blocking(m,c)) return -1;
619 if (!nodes_list[i-1]->init_output_blocking(m,c)) {
620 error("PIPE, init output blocking mode for node %d\n", i-1);
621 return -1;
622 }
623 // -----------------------------------------------------------
624 } else {
625 assert(prev_multi_multioutput || prev_multi_standard);
626
627 svector<ff_node*> w1(1);
628 nodes_list[i-1]->get_out_nodes(w1);
629 assert(w1.size());
630 // here we pretend to have point-to-point connections
631 if (w1.size() != W1.size()) {
632 error("PIPE, cannot connect stage %d with stage %d, because of different input/output cardinality (**)\n", i-1, i);
633 return -1;
634 }
635 if (a2a->create_input_buffer(in_buffer_entries, fixedsizeIN)<0) return -1;
636 // the previous node can be a farm or a all2all
637 for(size_t i=0;i<w1.size();++i) {
638 if (prev_multi_multioutput) {
639 if (w1[i]->set_output(W1[i])<0) return -1;
640 } else {
641 if (w1[i]->set_output_buffer(W1[i]->get_in_buffer())<0) return -1;
642 }
643 }
644 //if (nodes_list[i-1]->set_output(w1)<0) return -1;
645
646 // blocking stuff --------------------------------------------
647 svector<ff_node*> w(MAX_NUM_THREADS);
648 nodes_list[i-1]->get_out_nodes(w);
649 assert(w.size());
650 for(size_t i=0;i<w.size();++i) {
651 if (!W1[i]->init_input_blocking(m,c)) return -1;
652 w[i]->set_output_blocking(m,c);
653
654 if (!w[i]->init_output_blocking(m,c)) return -1;
655 }
656 // -----------------------------------------------------------
657 }
658 }
659 }
660 // it could be an all-to-all or a pipeline containing as first stage an all-to-all
661 if (curr_multi_multiinput) {
662 ff_node* a2a = get_node(i);
663 assert(a2a->isAll2All());
664 const svector<ff_node*>& W1 = isa2a_getfirstset(a2a);
665 assert(W1.size()>0);
666
667 if (prev_single_standard) {
668 if (W1.size()>1) {
669 error("PIPE, cannot connect stage %d with stage %d\n", i-1, i);
670 return -1;
671 }
672 if (a2a->create_input_buffer(in_buffer_entries, fixedsizeIN)<0) return -1;
673 svector<ff_node*> w(1);
674 nodes_list[i]->get_in_nodes(w);
675 assert(w.size() == 1);
676 if (nodes_list[i-1]->set_output_buffer(w[0]->get_in_buffer())<0) return -1;
677 }
678 if (prev_single_multioutput) {
679 for(size_t j=0;j<W1.size();++j) {
680 svector<ff_node*> w(MAX_NUM_THREADS);
681 W1[j]->get_in_nodes(w);
682 for(size_t k=0;k<w.size();++k) {
683 ff_node* t = new ff_buffernode(in_buffer_entries,fixedsizeIN|fixedsizeOUT, j);
684 internalSupportNodes.push_back(t);
685 w[k]->set_input(t);
686 nodes_list[i-1]->set_output(t);
687 }
688 }
689 }
690 if (prev_multi_standard) {
691
692 svector<ff_node*> w(MAX_NUM_THREADS);
693 nodes_list[i-1]->get_out_nodes(w);
694 assert(w.size());
695 // here we pretend to have point-to-point connections
696 if (w.size() != W1.size()) {
697 error("PIPE, cannot connect stage %d with stage %d, because of different input/output cardinality (**)\n", i-1, i);
698 return -1;
699 }
700 w.clear();
701 if (nodes_list[i-1]->create_output_buffer(in_buffer_entries, fixedsizeOUT)<0) return -1;
702 nodes_list[i-1]->get_out_nodes(w);
703 assert(w.size()== W1.size());
704
705 for(size_t j=0;j<W1.size();++j) {
706 W1[j]->set_input(w[j]);
707 }
708 }
709 if (prev_multi_multioutput) {
710 svector<ff_node*> w(MAX_NUM_THREADS);
711 nodes_list[i-1]->get_out_nodes(w);
712 assert(w.size());
713
714 for(size_t k=0;k<W1.size(); ++k) {
715 for(size_t j=0;j<w.size();++j) {
716 ff_node* t = new ff_buffernode(in_buffer_entries,fixedsizeIN|fixedsizeOUT, j);
717 internalSupportNodes.push_back(t);
718 W1[k]->set_input(t);
719 w[j]->set_output(t);
720 }
721 }
722 }
723
724 // blocking stuff --------------------------------------------
725 if (prev_multi_standard) {
726 svector<ff_node*> w(MAX_NUM_THREADS);
727 nodes_list[i-1]->get_out_nodes(w);
728 assert(w.size());
729 for(size_t i=0;i<w.size();++i) {
730 if (!W1[i]->init_input_blocking(m,c)) return -1;
731 w[i]->set_output_blocking(m,c);
732
733 if (!w[i]->init_output_blocking(m,c)) return -1;
734 }
735 } else {
736 if (!a2a->init_input_blocking(m,c)) {
737 error("PIPE, init input blocking mode for node %d\n", i);
738 return -1;
739 }
740 if (!nodes_list[i-1]->init_output_blocking(m,c)) {
741 error("PIPE, init output blocking mode for node %d\n", i-1);
742 return -1;
743 }
744 }
745 // -----------------------------------------------------------
746
747 }
748 }
749
750 // Preparation of buffers for the accelerator
751 int ret = 0;
752 if (has_input_channel) {
753 if (create_input_buffer(in_buffer_entries, fixedsizeIN)<0) {
754 error("PIPE, creating input buffer for the accelerator\n");
755 ret=-1;
756 } else {
757 if (get_out_buffer()) {
758 error("PIPE, output buffer already present for the accelerator\n");
759 ret=-1;
760 } else {
761 // NOTE: the last buffer is forced to be unbounded |
762 if (create_output_buffer(out_buffer_entries, false)<0) {
763 error("PIPE, creating output buffer for the accelerator\n");
764 ret = -1;
765 }
766 }
767 }
768
769 pthread_mutex_t *m = NULL;
770 pthread_cond_t *c = NULL;
771
772 // set blocking input for the first stage (cons_m,...)
773 if (!init_input_blocking(m,c)) {
774 error("PIPE, init input blocking mode for accelerator\n");
775 }
776 // set my pointers to the first stage input blocking stuff
777 ff_node::set_output_blocking(m,c);
778
779 m=NULL,c=NULL;
780
781 // set my blocking output (prod_m, ....)
782 if (!ff_node::init_output_blocking(m,c)) {
783 error("PIPE, init output blocking mode for accelerator\n");
784 }
785
786 m=NULL,c=NULL;
787
788 // pipeline's first stage blocking output stuff (prod_m, ....)
789 if (!init_output_blocking(m,c)) {
790 error("FARM, add_collector, init input blocking mode for accelerator\n");
791 }
792
793 m=NULL,c=NULL;
794
795 // set my blocking input (cons_m, ....)
796 if (!ff_node::init_input_blocking(m,c)) {
797 error("PIPE, init input blocking mode for accelerator\n");
798 }
799 // give pointers to my blocking input to the last pipeline stage (p_cons_m,...)
800 set_output_blocking(m,c);
801 }
802
803 prepared=true;
804 return ret;
805 }

950 /* remove one stage already inserted into the pipeline.
951 * NOTE: The node is not deleted if the pipeline cleanup flag is set.
952 * NOTE: If remove_from_cleanuplist is true the node is removed (if it is present)
953 * from the list of internalSupportNodes.
954 *
955 */
956 void remove_stage(int pos, bool remove_from_cleanuplist=false) {
957 if (prepared) {
958 error("PIPE, remove_stage, stage %d cannot be removed because the PIPE has already been prepared\n");
959 return;
960 }
961 if (pos<0 || pos>static_cast<int>(nodes_list.size())) {
962 error("PIPE, remove_stage, stage number %d does not exist\n", pos);
963 return;
964 }
965 svector<ff_node*>::iterator it=nodes_list.begin();
966 assert(it+pos < nodes_list.end());
967 if (remove_from_cleanuplist) {
968 ff_node* node = nodes_list[pos];
969 int pos2=-1;
970 for(size_t i=0;i<internalSupportNodes.size();++i)
971 if (internalSupportNodes[i] == node) { pos2 = i; break; }
972 if (pos2>=0) internalSupportNodes.erase(internalSupportNodes.begin()+pos2);
973 }
974 nodes_list.erase(it+pos);
975 }
