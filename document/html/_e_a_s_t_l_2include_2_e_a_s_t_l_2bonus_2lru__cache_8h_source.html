6// lru_cache is a container that simplifies caching of objects in a map.
7// Basically, you give the container a key, like a string, and the data you want.
8// The container provides callback mechanisms to generate data if it's missing
9// as well as delete data when it's purged from the cache. This container
10// uses a least recently used method: whatever the oldest item is will be
11// replaced with a new entry.
12//
13// Algorithmically, the container is a combination of a map and a list.
14// The list stores the age of the entries by moving the entry to the head
15// of the list on each access, either by a call to get() or to touch().
16// The map is just the map as one would expect.
17//
18// This is useful for caching off data that is expensive to generate,
19// for example text to speech wave files that are dynamically generated,
20// but that will need to be reused, as is the case in narration of menu
21// entries as a user scrolls through the entries.

140 bool insert(const key_type& k, const value_type& v)
141 {
142 if (m_map.find(k) == m_map.end())
143 {
144 make_space();
145
146 m_list.push_front(k);
147 m_map[k] = data_container_type(v, m_list.begin());
148
149 return true;
150 }

152 {
153 return false;

217 value_type& get(const key_type& k)
218 {
219 auto iter = m_map.find(k);
220
221 // The entry exists in the cache
222 if (iter != m_map.end())
223 {
224 touch(k);
225 return iter->second.first;
226 }
227 else // The entry doesn't exist in the cache, so create one
228 {
229 // Add the entry to the map
230 insert(k, m_create_callback ? m_create_callback(k) : value_type());
231
232 // return the new data
233 return m_map[k].first;
234 }
235 }

244 bool erase(const key_type& k)
245 {

294 void touch(iterator& iter)

389 bool trim()
390 {
391 if (size() <= m_capacity)
392 {
393 return false; // No trim necessary
394 }
395
396 // We need to trim
397 do
398 {
399 erase_oldest();
400 } while (m_list.size() > m_capacity);
401
402 return true;
403 }
