107 std::mt19937 g(0); // fixed the seed for graph generator
108 std::srand(0);

110 for(size_t l = 0; l < level; ++l){
111
112 std::vector<Node> cur_nodes;
113 std::vector<int> next_level_nodes;
114
115 for(size_t i = 0; i < length; i++){
116 next_level_nodes.push_back(i);

118
119 //shuffle nodes in the next level
120 std::shuffle(next_level_nodes.begin(), next_level_nodes.end(), g);
121
122 size_t edge_num = 1;
123 size_t start = 0, end = 0;
124 bool re_shuffle = false;
125
126 for(size_t i = 0; i < length; i++){
127 edge_num = 1 + (std::rand() % _edge_max);
128
129
130 if(start + edge_num >= length){
131 end = length;
132 re_shuffle = true;
133 }
134 else{
135 end = start + edge_num;
136 }
137
138 //std::cout << "Level\t" << l << "\tidx\t" << i << "\tedge_num\t" << edge_num << "\tstart\t" << start << "\tend\t" << end << std::endl;
139
140 std::vector<int> edges(next_level_nodes.begin()+start, next_level_nodes.begin()+end);
141
142 //choose a node to do some work
143 float rv = std::rand()/(RAND_MAX + 1u);
144 bool chosen = false;
145 if(rv < threshold){
146 chosen = true;
147 }
148
149 cur_nodes.emplace_back(*this, chosen, l, i, length, edges); //create nodes
150
151 if(re_shuffle){
152 std::shuffle(next_level_nodes.begin(), next_level_nodes.end(), g);
153 start = 0;
154 re_shuffle = false;
155 }
156 else{
157 start = end;
158 }
159 }

161 _graph.push_back(std::move(cur_nodes));
162
163 }
164
165 for(size_t l = 0; level > 0 && l < level-1; ++l){
166 for(size_t i = 0; i < length; ++i){

242 void BFS(const Node& dst){
243
244 std::unordered_set<int> idx_list;
245 std::queue<int> idx_queue;
246
247 int dst_idx = dst.level()*_length_num + dst.index();
248 idx_queue.push(dst_idx);
249 idx_list.insert(dst_idx);
250
251 while(!idx_queue.empty()){
252 int child = idx_queue.front();
253 int child_level = child / _length_num;
254 int child_index = child % _length_num;
255 //std::cout << "Node level: " << child_level << " idx: " << child_index << std::endl;
256
257 idx_queue.pop();
258 const Node& n = _graph[child_level][child_index];
259
260 for(size_t i = 0; child_level > 0 && i < n._in_edges.size(); i++){
261 int parent_level = child_level-1;
262 int parent_index = n._in_edges[i].first;
263 int parent = parent_level*_length_num + parent_index;
264
265 if(idx_list.find(parent) == idx_list.end()){
266 idx_queue.push(parent);
267 idx_list.insert(parent);
268 }
269 }
270
271 }
