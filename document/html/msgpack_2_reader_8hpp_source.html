58 template <class T>
59 rfl::Result<T> to_basic_type(const InputVarType& _var) const noexcept {
60 const auto type = _var.type;
61 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
62 if (type != MSGPACK_OBJECT_STR) {
63 return error("Could not cast to string.");
64 }
65 const auto str = _var.via.str;
66 return std::string(str.ptr, str.size);
67
68 } else if constexpr (std::is_same<std::remove_cvref_t<T>,
69 rfl::Bytestring>()) {
70 if (type != MSGPACK_OBJECT_BIN) {
71 return error("Could not cast to a bytestring.");
72 }
73 const auto bin = _var.via.bin;
74 const auto data = internal::ptr_cast<const std::byte*>(bin.ptr);
75 return rfl::Bytestring(data, data + bin.size);
76
77 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
78 if (type != MSGPACK_OBJECT_BOOLEAN) {
79 return error("Could not cast to boolean.");
80 }
81 return _var.via.boolean;
82
83 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
84 if (type == MSGPACK_OBJECT_FLOAT32 || type == MSGPACK_OBJECT_FLOAT64 ||
85 type == MSGPACK_OBJECT_FLOAT) {
86 return static_cast<T>(_var.via.f64);
87 }
88 return error(
89 "Could not cast to numeric value. The type must be float "
90 "or double.");
91
92 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
93 if (type == MSGPACK_OBJECT_POSITIVE_INTEGER) {
94 return static_cast<T>(_var.via.u64);
95 } else if (type == MSGPACK_OBJECT_NEGATIVE_INTEGER) {
96 return static_cast<T>(_var.via.i64);
97 }
98 return error(
99 "Could not cast to numeric value. The type must be integral, float "
100 "or double.");
101 } else {
102 static_assert(rfl::always_false_v<T>, "Unsupported type.");
103 }
104 }
