75protected:
76 tpcEnvironment():ctx(NULL),dev_ctx(NULL) {
77 tpcId = 0;
78
79 bool ok = false;
80 tpc_res_t r = tpc_init(&ctx);
81 if (r == TPC_SUCCESS) {
82 // FIX: need to support multiple TPC devices
83 r = tpc_create_device(ctx, 0, &dev_ctx, TPC_DEVICE_CREATE_FLAGS_NONE);
84 ok = r == TPC_SUCCESS;
85 }
86 if (! ok) {
87 tpc_deinit(ctx);
88 ctx = NULL, dev_ctx=NULL;
89 error("tpcEnvironment::tpcEnvironment FATAL ERROR, unable to create TPC device\n");
90 abort();
91 }
92 }
93
94public:
95 ~tpcEnvironment() {
96 if (ctx != NULL) {
97 tpc_destroy_device(ctx,dev_ctx);
98 tpc_deinit(ctx);
99 }
100 }

102 static inline tpcEnvironment * instance() {
103 while (!m_tpcEnvironment) {
104 pthread_mutex_lock(&tpcInstanceMutex);
105 if (!m_tpcEnvironment) {
106 m_tpcEnvironment = new tpcEnvironment();
107 }
108 assert(m_tpcEnvironment);
109 pthread_mutex_unlock(&tpcInstanceMutex);
110 }
111 return m_tpcEnvironment;
112 }
