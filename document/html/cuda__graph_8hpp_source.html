774// dump the graph
775template <typename Creator, typename Deleter>
776void cudaGraphBase<Creator, Deleter>::dump(std::ostream& os) {
777
778 // Generate a unique temporary filename in the system's temp directory using filesystem
779 auto temp_path = std::filesystem::temp_directory_path() / "graph_";
780 std::random_device rd;
781 std::uniform_int_distribution<int> dist(100000, 999999); // Generates a random number
782 temp_path += std::to_string(dist(rd)) + ".dot";
783
784 // Call the original function with the temporary file
785 TF_CHECK_CUDA(cudaGraphDebugDotPrint(this->get(), temp_path.string().c_str(), 0), "");
786
787 // Read the file and write to the output stream
788 std::ifstream file(temp_path);
789 if (file) {
790 os << file.rdbuf(); // Copy file contents to the stream
791 file.close();
792 std::filesystem::remove(temp_path); // Clean up the temporary file
793 } else {
794 TF_THROW("failed to open ", temp_path, " for dumping the CUDA graph");
795 }
796}
