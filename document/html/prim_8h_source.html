41// Allocate OS memory. Return NULL on error.
42// The `try_alignment` is just a hint and the returned pointer does not have to be aligned.
43// If `commit` is false, the virtual memory range only needs to be reserved (with no access)
44// which will later be committed explicitly using `_mi_prim_commit`.
45// `is_zero` is set to true if the memory was zero initialized (as on most OS's)
46// The `hint_addr` address is either `NULL` or a preferred allocation address but can be ignored.
47// pre: !commit => !allow_large
48// try_alignment >= _mi_os_page_size() and a power of 2
49int _mi_prim_alloc(void* hint_addr, size_t size, size_t try_alignment, bool commit, bool allow_large, bool* is_large, bool* is_zero, void** addr);

51// Commit memory. Returns error code or 0 on success.

53// `is_zero` is set to true if the memory was zero initialized (e.g. on Windows)
54int _mi_prim_commit(void* addr, size_t size, bool* is_zero);

56// Decommit memory. Returns error code or 0 on success. The `needs_recommit` result is true

125//-------------------------------------------------------------------
126// Access to TLS (thread local storage) slots.
127// We need fast access to both a unique thread id (in `free.c:mi_free`) and
128// to a thread-local heap pointer (in `alloc.c:mi_malloc`).
129// To achieve this we use specialized code for various platforms.
130//-------------------------------------------------------------------
131
132// On some libc + platform combinations we can directly access a thread-local storage (TLS) slot.
133// The TLS layout depends on both the OS and libc implementation so we use specific tests for each main platform.
134// If you test on another platform and it works please send a PR :-)
135// see also https://akkadia.org/drepper/tls.pdf for more info on the TLS register.
136//
137// Note: we would like to prefer `__builtin_thread_pointer()` nowadays instead of using assembly,
138// but unfortunately we can not detect support reliably (see issue #883)
139// We also use it on Apple OS as we use a TLS slot for the default heap there.
140#if defined(__GNUC__) && ( \
141 (defined(__GLIBC__) && (defined(__x86_64__) || defined(__i386__) || (defined(__arm__) && __ARM_ARCH >= 7) || defined(__aarch64__))) \
142 || (defined(__APPLE__) && (defined(__x86_64__) || defined(__aarch64__) || defined(__POWERPC__))) \
143 || (defined(__BIONIC__) && (defined(__x86_64__) || defined(__i386__) || (defined(__arm__) && __ARM_ARCH >= 7) || defined(__aarch64__))) \
144 || (defined(__FreeBSD__) && (defined(__x86_64__) || defined(__i386__) || defined(__aarch64__))) \
145 || (defined(__OpenBSD__) && (defined(__x86_64__) || defined(__i386__) || defined(__aarch64__))) \
146 )
147
148#define MI_HAS_TLS_SLOT 1

150static inline void* mi_prim_tls_slot(size_t slot) mi_attr_noexcept {
151 void* res;
152 const size_t ofs = (slot*sizeof(void*));
153 #if defined(__i386__)
154 __asm__("movl %%gs:%1, %0" : "=r" (res) : "m" (*((void**)ofs)) : ); // x86 32-bit always uses GS
155 #elif defined(__APPLE__) && defined(__x86_64__)
156 __asm__("movq %%gs:%1, %0" : "=r" (res) : "m" (*((void**)ofs)) : ); // x86_64 macOSX uses GS
157 #elif defined(__x86_64__) && (MI_INTPTR_SIZE==4)
158 __asm__("movl %%fs:%1, %0" : "=r" (res) : "m" (*((void**)ofs)) : ); // x32 ABI
159 #elif defined(__x86_64__)
160 __asm__("movq %%fs:%1, %0" : "=r" (res) : "m" (*((void**)ofs)) : ); // x86_64 Linux, BSD uses FS
161 #elif defined(__arm__)
162 void** tcb; MI_UNUSED(ofs);
163 __asm__ volatile ("mrc p15, 0, %0, c13, c0, 3\nbic %0, %0, #3" : "=r" (tcb));
164 res = tcb[slot];
165 #elif defined(__aarch64__)
166 void** tcb; MI_UNUSED(ofs);
167 #if defined(__APPLE__) // M1, issue #343
168 __asm__ volatile ("mrs %0, tpidrro_el0\nbic %0, %0, #7" : "=r" (tcb));
169 #else
170 __asm__ volatile ("mrs %0, tpidr_el0" : "=r" (tcb));
171 #endif
172 res = tcb[slot];
173 #elif defined(__APPLE__) && defined(__POWERPC__) // ppc, issue #781
174 MI_UNUSED(ofs);
175 res = pthread_getspecific(slot);
176 #endif
177 return res;
178}

189 #elif defined(__x86_64__)

211// On windows we can store the thread-local heap at a fixed TLS slot to avoid
212// thread-local initialization checks in the fast path. This uses a fixed location
213// in the TCB though (last user-reserved slot by default) which may clash with other applications.
214
215#define MI_HAS_TLS_SLOT 2 // 2 = we can reliably initialize the slot (saving a test on each malloc)
216
217#if MI_WIN_USE_FIXED_TLS > 1
218#define MI_TLS_SLOT (MI_WIN_USE_FIXED_TLS)
219#elif MI_SIZE_SIZE == 4
220#define MI_TLS_SLOT (0x710) // Last user-reserved slot <https://en.wikipedia.org/wiki/Win32_Thread_Information_Block>
221// #define MI_TLS_SLOT (0xF0C) // Last TlsSlot (might clash with other app reserved slot)
222#else
223#define MI_TLS_SLOT (0x888) // Last user-reserved slot <https://en.wikipedia.org/wiki/Win32_Thread_Information_Block>
224// #define MI_TLS_SLOT (0x1678) // Last TlsSlot (might clash with other app reserved slot)
225#endif

227static inline void* mi_prim_tls_slot(size_t slot) mi_attr_noexcept {
228 #if (_M_X64 || _M_AMD64) && !defined(_M_ARM64EC)
229 return (void*)__readgsqword((unsigned long)slot); // direct load at offset from gs
230 #elif _M_IX86 && !defined(_M_ARM64EC)
231 return (void*)__readfsdword((unsigned long)slot); // direct load at offset from fs
232 #else
233 return ((void**)NtCurrentTeb())[slot / sizeof(void*)];
234 #endif
235}
236static inline void mi_prim_tls_slot_set(size_t slot, void* value) mi_attr_noexcept {
237 ((void**)NtCurrentTeb())[slot / sizeof(void*)] = value;
238}

244//-------------------------------------------------------------------
245// Get a fast unique thread id.
246//
247// Getting the thread id should be performant as it is called in the
248// fast path of `_mi_free` and we specialize for various platforms as
249// inlined definitions. Regular code should call `init.c:_mi_thread_id()`.
250// We only require _mi_prim_thread_id() to return a unique id
251// for each thread (unequal to zero).
252//-------------------------------------------------------------------
253
254
255// Do we have __builtin_thread_pointer? This would be the preferred way to get a unique thread id
256// but unfortunately, it seems we cannot test for this reliably at this time (see issue #883)
257// Nevertheless, it seems needed on older graviton platforms (see issue #851).
258// For now, we only enable this for specific platforms.
259#if !defined(__APPLE__) /* on apple (M1) the wrong register is read (tpidr_el0 instead of tpidrro_el0) so fall back to TLS slot assembly (<https://github.com/microsoft/mimalloc/issues/343#issuecomment-763272369>)*/ \
260 && !defined(__CYGWIN__) \
261 && !defined(MI_LIBC_MUSL) \
262 && (!defined(__clang_major__) || __clang_major__ >= 14) /* older clang versions emit bad code; fall back to using the TLS slot (<https://lore.kernel.org/linux-arm-kernel/202110280952.352F66D8@keescook/T/>) */
263 #if (defined(__GNUC__) && (__GNUC__ >= 7) && defined(__aarch64__)) /* aarch64 for older gcc versions (issue #851) */ \
264 || (defined(__GNUC__) && (__GNUC__ >= 11) && defined(__x86_64__)) \
265 || (defined(__clang_major__) && (__clang_major__ >= 14) && (defined(__aarch64__) || defined(__x86_64__)))
266 #define MI_USE_BUILTIN_THREAD_POINTER 1
267 #endif
268#endif

332/* ----------------------------------------------------------------------------------------
333Get the thread local default heap: `_mi_prim_get_default_heap()`
334
335This is inlined here as it is on the fast path for allocation functions.
336
337On most platforms (Windows, Linux, FreeBSD, NetBSD, etc), this just returns a
338__thread local variable (`_mi_heap_default`). With the initial-exec TLS model this ensures
339that the storage will always be available (allocated on the thread stacks).
340
341On some platforms though we cannot use that when overriding `malloc` since the underlying
342TLS implementation (or the loader) will call itself `malloc` on a first access and recurse.
343We try to circumvent this in an efficient way:
344- macOSX : we use an unused TLS slot from the OS allocated slots (MI_TLS_SLOT). On OSX, the
345 loader itself calls `malloc` even before the modules are initialized.
346- OpenBSD: we use an unused slot from the pthread block (MI_TLS_PTHREAD_SLOT_OFS).
347- DragonFly: defaults are working but seem slow compared to freeBSD (see PR #323)
348------------------------------------------------------------------------------------------- */
349
350static inline mi_heap_t* mi_prim_get_default_heap(void);
351
352#if defined(MI_MALLOC_OVERRIDE)
353#if defined(__APPLE__) // macOS
354 #define MI_TLS_SLOT 89 // seems unused?
355 // other possible unused ones are 9, 29, __PTK_FRAMEWORK_JAVASCRIPTCORE_KEY4 (94), __PTK_FRAMEWORK_GC_KEY9 (112) and __PTK_FRAMEWORK_OLDGC_KEY9 (89)
356 // see <https://github.com/rweichler/substrate/blob/master/include/pthread_machdep.h>
357#elif defined(__OpenBSD__)
358 // use end bytes of a name; goes wrong if anyone uses names > 23 characters (ptrhread specifies 16)
359 // see <https://github.com/openbsd/src/blob/master/lib/libc/include/thread_private.h#L371>
360 #define MI_TLS_PTHREAD_SLOT_OFS (6*sizeof(int) + 4*sizeof(void*) + 24)
361 // #elif defined(__DragonFly__)
362 // #warning "mimalloc is not working correctly on DragonFly yet."
363 // #define MI_TLS_PTHREAD_SLOT_OFS (4 + 1*sizeof(void*)) // offset `uniqueid` (also used by gdb?) <https://github.com/DragonFlyBSD/DragonFlyBSD/blob/master/lib/libthread_xu/thread/thr_private.h#L458>
364#elif defined(__ANDROID__)
365 // See issue #381
366 #define MI_TLS_PTHREAD
367#endif
368#endif

372# if !defined(MI_HAS_TLS_SLOT)
373# error "trying to use a TLS slot for the default heap, but the mi_prim_tls_slot primitives are not defined"

376static inline mi_heap_t* mi_prim_get_default_heap(void) {
377 mi_heap_t* heap = (mi_heap_t*)mi_prim_tls_slot(MI_TLS_SLOT);
378 #if MI_HAS_TLS_SLOT == 1 // check if the TLS slot is initialized
379 if mi_unlikely(heap == NULL) {
380 #ifdef __GNUC__
381 __asm(""); // prevent conditional load of the address of _mi_heap_empty
382 #endif
383 heap = (mi_heap_t*)&_mi_heap_empty;
384 }
385 #endif
386 return heap;
387}

389#elif defined(MI_TLS_PTHREAD_SLOT_OFS)

391static inline mi_heap_t** mi_prim_tls_pthread_heap_slot(void) {
392 pthread_t self = pthread_self();
393 #if defined(__DragonFly__)
394 if (self==NULL) return NULL;
395 #endif
396 return (mi_heap_t**)((uint8_t*)self + MI_TLS_PTHREAD_SLOT_OFS);
397}

402 mi_heap_t* heap = *pheap;

404 return heap;
405}
