267 void do_test( Map& testMap )
268 {
269 typedef Inserter<Map> insert_thread;
270 typedef Extractor< typename Map::gc, Map > extract_thread;
271
272 m_nInsThreadCount.store( s_nInsThreadCount, atomics::memory_order_release );
273
274 {
275 std::vector<key_type> arr;
276 arr.resize( s_nMapSize );
277 for ( int i = 0; i < static_cast<int>( s_nMapSize ); ++i )
278 arr[i] = i;;
279 shuffle( arr.begin(), arr.end());
280
281 for ( key_type key : arr )
282 testMap.insert( key, key );
283 }
284
285 m_KeyMin = 0;
286 m_KeyMax = static_cast<int>( s_nMapSize - 1 );
287
288 cds_test::thread_pool& pool = get_pool();
289 pool.add( new insert_thread( pool, testMap ), s_nInsThreadCount );
290 pool.add( new extract_thread( pool, testMap ), s_nExtractThreadCount );
291
292 propout() << std::make_pair( "insert_thread_count", s_nInsThreadCount )
293 << std::make_pair( "extract_thread_count", s_nExtractThreadCount )
294 << std::make_pair( "map_size", s_nMapSize )
295 << std::make_pair( "pass_count", s_nPassCount );
296
297 std::chrono::milliseconds duration = pool.run();
298
299 propout() << std::make_pair( "duration", duration );
300
301 size_t nInsertMinSuccess = 0;
302 size_t nInsertMinFailed = 0;
303 size_t nInsertMaxSuccess = 0;
304 size_t nInsertMaxFailed = 0;
305 size_t nDeleteMin = 0;
306 size_t nDeleteMinFailed = 0;
307 size_t nDeleteMax = 0;
308 size_t nDeleteMaxFailed = 0;
309
310 for ( size_t i = 0; i < pool.size(); ++i ) {
311 cds_test::thread& thr = pool.get( i );
312 switch ( thr.type()) {
313 case inserter_thread:
314 {
315 insert_thread& inserter = static_cast<insert_thread&>(thr);
316 nInsertMinSuccess += inserter.m_nInsertMinSuccess;
317 nInsertMinFailed += inserter.m_nInsertMinFailed;
318 nInsertMaxSuccess += inserter.m_nInsertMaxSuccess;
319 nInsertMaxFailed += inserter.m_nInsertMaxFailed;
320 }
321 break;
322 case extractor_thread:
323 {
324 extract_thread& extractor = static_cast<extract_thread&>(thr);
325 nDeleteMin += extractor.m_nDeleteMin;
326 nDeleteMinFailed += extractor.m_nDeleteMinFailed;
327 nDeleteMax += extractor.m_nDeleteMax;
328 nDeleteMaxFailed += extractor.m_nDeleteMaxFailed;
329 }
330 break;
331 default:
332 assert( false );
333 }
334 }
335
336 EXPECT_EQ( nDeleteMinFailed, 0u );
337 EXPECT_EQ( nDeleteMaxFailed, 0u );
338
339 EXPECT_EQ( nDeleteMin, nInsertMinSuccess + 1 );
340 EXPECT_EQ( nDeleteMax, nInsertMaxSuccess + 1 );
341
342 propout()
343 << std::make_pair( "insert_min", nInsertMinSuccess + 1 )
344 << std::make_pair( "insert_min_double", nInsertMinFailed )
345 << std::make_pair( "insert_max", nInsertMaxSuccess + 1 )
346 << std::make_pair( "insert_max_double", nInsertMaxFailed )
347 << std::make_pair( "extract_min", nDeleteMin )
348 << std::make_pair( "extract_min_failed", nDeleteMinFailed )
349 << std::make_pair( "extract_max", nDeleteMax )
350 << std::make_pair( "extract_max_failed", nDeleteMaxFailed );
351
352 analyze( testMap );
353 }
