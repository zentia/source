387 void fc_apply( fc_record * pRec )
388 {
389 assert( pRec );
390
391 switch ( pRec->op()) {
392 case op_push_front:
393 assert( pRec->pValPush );
394 m_Deque.push_front( *(pRec->pValPush));
395 break;
396 case op_push_front_move:
397 assert( pRec->pValPush );
398 m_Deque.push_front( std::move( *(pRec->pValPush )));
399 break;
400 case op_push_back:
401 assert( pRec->pValPush );
402 m_Deque.push_back( *(pRec->pValPush));
403 break;
404 case op_push_back_move:
405 assert( pRec->pValPush );
406 m_Deque.push_back( std::move( *(pRec->pValPush )));
407 break;
408 case op_pop_front:
409 assert( pRec->pValPop );
410 pRec->bEmpty = m_Deque.empty();
411 if ( !pRec->bEmpty ) {
412 *(pRec->pValPop) = std::move( m_Deque.front());
413 m_Deque.pop_front();
414 }
415 break;
416 case op_pop_back:
417 assert( pRec->pValPop );
418 pRec->bEmpty = m_Deque.empty();
419 if ( !pRec->bEmpty ) {
420 *(pRec->pValPop) = std::move( m_Deque.back());
421 m_Deque.pop_back();
422 }
423 break;
424 case op_clear:
425 while ( !m_Deque.empty())
426 m_Deque.pop_front();
427 break;
428 default:
429 assert(false);
430 break;
431 }

435 void fc_process( typename fc_kernel::iterator itBegin, typename fc_kernel::iterator itEnd )
436 {
437 typedef typename fc_kernel::iterator fc_iterator;
438
439 for ( fc_iterator it = itBegin, itPrev = itEnd; it != itEnd; ++it ) {
440 switch ( it->op( atomics::memory_order_acquire )) {
441 case op_push_front:
442 if ( itPrev != itEnd
443 && (itPrev->op() == op_pop_front || (m_Deque.empty() && itPrev->op() == op_pop_back)))
444 {
445 collide( *it, *itPrev );
446 itPrev = itEnd;
447 }
448 else
449 itPrev = it;
450 break;
451 case op_push_front_move:
452 if ( itPrev != itEnd
453 && (itPrev->op() == op_pop_front || ( m_Deque.empty() && itPrev->op() == op_pop_back )))
454 {
455 collide_move( *it, *itPrev );
456 itPrev = itEnd;
457 }
458 else
459 itPrev = it;
460 break;
461 case op_push_back:
462 if ( itPrev != itEnd
463 && (itPrev->op() == op_pop_back || (m_Deque.empty() && itPrev->op() == op_pop_front)))
464 {
465 collide( *it, *itPrev );
466 itPrev = itEnd;
467 }
468 else
469 itPrev = it;
470 break;
471 case op_push_back_move:
472 if ( itPrev != itEnd
473 && (itPrev->op() == op_pop_back || ( m_Deque.empty() && itPrev->op() == op_pop_front )))
474 {
475 collide_move( *it, *itPrev );
476 itPrev = itEnd;
477 }
478 else
479 itPrev = it;
480 break;
481 case op_pop_front:
482 if ( itPrev != itEnd ) {
483 if ( m_Deque.empty()) {
484 switch ( itPrev->op()) {
485 case op_push_back:
486 collide( *itPrev, *it );
487 itPrev = itEnd;
488 break;
489 case op_push_back_move:
490 collide_move( *itPrev, *it );
491 itPrev = itEnd;
492 break;
493 default:
494 itPrev = it;
495 break;
496 }
497 }
498 else {
499 switch ( itPrev->op()) {
500 case op_push_front:
501 collide( *itPrev, *it );
502 itPrev = itEnd;
503 break;
504 case op_push_front_move:
505 collide_move( *itPrev, *it );
506 itPrev = itEnd;
507 break;
508 default:
509 itPrev = it;
510 break;
511 }
512 }
513 }
514 else
515 itPrev = it;
516 break;
517 case op_pop_back:
518 if ( itPrev != itEnd ) {
519 if ( m_Deque.empty()) {
520 switch ( itPrev->op()) {
521 case op_push_front:
522 collide( *itPrev, *it );
523 itPrev = itEnd;
524 break;
525 case op_push_front_move:
526 collide_move( *itPrev, *it );
527 itPrev = itEnd;
528 break;
529 default:
530 itPrev = it;
531 break;
532 }
533 }
534 else {
535 switch ( itPrev->op()) {
536 case op_push_back:
537 collide( *itPrev, *it );
538 itPrev = itEnd;
539 break;
540 case op_push_back_move:
541 collide_move( *itPrev, *it );
542 itPrev = itEnd;
543 break;
544 default:
545 itPrev = it;
546 break;
547 }
548 }
549 }
550 else
551 itPrev = it;
552 break;
553 }
554 }
555 }

559 //@cond
560 void collide( fc_record& recPush, fc_record& recPop )
561 {
562 *(recPop.pValPop) = *(recPush.pValPush);
563 recPop.bEmpty = false;
564 m_FlatCombining.operation_done( recPush );
565 m_FlatCombining.operation_done( recPop );
566 m_FlatCombining.internal_statistics().onCollide();
567 }
568
569 void collide_move( fc_record& recPush, fc_record& recPop )
570 {
571 *(recPop.pValPop) = std::move( *(recPush.pValPush));
572 recPop.bEmpty = false;
573 m_FlatCombining.operation_done( recPush );
574 m_FlatCombining.operation_done( recPop );
575 m_FlatCombining.internal_statistics().onCollide();
576 }
