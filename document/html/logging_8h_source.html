63 std::string error_message;
64 if constexpr(sizeof...(args) == 1u) {
65 error_message = std::string{std::forward<Args>(args)...};
66 } else {
67 error_message = fmt::format(std::forward<Args>(args)...);
68 }
69 auto trace = luisa::backtrace();
70 for (auto i = 0u; i < trace.size(); i++) {
71 auto &&t = trace[i];
72 fmt::format_to(std::back_inserter(error_message),
73 FMT_STRING("\n {:>2} {}"), i, t);
74 }
75 detail::default_logger().error(error_message);
76 std::abort();
77}
