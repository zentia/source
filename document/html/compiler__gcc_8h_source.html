92#define EASTL_GCC_ATOMIC_FETCH_INTRIN_N(integralType, fetchIntrinsic, type, ret, ptr, val, gccMemoryOrder) \
93 { \
94 integralType retIntegral; \
95 integralType valIntegral = EASTL_ATOMIC_TYPE_PUN_CAST(integralType, (val)); \
96 \
97 retIntegral = fetchIntrinsic(EASTL_ATOMIC_VOLATILE_INTEGRAL_CAST(integralType, (ptr)), valIntegral, gccMemoryOrder); \
98 \
99 ret = EASTL_ATOMIC_TYPE_PUN_CAST(type, retIntegral); \
100 }

102#define EASTL_GCC_ATOMIC_CMPXCHG_INTRIN_N(integralType, type, ret, ptr, expected, desired, weak, successOrder, failOrder) \
103 ret = __atomic_compare_exchange(EASTL_ATOMIC_VOLATILE_INTEGRAL_CAST(integralType, (ptr)), \
104 EASTL_ATOMIC_INTEGRAL_CAST(integralType, (expected)), \
105 EASTL_ATOMIC_INTEGRAL_CAST(integralType, &(desired)), \
106 weak, successOrder, failOrder)
107
108#define EASTL_GCC_ATOMIC_EXCHANGE_INTRIN_N(integralType, type, ret, ptr, val, gccMemoryOrder) \
109 { \
110 integralType retIntegral; \
111 integralType valIntegral = EASTL_ATOMIC_TYPE_PUN_CAST(integralType, (val)); \
112 \
113 __atomic_exchange(EASTL_ATOMIC_VOLATILE_INTEGRAL_CAST(integralType, (ptr)), \
114 &valIntegral, &retIntegral, gccMemoryOrder); \
115 \
116 ret = EASTL_ATOMIC_TYPE_PUN_CAST(type, retIntegral); \
117 }
