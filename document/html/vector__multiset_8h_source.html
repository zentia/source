6// This file implements vector_multiset. It acts much like std::multiset, except
7// its underlying representation is a random access container such as vector.
8// These containers are sometimes also known as "sorted vectors."
9// vector_sets have an advantage over conventional sets in that their memory
10// is contiguous and node-less. The result is that lookups are faster, more
11// cache friendly (which potentially more so benefits speed), and the container
12// uses less memory. The downside is that inserting new items into the container
13// is slower if they are inserted in random order instead of in sorted order.
14// This tradeoff is well-worth it for many cases. Note that vector_multiset allows
15// you to use a deque or other random access container which may perform
16// better for you than vector.
17//
18// Note that with vector_set, vector_multiset, vector_map, vector_multimap
19// that the modification of the container potentially invalidates all
20// existing iterators into the container, unlike what happens with conventional
21// sets and maps.
22//
23// This type could conceptually use a eastl::array as its underlying container,
24// however the current design requires an allocator aware container.
25// Consider using a fixed_vector instead.

221 eastl::pair<iterator, iterator> equal_range_small(const key_type& k)
222 {
223 // Defined inline because VC7.1 is broken for when it's defined outside.
224 const iterator itLower(lower_bound(k));
225 iterator itUpper(itLower);
226
227 while((itUpper != end()) && !value_compare::operator()(k, *itUpper))
228 ++itUpper;
229
230 return eastl::pair<iterator, iterator>(itLower, itUpper);

241 // NOTE(rparolin): It is undefined behaviour if user code fails to ensure the container
242 // invariants are respected by performing an explicit call to 'sort' before any other
243 // operations on the container are performed that do not clear the elements.
244 //
245 // 'push_back_unsorted' and 'emplace_back_unsorted' do not satisfy container invariants
246 // for being sorted. We provide these overloads explicitly labelled as '_unsorted' as an
247 // optimization opportunity when batch inserting elements so users can defer the cost of
248 // sorting the container once when all elements are contained. This was done to clarify
249 // the intent of code by leaving a trace that a manual call to sort is required.
250 //
251 template <typename... Args> decltype(auto) push_back_unsorted(Args&&... args)
252 { return base_type::push_back(eastl::forward<Args>(args)...); }
253 template <typename... Args> decltype(auto) emplace_back_unsorted(Args&&... args)
254 { return base_type::emplace_back(eastl::forward<Args>(args)...); }

467 template <typename K, typename C, typename A, typename RAC>
468 inline typename vector_multiset<K, C, A, RAC>::iterator
469 vector_multiset<K, C, A, RAC>::insert(const_iterator position, const value_type& value)
470 {
471 // We assume that the user knows what he is doing and has supplied us with
472 // a position that is right where value should be inserted (put in front of).
473 // We do a test to see if the position is correct. If so then we insert,
474 // if not then we ignore the input position. However,
475
476 if((position == end()) || !value_compare::operator()(*position, value)) // If value is <= the element at position...
477 {
478 if((position == begin()) || !value_compare::operator()(value, *(position - 1))) // If value is >= the element before position...
479 return base_type::insert(position, value);
480 }
481
482 // In this case we have an incorrect position. We fall back to the regular insert function.
483 return insert(value);
484 }
