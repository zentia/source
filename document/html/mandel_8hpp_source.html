44// Computes the value of the point (px, py), which represents the complex
45// number (px + i*py), based on whether it is part of the Mandelbrot set or not
46inline int escape_time(double px, double py, int n) {
47 // x and y represent a complex number (x + i*y)
48 double x = px;
49 double y = py;
50
51 // square both components
52 double x2 = x * x;
53 double y2 = y * y;
54
56 //double cr = -0.8;
57 //double ci = 0.156;
58
59 // Mandelbrot set:
60 double cr = px;
61 double ci = py;
62
63 double xtmp;
64
65 // We need i (number of iteration) after the loop, so don't declare the variable inside the for.
66 int i;
67 for (i = 0; (x2 + y2) <= 4.0 && i < MAX_ITERATION; i++) {
68 xtmp = pow((x*x+y*y), (n/2.0))*cos(n*atan2(y,x)) + cr;
69 y = pow((x*x+y*y), (n/2.0))*sin(n*atan2(y,x)) + ci;
70 x = xtmp;
71
72 // Update our temp variables
73 x2 = x * x;
74 y2 = y * y;
75 }
76
77 return i;
78}
