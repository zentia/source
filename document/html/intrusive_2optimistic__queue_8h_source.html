458 struct dequeue_result {
459 typename gc::template GuardArray<3> guards;
460
461 node_type * pHead;
462 node_type * pNext;
463 };
464
465 bool do_dequeue( dequeue_result& res )
466 {
467 node_type * pTail;
468 node_type * pHead;
469 node_type * pFirstNodePrev;
470 back_off bkoff;
471
472 while ( true ) { // Try till success or empty
473 pHead = res.guards.protect( 0, m_pHead, [](node_type * p) -> value_type * {return node_traits::to_value_ptr(p);});
474 pTail = res.guards.protect( 1, m_pTail, [](node_type * p) -> value_type * {return node_traits::to_value_ptr(p);});
475 assert( pHead != nullptr );
476 pFirstNodePrev = res.guards.protect( 2, pHead->m_pPrev, [](node_type * p) -> value_type * {return node_traits::to_value_ptr(p);});
477
478 if ( pHead == m_pHead.load(memory_model::memory_order_acquire)) {
479 if ( pTail != pHead ) {
480 if ( pFirstNodePrev == nullptr
481 || pFirstNodePrev->m_pNext.load(memory_model::memory_order_acquire) != pHead )
482 {
483 fix_list( pTail, pHead );
484 continue;
485 }
486 if ( m_pHead.compare_exchange_weak( pHead, pFirstNodePrev, memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
487 // dequeue success
488 break;
489 }
490 }
491 else {
492 // the queue is empty
493 m_Stat.onEmptyDequeue();
494 return false;
495 }
496 }
497
498 m_Stat.onDequeueRace();
499 bkoff();
500 }
501
502 --m_ItemCounter;
503 m_Stat.onDequeue();
504
505 res.pHead = pHead;
506 res.pNext = pFirstNodePrev;
507 return true;
508 }
509
510
512 void fix_list( node_type * pTail, node_type * pHead )
513 {
514 // pTail and pHead are already guarded
515
516 node_type * pCurNode;
517 node_type * pCurNodeNext;
518
519 typename gc::template GuardArray<2> guards;
520
521 pCurNode = pTail;
522 while ( pCurNode != pHead ) { // While not at head
523 pCurNodeNext = guards.protect(0, pCurNode->m_pNext, [](node_type * p) -> value_type * { return node_traits::to_value_ptr(p);});
524 if ( pHead != m_pHead.load(memory_model::memory_order_acquire))
525 break;
526 pCurNodeNext->m_pPrev.store( pCurNode, memory_model::memory_order_release );
527 guards.assign( 1, node_traits::to_value_ptr( pCurNode = pCurNodeNext ));
528 }
529
530 m_Stat.onFixList();
531 }

533 void dispose_result( dequeue_result& res )

538 void dispose_node( node_type * p )
539 {
540 assert( p != nullptr );
541
542 if ( p != &m_Dummy ) {
543 struct internal_disposer
544 {
545 void operator ()( value_type * p )
546 {
547 assert( p != nullptr );
548
549 OptimisticQueue::clear_links( node_traits::to_node_ptr( *p ));
550 disposer()(p);
551 }
552 };
553 gc::template retire<internal_disposer>( node_traits::to_value_ptr(p));
554 }
555 }

581 bool enqueue( value_type& val )
582 {
583 node_type * pNew = node_traits::to_node_ptr( val );
584 link_checker::is_empty( pNew );
585
586 typename gc::template GuardArray<2> guards;
587 back_off bkoff;
588
589 guards.assign( 1, &val );
590 while( true ) {
591 node_type * pTail = guards.protect( 0, m_pTail, []( node_type * p ) -> value_type * { return node_traits::to_value_ptr( p ); } ); // Read the tail
592 pNew->m_pNext.store( pTail, memory_model::memory_order_relaxed );
593 if ( m_pTail.compare_exchange_strong( pTail, pNew, memory_model::memory_order_release, atomics::memory_order_acquire )) { // Try to CAS the tail
594 pTail->m_pPrev.store( pNew, memory_model::memory_order_release ); // Success, write prev
595 ++m_ItemCounter;
596 m_Stat.onEnqueue();
597 break; // Enqueue done!
598 }
599 m_Stat.onEnqueueRace();
600 bkoff();
601 }
602 return true;
603 }
