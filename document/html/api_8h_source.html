69typedef void * SceneHandle;
70
71typedef struct {
72 int texturefunc; /* which texture function to use */
73 color col; /* base object color */
74 int shadowcast; /* does the object cast a shadow */
75 apiflt ambient; /* ambient lighting */
76 apiflt diffuse; /* diffuse reflection */
77 apiflt specular; /* specular reflection */
78 apiflt opacity; /* how opaque the object is */
79 vector ctr; /* origin of texture */
80 vector rot; /* rotation of texture around origin */
81 vector scale; /* scale of texture in x,y,z */
82 vector uaxs; /* planar map u axis */
83 vector vaxs; /* planar map v axis */
84 char imap[96]; /* name of image map */
85} apitexture;
86
87
88/*******************************************************************
89 * NOTE: The value passed in apitexture.texturefunc corresponds to
90 * the meanings given in this table:
91 *
92 * 0 - No texture function is applied other than standard lighting.
93 * 1 - 3D checkerboard texture. Red & Blue checkers through 3d space.
94 * 2 - Grit texture, roughens up the surface of the object a bit.
95 * 3 - 3D marble texture. Makes a 3D swirl pattern through the object.
96 * 4 - 3D wood texture. Makes a 3D wood pattern through the object.
97 * 5 - 3D gradient noise function.
98 * 6 - I've forgotten :-)
99 * 7 - Cylindrical Image Map **** IMAGE MAPS REQUIRE the filename
100 * 8 - Spherical Image Map of the image be put in imap[]
101 * 9 - Planar Image Map part of the texture...
102 * planar requires uaxs, and vaxs..
103 *
104 *******************************************************************/
