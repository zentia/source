6// This file implements the eastl::span which is part of the C++ standard
7// STL library specification.
8//
9// eastl::span is a non-owning container that refers to a contiguous block of
10// memory. It bundles up the classic pattern of a pointer and a size into a
11// single type. A span can either have a static extent, in which case the
12// number of elements in the sequence is known and encoded in the type, or a
13// dynamic extent.
14//
15// http://en.cppreference.com/w/cpp/container/span
16// http://eel.is/c++draft/views#span.syn

37 // HasSizeAndData
38 //
39 // custom type trait to determine if eastl::data(Container) and eastl::size(Container) are well-formed.
40 //
41 template <typename, typename = void>
42 struct HasSizeAndData : eastl::false_type {};

44 template <typename T>
45 struct HasSizeAndData<T, void_t<decltype(eastl::size(eastl::declval<T>())), decltype(eastl::data(eastl::declval<T>()))>> : eastl::true_type {};

47 // SubspanExtent
48 //
49 // Integral constant that calculates the resulting extent of a templated subspan operation.
50 //
51 // If Count is not dynamic_extent then SubspanExtent::value is Count,
52 // otherwise, if Extent is not dynamic_extent, SubspanExtent::value is (Extent - Offset),
53 // otherwise, SubspanExtent::value is dynamic_extent.
54 //
55 template<size_t Extent, size_t Offset, size_t Count>
56 struct SubspanExtent : eastl::integral_constant<size_t, (Count != dynamic_extent ? Count : (Extent != dynamic_extent ? (Extent - Offset) : dynamic_extent))> {};

58 // is_eastl_array<PossiblyArray, T>
59 //
60 // is an eastl::array<T, N> with elements T?
61 // NOT the same as is_array, which is a type trait for the C-style array type.
62 template<typename PossiblyArray, typename T>
63 struct is_eastl_array : public eastl::false_type {};
64
65 template<typename T, size_t N>
66 struct is_eastl_array<eastl::array<T, N>, T> : public eastl::true_type {};
