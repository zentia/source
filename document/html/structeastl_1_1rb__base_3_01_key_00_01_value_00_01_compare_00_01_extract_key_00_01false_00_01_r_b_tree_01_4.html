template<typename Key, typename Value, typename Compare, typename ExtractKey, typename RBTree>
struct eastl::rb_base< Key, Value, Compare, ExtractKey, false, RBTree >

rb_base This class is used for 'multiset'. In this case, Key and Value will be the same as each other and ExtractKey will be eastl::use_self.
