59 void make_tower( unsigned int nHeight, atomic_marked_ptr * nextTower )
60 {
61 assert( nHeight > 0 );
62 assert( (nHeight == 1 && nextTower == nullptr) // bottom-list node
63 || (nHeight > 1 && nextTower != nullptr) // node at level of more than 0
64 );
65
66 m_arrNext = nextTower;
67 m_nHeight = nHeight;
68 m_nUnlink.store( nHeight, atomics::memory_order_release );
69 }

74 m_arrNext = nullptr;

157 template <class RCU, typename NodeTraits, typename BackOff, bool IsConst>
158 class iterator< cds::urcu::gc< RCU >, NodeTraits, BackOff, IsConst >
159 {
160 public:
161 typedef cds::urcu::gc< RCU > gc;
162 typedef NodeTraits node_traits;
163 typedef BackOff back_off;
164 typedef typename node_traits::node_type node_type;
165 typedef typename node_traits::value_type value_type;
166 static bool const c_isConst = IsConst;
167
168 typedef typename std::conditional< c_isConst, value_type const &, value_type &>::type value_ref;
169
170 protected:
171 typedef typename node_type::marked_ptr marked_ptr;
172 typedef typename node_type::atomic_marked_ptr atomic_marked_ptr;
173
174 node_type * m_pNode;
175
176 protected:
177 void next()
178 {
179 back_off bkoff;
180
181 for (;;) {
182 if ( m_pNode->next( m_pNode->height() - 1 ).load( atomics::memory_order_acquire ).bits()) {
183 // Current node is marked as deleted. So, its next pointer can point to anything
184 // In this case we interrupt our iteration and returns end() iterator.
185 *this = iterator();
186 return;
187 }
188
189 marked_ptr p = m_pNode->next(0).load( atomics::memory_order_relaxed );
190 node_type * pp = p.ptr();
191 if ( p.bits()) {
192 // p is marked as deleted. Spin waiting for physical removal
193 bkoff();
194 continue;
195 }
196 else if ( pp && pp->next( pp->height() - 1 ).load( atomics::memory_order_relaxed ).bits()) {
197 // p is marked as deleted. Spin waiting for physical removal
198 bkoff();
199 continue;
200 }
201
202 m_pNode = pp;
203 break;
204 }
205 }
206
207 public: // for internal use only!!!
208 iterator( node_type& refHead )
209 : m_pNode( nullptr )
210 {
211 back_off bkoff;
212
213 for (;;) {
214 marked_ptr p = refHead.next(0).load( atomics::memory_order_relaxed );
215 if ( !p.ptr()) {
216 // empty skip-list
217 break;
218 }
219
220 node_type * pp = p.ptr();
221 // Logically deleted node is marked from highest level
222 if ( !pp->next( pp->height() - 1 ).load( atomics::memory_order_acquire ).bits()) {
223 m_pNode = pp;
224 break;
225 }
226
227 bkoff();
228 }
229 }
230
231 public:
232 iterator()
233 : m_pNode( nullptr )
234 {}
235
236 iterator( iterator const& s)
237 : m_pNode( s.m_pNode )
238 {}
239
240 value_type * operator ->() const
241 {
242 assert( m_pNode != nullptr );
243 assert( node_traits::to_value_ptr( m_pNode ) != nullptr );
244
245 return node_traits::to_value_ptr( m_pNode );
246 }
247
248 value_ref operator *() const
249 {
250 assert( m_pNode != nullptr );
251 assert( node_traits::to_value_ptr( m_pNode ) != nullptr );
252
253 return *node_traits::to_value_ptr( m_pNode );
254 }
255
257 iterator& operator ++()
258 {
259 next();
260 return *this;
261 }
262
263 iterator& operator = (const iterator& src)
264 {
265 m_pNode = src.m_pNode;
266 return *this;
267 }

276 {

757 template <typename Func>
758 bool insert( value_type& val, Func f )
759 {
760 check_deadlock_policy::check();
761
762 position pos;
763 bool bRet;
764
765 {
766 node_type * pNode = node_traits::to_node_ptr( val );
767 scoped_node_ptr scp( pNode );
768 unsigned int nHeight = pNode->height();
769 bool bTowerOk = nHeight > 1 && pNode->get_tower() != nullptr;
770 bool bTowerMade = false;
771
772 rcu_lock rcuLock;
773
774 while ( true )
775 {
776 bool bFound = find_position( val, pos, key_comparator(), true );
777 if ( bFound ) {
778 // scoped_node_ptr deletes the node tower if we create it
779 if ( !bTowerMade )
780 scp.release();
781
782 m_Stat.onInsertFailed();
783 bRet = false;
784 break;
785 }
786
787 if ( !bTowerOk ) {
788 build_node( pNode );
789 nHeight = pNode->height();
790 bTowerMade =
791 bTowerOk = true;
792 }
793
794 if ( !insert_at_position( val, pNode, pos, f )) {
795 m_Stat.onInsertRetry();
796 continue;
797 }
798
799 increase_height( nHeight );
800 ++m_ItemCounter;
801 m_Stat.onAddNode( nHeight );
802 m_Stat.onInsertSuccess();
803 scp.release();
804 bRet = true;
805 break;
806 }
807 }
808
809 return bRet;
810 }

842 template <typename Func>
843 std::pair<bool, bool> update( value_type& val, Func func, bool bInsert = true )
844 {
845 check_deadlock_policy::check();
846
847 position pos;
848 std::pair<bool, bool> bRet( true, false );
849
850 {
851 node_type * pNode = node_traits::to_node_ptr( val );
852 scoped_node_ptr scp( pNode );
853 unsigned int nHeight = pNode->height();
854 bool bTowerOk = nHeight > 1 && pNode->get_tower() != nullptr;
855 bool bTowerMade = false;
856
857 rcu_lock rcuLock;
858 while ( true )
859 {
860 bool bFound = find_position( val, pos, key_comparator(), true );
861 if ( bFound ) {
862 // scoped_node_ptr deletes the node tower if we create it before
863 if ( !bTowerMade )
864 scp.release();
865
866 func( false, *node_traits::to_value_ptr(pos.pCur), val );
867 m_Stat.onUpdateExist();
868 break;
869 }
870
871 if ( !bInsert ) {
872 scp.release();
873 bRet.first = false;
874 break;
875 }
876
877 if ( !bTowerOk ) {
878 build_node( pNode );
879 nHeight = pNode->height();
880 bTowerMade =
881 bTowerOk = true;
882 }
883
884 if ( !insert_at_position( val, pNode, pos, [&func]( value_type& item ) { func( true, item, item ); })) {
885 m_Stat.onInsertRetry();
886 continue;
887 }
888
889 increase_height( nHeight );
890 ++m_ItemCounter;
891 scp.release();
892 m_Stat.onAddNode( nHeight );
893 m_Stat.onUpdateNew();
894 bRet.second = true;
895 break;
896 }
897 }
898
899 return bRet;
