120static bool unix_detect_overcommit(void) {
121 bool os_overcommit = true;
122#if defined(__linux__)
123 int fd = mi_prim_open("/proc/sys/vm/overcommit_memory", O_RDONLY);
124 if (fd >= 0) {
125 char buf[32];
126 ssize_t nread = mi_prim_read(fd, &buf, sizeof(buf));
127 mi_prim_close(fd);
128 // <https://www.kernel.org/doc/Documentation/vm/overcommit-accounting>
129 // 0: heuristic overcommit, 1: always overcommit, 2: never overcommit (ignore NORESERVE)
130 if (nread >= 1) {
131 os_overcommit = (buf[0] == '0' || buf[0] == '1');
132 }
133 }
134#elif defined(__FreeBSD__)
135 int val = 0;
136 size_t olen = sizeof(val);
137 if (sysctlbyname("vm.overcommit", &val, &olen, NULL, 0) == 0) {
138 os_overcommit = (val != 0);
139 }
140#else
141 // default: overcommit is true
142#endif
143 return os_overcommit;
144}

146void _mi_prim_mem_init( mi_os_mem_config_t* config )
147{
148 long psize = sysconf(_SC_PAGESIZE);
149 if (psize > 0) {
150 config->page_size = (size_t)psize;
151 config->alloc_granularity = (size_t)psize;
152 #if defined(_SC_PHYS_PAGES)
153 long pphys = sysconf(_SC_PHYS_PAGES);
154 const size_t psize_in_kib = (size_t)psize / MI_KiB;
155 if (psize_in_kib > 0 && pphys > 0 && (size_t)pphys <= (SIZE_MAX/psize_in_kib)) {
156 config->physical_memory_in_kib = (size_t)pphys * psize_in_kib;
157 }
158 #endif
159 }
160 config->large_page_size = MI_UNIX_LARGE_PAGE_SIZE;
161 config->has_overcommit = unix_detect_overcommit();
162 config->has_partial_free = true; // mmap can free in parts
163 config->has_virtual_reserve = true; // todo: check if this true for NetBSD? (for anonymous mmap with PROT_NONE)
164
165 // disable transparent huge pages for this process?
166 #if (defined(__linux__) || defined(__ANDROID__)) && defined(PR_GET_THP_DISABLE)
167 #if defined(MI_NO_THP)
168 if (true)
169 #else
170 if (!mi_option_is_enabled(mi_option_allow_large_os_pages)) // disable THP also if large OS pages are not allowed in the options
171 #endif
172 {
173 int val = 0;
174 if (prctl(PR_GET_THP_DISABLE, &val, 0, 0, 0) != 0) {
175 // Most likely since distros often come with always/madvise settings.
176 val = 1;
177 // Disabling only for mimalloc process rather than touching system wide settings
178 (void)prctl(PR_SET_THP_DISABLE, &val, 0, 0, 0);
179 }
180 }
181 #endif
182}

210static void* unix_mmap_prim(void* addr, size_t size, size_t try_alignment, int protect_flags, int flags, int fd) {
211 MI_UNUSED(try_alignment);
212 void* p = NULL;
213 #if defined(MAP_ALIGNED) // BSD
214 if (addr == NULL && try_alignment > 1 && (try_alignment % _mi_os_page_size()) == 0) {
215 size_t idx;
216 size_t n = mi_bsr(try_alignment, &idx);
217 if (((size_t)1 << n) == try_alignment && n >= 12 && n <= 30) { // alignment is a power of 2 and 4096 <= alignment <= 1GiB
218 p = mmap(addr, size, protect_flags, flags | MAP_ALIGNED(n), fd, 0);
219 if (p==MAP_FAILED || !_mi_is_aligned(p,try_alignment)) {
220 int err = errno;
221 _mi_trace_message("unable to directly request aligned OS memory (error: %d (0x%x), size: 0x%zx bytes, alignment: 0x%zx, hint address: %p)\n", err, err, size, try_alignment, addr);
222 }
223 if (p!=MAP_FAILED) return p;
224 // fall back to regular mmap
225 }
226 }
227 #elif defined(MAP_ALIGN) // Solaris
228 if (addr == NULL && try_alignment > 1 && (try_alignment % _mi_os_page_size()) == 0) {
229 p = mmap((void*)try_alignment, size, protect_flags, flags | MAP_ALIGN, fd, 0); // addr parameter is the required alignment
230 if (p!=MAP_FAILED) return p;
231 // fall back to regular mmap
232 }
233 #endif
234 #if (MI_INTPTR_SIZE >= 8) && !defined(MAP_ALIGNED)
235 // on 64-bit systems, use the virtual address area after 2TiB for 4MiB aligned allocations
236 if (addr == NULL) {
237 void* hint = _mi_os_get_aligned_hint(try_alignment, size);
238 if (hint != NULL) {
239 p = mmap(hint, size, protect_flags, flags, fd, 0);
240 if (p==MAP_FAILED || !_mi_is_aligned(p,try_alignment)) {
241 #if MI_TRACK_ENABLED // asan sometimes does not instrument errno correctly?
242 int err = 0;
243 #else
244 int err = errno;
245 #endif
246 _mi_trace_message("unable to directly request hinted aligned OS memory (error: %d (0x%x), size: 0x%zx bytes, alignment: 0x%zx, hint address: %p)\n", err, err, size, try_alignment, hint);
247 }
248 if (p!=MAP_FAILED) return p;
249 // fall back to regular mmap
250 }
251 }
252 #endif
253 // regular mmap
254 p = mmap(addr, size, protect_flags, flags, fd, 0);
255 if (p!=MAP_FAILED) return p;
256 // failed to allocate
257 return NULL;
258}

271static void* unix_mmap(void* addr, size_t size, size_t try_alignment, int protect_flags, bool large_only, bool allow_large, bool* is_large) {
272 #if !defined(MAP_ANONYMOUS)
273 #define MAP_ANONYMOUS MAP_ANON
274 #endif
275 #if !defined(MAP_NORESERVE)
276 #define MAP_NORESERVE 0
277 #endif
278 void* p = NULL;
279 const int fd = unix_mmap_fd();
280 int flags = MAP_PRIVATE | MAP_ANONYMOUS;
281 if (_mi_os_has_overcommit()) {
282 flags |= MAP_NORESERVE;
283 }
284 #if defined(PROT_MAX)
285 protect_flags |= PROT_MAX(PROT_READ | PROT_WRITE); // BSD
286 #endif
287 // huge page allocation
288 if (allow_large && (large_only || (_mi_os_use_large_page(size, try_alignment) && mi_option_get(mi_option_allow_large_os_pages) == 1))) {
289 static _Atomic(size_t) large_page_try_ok; // = 0;
290 size_t try_ok = mi_atomic_load_acquire(&large_page_try_ok);
291 if (!large_only && try_ok > 0) {
292 // If the OS is not configured for large OS pages, or the user does not have
293 // enough permission, the `mmap` will always fail (but it might also fail for other reasons).
294 // Therefore, once a large page allocation failed, we don't try again for `large_page_try_ok` times
295 // to avoid too many failing calls to mmap.
296 mi_atomic_cas_strong_acq_rel(&large_page_try_ok, &try_ok, try_ok - 1);
297 }
298 else {
299 int lflags = flags & ~MAP_NORESERVE; // using NORESERVE on huge pages seems to fail on Linux
300 int lfd = fd;
301 #ifdef MAP_ALIGNED_SUPER
302 lflags |= MAP_ALIGNED_SUPER;
303 #endif
304 #ifdef MAP_HUGETLB
305 lflags |= MAP_HUGETLB;
306 #endif
307 #ifdef MAP_HUGE_1GB
308 static bool mi_huge_pages_available = true;
309 if (large_only && (size % MI_GiB) == 0 && mi_huge_pages_available) {
310 lflags |= MAP_HUGE_1GB;
311 }
312 else
313 #endif
314 {
315 #ifdef MAP_HUGE_2MB
316 lflags |= MAP_HUGE_2MB;
317 #endif
318 }
319 #ifdef VM_FLAGS_SUPERPAGE_SIZE_2MB
320 lfd |= VM_FLAGS_SUPERPAGE_SIZE_2MB;
321 #endif
322 if (large_only || lflags != flags) {
323 // try large OS page allocation
324 *is_large = true;
325 p = unix_mmap_prim(addr, size, try_alignment, protect_flags, lflags, lfd);
326 #ifdef MAP_HUGE_1GB
327 if (p == NULL && (lflags & MAP_HUGE_1GB) == MAP_HUGE_1GB) {
328 mi_huge_pages_available = false; // don't try huge 1GiB pages again
329 if (large_only) {
330 _mi_warning_message("unable to allocate huge (1GiB) page, trying large (2MiB) pages instead (errno: %i)\n", errno);
331 }
332 lflags = ((lflags & ~MAP_HUGE_1GB) | MAP_HUGE_2MB);
333 p = unix_mmap_prim(addr, size, try_alignment, protect_flags, lflags, lfd);
334 }
335 #endif
336 if (large_only) return p;
337 if (p == NULL) {
338 mi_atomic_store_release(&large_page_try_ok, (size_t)8); // on error, don't try again for the next N allocations
339 }
340 }
341 }
342 }
343 // regular allocation
344 if (p == NULL) {
345 *is_large = false;
346 p = unix_mmap_prim(addr, size, try_alignment, protect_flags, flags, fd);
347 if (p != NULL) {
348 #if defined(MADV_HUGEPAGE)
349 // Many Linux systems don't allow MAP_HUGETLB but they support instead
350 // transparent huge pages (THP). Generally, it is not required to call `madvise` with MADV_HUGE
351 // though since properly aligned allocations will already use large pages if available
352 // in that case -- in particular for our large regions (in `memory.c`).
353 // However, some systems only allow THP if called with explicit `madvise`, so
354 // when large OS pages are enabled for mimalloc, we call `madvise` anyways.
355 if (allow_large && _mi_os_use_large_page(size, try_alignment)) {
356 if (unix_madvise(p, size, MADV_HUGEPAGE) == 0) {
357 // *is_large = true; // possibly
358 };
359 }
360 #elif defined(__sun)
361 if (allow_large && _mi_os_use_large_page(size, try_alignment)) {
362 struct memcntl_mha cmd = {0};
363 cmd.mha_pagesize = _mi_os_large_page_size();
364 cmd.mha_cmd = MHA_MAPSIZE_VA;
365 if (memcntl((caddr_t)p, size, MC_HAT_ADVISE, (caddr_t)&cmd, 0, 0) == 0) {
366 // *is_large = true; // possibly
367 }
368 }
369 #endif
370 }
371 }
372 return p;
373}

375// Note: the `try_alignment` is just a hint and the returned pointer is not guaranteed to be aligned.
376int _mi_prim_alloc(void* hint_addr, size_t size, size_t try_alignment, bool commit, bool allow_large, bool* is_large, bool* is_zero, void** addr) {
377 mi_assert_internal(size > 0 && (size % _mi_os_page_size()) == 0);
378 mi_assert_internal(commit || !allow_large);
379 mi_assert_internal(try_alignment > 0);
380 if (hint_addr == NULL && size >= 8*MI_UNIX_LARGE_PAGE_SIZE && try_alignment > 1 && _mi_is_power_of_two(try_alignment) && try_alignment < MI_UNIX_LARGE_PAGE_SIZE) {
381 try_alignment = MI_UNIX_LARGE_PAGE_SIZE; // try to align along large page size for larger allocations
382 }
383
384 *is_zero = true;
385 int protect_flags = (commit ? (PROT_WRITE | PROT_READ) : PROT_NONE);
386 *addr = unix_mmap(hint_addr, size, try_alignment, protect_flags, false, allow_large, is_large);
387 return (*addr != NULL ? 0 : errno);
388}

411int _mi_prim_commit(void* start, size_t size, bool* is_zero) {
412 // commit: ensure we can access the area
413 // note: we may think that *is_zero can be true since the memory
414 // was either from mmap PROT_NONE, or from decommit MADV_DONTNEED, but
415 // we sometimes call commit on a range with still partially committed
416 // memory and `mprotect` does not zero the range.
417 *is_zero = false;
418 int err = mprotect(start, size, (PROT_READ | PROT_WRITE));
419 if (err != 0) {
420 err = errno;
421 unix_mprotect_hint(err);
422 }
423 return err;
424}

426int _mi_prim_decommit(void* start, size_t size, bool* needs_recommit) {
427 int err = 0;
428 // decommit: use MADV_DONTNEED as it decreases rss immediately (unlike MADV_FREE)
429 err = unix_madvise(start, size, MADV_DONTNEED);
430 #if !MI_DEBUG && MI_SECURE<=2
431 *needs_recommit = false;
432 #else
433 *needs_recommit = true;
434 mprotect(start, size, PROT_NONE);
435 #endif
436 /*
437 // decommit: use mmap with MAP_FIXED and PROT_NONE to discard the existing memory (and reduce rss)
438 *needs_recommit = true;
439 const int fd = unix_mmap_fd();
440 void* p = mmap(start, size, PROT_NONE, (MAP_FIXED | MAP_PRIVATE | MAP_ANONYMOUS | MAP_NORESERVE), fd, 0);
441 if (p != start) { err = errno; }
442 */
443 return err;
444}

446int _mi_prim_reset(void* start, size_t size) {
447 // We try to use `MADV_FREE` as that is the fastest. A drawback though is that it
448 // will not reduce the `rss` stats in tools like `top` even though the memory is available
449 // to other processes. With the default `MIMALLOC_PURGE_DECOMMITS=1` we ensure that by
450 // default `MADV_DONTNEED` is used though.
451 #if defined(MADV_FREE)
452 static _Atomic(size_t) advice = MI_ATOMIC_VAR_INIT(MADV_FREE);
453 int oadvice = (int)mi_atomic_load_relaxed(&advice);
454 int err;
455 while ((err = unix_madvise(start, size, oadvice)) != 0 && errno == EAGAIN) { errno = 0; };
456 if (err != 0 && errno == EINVAL && oadvice == MADV_FREE) {
457 // if MADV_FREE is not supported, fall back to MADV_DONTNEED from now on
458 mi_atomic_store_release(&advice, (size_t)MADV_DONTNEED);
459 err = unix_madvise(start, size, MADV_DONTNEED);
460 }
461 #else
462 int err = unix_madvise(start, size, MADV_DONTNEED);
463 #endif
464 return err;
465}

470 unix_mprotect_hint(err);

743bool _mi_prim_getenv(const char* name, char* result, size_t result_size) {
744 if (name==NULL) return false;
745 const size_t len = _mi_strlen(name);
746 if (len == 0) return false;
747 char** env = mi_get_environ();
748 if (env == NULL) return false;
749 // compare up to 10000 entries
750 for (int i = 0; i < 10000 && env[i] != NULL; i++) {
751 const char* s = env[i];
752 if (_mi_strnicmp(name, s, len) == 0 && s[len] == '=') { // case insensitive
753 // found it
754 _mi_strlcpy(result, s + len + 1, result_size);
755 return true;
756 }
757 }
758 return false;
759}
760#else
761// fallback: use standard C `getenv` but this cannot be used while initializing the C runtime
762bool _mi_prim_getenv(const char* name, char* result, size_t result_size) {
763 // cannot call getenv() when still initializing the C runtime.
764 if (_mi_preloading()) return false;
765 const char* s = getenv(name);
766 if (s == NULL) {
767 // we check the upper case name too.
768 char buf[64+1];
769 size_t len = _mi_strnlen(name,sizeof(buf)-1);
770 for (size_t i = 0; i < len; i++) {
771 buf[i] = _mi_toupper(name[i]);
772 }
773 buf[len] = 0;
774 s = getenv(buf);
775 }
776 if (s == NULL || _mi_strnlen(s,result_size) >= result_size) return false;
777 _mi_strlcpy(result, s, result_size);
778 return true;
779}

809#include <sys/types.h>
810#include <sys/stat.h>
811#include <errno.h>
812
813bool _mi_prim_random_buf(void* buf, size_t buf_len) {
814 // Modern Linux provides `getrandom` but different distributions either use `sys/random.h` or `linux/random.h`
815 // and for the latter the actual `getrandom` call is not always defined.
816 // (see <https://stackoverflow.com/questions/45237324/why-doesnt-getrandom-compile>)
817 // We therefore use a syscall directly and fall back dynamically to /dev/urandom when needed.
818 #if defined(MI_HAS_SYSCALL_H) && defined(SYS_getrandom)
819 #ifndef GRND_NONBLOCK
820 #define GRND_NONBLOCK (1)
821 #endif
822 static _Atomic(uintptr_t) no_getrandom; // = 0
823 if (mi_atomic_load_acquire(&no_getrandom)==0) {
824 ssize_t ret = syscall(SYS_getrandom, buf, buf_len, GRND_NONBLOCK);
825 if (ret >= 0) return (buf_len == (size_t)ret);
826 if (errno != ENOSYS) return false;
827 mi_atomic_store_release(&no_getrandom, (uintptr_t)1); // don't call again, and fall back to /dev/urandom
828 }
829 #endif
830 int flags = O_RDONLY;
831 #if defined(O_CLOEXEC)
832 flags |= O_CLOEXEC;
833 #endif
834 int fd = mi_prim_open("/dev/urandom", flags);
835 if (fd < 0) return false;
836 size_t count = 0;
837 while(count < buf_len) {
838 ssize_t ret = mi_prim_read(fd, (char*)buf + count, buf_len - count);
839 if (ret<=0) {
840 if (errno!=EAGAIN && errno!=EINTR) break;
841 }
842 else {
843 count += ret;
844 }
845 }
846 mi_prim_close(fd);
847 return (count==buf_len);
848}
