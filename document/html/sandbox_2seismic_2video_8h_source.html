157inline color_t video::get_color(colorcomp_t red, colorcomp_t green, colorcomp_t blue) const
158{
159 if(red_shift == 16) // only for depth == 24 && red_shift > blue_shift
160 return (red<<16) | (green<<8) | blue;
161 else if(depth >= 24)
162 return
163#if __ANDROID__
164 // Setting Alpha to 0xFF
165 0xFF000000 |
166#endif
167 (red<<red_shift) | (green<<green_shift) | (blue<<blue_shift);
168 else if(depth > 0) {
169 depth_t bs = blue_shift, rs = red_shift;
170 if(blue_shift < 0) blue >>= -bs, bs = 0;
171 else /*red_shift < 0*/ red >>= -rs, rs = 0;
172 return ((red<<rs)&red_mask) | ((green<<green_shift)&green_mask) | ((blue<<bs)&blue_mask);
173 } else { // UYVY colorspace
174 unsigned y, u, v;
175 y = red * 77 + green * 150 + blue * 29; // sum(77+150+29=256) * max(=255): limit->2^16
176 u = (2048 + (blue << 3) - (y >> 5)) >> 4; // (limit->2^12)>>4
177 v = (2048 + (red << 3) - (y >> 5)) >> 4;
178 y = y >> 8;
179 return u | (y << 8) | (v << 16) | (y << 24);
180 }
181}

183inline drawing_area::drawing_area(int x, int y, int sizex, int sizey, const drawing_memory &dmem)

199inline void drawing_area::put_pixel(color_t color)
200{
201 assert(index < max_index);
202 if(pixel_depth > 16) ptr32[index++] = color;
203 else if(pixel_depth > 0)
204 ((unsigned short*)ptr32)[index++] = (unsigned short)color;
205 else { // UYVY colorspace
206 if(index&1) color >>= 16;
207 ((unsigned short*)ptr32)[index++] = (unsigned short)color;
208 }
209}
