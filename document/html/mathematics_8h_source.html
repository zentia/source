213template<typename A, typename B, typename C,
214 std::enable_if_t<std::disjunction_v<
215 is_vector<A>, is_vector<B>, is_vector<C>>,
216 int> = 0>
217[[nodiscard]] inline auto fma(A a, B b, C c) noexcept {
218 constexpr auto dim_a = vector_dimension_v<A>;
219 constexpr auto dim_b = vector_dimension_v<B>;
220 constexpr auto dim_c = vector_dimension_v<C>;
221 constexpr auto dim = (std::max)({dim_a, dim_b, dim_c});
222 static_assert((dim_a == 1 || dim_a == dim) &&
223 (dim_b == 1 || dim_b == dim) &&
224 (dim_c == 1 || dim_c == dim));
225 static_assert(std::is_same_v<vector_element_t<A>, float> &&
226 std::is_same_v<vector_element_t<B>, float> &&
227 std::is_same_v<vector_element_t<C>, float>);
228 using V = Vector<float, dim>;
229 auto va = V{a};
230 auto vb = V{b};
231 auto vc = V{c};
232 V result{};
233 for (auto i = 0; i < dim; ++i) {
234 result[i] = fma(va[i], vb[i], vc[i]);
235 }
236 return result;
237};
