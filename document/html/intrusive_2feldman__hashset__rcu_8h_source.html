175 template <typename Func>
176 bool insert( value_type& val, Func f )
177 {
178 hash_type const& hash = hash_accessor()( val );
179 traverse_data pos( hash, *this );
180 hash_comparator cmp;
181
182 while (true) {
183 rcu_lock rcuLock;
184
185 node_ptr slot = base_class::traverse( pos );
186 assert(slot.bits() == 0);
187
188 if ( pos.pArr->nodes[pos.nSlot].load(memory_model::memory_order_acquire) == slot) {
189 if (slot.ptr()) {
190 if ( cmp( hash, hash_accessor()(*slot.ptr())) == 0 ) {
191 // the item with that hash value already exists
192 stats().onInsertFailed();
193 return false;
194 }
195
196 // the slot must be expanded
197 base_class::expand_slot( pos, slot );
198 }
199 else {
200 // the slot is empty, try to insert data node
201 node_ptr pNull;
202 if ( pos.pArr->nodes[pos.nSlot].compare_exchange_strong(pNull, node_ptr(&val), memory_model::memory_order_release, atomics::memory_order_relaxed))
203 {
204 // the new data node has been inserted
205 f(val);
206 ++m_ItemCounter;
207 stats().onInsertSuccess();
208 stats().height( pos.nHeight );
209 return true;
210 }
211
212 // insert failed - slot has been changed by another thread
213 // retry inserting
214 stats().onInsertRetry();
215 }
216 }
217 else
218 stats().onSlotChanged();
219 }
220 }

574 void forward()
575 {
576 assert(m_set != nullptr);
577 assert(m_pNode != nullptr);
578
579 size_t const arrayNodeSize = m_set->array_node_size();
580 size_t const headSize = m_set->head_size();
581 array_node * pNode = m_pNode;
582 size_t idx = m_idx + 1;
583 size_t nodeSize = m_pNode->pParent ? arrayNodeSize : headSize;
584
585 for (;;) {
586 if (idx < nodeSize) {
587 node_ptr slot = pNode->nodes[idx].load(memory_model::memory_order_acquire);
588 if (slot.bits() == base_class::flag_array_node ) {
589 // array node, go down the tree
590 assert(slot.ptr() != nullptr);
591 pNode = to_array(slot.ptr());
592 idx = 0;
593 nodeSize = arrayNodeSize;
594 }
595 else if (slot.bits() == base_class::flag_array_converting ) {
596 // the slot is converting to array node right now - skip the node
597 ++idx;
598 }
599 else {
600 if (slot.ptr()) {
601 // data node
602 m_pNode = pNode;
603 m_idx = idx;
604 m_pValue = slot.ptr();
605 return;
606 }
607 ++idx;
608 }
609 }
610 else {
611 // up to parent node
612 if (pNode->pParent) {
613 idx = pNode->idxParent + 1;
614 pNode = pNode->pParent;
615 nodeSize = pNode->pParent ? arrayNodeSize : headSize;
616 }
617 else {
618 // end()
619 assert(pNode == m_set->head());
620 assert(idx == headSize);
621 m_pNode = pNode;
622 m_idx = idx;
623 m_pValue = nullptr;
624 return;
625 }
626 }
627 }
628 }
629
630 void backward()
631 {
632 assert(m_set != nullptr);
633 assert(m_pNode != nullptr);
634
635 size_t const arrayNodeSize = m_set->array_node_size();
636 size_t const headSize = m_set->head_size();
637 size_t const endIdx = size_t(0) - 1;
638
639 array_node * pNode = m_pNode;
640 size_t idx = m_idx - 1;
641 size_t nodeSize = m_pNode->pParent ? arrayNodeSize : headSize;
642
643 for (;;) {
644 if (idx != endIdx) {
645 node_ptr slot = pNode->nodes[idx].load(memory_model::memory_order_acquire);
646 if (slot.bits() == base_class::flag_array_node ) {
647 // array node, go down the tree
648 assert(slot.ptr() != nullptr);
649 pNode = to_array(slot.ptr());
650 nodeSize = arrayNodeSize;
651 idx = nodeSize - 1;
652 }
653 else if (slot.bits() == base_class::flag_array_converting ) {
654 // the slot is converting to array node right now - skip the node
655 --idx;
656 }
657 else {
658 if (slot.ptr()) {
659 // data node
660 m_pNode = pNode;
661 m_idx = idx;
662 m_pValue = slot.ptr();
663 return;
664 }
665 --idx;
666 }
667 }
668 else {
669 // up to parent node
670 if (pNode->pParent) {
671 idx = pNode->idxParent - 1;
672 pNode = pNode->pParent;
673 nodeSize = pNode->pParent ? arrayNodeSize : headSize;
674 }
675 else {
676 // rend()
677 assert(pNode == m_set->head());
678 assert(idx == endIdx);
679 m_pNode = pNode;
680 m_idx = idx;
681 m_pValue = nullptr;
682 return;
683 }
684 }
685 }

1023 template <typename Func>
1024 std::pair<bool, bool> do_update(value_type& val, Func f, bool bInsert = true)
1025 {
1026 hash_type const& hash = hash_accessor()(val);
1027 traverse_data pos( hash, *this );
1028 hash_comparator cmp;
1029 value_type * pOld;
1030
1031 while ( true ) {
1032 rcu_lock rcuLock;
1033
1034 node_ptr slot = base_class::traverse( pos );
1035 assert(slot.bits() == 0);
1036
1037 pOld = nullptr;
1038 if ( pos.pArr->nodes[pos.nSlot].load(memory_model::memory_order_acquire) == slot) {
1039 if ( slot.ptr()) {
1040 if ( cmp( hash, hash_accessor()(*slot.ptr())) == 0 ) {
1041 // the item with that hash value already exists
1042 // Replace it with val
1043 if ( slot.ptr() == &val ) {
1044 stats().onUpdateExisting();
1045 return std::make_pair(true, false);
1046 }
1047
1048 if ( pos.pArr->nodes[pos.nSlot].compare_exchange_strong(slot, node_ptr(&val), memory_model::memory_order_release, atomics::memory_order_relaxed)) {
1049 // slot can be disposed
1050 f( val, slot.ptr());
1051 pOld = slot.ptr();
1052 stats().onUpdateExisting();
1053 goto update_existing_done;
1054 }
1055
1056 stats().onUpdateRetry();
1057 }
1058 else {
1059 if ( bInsert ) {
1060 // the slot must be expanded
1061 base_class::expand_slot( pos, slot );
1062 }
1063 else {
1064 stats().onUpdateFailed();
1065 return std::make_pair(false, false);
1066 }
1067 }
1068 }
1069 else {
1070 // the slot is empty, try to insert data node
1071 if (bInsert) {
1072 node_ptr pNull;
1073 if ( pos.pArr->nodes[pos.nSlot].compare_exchange_strong(pNull, node_ptr(&val), memory_model::memory_order_release, atomics::memory_order_relaxed))
1074 {
1075 // the new data node has been inserted
1076 f(val, nullptr);
1077 ++m_ItemCounter;
1078 stats().onUpdateNew();
1079 stats().height( pos.nHeight );
1080 return std::make_pair(true, true);
1081 }
1082 }
1083 else {
1084 stats().onUpdateFailed();
1085 return std::make_pair(false, false);
1086 }
1087
1088 // insert failed - slot has been changed by another thread
1089 // retry updating
1090 stats().onUpdateRetry();
1091 }
1092 }
1093 else
1094 stats().onSlotChanged();
1095 } // while
1096
1097 // update success
1098 // retire_ptr must be called only outside of RCU lock
1099 update_existing_done:
1100 if (pOld)
1101 gc::template retire_ptr<disposer>(pOld);
1102 return std::make_pair(true, false);
1103 }

1105 template <typename Predicate>
1106 value_type * do_erase( hash_type const& hash, Predicate pred)
1107 {
1108 assert(gc::is_locked());
1109 traverse_data pos( hash, *this );
1110 hash_comparator cmp;
1111
1112 while ( true ) {
1113 node_ptr slot = base_class::traverse( pos );
1114 assert( slot.bits() == 0 );
1115
1116 if ( pos.pArr->nodes[pos.nSlot].load( memory_model::memory_order_acquire ) == slot ) {
1117 if ( slot.ptr()) {
1118 if ( cmp( hash, hash_accessor()(*slot.ptr())) == 0 && pred( *slot.ptr())) {
1119 // item found - replace it with nullptr
1120 if ( pos.pArr->nodes[pos.nSlot].compare_exchange_strong( slot, node_ptr( nullptr ), memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
1121 --m_ItemCounter;
1122 stats().onEraseSuccess();
1123
1124 return slot.ptr();
1125 }
1126 stats().onEraseRetry();
1127 continue;
1128 }
1129 stats().onEraseFailed();
1130 return nullptr;
1131 }
1132 else {
1133 // the slot is empty
1134 stats().onEraseFailed();
1135 return nullptr;
1136 }
1137 }
1138 else
1139 stats().onSlotChanged();
1140 }

1143 value_type * search(hash_type const& hash )
1144 {
1145 assert( gc::is_locked());
1146 traverse_data pos( hash, *this );
1147 hash_comparator cmp;
1148
1149 while ( true ) {
1150 node_ptr slot = base_class::traverse( pos );
1151 assert( slot.bits() == 0 );
1152
1153 if ( pos.pArr->nodes[pos.nSlot].load( memory_model::memory_order_acquire ) != slot ) {
1154 // slot value has been changed - retry
1155 stats().onSlotChanged();
