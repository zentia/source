44class ff_oclallocator {
45 typedef std::map<const void*, cl_mem> inner_map_t;
46 typedef std::map<cl_context, inner_map_t> outer_map_t;
47public:
48
49 cl_mem createBuffer(const void *key,
50 cl_context ctx, cl_mem_flags flags, size_t size, cl_int *status) {
51 cl_mem ptr = clCreateBuffer(ctx,flags,size, NULL,status);
52 if (*status == CL_SUCCESS) allocated[ctx][key] = ptr;
53 return ptr;
54 }
55
56 cl_mem createBufferUnique(const void *key,
57 cl_context ctx, cl_mem_flags flags, size_t size, cl_int *status) {
58 if (allocated.find(ctx) != allocated.end()) {
59 inner_map_t::iterator it = allocated[ctx].find(key);
60 if (it != allocated[ctx].end()) {
61 *status = CL_SUCCESS;
62 return it->second;
63 }
64 }
65 return createBuffer(key,ctx,flags,size,status);
66 }
67
68 void updateKey(const void *oldkey, const void *newkey, cl_context ctx) {
69 assert(allocated.find(ctx) != allocated.end());
70 assert(allocated[ctx].find(oldkey) != allocated[ctx].end());
71
72 cl_mem ptr = allocated[ctx][oldkey];
73 allocated[ctx][newkey] = ptr;
74 allocated[ctx].erase(oldkey);
75 }
76
77
78 cl_int releaseBuffer(const void *key, cl_context ctx, cl_mem ptr) {
79 assert(allocated.find(ctx) != allocated.end());
80 assert(allocated[ctx].find(key) != allocated[ctx].end());
81 if (clReleaseMemObject(ptr) == CL_SUCCESS) {
82 allocated[ctx].erase(key);
83 return CL_SUCCESS;
84 }
85 return CL_INVALID_MEM_OBJECT;
86 }
87
88 cl_int releaseAllBuffers(cl_context ctx) {
89 outer_map_t::iterator it = allocated.find(ctx);
90 if (it == allocated.end()) return CL_SUCCESS;
91
92 inner_map_t::iterator b=it->second.begin();
93 inner_map_t::iterator e=it->second.end();
94 while(b != e) {
95 if (clReleaseMemObject(b->second) != CL_SUCCESS) return CL_INVALID_MEM_OBJECT;
96 ++b;
97 }
98 it->second.clear();
99 return CL_SUCCESS;
100 }
101
102protected:
103 std::map<cl_context, std::map<const void*, cl_mem> > allocated;
104};
