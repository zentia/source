44template <auto e>
45consteval auto get_enum_name_str_view() {
46#if __cpp_lib_source_location >= 201907L
47 const auto func_name =
48 std::string_view{std::source_location::current().function_name()};
49#elif defined(_MSC_VER)
50 // Officially, we only support MSVC versions that are modern enough to contain
51 // <source_location>, but inofficially, this might work.
52 const auto func_name = std::string_view{__FUNCSIG__};
53#else
54 const auto func_name = std::string_view{__PRETTY_FUNCTION__};
55#endif
56#if defined(__clang__)
57 const auto split = func_name.substr(0, func_name.size() - 1);

59#elif defined(__GNUC__)
60 const auto split = func_name.substr(0, func_name.size() - 1);

66 static_assert(false,

68 "or MSVC or use rfl::Literal.");

81template <class T, bool _is_flag>

83 using U = std::underlying_type_t<T>;
84 if constexpr (_is_flag) {
85 return 0;
86 } else {

90template <class T, bool _is_flag>
91consteval auto get_range_max() {
92 using U = std::underlying_type_t<T>;
93 if constexpr (_is_flag) {
94 if constexpr (std::is_signed_v<T>) {
95 return (sizeof(U) * 8 - 2);
96 } else {
97 return (sizeof(U) * 8 - 1);
98 }
99 } else {
100 return (std::min)((std::numeric_limits<U>::max)(), range_max<T>::value);
101 }
102}

138template <int ChunkSize = 256>
139struct CombineNames {
140 template <class NamesType, int Start, int End>
141 static consteval auto apply() {
142 if constexpr (End - Start < ChunkSize) {
143 return []<int... Is>(std::integer_sequence<int, Is...>) {
144 return (NamesType{} | ... | std::integral_constant<int, Start + Is>());
145 }(std::make_integer_sequence<int, End - Start + 1>{});
146 } else {
147 constexpr int Mid = Start + (End - Start) / 2;
148 constexpr auto left = CombineNames<ChunkSize>::template apply<NamesType, Start, Mid>();
149 constexpr auto right = CombineNames<ChunkSize>::template apply<NamesType, Mid + 1, End>();
150 return left | right;
151 }
152 }
153};

155template <class EnumType, bool _is_flag>
156consteval auto get_enum_names() {
157 static_assert(is_scoped_enum<EnumType>,
158 "You must use scoped enums (using class or struct) for the parsing to work!");
159 static_assert(std::is_integral_v<std::underlying_type_t<EnumType>>,
160 "The underlying type of any Enum must be integral!");
161
162 constexpr auto max = get_range_max<EnumType, _is_flag>();
163 constexpr auto min = get_range_min<EnumType, _is_flag>();
164 constexpr auto range_size = max - min + 1;
165
166 static_assert(range_size > 0,
167 "enum_range requires a valid range size. Ensure that max is greater than min.");
168
169 using EmptyNames = Names<EnumType, rfl::Literal<"">, 0, _is_flag>;
170
171 return CombineNames<>::template apply<EmptyNames, min, max>();
172}
