1161inline void Executor::_invoke(Worker& worker, Node* node) {
1162
1163 int state;
1164 SmallVector<int> conds;
1165
1166 // synchronize all outstanding memory operations caused by reordering
1167 do {
1168 state = node->_state.load(std::memory_order_acquire);
1169 } while(! (state & Node::READY));
1170
1171 // unwind stack for deferred node
1172 if(state & Node::DEFERRED) {
1173 node->_state.fetch_and(~Node::DEFERRED, std::memory_order_relaxed);
1174 goto invoke_epilogue;
1175 }
1176
1177 //while(!(node->_state.load(std::memory_order_acquire) & Node::READY));
1178
1179 invoke_prologue:
1180
1181 // no need to do other things if the topology is cancelled
1182 if(node->_is_cancelled()) {
1183 _cancel_invoke(worker, node);
1184 return;
1185 }
1186
1187 // if acquiring semaphore(s) exists, acquire them first
1188 if(node->_semaphores && !node->_semaphores->to_acquire.empty()) {
1189 SmallVector<Node*> nodes;
1190 if(!node->_acquire_all(nodes)) {
1191 _schedule(worker, nodes);
1192 return;
1193 }
1194 node->_state.fetch_or(Node::ACQUIRED, std::memory_order_release);
1195 }
1196
1197 // condition task
1198 //int cond = -1;
1199 //SmallVector<int> conds = { -1 };
1200
1201 // switch is faster than nested if-else due to jump table
1202 switch(node->_handle.index()) {
1203 // static task
1204 case Node::STATIC:{
1205 _invoke_static_task(worker, node);
1206 }
1207 break;
1208
1209 // dynamic task
1210 case Node::DYNAMIC: {
1211 _invoke_dynamic_task(worker, node);
1212 }
1213 break;
1214
1215 // condition task
1216 case Node::CONDITION: {
1217 _invoke_condition_task(worker, node, conds);
1218 }
1219 break;
1220
1221 // multi-condition task
1222 case Node::MULTI_CONDITION: {
1223 _invoke_multi_condition_task(worker, node, conds);
1224 }
1225 break;
1226
1227 // module task
1228 case Node::MODULE: {
1229 bool deferred = false;
1230 _invoke_module_task(worker, node, deferred);
1231 if(deferred) {
1232 return;
1233 }
1234 }
1235 break;
1236
1237 // async task
1238 case Node::ASYNC: {
1239 _invoke_async_task(worker, node);
1240 _tear_down_async(node);
1241 return ;
1242 }
1243 break;
1244
1245 // silent async task
1246 case Node::SILENT_ASYNC: {
1247 _invoke_silent_async_task(worker, node);
1248 _tear_down_async(node);
1249 return ;
1250 }
1251 break;
1252
1253 // cudaflow task
1254 case Node::CUDAFLOW: {
1255 _invoke_cudaflow_task(worker, node);
1256 }
1257 break;
1258
1259 // syclflow task
1260 case Node::SYCLFLOW: {
1261 _invoke_syclflow_task(worker, node);
1262 }
1263 break;
1264
1265 // runtime task
1266 case Node::RUNTIME: {
1267 _invoke_runtime_task(worker, node);
1268 }
1269 break;
1270
1271 // monostate (placeholder)
1272 default:
1273 break;
1274 }
1275
1276 invoke_epilogue:
1277
1278 // if releasing semaphores exist, release them
1279 if(node->_semaphores && !node->_semaphores->to_release.empty()) {
1280 _schedule(worker, node->_release_all());
1281 }
1282
1283 // We MUST recover the dependency since the graph may have cycles.
1284 // This must be done before scheduling the successors, otherwise this might cause
1285 // race condition on the _dependents
1286 if((node->_state.load(std::memory_order_relaxed) & Node::CONDITIONED)) {
1287 node->_join_counter = node->num_strong_dependents();
1288 }
1289 else {
1290 node->_join_counter = node->num_dependents();
1291 }
1292
1293 // acquire the parent flow counter
1294 auto& j = (node->_parent) ? node->_parent->_join_counter :
1295 node->_topology->_join_counter;
1296
1297 Node* cache {nullptr};
1298
1299 // At this point, the node storage might be destructed (to be verified)
1300 // case 1: non-condition task
1301 switch(node->_handle.index()) {
1302
1303 // condition and multi-condition tasks
1304 case Node::CONDITION:
1305 case Node::MULTI_CONDITION: {
1306 for(auto cond : conds) {
1307 if(cond >= 0 && static_cast<size_t>(cond) < node->_successors.size()) {
1308 auto s = node->_successors[cond];
1309 // zeroing the join counter for invariant
1310 s->_join_counter.store(0, std::memory_order_relaxed);
1311 j.fetch_add(1);
1312 if(cache) {
1313 _schedule(worker, cache);
1314 }
1315 cache = s;
1316 }
1317 }
1318 }
1319 break;
1320
1321 // non-condition task
1322 default: {
1323 for(size_t i=0; i<node->_successors.size(); ++i) {
1324 if(--(node->_successors[i]->_join_counter) == 0) {
1325 j.fetch_add(1);
1326 if(cache) {
1327 _schedule(worker, cache);
1328 }
1329 cache = node->_successors[i];
1330 }
1331 }
1332 }
1333 break;
1334 }
1335
1336 // tear_down the invoke
1337 _tear_down_invoke(worker, node);
1338
1339 // perform tail recursion elimination for the right-most child to reduce
1340 // the number of expensive pop/push operations through the task queue
1341 if(cache) {
1342 node = cache;
1343 //node->_state.fetch_or(Node::READY, std::memory_order_release);
1344 goto invoke_prologue;
1345 }
1346}

1348// Procedure: _tear_down_async
1349inline void Executor::_tear_down_async(Node* node) {
1350 if(node->_parent) {
1351 node->_parent->_join_counter.fetch_sub(1);
1352 }
1353 else {
1354 _decrement_topology_and_notify();
1355 }
1356 node_pool.recycle(node);
1357}

1359// Proecdure: _tear_down_invoke
1360inline void Executor::_tear_down_invoke(Worker& worker, Node* node) {
1361 // we must check parent first before substracting the join counter,
1362 // or it can introduce data race
1363 if(auto parent = node->_parent; parent == nullptr) {
1364 if(node->_topology->_join_counter.fetch_sub(1) == 1) {
1365 _tear_down_topology(worker, node->_topology);
1366 }
1367 }
1368 else {
1369 // prefetch the deferred status, as subtracting the join counter can
1370 // immediately cause the other worker to release the subflow
1371 auto deferred = parent->_state.load(std::memory_order_relaxed) & Node::DEFERRED;
1372 if(parent->_join_counter.fetch_sub(1) == 1 && deferred) {
1373 _schedule(worker, parent);
1374 }
1375 }
1376}

1378// Procedure: _cancel_invoke
1379inline void Executor::_cancel_invoke(Worker& worker, Node* node) {
1380
1381 switch(node->_handle.index()) {
1382 // async task needs to carry out the promise
1383 case Node::ASYNC:
1384 std::get_if<Node::Async>(&(node->_handle))->work(true);
1385 _tear_down_async(node);
1386 break;
1387
1388 // silent async doesn't need to carry out the promise
1389 case Node::SILENT_ASYNC:
1390 _tear_down_async(node);
1391 break;
1392
1393 // tear down topology if the node is the last leaf
1394 default: {
1395 _tear_down_invoke(worker, node);
1396 }
1397 break;
1398 }

1442// Procedure: _invoke_dynamic_task_external
1443inline void Executor::_invoke_dynamic_task_external(
1444 Worker& w, Node* p, Graph& g, bool detach
1445) {
1446
1447 // graph is empty and has no async tasks
1448 if(g.empty() && p->_join_counter == 0) {
1449 return;
1450 }
1451
1452 SmallVector<Node*> src;
1453
1454 for(auto n : g._nodes) {
1455
1456 n->_topology = p->_topology;
1457 n->_state.store(0, std::memory_order_relaxed);
1458 n->_set_up_join_counter();
1459
1460 if(detach) {
1461 n->_parent = nullptr;
1462 n->_state.fetch_or(Node::DETACHED, std::memory_order_relaxed);
1463 }
1464 else {
1465 n->_parent = p;
1466 }
1467
1468 if(n->num_dependents() == 0) {
1469 src.push_back(n);
1470 }
1471 }
1472
1473 // detach here
1474 if(detach) {
1475
1476 {
1477 std::lock_guard<std::mutex> lock(p->_topology->_taskflow._mutex);
1478 p->_topology->_taskflow._graph._merge(std::move(g));
1479 }
1480
1481 p->_topology->_join_counter.fetch_add(src.size());
1482 _schedule(w, src);
1483 }
1484 // join here
1485 else {
1486 p->_join_counter.fetch_add(src.size());
1487 _schedule(w, src);
1488 _consume_task(w, p);
1489 }
1490}

1670// Function: run_until
1671template <typename P, typename C>
1672tf::Future<void> Executor::run_until(Taskflow& f, P&& p, C&& c) {
1673
1674 _increment_topology();
1675
1676 // Need to check the empty under the lock since dynamic task may
1677 // define detached blocks that modify the taskflow at the same time
1678 bool empty;
1679 {
1680 std::lock_guard<std::mutex> lock(f._mutex);
1681 empty = f.empty();
1682 }
1683
1684 // No need to create a real topology but returns an dummy future
1685 if(empty || p()) {
1686 c();
1687 std::promise<void> promise;
1688 promise.set_value();
1689 _decrement_topology_and_notify();
1690 return tf::Future<void>(promise.get_future(), std::monostate{});
1691 }
1692
1693 // create a topology for this run
1694 auto t = std::make_shared<Topology>(f, std::forward<P>(p), std::forward<C>(c));
1695
1696 // need to create future before the topology got torn down quickly
1697 tf::Future<void> future(t->_promise.get_future(), t);
1698
1699 // modifying topology needs to be protected under the lock
1700 {
1701 std::lock_guard<std::mutex> lock(f._mutex);
1702 f._topologies.push(t);
1703 if(f._topologies.size() == 1) {
1704 _set_up_topology(_this_worker(), t.get());
1705 }
1706 }
1707
1708 return future;
1709}

1714
1715 std::list<Taskflow>::iterator itr;
1716
1717 {
1718 std::scoped_lock<std::mutex> lock(_taskflow_mutex);
1719 itr = _taskflows.emplace(_taskflows.end(), std::move(f));
1720 itr->_satellite = itr;
1721 }
1722
1723 return run_until(*itr, std::forward<P>(pred), std::forward<C>(c));
1724}

1752// Function: _set_up_topology
1753inline void Executor::_set_up_topology(Worker* worker, Topology* tpg) {
1754
1755 // ---- under taskflow lock ----
1756
1757 tpg->_sources.clear();
1758 tpg->_taskflow._graph._clear_detached();
1759
1760 // scan each node in the graph and build up the links
1761 for(auto node : tpg->_taskflow._graph._nodes) {
1762
1763 node->_topology = tpg;
1764 node->_state.store(0, std::memory_order_relaxed);
1765
1766 if(node->num_dependents() == 0) {
1767 tpg->_sources.push_back(node);
1768 }
1769
1770 node->_set_up_join_counter();
1771 }
1772
1773 tpg->_join_counter = tpg->_sources.size();
1774
1775 if(worker) {
1776 _schedule(*worker, tpg->_sources);
1777 }
1778 else {
1779 _schedule(tpg->_sources);
1780 }
1781}

1783// Function: _tear_down_topology
1784inline void Executor::_tear_down_topology(Worker& worker, Topology* tpg) {
1785
1786 auto &f = tpg->_taskflow;
1787
1788 //assert(&tpg == &(f._topologies.front()));
1789
1790 // case 1: we still need to run the topology again
1791 if(!tpg->_is_cancelled && !tpg->_pred()) {
1792 //assert(tpg->_join_counter == 0);
1793 std::lock_guard<std::mutex> lock(f._mutex);
1794 tpg->_join_counter = tpg->_sources.size();
1795 _schedule(worker, tpg->_sources);
1796 }
1797 // case 2: the final run of this topology
1798 else {
1799
1800 // TODO: if the topology is cancelled, need to release all semaphores
1801
1802 if(tpg->_call != nullptr) {
1803 tpg->_call();
1804 }
1805
1806 // If there is another run (interleave between lock)
1807 if(std::unique_lock<std::mutex> lock(f._mutex); f._topologies.size()>1) {
1808 //assert(tpg->_join_counter == 0);
1809
1810 // Set the promise
1811 tpg->_promise.set_value();
1812 f._topologies.pop();
1813 tpg = f._topologies.front().get();
1814
1815 // decrement the topology but since this is not the last we don't notify
1816 _decrement_topology();
1817
1818 // set up topology needs to be under the lock or it can
1819 // introduce memory order error with pop
1820 _set_up_topology(&worker, tpg);
1821 }
1822 else {
1823 //assert(f._topologies.size() == 1);
1824
1825 // Need to back up the promise first here becuz taskflow might be
1826 // destroy soon after calling get
1827 auto p {std::move(tpg->_promise)};
1828
1829 // Back up lambda capture in case it has the topology pointer,
1830 // to avoid it releasing on pop_front ahead of _mutex.unlock &
1831 // _promise.set_value. Released safely when leaving scope.
1832 auto c {std::move(tpg->_call)};
1833
1834 // Get the satellite if any
1835 auto s {f._satellite};
1836
1837 // Now we remove the topology from this taskflow
1838 f._topologies.pop();
1839
1840 //f._mutex.unlock();
1841 lock.unlock();
1842
1843 // We set the promise in the end in case taskflow leaves the scope.
1844 // After set_value, the caller will return from wait
1845 p.set_value();
1846
1847 _decrement_topology_and_notify();
1848
1849 // remove the taskflow if it is managed by the executor
1850 // TODO: in the future, we may need to synchronize on wait
1851 // (which means the following code should the moved before set_value)
1852 if(s) {
1853 std::scoped_lock<std::mutex> lock(_taskflow_mutex);
1854 _taskflows.erase(*s);
1855 }
1856 }
1857 }
1858}

1898// Function: _named_async
1899template <typename F, typename... ArgsT>
1900auto Subflow::_named_async(
1901 Worker& w,
1902 const std::string& name,
1903 F&& f,
1904 ArgsT&&... args
1905) {
1906
1907 _parent->_join_counter.fetch_add(1);
1908
1909 using T = std::invoke_result_t<F, ArgsT...>;
1910 using R = std::conditional_t<std::is_same_v<T, void>, void, std::optional<T>>;
1911
1912 std::promise<R> p;
1913
1914 auto tpg = std::make_shared<AsyncTopology>();
1915
1916 Future<R> fu(p.get_future(), tpg);
1917
1918 auto node = node_pool.animate(
1919 std::in_place_type_t<Node::Async>{},
1920 [p=make_moc(std::move(p)), f=std::forward<F>(f), args...]
1921 (bool cancel) mutable {
1922 if constexpr(std::is_same_v<R, void>) {
1923 if(!cancel) {
1924 f(args...);
1925 }
1926 p.object.set_value();
1927 }
1928 else {
1929 p.object.set_value(cancel ? std::nullopt : std::make_optional(f(args...)));
1930 }
1931 },
1932 std::move(tpg)
1933 );
1934
1935 node->_name = name;
1936 node->_topology = _parent->_topology;
1937 node->_parent = _parent;
1938
1939 _executor._schedule(w, node);
1940
1941 return fu;
1942}
