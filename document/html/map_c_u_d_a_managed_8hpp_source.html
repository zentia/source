113template<typename T, typename kernelF>
114class ff_mapCUDAManaged: public ff_node {
115public:
116
117 ff_mapCUDAManaged(kernelF *mapF, void * env, void * in, void * out, size_t s):
118 oneshot(true),
119 Task( (typename T::env_type *)env, (typename T::inout_type *)in, (typename T::inout_type *)out, s ),
120 kernel(mapF) {
121 assert(in);
122 ff_node::skipfirstpop(true);
123 maxThreads=maxBlocks=0;
124 oldSize=0;
125 inPtr = outPtr = NULL;
126 envPtr = NULL;
127 }

129 int run(bool=false) { return ff_node::run(); }
130 int wait() { return ff_node::wait(); }

134 if (wait()<0) return -1;
135 return 0;
136 }

138 double ffTime() { return ff_node::ffTime(); }
139 double ffwTime() { return ff_node::wffTime(); }
140
141 const T* getTask() const { return &Task; }
142
143 void cleanup() { if (kernel) delete kernel; }

145protected:
146
147 int svc_init() {
148 int deviceID = 0; // FIX: we have to manage multiple devices
149 cudaDeviceProp deviceProp;
150
151 cudaSetDevice(deviceID);
152 if (cudaGetDeviceProperties(&deviceProp, deviceID) != cudaSuccess)
153 error("mapCUDA, error getting device properties\n");
154
155 if(deviceProp.major == 1 && deviceProp.minor < 2)
156 maxThreads = 256;
157 else
158 maxThreads = deviceProp.maxThreadsPerBlock;
159 maxBlocks = deviceProp.maxGridSize[0];
160
161 if(cudaStreamCreate(&stream) != cudaSuccess)
162 error("mapCUDA, error creating stream\n");
163
164 return 0;
165 }
166
167 void * svc(void* task) {
168 Task.setTask(task);
169 size_t size = Task.size();
170 inPtr = (typename T::inout_type*) Task.getInPtr();
171 outPtr = (typename T::inout_type*) Task.newOutPtr();
172 envPtr = (typename T::env_type*) Task.getEnvPtr();
173
174 size_t thxblock = std::min(maxThreads, size);
175 size_t blockcnt = std::min(size/thxblock + (size%thxblock == 0 ?0:1), maxBlocks);
176
177 mapCUDAKernelManaged<typename T::env_type, typename T::inout_type, kernelF>
178 <<<blockcnt,thxblock,0,stream>>>(*kernel, envPtr, inPtr, outPtr, size);
179
180 cudaStreamSynchronize(stream);
181
182 // return (NULL);
183 return (oneshot?NULL:outPtr);
184 }
185
186 void svc_end() {
187
188 if(cudaStreamDestroy(stream) != cudaSuccess)
189 error("mapCUDA, error destroying stream\n");
190 }
191private:
192 const bool oneshot;
193 T Task;
194 kernelF *kernel; // user function
195 cudaStream_t stream;
196 size_t maxThreads;
197 size_t maxBlocks;
198 size_t oldSize;
199 typename T::inout_type* inPtr;
200 typename T::inout_type* outPtr;
201 typename T::env_type* envPtr;
202};
