6// This file implements sorting algorithms. Some of these are equivalent to
7// std C++ sorting algorithms, while others don't have equivalents in the
8// C++ standard. We implement the following sorting algorithms:
9// is_sorted --
10// sort -- Unstable. The implementation of this is mapped to quick_sort by default.
11// quick_sort -- Unstable. This is actually an intro-sort (quick sort with switch to insertion sort).
12// tim_sort -- Stable.
13// tim_sort_buffer -- Stable.
14// partial_sort -- Unstable.
15// insertion_sort -- Stable.
16// shell_sort -- Unstable.
17// heap_sort -- Unstable.
18// stable_sort -- Stable. The implementation of this is simply mapped to merge_sort.
19// merge --
20// merge_sort -- Stable.
21// merge_sort_buffer -- Stable.
22// nth_element -- Unstable.
23// radix_sort -- Stable. Important and useful sort for integral data, and faster than all others for this.
24// comb_sort -- Unstable. Possibly the best combination of small code size but fast sort.
25// bubble_sort -- Stable. Useful in practice for sorting tiny sets of data (<= 10 elements).
26// selection_sort* -- Unstable.
27// shaker_sort* -- Stable.
28// bucket_sort* -- Stable.

30// * Found in sort_extra.h.

176 template <typename InputIterator1, typename InputIterator2, typename OutputIterator, typename Compare>
177 OutputIterator merge(InputIterator1 first1, InputIterator1 last1, InputIterator2 first2, InputIterator2 last2, OutputIterator result, Compare compare)
178 {
179 while((first1 != last1) && (first2 != last2))
180 {
181 if(compare(*first2, *first1))
182 {
183 EASTL_VALIDATE_COMPARE(!compare(*first1, *first2)); // Validate that the compare function is sane.
184 *result = *first2;
185 ++first2;
186 }
187 else
188 {
189 *result = *first1;
190 ++first1;
191 }
192 ++result;
193 }
194
195 // Check which list is empty and explicitly copy remaining items from the other list.
196 // For performance reasons, only a single copy operation is invoked to avoid the potential overhead
197 // introduced by chaining two copy operations together. Even if a copy is of zero size there can
198 // be overhead from calling memmove with a zero size copy.
199 if (first1 == last1)
200 {
201 return eastl::copy(first2, last2, result);
202 }
203 else
204 {
205 return eastl::copy(first1, last1, result);
206 }
207 }

232 template <typename BidirectionalIterator, typename StrictWeakOrdering>
233 void insertion_sort(BidirectionalIterator first, BidirectionalIterator last, StrictWeakOrdering compare)
234 {
235 typedef typename eastl::iterator_traits<BidirectionalIterator>::value_type value_type;
236
237 if (first != last)
238 {
239 BidirectionalIterator i = first;
240
241 for (++i; i != last; ++i)
242 {
243 value_type insertValue(eastl::move(*i));
244 BidirectionalIterator insertPosition = i;
245
246 for (BidirectionalIterator movePosition = i; movePosition != first && compare(insertValue, *(--movePosition)); --insertPosition)
247 {
248 EASTL_VALIDATE_COMPARE(!compare(*movePosition, insertValue));
249 *insertPosition = eastl::move(*movePosition);
250 }
251
252 *insertPosition = eastl::move(insertValue);
253 }
254 }

278 template <typename RandomAccessIterator, typename StrictWeakOrdering>
279 void shell_sort(RandomAccessIterator first, RandomAccessIterator last, StrictWeakOrdering compare)
280 {
281 typedef typename eastl::iterator_traits<RandomAccessIterator>::difference_type difference_type;
282
283 // We use the Knuth 'h' sequence below, as it is easy to calculate at runtime.
284 // However, possibly we are better off using a different sequence based on a table.
285 // One such sequence which averages slightly better than Knuth is:
286 // 1, 5, 19, 41, 109, 209, 505, 929, 2161, 3905, 8929, 16001, 36289,
287 // 64769, 146305, 260609, 587521, 1045505, 2354689, 4188161, 9427969, 16764929
288
289 if(first != last)
290 {
291 const difference_type nSize = last - first;
292 difference_type nSpace = 1; // nSpace is the 'h' value of the ShellSort algorithm.
293
294 while(nSpace < nSize)
295 nSpace = (nSpace * 3) + 1; // This is the Knuth 'h' sequence: 1, 4, 13, 40, 121, 364, 1093, 3280, 9841, 29524, 88573, 265720, 797161, 2391484, 7174453, 21523360, 64570081, 193710244,
296
297 for(nSpace = (nSpace - 1) / 3; nSpace >= 1; nSpace = (nSpace - 1) / 3) // Integer division is less than ideal.
298 {
299 for(difference_type i = 0; i < nSpace; i++)
300 {
301 const RandomAccessIterator iInsertFirst = first + i;
302
303 // Note: we can only move the iterator forward if we know we won't overrun the
304 // end(), otherwise we can invoke undefined behaviour. So we need to check we
305 // have enough space before moving the iterator.
306 RandomAccessIterator iSorted = iInsertFirst;
307 while(distance(iSorted, last) > nSpace)
308 {
309 iSorted += nSpace;
310
311 RandomAccessIterator iCurrent = iSorted;
312 for(RandomAccessIterator iBack = iSorted - nSpace; (iCurrent != iInsertFirst) && compare(*iCurrent, *iBack); iCurrent = iBack, iBack -= nSpace)
313 {
314 EASTL_VALIDATE_COMPARE(!compare(*iBack, *iCurrent)); // Validate that the compare function is sane.
315 eastl::iter_swap(iCurrent, iBack);
316 }
317 }
318 }
319 }
320 }
321 } // shell_sort

359 // Sorts a range whose initial (start - first) entries are already sorted.
360 // This function is a useful helper to the tim_sort function.
361 // This is the same as insertion_sort except that it has a start parameter which indicates
362 // where the start of the unsorted data is.
363 template <typename BidirectionalIterator, typename StrictWeakOrdering>
364 void insertion_sort_already_started(BidirectionalIterator first, BidirectionalIterator last, BidirectionalIterator start, StrictWeakOrdering compare)
365 {
366 typedef typename eastl::iterator_traits<BidirectionalIterator>::value_type value_type;
367
368 if (first != last) // if the range is non-empty...
369 {
370 BidirectionalIterator iCurrent, iNext, iSorted = start - 1;
371
372 for (++iSorted; iSorted != last; ++iSorted)
373 {
374 const value_type temp(*iSorted);
375
376 iNext = iCurrent = iSorted;
377
378 for (--iCurrent; (iNext != first) && compare(temp, *iCurrent); --iNext, --iCurrent)
379 {
380 EASTL_VALIDATE_COMPARE(!compare(*iCurrent, temp)); // Validate that the compare function is sane.
381 *iNext = *iCurrent;
382 }
383
384 *iNext = temp;
385 }
386 }
387 }

409 template <typename RandomAccessIterator, typename T, typename StrictWeakOrdering, typename difference_type, int InsertionSortLimit>
410 class MergeSorter
411 {
412 public:
413 static void sort(RandomAccessIterator first, RandomAccessIterator last, T* pBuffer, StrictWeakOrdering compare)
414 {
415 if (sort_impl(first, last, pBuffer, difference_type(0), compare) == RL_Buffer)
416 {
417 const difference_type nCount = last - first;
418 eastl::copy<T*, RandomAccessIterator>(pBuffer, pBuffer + nCount, first);
419 }
420 EASTL_DEV_ASSERT((eastl::is_sorted<RandomAccessIterator, StrictWeakOrdering>(first, last, compare)));
421 }
422
423 private:
424 static_assert(InsertionSortLimit > 1, "Sequences of length 1 are already sorted. Use a larger value for InsertionSortLimit");
425
426 enum ResultLocation
427 {
428 RL_SourceRange, // i.e. result is in the range defined by [first, last)
429 RL_Buffer, // i.e. result is in pBuffer
430 };
431
432 // sort_impl
433 //
434 // This sort routine sorts the data in [first, last) and places the result in pBuffer or in the original range of the input. The actual
435 // location of the data is indicated by the enum returned.
436 //
437 // lastSortedEnd is used to specify a that data in the range [first, first + lastSortedEnd] is already sorted. This information is used
438 // to avoid unnecessary merge sorting of already sorted data. lastSortedEnd is a hint, and can be an under estimate of the sorted elements
439 // (i.e. it is legal to pass 0).
440 static ResultLocation sort_impl(RandomAccessIterator first, RandomAccessIterator last, T* pBuffer, difference_type lastSortedEnd, StrictWeakOrdering compare)
441 {
442 const difference_type nCount = last - first;
443
444 if (lastSortedEnd < 1)
445 {
446 lastSortedEnd = eastl::is_sorted_until<RandomAccessIterator, StrictWeakOrdering>(first, last, compare) - first;
447 }
448
449 // Sort the region unless lastSortedEnd indicates it is already sorted.
450 if (lastSortedEnd < nCount)
451 {
452 // If the size is less than or equal to InsertionSortLimit use insertion sort instead of recursing further.
453 if (nCount <= InsertionSortLimit)
454 {
455 eastl::Internal::insertion_sort_already_started<RandomAccessIterator, StrictWeakOrdering>(first, last, first + lastSortedEnd, compare);
456 return RL_SourceRange;
457 }
458 else
459 {
460 const difference_type nMid = nCount / 2;
461
462 ResultLocation firstHalfLocation = RL_SourceRange;
463 // Don't sort the first half if it is already sorted.
464 if (lastSortedEnd < nMid)
465 {
466 firstHalfLocation = sort_impl(first, first + nMid, pBuffer, lastSortedEnd, compare);
467 }
468
469 ResultLocation secondHalfLocation = sort_impl(first + nMid, last, pBuffer + nMid, lastSortedEnd - nMid, compare);
470
471 return merge_halves(first, last, nMid, pBuffer, firstHalfLocation, secondHalfLocation, compare);
472 }
473 }
474 else
475 {
476 EASTL_DEV_ASSERT((eastl::is_sorted<RandomAccessIterator, StrictWeakOrdering>(first, last, compare)));
477 return RL_SourceRange;
478 }
479 }
480
481 // merge_halves
482 //
483 // Merge two sorted regions of elements.
484 // The inputs to this method effectively define two large buffers. The variables 'firstHalfLocation' and 'secondHalfLocation' define where the data to be
485 // merged is located within the two buffers. It is entirely possible that the two areas to be merged could be entirely located in either of the larger buffers.
486 // Upon returning the merged results will be in one of the two buffers (indicated by the return result).
487 static ResultLocation merge_halves(RandomAccessIterator first, RandomAccessIterator last, difference_type nMid, T* pBuffer, ResultLocation firstHalfLocation, ResultLocation secondHalfLocation, StrictWeakOrdering compare)
488 {
489 const difference_type nCount = last - first;
490 if (firstHalfLocation == RL_SourceRange)
491 {
492 if (secondHalfLocation == RL_SourceRange)
493 {
494 eastl::merge<RandomAccessIterator, RandomAccessIterator, T*, StrictWeakOrdering>(first, first + nMid, first + nMid, last, pBuffer, compare);
495 EASTL_DEV_ASSERT((eastl::is_sorted<T*, StrictWeakOrdering>(pBuffer, pBuffer + nCount, compare)));
496 return RL_Buffer;
497 }
498 else
499 {
500 eastl::copy(first, first + nMid, pBuffer);
501 eastl::merge<T*, T*, RandomAccessIterator, StrictWeakOrdering>(pBuffer, pBuffer + nMid, pBuffer + nMid, pBuffer + nCount, first, compare);
502 EASTL_DEV_ASSERT((eastl::is_sorted<RandomAccessIterator, StrictWeakOrdering>(first, last, compare)));
503 return RL_SourceRange;
504 }
505 }
506 else
507 {
508 if (secondHalfLocation == RL_SourceRange)
509 {
510 eastl::copy(first + nMid, last, pBuffer + nMid);
511 eastl::merge<T*, T*, RandomAccessIterator, StrictWeakOrdering>(pBuffer, pBuffer + nMid, pBuffer + nMid, pBuffer + nCount, first, compare);
512 EASTL_DEV_ASSERT((eastl::is_sorted<RandomAccessIterator, StrictWeakOrdering>(first, last, compare)));
513 return RL_SourceRange;
514 }
515 else
516 {
517 eastl::merge<T*, T*, RandomAccessIterator, StrictWeakOrdering>(pBuffer, pBuffer + nMid, pBuffer + nMid, pBuffer + nCount, first, compare);
518 EASTL_DEV_ASSERT((eastl::is_sorted<RandomAccessIterator, StrictWeakOrdering>(first, last, compare)));
519 return RL_SourceRange;
520 }
521 }
522 }

553 void merge_sort(RandomAccessIterator first, RandomAccessIterator last, Allocator& allocator, StrictWeakOrdering compare)
554 {
555 typedef typename eastl::iterator_traits<RandomAccessIterator>::difference_type difference_type;
556 typedef typename eastl::iterator_traits<RandomAccessIterator>::value_type value_type;
557
558 const difference_type nCount = last - first;
559
560 if(nCount > 1)
561 {
562 // We need to allocate an array of nCount value_type objects as a temporary buffer.
563 value_type* const pBuffer = (value_type*)allocate_memory(allocator, nCount * sizeof(value_type), EASTL_ALIGN_OF(value_type), 0);
564 eastl::uninitialized_fill(pBuffer, pBuffer + nCount, value_type());
565
566 eastl::merge_sort_buffer<RandomAccessIterator, value_type, StrictWeakOrdering>
567 (first, last, pBuffer, compare);
568
569 eastl::destruct(pBuffer, pBuffer + nCount);
570 EASTLFree(allocator, pBuffer, nCount * sizeof(value_type));
571 }

598 template<typename InputIterator, typename Predicate>
599 InputIterator partition(InputIterator begin, InputIterator end, Predicate predicate)
600 {
601 if(begin != end)
602 {
603 while(predicate(*begin))
604 {
605 if(++begin == end)
606 return begin;
607 }
608
609 InputIterator middle = begin;
610
611 while(++middle != end)
612 {
613 if(predicate(*middle))
614 {
615 eastl::swap(*begin, *middle);
616 ++begin;
617 }
618 }
619 }
620
621 return begin;
622 }

629 template <typename ForwardIterator, typename Predicate>
630 ForwardIterator stable_partition(ForwardIterator first, ForwardIterator last, Predicate pred)
631 {
632 first = eastl::find_if_not(first, last, pred);
633
634 if (first == last)
635 return first;
636
637 typedef typename iterator_traits<ForwardIterator>::value_type value_type;
638
639 const auto requested_size = eastl::distance(first, last);
640
641 auto allocator = *get_default_allocator(0);
642 value_type* const buffer =
643 (value_type*)allocate_memory(allocator, requested_size * sizeof(value_type), EASTL_ALIGN_OF(value_type), 0);
644 eastl::uninitialized_fill(buffer, buffer + requested_size, value_type());
645
646 ForwardIterator result1 = first;
647 value_type* result2 = buffer;
648
649 *result2 = eastl::move(*first);
650 ++result2;
651 ++first;
652 for (; first != last; ++first)
653 {
654 if (pred(*first))
655 {
656 *result1 = eastl::move(*first);
657 ++result1;
658 }
659 else
660 {
661 *result2 = eastl::move(*first);
662 ++result2;
663 }
664 }
665
666 eastl::copy(buffer, result2, result1);
667
668 eastl::destruct(buffer, buffer + requested_size);
669 EASTLFree(allocator, buffer, requested_size * sizeof(value_type));
670
671 return result1;
672 }

676 //
677 // We do the "introspection sort" variant of quick sort which is now
678 // well-known and understood. You can read about this algorithm in
679 // many articles on quick sort, but briefly what it does is a median-
680 // of-three quick sort whereby the recursion depth is limited to a
681 // some value (after which it gives up on quick sort and switches to
682 // a heap sort) and whereby after a certain amount of sorting the

719 template <typename RandomAccessIterator, typename T>
720 inline RandomAccessIterator get_partition_impl(RandomAccessIterator first, RandomAccessIterator last, T&& pivotValue)
721 {
722 for(; ; ++first)
723 {
724 while(*first < pivotValue)
725 {
726 EASTL_VALIDATE_COMPARE(!(pivotValue < *first)); // Validate that the compare function is sane.
727 ++first;
728 }
729 --last;
730
731 while(pivotValue < *last)

733 EASTL_VALIDATE_COMPARE(!(*last < pivotValue)); // Validate that the compare function is sane.
734 --last;
735 }
736
737 if(first >= last) // Random access iterators allow operator >=
738 return first;
739
740 eastl::iter_swap(first, last);
741 }

1169 // tim_sort_count_run
1170 //
1171 // Finds the length of a run which is already sorted (either up or down).
1172 // If the run is in reverse order, this function puts it in regular order.
1173 //
1174 template <typename RandomAccessIterator, typename StrictWeakOrdering>
1175 intptr_t tim_sort_count_run(const RandomAccessIterator first, const intptr_t start, const intptr_t size, StrictWeakOrdering compare)
1176 {
1177 if((size - start) > 1) // If there is anything in the set...
1178 {
1179 intptr_t curr = (start + 2);
1180
1181 if(!compare(*(first + start + 1), *(first + start))) // If (first[start + 1] >= first[start]) (If the run is increasing) ...
1182 {
1183 for(;; ++curr)
1184 {
1185 if(curr >= (size - 1)) // If we are at the end of the data... this run is done.
1186 break;
1187
1188 if(compare(*(first + curr), *(first + curr - 1))) // If this item is not in order... this run is done.
1189 break;
1190 }
1191 }
1192 else // Else it is decreasing.
1193 {
1194 for(;; ++curr)
1195 {
1196 if(curr >= (size - 1)) // If we are at the end of the data... this run is done.
1197 break;
1198
1199 if(!compare(*(first + curr), *(first + curr - 1))) // If this item is not in order... this run is done.
1200 break; // Note that we intentionally compare against <= 0 and not just < 0. This is because
1201 } // The reverse_elements call below could reverse two equal elements and break our stability requirement.
1202
1203 reverse_elements(first, start, curr - 1);
1204 }
1205
1206 return (curr - start);
1207 }
1208
1209 // Else we have just one item in the set.
1210 return 1;
1211 }

1253 template <typename RandomAccessIterator, typename T, typename StrictWeakOrdering>
1254 void tim_sort_merge(RandomAccessIterator first, const tim_sort_run* run_stack, const intptr_t stack_curr,
1255 T* pBuffer, StrictWeakOrdering compare)
1256 {
1257 const intptr_t A = run_stack[stack_curr - 2].length;
1258 const intptr_t B = run_stack[stack_curr - 1].length;
1259 const intptr_t curr = run_stack[stack_curr - 2].start;
1260
1261 EASTL_DEV_ASSERT((A < 10000000) && (B < 10000000) && (curr < 10000000)); // Sanity check.
1262
1263 if(A < B) // If the first run is shorter than the second run... merge left.
1264 {
1265 // Copy to another location so we have room in the main array to put the sorted items.
1266 eastl::copy(first + curr, first + curr + A, pBuffer);
1267
1268 #if EASTL_DEV_DEBUG
1269 typedef typename eastl::iterator_traits<RandomAccessIterator>::value_type value_type;
1270
1271 for(intptr_t i = 0; i < A; i++)
1272 *(first + curr + i) = value_type();
1273 #endif
1274
1275 intptr_t i = 0;
1276 intptr_t j = curr + A;
1277
1278 for(intptr_t k = curr; k < curr + A + B; k++)
1279 {
1280 if((i < A) && (j < (curr + A + B)))
1281 {
1282 if(!compare(*(first + j), *(pBuffer + i))) // If (first[j] >= pBuffer[i])...
1283 *(first + k) = *(pBuffer + i++);
1284 else
1285 *(first + k) = *(first + j++);
1286 }
1287 else if(i < A)
1288 *(first + k) = *(pBuffer + i++);
1289 else
1290 *(first + k) = *(first + j++);
1291 }
1292 }
1293 else // Else the second run is equal or shorter... merge right.
1294 {
1295 eastl::copy(first + curr + A, first + curr + A + B, pBuffer);
1296
1297 intptr_t i = B - 1;
1298 intptr_t j = curr + A - 1;
1299
1300 for(intptr_t k = curr + A + B - 1; k >= curr; k--)
1301 {
1302 if((i >= 0) && (j >= curr))
1303 {
1304 if(compare(*(pBuffer + i), *(first + j))) // If (pBuffer[i] < first[j]) ...
1305 *(first + k) = *(first + j--);
1306 else
1307 *(first + k) = *(pBuffer + i--);
1308 }
1309 else if(i >= 0)
1310 *(first + k) = *(pBuffer + i--);
1311 else
1312 *(first + k) = *(first + j--);

1319 //
1320 // ------------------------------------------------------------------------
1321 // What turned out to be a good compromise maintains two invariants on the
1322 // stack entries, where A, B and C are the lengths of the three righmost
1323 // not-yet merged slices:
1324 // 1. A > B+C
1325 // 2. B > C
1326 // ------------------------------------------------------------------------
1327 //
1328 static inline bool timsort_check_invariant(tim_sort_run* run_stack, const intptr_t stack_curr)
1329 {
1330 // To do: Optimize this for the most common type of values.
1331 if(stack_curr > 2)
1332 {
1333 const intptr_t A = run_stack[stack_curr - 3].length;
1334 const intptr_t B = run_stack[stack_curr - 2].length;
1335 const intptr_t C = run_stack[stack_curr - 1].length;
1336
1337 EASTL_DEV_ASSERT((A < 10000000) && (B < 10000000) && (C < 10000000)); // Sanity check.
1338
1339 if((A <= (B + C)) || (B <= C))
1340 return true; // Merge the right-most runs.
1341 }
1342 else if(stack_curr == 2)
1343 {
1344 const intptr_t A = run_stack[stack_curr - 2].length;
1345 const intptr_t B = run_stack[stack_curr - 1].length;
1346
1347 EASTL_DEV_ASSERT((A < 10000000) && (B < 10000000)); // Sanity check.
1348
1349 if(A <= B)
1350 return true; // Merge the right-most runs.
1351 }
1352
1353 return false; // Don't merge the right-most runs.
1354 }

1357 template <typename RandomAccessIterator, typename T, typename StrictWeakOrdering>
1358 intptr_t tim_sort_collapse(RandomAccessIterator first, tim_sort_run* run_stack, intptr_t stack_curr,
1359 T* pBuffer, const intptr_t size, StrictWeakOrdering compare)
1360 {
1361 // If the run_stack only has one thing on it, we are done with the collapse.
1362 while(stack_curr > 1)
1363 {
1364 // If this is the last merge, just do it.
1365 if((stack_curr == 2) && ((run_stack[0].length + run_stack[1].length) == size))
1366 {
1367 tim_sort_merge<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr, pBuffer, compare);
1368 run_stack[0].length += run_stack[1].length;
1369 stack_curr--;
1370
1371 #if EASTL_DEV_DEBUG
1372 memset(&run_stack[stack_curr], 0, sizeof(run_stack[stack_curr]));
1373 #endif
1374
1375 break;
1376 }
1377 // Check if the invariant is off for a run_stack of 2 elements.
1378 else if((stack_curr == 2) && (run_stack[0].length <= run_stack[1].length))
1379 {
1380 tim_sort_merge<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr, pBuffer, compare);
1381 run_stack[0].length += run_stack[1].length;
1382 stack_curr--;
1383
1384 #if EASTL_DEV_DEBUG
1385 memset(&run_stack[stack_curr], 0, sizeof(run_stack[stack_curr]));
1386 #endif
1387
1388 break;
1389 }
1390 else if (stack_curr == 2)
1391 break;
1392
1393 const intptr_t A = run_stack[stack_curr - 3].length;
1394 const intptr_t B = run_stack[stack_curr - 2].length;
1395 const intptr_t C = run_stack[stack_curr - 1].length;
1396
1397 if(A <= (B + C)) // Check first invariant.
1398 {
1399 if(A < C)
1400 {
1401 tim_sort_merge<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr - 1, pBuffer, compare);
1402
1403 stack_curr--;
1404 run_stack[stack_curr - 2].length += run_stack[stack_curr - 1].length; // Merge A and B.
1405 run_stack[stack_curr - 1] = run_stack[stack_curr];
1406
1407 #if EASTL_DEV_DEBUG
1408 EASTL_DEV_ASSERT((run_stack[stack_curr - 2].start + run_stack[stack_curr - 2].length) <= size);
1409 EASTL_DEV_ASSERT((run_stack[stack_curr - 1].start + run_stack[stack_curr - 1].length) <= size);
1410 memset(&run_stack[stack_curr], 0, sizeof(run_stack[stack_curr]));
1411 #endif
1412 }
1413 else
1414 {
1415 tim_sort_merge<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr, pBuffer, compare); // Merge B and C.
1416
1417 stack_curr--;
1418 run_stack[stack_curr - 1].length += run_stack[stack_curr].length;
1419
1420 #if EASTL_DEV_DEBUG
1421 EASTL_DEV_ASSERT((run_stack[stack_curr - 1].start + run_stack[stack_curr - 1].length) <= size);
1422 memset(&run_stack[stack_curr], 0, sizeof(run_stack[stack_curr]));
1423 #endif
1424 }
1425 }
1426 else if(B <= C) // Check second invariant
1427 {
1428 tim_sort_merge<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr, pBuffer, compare);
1429

1450 template <typename RandomAccessIterator, typename T, typename StrictWeakOrdering>
1451 bool tim_sort_add_run(tim_sort_run* run_stack, RandomAccessIterator first, T* pBuffer, const intptr_t size, const intptr_t minrun,
1452 intptr_t& len, intptr_t& run, intptr_t& curr, intptr_t& stack_curr, StrictWeakOrdering compare)
1453 {

1455 run = minrun;
1456

1458 run = minrun;
1459
1460 if(run > (size - curr)) // But if there isn't minrun data remaining, just sort what's remaining.
1461 run = (size - curr);
1462
1463 if(run > len) // If there is any additional data we want to sort to bring up the run length to minrun.
1464 {
1465 insertion_sort_already_started<RandomAccessIterator, StrictWeakOrdering>(first + curr, first + curr + run, first + curr + len, compare);
1466 len = run;
1467 }
1468
1469 // At this point, run will be equal to minrun or will go to the end of our data.
1470 // Add this run to our stack of runs.
1471 EASTL_DEV_ASSERT(stack_curr < kTimSortStackSize);
1472 EASTL_DEV_ASSERT((curr >= 0) && (curr < size) && ((curr + len) <= size));
1473
1474 run_stack[stack_curr].start = curr;
1475 run_stack[stack_curr].length = len;
1476 stack_curr++;
1477
1478 // Move to the beginning of the next run in the data.
1479 curr += len;
1480
1481 if(curr == size) // If we have hit the end of the data...
1482 {
1483 while(stack_curr > 1) // If there is any more than one run... (else all the data is sorted)
1484 {
1485 tim_sort_merge<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr, pBuffer, compare);
1486
1487 run_stack[stack_curr - 2].length += run_stack[stack_curr - 1].length;
1488 stack_curr--;
1489
1490 #if EASTL_DEV_DEBUG
1491 EASTL_DEV_ASSERT((run_stack[stack_curr - 1].start + run_stack[stack_curr - 1].length) <= size);
1492 memset(&run_stack[stack_curr], 0, sizeof(run_stack[stack_curr]));
1493 #endif
1494 }
1495
1496 return true; // We are done with sorting.
1497 }
1498
1499 return false;

1505 // tim_sort_buffer
1506 //
1508 // Implements the tim-sort sorting algorithm with a user-provided scratch buffer.
1509 // http://en.wikipedia.org/wiki/Timsort
1510 // This sort is the fastest sort when sort stability (maintaining order of equal values) is required and
1511 // data sets are non-trivial (size >= 15). It's also the fastest sort (e.g. faster than quick_sort) for
1512 // the case that at at least half your data is already sorted. Otherwise, eastl::quick_sort is about 10%
1513 // faster than tim_sort_buffer but is not a stable sort. There are some reports that tim_sort outperforms
1514 // quick_sort but most of these aren't taking into account that optimal quick_sort implementations use
1515 // a hybrid approach called "introsort" (http://en.wikipedia.org/wiki/Introsort) which improves quick_sort
1516 // considerably in practice.
1517 //
1518 // Strengths:
1519 // - Fastest stable sort for most sizes of data.
1520 // - Fastest sort for containers of data already mostly sorted.
1521 // - Simpler to understand than quick_sort.
1522 //
1523 // Weaknesses:
1524 // - User must provide a scratch buffer, otherwise the buffer is dynamically allocated during runtime.
1525 // - Not as fast as quick_sort for the general case of randomized data.
1526 // - Requires a RandomAccessIterator; thus must be on an array container type and not a list container type.
1527 // - Uses a lot of code to implement; thus it's not great when there is little room for more code.
1528 //
1529 // The pBuffer parameter must hold at least ((last-first)/2) elements (i.e. half the elements of the container).
1530 // This minimum size is a worst-case size requirement, but handles all possible cases. pBuffer is just a scratch
1531 // buffer and is not needed after the return of this function, and doesn't need to be seeded with any particular
1532 // values upon entering this function.
1533 //
1534 // Example usage:
1535 // int intArray[64];
1536 // int buffer[32];
1537 // ...
1538 // tim_sort_buffer(intArray, intArray + 64, buffer);
1539 //
1540 template <typename RandomAccessIterator, typename T, typename StrictWeakOrdering>
1541 void tim_sort_buffer(RandomAccessIterator first, RandomAccessIterator last, T* pBuffer, StrictWeakOrdering compare)
1542 {
1543 using namespace Internal;
1544
1545 // To consider: Convert the implementation to use first/last instead of first/size.
1546 const intptr_t size = (intptr_t)(last - first);
1547 if (size == 0)
1548 {
1549 // This branch is necessary because the expression `first + 1` below is undefined
1550 // behaviour when first is nullptr (for example when it is the begin() iterator of an
1551 // empty vector).
1552 return;
1553 }
1554 else if (size < 64)
1555 {
1556 insertion_sort_already_started(first, first + size, first + 1, compare);
1557 }
1558 else
1559 {
1560 tim_sort_run run_stack[kTimSortStackSize];
1561 intptr_t stack_curr = 0;
1562 intptr_t len, run;
1563 intptr_t curr = 0;
1564 const intptr_t minrun = timsort_compute_minrun(size);
1565
1566 #if EASTL_DEV_DEBUG
1567 memset(run_stack, 0, sizeof(run_stack));
1568 #endif
1569
1570 if(tim_sort_add_run<RandomAccessIterator, T, StrictWeakOrdering>(run_stack, first, pBuffer, size, minrun, len, run, curr, stack_curr, compare))
1571 return;
1572 if(tim_sort_add_run<RandomAccessIterator, T, StrictWeakOrdering>(run_stack, first, pBuffer, size, minrun, len, run, curr, stack_curr, compare))
1573 return;
1574 if(tim_sort_add_run<RandomAccessIterator, T, StrictWeakOrdering>(run_stack, first, pBuffer, size, minrun, len, run, curr, stack_curr, compare))
1575 return;
1576
1577 for(;;)
1578 {
1579 if(timsort_check_invariant(run_stack, stack_curr))
1580 stack_curr = tim_sort_collapse<RandomAccessIterator, T, StrictWeakOrdering>(first, run_stack, stack_curr, pBuffer, size, compare);
1581 else
1582 {
1583 if(tim_sort_add_run<RandomAccessIterator, T, StrictWeakOrdering>(run_stack, first, pBuffer, size, minrun, len, run, curr, stack_curr, compare))
1584 break;
1585 }
1586 }
1587 }

1653 // The radix_sort implementation uses two optimizations that are not part of a typical radix sort implementation.
1654 // 1. Computing a histogram (i.e. finding the number of elements per bucket) for the next pass is done in parallel with the loop that "scatters"
1655 // elements in the current pass. The advantage is that it avoids the memory traffic / cache pressure of reading keys in a separate operation.
1656 // Note: It would also be possible to compute all histograms in a single pass. However, that would increase the amount of stack space used and
1657 // also increase cache pressure slightly. However, it could still be faster under some situations.
1658 // 2. If all elements are mapped to a single bucket, then there is no need to perform a scatter operation. Instead the elements are left in place
1659 // and only copied if they need to be copied to the final output buffer.
1660 template <typename RandomAccessIterator, typename ExtractKey, int DigitBits, typename IntegerType>
1661 void radix_sort_impl(RandomAccessIterator first,
1662 RandomAccessIterator last,
1663 RandomAccessIterator buffer,
1664 ExtractKey extractKey,
1665 IntegerType)
1666 {
1667 RandomAccessIterator srcFirst = first;
1668 EA_CONSTEXPR_OR_CONST size_t numBuckets = 1 << DigitBits;
1669 EA_CONSTEXPR_OR_CONST IntegerType bucketMask = numBuckets - 1;
1670
1671 // The alignment of this variable isn't required; it merely allows the code below to be faster on some platforms.
1672 uint32_t EA_PREFIX_ALIGN(EASTL_PLATFORM_PREFERRED_ALIGNMENT) bucketSize[numBuckets];
1673 uint32_t EA_PREFIX_ALIGN(EASTL_PLATFORM_PREFERRED_ALIGNMENT) bucketPosition[numBuckets];
1674
1675 RandomAccessIterator temp;
1676 uint32_t i;
1677 bool doSeparateHistogramCalculation = true;
1678
1679 constexpr uint32_t kMaxDigitBits = 8 * sizeof(IntegerType);
1680 for (uint32_t j = 0; j < kMaxDigitBits; j += DigitBits)
1681 {
1682 if (doSeparateHistogramCalculation)
1683 {
1684 memset(bucketSize, 0, sizeof(bucketSize));
1685 // Calculate histogram for the first scatter operation
1686 for (temp = srcFirst; temp != last; ++temp)
1687 ++bucketSize[(extractKey(*temp) >> j) & bucketMask];
1688 }
1689
1690 // If a single bucket contains all of the elements, then don't bother redistributing all elements to the
1691 // same bucket.
1692 if (bucketSize[((extractKey(*srcFirst) >> j) & bucketMask)] == uint32_t(last - srcFirst))
1693 {
1694 // Set flag to ensure histogram is computed for next digit position.
1695 doSeparateHistogramCalculation = true;
1696 }
1697 else
1698 {
1699 // The histogram is either not needed or it will be calculated in parallel with the scatter operation below for better cache efficiency.
1700 doSeparateHistogramCalculation = false;
1701
1702 // If this is the last digit position, then don't calculate a histogram
1703 const uint32_t jNext = j + DigitBits;
1704 if (jNext >= kMaxDigitBits)
1705 {
1706 bucketPosition[0] = 0;
1707 for (i = 0; i < numBuckets - 1; i++)
1708 {
1709 bucketPosition[i + 1] = bucketPosition[i] + bucketSize[i];
1710 }
1711
1712 for (temp = srcFirst; temp != last; ++temp)
1713 {
1714 IntegerType key = extractKey(*temp);
1715 const size_t digit = (key >> j) & bucketMask;
1716 buffer[bucketPosition[digit]++] = *temp;
1717 }
1718 }
1719 // Compute the histogram while performing the scatter operation
1720 else
1721 {
1722 bucketPosition[0] = 0;
1723 for (i = 0; i < numBuckets - 1; i++)
1724 {
1725 bucketPosition[i + 1] = bucketPosition[i] + bucketSize[i];
1726 bucketSize[i] = 0; // Clear the bucket for the next pass
1727 }
1728 bucketSize[numBuckets - 1] = 0;
1729
1730 for (temp = srcFirst; temp != last; ++temp)
1731 {
1732 const IntegerType key = extractKey(*temp);
1733 const size_t digit = (key >> j) & bucketMask;
1734 buffer[bucketPosition[digit]++] = *temp;
1735
1736 // Update histogram for the next scatter operation
1737 ++bucketSize[(key >> jNext) & bucketMask];
1738 }
1739 }
1740
1741 last = buffer + (last - srcFirst);
1742 temp = srcFirst;
1743 srcFirst = buffer;
1744 buffer = temp;
1745 }
1746 }
1747
1748 if (srcFirst != first)
1749 {
1750 // Copy values back into the expected buffer
1751 for (temp = srcFirst; temp != last; ++temp)
1752 *buffer++ = *temp;
1753 }
1754 }
