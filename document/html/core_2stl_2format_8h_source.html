53 return ctx.end();

55 template<typename FormatContext>
56 auto format(const luisa::Vector<T, N> &v, FormatContext &ctx) const noexcept {
57 using namespace std::string_view_literals;
58 using luisa::uint;
59 using luisa::ushort;
60 using luisa::slong;
61 using luisa::ulong;
62 using luisa::half;
63 constexpr auto type_name =
64 std::is_same_v<T, bool> ? "bool"sv :
65 std::is_same_v<T, short> ? "short"sv :
66 std::is_same_v<T, ushort> ? "ushort"sv :
67 std::is_same_v<T, int> ? "int"sv :
68 std::is_same_v<T, uint> ? "uint"sv :
69 std::is_same_v<T, slong> ? "slong"sv :
70 std::is_same_v<T, ulong> ? "ulong"sv :
71 std::is_same_v<T, half> ? "half"sv :
72 std::is_same_v<T, float> ? "float"sv :
73 std::is_same_v<T, double> ? "double"sv :
74 "unknown"sv;
75 if constexpr (N == 2u) {
76 return fmt::format_to(
77 ctx.out(),
78 FMT_STRING("{}2({}, {})"),
79 type_name, v.x, v.y);
80 } else if constexpr (N == 3u) {
81 return fmt::format_to(
82 ctx.out(),
83 FMT_STRING("{}3({}, {}, {})"),
84 type_name, v.x, v.y, v.z);
85 } else if constexpr (N == 4u) {
86 return fmt::format_to(
87 ctx.out(),
88 FMT_STRING("{}4({}, {}, {}, {})"),
89 type_name, v.x, v.y, v.z, v.w);
90 } else {
91 static_assert(luisa::always_false_v<T>);
92 }
93 }
