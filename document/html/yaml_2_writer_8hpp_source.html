87 OutputVarType insert_value(const std::string_view& _name,
88 const T& _var) const noexcept {
89 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>() ||
90 std::is_same<std::remove_cvref_t<T>, bool>() ||
91 std::is_same<std::remove_cvref_t<T>,
92 std::remove_cvref_t<decltype(YAML::Null)>>()) {
93 (*out_) << YAML::Key << _name.data() << YAML::Value << _var;
94 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
95 // std::to_string is necessary to ensure that floating point values are
96 // always written as floats.
97 (*out_) << YAML::Key << _name.data() << YAML::Value
98 << std::to_string(_var);
99 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
100 (*out_) << YAML::Key << _name.data() << YAML::Value
101 << static_cast<int64_t>(_var);
102 } else {
103 static_assert(rfl::always_false_v<T>, "Unsupported type.");
104 }
105 return OutputVarType{};
106 }
