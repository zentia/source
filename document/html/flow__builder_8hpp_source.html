1306inline void FlowBuilder::erase(Task task) {
1307
1308 if (!task._node) {
1309 return;
1310 }
1311
1312 // remove task from its successors' predecessor list
1313 for(size_t i=0; i<task._node->_num_successors; ++i) {
1314 task._node->_edges[i]->_remove_predecessors(task._node);
1315 }
1316
1317 // remove task from its precedessors' successor list
1318 for(size_t i=task._node->_num_successors; i<task._node->_edges.size(); ++i) {
1319 task._node->_edges[i]->_remove_successors(task._node);
1320 }
1321
1322 _graph._erase(task._node);
1323}
