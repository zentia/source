220 template <typename Value, typename Hash, typename Predicate, typename Allocator, bool bCacheHashCode, typename UserPredicate>
221 typename eastl::hash_set<Value, Hash, Predicate, Allocator, bCacheHashCode>::size_type erase_if(eastl::hash_set<Value, Hash, Predicate, Allocator, bCacheHashCode>& c, UserPredicate predicate)
222 {
223 auto oldSize = c.size();
224 // Erases all elements that satisfy the predicate pred from the container.
225 for (auto i = c.begin(), last = c.end(); i != last;)
226 {
227 if (predicate(*i))
228 {
229 i = c.erase(i);
230 }
231 else
232 {
233 ++i;
234 }
235 }
236 return oldSize - c.size();
237 }

381 template <typename Value, typename Hash, typename Predicate, typename Allocator, bool bCacheHashCode>
382 inline bool operator==(const hash_set<Value, Hash, Predicate, Allocator, bCacheHashCode>& a,
383 const hash_set<Value, Hash, Predicate, Allocator, bCacheHashCode>& b)
384 {
385 typedef typename hash_set<Value, Hash, Predicate, Allocator, bCacheHashCode>::const_iterator const_iterator;
386
387 // We implement branching with the assumption that the return value is usually false.
388 if(a.size() != b.size())
389 return false;
390
391 // For set (with its unique keys), we need only test that each element in a can be found in b,
392 // as there can be only one such pairing per element. multiset needs to do a something more elaborate.
393 for(const_iterator ai = a.begin(), aiEnd = a.end(), biEnd = b.end(); ai != aiEnd; ++ai)
394 {
395 const_iterator bi = b.find(*ai);
396
397 if((bi == biEnd) || !(*ai == *bi)) // We have to compare values in addition to making sure the lookups succeeded. This is because the lookup is done via the user-supplised Predicate
398 return false; // which isn't strictly required to be identical to the Value operator==, though 99% of the time it will be so.
399 }
400
401 return true;
402 }

413 template <typename Value, typename Hash, typename Predicate, typename Allocator, bool bCacheHashCode>
414 inline bool operator==(const hash_multiset<Value, Hash, Predicate, Allocator, bCacheHashCode>& a,
415 const hash_multiset<Value, Hash, Predicate, Allocator, bCacheHashCode>& b)
416 {
417 typedef typename hash_multiset<Value, Hash, Predicate, Allocator, bCacheHashCode>::const_iterator const_iterator;
418 typedef typename eastl::iterator_traits<const_iterator>::difference_type difference_type;
419
420 // We implement branching with the assumption that the return value is usually false.
421 if(a.size() != b.size())
422 return false;
423
424 // We can't simply search for each element of a in b, as it may be that the bucket for
425 // two elements in a has those same two elements in b but in different order (which should
426 // still result in equality). Also it's possible that one bucket in a has two elements which
427 // both match a solitary element in the equivalent bucket in b (which shouldn't result in equality).
428 eastl::pair<const_iterator, const_iterator> aRange;
429 eastl::pair<const_iterator, const_iterator> bRange;
430
431 for(const_iterator ai = a.begin(), aiEnd = a.end(); ai != aiEnd; ai = aRange.second) // For each element in a...
432 {
433 aRange = a.equal_range(*ai); // Get the range of elements in a that are equal to ai.
434 bRange = b.equal_range(*ai); // Get the range of elements in b that are equal to ai.
435
436 // We need to verify that aRange == bRange. First make sure the range sizes are equivalent...
437 const difference_type aDistance = eastl::distance(aRange.first, aRange.second);
438 const difference_type bDistance = eastl::distance(bRange.first, bRange.second);
439
440 if(aDistance != bDistance)
441 return false;
442
443 // At this point, aDistance > 0 and aDistance == bDistance.
444 // Implement a fast pathway for the case that there's just a single element.
445 if(aDistance == 1)
446 {
447 if(!(*aRange.first == *bRange.first)) // We have to compare values in addition to making sure the distance (element count) was equal. This is because the lookup is done via the user-supplised Predicate
448 return false; // which isn't strictly required to be identical to the Value operator==, though 99% of the time it will be so. Ditto for the is_permutation usage below.
449 }
450 else
451 {
452 // Check to see if these aRange and bRange are any permutation of each other.
453 // This check gets slower as there are more elements in the range.
454 if(!eastl::is_permutation(aRange.first, aRange.second, bRange.first))
455 return false;
456 }
457 }
458
459 return true;
460 }
