2950static xxh_u32 XXH32_round(xxh_u32 acc, xxh_u32 input)
2951{
2952 acc += input * XXH_PRIME32_2;
2953 acc = XXH_rotl32(acc, 13);
2954 acc *= XXH_PRIME32_1;
2955#if (defined(__SSE4_1__) || defined(__aarch64__) || defined(__wasm_simd128__)) && !defined(XXH_ENABLE_AUTOVECTORIZE)
2956 /*
2957 * UGLY HACK:
2958 * A compiler fence is used to prevent GCC and Clang from
2959 * autovectorizing the XXH32 loop (pragmas and attributes don't work for some
2960 * reason) without globally disabling SSE4.1.
2961 *
2962 * The reason we want to avoid vectorization is because despite working on
2963 * 4 integers at a time, there are multiple factors slowing XXH32 down on
2964 * SSE4:
2965 * - There's a ridiculous amount of lag from pmulld (10 cycles of latency on
2966 * newer chips!) making it slightly slower to multiply four integers at
2967 * once compared to four integers independently. Even when pmulld was
2968 * fastest, Sandy/Ivy Bridge, it is still not worth it to go into SSE
2969 * just to multiply unless doing a long operation.
2970 *
2971 * - Four instructions are required to rotate,
2972 * movqda tmp, v // not required with VEX encoding
2973 * pslld tmp, 13 // tmp <<= 13
2974 * psrld v, 19 // x >>= 19
2975 * por v, tmp // x |= tmp
2976 * compared to one for scalar:
2977 * roll v, 13 // reliably fast across the board
2978 * shldl v, v, 13 // Sandy Bridge and later prefer this for some reason
2979 *
2980 * - Instruction level parallelism is actually more beneficial here because
2981 * the SIMD actually serializes this operation: While v1 is rotating, v2
2982 * can load data, while v3 can multiply. SSE forces them to operate
2983 * together.
2984 *
2985 * This is also enabled on AArch64, as Clang is *very aggressive* in vectorizing
2986 * the loop. NEON is only faster on the A53, and with the newer cores, it is less
2987 * than half the speed.
2988 *
2989 * Additionally, this is used on WASM SIMD128 because it JITs to the same
2990 * SIMD instructions and has the same issue.
2991 */
2992 XXH_COMPILER_GUARD(acc);
2993#endif
2994 return acc;
2995}

3089static XXH_PUREF xxh_u32
3090XXH32_finalize(xxh_u32 hash, const xxh_u8* ptr, size_t len, XXH_alignment align)
3091{
3092#define XXH_PROCESS1 do { \
3093 hash += (*ptr++) * XXH_PRIME32_5; \
3094 hash = XXH_rotl32(hash, 11) * XXH_PRIME32_1; \
3095} while (0)
3096
3097#define XXH_PROCESS4 do { \
3098 hash += XXH_get32bits(ptr) * XXH_PRIME32_3; \
3099 ptr += 4; \
3100 hash = XXH_rotl32(hash, 17) * XXH_PRIME32_4; \
3101} while (0)
3102
3103 if (ptr==NULL) XXH_ASSERT(len == 0);
3104
3105 /* Compact rerolled version; generally faster */
3106 if (!XXH32_ENDJMP) {
3107 len &= 15;
3108 while (len >= 4) {
3109 XXH_PROCESS4;
3110 len -= 4;
3111 }
3112 while (len > 0) {
3113 XXH_PROCESS1;
3114 --len;
3115 }
3116 return XXH32_avalanche(hash);
3117 } else {
3118 switch(len&15) /* or switch(bEnd - p) */ {
3119 case 12: XXH_PROCESS4;
3120 XXH_FALLTHROUGH; /* fallthrough */
3121 case 8: XXH_PROCESS4;
3122 XXH_FALLTHROUGH; /* fallthrough */
3123 case 4: XXH_PROCESS4;
3124 return XXH32_avalanche(hash);
3125
3126 case 13: XXH_PROCESS4;
3127 XXH_FALLTHROUGH; /* fallthrough */
3128 case 9: XXH_PROCESS4;
3129 XXH_FALLTHROUGH; /* fallthrough */
3130 case 5: XXH_PROCESS4;
3131 XXH_PROCESS1;
3132 return XXH32_avalanche(hash);
3133
3134 case 14: XXH_PROCESS4;
3135 XXH_FALLTHROUGH; /* fallthrough */
3136 case 10: XXH_PROCESS4;
3137 XXH_FALLTHROUGH; /* fallthrough */
3138 case 6: XXH_PROCESS4;
3139 XXH_PROCESS1;
3140 XXH_PROCESS1;
3141 return XXH32_avalanche(hash);
3142
3143 case 15: XXH_PROCESS4;
3144 XXH_FALLTHROUGH; /* fallthrough */
3145 case 11: XXH_PROCESS4;
3146 XXH_FALLTHROUGH; /* fallthrough */
3147 case 7: XXH_PROCESS4;
3148 XXH_FALLTHROUGH; /* fallthrough */
3149 case 3: XXH_PROCESS1;
3150 XXH_FALLTHROUGH; /* fallthrough */
3151 case 2: XXH_PROCESS1;
3152 XXH_FALLTHROUGH; /* fallthrough */
3153 case 1: XXH_PROCESS1;
3154 XXH_FALLTHROUGH; /* fallthrough */
3155 case 0: return XXH32_avalanche(hash);
3156 }
3157 XXH_ASSERT(0);
3158 return hash; /* reaching this point is deemed impossible */
3159 }

3253XXH_PUBLIC_API XXH_errorcode
3254XXH32_update(XXH32_state_t* state, const void* input, size_t len)
3255{
3256 if (input==NULL) {
3257 XXH_ASSERT(len == 0);
3258 return XXH_OK;
3259 }
3260
3261 state->total_len_32 += (XXH32_hash_t)len;
3262 state->large_len |= (XXH32_hash_t)((len>=16) | (state->total_len_32>=16));
3263
3264 XXH_ASSERT(state->bufferedSize < sizeof(state->buffer));
3265 if (len < sizeof(state->buffer) - state->bufferedSize) { /* fill in tmp buffer */
3266 XXH_memcpy(state->buffer + state->bufferedSize, input, len);
3267 state->bufferedSize += (XXH32_hash_t)len;
3268 return XXH_OK;
3269 }
3270
3271 { const xxh_u8* xinput = (const xxh_u8*)input;
3272 const xxh_u8* const bEnd = xinput + len;
3273
3274 if (state->bufferedSize) { /* non-empty buffer: complete first */
3275 XXH_memcpy(state->buffer + state->bufferedSize, xinput, sizeof(state->buffer) - state->bufferedSize);
3276 xinput += sizeof(state->buffer) - state->bufferedSize;
3277 /* then process one round */
3278 (void)XXH32_consumeLong(state->acc, state->buffer, sizeof(state->buffer), XXH_aligned);
3279 state->bufferedSize = 0;
3280 }
3281
3282 XXH_ASSERT(xinput <= bEnd);
3283 if ((size_t)(bEnd - xinput) >= sizeof(state->buffer)) {
3284 /* Process the remaining data */
3285 xinput = XXH32_consumeLong(state->acc, xinput, (size_t)(bEnd - xinput), XXH_unaligned);
3286 }
3287
3288 if (xinput < bEnd) {
3289 /* Copy the leftover to the tmp buffer */
3290 XXH_memcpy(state->buffer, xinput, (size_t)(bEnd-xinput));
3291 state->bufferedSize = (unsigned)(bEnd-xinput);
3292 }
3293 }

3295 return XXH_OK;

3558XXH_FORCE_INLINE const xxh_u8 *
3559XXH64_consumeLong(
3560 xxh_u64 *XXH_RESTRICT acc,
3561 xxh_u8 const *XXH_RESTRICT input,
3562 size_t len,
3563 XXH_alignment align
3564)
3565{

3567 const xxh_u8* const limit = bEnd - 31;

3569 XXH_ASSERT(input != NULL);
3570 XXH_ASSERT(len >= 32);

3574 size_t i;
3575 for (i = 0; i < 4; i++) {

3632XXH_STATIC XXH_PUREF xxh_u64
3633XXH64_finalize(xxh_u64 hash, const xxh_u8* ptr, size_t len, XXH_alignment align)
3634{
3635 if (ptr==NULL) XXH_ASSERT(len == 0);
3636 len &= 31;
3637 while (len >= 8) {
3638 xxh_u64 const k1 = XXH64_round(0, XXH_get64bits(ptr));
3639 ptr += 8;
3640 hash ^= k1;
3641 hash = XXH_rotl64(hash,27) * XXH_PRIME64_1 + XXH_PRIME64_4;
3642 len -= 8;
3643 }
3644 if (len >= 4) {
3645 hash ^= (xxh_u64)(XXH_get32bits(ptr)) * XXH_PRIME64_1;
3646 ptr += 4;
3647 hash = XXH_rotl64(hash, 23) * XXH_PRIME64_2 + XXH_PRIME64_3;
3648 len -= 4;
3649 }
3650 while (len > 0) {
3651 hash ^= (*ptr++) * XXH_PRIME64_5;
3652 hash = XXH_rotl64(hash, 11) * XXH_PRIME64_1;
3653 --len;
3654 }
3655 return XXH64_avalanche(hash);
3656}

3751{

3753 XXH_ASSERT(len == 0);
3754 return XXH_OK;
3755 }
3756
3757 state->total_len += len;
3758
3759 XXH_ASSERT(state->bufferedSize <= sizeof(state->buffer));
3760 if (len < sizeof(state->buffer) - state->bufferedSize) { /* fill in tmp buffer */
3761 XXH_memcpy(state->buffer + state->bufferedSize, input, len);
3762 state->bufferedSize += (XXH32_hash_t)len;
3763 return XXH_OK;
3764 }
3765
3766 { const xxh_u8* xinput = (const xxh_u8*)input;
3767 const xxh_u8* const bEnd = xinput + len;
3768

3772 /* and process one round */
3773 (void)XXH64_consumeLong(state->acc, state->buffer, sizeof(state->buffer), XXH_aligned);
3774 state->bufferedSize = 0;

3776
3777 XXH_ASSERT(xinput <= bEnd);
3778 if ((size_t)(bEnd - xinput) >= sizeof(state->buffer)) {
3779 /* Process the remaining data */
3780 xinput = XXH64_consumeLong(state->acc, xinput, (size_t)(bEnd - xinput), XXH_unaligned);
3781 }
3782
3783 if (xinput < bEnd) {
3784 /* Copy the leftover to the tmp buffer */
3785 XXH_memcpy(state->buffer, xinput, (size_t)(bEnd-xinput));
3786 state->bufferedSize = (unsigned)(bEnd-xinput);
3787 }
3788 }
3789
3790 return XXH_OK;
3791}

3892/*
3893 * One goal of XXH3 is to make it fast on both 32-bit and 64-bit, while
3894 * remaining a true 64-bit/128-bit hash function.
3895 *
3896 * This is done by prioritizing a subset of 64-bit operations that can be
3897 * emulated without too many steps on the average 32-bit machine.
3898 *
3899 * For example, these two lines seem similar, and run equally fast on 64-bit:
3900 *
3901 * xxh_u64 x;
3902 * x ^= (x >> 47); // good
3903 * x ^= (x >> 13); // bad
3904 *
3905 * However, to a 32-bit machine, there is a major difference.
3906 *
3907 * x ^= (x >> 47) looks like this:
3908 *
3909 * x.lo ^= (x.hi >> (47 - 32));
3910 *
3911 * while x ^= (x >> 13) looks like this:
3912 *
3913 * // note: funnel shifts are not usually cheap.
3914 * x.lo ^= (x.lo >> 13) | (x.hi << (32 - 13));
3915 * x.hi ^= (x.hi >> 13);
3916 *
3917 * The first one is significantly faster than the second, simply because the
3918 * shift is larger than 32. This means:
3919 * - All the bits we need are in the upper 32 bits, so we can ignore the lower
3920 * 32 bits in the shift.
3921 * - The shift result will always fit in the lower 32 bits, and therefore,
3922 * we can ignore the upper 32 bits in the xor.
3923 *
3924 * Thanks to this optimization, XXH3 only requires these features to be efficient:
3925 *
3926 * - Usable unaligned access
3927 * - A 32-bit or 64-bit ALU
3928 * - If 32-bit, a decent ADC instruction
3929 * - A 32 or 64-bit multiply with a 64-bit result
3930 * - For the 128-bit variant, a decent byteswap helps short inputs.
3931 *
3932 * The first two are already required by XXH32, and almost all 32-bit and 64-bit
3933 * platforms which can run XXH32 can run XXH3 efficiently.
3934 *
3935 * Thumb-1, the classic 16-bit only subset of ARM's instruction set, is one
3936 * notable exception.
3937 *
3938 * First of all, Thumb-1 lacks support for the UMULL instruction which
3939 * performs the important long multiply. This means numerous __aeabi_lmul
3940 * calls.
3941 *
3942 * Second of all, the 8 functional registers are just not enough.
3943 * Setup for __aeabi_lmul, byteshift loads, pointers, and all arithmetic need
3944 * Lo registers, and this shuffling results in thousands more MOVs than A32.
3945 *
3946 * A32 and T32 don't have this limitation. They can access all 14 registers,
3947 * do a 32->64 multiply with UMULL, and the flexible operand allowing free
3948 * shifts is helpful, too.
3949 *
3950 * Therefore, we do a quick sanity check.
3951 *
3952 * If compiling Thumb-1 for a target which supports ARM instructions, we will
3953 * emit a warning, as it is not a "sane" platform to compile for.
3954 *
3955 * Usually, if this happens, it is because of an accident and you probably need
3956 * to specify -march, as you likely meant to compile for a newer architecture.
3957 *
3958 * Credit: large sections of the vectorial and asm source code paths
3959 * have been contributed by @easyaspi314
3960 */
3961#if defined(__thumb__) && !defined(__thumb2__) && defined(__ARM_ARCH_ISA_ARM)
3962# warning "XXH3 is highly inefficient without ARM or Thumb-2."
3963#endif

4463static XXH128_hash_t
4464XXH_mult64to128(xxh_u64 lhs, xxh_u64 rhs)
4465{
4466 /*
4467 * GCC/Clang __uint128_t method.
4468 *
4469 * On most 64-bit targets, GCC and Clang define a __uint128_t type.
4470 * This is usually the best way as it usually uses a native long 64-bit
4471 * multiply, such as MULQ on x86_64 or MUL + UMULH on aarch64.
4472 *
4473 * Usually.
4474 *
4475 * Despite being a 32-bit platform, Clang (and emscripten) define this type
4476 * despite not having the arithmetic for it. This results in a laggy
4477 * compiler builtin call which calculates a full 128-bit multiply.
4478 * In that case it is best to use the portable one.
4479 * https://github.com/Cyan4973/xxHash/issues/211#issuecomment-515575677
4480 */
4481#if (defined(__GNUC__) || defined(__clang__)) && !defined(__wasm__) \
4482 && defined(__SIZEOF_INT128__) \
4483 || (defined(_INTEGRAL_MAX_BITS) && _INTEGRAL_MAX_BITS >= 128)
4484
4485 __uint128_t const product = (__uint128_t)lhs * (__uint128_t)rhs;
4486 XXH128_hash_t r128;
4487 r128.low64 = (xxh_u64)(product);
4488 r128.high64 = (xxh_u64)(product >> 64);
4489 return r128;
4490
4491 /*
4492 * MSVC for x64's _umul128 method.
4493 *
4494 * xxh_u64 _umul128(xxh_u64 Multiplier, xxh_u64 Multiplicand, xxh_u64 *HighProduct);
4495 *
4496 * This compiles to single operand MUL on x64.
4497 */
4498#elif (defined(_M_X64) || defined(_M_IA64)) && !defined(_M_ARM64EC)
4499
4500#ifndef _MSC_VER
4501# pragma intrinsic(_umul128)
4502#endif
4503 xxh_u64 product_high;
4504 xxh_u64 const product_low = _umul128(lhs, rhs, &product_high);
4505 XXH128_hash_t r128;
4506 r128.low64 = product_low;
4507 r128.high64 = product_high;
4508 return r128;
4509
4510 /*
4511 * MSVC for ARM64's __umulh method.
4512 *
4513 * This compiles to the same MUL + UMULH as GCC/Clang's __uint128_t method.
4514 */
4515#elif defined(_M_ARM64) || defined(_M_ARM64EC)
4516
4517#ifndef _MSC_VER
4518# pragma intrinsic(__umulh)
4519#endif
4520 XXH128_hash_t r128;
4521 r128.low64 = lhs * rhs;
4522 r128.high64 = __umulh(lhs, rhs);
4523 return r128;
4524
4525#else
4526 /*
4527 * Portable scalar method. Optimized for 32-bit and 64-bit ALUs.
4528 *
4529 * This is a fast and simple grade school multiply, which is shown below
4530 * with base 10 arithmetic instead of base 0x100000000.
4531 *
4532 * 9 3 // D2 lhs = 93
4533 * x 7 5 // D2 rhs = 75
4534 * ----------
4535 * 1 5 // D2 lo_lo = (93 % 10) * (75 % 10) = 15
4536 * 4 5 | // D2 hi_lo = (93 / 10) * (75 % 10) = 45
4537 * 2 1 | // D2 lo_hi = (93 % 10) * (75 / 10) = 21
4538 * + 6 3 | | // D2 hi_hi = (93 / 10) * (75 / 10) = 63
4539 * ---------
4540 * 2 7 | // D2 cross = (15 / 10) + (45 % 10) + 21 = 27
4541 * + 6 7 | | // D2 upper = (27 / 10) + (45 / 10) + 63 = 67
4542 * ---------
4543 * 6 9 7 5 // D4 res = (27 * 10) + (15 % 10) + (67 * 100) = 6975
4544 *
4545 * The reasons for adding the products like this are:
4546 * 1. It avoids manual carry tracking. Just like how
4547 * (9 * 9) + 9 + 9 = 99, the same applies with this for UINT64_MAX.
4548 * This avoids a lot of complexity.
4549 *
4550 * 2. It hints for, and on Clang, compiles to, the powerful UMAAL
4551 * instruction available in ARM's Digital Signal Processing extension
4552 * in 32-bit ARMv6 and later, which is shown below:
4553 *
4554 * void UMAAL(xxh_u32 *RdLo, xxh_u32 *RdHi, xxh_u32 Rn, xxh_u32 Rm)
4555 * {
4556 * xxh_u64 product = (xxh_u64)*RdLo * (xxh_u64)*RdHi + Rn + Rm;
4557 * *RdLo = (xxh_u32)(product & 0xFFFFFFFF);
4558 * *RdHi = (xxh_u32)(product >> 32);
4559 * }
4560 *
4561 * This instruction was designed for efficient long multiplication, and
4562 * allows this to be calculated in only 4 instructions at speeds
4563 * comparable to some 64-bit ALUs.
4564 *
4565 * 3. It isn't terrible on other platforms. Usually this will be a couple
4566 * of 32-bit ADD/ADCs.
4567 */
4568
4569 /* First calculate all of the cross products. */
4570 xxh_u64 const lo_lo = XXH_mult32to64(lhs & 0xFFFFFFFF, rhs & 0xFFFFFFFF);
4571 xxh_u64 const hi_lo = XXH_mult32to64(lhs >> 32, rhs & 0xFFFFFFFF);
4572 xxh_u64 const lo_hi = XXH_mult32to64(lhs & 0xFFFFFFFF, rhs >> 32);
4573 xxh_u64 const hi_hi = XXH_mult32to64(lhs >> 32, rhs >> 32);
4574
4575 /* Now add the products together. These will never overflow. */
4576 xxh_u64 const cross = (lo_lo >> 32) + (hi_lo & 0xFFFFFFFF) + lo_hi;
4577 xxh_u64 const upper = (hi_lo >> 32) + (cross >> 32) + hi_hi;
4578 xxh_u64 const lower = (cross << 32) | (lo_lo & 0xFFFFFFFF);
4579
4580 XXH128_hash_t r128;
4581 r128.low64 = lower;
4582 r128.high64 = upper;
4583 return r128;
4584#endif
4585}

4639/* ==========================================
4640 * Short keys
4641 * ==========================================
4642 * One of the shortcomings of XXH32 and XXH64 was that their performance was
4643 * sub-optimal on short lengths. It used an iterative algorithm which strongly
4644 * favored lengths that were a multiple of 4 or 8.
4645 *
4646 * Instead of iterating over individual inputs, we use a set of single shot
4647 * functions which piece together a range of lengths and operate in constant time.
4648 *
4649 * Additionally, the number of multiplies has been significantly reduced. This
4650 * reduces latency, especially when emulating 64-bit multiplies on 32-bit.
4651 *
4652 * Depending on the platform, this may or may not be faster than XXH32, but it
4653 * is almost guaranteed to be faster than XXH64.
4654 */
4655
4656/*
4657 * At very short lengths, there isn't enough input to fully hide secrets, or use
4658 * the entire secret.
4659 *
4660 * There is also only a limited amount of mixing we can do before significantly
4661 * impacting performance.
4662 *
4663 * Therefore, we use different sections of the secret and always mix two secret
4664 * samples with an XOR. This should have no effect on performance on the
4665 * seedless or withSeed variants because everything _should_ be constant folded
4666 * by modern compilers.
4667 *
4668 * The XOR mixing hides individual parts of the secret and increases entropy.
4669 *
4670 * This adds an extra layer of strength for custom secrets.
4671 */

4680 * len = 2: combined = { input[1], 0x02, input[0], input[1] }
4681 * len = 3: combined = { input[2], 0x03, input[0], input[1] }

4687 | ((xxh_u32)c3 << 0) | ((xxh_u32)len << 8);

4690 return XXH64_avalanche(keyed);

4738/*
4739 * DISCLAIMER: There are known *seed-dependent* multicollisions here due to
4740 * multiplication by zero, affecting hashes of lengths 17 to 240.
4741 *
4742 * However, they are very unlikely.
4743 *
4744 * Keep this in mind when using the unseeded XXH3_64bits() variant: As with all
4745 * unseeded non-cryptographic hashes, it does not attempt to defend itself
4746 * against specially crafted inputs, only random inputs.
4747 *
4748 * Compared to classic UMAC where a 1 in 2^31 chance of 4 consecutive bytes
4749 * cancelling out the secret is taken an arbitrary number of times (addressed
4750 * in XXH3_accumulate_512), this collision is very unlikely with random inputs
4751 * and/or proper seeding:
4752 *
4753 * This only has a 1 in 2^63 chance of 8 consecutive bytes cancelling out, in a
4754 * function that is only called up to 16 times per hash with up to 240 bytes of
4755 * input.
4756 *
4757 * This is not too bad for a non-cryptographic hash function, especially with
4758 * only 64 bit outputs.
4759 *
4760 * The 128-bit variant (which trades some speed for strength) is NOT affected
4761 * by this, although it is always a good idea to use a proper seed if you care
4762 * about strength.
4763 */
4764XXH_FORCE_INLINE xxh_u64 XXH3_mix16B(const xxh_u8* XXH_RESTRICT input,
4765 const xxh_u8* XXH_RESTRICT secret, xxh_u64 seed64)
4766{
4767#if defined(__GNUC__) && !defined(__clang__) /* GCC, not Clang */ \
4768 && defined(__i386__) && defined(__SSE2__) /* x86 + SSE2 */ \
4769 && !defined(XXH_ENABLE_AUTOVECTORIZE) /* Define to disable like XXH32 hack */
4770 /*
4771 * UGLY HACK:
4772 * GCC for x86 tends to autovectorize the 128-bit multiply, resulting in
4773 * slower code.
4774 *
4775 * By forcing seed64 into a register, we disrupt the cost model and
4776 * cause it to scalarize. See `XXH32_round()`
4777 *
4778 * FIXME: Clang's output is still _much_ faster -- On an AMD Ryzen 3600,
4779 * XXH3_64bits @ len=240 runs at 4.6 GB/s with Clang 9, but 3.3 GB/s on
4780 * GCC 9.2, despite both emitting scalar code.
4781 *
4782 * GCC generates much better scalar code than Clang for the rest of XXH3,
4783 * which is why finding a more optimal codepath is an interest.
4784 */
4785 XXH_COMPILER_GUARD(seed64);
4786#endif
4787 { xxh_u64 const input_lo = XXH_readLE64(input);
4788 xxh_u64 const input_hi = XXH_readLE64(input+8);

4966/*
4967 * XXH3_accumulate_512 is the tightest loop for long inputs, and it is the most optimized.
4968 *
4969 * It is a hardened version of UMAC, based off of FARSH's implementation.
4970 *
4971 * This was chosen because it adapts quite well to 32-bit, 64-bit, and SIMD
4972 * implementations, and it is ridiculously fast.
4973 *
4974 * We harden it by mixing the original input to the accumulators as well as the product.
4975 *
4976 * This means that in the (relatively likely) case of a multiply by zero, the
4977 * original input is preserved.
4978 *
4979 * On 128-bit inputs, we swap 64-bit pairs when we add the input to improve
4980 * cross-pollination, as otherwise the upper and lower halves would be
4981 * essentially independent.
4982 *
4983 * This doesn't matter on 64-bit hashes since they all get merged together in
4984 * the end, so we skip the extra step.
4985 *
4986 * Both XXH3_64bits and XXH3_128bits use this subroutine.
4987 */

4996XXH_FORCE_INLINE XXH_TARGET_AVX512 void

4998 const void* XXH_RESTRICT input,
4999 const void* XXH_RESTRICT secret)
5000{

5002 XXH_ASSERT((((size_t)acc) & 63) == 0);
5003 XXH_STATIC_ASSERT(XXH_STRIPE_LEN == sizeof(__m512i));
5004
5005 {
5006 /* data_vec = input[0]; */
5007 __m512i const data_vec = _mm512_loadu_si512 (input);
5008 /* key_vec = secret[0]; */
5009 __m512i const key_vec = _mm512_loadu_si512 (secret);
5010 /* data_key = data_vec ^ key_vec; */
5011 __m512i const data_key = _mm512_xor_si512 (data_vec, key_vec);
5012 /* data_key_lo = data_key >> 32; */
5013 __m512i const data_key_lo = _mm512_srli_epi64 (data_key, 32);
5014 /* product = (data_key & 0xffffffff) * (data_key_lo & 0xffffffff); */
5015 __m512i const product = _mm512_mul_epu32 (data_key, data_key_lo);
5016 /* xacc[0] += swap(data_vec); */
5017 __m512i const data_swap = _mm512_shuffle_epi32(data_vec, (_MM_PERM_ENUM)_MM_SHUFFLE(1, 0, 3, 2));
5018 __m512i const sum = _mm512_add_epi64(*xacc, data_swap);
5019 /* xacc[0] += product; */

5021 }

5025/*
5026 * XXH3_scrambleAcc: Scrambles the accumulators to improve mixing.
5027 *
5028 * Multiplication isn't perfect, as explained by Google in HighwayHash:
5029 *
5030 * // Multiplication mixes/scrambles bytes 0-7 of the 64-bit result to
5031 * // varying degrees. In descending order of goodness, bytes
5032 * // 3 4 2 5 1 6 0 7 have quality 228 224 164 160 100 96 36 32.
5033 * // As expected, the upper and lower bytes are much worse.
5034 *
5035 * Source: https://github.com/google/highwayhash/blob/0aaf66b/highwayhash/hh_avx2.h#L291
5036 *
5037 * Since our algorithm uses a pseudorandom secret to add some variance into the
5038 * mix, we don't need to (or want to) mix as often or as much as HighwayHash does.
5039 *
5040 * This isn't as tight as XXH3_accumulate, but still written in SIMD to avoid
5041 * extraction.
5042 *
5043 * Both XXH3_64bits and XXH3_128bits use this subroutine.
5044 */

5048{

5050 XXH_STATIC_ASSERT(XXH_STRIPE_LEN == sizeof(__m512i));
5051 { __m512i* const xacc = (__m512i*) acc;
5052 const __m512i prime32 = _mm512_set1_epi32((int)XXH_PRIME32_1);
5053
5054 /* xacc[0] ^= (xacc[0] >> 47) */
5055 __m512i const acc_vec = *xacc;
5056 __m512i const shifted = _mm512_srli_epi64 (acc_vec, 47);
5057 /* xacc[0] ^= secret; */
5058 __m512i const key_vec = _mm512_loadu_si512 (secret);
5059 __m512i const data_key = _mm512_ternarylogic_epi32(key_vec, acc_vec, shifted, 0x96 /* key_vec ^ acc_vec ^ shifted */);
5060
5061 /* xacc[0] *= XXH_PRIME32_1; */
5062 __m512i const data_key_hi = _mm512_srli_epi64 (data_key, 32);
5063 __m512i const prod_lo = _mm512_mul_epu32 (data_key, prime32);
5064 __m512i const prod_hi = _mm512_mul_epu32 (data_key_hi, prime32);
5065 *xacc = _mm512_add_epi64(prod_lo, _mm512_slli_epi64(prod_hi, 32));
5066 }

5341XXH3_accumulate_512_neon( void* XXH_RESTRICT acc,
5342 const void* XXH_RESTRICT input,
5343 const void* XXH_RESTRICT secret)
5344{
5345 XXH_ASSERT((((size_t)acc) & 15) == 0);
5346 XXH_STATIC_ASSERT(XXH3_NEON_LANES > 0 && XXH3_NEON_LANES <= XXH_ACC_NB && XXH3_NEON_LANES % 2 == 0);
5347 { /* GCC for darwin arm64 does not like aliasing here */
5348 xxh_aliasing_uint64x2_t* const xacc = (xxh_aliasing_uint64x2_t*) acc;
5349 /* We don't use a uint32x4_t pointer because it causes bus errors on ARMv7. */
5350 uint8_t const* xinput = (const uint8_t *) input;
5351 uint8_t const* xsecret = (const uint8_t *) secret;
5352
5353 size_t i;
5354#ifdef __wasm_simd128__
5355 /*
5356 * On WASM SIMD128, Clang emits direct address loads when XXH3_kSecret
5357 * is constant propagated, which results in it converting it to this
5358 * inside the loop:
5359 *
5360 * a = v128.load(XXH3_kSecret + 0 + $secret_offset, offset = 0)
5361 * b = v128.load(XXH3_kSecret + 16 + $secret_offset, offset = 0)
5362 * ...
5363 *
5364 * This requires a full 32-bit address immediate (and therefore a 6 byte
5365 * instruction) as well as an add for each offset.
5366 *
5367 * Putting an asm guard prevents it from folding (at the cost of losing
5368 * the alignment hint), and uses the free offset in `v128.load` instead
5369 * of adding secret_offset each time which overall reduces code size by
5370 * about a kilobyte and improves performance.
5371 */
5372 XXH_COMPILER_GUARD(xsecret);
5373#endif
5374 /* Scalar lanes use the normal scalarRound routine */
5375 for (i = XXH3_NEON_LANES; i < XXH_ACC_NB; i++) {
5376 XXH3_scalarRound(acc, input, secret, i);
5377 }
5378 i = 0;
5379 /* 4 NEON lanes at a time. */
5380 for (; i+1 < XXH3_NEON_LANES / 2; i+=2) {
5381 /* data_vec = xinput[i]; */
5382 uint64x2_t data_vec_1 = XXH_vld1q_u64(xinput + (i * 16));
5383 uint64x2_t data_vec_2 = XXH_vld1q_u64(xinput + ((i+1) * 16));
5384 /* key_vec = xsecret[i]; */
5385 uint64x2_t key_vec_1 = XXH_vld1q_u64(xsecret + (i * 16));
5386 uint64x2_t key_vec_2 = XXH_vld1q_u64(xsecret + ((i+1) * 16));
5387 /* data_swap = swap(data_vec) */
5388 uint64x2_t data_swap_1 = vextq_u64(data_vec_1, data_vec_1, 1);
5389 uint64x2_t data_swap_2 = vextq_u64(data_vec_2, data_vec_2, 1);
5390 /* data_key = data_vec ^ key_vec; */
5391 uint64x2_t data_key_1 = veorq_u64(data_vec_1, key_vec_1);
5392 uint64x2_t data_key_2 = veorq_u64(data_vec_2, key_vec_2);
5393
5394 /*
5395 * If we reinterpret the 64x2 vectors as 32x4 vectors, we can use a
5396 * de-interleave operation for 4 lanes in 1 step with `vuzpq_u32` to
5397 * get one vector with the low 32 bits of each lane, and one vector
5398 * with the high 32 bits of each lane.
5399 *
5400 * The intrinsic returns a double vector because the original ARMv7-a
5401 * instruction modified both arguments in place. AArch64 and SIMD128 emit
5402 * two instructions from this intrinsic.
5403 *
5404 * [ dk11L | dk11H | dk12L | dk12H ] -> [ dk11L | dk12L | dk21L | dk22L ]
5405 * [ dk21L | dk21H | dk22L | dk22H ] -> [ dk11H | dk12H | dk21H | dk22H ]
5406 */
5407 uint32x4x2_t unzipped = vuzpq_u32(
5408 vreinterpretq_u32_u64(data_key_1),
5409 vreinterpretq_u32_u64(data_key_2)
5410 );
5411 /* data_key_lo = data_key & 0xFFFFFFFF */
5412 uint32x4_t data_key_lo = unzipped.val[0];
5413 /* data_key_hi = data_key >> 32 */
5414 uint32x4_t data_key_hi = unzipped.val[1];
5415 /*
5416 * Then, we can split the vectors horizontally and multiply which, as for most
5417 * widening intrinsics, have a variant that works on both high half vectors
5418 * for free on AArch64. A similar instruction is available on SIMD128.
5419 *
5420 * sum = data_swap + (u64x2) data_key_lo * (u64x2) data_key_hi
5421 */
5422 uint64x2_t sum_1 = XXH_vmlal_low_u32(data_swap_1, data_key_lo, data_key_hi);
5423 uint64x2_t sum_2 = XXH_vmlal_high_u32(data_swap_2, data_key_lo, data_key_hi);
5424 /*
5425 * Clang reorders
5426 * a += b * c; // umlal swap.2d, dkl.2s, dkh.2s
5427 * c += a; // add acc.2d, acc.2d, swap.2d
5428 * to
5429 * c += a; // add acc.2d, acc.2d, swap.2d
5430 * c += b * c; // umlal acc.2d, dkl.2s, dkh.2s
5431 *
5432 * While it would make sense in theory since the addition is faster,
5433 * for reasons likely related to umlal being limited to certain NEON
5434 * pipelines, this is worse. A compiler guard fixes this.
5435 */
5436 XXH_COMPILER_GUARD_CLANG_NEON(sum_1);
5437 XXH_COMPILER_GUARD_CLANG_NEON(sum_2);
5438 /* xacc[i] = acc_vec + sum; */
5439 xacc[i] = vaddq_u64(xacc[i], sum_1);
5440 xacc[i+1] = vaddq_u64(xacc[i+1], sum_2);
5441 }
5442 /* Operate on the remaining NEON lanes 2 at a time. */
5443 for (; i < XXH3_NEON_LANES / 2; i++) {
5444 /* data_vec = xinput[i]; */
5445 uint64x2_t data_vec = XXH_vld1q_u64(xinput + (i * 16));
5446 /* key_vec = xsecret[i]; */
5447 uint64x2_t key_vec = XXH_vld1q_u64(xsecret + (i * 16));
5448 /* acc_vec_2 = swap(data_vec) */
5449 uint64x2_t data_swap = vextq_u64(data_vec, data_vec, 1);
5450 /* data_key = data_vec ^ key_vec; */
5451 uint64x2_t data_key = veorq_u64(data_vec, key_vec);
5452 /* For two lanes, just use VMOVN and VSHRN. */
5453 /* data_key_lo = data_key & 0xFFFFFFFF; */
5454 uint32x2_t data_key_lo = vmovn_u64(data_key);
5455 /* data_key_hi = data_key >> 32; */
5456 uint32x2_t data_key_hi = vshrn_n_u64(data_key, 32);
5457 /* sum = data_swap + (u64x2) data_key_lo * (u64x2) data_key_hi; */
5458 uint64x2_t sum = vmlal_u32(data_swap, data_key_lo, data_key_hi);
5459 /* Same Clang workaround as before */
5460 XXH_COMPILER_GUARD_CLANG_NEON(sum);
5461 /* xacc[i] = acc_vec + sum; */
5462 xacc[i] = vaddq_u64 (xacc[i], sum);
5463 }
5464 }

5949XXH_FORCE_INLINE void
5950XXH3_initCustomSecret_scalar(void* XXH_RESTRICT customSecret, xxh_u64 seed64)
5951{
5952 /*
5953 * We need a separate pointer for the hack below,
5954 * which requires a non-const pointer.
5955 * Any decent compiler will optimize this out otherwise.
5956 */
5957 const xxh_u8* kSecretPtr = XXH3_kSecret;
5958 XXH_STATIC_ASSERT((XXH_SECRET_DEFAULT_SIZE & 15) == 0);
5959
5960#if defined(__GNUC__) && defined(__aarch64__)
5961 /*
5962 * UGLY HACK:
5963 * GCC and Clang generate a bunch of MOV/MOVK pairs for aarch64, and they are
5964 * placed sequentially, in order, at the top of the unrolled loop.
5965 *
5966 * While MOVK is great for generating constants (2 cycles for a 64-bit
5967 * constant compared to 4 cycles for LDR), it fights for bandwidth with
5968 * the arithmetic instructions.
5969 *
5970 * I L S
5971 * MOVK
5972 * MOVK
5973 * MOVK
5974 * MOVK
5975 * ADD
5976 * SUB STR
5977 * STR
5978 * By forcing loads from memory (as the asm line causes the compiler to assume
5979 * that XXH3_kSecretPtr has been changed), the pipelines are used more
5980 * efficiently:
5981 * I L S
5982 * LDR
5983 * ADD LDR
5984 * SUB STR
5985 * STR
5986 *
5987 * See XXH3_NEON_LANES for details on the pipsline.
5988 *
5989 * XXH3_64bits_withSeed, len == 256, Snapdragon 835
5990 * without hack: 2654.4 MB/s
5991 * with hack: 3202.9 MB/s
5992 */
5993 XXH_COMPILER_GUARD(kSecretPtr);
5994#endif
5995 { int const nbRounds = XXH_SECRET_DEFAULT_SIZE / 16;
5996 int i;
5997 for (i=0; i < nbRounds; i++) {
5998 /*
5999 * The asm hack causes the compiler to assume that kSecretPtr aliases with
6000 * customSecret, and on aarch64, this prevented LDP from merging two
6001 * loads together for free. Putting the loads together before the stores
6002 * properly generates LDP.
6003 */
6004 xxh_u64 lo = XXH_readLE64(kSecretPtr + 16*i) + seed64;
6005 xxh_u64 hi = XXH_readLE64(kSecretPtr + 16*i + 8) - seed64;
6006 XXH_writeLE64((xxh_u8*)customSecret + 16*i, lo);
6007 XXH_writeLE64((xxh_u8*)customSecret + 16*i + 8, hi);
6008 } }
6009}

6084XXH_FORCE_INLINE void
6085XXH3_hashLong_internal_loop(xxh_u64* XXH_RESTRICT acc,
6086 const xxh_u8* XXH_RESTRICT input, size_t len,

6088 XXH3_f_accumulate f_acc,
6089 XXH3_f_scrambleAcc f_scramble)
6090{
6091 size_t const nbStripesPerBlock = (secretSize - XXH_STRIPE_LEN) / XXH_SECRET_CONSUME_RATE;
6092 size_t const block_len = XXH_STRIPE_LEN * nbStripesPerBlock;
6093 size_t const nb_blocks = (len - 1) / block_len;
6094
6095 size_t n;
6096
6097 XXH_ASSERT(secretSize >= XXH3_SECRET_SIZE_MIN);
6098
6099 for (n = 0; n < nb_blocks; n++) {
6100 f_acc(acc, input + n*block_len, secret, nbStripesPerBlock);
6101 f_scramble(acc, secret + secretSize - XXH_STRIPE_LEN);
6102 }
6103
6104 /* last partial block */
6105 XXH_ASSERT(len > XXH_STRIPE_LEN);
6106 { size_t const nbStripes = ((len - 1) - (block_len * nb_blocks)) / XXH_STRIPE_LEN;
6107 XXH_ASSERT(nbStripes <= (secretSize / XXH_SECRET_CONSUME_RATE));
6108 f_acc(acc, input + nb_blocks*block_len, secret, nbStripes);
6109
6110 /* last stripe */
6111 { const xxh_u8* const p = input + len - XXH_STRIPE_LEN;
6112#define XXH_SECRET_LASTACC_START 7 /* not aligned on 8, last secret is different from acc & scrambler */
6113 XXH3_accumulate_512(acc, p, secret + secretSize - XXH_STRIPE_LEN - XXH_SECRET_LASTACC_START);
6114 } }
6115}

6117XXH_FORCE_INLINE xxh_u64
6118XXH3_mix2Accs(const xxh_u64* XXH_RESTRICT acc, const xxh_u8* XXH_RESTRICT secret)
6119{
6120 return XXH3_mul128_fold64(
6121 acc[0] ^ XXH_readLE64(secret),
6122 acc[1] ^ XXH_readLE64(secret+8) );
6123}

6125static XXH_PUREF XXH64_hash_t
6126XXH3_mergeAccs(const xxh_u64* XXH_RESTRICT acc, const xxh_u8* XXH_RESTRICT secret, xxh_u64 start)
6127{
6128 xxh_u64 result64 = start;
6129 size_t i = 0;
6130
6131 for (i = 0; i < 4; i++) {
6132 result64 += XXH3_mix2Accs(acc+2*i, secret + 16*i);
6133#if defined(__clang__) /* Clang */ \
6134 && (defined(__arm__) || defined(__thumb__)) /* ARMv7 */ \
6135 && (defined(__ARM_NEON) || defined(__ARM_NEON__)) /* NEON */ \
6136 && !defined(XXH_ENABLE_AUTOVECTORIZE) /* Define to disable */
6137 /*
6138 * UGLY HACK:
6139 * Prevent autovectorization on Clang ARMv7-a. Exact same problem as
6140 * the one in XXH3_len_129to240_64b. Speeds up shorter keys > 240b.
6141 * XXH3_64bits, len == 256, Snapdragon 835:
6142 * without hack: 2063.7 MB/s
6143 * with hack: 2560.7 MB/s
6144 */
6145 XXH_COMPILER_GUARD(result64);
6146#endif
6147 }
6148
6149 return XXH3_avalanche(result64);
6150}

6311#ifndef XXH_NO_STREAM
6312/*
6313 * Malloc's a pointer that is always aligned to @align.
6314 *
6315 * This must be freed with `XXH_alignedFree()`.
6316 *
6317 * malloc typically guarantees 16 byte alignment on 64-bit systems and 8 byte
6318 * alignment on 32-bit. This isn't enough for the 32 byte aligned loads in AVX2
6319 * or on 32-bit, the 16 byte aligned loads in SSE2 and NEON.
6320 *
6321 * This underalignment previously caused a rather obvious crash which went
6322 * completely unnoticed due to XXH3_createState() not actually being tested.
6323 * Credit to RedSpah for noticing this bug.
6324 *
6325 * The alignment is done manually: Functions like posix_memalign or _mm_malloc
6326 * are avoided: To maintain portability, we would have to write a fallback
6327 * like this anyways, and besides, testing for the existence of library
6328 * functions without relying on external build tools is impossible.
6329 *
6330 * The method is simple: Overallocate, manually align, and store the offset
6331 * to the original behind the returned pointer.
6332 *
6333 * Align must be a power of 2 and 8 <= align <= 128.
6334 */
6335static XXH_MALLOCF void* XXH_alignedMalloc(size_t s, size_t align)
6336{
6337 XXH_ASSERT(align <= 128 && align >= 8); /* range check */
6338 XXH_ASSERT((align & (align-1)) == 0); /* power of 2 */
6339 XXH_ASSERT(s != 0 && s < (s + align)); /* empty/overflow */
6340 { /* Overallocate to make room for manual realignment and an offset byte */
6341 xxh_u8* base = (xxh_u8*)XXH_malloc(s + align);
6342 if (base != NULL) {
6343 /*
6344 * Get the offset needed to align this pointer.
6345 *
6346 * Even if the returned pointer is aligned, there will always be
6347 * at least one byte to store the offset to the original pointer.
6348 */
6349 size_t offset = align - ((size_t)base & (align - 1)); /* base % align */
6350 /* Add the offset for the now-aligned pointer */
6351 xxh_u8* ptr = base + offset;
6352
6353 XXH_ASSERT((size_t)ptr % align == 0);
6354
6355 /* Store the offset immediately before the returned pointer. */
6356 ptr[-1] = (xxh_u8)offset;
6357 return ptr;
6358 }
6359 return NULL;
6360 }
6361}
6362/*
6363 * Frees an aligned pointer allocated by XXH_alignedMalloc(). Don't pass
6364 * normal malloc'd pointers, XXH_alignedMalloc has a specific data layout.
6365 */
6366static void XXH_alignedFree(void* p)
6367{
6368 if (p != NULL) {
6369 xxh_u8* ptr = (xxh_u8*)p;
6370 /* Get the offset byte we added in XXH_malloc. */
6371 xxh_u8 offset = ptr[-1];
6372 /* Free the original malloc'd pointer */
6373 xxh_u8* base = ptr - offset;
6374 XXH_free(base);
6375 }

6509XXH_FORCE_INLINE const xxh_u8 *
6510XXH3_consumeStripes(xxh_u64* XXH_RESTRICT acc,
6511 size_t* XXH_RESTRICT nbStripesSoFarPtr, size_t nbStripesPerBlock,
6512 const xxh_u8* XXH_RESTRICT input, size_t nbStripes,
6513 const xxh_u8* XXH_RESTRICT secret, size_t secretLimit,
6514 XXH3_f_accumulate f_acc,
6515 XXH3_f_scrambleAcc f_scramble)
6516{
6517 const xxh_u8* initialSecret = secret + *nbStripesSoFarPtr * XXH_SECRET_CONSUME_RATE;
6518 /* Process full blocks */
6519 if (nbStripes >= (nbStripesPerBlock - *nbStripesSoFarPtr)) {
6520 /* Process the initial partial block... */
6521 size_t nbStripesThisIter = nbStripesPerBlock - *nbStripesSoFarPtr;
6522
6523 do {
6524 /* Accumulate and scramble */
6525 f_acc(acc, input, initialSecret, nbStripesThisIter);
6526 f_scramble(acc, secret + secretLimit);
6527 input += nbStripesThisIter * XXH_STRIPE_LEN;
6528 nbStripes -= nbStripesThisIter;
6529 /* Then continue the loop with the full block size */
6530 nbStripesThisIter = nbStripesPerBlock;
6531 initialSecret = secret;
6532 } while (nbStripes >= nbStripesPerBlock);
6533 *nbStripesSoFarPtr = 0;
6534 }
6535 /* Process a partial block */
6536 if (nbStripes > 0) {
6537 f_acc(acc, input, initialSecret, nbStripes);
6538 input += nbStripes * XXH_STRIPE_LEN;
6539 *nbStripesSoFarPtr += nbStripes;
6540 }
6541 /* Return end pointer */
6542 return input;
6543}

6545#ifndef XXH3_STREAM_USE_STACK
6546# if XXH_SIZE_OPT <= 0 && !defined(__clang__) /* clang doesn't need additional stack space */
6547# define XXH3_STREAM_USE_STACK 1
6548# endif
6549#endif
6550/* This function accepts f_acc and f_scramble as function pointers,
6551 * making it possible to implement multiple variants with different acc & scramble stages.
6552 * This is notably useful to implement multiple vector variants with different intrinsics.
6553 */
6554XXH_FORCE_INLINE XXH_errorcode
6555XXH3_update(XXH3_state_t* XXH_RESTRICT const state,
6556 const xxh_u8* XXH_RESTRICT input, size_t len,
6557 XXH3_f_accumulate f_acc,
6558 XXH3_f_scrambleAcc f_scramble)
6559{
6560 if (input==NULL) {
6561 XXH_ASSERT(len == 0);
6562 return XXH_OK;
6563 }
6564
6565 XXH_ASSERT(state != NULL);
6566 { const xxh_u8* const bEnd = input + len;
6567 const unsigned char* const secret = (state->extSecret == NULL) ? state->customSecret : state->extSecret;
6568#if defined(XXH3_STREAM_USE_STACK) && XXH3_STREAM_USE_STACK >= 1
6569 /* For some reason, gcc and MSVC seem to suffer greatly
6570 * when operating accumulators directly into state.
6571 * Operating into stack space seems to enable proper optimization.
6572 * clang, on the other hand, doesn't seem to need this trick */
6573 XXH_ALIGN(XXH_ACC_ALIGN) xxh_u64 acc[8];
6574 XXH_memcpy(acc, state->acc, sizeof(acc));
6575#else
6576 xxh_u64* XXH_RESTRICT const acc = state->acc;
6577#endif
6578 state->totalLen += len;
6579 XXH_ASSERT(state->bufferedSize <= XXH3_INTERNALBUFFER_SIZE);
6580
6581 /* small input : just fill in tmp buffer */
6582 if (len <= XXH3_INTERNALBUFFER_SIZE - state->bufferedSize) {
6583 XXH_memcpy(state->buffer + state->bufferedSize, input, len);
6584 state->bufferedSize += (XXH32_hash_t)len;
6585 return XXH_OK;
6586 }
6587
6588 /* total input is now > XXH3_INTERNALBUFFER_SIZE */
6589 #define XXH3_INTERNALBUFFER_STRIPES (XXH3_INTERNALBUFFER_SIZE / XXH_STRIPE_LEN)
6590 XXH_STATIC_ASSERT(XXH3_INTERNALBUFFER_SIZE % XXH_STRIPE_LEN == 0); /* clean multiple */
6591
6592 /*
6593 * Internal buffer is partially filled (always, except at beginning)
6594 * Complete it, then consume it.
6595 */
6596 if (state->bufferedSize) {
6597 size_t const loadSize = XXH3_INTERNALBUFFER_SIZE - state->bufferedSize;
6598 XXH_memcpy(state->buffer + state->bufferedSize, input, loadSize);
6599 input += loadSize;
6600 XXH3_consumeStripes(acc,
6601 &state->nbStripesSoFar, state->nbStripesPerBlock,
6602 state->buffer, XXH3_INTERNALBUFFER_STRIPES,
6603 secret, state->secretLimit,
6604 f_acc, f_scramble);
6605 state->bufferedSize = 0;
6606 }
6607 XXH_ASSERT(input < bEnd);
6608 if (bEnd - input > XXH3_INTERNALBUFFER_SIZE) {
6609 size_t nbStripes = (size_t)(bEnd - 1 - input) / XXH_STRIPE_LEN;
6610 input = XXH3_consumeStripes(acc,
6611 &state->nbStripesSoFar, state->nbStripesPerBlock,
6612 input, nbStripes,
6613 secret, state->secretLimit,
6614 f_acc, f_scramble);
6615 XXH_memcpy(state->buffer + sizeof(state->buffer) - XXH_STRIPE_LEN, input - XXH_STRIPE_LEN, XXH_STRIPE_LEN);
6616
6617 }
6618 /* Some remaining input (always) : buffer it */
6619 XXH_ASSERT(input < bEnd);
6620 XXH_ASSERT(bEnd - input <= XXH3_INTERNALBUFFER_SIZE);
6621 XXH_ASSERT(state->bufferedSize == 0);
6622 XXH_memcpy(state->buffer, input, (size_t)(bEnd-input));
6623 state->bufferedSize = (XXH32_hash_t)(bEnd-input);
6624#if defined(XXH3_STREAM_USE_STACK) && XXH3_STREAM_USE_STACK >= 1
6625 /* save stack accumulators into state */
6626 XXH_memcpy(state->acc, acc, sizeof(acc));
6627#endif
6628 }
6629
6630 return XXH_OK;
6631}

6651XXH_FORCE_INLINE void
6652XXH3_digest_long (XXH64_hash_t* acc,
6653 const XXH3_state_t* state,
6654 const unsigned char* secret)
6655{
6656 xxh_u8 lastStripe[XXH_STRIPE_LEN];
6657 const xxh_u8* lastStripePtr;
6658
6659 /*
6660 * Digest on a local copy. This way, the state remains unaltered, and it can
6661 * continue ingesting more input afterwards.
6662 */
6663 XXH_memcpy(acc, state->acc, sizeof(state->acc));
6664 if (state->bufferedSize >= XXH_STRIPE_LEN) {
6665 /* Consume remaining stripes then point to remaining data in buffer */
6666 size_t const nbStripes = (state->bufferedSize - 1) / XXH_STRIPE_LEN;
6667 size_t nbStripesSoFar = state->nbStripesSoFar;
6668 XXH3_consumeStripes(acc,
6669 &nbStripesSoFar, state->nbStripesPerBlock,
6670 state->buffer, nbStripes,
6671 secret, state->secretLimit,
6672 XXH3_accumulate, XXH3_scrambleAcc);
6673 lastStripePtr = state->buffer + state->bufferedSize - XXH_STRIPE_LEN;
6674 } else { /* bufferedSize < XXH_STRIPE_LEN */
6675 /* Copy to temp buffer */
6676 size_t const catchupSize = XXH_STRIPE_LEN - state->bufferedSize;
6677 XXH_ASSERT(state->bufferedSize > 0); /* there is always some input buffered */
6678 XXH_memcpy(lastStripe, state->buffer + sizeof(state->buffer) - catchupSize, catchupSize);
6679 XXH_memcpy(lastStripe + catchupSize, state->buffer, state->bufferedSize);
6680 lastStripePtr = lastStripe;
6681 }
6682 /* Last stripe */
6683 XXH3_accumulate_512(acc,
6684 lastStripePtr,
6685 secret + state->secretLimit - XXH_SECRET_LASTACC_START);
6686}

6706/* ==========================================
6707 * XXH3 128 bits (a.k.a XXH128)
6708 * ==========================================
6709 * XXH3's 128-bit variant has better mixing and strength than the 64-bit variant,
6710 * even without counting the significantly larger output size.
6711 *
6712 * For example, extra steps are taken to avoid the seed-dependent collisions
6713 * in 17-240 byte inputs (See XXH3_mix16B and XXH128_mix32B).
6714 *
6715 * This strength naturally comes at the cost of some speed, especially on short
6716 * lengths. Note that longer hashes are about as fast as the 64-bit version
6717 * due to it using only a slight modification of the 64-bit loop.
6718 *
6719 * XXH128 is also more oriented towards 64-bit machines. It is still extremely
6720 * fast for a _128-bit_ hash on 32-bit (it usually clears XXH64).
6721 */
6722
6723XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6724XXH3_len_1to3_128b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
6725{
6726 /* A doubled version of 1to3_64b with different constants. */
6727 XXH_ASSERT(input != NULL);
6728 XXH_ASSERT(1 <= len && len <= 3);
6729 XXH_ASSERT(secret != NULL);
6730 /*
6731 * len = 1: combinedl = { input[0], 0x01, input[0], input[0] }
6732 * len = 2: combinedl = { input[1], 0x02, input[0], input[1] }
6733 * len = 3: combinedl = { input[2], 0x03, input[0], input[1] }
6734 */
6735 { xxh_u8 const c1 = input[0];
6736 xxh_u8 const c2 = input[len >> 1];
6737 xxh_u8 const c3 = input[len - 1];
6738 xxh_u32 const combinedl = ((xxh_u32)c1 <<16) | ((xxh_u32)c2 << 24)
6739 | ((xxh_u32)c3 << 0) | ((xxh_u32)len << 8);
6740 xxh_u32 const combinedh = XXH_rotl32(XXH_swap32(combinedl), 13);
6741 xxh_u64 const bitflipl = (XXH_readLE32(secret) ^ XXH_readLE32(secret+4)) + seed;
6742 xxh_u64 const bitfliph = (XXH_readLE32(secret+8) ^ XXH_readLE32(secret+12)) - seed;
6743 xxh_u64 const keyed_lo = (xxh_u64)combinedl ^ bitflipl;
6744 xxh_u64 const keyed_hi = (xxh_u64)combinedh ^ bitfliph;
6745 XXH128_hash_t h128;
6746 h128.low64 = XXH64_avalanche(keyed_lo);
6747 h128.high64 = XXH64_avalanche(keyed_hi);
6748 return h128;
6749 }
6750}
6751
6752XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6753XXH3_len_4to8_128b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
6754{
6755 XXH_ASSERT(input != NULL);
6756 XXH_ASSERT(secret != NULL);
6757 XXH_ASSERT(4 <= len && len <= 8);
6758 seed ^= (xxh_u64)XXH_swap32((xxh_u32)seed) << 32;
6759 { xxh_u32 const input_lo = XXH_readLE32(input);
6760 xxh_u32 const input_hi = XXH_readLE32(input + len - 4);
6761 xxh_u64 const input_64 = input_lo + ((xxh_u64)input_hi << 32);
6762 xxh_u64 const bitflip = (XXH_readLE64(secret+16) ^ XXH_readLE64(secret+24)) + seed;
6763 xxh_u64 const keyed = input_64 ^ bitflip;
6764
6765 /* Shift len to the left to ensure it is even, this avoids even multiplies. */
6766 XXH128_hash_t m128 = XXH_mult64to128(keyed, XXH_PRIME64_1 + (len << 2));
6767
6768 m128.high64 += (m128.low64 << 1);
6769 m128.low64 ^= (m128.high64 >> 3);
6770
6771 m128.low64 = XXH_xorshift64(m128.low64, 35);
6772 m128.low64 *= PRIME_MX2;
6773 m128.low64 = XXH_xorshift64(m128.low64, 28);
6774 m128.high64 = XXH3_avalanche(m128.high64);
6775 return m128;
6776 }
6777}

6779XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6780XXH3_len_9to16_128b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
6781{
6782 XXH_ASSERT(input != NULL);
6783 XXH_ASSERT(secret != NULL);
6784 XXH_ASSERT(9 <= len && len <= 16);
6785 { xxh_u64 const bitflipl = (XXH_readLE64(secret+32) ^ XXH_readLE64(secret+40)) - seed;
6786 xxh_u64 const bitfliph = (XXH_readLE64(secret+48) ^ XXH_readLE64(secret+56)) + seed;
6787 xxh_u64 const input_lo = XXH_readLE64(input);
6788 xxh_u64 input_hi = XXH_readLE64(input + len - 8);
6789 XXH128_hash_t m128 = XXH_mult64to128(input_lo ^ input_hi ^ bitflipl, XXH_PRIME64_1);
6790 /*
6791 * Put len in the middle of m128 to ensure that the length gets mixed to
6792 * both the low and high bits in the 128x64 multiply below.
6793 */
6794 m128.low64 += (xxh_u64)(len - 1) << 54;
6795 input_hi ^= bitfliph;
6796 /*
6797 * Add the high 32 bits of input_hi to the high 32 bits of m128, then
6798 * add the long product of the low 32 bits of input_hi and XXH_PRIME32_2 to
6799 * the high 64 bits of m128.
6800 *
6801 * The best approach to this operation is different on 32-bit and 64-bit.
6802 */
6803 if (sizeof(void *) < sizeof(xxh_u64)) { /* 32-bit */
6804 /*
6805 * 32-bit optimized version, which is more readable.
6806 *
6807 * On 32-bit, it removes an ADC and delays a dependency between the two
6808 * halves of m128.high64, but it generates an extra mask on 64-bit.
6809 */
6810 m128.high64 += (input_hi & 0xFFFFFFFF00000000ULL) + XXH_mult32to64((xxh_u32)input_hi, XXH_PRIME32_2);
6811 } else {
6812 /*
6813 * 64-bit optimized (albeit more confusing) version.
6814 *
6815 * Uses some properties of addition and multiplication to remove the mask:
6816 *
6817 * Let:
6818 * a = input_hi.lo = (input_hi & 0x00000000FFFFFFFF)
6819 * b = input_hi.hi = (input_hi & 0xFFFFFFFF00000000)
6820 * c = XXH_PRIME32_2
6821 *
6822 * a + (b * c)
6823 * Inverse Property: x + y - x == y
6824 * a + (b * (1 + c - 1))
6825 * Distributive Property: x * (y + z) == (x * y) + (x * z)
6826 * a + (b * 1) + (b * (c - 1))
6827 * Identity Property: x * 1 == x
6828 * a + b + (b * (c - 1))
6829 *
6830 * Substitute a, b, and c:
6831 * input_hi.hi + input_hi.lo + ((xxh_u64)input_hi.lo * (XXH_PRIME32_2 - 1))
6832 *
6833 * Since input_hi.hi + input_hi.lo == input_hi, we get this:
6834 * input_hi + ((xxh_u64)input_hi.lo * (XXH_PRIME32_2 - 1))
6835 */
6836 m128.high64 += input_hi + XXH_mult32to64((xxh_u32)input_hi, XXH_PRIME32_2 - 1);
6837 }
6838 /* m128 ^= XXH_swap64(m128 >> 64); */
6839 m128.low64 ^= XXH_swap64(m128.high64);
6840
6841 { /* 128x64 multiply: h128 = m128 * XXH_PRIME64_2; */
6842 XXH128_hash_t h128 = XXH_mult64to128(m128.low64, XXH_PRIME64_2);
6843 h128.high64 += m128.high64 * XXH_PRIME64_2;
6844
6845 h128.low64 = XXH3_avalanche(h128.low64);
6846 h128.high64 = XXH3_avalanche(h128.high64);
6847 return h128;
6848 } }
6849}

6870/*
6871 * A bit slower than XXH3_mix16B, but handles multiply by zero better.
6872 */
6873XXH_FORCE_INLINE XXH128_hash_t
6874XXH128_mix32B(XXH128_hash_t acc, const xxh_u8* input_1, const xxh_u8* input_2,
6875 const xxh_u8* secret, XXH64_hash_t seed)
6876{
6877 acc.low64 += XXH3_mix16B (input_1, secret+0, seed);
6878 acc.low64 ^= XXH_readLE64(input_2) + XXH_readLE64(input_2 + 8);
6879 acc.high64 += XXH3_mix16B (input_2, secret+16, seed);
6880 acc.high64 ^= XXH_readLE64(input_1) + XXH_readLE64(input_1 + 8);
6881 return acc;

6885XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6886XXH3_len_17to128_128b(const xxh_u8* XXH_RESTRICT input, size_t len,
6887 const xxh_u8* XXH_RESTRICT secret, size_t secretSize,
6888 XXH64_hash_t seed)
6889{
6890 XXH_ASSERT(secretSize >= XXH3_SECRET_SIZE_MIN); (void)secretSize;
6891 XXH_ASSERT(16 < len && len <= 128);
6892
6893 { XXH128_hash_t acc;
6894 acc.low64 = len * XXH_PRIME64_1;
6895 acc.high64 = 0;
6896
6897#if XXH_SIZE_OPT >= 1
6898 {
6899 /* Smaller, but slightly slower. */
6900 unsigned int i = (unsigned int)(len - 1) / 32;
6901 do {
6902 acc = XXH128_mix32B(acc, input+16*i, input+len-16*(i+1), secret+32*i, seed);
6903 } while (i-- != 0);
6904 }
6905#else
6906 if (len > 32) {
6907 if (len > 64) {
6908 if (len > 96) {
6909 acc = XXH128_mix32B(acc, input+48, input+len-64, secret+96, seed);
6910 }
6911 acc = XXH128_mix32B(acc, input+32, input+len-48, secret+64, seed);
6912 }
6913 acc = XXH128_mix32B(acc, input+16, input+len-32, secret+32, seed);
6914 }
6915 acc = XXH128_mix32B(acc, input, input+len-16, secret, seed);
6916#endif
6917 { XXH128_hash_t h128;
6918 h128.low64 = acc.low64 + acc.high64;
6919 h128.high64 = (acc.low64 * XXH_PRIME64_1)
6920 + (acc.high64 * XXH_PRIME64_4)
6921 + ((len - seed) * XXH_PRIME64_2);
6922 h128.low64 = XXH3_avalanche(h128.low64);
6923 h128.high64 = (XXH64_hash_t)0 - XXH3_avalanche(h128.high64);
6924 return h128;
6925 }
6926 }

6929XXH_NO_INLINE XXH_PUREF XXH128_hash_t
6930XXH3_len_129to240_128b(const xxh_u8* XXH_RESTRICT input, size_t len,
6931 const xxh_u8* XXH_RESTRICT secret, size_t secretSize,
6932 XXH64_hash_t seed)
6933{
6934 XXH_ASSERT(secretSize >= XXH3_SECRET_SIZE_MIN); (void)secretSize;
6935 XXH_ASSERT(128 < len && len <= XXH3_MIDSIZE_MAX);
6936
6937 { XXH128_hash_t acc;
6938 unsigned i;
6939 acc.low64 = len * XXH_PRIME64_1;
6940 acc.high64 = 0;
6941 /*
6942 * We set as `i` as offset + 32. We do this so that unchanged
6943 * `len` can be used as upper bound. This reaches a sweet spot
6944 * where both x86 and aarch64 get simple agen and good codegen
6945 * for the loop.
6946 */
6947 for (i = 32; i < 160; i += 32) {
6948 acc = XXH128_mix32B(acc,
6949 input + i - 32,
6950 input + i - 16,
6951 secret + i - 32,
6952 seed);
6953 }
6954 acc.low64 = XXH3_avalanche(acc.low64);
6955 acc.high64 = XXH3_avalanche(acc.high64);
6956 /*
6957 * NB: `i <= len` will duplicate the last 32-bytes if
6958 * len % 32 was zero. This is an unfortunate necessity to keep
6959 * the hash result stable.
6960 */
6961 for (i=160; i <= len; i += 32) {
6962 acc = XXH128_mix32B(acc,
6963 input + i - 32,
6964 input + i - 16,
6965 secret + XXH3_MIDSIZE_STARTOFFSET + i - 160,
6966 seed);
6967 }
6968 /* last bytes */
6969 acc = XXH128_mix32B(acc,
6970 input + len - 16,
6971 input + len - 32,
6972 secret + XXH3_SECRET_SIZE_MIN - XXH3_MIDSIZE_LASTOFFSET - 16,
6973 (XXH64_hash_t)0 - seed);
6974
6975 { XXH128_hash_t h128;
6976 h128.low64 = acc.low64 + acc.high64;
6977 h128.high64 = (acc.low64 * XXH_PRIME64_1)
6978 + (acc.high64 * XXH_PRIME64_4)
6979 + ((len - seed) * XXH_PRIME64_2);
6980 h128.low64 = XXH3_avalanche(h128.low64);
6981 h128.high64 = (XXH64_hash_t)0 - XXH3_avalanche(h128.high64);
6982 return h128;
6983 }
6984 }
