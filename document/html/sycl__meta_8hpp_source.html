61// Check range.
62template<unsigned nt, unsigned vt, unsigned vt0 = vt, typename F>
63void sycl_strided_iterate(F f, unsigned tid, unsigned count) {
64 // Unroll the first vt0 elements of each thread.
65 if(vt0 > 1 && count >= nt * vt0) {
66 sycl_strided_iterate<nt, vt0>(f, tid); // No checking
67 } else {
68 sycl_iterate<vt0>([=](auto i) {
69 auto j = nt * i + tid;
70 if(j < count) f(i, j);
71 });
72 }
73
74 // TODO: seems dummy when vt0 == vt
75 sycl_iterate<vt0, vt>([=](auto i) {
76 auto j = nt * i + tid;
77 if(j < count) f(i, j);
78 });
79}
