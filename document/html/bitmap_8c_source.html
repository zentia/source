76// mask of `bit_count` bits set shifted to the left by `shiftl`
77static inline mi_bfield_t mi_bfield_mask(size_t bit_count, size_t shiftl) {
78 mi_assert_internal(bit_count > 0);
79 mi_assert_internal(bit_count + shiftl <= MI_BFIELD_BITS);
80 const mi_bfield_t mask0 = (bit_count < MI_BFIELD_BITS ? (mi_bfield_one() << bit_count)-1 : mi_bfield_all_set());
81 return (mask0 << shiftl);
82}

89// Set a bit atomically. Returns `true` if the bit transitioned from 0 to 1
90static inline bool mi_bfield_atomic_set(_Atomic(mi_bfield_t)*b, size_t idx) {
91 mi_assert_internal(idx < MI_BFIELD_BITS);
92 const mi_bfield_t mask = mi_bfield_mask(1, idx);;
93 const mi_bfield_t old = mi_atomic_or_acq_rel(b, mask);
94 return ((old&mask) == 0);
95}

97// Clear a bit atomically. Returns `true` if the bit transitioned from 1 to 0.
98// `all_clear` is set if the new bfield is zero.
99static inline bool mi_bfield_atomic_clear(_Atomic(mi_bfield_t)*b, size_t idx, bool* all_clear) {
100 mi_assert_internal(idx < MI_BFIELD_BITS);
101 const mi_bfield_t mask = mi_bfield_mask(1, idx);;
102 mi_bfield_t old = mi_atomic_and_acq_rel(b, ~mask);
103 if (all_clear != NULL) { *all_clear = ((old&~mask)==0); }
104 return ((old&mask) == mask);
105}

107// Clear a bit but only when/once it is set. This is used by concurrent free's while
108// the page is abandoned and mapped. This can incure a busy wait :-( but it should
109// happen almost never (and is accounted for in the stats)
110static inline void mi_bfield_atomic_clear_once_set(_Atomic(mi_bfield_t)*b, size_t idx) {
111 mi_assert_internal(idx < MI_BFIELD_BITS);
112 const mi_bfield_t mask = mi_bfield_mask(1, idx);;
113 mi_bfield_t old = mi_atomic_load_relaxed(b);
114 do {
115 if mi_unlikely((old&mask) == 0) {
116 old = mi_atomic_load_acquire(b);
117 if ((old&mask)==0) {
118 mi_subproc_stat_counter_increase(_mi_subproc(), pages_unabandon_busy_wait, 1);
119 }
120 while ((old&mask)==0) { // busy wait
121 mi_atomic_yield();
122 old = mi_atomic_load_acquire(b);
123 }
124 }
125 } while (!mi_atomic_cas_weak_acq_rel(b,&old, (old&~mask)));
126 mi_assert_internal((old&mask)==mask); // we should only clear when it was set
127}

129// Set a mask set of bits atomically, and return true of the mask bits transitioned from all 0's to 1's.
130// `already_set` contains the count of bits that were already set (used when committing ranges to account
131// statistics correctly).
132static inline bool mi_bfield_atomic_set_mask(_Atomic(mi_bfield_t)*b, mi_bfield_t mask, size_t* already_set) {
133 mi_assert_internal(mask != 0);
134 mi_bfield_t old = mi_atomic_load_relaxed(b);
135 while (!mi_atomic_cas_weak_acq_rel(b, &old, old|mask)) {}; // try to atomically set the mask bits until success
136 if (already_set!=NULL) { *already_set = mi_bfield_popcount(old&mask); }
137 return ((old&mask) == 0);
138}

140// Clear a mask set of bits atomically, and return true of the mask bits transitioned from all 1's to 0's
141// `all_clear` is set to `true` if the new bfield became zero.
142static inline bool mi_bfield_atomic_clear_mask(_Atomic(mi_bfield_t)*b, mi_bfield_t mask, bool* all_clear) {
143 mi_assert_internal(mask != 0);
144 mi_bfield_t old = mi_atomic_load_relaxed(b);
145 while (!mi_atomic_cas_weak_acq_rel(b, &old, old&~mask)) {}; // try to atomically clear the mask bits until success
146 if (all_clear != NULL) { *all_clear = ((old&~mask)==0); }
147 return ((old&mask) == mask);
148}

165// Tries to clear a mask atomically, and returns true if the mask bits atomically transitioned from mask to 0
166// and false otherwise (leaving the bit field as is).
167// `all_clear` is set to `true` if the new bfield became zero.
168static inline bool mi_bfield_atomic_try_clear_mask_of(_Atomic(mi_bfield_t)*b, mi_bfield_t mask, mi_bfield_t expect, bool* all_clear) {
169 mi_assert_internal(mask != 0);
170 // try to atomically clear the mask bits
171 do {
172 if ((expect & mask) != mask) { // are all bits still set?
173 if (all_clear != NULL) { *all_clear = (expect == 0); }
174 return false;
175 }
176 } while (!mi_atomic_cas_weak_acq_rel(b, &expect, expect & ~mask));
177 if (all_clear != NULL) { *all_clear = ((expect & ~mask) == 0); }
178 return true;
179}

205// Try to clear a full field of bits atomically, and return true all bits transitioned from all 1's to 0's.
206// and false otherwise leaving the bit field as-is.
207// `all_clear` is set to true if the new bfield became zero (which is always the case if successful).
208static inline bool mi_bfield_atomic_try_clearX(_Atomic(mi_bfield_t)*b, bool* all_clear) {
209 mi_bfield_t old = mi_bfield_all_set();
210 if (mi_atomic_cas_strong_acq_rel(b, &old, mi_bfield_zero())) {
211 if (all_clear != NULL) { *all_clear = true; }
212 return true;
213 }
214 else return false;

277// Set `0 < n <= MI_BFIELD_BITS`, and return true of the mask bits transitioned from all 0's to 1's.
278// `already_set` contains the count of bits that were already set (used when committing ranges to account
279// statistics correctly).
280// Can cross over two bfields.
281static inline bool mi_bchunk_setNX(mi_bchunk_t* chunk, size_t cidx, size_t n, size_t* already_set) {
282 mi_assert_internal(cidx < MI_BCHUNK_BITS);
283 mi_assert_internal(n > 0 && n <= MI_BFIELD_BITS);
284 const size_t i = cidx / MI_BFIELD_BITS;
285 const size_t idx = cidx % MI_BFIELD_BITS;
286 if mi_likely(idx + n <= MI_BFIELD_BITS) {
287 // within one field
288 return mi_bfield_atomic_set_mask(&chunk->bfields[i], mi_bfield_mask(n,idx), already_set);
289 }
290 else {
291 // spanning two fields
292 const size_t m = MI_BFIELD_BITS - idx; // bits to clear in the first field
293 mi_assert_internal(m < n);
294 mi_assert_internal(i < MI_BCHUNK_FIELDS - 1);
295 mi_assert_internal(idx + m <= MI_BFIELD_BITS);
296 size_t already_set1;
297 const bool all_set1 = mi_bfield_atomic_set_mask(&chunk->bfields[i], mi_bfield_mask(m, idx), &already_set1);
298 mi_assert_internal(n - m > 0);
299 mi_assert_internal(n - m < MI_BFIELD_BITS);
300 size_t already_set2;
301 const bool all_set2 = mi_bfield_atomic_set_mask(&chunk->bfields[i+1], mi_bfield_mask(n - m, 0), &already_set2);
302 if (already_set != NULL) { *already_set = already_set1 + already_set2; }
303 return (all_set1 && all_set2);
304 }

307// Set a sequence of `n` bits within a chunk.
308// Returns true if all bits transitioned from 0 to 1 (or 1 to 0).
309mi_decl_noinline static bool mi_bchunk_xsetN_(mi_xset_t set, mi_bchunk_t* chunk, size_t cidx, size_t n, size_t* palready_set, bool* pmaybe_all_clear) {
310 mi_assert_internal(cidx + n <= MI_BCHUNK_BITS);
311 mi_assert_internal(n>0);
312 bool all_transition = true;
313 bool maybe_all_clear = true;
314 size_t total_already_set = 0;
315 size_t idx = cidx % MI_BFIELD_BITS;
316 size_t field = cidx / MI_BFIELD_BITS;
317 while (n > 0) {
318 size_t m = MI_BFIELD_BITS - idx; // m is the bits to xset in this field
319 if (m > n) { m = n; }
320 mi_assert_internal(idx + m <= MI_BFIELD_BITS);
321 mi_assert_internal(field < MI_BCHUNK_FIELDS);
322 const mi_bfield_t mask = mi_bfield_mask(m, idx);
323 size_t already_set = 0;
324 bool all_clear = false;
325 const bool transition = (set ? mi_bfield_atomic_set_mask(&chunk->bfields[field], mask, &already_set)
326 : mi_bfield_atomic_clear_mask(&chunk->bfields[field], mask, &all_clear));
327 mi_assert_internal((transition && already_set == 0) || (!transition && already_set > 0));
328 all_transition = all_transition && transition;
329 total_already_set += already_set;
330 maybe_all_clear = maybe_all_clear && all_clear;
331 // next field
332 field++;
333 idx = 0;
334 mi_assert_internal(m <= n);
335 n -= m;
336 }
337 if (palready_set!=NULL) { *palready_set = total_already_set; }
338 if (pmaybe_all_clear!=NULL) { *pmaybe_all_clear = maybe_all_clear; }
339 return all_transition;
340}

382 if (!mi_bfield_atomic_is_xset_mask(set, &chunk->bfields[field_idx], mask)) {

384 }
385 // next field
386 field_idx++;
387 idx = 0;
388 n -= m;
389 }
390 return true;
391}

408// Clear `0 < n <= MI_BITFIELD_BITS`. Can cross over a bfield boundary.
409static inline bool mi_bchunk_try_clearNX(mi_bchunk_t* chunk, size_t cidx, size_t n, bool* pmaybe_all_clear) {
410 mi_assert_internal(cidx < MI_BCHUNK_BITS);
411 mi_assert_internal(n <= MI_BFIELD_BITS);
412 const size_t i = cidx / MI_BFIELD_BITS;
413 const size_t idx = cidx % MI_BFIELD_BITS;
414 if mi_likely(idx + n <= MI_BFIELD_BITS) {
415 // within one field
416 return mi_bfield_atomic_try_clear_mask(&chunk->bfields[i], mi_bfield_mask(n, idx), pmaybe_all_clear);
417 }
418 else {
419 // spanning two fields (todo: use double-word atomic ops?)
420 const size_t m = MI_BFIELD_BITS - idx; // bits to clear in the first field
421 mi_assert_internal(m < n);
422 mi_assert_internal(i < MI_BCHUNK_FIELDS - 1);
423 bool field1_is_clear;
424 if (!mi_bfield_atomic_try_clear_mask(&chunk->bfields[i], mi_bfield_mask(m, idx), &field1_is_clear)) return false;
425 // try the second field as well
426 mi_assert_internal(n - m > 0);
427 mi_assert_internal(n - m < MI_BFIELD_BITS);
428 bool field2_is_clear;
429 if (!mi_bfield_atomic_try_clear_mask(&chunk->bfields[i+1], mi_bfield_mask(n - m, 0), &field2_is_clear)) {
430 // we failed to clear the second field, restore the first one
431 mi_bfield_atomic_set_mask(&chunk->bfields[i], mi_bfield_mask(m, idx), NULL);
432 return false;
433 }
434 if (pmaybe_all_clear != NULL) { *pmaybe_all_clear = field1_is_clear && field2_is_clear; }
435 return true;
436 }

447// Try to atomically clear a sequence of `n` bits within a chunk.
448// Returns true if all bits transitioned from 1 to 0,
449// and false otherwise leaving all bit fields as is.
450// Note: this is the complex one as we need to unwind partial atomic operations if we fail halfway..
451// `maybe_all_clear` is set to `true` if all the bfields involved become zero.
452mi_decl_noinline static bool mi_bchunk_try_clearN_(mi_bchunk_t* chunk, size_t cidx, size_t n, bool* pmaybe_all_clear) {
453 mi_assert_internal(cidx + n <= MI_BCHUNK_BITS);
454 mi_assert_internal(n>0);
455 if (pmaybe_all_clear != NULL) { *pmaybe_all_clear = true; }
456 if (n==0) return true;
457
458 // first field
459 const size_t start_idx = cidx % MI_BFIELD_BITS;
460 const size_t start_field = cidx / MI_BFIELD_BITS;
461 size_t field = start_field;
462 size_t m = MI_BFIELD_BITS - start_idx; // m are the bits to clear in this field
463 if (m > n) { m = n; }
464 mi_assert_internal(start_idx + m <= MI_BFIELD_BITS);
465 mi_assert_internal(start_field < MI_BCHUNK_FIELDS);
466 const mi_bfield_t mask_start = mi_bfield_mask(m, start_idx);
467 bool maybe_all_clear;
468 if (!mi_bfield_atomic_try_clear_mask(&chunk->bfields[field], mask_start, &maybe_all_clear)) return false;
469
470 // done?
471 mi_assert_internal(m <= n);
472 n -= m;
473
474 // continue with mid fields and last field: if these fail we need to recover by unsetting previous fields
475 // mid fields?
476 while (n >= MI_BFIELD_BITS) {
477 field++;
478 mi_assert_internal(field < MI_BCHUNK_FIELDS);
479 bool field_is_clear;
480 if (!mi_bfield_atomic_try_clearX(&chunk->bfields[field], &field_is_clear)) goto restore;
481 maybe_all_clear = maybe_all_clear && field_is_clear;
482 n -= MI_BFIELD_BITS;
483 }
484
485 // last field?
486 if (n > 0) {
487 mi_assert_internal(n < MI_BFIELD_BITS);
488 field++;
489 mi_assert_internal(field < MI_BCHUNK_FIELDS);
490 const mi_bfield_t mask_end = mi_bfield_mask(n, 0);
491 bool field_is_clear;
492 if (!mi_bfield_atomic_try_clear_mask(&chunk->bfields[field], mask_end, &field_is_clear)) goto restore;
493 maybe_all_clear = maybe_all_clear && field_is_clear;
494 }
495
496 if (pmaybe_all_clear != NULL) { *pmaybe_all_clear = maybe_all_clear; }
497 return true;
498
499restore:
500 // `field` is the index of the field that failed to set atomically; we need to restore all previous fields
501 mi_assert_internal(field > start_field);
502 while( field > start_field) {
503 field--;
504 if (field == start_field) {
505 mi_bfield_atomic_set_mask(&chunk->bfields[field], mask_start, NULL);
506 }
507 else {
508 mi_bfield_atomic_setX(&chunk->bfields[field], NULL); // mid-field: set all bits again
509 }
510 }
511 return false;
512}

756// find a sequence of `n` bits in a chunk with `0 < n <= MI_BFIELD_BITS` with all bits set,
757// and try to clear them atomically.
758// set `*pidx` to its bit index (0 <= *pidx <= MI_BCHUNK_BITS - n) on success.
759// will cross bfield boundaries.
760mi_decl_noinline static bool mi_bchunk_try_find_and_clearNX(mi_bchunk_t* chunk, size_t n, size_t* pidx) {
761 if (n == 0 || n > MI_BFIELD_BITS) return false;
762 const mi_bfield_t mask = mi_bfield_mask(n, 0);
763 // for all fields in the chunk
764 for (int i = 0; i < MI_BCHUNK_FIELDS; i++) {
765 mi_bfield_t b0 = mi_atomic_load_relaxed(&chunk->bfields[i]);
766 mi_bfield_t b = b0;
767 size_t idx;
768
769 // is there a range inside the field?
770 while (mi_bfield_find_least_bit(b, &idx)) { // find least 1-bit
771 if (idx + n > MI_BFIELD_BITS) break; // too short: maybe cross over, or continue with the next field
772
773 const size_t bmask = mask<<idx;
774 mi_assert_internal(bmask>>idx == mask);
775 if ((b&bmask) == bmask) { // found a match with all bits set, try clearing atomically
776 if mi_likely(mi_bfield_atomic_try_clear_mask_of(&chunk->bfields[i], bmask, b0, NULL)) {
777 *pidx = (i*MI_BFIELD_BITS) + idx;
778 mi_assert_internal(*pidx < MI_BCHUNK_BITS);
779 mi_assert_internal(*pidx + n <= MI_BCHUNK_BITS);
780 return true;
781 }
782 else {
783 // if we failed to atomically commit, reload b and try again from the start
784 b = b0 = mi_atomic_load_acquire(&chunk->bfields[i]);
785 }
786 }
787 else {
788 // advance by clearing the least run of ones, for example, with n>=4, idx=2:
789 // b = 1111 1101 1010 1100
790 // .. + (1<<idx) = 1111 1101 1011 0000
791 // .. & b = 1111 1101 1010 0000
792 b = b & (b + (mi_bfield_one() << idx));
793 }
794 }
795
796 // check if we can cross into the next bfield
797 if (b!=0 && i < MI_BCHUNK_FIELDS-1) {
798 const size_t post = mi_bfield_clz(~b);

818// find a sequence of `n` bits in a chunk with `n < MI_BCHUNK_BITS` with all bits set,
819// and try to clear them atomically.
820// set `*pidx` to its bit index (0 <= *pidx <= MI_BCHUNK_BITS - n) on success.
821// This can cross bfield boundaries.
822static mi_decl_noinline bool mi_bchunk_try_find_and_clearN_(mi_bchunk_t* chunk, size_t n, size_t* pidx) {
823 if (n == 0 || n > MI_BCHUNK_BITS) return false; // cannot be more than a chunk
824
825 // we first scan ahead to see if there is a range of `n` set bits, and only then try to clear atomically
826 mi_assert_internal(n>0);
827 const size_t skip_count = (n-1)/MI_BFIELD_BITS;
828 size_t cidx;
829 for (size_t i = 0; i < MI_BCHUNK_FIELDS - skip_count; i++)
830 {
831 size_t m = n; // bits to go
832
833 // first field
834 mi_bfield_t b = mi_atomic_load_relaxed(&chunk->bfields[i]);
835 size_t ones = mi_bfield_clz(~b);
836 cidx = (i*MI_BFIELD_BITS) + (MI_BFIELD_BITS - ones); // start index
837 if (ones >= m) {
838 // we found enough bits!
839 m = 0;
840 }
841 else {
842 m -= ones;
843
844 // keep scanning further fields?
845 size_t j = 1; // field count from i
846 while (i+j < MI_BCHUNK_FIELDS) {
847 mi_assert_internal(m > 0);
848 b = mi_atomic_load_relaxed(&chunk->bfields[i+j]);
849 ones = mi_bfield_ctz(~b);
850 if (ones >= m) {
851 // we found enough bits
852 m = 0;
853 break;
854 }
855 else if (ones == MI_BFIELD_BITS) {
856 // not enough yet, proceed to the next field
857 j++;
858 m -= MI_BFIELD_BITS;
859 }
860 else {
861 // the range was not enough, start from scratch
862 i = i + j - 1; // no need to re-scan previous fields, except the last one (with clz this time)
863 mi_assert_internal(m>0);
864 break;
865 }
866 }
867 }
868
869 // did we find a range?
870 if (m==0) {
871 if (mi_bchunk_try_clearN(chunk, cidx, n, NULL)) {
872 // we cleared all atomically
873 *pidx = cidx;
874 mi_assert_internal(*pidx < MI_BCHUNK_BITS);
875 mi_assert_internal(*pidx + n <= MI_BCHUNK_BITS);
876 return true;
877 }
878 // note: if we fail for a small `n` on the first field, we don't rescan that field (as `i` is incremented)
879 }
880 // otherwise continue searching
881 }
882 return false;
883}

912 const uint64x2_t v1 = vld1q_u64((uint64_t*)chunk->bfields + 2);
913 const uint64x2_t v2 = vld1q_u64((uint64_t*)chunk->bfields + 4);
914 const uint64x2_t v3 = vld1q_u64((uint64_t*)chunk->bfields + 6);
915 const uint64x2_t v = vorrq_u64(vorrq_u64(v0,v1),vorrq_u64(v2,v3));
916 return (vmaxvq_u32(vreinterpretq_u32_u64(v)) == 0);
917 #else
918 for (int i = 0; i < MI_BCHUNK_FIELDS; i++) {
919 if (mi_atomic_load_relaxed(&chunk->bfields[i]) != 0) return false;
920 }
921 return true;
922 #endif
923}

925// are all bits in a bitmap chunk set?
926static inline bool mi_bchunk_all_are_set_relaxed(mi_bchunk_t* chunk) {
927#if MI_OPT_SIMD && defined(__AVX2__) && (MI_BCHUNK_BITS==256)
928 const __m256i vec = _mm256_load_si256((const __m256i*)chunk->bfields);
929 return mi_mm256_is_ones(vec);
930#elif MI_OPT_SIMD && defined(__AVX2__) && (MI_BCHUNK_BITS==512)
931 // a 64b cache-line contains the entire chunk anyway so load both at once
932 const __m256i vec1 = _mm256_load_si256((const __m256i*)chunk->bfields);
933 const __m256i vec2 = _mm256_load_si256(((const __m256i*)chunk->bfields)+1);
934 return (mi_mm256_is_ones(_mm256_and_si256(vec1, vec2)));
935#elif MI_OPT_SIMD && (MI_BCHUNK_BITS==512) && MI_ARCH_ARM64
936 const uint64x2_t v0 = vld1q_u64((uint64_t*)chunk->bfields);
937 const uint64x2_t v1 = vld1q_u64((uint64_t*)chunk->bfields + 2);
938 const uint64x2_t v2 = vld1q_u64((uint64_t*)chunk->bfields + 4);
939 const uint64x2_t v3 = vld1q_u64((uint64_t*)chunk->bfields + 6);
940 const uint64x2_t v = vandq_u64(vandq_u64(v0,v1),vandq_u64(v2,v3));
941 return (vminvq_u32(vreinterpretq_u32_u64(v)) == 0xFFFFFFFFUL);
942#else
943 for (int i = 0; i < MI_BCHUNK_FIELDS; i++) {
944 if (~mi_atomic_load_relaxed(&chunk->bfields[i]) != 0) return false;
945 }
946 return true;
947#endif
948}

973static bool mi_bitmap_chunkmap_try_clear(mi_bitmap_t* bitmap, size_t chunk_idx) {
974 mi_assert(chunk_idx < mi_bitmap_chunk_count(bitmap));
975 // check if the corresponding chunk is all clear

982 mi_bchunk_set(&bitmap->chunkmap, chunk_idx, NULL);

984 }

1021// Set a sequence of `n` bits in the bitmap (and can cross chunks). Not atomic so only use if local to a thread.
1022static void mi_bchunks_unsafe_setN(mi_bchunk_t* chunks, mi_bchunkmap_t* cmap, size_t idx, size_t n) {
1023 mi_assert_internal(n>0);
1024
1025 // start chunk and index
1026 size_t chunk_idx = idx / MI_BCHUNK_BITS;
1027 const size_t cidx = idx % MI_BCHUNK_BITS;
1028 const size_t ccount = _mi_divide_up(n, MI_BCHUNK_BITS);
1029
1030 // first update the chunkmap
1031 mi_bchunk_setN(cmap, chunk_idx, ccount, NULL);
1032
1033 // first chunk
1034 size_t m = MI_BCHUNK_BITS - cidx;
1035 if (m > n) { m = n; }
1036 mi_bchunk_setN(&chunks[chunk_idx], cidx, m, NULL);
1037
1038 // n can be large so use memset for efficiency for all in-between chunks
1039 chunk_idx++;
1040 n -= m;
1041 const size_t mid_chunks = n / MI_BCHUNK_BITS;
1042 if (mid_chunks > 0) {
1043 _mi_memset(&chunks[chunk_idx], ~0, mid_chunks * MI_BCHUNK_SIZE);
1044 chunk_idx += mid_chunks;
1045 n -= (mid_chunks * MI_BCHUNK_BITS);
1046 }
1047
1048 // last chunk
1049 if (n > 0) {
1050 mi_assert_internal(n < MI_BCHUNK_BITS);
1051 mi_assert_internal(chunk_idx < MI_BCHUNK_FIELDS);
1052 mi_bchunk_setN(&chunks[chunk_idx], 0, n, NULL);
1053 }
1054}

1135/* --------------------------------------------------------------------------------
1136 Iterate through a bfield
1137-------------------------------------------------------------------------------- */
1138
1139// Cycle iteration through a bitfield. This is used to space out threads
1140// so there is less chance of contention. When searching for a free page we
1141// like to first search only the accessed part (so we reuse better). This
1142// high point is called the `cycle`.
1143//
1144// We then iterate through the bitfield as:
1145// first: [start, cycle>
1146// then : [0, start>
1147// then : [cycle, MI_BFIELD_BITS>
1148//
1149// The start is determined usually as `tseq % cycle` to have each thread
1150// start at a different spot.
1151// - We use `popcount` to improve branch prediction (maybe not needed? can we simplify?)
1152// - The `cycle_mask` is the part `[start, cycle>`.
1153#define mi_bfield_iterate(bfield,start,cycle,name_idx,SUF) { \
1154 mi_assert_internal(start <= cycle); \
1155 mi_assert_internal(start < MI_BFIELD_BITS); \
1156 mi_assert_internal(cycle <= MI_BFIELD_BITS); \
1157 mi_bfield_t _cycle_mask##SUF = mi_bfield_mask(cycle - start, start); \
1158 size_t _bcount##SUF = mi_bfield_popcount(bfield); \
1159 mi_bfield_t _b##SUF = bfield & _cycle_mask##SUF; /* process [start, cycle> first*/\
1160 while(_bcount##SUF > 0) { \
1161 _bcount##SUF--;\
1162 if (_b##SUF==0) { _b##SUF = bfield & ~_cycle_mask##SUF; } /* process [0,start> + [cycle, MI_BFIELD_BITS> next */ \
1163 /* size_t name_idx; */ \
1164 bool _found##SUF = mi_bfield_find_least_bit(_b##SUF,&name_idx); \
1165 mi_assert_internal(_found##SUF); MI_UNUSED(_found##SUF); \
1166 { \
1167
1168#define mi_bfield_iterate_end(SUF) \
1169 } \
1170 _b##SUF = mi_bfield_clear_least_bit(_b##SUF); \
1171 } \
1172}
1173
1174#define mi_bfield_cycle_iterate(bfield,tseq,cycle,name_idx,SUF) { \
1175 const size_t _start##SUF = (uint32_t)(tseq) % (uint32_t)(cycle); /* or: 0 to always search from the start? */\
1176 mi_bfield_iterate(bfield,_start##SUF,cycle,name_idx,SUF)
1177
1178#define mi_bfield_cycle_iterate_end(SUF) \
1179 mi_bfield_iterate_end(SUF); }

1182/* --------------------------------------------------------------------------------
1183 mi_bitmap_find
1184 (used to find free pages)
1185-------------------------------------------------------------------------------- */
1186
1187typedef bool (mi_bitmap_visit_fun_t)(mi_bitmap_t* bitmap, size_t chunk_idx, size_t n, size_t* idx, void* arg1, void* arg2);
1188
1189// Go through the bitmap and for every sequence of `n` set bits, call the visitor function.
1190// If it returns `true` stop the search.
1191static inline bool mi_bitmap_find(mi_bitmap_t* bitmap, size_t tseq, size_t n, size_t* pidx, mi_bitmap_visit_fun_t* on_find, void* arg1, void* arg2)
1192{
1193 const size_t chunkmap_max = _mi_divide_up(mi_bitmap_chunk_count(bitmap), MI_BFIELD_BITS);
1194 for (size_t i = 0; i < chunkmap_max; i++) {
1195 // and for each chunkmap entry we iterate over its bits to find the chunks
1196 const mi_bfield_t cmap_entry = mi_atomic_load_relaxed(&bitmap->chunkmap.bfields[i]);
1197 size_t hi;
1198 if (mi_bfield_find_highest_bit(cmap_entry, &hi)) {
1199 size_t eidx = 0;
1200 mi_bfield_cycle_iterate(cmap_entry, tseq%8, hi+1, eidx, Y) // reduce the tseq to 8 bins to reduce using extra memory (see `mstress`)
1201 {
1202 mi_assert_internal(eidx <= MI_BFIELD_BITS);
1203 const size_t chunk_idx = i*MI_BFIELD_BITS + eidx;
1204 mi_assert_internal(chunk_idx < mi_bitmap_chunk_count(bitmap));
1205 if ((*on_find)(bitmap, chunk_idx, n, pidx, arg1, arg2)) {
1206 return true;
1207 }
1208 }
1209 mi_bfield_cycle_iterate_end(Y);
1210 }
1211 }
1212 return false;
1213}

1216/* --------------------------------------------------------------------------------
1217 Bitmap: try_find_and_claim -- used to allocate abandoned pages
1218 note: the compiler will fully inline the indirect function call
1219-------------------------------------------------------------------------------- */
1220
1221typedef struct mi_claim_fun_data_s {
1222 mi_arena_t* arena;
1223 mi_heaptag_t heap_tag;
1224} mi_claim_fun_data_t;
1225
1226static bool mi_bitmap_try_find_and_claim_visit(mi_bitmap_t* bitmap, size_t chunk_idx, size_t n, size_t* pidx, void* arg1, void* arg2)
1227{
1228 mi_assert_internal(n==1); MI_UNUSED(n);
1229 mi_claim_fun_t* claim_fun = (mi_claim_fun_t*)arg1;
1230 mi_claim_fun_data_t* claim_data = (mi_claim_fun_data_t*)arg2;
1231 size_t cidx;
1232 if mi_likely(mi_bchunk_try_find_and_clear(&bitmap->chunks[chunk_idx], &cidx)) {
1233 const size_t slice_index = (chunk_idx * MI_BCHUNK_BITS) + cidx;
1234 mi_assert_internal(slice_index < mi_bitmap_max_bits(bitmap));
1235 bool keep_set = true;
1236 if ((*claim_fun)(slice_index, claim_data->arena, claim_data->heap_tag, &keep_set)) {
1237 // success!
1238 mi_assert_internal(!keep_set);
1239 *pidx = slice_index;
1240 return true;
1241 }
1242 else {
1243 // failed to claim it, set abandoned mapping again (unless the page was freed)
1244 if (keep_set) {
1245 const bool wasclear = mi_bchunk_set(&bitmap->chunks[chunk_idx], cidx, NULL);
1246 mi_assert_internal(wasclear); MI_UNUSED(wasclear);
1247 }
1248 }
1249 }
1250 else {
1251 // we may find that all are cleared only on a second iteration but that is ok as
1252 // the chunkmap is a conservative approximation.
1253 mi_bitmap_chunkmap_try_clear(bitmap, chunk_idx);
1254 }
1255 return false;
1256}

1298// Visit all set bits in a bitmap.
1299// todo: optimize further? maybe use avx512 to directly get all indices using a mask_compressstore?
1300bool _mi_bitmap_forall_set(mi_bitmap_t* bitmap, mi_forall_set_fun_t* visit, mi_arena_t* arena, void* arg) {
1301 // for all chunkmap entries
1302 const size_t chunkmap_max = _mi_divide_up(mi_bitmap_chunk_count(bitmap), MI_BFIELD_BITS);
1303 for(size_t i = 0; i < chunkmap_max; i++) {
1304 mi_bfield_t cmap_entry = mi_atomic_load_relaxed(&bitmap->chunkmap.bfields[i]);
1305 size_t cmap_idx;
1306 // for each chunk (corresponding to a set bit in a chunkmap entry)
1307 while (mi_bfield_foreach_bit(&cmap_entry, &cmap_idx)) {
1308 const size_t chunk_idx = i*MI_BFIELD_BITS + cmap_idx;
1309 // for each chunk field
1310 mi_bchunk_t* const chunk = &bitmap->chunks[chunk_idx];
1311 for (size_t j = 0; j < MI_BCHUNK_FIELDS; j++) {
1312 const size_t base_idx = (chunk_idx*MI_BCHUNK_BITS) + (j*MI_BFIELD_BITS);
1313 mi_bfield_t b = mi_atomic_load_relaxed(&chunk->bfields[j]);
1314 size_t bidx;
1315 while (mi_bfield_foreach_bit(&b, &bidx)) {
1316 const size_t idx = base_idx + bidx;
1317 if (!visit(idx, 1, arena, arg)) return false;
1318 }
1319 }
1320 }
1321 }
1322 return true;
1323}

1325// Visit all set bits in a bitmap but try to return ranges (within bfields) if possible.
1326// Also clear those ranges atomically.
1327// Used by purging to purge larger ranges when possible
1328// todo: optimize further? maybe use avx512 to directly get all indices using a mask_compressstore?
1329bool _mi_bitmap_forall_setc_ranges(mi_bitmap_t* bitmap, mi_forall_set_fun_t* visit, mi_arena_t* arena, void* arg) {
1330 // for all chunkmap entries
1331 const size_t chunkmap_max = _mi_divide_up(mi_bitmap_chunk_count(bitmap), MI_BFIELD_BITS);
1332 for (size_t i = 0; i < chunkmap_max; i++) {
1333 mi_bfield_t cmap_entry = mi_atomic_load_relaxed(&bitmap->chunkmap.bfields[i]);
1334 size_t cmap_idx;
1335 // for each chunk (corresponding to a set bit in a chunkmap entry)
1336 while (mi_bfield_foreach_bit(&cmap_entry, &cmap_idx)) {
1337 const size_t chunk_idx = i*MI_BFIELD_BITS + cmap_idx;
1338 // for each chunk field
1339 mi_bchunk_t* const chunk = &bitmap->chunks[chunk_idx];
1340 for (size_t j = 0; j < MI_BCHUNK_FIELDS; j++) {
1341 const size_t base_idx = (chunk_idx*MI_BCHUNK_BITS) + (j*MI_BFIELD_BITS);
1342 mi_bfield_t b = mi_atomic_exchange_acq_rel(&chunk->bfields[j], 0); // can be relaxed?
1343 #if MI_DEBUG > 1
1344 const size_t bpopcount = mi_popcount(b);
1345 size_t rngcount = 0;
1346 #endif
1347 size_t bidx;
1348 while (mi_bfield_find_least_bit(b, &bidx)) {
1349 const size_t rng = mi_ctz(~(b>>bidx)); // all the set bits from bidx
1350 #if MI_DEBUG > 1
1351 rngcount += rng;
1352 #endif
1353 mi_assert_internal(rng>=1 && rng<=MI_BFIELD_BITS);
1354 const size_t idx = base_idx + bidx;
1355 mi_assert_internal((idx % MI_BFIELD_BITS) + rng <= MI_BFIELD_BITS);
1356 mi_assert_internal((idx / MI_BCHUNK_BITS) < mi_bitmap_chunk_count(bitmap));
1357 if (!visit(idx, rng, arena, arg)) return false;
1358 // clear rng bits in b
1359 b = b & ~mi_bfield_mask(rng, bidx);
1360 }
1361 mi_assert_internal(rngcount == bpopcount);
1362 }
1363 }
1364 }
1365 return true;

1538 (used to find free pages)
1539-------------------------------------------------------------------------------- */
1540
1541typedef bool (mi_bchunk_try_find_and_clear_fun_t)(mi_bchunk_t* chunk, size_t n, size_t* idx);
1542
1543// Go through the bbitmap and for every sequence of `n` set bits, call the visitor function.
1544// If it returns `true` stop the search.
1545//
1546// This is used for finding free blocks and it is important to be efficient (with 2-level bitscan)
1547// but also reduce fragmentation (through size bins).
1548static inline bool mi_bbitmap_try_find_and_clear_generic(mi_bbitmap_t* bbitmap, size_t tseq, size_t n, size_t* pidx, mi_bchunk_try_find_and_clear_fun_t* on_find)
1549{
1550 // we space out threads to reduce contention
1551 const size_t cmap_max_count = _mi_divide_up(mi_bbitmap_chunk_count(bbitmap),MI_BFIELD_BITS);
1552 const size_t chunk_acc = mi_atomic_load_relaxed(&bbitmap->chunk_max_accessed);
1553 const size_t cmap_acc = chunk_acc / MI_BFIELD_BITS;
1554 const size_t cmap_acc_bits = 1 + (chunk_acc % MI_BFIELD_BITS);
1555
1556 // create a mask over the chunkmap entries to iterate over them efficiently
1557 mi_assert_internal(MI_BFIELD_BITS >= MI_BCHUNK_FIELDS);
1558 const mi_bfield_t cmap_mask = mi_bfield_mask(cmap_max_count,0);
1559 const size_t cmap_cycle = cmap_acc+1;
1560 const mi_bbin_t bbin = mi_bbin_of(n);
1561 // visit each cmap entry
1562 size_t cmap_idx = 0;
1563 mi_bfield_cycle_iterate(cmap_mask, tseq, cmap_cycle, cmap_idx, X)
1564 {
1565 // and for each chunkmap entry we iterate over its bits to find the chunks
1566 const mi_bfield_t cmap_entry = mi_atomic_load_relaxed(&bbitmap->chunkmap.bfields[cmap_idx]);
1567 const size_t cmap_entry_cycle = (cmap_idx != cmap_acc ? MI_BFIELD_BITS : cmap_acc_bits);
1568 if (cmap_entry == 0) continue;
1569
1570 // get size bin masks
1571 mi_bfield_t cmap_bins[MI_BBIN_COUNT] = { 0 };
1572 cmap_bins[MI_BBIN_NONE] = cmap_entry;
1573 for (mi_bbin_t ibin = MI_BBIN_SMALL; ibin < MI_BBIN_NONE; ibin = mi_bbin_inc(ibin)) {
1574 const mi_bfield_t cmap_bin = mi_atomic_load_relaxed(&bbitmap->chunkmap_bins[ibin].bfields[cmap_idx]);
1575 cmap_bins[ibin] = cmap_bin & cmap_entry;
1576 cmap_bins[MI_BBIN_NONE] &= ~cmap_bin; // clear bits that are in an assigned size bin
1577 }
1578
1579 // consider only chunks for a particular size bin at a time
1580 // this picks the best bin only within a cmap entry (~ 1GiB address space), but avoids multiple
1581 // iterations through all entries.
1582 mi_assert_internal(bbin < MI_BBIN_NONE);
1583 for (mi_bbin_t ibin = MI_BBIN_SMALL; ibin <= MI_BBIN_NONE;
1584 // skip from bbin to NONE (so, say, a SMALL will never be placed in a OTHER, MEDIUM, or LARGE chunk to reduce fragmentation)
1585 ibin = (ibin == bbin ? MI_BBIN_NONE : mi_bbin_inc(ibin)))
1586 {
1587 mi_assert_internal(ibin < MI_BBIN_COUNT);
1588 const mi_bfield_t cmap_bin = cmap_bins[ibin];
1589 size_t eidx = 0;
1590 mi_bfield_cycle_iterate(cmap_bin, tseq, cmap_entry_cycle, eidx, Y)
1591 {
1592 // assertion doesn't quite hold as the max_accessed may be out-of-date
1593 // mi_assert_internal(cmap_entry_cycle > eidx || ibin == MI_BBIN_NONE);
1594
1595 // get the chunk
1596 const size_t chunk_idx = cmap_idx*MI_BFIELD_BITS + eidx;
1597 mi_bchunk_t* chunk = &bbitmap->chunks[chunk_idx];
1598
1599 size_t cidx;
1600 if ((*on_find)(chunk, n, &cidx)) {
1601 if (cidx==0 && ibin == MI_BBIN_NONE) { // only the first block determines the size bin
1602 // this chunk is now reserved for the `bbin` size class
1603 mi_bbitmap_set_chunk_bin(bbitmap, chunk_idx, bbin);
1604 }
1605 *pidx = (chunk_idx * MI_BCHUNK_BITS) + cidx;
1606 mi_assert_internal(*pidx + n <= mi_bbitmap_max_bits(bbitmap));
1607 return true;
1608 }
1609 else {
1610 // todo: should _on_find_ return a boolen if there is a chance all are clear to avoid calling `try_clear?`
1611 // we may find that all are cleared only on a second iteration but that is ok as the chunkmap is a conservative approximation.
1612 mi_bbitmap_chunkmap_try_clear(bbitmap, chunk_idx);
1613 }
1614 }
1615 mi_bfield_cycle_iterate_end(Y);
1616 }
1617 }
1618 mi_bfield_cycle_iterate_end(X);
1619 return false;
1620}
