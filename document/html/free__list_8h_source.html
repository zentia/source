98 void put( node * pNode )
99 {
100 // We know that the should-be-on-freelist bit is 0 at this point, so it's safe to
101 // set it using a fetch_add
102 if ( pNode->m_freeListRefs.fetch_add( c_ShouldBeOnFreeList, atomics::memory_order_release ) == 0 ) {
103 // Oh look! We were the last ones referencing this node, and we know
104 // we want to add it to the free list, so let's do it!
105 add_knowing_refcount_is_zero( pNode );
106 }
107 }
108
110 node * get()
111 {
112 auto head = m_Head.load( atomics::memory_order_acquire );
113 while ( head != nullptr ) {
114 auto prevHead = head;
115 auto refs = head->m_freeListRefs.load( atomics::memory_order_relaxed );
116
117 if ( cds_unlikely( (refs & c_RefsMask) == 0 || !head->m_freeListRefs.compare_exchange_strong( refs, refs + 1,
118 atomics::memory_order_acquire, atomics::memory_order_relaxed )))
119 {
120 head = m_Head.load( atomics::memory_order_acquire );
121 continue;
122 }
123
124 // Good, reference count has been incremented (it wasn't at zero), which means
125 // we can read the next and not worry about it changing between now and the time
126 // we do the CAS
127 node * next = head->m_freeListNext.load( atomics::memory_order_relaxed );
128 if ( cds_likely( m_Head.compare_exchange_strong( head, next, atomics::memory_order_acquire, atomics::memory_order_relaxed ))) {
129 // Yay, got the node. This means it was on the list, which means
130 // shouldBeOnFreeList must be false no matter the refcount (because
131 // nobody else knows it's been taken off yet, it can't have been put back on).
132 assert( (head->m_freeListRefs.load( atomics::memory_order_relaxed ) & c_ShouldBeOnFreeList) == 0 );
133
134 // Decrease refcount twice, once for our ref, and once for the list's ref
135 head->m_freeListRefs.fetch_sub( 2, atomics::memory_order_relaxed );
136
137 return head;
138 }
139
140 // OK, the head must have changed on us, but we still need to decrease the refcount we
141 // increased
142 refs = prevHead->m_freeListRefs.fetch_sub( 1, atomics::memory_order_acq_rel );
143 if ( refs == c_ShouldBeOnFreeList + 1 )
144 add_knowing_refcount_is_zero( prevHead );
145 }
146
147 return nullptr;
148 }

169 template <typename Disposer>
170 void clear( Disposer disp )
171 {
172 node * head = m_Head.load( atomics::memory_order_relaxed );
173 m_Head.store( nullptr, atomics::memory_order_relaxed );
174 while ( head ) {
175 node * next = head->m_freeListNext.load( atomics::memory_order_relaxed );
176 disp( head );
177 head = next;
178 }
179 }

181 private:
182 //@cond
183 void add_knowing_refcount_is_zero( node * pNode )
184 {
185 // Since the refcount is zero, and nobody can increase it once it's zero (except us, and we
186 // run only one copy of this method per node at a time, i.e. the single thread case), then we
187 // know we can safely change the next pointer of the node; however, once the refcount is back
188 // above zero, then other threads could increase it (happens under heavy contention, when the
189 // refcount goes to zero in between a load and a refcount increment of a node in try_get, then
190 // back up to something non-zero, then the refcount increment is done by the other thread) --
191 // so, if the CAS to add the node to the actual list fails, decrease the refcount and leave
192 // the add operation to the next thread who puts the refcount back at zero (which could be us,
193 // hence the loop).
194 node * head = m_Head.load( atomics::memory_order_relaxed );
195 while ( true ) {
196 pNode->m_freeListNext.store( head, atomics::memory_order_relaxed );
197 pNode->m_freeListRefs.store( 1, atomics::memory_order_release );
198 if ( cds_unlikely( !m_Head.compare_exchange_strong( head, pNode, atomics::memory_order_release, atomics::memory_order_relaxed ))) {
199 // Hmm, the add failed, but we can only try again when the refcount goes back to zero
200 if ( pNode->m_freeListRefs.fetch_add( c_ShouldBeOnFreeList - 1, atomics::memory_order_release ) == 1 )
201 continue;
202 }
203 return;
204 }
205 }
206 //@endcond
207
208 private:
209 //@cond
210 static constexpr uint32_t const c_RefsMask = 0x7FFFFFFF;
211 static constexpr uint32_t const c_ShouldBeOnFreeList = 0x80000000;
212
213 // Implemented like a stack, but where node order doesn't matter (nodes are
214 // inserted out of order under contention)
215 atomics::atomic<node *> m_Head;
216 //@endcond
217 };
