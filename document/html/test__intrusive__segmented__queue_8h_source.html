63 template <typename Queue, typename Data>
64 void test( Queue& q, Data& val )
65 {
66 typedef typename Queue::value_type value_type;
67 val.resize( 100 );
68 for ( size_t i = 0; i < val.size(); ++i )
69 val[i].nValue = static_cast<int>( i );
70
71 ASSERT_TRUE( q.empty());
72 ASSERT_CONTAINER_SIZE( q, 0u );
73
74 // push/enqueue
75 for ( size_t i = 0; i < val.size(); ++i ) {
76 if ( i & 1 ) {
77 ASSERT_TRUE( q.push( val[i] ));
78 }
79 else {
80 ASSERT_TRUE( q.enqueue( val[i] ));
81 }
82
83 ASSERT_CONTAINER_SIZE( q, i + 1 );
84 }
85 EXPECT_TRUE( !q.empty());
86
87 // pop/dequeue
88 size_t nCount = 0;
89 while ( !q.empty()) {
90 value_type * pVal;
91 if ( nCount & 1 )
92 pVal = q.pop();
93 else
94 pVal = q.dequeue();
95
96 ASSERT_TRUE( pVal != nullptr );
97
98 int nSegment = int( nCount / q.quasi_factor());
99 int nMin = nSegment * int( q.quasi_factor());
100 int nMax = nMin + int( q.quasi_factor()) - 1;
101 EXPECT_TRUE( nMin <= pVal->nValue && pVal->nValue <= nMax ) << nMin << " <= " << pVal->nValue << " <= " << nMax;
102
103 ++nCount;
104 EXPECT_CONTAINER_SIZE( q, val.size() - nCount );
105 }
106 EXPECT_EQ( nCount, val.size());
107 EXPECT_TRUE( q.empty());
108 EXPECT_CONTAINER_SIZE( q, 0u );
109
110 // pop from empty queue
111 ASSERT_TRUE( q.pop() == nullptr );
112 EXPECT_TRUE( q.empty());
113 EXPECT_CONTAINER_SIZE( q, 0u );
114
115 // check that Disposer has not been called
116 Queue::gc::force_dispose();
117 for ( size_t i = 0; i < val.size(); ++i ) {
118 EXPECT_EQ( val[i].nDisposeCount, 0u );
119 EXPECT_EQ( val[i].nDispose2Count, 0u );
120 }
121
122 // clear
123 for ( size_t i = 0; i < val.size(); ++i )
124 EXPECT_TRUE( q.push( val[i] ));
125 EXPECT_CONTAINER_SIZE( q, val.size());
126 EXPECT_TRUE( !q.empty());
127
128 q.clear();
129 EXPECT_CONTAINER_SIZE( q, 0u );
130 EXPECT_TRUE( q.empty());
131
132 // check if Disposer has been called
133 Queue::gc::force_dispose();
134 for ( size_t i = 0; i < val.size(); ++i ) {
135 EXPECT_EQ( val[i].nDisposeCount, 1u );
136 EXPECT_EQ( val[i].nDispose2Count, 0u );
137 }
138
139 // clear_with
140 for ( size_t i = 0; i < val.size(); ++i )
141 EXPECT_TRUE( q.push( val[i] ));
142 EXPECT_CONTAINER_SIZE( q, val.size());
143 EXPECT_TRUE( !q.empty());
144
145 q.clear_with( Disposer2());
146 EXPECT_CONTAINER_SIZE( q, 0u );
147 EXPECT_TRUE( q.empty());
148
149 // check if Disposer has been called
150 Queue::gc::force_dispose();
151 for ( size_t i = 0; i < val.size(); ++i ) {
152 EXPECT_EQ( val[i].nDisposeCount, 1u );
153 EXPECT_EQ( val[i].nDispose2Count, 1u );
154 }
155
156 // check clear on destruct
157 for ( size_t i = 0; i < val.size(); ++i )
158 EXPECT_TRUE( q.push( val[i] ));
159 EXPECT_CONTAINER_SIZE( q, val.size());
160 EXPECT_TRUE( !q.empty());
161 }
162 };
