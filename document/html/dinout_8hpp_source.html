79 virtual inline bool push(void * ptr) {
80 if (skipdnode || !P) return ff_node::push(ptr);
81
82 // gets the peers involved in one single communication
83 const int peers=com.getDescriptor()->getPeers();
84 if (ptr == (void*)FF_EOS) {
85 //cerr << "DNODE prepare to send FF_DEOS to " << peers <<" peers\n";
86 for(int i=0;i<peers;++i) {
87 msg_t msg;
88 msg.init(FF_DEOS,msg_t::HEADER_LENGHT);
89 //cerr << "DNODE sends FF_DEOS to " << i <<"\n";
90 if (!com.put(msg,i)) return false;
91 }
92 return true;
93 }
94
95 if (CommImpl::MULTIPUT) {
96 svector<iovec> v;
97 for(int i=0;i<peers;++i) {
98 v.clear();
99 callbackArg.resize(0);
100 prepare(v, ptr, i);
101
102 msg_t hdr(new uint32_t(v.size()), msg_t::HEADER_LENGHT, freeHdr);
103 com.putmore(hdr,i);
104 callbackArg.resize(v.size());
105 for(size_t j=0;j<v.size()-1;++j) {
106 msg_t msg(v[j].iov_base, v[j].iov_len,freeMsg,callbackArg[j]);
107 com.putmore(msg,i);
108 }
109 msg_t msg(v[v.size()-1].iov_base, v[v.size()-1].iov_len,freeMsg,callbackArg[v.size()-1]);
110 if (!com.put(msg,i)) return false;
111 }
112 } else {
113 svector<iovec> v;
114 callbackArg.resize(0);
115 prepare(v, ptr);
116
117 msg_t hdr(new uint32_t(v.size()), msg_t::HEADER_LENGHT, freeHdr);
118 com.putmore(hdr);
119 callbackArg.resize(v.size());
120 for(size_t j=0;j<v.size()-1;++j) {
121 msg_t msg(v[j].iov_base, v[j].iov_len,freeMsg,callbackArg[j]);
122 com.putmore(msg);
123 }
124 msg_t msg(v[v.size()-1].iov_base, v[v.size()-1].iov_len,freeMsg,callbackArg[v.size()-1]);
125 if (!com.put(msg)) return false;
126 }
127 return true;
128 }

131 virtual inline bool pop(void ** ptr) {
132 if (skipdnode || P) return ff_node::pop(ptr);
133
134 // gets the peers involved in one single communication
135 const int sendingPeers=com.getToWait();
136
137 svector<msg_t*>* v[sendingPeers];
138 for(int i=0;i<sendingPeers;++i) {
139 msg_t hdr;
140 int sender=-1;
141 if (!com.gethdr(hdr, sender)) {
142 error("dnode:pop: ERROR: receiving header from peer");
143 return false;
144 }
145 if (isEos(static_cast<char *>(hdr.getData()))) {
146 //std::cerr << "DNODE gets DEOS\n";
147 if (++neos==com.getDescriptor()->getPeers()) {
148 com.done();
149 *ptr = (void*)FF_EOS;
150 neos=0;
151 return true;
152 }
153 if (sendingPeers==1) i=-1; // reset for index
154 continue;
155 }
156 uint32_t len = *static_cast<uint32_t*>(hdr.getData());
157 int ventry = (sendingPeers==1)?0:sender;
158 prepare(v[ventry], len, sender);
159 assert(v[ventry]->size() == len);
160
161 for(size_t j=0;j<len;++j)
162 if (!com.get(*(v[ventry]->operator[](j)),sender)) {
163 error("dnode:pop: ERROR: receiving data from peer");
164 return false;
165 }
166 }
167 com.done();
168
169 unmarshalling(v, sendingPeers, *ptr);
170 return true;
171 }

189 v.push_back(iov);
190 setCallbackArg(NULL);

201 /*
202 * Used to give to the run-time a pool of messages on which
203 * input message frames will be received
204 * \param len the number of input messages expected
205 * \param sender the message sender
206 * \param v vector containing the pool of messages
207 */
208 virtual void prepare(svector<msg_t*>*& v, size_t len, const int sender=-1) {
209 svector<msg_t*> * v2 = new svector<msg_t*>(len);
210 assert(v2);
211 for(size_t i=0;i<len;++i) {
212 msg_t * m = new msg_t;
213 assert(m);
214 v2->push_back(m);
215 }
216 v = v2;
217 }
218
219 /*
220 * This method is called once, when all frames composing the message have been received
221 * by the run-time. Within that method, it is possible to convert or re-arrange
222 * all the frames back to their original data or object layout.
223 */
224 virtual void unmarshalling(svector<msg_t*>* const v[], const int vlen, void *& task) {
225 assert(vlen==1 && v[0]->size()==1);
226 task = v[0]->operator[](0)->getData();
227 delete v[0];
228 }

233 * producer.
