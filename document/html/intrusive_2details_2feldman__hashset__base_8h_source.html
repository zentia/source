342 struct metrics {
343 size_t head_node_size; // power-of-two
344 size_t head_node_size_log; // log2( head_node_size )
345 size_t array_node_size; // power-of-two
346 size_t array_node_size_log;// log2( array_node_size )
347
348 static metrics make(size_t head_bits, size_t array_bits, size_t hash_size )
349 {
350 size_t const hash_bits = hash_size * 8;
351
352 if (array_bits < 2)
353 array_bits = 2;
354 if (head_bits < 4)
355 head_bits = 4;
356 if (head_bits > hash_bits)
357 head_bits = hash_bits;
358 if ((hash_bits - head_bits) % array_bits != 0)
359 head_bits += (hash_bits - head_bits) % array_bits;
360
361 assert((hash_bits - head_bits) % array_bits == 0);
362
363 metrics m;
364 m.head_node_size_log = head_bits;
365 m.head_node_size = size_t(1) << head_bits;
366 m.array_node_size_log = array_bits;
367 m.array_node_size = size_t(1) << array_bits;
368 return m;
369 }
370 };

482 node_ptr traverse(traverse_data& pos)
483 {
484 back_off bkoff;
485 while (true) {
486 node_ptr slot = pos.pArr->nodes[pos.nSlot].load(memory_model::memory_order_acquire);
487 if ( slot.bits() == flag_array_node ) {
488 // array node, go down the tree
489 assert(slot.ptr() != nullptr);
490 assert( !pos.splitter.eos());
491 pos.nSlot = pos.splitter.cut( static_cast<unsigned>( metrics().array_node_size_log ));
492 assert( static_cast<size_t>( pos.nSlot ) < metrics().array_node_size );
493 pos.pArr = to_array(slot.ptr());
494 ++pos.nHeight;
495 }
496 else if (slot.bits() == flag_array_converting) {
497 // the slot is converting to array node right now
498 bkoff();
499 stats().onSlotConverting();
500 }
501 else {
502 // data node
503 assert(slot.bits() == 0);
504 return slot;
505 }
506 } // while

604 void gather_level_statistics(std::vector<feldman_hashset::level_statistics>& stat, size_t nLevel, array_node * pArr, size_t nSize) const
605 {
606 if (stat.size() <= nLevel) {
607 stat.resize(nLevel + 1);
608 stat[nLevel].node_capacity = nSize;
609 }
610
611 ++stat[nLevel].array_node_count;
612 for (atomic_node_ptr * p = pArr->nodes, *pLast = p + nSize; p != pLast; ++p) {
613 node_ptr slot = p->load(memory_model::memory_order_relaxed);
614 if (slot.bits()) {
615 ++stat[nLevel].array_cell_count;
616 if (slot.bits() == flag_array_node)
617 gather_level_statistics(stat, nLevel + 1, to_array(slot.ptr()), array_node_size());
618 }
619 else if (slot.ptr())
620 ++stat[nLevel].data_cell_count;
621 else
622 ++stat[nLevel].empty_cell_count;
623 }

626 bool expand_slot( traverse_data& pos, node_ptr current)
627 {
628 assert( !pos.splitter.eos());
629 return expand_slot( pos.pArr, pos.nSlot, current, pos.splitter.bit_offset());
630 }

632 private:
633 bool expand_slot(array_node * pParent, size_t idxParent, node_ptr current, size_t nOffset)
634 {
635 assert(current.bits() == 0);
636 assert(current.ptr());
637
638 array_node * pArr = alloc_array_node(pParent, idxParent);
639
640 node_ptr cur(current.ptr());
641 atomic_node_ptr& slot = pParent->nodes[idxParent];
642 if (!slot.compare_exchange_strong(cur, cur | flag_array_converting, memory_model::memory_order_release, atomics::memory_order_relaxed))
643 {
644 stats().onExpandNodeFailed();
645 free_array_node( pArr, array_node_size());
646 return false;
647 }
648
649 typename hash_splitter::uint_type idx = hash_splitter( hash_accessor()(*current.ptr()), nOffset ).cut(
650 static_cast<unsigned>( m_Metrics.array_node_size_log ));
651 pArr->nodes[idx].store(current, memory_model::memory_order_release);
652
653 cur = cur | flag_array_converting;
654 CDS_VERIFY(
655 slot.compare_exchange_strong(cur, node_ptr(to_node(pArr), flag_array_node), memory_model::memory_order_release, atomics::memory_order_relaxed)
656 );
657
658 stats().onExpandNodeSuccess();
659 stats().onArrayNodeCreated();
660 return true;
661 }
