107static inline unsigned int fnv_hash_function( void *key, int len ) {
108 unsigned char *p = (unsigned char*)key;
109 unsigned int h = 2166136261u;
110 int i;
111 for ( i = 0; i < len; i++ )
112 h = ( h * 16777619 ) ^ p[i];
113 return h;
114}

120static inline unsigned int address_hash_function(void *address) {
121 int len = sizeof(void *);
122 unsigned int hashval = fnv_hash_function( &address, len );
123 return hashval;

160// Parallelism priority
161struct CompareTask_Par {
162 // Returns true if t1 is earlier than t2
163 bool operator()(hash_task_t * &t1, hash_task_t* &t2) {
164 if (t1->unblock_numb < t2->unblock_numb) return true;
165 return false;
166 }
167};
168// LIFO priority
169struct CompareTask_LIFO {
170 bool operator()(hash_task_t * &t1, hash_task_t* &t2) {
171 if (t1->id < t2->id) return true;
172 return false;
173 }
174};
175// FIFO priority
176struct CompareTask_FIFO {
177 bool operator()(hash_task_t * &t1, hash_task_t* &t2) {
178 if (t1->id > t2->id) return true;
179 return false;
180 }
181};

223 inline hash_task_t* createTask(unsigned long id, task_status_t status, base_f_t *wtask) {
224 hash_task_t *t=(hash_task_t*)TASK_MALLOC(sizeof(hash_task_t));

227 t->unblock_numb=0; t->wtask=wtask; t->is_dummy=false;
228 t->unblock_task_ids=(unsigned long *)TASK_MALLOC(UNBLOCK_SIZE*sizeof(unsigned long));

233 inline hash_task_t *insertTask(task_f_t *const msg,
234 hash_task_t *waittask=nullptr) {
235 unsigned long act_id=task_id++;
236 hash_task_t *act_task=createTask(act_id,NOT_READY,msg->wtask);
237 icl_hash_insert(task_set, &act_task->id, act_task);
238
239 for (auto p: msg->P) {
240 auto d = p.tag;
241 auto dir = p.dir;
242 if(dir==INPUT) {
243 //hash_task_t * t=(hash_task_t *)icl_hash_find(address_set,(void*)d);
244 unsigned long t_id=(unsigned long)icl_hash_find(address_set,(void *)d);
245 hash_task_t * t=NULL;
246 if(t_id) //t_id==0 if the hash table does not contains info for d
247 t=(hash_task_t *)icl_hash_find(task_set,&t_id);
248
249
250
251 if(t==NULL) { // no writer for this tag
252 hash_task_t *dummy=createTask(task_id,DONE,NULL);
253 dummy->is_dummy=true;
254 // the dummy task uses current data
255 icl_hash_insert(address_set,(void*)d,(void*)(dummy->id));
256 // the dummy task unblocks the current data
257 dummy->unblock_task_ids[dummy->unblock_numb]=act_id;
258 dummy->unblock_numb++;
259 dummy->num_out++;
260 icl_hash_insert(task_set,&dummy->id,dummy);
261 task_id++;
262 } else {
263 if(t->unblock_numb == t->unblock_act_numb) {
264 t->unblock_act_numb+=UNBLOCK_SIZE;
265 t->unblock_task_ids=(unsigned long *)TASK_REALLOC(t->unblock_task_ids,t->unblock_act_numb*sizeof(unsigned long));
266 }
267 t->unblock_task_ids[t->unblock_numb]=act_id;
268 t->unblock_numb++;
269 if(t->status!=DONE) act_task->remaining_dep++;
270 }
271 } else
272 if (dir==OUTPUT) {
273 hash_task_t * t=NULL;
274 unsigned long t_id = (unsigned long)icl_hash_find(address_set,(void*)d);
275
276 if (t_id) t=(hash_task_t *)icl_hash_find(task_set,&t_id);
277
278 // the task has been already written
279 if(t != NULL) {
280 if (t->unblock_numb>0) {
281 // for each unblocked task, checks if that task unblock also act_task (WAR dependency)
282 for(long ii=0;ii<t->unblock_numb;ii++) {
283 hash_task_t* t2=(hash_task_t*)icl_hash_find(task_set,&t->unblock_task_ids[ii]);
284 if(t2!=NULL && t2!=act_task && t2->status!=DONE) {
285 if(t2->unblock_numb == t2->unblock_act_numb) {
286 t2->unblock_act_numb+=UNBLOCK_SIZE;
287 t2->unblock_task_ids=(unsigned long *)TASK_REALLOC(t2->unblock_task_ids,t2->unblock_act_numb*sizeof(unsigned long));
288 }
289 t2->unblock_task_ids[t2->unblock_numb]=act_id;
290 t2->unblock_numb++;
291 act_task->remaining_dep++;
292 }
293 }
294 } else {
295 if(t->status!=DONE) {
296 t->unblock_task_ids[t->unblock_numb]=act_id;
297 t->unblock_numb++;
298 act_task->remaining_dep++;
299 }
300 }
301 t->num_out--;
302 if (t->status==DONE && t->num_out==0)
303 task_hash_delete(t);
304 }
305 if(t_id) icl_hash_delete(address_set,(void *)d,NULL,NULL);
306 icl_hash_insert(address_set, (void*)d,(void *)(act_task->id));
307 act_task->num_out++;
308 }
309 }
310 if ((act_task->remaining_dep==0) && !waittask) {
311 act_task->status=READY;
312 readytasks++;
313 ready_queues[m].push(act_task);
314 m = (m + 1) % runningworkers;
315 }
316 return act_task;
317 }

319 // try to send at least one task to workers
320 inline bool schedule_task(const unsigned long th) {
321 bool ret = false;
322 for(size_t i=0;(readytasks>0)&&(i<runningworkers);i++){
323 if(nscheduled[i]<=th){
324 if(ready_queues[i].size()>0){
325 hash_task_t *top = ready_queues[i].top();
326 assert(top->status == READY);
327 //printf("schedule task %ld to %ld\n", top->id, i);
328 lb->ff_send_out_to(top,i);
329 //taskscheduled[i].push_back(top);// remember the task just sent
330 ready_queues[i].pop();
331 ++nscheduled[i], --readytasks;
332 ret = true;
333 } else{
334 bool found = false;
335 for(size_t j=0; !found && (j<runningworkers);j++){
336 if(ready_queues[mmax].size()>0){
337 hash_task_t *top = ready_queues[mmax].top();
338 assert(top->status == READY);
339 //printf("schedule task %ld to %ld\n", top->id, i);
340 lb->ff_send_out_to(top,i);
341 //taskscheduled[i].push_back(top); // remember the task just sent
342 ready_queues[mmax].pop();
343 ++nscheduled[i], --readytasks;
344 ret = found = true;
345 }
346 mmax = (mmax + 1) % runningworkers;
347 }
348 }
349 }
350 }
351 return ret;

354 inline void handleTask(hash_task_t *t, int workerid) {
355 for(long i=0;i<t->unblock_numb;i++) {
356 hash_task_t *tmp=(hash_task_t*)icl_hash_find(task_set,&t->unblock_task_ids[i]);
357 assert(tmp);
358 tmp->remaining_dep--;
359 if(tmp->remaining_dep==0) {
360 if (tmp->status == PENDING_DONE) {
361 handleTask(tmp,workerid);
362 } else {
363 if (tmp->status == NOT_READY) {
364 tmp->status=READY;
365 ++readytasks;
366 ready_queues[workerid].push(tmp);
367 }
368 }
369 }
370 }
371 schedule_task(0);
372
373 t->status=DONE;
374 if(!t->num_out) {
375 delete t->wtask;
376 task_hash_delete(t);
377 }
378 }
