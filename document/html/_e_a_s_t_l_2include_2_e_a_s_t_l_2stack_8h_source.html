56 template <typename T, typename Container = eastl::vector<T> >
57 class stack
58 {
59 public:
60 typedef stack<T, Container> this_type;
61 typedef Container container_type;
62 //typedef typename Container::allocator_type allocator_type; // We can't currently declare this because the container may be a type that doesn't have an allocator.
63 typedef typename Container::value_type value_type;
64 typedef typename Container::reference reference;
65 typedef typename Container::const_reference const_reference;
66 typedef typename Container::size_type size_type;
67
68 public: // We declare public so that global comparison operators can be implemented without adding an inline level and without tripping up GCC 2.x friend declaration failures. GCC (through at least v4.0) is poor at inlining and performance wins over correctness.
69 container_type c; // The C++ standard specifies that you declare a protected member variable of type Container called 'c'.
70
71 public:
72 stack();
73
74 // Allocator is templated here because we aren't allowed to infer the allocator_type from the Container, as some containers (e.g. array) don't
75 // have allocators. For containers that don't have allocator types, you could use void or char as the Allocator template type.

163 template <typename T, typename Container>
164 inline stack<T, Container>::stack(std::initializer_list<value_type> ilist)
165 : c() // We could alternatively use c(ilist) here, but that would require c to have an ilist constructor.
166 {
167 // Better solution but requires an insert function.
168 // c.insert(ilist.begin(), ilist.end());
169
170 // Possibly slower solution but doesn't require an insert function.
171 for(const auto& value : ilist)
172 {
173 c.push_back(value);
174 }
