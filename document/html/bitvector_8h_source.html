457 template <typename Element>
458 bitvector_const_iterator<Element>&
459 bitvector_const_iterator<Element>::operator+=(difference_type n)
460 {
461 n += mReference.mnBitIndex;
462
463 if(n >= difference_type(0))
464 {
465 mReference.mpBitWord += n / kBitCount;
466 mReference.mnBitIndex = (size_type)(n % kBitCount);
467 }
468 else
469 {
470 // backwards is tricky
471 // figure out how many full words backwards we need to move
472 // n = [-1..-32] => 1
473 // n = [-33..-64] => 2
474 const size_type backwards = (size_type)(-n + kBitCount - 1);
475 mReference.mpBitWord -= backwards / kBitCount;
476
477 // -1 => 31; backwards = 32; 31 - (backwards % 32) = 31
478 // -2 => 30; backwards = 33; 31 - (backwards % 32) = 30
479 // -3 => 29; backwards = 34
480 // ..
481 // -32 => 0; backwards = 63; 31 - (backwards % 32) = 0
482 // -33 => 31; backwards = 64; 31 - (backwards % 32) = 31
483 mReference.mnBitIndex = (kBitCount - 1) - (backwards % kBitCount);
484 }
485
486 return *this;
487 }

592 template <typename Element>
593 int bitvector_const_iterator<Element>::validate(const Element* pStart, const Element* pEnd, eastl_size_t nExtraBits) const
594 {
595 const Element* const pCurrent = mReference.mpBitWord;
596
597 if(pCurrent >= pStart)
598 {
599 if(nExtraBits == 0)
600 {
601 if(pCurrent == pEnd && mReference)
602 return eastl::isf_valid | eastl::isf_current;
603 else if(pCurrent < pEnd)
604 return eastl::isf_valid | eastl::isf_current | eastl::isf_can_dereference;
605 }
606 else if(pCurrent == (pEnd - 1))
607 {
608 const size_type bit = mReference.mnBitIndex;
609 const size_type lastbit = kBitCount - nExtraBits;
610
611 if(bit == lastbit)
612 return eastl::isf_valid | eastl::isf_current;
613 else if(bit < lastbit)
614 return eastl::isf_valid | eastl::isf_current | eastl::isf_can_dereference;
615 }
616 else if(pCurrent < pEnd)
617 {
618 return eastl::isf_valid | eastl::isf_current | eastl::isf_can_dereference;
619 }
620 }
621
622 return eastl::isf_none;
623 }

981 void bitvector<Allocator, Element, Container>::resize(size_type n, value_type value)
982 {
983 const size_type s = size();
984 if(n < s)
985 resize(n);
986
987 // Fill up to the end of a word
988 size_type newbits = n - s;
989
990 while(mFreeBitCount && newbits)
991 {
992 push_back(value);
993 --newbits;
994 }
995
996 // Fill the rest a word at a time
997 if(newbits)
998 {
999 element_type element(0);
1000 if(value)
1001 element = ~element;
1002
1003 const size_type words = (n + kBitCount - 1) / kBitCount;
1004 const size_type extra = words * kBitCount - n;
1005 mContainer.resize(words, element);
1006 mFreeBitCount = extra;
1007 }
