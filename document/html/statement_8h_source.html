518// helper function for easy traversal over the ASTs
519template<bool recurse_subexpr,
520 typename F,
521 typename EnterStmt,
522 typename ExitStmt>
523void traverse_expressions(
524 const Statement *stmt, const F &visit,
525 const EnterStmt &enter_stmt,
526 const ExitStmt &exit_stmt) noexcept {
527
528 auto do_visit = [&visit](auto expr) noexcept {
529 if constexpr (recurse_subexpr) {
530 traverse_subexpressions(expr, visit, [](auto) noexcept {});
531 } else {
532 visit(expr);
533 }
534 };
535
536 enter_stmt(stmt);
537 switch (stmt->tag()) {
538 case Statement::Tag::BREAK:
539 case Statement::Tag::CONTINUE: break;
540 case Statement::Tag::RETURN: {
541 auto return_stmt = static_cast<const ReturnStmt *>(stmt);
542 if (auto value = return_stmt->expression()) { do_visit(value); }
543 break;
544 }
545 case Statement::Tag::SCOPE: {
546 auto scope_stmt = static_cast<const ScopeStmt *>(stmt);
547 for (auto s : scope_stmt->statements()) {
548 traverse_expressions<recurse_subexpr>(
549 s, visit, enter_stmt, exit_stmt);
550 }
551 break;
552 }
553 case Statement::Tag::IF: {
554 auto if_stmt = static_cast<const IfStmt *>(stmt);
555 do_visit(if_stmt->condition());
556 traverse_expressions<recurse_subexpr>(
557 if_stmt->true_branch(), visit, enter_stmt, exit_stmt);
558 traverse_expressions<recurse_subexpr>(
559 if_stmt->false_branch(), visit, enter_stmt, exit_stmt);
560 break;
561 }
562 case Statement::Tag::LOOP: {
563 auto loop_stmt = static_cast<const LoopStmt *>(stmt);
564 traverse_expressions<recurse_subexpr>(
565 loop_stmt->body(), visit, enter_stmt, exit_stmt);
566 break;
567 }
568 case Statement::Tag::EXPR: {
569 auto expr_stmt = static_cast<const ExprStmt *>(stmt);
570 do_visit(expr_stmt->expression());
571 break;
572 }
573 case Statement::Tag::SWITCH: {
574 auto switch_stmt = static_cast<const SwitchStmt *>(stmt);
575 do_visit(switch_stmt->expression());
576 traverse_expressions<recurse_subexpr>(
577 switch_stmt->body(), visit, enter_stmt, exit_stmt);
578 break;
579 }
580 case Statement::Tag::SWITCH_CASE: {
581 auto case_stmt = static_cast<const SwitchCaseStmt *>(stmt);
582 traverse_expressions<recurse_subexpr>(
583 case_stmt->body(), visit, enter_stmt, exit_stmt);
584 break;
585 }
586 case Statement::Tag::SWITCH_DEFAULT: {
587 auto default_stmt = static_cast<const SwitchDefaultStmt *>(stmt);
588 traverse_expressions<recurse_subexpr>(
589 default_stmt->body(), visit, enter_stmt, exit_stmt);
590 break;
591 }
592 case Statement::Tag::ASSIGN: {
593 auto assign_stmt = static_cast<const AssignStmt *>(stmt);
594 do_visit(assign_stmt->lhs());
595 do_visit(assign_stmt->rhs());
596 break;
597 }
598 case Statement::Tag::FOR: {
599 auto for_stmt = static_cast<const ForStmt *>(stmt);
600 do_visit(for_stmt->variable());
601 do_visit(for_stmt->condition());
602 do_visit(for_stmt->step());
603 traverse_expressions<recurse_subexpr>(
604 for_stmt->body(), visit, enter_stmt, exit_stmt);
605 break;
606 }
607 case Statement::Tag::COMMENT: break;
608 case Statement::Tag::RAY_QUERY: {
609 auto rq_stmt = static_cast<const RayQueryStmt *>(stmt);
610 do_visit(rq_stmt->query());
611 traverse_expressions<recurse_subexpr>(
612 rq_stmt->on_triangle_candidate(), visit, enter_stmt, exit_stmt);
613 traverse_expressions<recurse_subexpr>(
614 rq_stmt->on_procedural_candidate(), visit, enter_stmt, exit_stmt);
615 break;
616 }
617 case Statement::Tag::AUTO_DIFF: {
618 auto ad_stmt = static_cast<const AutoDiffStmt *>(stmt);
619 traverse_expressions<recurse_subexpr>(
620 ad_stmt->body(), visit, enter_stmt, exit_stmt);
621 break;
622 }
623 case Statement::Tag::PRINT: {
624 auto print_stmt = static_cast<const PrintStmt *>(stmt);
625 for (auto arg : print_stmt->arguments()) { do_visit(arg); }
626 break;
627 }
628 }
629 exit_stmt(stmt);
630}
