55 template <
56 class RCU,
57 typename NodeType,
58 typename ValueType,
59 typename Disposer,
60#ifdef CDS_DOXYGEN_INVOKED
61 typename Cast
62#else
63 typename Cast=details::conventional_exempt_member_cast<NodeType, ValueType>
64#endif
65 >
66 class exempt_ptr
67 {
68 //@cond
69 struct trivial_cast {
70 ValueType * operator()( NodeType * p ) const
71 {
72 return p;
73 }
74 };
75 //@endcond
76 public:
77 typedef RCU rcu ;
78 typedef NodeType node_type ;
79 typedef ValueType value_type ;
80 typedef Disposer disposer ;
82 typedef typename std::conditional< std::is_same< Cast, void >::value, trivial_cast, Cast>::type node_to_value_cast;
83
84 private:
85 //@cond
86 node_type * m_pNode;
87 //@endcond
88
89 public:
91 exempt_ptr() noexcept
92 : m_pNode( nullptr )
93 {}
94
95 //@cond
97 explicit exempt_ptr( node_type * pNode ) noexcept
98 : m_pNode( pNode )
99 {}
100 explicit exempt_ptr( std::nullptr_t ) noexcept
101 : m_pNode( nullptr )
102 {}
103 //@endcond
104
106 exempt_ptr( exempt_ptr&& p ) noexcept
107 : m_pNode( p.m_pNode )
108 {
109 p.m_pNode = nullptr;
110 }
111
113 exempt_ptr( exempt_ptr const& ) = delete;
114
116 ~exempt_ptr()
117 {
118 release();
119 }
120
122 bool empty() const noexcept
123 {
124 return m_pNode == nullptr;
125 }
126
128 explicit operator bool() const noexcept
129 {
130 return !empty();
131 }
132
134 value_type * operator->() const noexcept
135 {
136 return !empty() ? node_to_value_cast()(m_pNode) : nullptr;
137 }
138
140 value_type& operator *() noexcept
141 {
142 assert( !empty());
143 return *node_to_value_cast()( m_pNode );
144 }
145
147 exempt_ptr& operator =( exempt_ptr&& p ) noexcept
148 {
149 release();
150 m_pNode = p.m_pNode;
151 p.m_pNode = nullptr;
152 return *this;
153 }
154
156 exempt_ptr& operator=(exempt_ptr const&) = delete;
157
159 void release()
160 {
161 if ( !empty()) {
162 assert( !rcu::is_locked());
163 rcu::template retire_ptr<disposer>( m_pNode );
164 m_pNode = nullptr;
165 }
166 }
167 };
