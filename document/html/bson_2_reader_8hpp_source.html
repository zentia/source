52 rfl::Result<InputVarType> get_field_from_array(
53 const size_t _idx, const InputArrayType& _arr) const noexcept {
54 bson_t b;
55 bson_iter_t iter;
56 const auto doc = _arr.val_->value.v_doc;
57 if (bson_init_static(&b, doc.data, doc.data_len)) {
58 if (bson_iter_init(&iter, &b)) {
59 size_t i = 0;
60 while (bson_iter_next(&iter)) {
61 if (i == _idx) {
62 return to_input_var(&iter);
63 }
64 ++i;
65 }
66 } else {
67 return error("Could not init the array iteration.");
68 }
69 } else {
70 return error("Could not init array.");
71 }
72 return error("Index " + std::to_string(_idx) + " of of bounds.");
73 }

97 template <class T>
98 rfl::Result<T> to_basic_type(const InputVarType& _var) const noexcept {
99 const auto btype = _var.val_->value_type;
100 const auto value = _var.val_->value;
101 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
102 switch (btype) {
103 case BSON_TYPE_UTF8:
104 return std::string(value.v_utf8.str, value.v_utf8.len);
105
106 case BSON_TYPE_SYMBOL:
107 return std::string(value.v_symbol.symbol, value.v_symbol.len);
108
109 default:
110 return error(
111 "Could not cast to string. The type must be UTF8 or symbol.");
112 }
113
114 } else if constexpr (std::is_same<std::remove_cvref_t<T>,
115 rfl::Bytestring>()) {
116 if (btype != BSON_TYPE_BINARY) {
117 return error("Could not cast to bytestring.");
118 }
119 if (value.v_binary.subtype != BSON_SUBTYPE_BINARY) {
120 return error(
121 "The BSON subtype must be a binary in order to read into a "
122 "bytestring.");
123 }
124 const auto data =
125 internal::ptr_cast<const std::byte*>(value.v_binary.data);
126 return rfl::Bytestring(data, data + value.v_binary.data_len);
127
128 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
129 if (btype != BSON_TYPE_BOOL) {
130 return error("Could not cast to boolean.");
131 }
132 return value.v_bool;
133
134 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
135 if (btype != BSON_TYPE_DOUBLE) {
136 return error(
137 "Could not cast to numeric value. The type must be double, "
138 "int32, int64 or date_time.");
139 }
140 return static_cast<T>(value.v_double);
141
142 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
143 switch (btype) {
144 case BSON_TYPE_INT32:
145 return static_cast<T>(value.v_int32);
146
147 case BSON_TYPE_INT64:
148 return static_cast<T>(value.v_int64);
149
150 case BSON_TYPE_DATE_TIME:
151 return static_cast<T>(value.v_datetime);
152
153 default:
154 return error(
155 "Could not cast to numeric value. The type must be "
156 "int32, int64 or date_time.");
157 }
158
159 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bson_oid_t>()) {
160 if (btype != BSON_TYPE_OID) {
161 return error("Could not cast to OID.");
162 }
163 return value.v_oid;
164 } else {
165 static_assert(rfl::always_false_v<T>, "Unsupported type.");
166 }
167 }

178 template <class ArrayReader>
179 std::optional<Error> read_array(const ArrayReader& _array_reader,
180 const InputArrayType& _arr) const noexcept {
181 bson_t b;
182 bson_iter_t iter;
183 const auto doc = _arr.val_->value.v_doc;
184 if (bson_init_static(&b, doc.data, doc.data_len)) {
185 if (bson_iter_init(&iter, &b)) {
186 while (bson_iter_next(&iter)) {
187 const auto err = _array_reader.read(to_input_var(&iter));
188 if (err) {
189 return err;
190 }
191 }
192 } else {
193 return Error("Could not init the array iteration.");
194 }
195 } else {
196 return Error("Could not init array.");
197 }
198 return std::nullopt;
199 }

206 const auto doc = _obj.val_->value.v_doc;

208 if (bson_iter_init(&iter, &b)) {

212 }

214 return Error("Could not init the object iteration.");
215 }
216 } else {
217 return Error("Could not init object.");
