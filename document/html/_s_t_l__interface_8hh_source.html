136 static inline void symv(gene_matrix & A, gene_vector & B, gene_vector & X, int N)
137 {
138 for (int j=0; j<N; ++j)
139 X[j] = 0;
140 for (int j=0; j<N; ++j)
141 {
142 real t1 = B[j];
143 real t2 = 0;
144 X[j] += t1 * A[j][j];
145 for (int i=j+1; i<N; ++i) {
146 X[i] += t1 * A[j][i];
147 t2 += A[j][i] * B[i];
148 }
149 X[j] += t2;
150 }
151 }
