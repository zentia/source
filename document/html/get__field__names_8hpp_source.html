40template <class T, auto ptr>
41consteval auto get_field_name_str_view() {
42#if __cpp_lib_source_location >= 201907L
43 const auto func_name =
44 std::string_view{std::source_location::current().function_name()};
45#elif defined(_MSC_VER)
46 // Officially, we only support MSVC versions that are modern enough to contain
47 // <source_location>, but inofficially, this might work.
48 const auto func_name = std::string_view{__FUNCSIG__};
49#else
50 const auto func_name = std::string_view{__PRETTY_FUNCTION__};
51#endif
52#if defined(__clang__)
53 const auto split = func_name.substr(0, func_name.size() - 2);
54 return split.substr(split.find_last_of(":.") + 1);
55#elif defined(__GNUC__)
56 const auto split = func_name.substr(0, func_name.size() - 2);
57 return split.substr(split.find_last_of(":") + 1);
58#elif defined(_MSC_VER)
59 const auto split = func_name.substr(0, func_name.size() - 7);
60 return split.substr(split.rfind("->") + 2);
61#else
62 static_assert(false,
63 "You are using an unsupported compiler. Please use GCC, Clang "
64 "or MSVC or switch to the rfl::Field-syntax.");
65#endif
66}

80template <class T, auto ptr>
81auto get_field_name() {
82#if defined(__clang__)
83 using Type = std::remove_cvref_t<std::remove_pointer_t<
84 typename std::remove_pointer_t<decltype(ptr)>::Type>>;
85#else
86 using Type = std::remove_cvref_t<std::remove_pointer_t<decltype(ptr)>>;
87#endif
88 if constexpr (is_rename_v<Type>) {
89 using Name = typename Type::Name;
90 return Name();

92 return get_field_names<std::remove_cvref_t<typename Type::Type>>();
93 } else {
