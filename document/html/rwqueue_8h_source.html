319 template <typename Func>
320 bool dequeue_with( Func f )
321 {
322 node_type * pNode;
323 {
324 scoped_lock lock( m_Head.lock );
325 pNode = m_Head.ptr;
326 node_type * pNewHead = pNode->m_pNext.load( atomics::memory_order_acquire );
327 if ( pNewHead == nullptr )
328 return false;
329 f( pNewHead->m_value );
330 m_Head.ptr = pNewHead;
331 } // unlock here
332 --m_ItemCounter;
333 free_node( pNode );
334 return true;
335 }

363 node_type * pHead = m_Head.ptr;
