146 void* svc(void * task) {
147 if (task) input = ((std::vector<T>*)task);
148
149 cudaEvolTask evolTask;
150 cudaEvolTask *pevolTask = &evolTask;
151
152 pipeevol.run_then_freeze();
153 while(!termination(*input)) {
154 // selection phase
155 buffer.clear();
156 selection(input->begin(),input->end(), buffer);
157
158 // evolution phase
159 evolTask.buffer = buffer.data();
160 evolTask.size = buffer.size();
161 pipeevol.offload((void*)pevolTask);
162 pipeevol.load_result((void**)&pevolTask);
163
164 // filtering phase
165 input->clear();
166 filter(buffer.begin(),buffer.end(), *input);
167 }
168 if (task) ff_send_out(task);
169 pipeevol.offload(GO_OUT);
170 pipeevol.wait_freezing();
171 return (task?GO_ON:NULL);
172 }
