792 position pos;

794 for ( pos.pCur = m_Head.next.load( memory_model::memory_order_relaxed ); pos.pCur != pos.pPrev; pos.pCur = pos.pCur->next.load( memory_model::memory_order_relaxed )) {
795 while ( true ) {

913 template <typename Func>
914 std::pair<bool, bool> update_at( node_type* pHead, value_type& val, Func func, bool bInsert )
915 {
916 insert_position pos;
917
918 typename gc::Guard guard;
919 guard.assign( &val );
920
921 while ( true ) {
922 if ( inserting_search( pHead, val, pos, key_comparator())) {
923 // try to replace pCur->data with val
924 assert( pos.pFound != nullptr );
925 assert( key_comparator()(*pos.pFound, val) == 0 );
926
927 marked_data_ptr pFound( pos.pFound );
928 if ( cds_likely( pos.pCur->data.compare_exchange_strong( pFound, marked_data_ptr( &val ),
929 memory_model::memory_order_release, atomics::memory_order_relaxed )))
930 {
931 if ( pos.pFound != &val ) {
932 retire_data( pos.pFound );
933 func( val, pos.pFound );
934 }
935 m_Stat.onUpdateExisting();
936 return std::make_pair( true, false );
937 }
938 }
939 else {
940 if ( !bInsert ) {
941 m_Stat.onUpdateFailed();
942 return std::make_pair( false, false );
943 }
944
945 if ( link_data( &val, pos, pHead )) {
946 func( val, static_cast<value_type*>( nullptr ));
947 ++m_ItemCounter;
948 m_Stat.onUpdateNew();
949 return std::make_pair( true, true );
950 }
951 }
952
953 m_Stat.onUpdateRetry();
954 }

1113 template <typename Q, typename Compare >
1114 bool search( node_type const* pHead, Q const& val, position& pos, Compare cmp ) const
1115 {
1116 pos.pHead = pHead;
1117 node_type* pPrev = const_cast<node_type*>( pHead );
1118
1119 while ( true ) {
1120 node_type * pCur = pPrev->next.load( memory_model::memory_order_relaxed );
1121
1122 if ( pCur == pCur->next.load( memory_model::memory_order_acquire )) {
1123 // end-of-list
1124 pos.pPrev = pPrev;
1125 pos.pCur = pCur;
1126 pos.pFound = nullptr;
1127 return false;
1128 }
1129
1130 value_type * pVal = pos.guard.protect( pCur->data,
1131 []( marked_data_ptr p ) -> value_type*
1132 {
1133 return p.ptr();
1134 }).ptr();
1135
1136 if ( pVal ) {
1137 int const nCmp = cmp( *pVal, val );
1138 if ( nCmp >= 0 ) {
1139 pos.pPrev = pPrev;
1140 pos.pCur = pCur;
1141 pos.pFound = pVal;
1142 return nCmp == 0;
1143 }
1144 }
1145
1146 pPrev = pCur;
1147 }

1194 void destroy( Predicate pred )
1195 {
1196 node_type * pNode = m_Head.next.load( memory_model::memory_order_relaxed );
1197 while ( pNode != pNode->next.load( memory_model::memory_order_relaxed )) {
1198 value_type * pVal = pNode->data.load( memory_model::memory_order_relaxed ).ptr();
1199 node_type * pNext = pNode->next.load( memory_model::memory_order_relaxed );
1200 bool const is_regular_node = !pVal || pred( pVal );
1201 if ( is_regular_node ) {
1202 if ( pVal )
1203 retire_data( pVal );
1204 delete_node( pNode );
1205 }
1206 pNode = pNext;

1253 bool link_data( value_type* pVal, insert_position& pos, node_type* pHead )
1254 {
1255 assert( pos.pPrev != nullptr );
1256 assert( pos.pCur != nullptr );
1257
1258 // We need pos.pCur data should be unchanged, otherwise ordering violation can be possible
1259 // if current thread will be preempted and another thread will delete pos.pCur data
1260 // and then set it to another.
1261 // To prevent this we mark pos.pCur data as undeletable by setting LSB
1262 marked_data_ptr valCur( pos.pFound );
1263 if ( !pos.pCur->data.compare_exchange_strong( valCur, valCur | 1, memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
1264 // oops, pos.pCur data has been changed or another thread is setting pos.pPrev data
1265 m_Stat.onNodeMarkFailed();
1266 return false;
1267 }
1268
1269 marked_data_ptr valPrev( pos.pPrevVal );
1270 if ( !pos.pPrev->data.compare_exchange_strong( valPrev, valPrev | 1, memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
1271 pos.pCur->data.store( valCur, memory_model::memory_order_relaxed );
1272
1273 m_Stat.onNodeMarkFailed();
1274 return false;
1275 }
1276
1277 // checks if link pPrev -> pCur is broken
1278 if ( pos.pPrev->next.load( memory_model::memory_order_acquire ) != pos.pCur ) {
1279 // sequence pPrev - pCur is broken
1280 pos.pPrev->data.store( valPrev, memory_model::memory_order_relaxed );
1281 pos.pCur->data.store( valCur, memory_model::memory_order_relaxed );
1282
1283 m_Stat.onNodeSeqBreak();
1284 return false;
1285 }
1286
1287 if ( pos.pPrevVal == nullptr ) {
1288 // Check ABA-problem for prev
1289 // There is a possibility that the current thread was preempted
1290 // on entry of this function. Other threads can link data to prev
1291 // and then remove it. As a result, the order of items may be changed
1292 if ( find_prev( pHead, *pVal ) != pos.pPrev ) {
1293 pos.pPrev->data.store( valPrev, memory_model::memory_order_relaxed );
1294 pos.pCur->data.store( valCur, memory_model::memory_order_relaxed );
1295
1296 m_Stat.onNullPrevABA();
1297 return false;
1298 }
1299 }
1300
1301 if ( pos.pPrev != pos.pHead && pos.pPrevVal == nullptr ) {
1302 // reuse pPrev
1303
1304 // Set pos.pPrev data if it is null
1305 valPrev |= 1;
1306 bool result = pos.pPrev->data.compare_exchange_strong( valPrev, marked_data_ptr( pVal ),
1307 memory_model::memory_order_release, atomics::memory_order_relaxed );
1308
1309 // Clears data marks
1310 pos.pCur->data.store( valCur, memory_model::memory_order_relaxed );
1311
1312 if ( result ) {
1313 m_Stat.onReuseNode();
1314 return result;
1315 }
1316 }
1317 else {
1318 // insert new node between pos.pPrev and pos.pCur
1319 node_type * pNode = alloc_node( pVal );
1320 pNode->next.store( pos.pCur, memory_model::memory_order_relaxed );
1321
1322 bool result = pos.pPrev->next.compare_exchange_strong( pos.pCur, pNode, memory_model::memory_order_release, atomics::memory_order_relaxed );
1323
1324 // Clears data marks
1325 pos.pPrev->data.store( valPrev, memory_model::memory_order_relaxed );
1326 pos.pCur->data.store( valCur, memory_model::memory_order_relaxed );
1327
1328 if ( result ) {
1329 m_Stat.onNewNodeCreated();
1330 return result;
1331 }
1332
1333 delete_node( pNode );
1334 }
1335
1336 return false;
1337 }
