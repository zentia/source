15/*
16 * ARMv6 UP and SMP safe atomic ops. We use load exclusive and
17 * store exclusive to ensure that these are atomic. We may loop
18 * to ensure that the update happens. Writing to 'v->counter'
19 * without using the following operations WILL break the atomic
20 * nature of these ops.
21 */
22static inline void atomic_set(atomic_t *v, int i)
23{
24 unsigned long tmp;
25
26 __asm__ __volatile__("@ atomic_set\n"
27"1: ldrex %0, [%1]\n" // load ex: tmp <- v->counter
28" strex %0, %2, [%1]\n" // store ex: tmp(status), i -> v->counter ; tmp=0 se tutto ok
29" teq %0, #0\n" // controlla se tmp contiene 0 e loop se !=
30" bne 1b"
31 : "=&r" (tmp) // output: %0
32 : "r" (&v->counter), "r" (i) // input: %1, %2
33 : "cc", "memory");
34}

36static inline int atomic_add_return(int i, atomic_t *v)
37{
38 unsigned long tmp;
39 int result;

41 __asm__ __volatile__("@ atomic_add_return\n"
42"1: ldrex %0, [%2]\n" // load ex: result <- v->counter

44" strex %1, %0, [%2]\n" // store ex: tmp(status), result -> v->counter ; tmp=0 se tutto ok

77static inline int atomic_cmpxchg(atomic_t *ptr, int old, int New)
78{
79 unsigned long oldval, res;

87 : "=&r" (res), "=&r" (oldval)

89 : "cc", "memory");
