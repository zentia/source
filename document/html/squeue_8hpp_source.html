96 inline void push_back(const elem_type & elem) {
97 T * current = data[tail].entry;
98 int current_tail = data[tail].t++;
99 if ((unsigned)current_tail == (chunk-1)) {
100 if (tail == (datacap-1)) {
101 datacap += DATA_CHUNK;
102 data = (data_type *)realloc(data, datacap*sizeof(data_type));
103 }
104
105 T * v = newchunk();
106 data[++tail] = data_type(0,0,v);
107 current = v;
108 current_tail=-1;
109 }
110 current[current_tail+1] = elem;
111 ++nelements;
112 }

115 inline void pop_back() {
116 if (!nelements) return;
117
118 T * current = data[tail].entry;
119 int current_tail = data[tail].t--;
120
121 current[current_tail].~T();
122
123 --nelements;
124 if (!current_tail && (tail>0)) {
125 deletechunk(tail);
126 --tail;
127 data[tail].t = chunk-1;
128 }
129
130 }

132 inline elem_type& back() {
133 if (!nelements) return *(elem_type*)0;
134
135 T * current = data[tail].entry;
136 int current_tail = data[tail].t;
137 return current[current_tail];
138 }

141 inline void pop_front() {
142 if (!nelements) return;
143
144 T * current = data[head].entry;
145 int current_head = data[head].h++;
146
147 current[current_head].~T();
148
149 --nelements;
150 if (((unsigned)current_head==(chunk-1)) && (tail>head)) {
151 deletechunk(head);
152 ++head;
153 }
