35 alignas(

37 auto ptr = internal::ptr_cast<std::array<T, _size>*>(&buf);

39 ArrayReader<R, W, ProcessorsType, T, _size>(&_r, ptr);
40 auto err = _r.read_array(array_reader, _arr);
41 if (err) {
42 call_destructors_on_array_where_necessary(array_reader.num_set(), ptr);
43 return error(*err);
44 }
45 err = array_reader.check_size();
46 if (err) {
47 call_destructors_on_array_where_necessary(array_reader.num_set(), ptr);
48 return error(*err);

50 auto result = Result<std::array<T, _size>>(std::move(*ptr));
51 call_destructors_on_array_where_necessary(array_reader.num_set(), ptr);
52 return result;
53 };
