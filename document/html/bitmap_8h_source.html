15/* --------------------------------------------------------------------------------
16 Atomic bitmaps with release/acquire guarantees:
17
18 `mi_bfield_t`: is a single machine word that can efficiently be bit counted (usually `size_t`)
19 each bit usually represents a single MI_ARENA_SLICE_SIZE in an arena (64 KiB).
20 We need 16K bits to represent a 1GiB arena.
21
22 `mi_bchunk_t`: a chunk of bfield's of a total of MI_BCHUNK_BITS (= 512 on 64-bit, 256 on 32-bit)
23 allocations never span across chunks -- so MI_ARENA_MAX_OBJ_SIZE is the number
24 of bits in a chunk times the MI_ARENA_SLICE_SIZE (512 * 64KiB = 32 MiB).
25 These chunks are cache-aligned and we can use AVX2/AVX512/NEON/SVE/SVE2/etc. instructions
26 to scan for bits (perhaps) more efficiently.
27
28 We allocate byte-sized ranges aligned to bytes in the bfield, and bfield-sized
29 ranges aligned to a bfield.
30
31 Searching linearly through the chunks would be too slow (16K bits per GiB).
32 Instead we add a "chunkmap" to do a two-level search (more or less a btree of depth 2).
33
34 `mi_bchunkmap_t` (== `mi_bchunk_t`): for each chunk we track if it has (potentially) any bit set.
35 The chunkmap has 1 bit per chunk that is set if the chunk potentially has a bit set.
36 This is used to avoid scanning every chunk. (and thus strictly an optimization)
37 It is conservative: it is fine to set a bit in the chunk map even if the chunk turns out
38 to have no bits set. It is also allowed to briefly have a clear bit even if the
39 chunk has bits set -- as long as we guarantee that the bit will be set later on;
40 (this allows us to set the chunkmap bit right after we set a bit in the corresponding chunk).
41
42 However, when we clear a bit in a chunk, and the chunk is indeed all clear, we
43 cannot safely clear the bit corresponding to the chunk in the chunkmap since it
44 may race with another thread setting a bit in the same chunk. Therefore, when
45 clearing, we first test if a chunk is clear, then clear the chunkmap bit, and
46 then test again to catch any set bits that we may have missed.
47
48 Since the chunkmap may thus be briefly out-of-sync, this means that we may sometimes
49 not find a free page even though it's there (but we accept this as we avoid taking
50 full locks). (Another way to do this is to use an epoch but we like to avoid that complexity
51 for now).
52
53 `mi_bitmap_t`: a bitmap with N chunks. A bitmap has a chunkmap of MI_BCHUNK_BITS (512)
54 and thus has at most 512 chunks (=2^18 bits x 64 KiB slices = 16 GiB max arena size).
55 The minimum is 1 chunk which is a 32 MiB arena.
56
57 For now, the implementation assumes MI_HAS_FAST_BITSCAN and uses trailing-zero-count
58 and pop-count (but we think it can be adapted work reasonably well on older hardware too)
59--------------------------------------------------------------------------------------------- */
60
61// A word-size bit field.
62typedef size_t mi_bfield_t;
63
64#define MI_BFIELD_BITS_SHIFT (MI_SIZE_SHIFT+3)
65#define MI_BFIELD_BITS (1 << MI_BFIELD_BITS_SHIFT)
66#define MI_BFIELD_SIZE (MI_BFIELD_BITS/8)
67#define MI_BFIELD_LO_BIT8 (((~(mi_bfield_t)0))/0xFF) // 0x01010101 ..
68#define MI_BFIELD_HI_BIT8 (MI_BFIELD_LO_BIT8 << 7) // 0x80808080 ..
69
70#define MI_BCHUNK_SIZE (MI_BCHUNK_BITS / 8)
71#define MI_BCHUNK_FIELDS (MI_BCHUNK_BITS / MI_BFIELD_BITS) // 8 on both 64- and 32-bit

74// A bitmap chunk contains 512 bits on 64-bit (256 on 32-bit)
75typedef mi_decl_align(MI_BCHUNK_SIZE) struct mi_bchunk_s {
76 _Atomic(mi_bfield_t) bfields[MI_BCHUNK_FIELDS];
77} mi_bchunk_t;

80// The chunkmap has one bit per corresponding chunk that is set if the chunk potentially has bits set.
81// The chunkmap is itself a chunk.

94#define MI_BITMAP_MIN_BIT_COUNT (MI_BITMAP_MIN_CHUNK_COUNT * MI_BCHUNK_BITS) // 32 MiB arena
95#define MI_BITMAP_DEFAULT_BIT_COUNT (MI_BITMAP_DEFAULT_CHUNK_COUNT * MI_BCHUNK_BITS) // 2 GiB arena

98// An atomic bitmap
99typedef mi_decl_align(MI_BCHUNK_SIZE) struct mi_bitmap_s {
100 _Atomic(size_t) chunk_count; // total count of chunks (0 < N <= MI_BCHUNKMAP_BITS)

209/* ----------------------------------------------------------------------------
210 Binned concurrent bitmap
211 Assigns a size class to each chunk such that small blocks don't cause too
212 much fragmentation since we keep chunks for larger blocks separate.
213---------------------------------------------------------------------------- */
214
215// Size bins; larger bins are allowed to go into smaller bins.
216// SMALL can only be in small (and NONE), so they cannot fragment the larger bins.
217typedef enum mi_bbin_e {
218 MI_BBIN_SMALL, // slice_count == 1
219 MI_BBIN_OTHER, // slice_count: any other from the other bins, and 1 <= slice_count <= MI_BCHUNK_BITS
220 MI_BBIN_MEDIUM, // slice_count == 8
221 MI_BBIN_LARGE, // slice_count == MI_BFIELD_BITS -- only used if MI_ENABLE_LARGE_PAGES is 1
222 MI_BBIN_NONE, // no bin assigned yet (the chunk is completely free)
223 MI_BBIN_COUNT
224} mi_bbin_t;
225
226static inline mi_bbin_t mi_bbin_inc(mi_bbin_t bbin) {
227 mi_assert_internal(bbin < MI_BBIN_COUNT);
228 return (mi_bbin_t)((int)bbin + 1);
229}
230
231static inline mi_bbin_t mi_bbin_dec(mi_bbin_t bbin) {
232 mi_assert_internal(bbin > MI_BBIN_NONE);
233 return (mi_bbin_t)((int)bbin - 1);
234}
235
236static inline mi_bbin_t mi_bbin_of(size_t slice_count) {
237 if (slice_count==1) return MI_BBIN_SMALL;
238 if (slice_count==8) return MI_BBIN_MEDIUM;
239 #if MI_ENABLE_LARGE_PAGES
240 if (slice_count==MI_BFIELD_BITS) return MI_BBIN_LARGE;
241 #endif
242 return MI_BBIN_OTHER;
243}
