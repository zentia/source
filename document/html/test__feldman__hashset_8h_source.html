264 void test( Set& s )
265 {
266 // Precondition: set is empty
267 // Postcondition: set is empty
268
269 ASSERT_TRUE( s.empty());
270 ASSERT_CONTAINER_SIZE( s, 0 );
271 size_t const nSetSize = kSize;
272
273 typedef typename Set::value_type value_type;
274
275 std::vector< value_type > data;
276 std::vector< size_t> indices;
277 data.reserve( kSize );
278 indices.reserve( kSize );
279 for ( size_t key = 0; key < kSize; ++key ) {
280 data.push_back( value_type( static_cast<int>(key)));
281 indices.push_back( key );
282 }
283 shuffle( indices.begin(), indices.end());
284
285 // insert/find
286 for ( auto idx : indices ) {
287 auto& i = data[idx];
288
289 ASSERT_FALSE( s.contains( i.nKey ));
290 ASSERT_FALSE( s.find( i.nKey, []( value_type& ) {} ));
291
292 std::pair<bool, bool> updResult;
293
294 std::string str;
295 updResult = s.update( i.key(), []( value_type&, value_type * )
296 {
297 ASSERT_TRUE( false );
298 }, false );
299 EXPECT_FALSE( updResult.first );
300 EXPECT_FALSE( updResult.second );
301
302 switch ( idx % 8 ) {
303 case 0:
304 ASSERT_TRUE( s.insert( i ));
305 ASSERT_FALSE( s.insert( i ));
306 updResult = s.update( i, []( value_type& val, value_type * prev )
307 {
308 ASSERT_TRUE( prev != nullptr );
309 EXPECT_EQ( val.key(), prev->key());
310 }, false );
311 EXPECT_TRUE( updResult.first );
312 EXPECT_FALSE( updResult.second );
313 break;
314 case 1:
315 ASSERT_TRUE( s.insert( i.key()));
316 ASSERT_FALSE( s.insert( i.key()));
317 updResult = s.update( i.key(), []( value_type& val, value_type * prev )
318 {
319 ASSERT_TRUE( prev != nullptr );
320 EXPECT_EQ( val.key(), prev->key());
321 }, false );
322 EXPECT_TRUE( updResult.first );
323 EXPECT_FALSE( updResult.second );
324 break;
325 case 2:
326 ASSERT_TRUE( s.insert( i, []( value_type& v ) { ++v.nFindCount; } ));
327 ASSERT_FALSE( s.insert( i, []( value_type& v ) { ++v.nFindCount; } ));
328 ASSERT_TRUE( s.find( i.nKey, []( value_type const& v )
329 {
330 EXPECT_EQ( v.nFindCount, 1u );
331 }));
332 break;
333 case 3:
334 ASSERT_TRUE( s.insert( i.key(), []( value_type& v ) { ++v.nFindCount; } ));
335 ASSERT_FALSE( s.insert( i.key(), []( value_type& v ) { ++v.nFindCount; } ));
336 ASSERT_TRUE( s.find( i.nKey, []( value_type const& v )
337 {
338 EXPECT_EQ( v.nFindCount, 1u );
339 }));
340 break;
341 case 4:
342 updResult = s.update( i, []( value_type& v, value_type * prev )
343 {
344 EXPECT_TRUE( prev == nullptr );
345 ++v.nUpdateNewCount;
346 });
347 EXPECT_TRUE( updResult.first );
348 EXPECT_TRUE( updResult.second );
349
350 updResult = s.update( i, []( value_type& v, value_type * prev )
351 {
352 ASSERT_TRUE( prev != nullptr );
353 EXPECT_EQ( prev->nUpdateNewCount, 1u );
354 EXPECT_EQ( v.key(), prev->key());
355 ++v.nUpdateCount;
356 }, false );
357 EXPECT_TRUE( updResult.first );
358 EXPECT_FALSE( updResult.second );
359
360 ASSERT_TRUE( s.find( i.nKey, []( value_type const& v )
361 {
362 EXPECT_EQ( v.nUpdateCount, 1u );
363 EXPECT_EQ( v.nUpdateNewCount, 0u );
364 }));
365 break;
366 case 5:
367 updResult = s.update( i.key(), []( value_type& v, value_type * prev )
368 {
369 EXPECT_TRUE( prev == nullptr );
370 ++v.nUpdateNewCount;
371 });
372 EXPECT_TRUE( updResult.first );
373 EXPECT_TRUE( updResult.second );
374
375 updResult = s.update( i.key(), []( value_type& v, value_type * prev )
376 {
377 ASSERT_TRUE( prev != nullptr );
378 EXPECT_EQ( v.key(), prev->key());
379 EXPECT_EQ( prev->nUpdateNewCount, 1u );
380 EXPECT_EQ( v.nUpdateNewCount, 0u );
381 ++v.nUpdateNewCount;
382 }, false );
383 EXPECT_TRUE( updResult.first );
384 EXPECT_FALSE( updResult.second );
385
386 ASSERT_TRUE( s.find( i.key(), []( value_type const& v )
387 {
388 EXPECT_EQ( v.nUpdateNewCount, 1u );
389 } ));
390 break;
391 case 6:
392 ASSERT_TRUE( s.emplace( i.key()));
393 ASSERT_TRUE( s.contains( i.key()));
394 break;
395 case 7:
396 str = "Hello!";
397 ASSERT_TRUE( s.emplace( i.key(), std::move( str )));
398 EXPECT_TRUE( str.empty());
399 ASSERT_TRUE( s.find( i.key(), []( value_type const& v )
400 {
401 EXPECT_EQ( v.strVal, std::string( "Hello!" ));
402 } ));
403 break;
404 default:
405 // forgot anything?..
406 ASSERT_TRUE( false );
407 }
408
409 ASSERT_TRUE( s.contains( i.nKey ));
410 ASSERT_TRUE( s.find( i.nKey, []( value_type& ) {} ));
411 }
412
413 ASSERT_FALSE( s.empty());
414 ASSERT_CONTAINER_SIZE( s, nSetSize );
415
416 // erase
417 shuffle( indices.begin(), indices.end());
418 for ( auto idx : indices ) {
419 auto& i = data[idx];
420
421 ASSERT_TRUE( s.contains( i.nKey ));
422 ASSERT_TRUE( s.find( i.nKey, []( value_type& v )
423 {
424 v.nFindCount = 1;
425 } ));
426
427 int nKey = i.key() - 1;
428 switch ( idx % 2 ) {
429 case 0:
430 ASSERT_TRUE( s.erase( i.key()));
431 ASSERT_FALSE( s.erase( i.key()));
432 break;
433 case 1:
434 ASSERT_TRUE( s.erase( i.key(), [&nKey]( value_type const& v )
435 {
436 EXPECT_EQ( v.nFindCount, 1u );
437 nKey = v.key();
438 } ));
439 EXPECT_EQ( i.key(), nKey );
440
441 nKey = i.key() - 1;
442 ASSERT_FALSE( s.erase( i.key(), [&nKey]( value_type const& v )
443 {
444 nKey = v.key();
445 } ));
446 EXPECT_EQ( i.key(), nKey + 1 );
447 break;
448 }
449
450 ASSERT_FALSE( s.contains( i.nKey ));
451 ASSERT_FALSE( s.find( i.nKey, []( value_type const& ) {} ));
452 }
453 ASSERT_TRUE( s.empty());
454 ASSERT_CONTAINER_SIZE( s, 0u );
455
456
457 // clear
458 for ( auto& i : data ) {
459 ASSERT_TRUE( s.insert( i ));
460 }
461
462 {
463 typename Set::stat const& statistics = s.statistics();
464 CDS_UNUSED( statistics );
465
466 std::vector< typename Set::level_statistics > lstat;
467 s.get_level_statistics( lstat );
468 EXPECT_EQ( lstat[0].node_capacity, s.head_size());
469 for ( size_t i = 1; i < lstat.size(); ++i ) {
470 EXPECT_EQ( lstat[i].node_capacity, s.array_node_size());
471 }
472 }
473
474 ASSERT_FALSE( s.empty());
475 ASSERT_CONTAINER_SIZE( s, nSetSize );
476
477 s.clear();
478
479 ASSERT_TRUE( s.empty());
480 ASSERT_CONTAINER_SIZE( s, 0u );
481
482 ASSERT_TRUE( s.begin() == s.end());
483 ASSERT_TRUE( s.cbegin() == s.cend());
484 ASSERT_TRUE( s.rbegin() == s.rend());
485 ASSERT_TRUE( s.crbegin() == s.crend());
486 }
487 };
