32void concurrent_monitor::prepare_wait( thread_context& thr, uintptr_t ctx ) {
33 if( !thr.ready )
34 thr.init();
35 // this is good place to pump previous skipped wakeup
36 else if( thr.skipped_wakeup ) {
37 thr.skipped_wakeup = false;
38 thr.semaphore().P();
39 }
40 thr.context = ctx;
41 thr.in_waitset = true;
42 {
43 tbb::spin_mutex::scoped_lock l( mutex_ec );
44 __TBB_store_relaxed( thr.epoch, __TBB_load_relaxed(epoch) );
45 waitset_ec.add( (waitset_t::node_t*)&thr );
46 }
47 atomic_fence();
48}

50void concurrent_monitor::cancel_wait( thread_context& thr ) {
51 // possible skipped wakeup will be pumped in the following prepare_wait()
52 thr.skipped_wakeup = true;
53 // try to remove node from waitset
54 bool th_in_waitset = thr.in_waitset;
55 if( th_in_waitset ) {
56 tbb::spin_mutex::scoped_lock l( mutex_ec );
57 if (thr.in_waitset) {
58 waitset_ec.remove( (waitset_t::node_t&)thr );
59 // node is removed from waitset, so there will be no wakeup
60 thr.in_waitset = false;
61 thr.skipped_wakeup = false;
62 }
63 }

66void concurrent_monitor::notify_one_relaxed() {
67 if( waitset_ec.empty() )
68 return;
69 waitset_node_t* n;
70 const waitset_node_t* end = waitset_ec.end();
71 {
72 tbb::spin_mutex::scoped_lock l( mutex_ec );
73 __TBB_store_relaxed( epoch, __TBB_load_relaxed(epoch) + 1 );
74 n = waitset_ec.front();
75 if( n!=end ) {
76 waitset_ec.remove( *n );
77 to_thread_context(n)->in_waitset = false;
78 }
79 }
80 if( n!=end )
81 to_thread_context(n)->semaphore().V();
82}

84void concurrent_monitor::notify_all_relaxed() {
85 if( waitset_ec.empty() )
86 return;
87 waitset_t temp;
88 const waitset_node_t* end;
89 {
90 tbb::spin_mutex::scoped_lock l( mutex_ec );
91 __TBB_store_relaxed( epoch, __TBB_load_relaxed(epoch) + 1 );
92 waitset_ec.flush_to( temp );
93 end = temp.end();
94 for( waitset_node_t* n=temp.front(); n!=end; n=n->next )
95 to_thread_context(n)->in_waitset = false;
96 }
97 waitset_node_t* nxt;
98 for( waitset_node_t* n=temp.front(); n!=end; n=nxt ) {
99 nxt = n->next;
100 to_thread_context(n)->semaphore().V();
101 }
102#if TBB_USE_ASSERT
103 temp.clear();
104#endif
105}

113 tbb::spin_mutex::scoped_lock l( mutex_ec );
