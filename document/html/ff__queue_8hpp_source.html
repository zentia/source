90 ff_queue (size_t bucket_size, size_t max_bucket_count)
91 : bucket_size_ (bucket_size)
92 , max_bucket_count_ (max_bucket_count)
93 {
94 bucket_count_ = 0;
95 bucket_t* bucket = alloc_bucket(bucket_size_);
96 head_pos_ = bucket->data;
97 tail_pos_ = bucket->data;
98 tail_end_ = bucket->data + bucket_size_;
99 tail_next_ = 0;
100 tail_bucket_ = bucket;
101 last_bucket_ = bucket;
102 *(void**)head_pos_ = (void*)1;
103 pad_[0] = 0;
104 }

107 ~ff_queue ()
108 {
109 bucket_t* bucket = last_bucket_;
110 while (bucket != 0)
111 {
112 bucket_t* next_bucket = bucket->next;
113 aligned_free(bucket);
114 bucket = next_bucket;
115 }
116 }

118 char* enqueue_prepare (size_t sz)
119 {
120 assert(((uintptr_t)tail_pos_ % sizeof(void*)) == 0);
121 size_t msg_size = ((uintptr_t)(sz + sizeof(void*) - 1) & ~(sizeof(void*) - 1)) + sizeof(void*);
122 if ((size_t)(tail_end_ - tail_pos_) >= msg_size + sizeof(void*))
123 {
124 tail_next_ = tail_pos_ + msg_size;
125 return tail_pos_ + sizeof(void*);
126 }
127 else
128 {
129 return enqueue_prepare_slow(sz);
130 }
131 }
132
133 void enqueue_commit ()
134 {
135 *(char* volatile*)tail_next_ = (char*)1;
136 atomic_addr_store_release((void* volatile*)tail_pos_, tail_next_);
137 tail_pos_ = tail_next_;
138 }

140
141 char* dequeue_prepare ()
142 {
143 assert(((uintptr_t)head_pos_ % sizeof(void*)) == 0);
144 void* next = atomic_addr_load_acquire((void* volatile*)head_pos_);
145 if (((uintptr_t)next & 1) == 0)
146 {
147 char* msg = head_pos_ + sizeof(void*);
148 return msg;
149 }
150 else if (((uintptr_t)next & ~1) == 0)
151 {
152 return 0;
153 }
154 else
155 {
156 atomic_addr_store_release((void* volatile*)&head_pos_, (char*)((uintptr_t)next & ~1));
157 return dequeue_prepare();
158 }
159 }

162 void dequeue_commit ()
163 {
164 char* next = *(char* volatile*)head_pos_;
165 assert(next != 0);
166 atomic_addr_store_release((void* volatile*)&head_pos_, next);
167 }

191 bucket_t* alloc_bucket (size_t sz)
192 {
193 bucket_t* bucket = (bucket_t*)aligned_malloc(sizeof(bucket_t) + sz);
194 if (bucket == 0)
195 throw std::bad_alloc();
196 bucket->next = 0;
197 bucket->size = sz;
198 bucket_count_ += 1;

203 NOINLINE
204 char* enqueue_prepare_slow (size_t sz)
205 {
206 size_t bucket_size = bucket_size_;
207 if (bucket_size < sz + 2 * sizeof(void*))
208 bucket_size = sz + 2 * sizeof(void*);
209
210 bucket_t* bucket = 0;
211 char* head_pos = (char*)atomic_addr_load_acquire((void* volatile*)&head_pos_);
212 while (head_pos < last_bucket_->data || head_pos >= last_bucket_->data + last_bucket_->size)
213 {
214 bucket = last_bucket_;
215 last_bucket_ = bucket->next;
216 bucket->next = 0;
217 assert(last_bucket_ != 0);
218
219 if ((bucket->size < bucket_size)
220 || (bucket_count_ > max_bucket_count_
221 && (head_pos < last_bucket_->data || head_pos >= last_bucket_->data + last_bucket_->size)))
222 {
223 aligned_free(bucket);
224 bucket = 0;
225 continue;
226 }
227 break;
228 }
229
230 if (bucket == 0)
231 bucket = alloc_bucket(bucket_size);
232 *(void* volatile*)bucket->data = (void*)1;
233 atomic_addr_store_release((void* volatile*)tail_pos_, (void*)((uintptr_t)bucket->data | 1));
234 tail_pos_ = bucket->data;
235 tail_end_ = tail_pos_ + bucket_size;
236 tail_bucket_->next = bucket;
237 tail_bucket_ = bucket;
238 return enqueue_prepare(sz);
239 }
