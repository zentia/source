66 S {new std::atomic<T>[static_cast<size_t>(C)]} {
67 }
68

77 void push(int64_t i, T o) noexcept {
78 S[i & M].store(o, std::memory_order_relaxed);
79 }
80

82 return S[i & M].load(std::memory_order_relaxed);

86 Array* ptr = new Array {2*C};

210// Function: push
211template <typename T>
212void UnboundedTaskQueue<T>::push(T o) {
213
214 int64_t b = _bottom.load(std::memory_order_relaxed);
215 int64_t t = _top.load(std::memory_order_acquire);
216 Array* a = _array.load(std::memory_order_relaxed);
217
218 // queue is full with one additional item (b-t+1)
219 if TF_UNLIKELY(a->capacity() - 1 < (b - t)) {
220 a = resize_array(a, b, t);
221 }
222
223 a->push(b, o);
224 std::atomic_thread_fence(std::memory_order_release);
225
226 // original paper uses relaxed here but tsa complains
227 _bottom.store(b + 1, std::memory_order_release);
228}

230// Function: pop
231template <typename T>
232T UnboundedTaskQueue<T>::pop() {
233
234 int64_t b = _bottom.load(std::memory_order_relaxed) - 1;
235 Array* a = _array.load(std::memory_order_relaxed);
236 _bottom.store(b, std::memory_order_relaxed);
237 std::atomic_thread_fence(std::memory_order_seq_cst);
238 int64_t t = _top.load(std::memory_order_relaxed);
239
240 T item {nullptr};
241
242 if(t <= b) {
243 item = a->pop(b);
244 if(t == b) {
245 // the last item just got stolen
246 if(!_top.compare_exchange_strong(t, t+1,
247 std::memory_order_seq_cst,
248 std::memory_order_relaxed)) {
249 item = nullptr;
250 }
251 _bottom.store(b + 1, std::memory_order_relaxed);
252 }
253 }
254 else {
255 _bottom.store(b + 1, std::memory_order_relaxed);
256 }
257
258 return item;
259}

467// Function: try_push
468template <typename T, size_t LogSize>
469template <typename O>
470bool BoundedTaskQueue<T, LogSize>::try_push(O&& o) {
471
472 int64_t b = _bottom.load(std::memory_order_relaxed);
473 int64_t t = _top.load(std::memory_order_acquire);
474
475 // queue is full with one additional item (b-t+1)
476 if TF_UNLIKELY((b - t) > BufferSize - 1) {
477 return false;
478 }
479
480 _buffer[b & BufferMask].store(std::forward<O>(o), std::memory_order_relaxed);
481
482 std::atomic_thread_fence(std::memory_order_release);
483
484 // original paper uses relaxed here but tsa complains
485 _bottom.store(b + 1, std::memory_order_release);
486
487 return true;
488}
