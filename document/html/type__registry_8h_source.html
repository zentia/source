250const Type *Type::of() noexcept {
251 if constexpr (std::is_same_v<T, void>) {
252 return nullptr;
253 } else {
254 auto desc = detail::TypeDesc<std::remove_cvref_t<T>>::description();
255 if constexpr (is_custom_struct_v<T>) {

260 return t;

268struct is_valid_reflection : std::false_type {};

270template<typename S, typename... M, typename O, O... os>
271struct is_valid_reflection<S, std::tuple<M...>, std::integer_sequence<O, os...>> {
272
273 static_assert(alignof(S) >= 4u, "Structs must be aligned to at least 4 bytes.");
274
275private:
276 [[nodiscard]] constexpr static auto _check() noexcept {
277 constexpr auto count = sizeof...(M);
278 static_assert(sizeof...(os) == count);
279 constexpr std::array<size_t, count> sizes{sizeof(M)...};
280 constexpr std::array<size_t, count> alignments{alignof(M)...};
281 constexpr std::array<size_t, count> offsets{os...};
282 auto current_offset = 0u;
283 for (auto i = 0u; i < count; i++) {
284 auto offset = offsets[i];
285 auto size = sizes[i];
286 auto alignment = alignments[i];
287 current_offset = (current_offset + alignment - 1u) /
288 alignment *
289 alignment;
290 if (current_offset != offset) { return false; }
291 current_offset += size;
292 }
293 constexpr auto struct_size = sizeof(S);
294 constexpr auto struct_alignment = alignof(S);
295 current_offset = (current_offset + struct_alignment - 1u) /
296 struct_alignment *
297 struct_alignment;
298 return current_offset == struct_size;
299 };
300
301public:
302 static constexpr auto value = _check();
303};
