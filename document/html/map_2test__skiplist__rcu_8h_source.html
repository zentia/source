21 template <typename Map>
22 void test( Map& m )
23 {
24 // Precondition: map is empty
25 // Postcondition: map is empty
26
27 base_class::test( m );
28
29 ASSERT_TRUE( m.empty());
30 ASSERT_CONTAINER_SIZE( m, 0 );
31
32 typedef typename Map::exempt_ptr exempt_ptr;
33 size_t const kkSize = base_class::kSize;
34
35 // get_min
36 for ( int i = static_cast<int>(kkSize); i > 0; --i )
37 ASSERT_TRUE( m.insert( i ));
38
39 exempt_ptr xp;
40
41 size_t nCount = 0;
42 int nKey = 0;
43 while ( !m.empty()) {
44 xp = m.extract_min();
45 ASSERT_FALSE( !xp );
46 EXPECT_EQ( xp->first.nKey, nKey + 1 );
47 nKey = xp->first.nKey;
48 ++nCount;
49 }
50 xp = m.extract_min();
51 ASSERT_TRUE( !xp );
52 xp = m.extract_max();
53 ASSERT_TRUE( !xp );
54 EXPECT_EQ( kkSize, nCount );
55 ASSERT_TRUE( m.empty());
56 ASSERT_CONTAINER_SIZE( m, 0 );
57
58 // get_max
59 for ( int i = 0; i < static_cast<int>(kkSize); ++i )
60 ASSERT_TRUE( m.insert( i ));
61
62 nKey = kkSize;
63 nCount = 0;
64 while ( !m.empty()) {
65 xp = m.extract_max();
66 ASSERT_FALSE( !xp );
67 EXPECT_EQ( xp->first.nKey, nKey - 1 );
68 nKey = xp->first.nKey;
69 ++nCount;
70 }
71 xp = m.extract_min();
72 ASSERT_TRUE( !xp );
73 xp = m.extract_max();
74 ASSERT_TRUE( !xp );
75 EXPECT_EQ( kkSize, nCount );
76 ASSERT_TRUE( m.empty());
77 ASSERT_CONTAINER_SIZE( m, 0 );
78 }
