28// Notifier allows to wait for arbitrary predicates in non-blocking
29// algorithms. Think of condition variable, but wait predicate does not need to
30// be protected by a mutex. Usage:
31// Waiting thread does:
32//
33// if (predicate)
34// return act();
35// Notifier::Waiter& w = waiters[my_index];
36// ec.prepare_wait(&w);
37// if (predicate) {
38// ec.cancel_wait(&w);
39// return act();
40// }
41// ec.commit_wait(&w);
42//
43// Notifying thread does:
44//
45// predicate = true;
46// ec.notify(true);
47//
48// notify is cheap if there are no waiting threads. prepare_wait/commit_wait are not
49// cheap, but they are executed only if the preceding predicate check has
50// failed.
51//
52// Algorithm outline:
53// There are two main variables: predicate (managed by user) and _state.
54// Operation closely resembles Dekker mutual algorithm:
55// https://en.wikipedia.org/wiki/Dekker%27s_algorithm
56// Waiting thread sets _state then checks predicate, Notifying thread sets
57// predicate then checks _state. Due to seq_cst fences in between these
58// operations it is guaranteed than either waiter will see predicate change
59// and won't block, or notifying thread will see _state change and will unblock
60// the waiter, or both. But it can't happen that both threads don't see each
61// other changes, which would lead to deadlock.

98 // prepare_wait prepares for waiting.
99 // After calling this function the thread must re-check the wait predicate
100 // and call either cancel_wait or commit_wait passing the same Waiter object.
101 void prepare_wait(Waiter* w) {
102 w->epoch = _state.fetch_add(kWaiterInc, std::memory_order_relaxed);
103 std::atomic_thread_fence(std::memory_order_seq_cst);
104 }
105
106 // commit_wait commits waiting.
107 // only the waiter itself can call
108 void commit_wait(Waiter* w) {
109#if __cplusplus >= TF_CPP20
110 w->state.store(Waiter::kNotSignaled, std::memory_order_relaxed);
111#else
112 w->state = Waiter::kNotSignaled;
113#endif
114 // Modification epoch of this waiter.
115 uint64_t epoch =
116 (w->epoch & kEpochMask) +
117 (((w->epoch & kWaiterMask) >> kWaiterShift) << kEpochShift);
118 uint64_t state = _state.load(std::memory_order_seq_cst);
119 for (;;) {
120 if (int64_t((state & kEpochMask) - epoch) < 0) {
121 // The preceding waiter has not decided on its fate. Wait until it
122 // calls either cancel_wait or commit_wait, or is notified.
123 std::this_thread::yield();
124 state = _state.load(std::memory_order_seq_cst);
125 continue;
126 }
127 // We've already been notified.
128 if (int64_t((state & kEpochMask) - epoch) > 0) return;
129 // Remove this thread from prewait counter and add it to the waiter list.
130 assert((state & kWaiterMask) != 0);
131 uint64_t newstate = state - kWaiterInc + kEpochInc;
132 //newstate = (newstate & ~kStackMask) | (w - &_waiters[0]);
133 newstate = static_cast<uint64_t>((newstate & ~kStackMask) | static_cast<uint64_t>(w - &_waiters[0]));
134 if ((state & kStackMask) == kStackMask)
135 w->next.store(nullptr, std::memory_order_relaxed);
136 else
137 w->next.store(&_waiters[state & kStackMask], std::memory_order_relaxed);
138 if (_state.compare_exchange_weak(state, newstate,
139 std::memory_order_release))
140 break;
141 }
142 _park(w);
143 }

145 // cancel_wait cancels effects of the previous prepare_wait call.
146 void cancel_wait(Waiter* w) {
147 uint64_t epoch =
148 (w->epoch & kEpochMask) +
149 (((w->epoch & kWaiterMask) >> kWaiterShift) << kEpochShift);
150 uint64_t state = _state.load(std::memory_order_relaxed);
151 for (;;) {
152 if (int64_t((state & kEpochMask) - epoch) < 0) {
153 // The preceding waiter has not decided on its fate. Wait until it
154 // calls either cancel_wait or commit_wait, or is notified.
155 std::this_thread::yield();
156 state = _state.load(std::memory_order_relaxed);
157 continue;
158 }
159 // We've already been notified.
160 if (int64_t((state & kEpochMask) - epoch) > 0) return;
161 // Remove this thread from prewait counter.
162 assert((state & kWaiterMask) != 0);
163 if (_state.compare_exchange_weak(state, state - kWaiterInc + kEpochInc,
164 std::memory_order_relaxed))
165 return;
166 }
167 }

213 void _park(Waiter* w) {
214#if __cplusplus >= TF_CPP20
215 unsigned target = Waiter::kNotSignaled;
216 if(w->state.compare_exchange_strong(target, Waiter::kWaiting,
217 std::memory_order_relaxed,

221#else

230 void _unpark(Waiter* waiters) {
231 Waiter* next = nullptr;
232 for (Waiter* w = waiters; w; w = next) {
233 next = w->next.load(std::memory_order_relaxed);
234#if __cplusplus >= TF_CPP20
235 // We only notify if the other is waiting - this is why we use tri-state
236 // variable instead of binary-state variable (i.e., atomic_flag)
237 // Performance is about 0.1% faster
238 if(w->state.exchange(Waiter::kSignaled, std::memory_order_relaxed) ==
239 Waiter::kWaiting) {
240 w->state.notify_one();
241 }
242#else
243 unsigned state;
244 {
245 std::unique_lock<std::mutex> lock(w->mu);
246 state = w->state;
247 w->state = Waiter::kSignaled;
248 }
249 // Avoid notifying if it wasn't waiting.
250 if (state == Waiter::kWaiting) w->cv.notify_one();
251#endif
252 }

255 // notify wakes one or all waiting threads.
256 // Must be called after changing the associated wait predicate.
257 template <bool all>
258 void _notify() {
259 std::atomic_thread_fence(std::memory_order_seq_cst);
260 uint64_t state = _state.load(std::memory_order_acquire);
261 for (;;) {
262 // Easy case: no waiters.
263 if ((state & kStackMask) == kStackMask && (state & kWaiterMask) == 0) {
264 return;
265 }
266 uint64_t waiters = (state & kWaiterMask) >> kWaiterShift;
267 uint64_t newstate;
268 if (all) {
269 // Reset prewait counter and empty wait list.
270 newstate = (state & kEpochMask) + (kEpochInc * waiters) + kStackMask;
271 } else if (waiters) {
272 // There is a thread in pre-wait state, unblock it.
273 newstate = state + kEpochInc - kWaiterInc;
274 } else {
275 // Pop a waiter from list and unpark it.
276 Waiter* w = &_waiters[state & kStackMask];
277 Waiter* wnext = w->next.load(std::memory_order_relaxed);
278 uint64_t next = kStackMask;
279 //if (wnext != nullptr) next = wnext - &_waiters[0];
280 if (wnext != nullptr) {
281 next = static_cast<uint64_t>(wnext - &_waiters[0]);
282 }
283 // Note: we don't add kEpochInc here. ABA problem on the lock-free stack
284 // can't happen because a waiter is re-pushed onto the stack only after
285 // it was in the pre-wait state which inevitably leads to epoch
286 // increment.
287 newstate = (state & kEpochMask) + next;
288 }
289 if (_state.compare_exchange_weak(state, newstate,
290 std::memory_order_acquire)) {
291 if (!all && waiters) return; // unblocked pre-wait thread
292 if ((state & kStackMask) == kStackMask) return;
293 Waiter* w = &_waiters[state & kStackMask];
294 if (!all) {
295 w->next.store(nullptr, std::memory_order_relaxed);
296 }
297 _unpark(w);
298 return;
299 }
300 }
301 }

499 // Notify wakes one or all waiting threads.
500 // Must be called after changing the associated wait predicate.
501 template <bool notifyAll>
502 void _notify() {
503 std::atomic_thread_fence(std::memory_order_seq_cst);
504 uint64_t state = _state.load(std::memory_order_acquire);
505 for (;;) {
506 //_check_state(state);
507 const uint64_t waiters = (state & kWaiterMask) >> kWaiterShift;
508 const uint64_t signals = (state & kSignalMask) >> kSignalShift;
509 // Easy case: no waiters.
510 if ((state & kStackMask) == kStackMask && waiters == signals) return;
511 uint64_t newstate;
512 if (notifyAll) {
513 // Empty wait stack and set signal to number of pre-wait threads.
514 newstate = (state & kWaiterMask) | (waiters << kSignalShift) | kStackMask;
515 } else if (signals < waiters) {
516 // There is a thread in pre-wait state, unblock it.
517 newstate = state + kSignalInc;
518 } else {
519 // Pop a waiter from list and unpark it.
520 Waiter* w = &_waiters[state & kStackMask];
521 uint64_t next = w->next.load(std::memory_order_relaxed);
522 newstate = (state & (kWaiterMask | kSignalMask)) | next;
523 }
524 //_check_state(newstate);
525 if (_state.compare_exchange_weak(state, newstate, std::memory_order_acq_rel)) {
526 if (!notifyAll && (signals < waiters)) return; // unblocked pre-wait thread
527 if ((state & kStackMask) == kStackMask) return;
528 Waiter* w = &_waiters[state & kStackMask];
529 if (!notifyAll) w->next.store(kStackMask, std::memory_order_relaxed);
530 _unpark(w);
531 return;
532 }
