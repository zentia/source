768 void clear()
769 {
770 if( !empty()) {
771 deadlock_policy::check();
772
773 node_type * pHead;
774 for (;;) {
775 {
776 rcu_lock l;
777 pHead = m_Head.m_pNext.load(memory_model::memory_order_acquire).ptr();
778 if ( pHead == &m_Tail )
779 break;
780
781 m_Head.m_Lock.lock();
782 pHead->m_Lock.lock();
783
784 if ( m_Head.m_pNext.load(memory_model::memory_order_relaxed).all() == pHead )
785 unlink_node( &m_Head, pHead, &m_Head );
786
787 pHead->m_Lock.unlock();
788 m_Head.m_Lock.unlock();
789 }
790
791 --m_ItemCounter;
792 dispose_node( pHead );
793 }
794 }
795 }

847 {
848 assert( pPred->m_pNext.load( memory_model::memory_order_relaxed ).ptr() == pCur );

880 template <typename Func>
881 bool insert_at( node_type * pHead, value_type& val, Func f )
882 {
883 position pos;
884 key_comparator cmp;
885
886 rcu_lock l;
887 while ( true ) {
888 search( pHead, val, pos );
889 {
890 scoped_position_lock sl( pos );
891 if ( validate( pos.pPred, pos.pCur )) {
892 if ( pos.pCur != &m_Tail && cmp( *node_traits::to_value_ptr( *pos.pCur ), val ) == 0 ) {
893 // failed: key already in list
894 m_Stat.onInsertFailed();
895 return false;
896 }
897
898 f( val );
899 link_node( node_traits::to_node_ptr( val ), pos.pPred, pos.pCur );
900 break;
901 }
902 }
903
904 m_Stat.onInsertRetry();
905 }
906
907 ++m_ItemCounter;
908 m_Stat.onInsertSuccess();
909 return true;
910 }

936 bool unlink_at( node_type * pHead, value_type& val )
937 {
938 position pos;
939 key_comparator cmp;
940 deadlock_policy::check();
941
942 while ( true ) {
943 int nResult = 0;
944 {
945 rcu_lock l;
946 search( pHead, val, pos );
947 {
948 scoped_position_lock alp( pos );
949 if ( validate( pos.pPred, pos.pCur )) {
950 if ( pos.pCur != &m_Tail
951 && cmp( *node_traits::to_value_ptr( *pos.pCur ), val ) == 0
952 && node_traits::to_value_ptr( pos.pCur ) == &val )
953 {
954 // item found
955 unlink_node( pos.pPred, pos.pCur, pHead );
956 nResult = 1;
957 }
958 else
959 nResult = -1;
960 }
961 }
962 }
963
964 if ( nResult ) {
965 if ( nResult > 0 ) {
966 --m_ItemCounter;
967 dispose_node( pos.pCur );
968 m_Stat.onEraseSuccess();
969 return true;
970 }
971
972 m_Stat.onEraseFailed();
973 return false;
974 }
975
976 m_Stat.onEraseRetry();
977 }
