184 template <typename T, bool autoConstruct = true, bool autoDestruct = true>
185 class late_constructed
186 {
187 public:
188 using this_type = late_constructed<T, autoConstruct, autoDestruct>;
189 using value_type = T;
190 using storage_type = eastl::aligned_storage_t<sizeof(value_type), eastl::alignment_of_v<value_type>>;
191
192 late_constructed() EA_NOEXCEPT // In the case of the late_constructed instance being at global scope, we rely on the
193 : mStorage(), mpValue(nullptr) {} // compiler executing this constructor or placing the instance in auto-zeroed-at-startup memory.
194
195 ~late_constructed()
196 {
197 if (autoDestruct && mpValue)
198 (*mpValue).~value_type();
199 }
200
201 template <typename... Args>
202 void construct(Args&&... args)
203 {
204 if(!mpValue)
205 mpValue = new (&mStorage) value_type(eastl::forward<Args>(args)...);
206 }
207
208 bool is_constructed() const EA_NOEXCEPT
209 { return mpValue != nullptr; }
210
211 void destruct()
212 {
213 if(mpValue)
214 {
215 (*mpValue).~value_type();
216 mpValue = nullptr;
217 }
218 }
219
220 value_type& operator*() EA_NOEXCEPT
221 {
222 if(!mpValue)
223 construct();
224
225 EA_ANALYSIS_ASSUME(mpValue);
226 return *mpValue;
227 }
228
229 const value_type& operator*() const EA_NOEXCEPT
230 {
231 if(!mpValue)
232 construct();
233
234 EA_ANALYSIS_ASSUME(mpValue);
235 return *mpValue;
236 }
237
238 value_type* operator->() EA_NOEXCEPT
239 {
240 if(!mpValue)
241 construct();
242 return mpValue;
243 }
244
245 const value_type* operator->() const EA_NOEXCEPT
246 {
247 if(!mpValue)
248 construct();
249 return mpValue;
250 }
251
252 value_type* get() EA_NOEXCEPT
253 {
254 if(!mpValue)
255 construct();
256 return mpValue;
257 }
258
259 const value_type* get() const EA_NOEXCEPT
260 {
261 if(!mpValue)
262 construct();
263 return mpValue;
264 }
265
266 protected:
267 storage_type mStorage; // Declared first because it may have aligment requirements, and it would be more space-efficient if it was first.
268 value_type* mpValue;
269 };

474 template <typename InputIterator, typename ForwardIterator>
475 inline ForwardIterator uninitialized_copy(InputIterator first, InputIterator last, ForwardIterator result)
476 {
477 typedef typename eastl::iterator_traits<InputIterator>::iterator_category IIC;
478 typedef typename eastl::iterator_traits<ForwardIterator>::iterator_category OIC;
479 typedef typename eastl::iterator_traits<InputIterator>::value_type value_type_input;
480 typedef typename eastl::iterator_traits<ForwardIterator>::value_type value_type_output;
481
482 // isTriviallyCopyable identifies if (non-overlapping) objects may be safely copied by means of memcpy/memmove.
483 const bool isTriviallyCopyable = eastl::is_same<value_type_input, value_type_output>::value && eastl::is_trivially_copyable<value_type_output>::value;
484 // ie. is eastl::addressof(*first) valid? ie. invalid for iterators that return value_type&&.
485 const bool isInputIteratorReferenceAddressable = eastl::is_convertible<typename eastl::add_lvalue_reference<value_type_input>::type, typename eastl::iterator_traits<InputIterator>::reference>::value;
486 // can memcpy/memmove a contiguous block, not just the individual elements?
487 const bool areIteratorsContiguous = (eastl::is_pointer<InputIterator>::value || internal::is_contiguous_iterator<IIC>::value) &&
488 (eastl::is_pointer<ForwardIterator>::value || internal::is_contiguous_iterator<OIC>::value);
489
490 return Internal::uninitialized_copy_impl<isTriviallyCopyable, isInputIteratorReferenceAddressable, areIteratorsContiguous>::impl(first, last, result);
491 }

579 template <typename InputIterator, typename ForwardIterator>
580 inline ForwardIterator uninitialized_move_impl(InputIterator first, InputIterator last, ForwardIterator dest, false_type)
581 {
582 typedef typename eastl::iterator_traits<ForwardIterator>::value_type value_type;
583 ForwardIterator currentDest(dest);
584
585 // We must run a loop over every element and move-construct it at the new location.
586 #if EASTL_EXCEPTIONS_ENABLED
587 try
588 {
589 #endif
590 for(; first != last; ++first, ++currentDest)
591 ::new((void*)eastl::addressof(*currentDest)) value_type(eastl::move(*first)); // If value_type has a move constructor then it will be used here.
592 #if EASTL_EXCEPTIONS_ENABLED
593 }
594 catch(...)
595 {
596 // We have a problem here: If an exception occurs while doing the loop below then we will
597 // have values that were moved from the source to the dest that may need to be moved back
598 // in the catch. What does the C++11 Standard say about this? And what happens if there's an
599 // exception while moving them back? We may want to trace through a conforming C++11 Standard
600 // Library to see what it does and do something similar. Given that rvalue references are
601 // objects that are going away, we may not need to move the values back, though that has the
602 // side effect of a certain kind of lost elements problem.
603 for(; dest < currentDest; ++dest)
604 (*dest).~value_type();
605 throw;
606 }
607 #endif
608
609 return currentDest;
610 }
