290 struct value_container
291 {
292 size_t nKeyExpected;
293
294 size_t nSuccessItem = 0;
295 size_t nFailedItem = 0;
296 };
297
298 struct erase_functor {
299 value_container m_cnt;
300
301 void operator ()( keyval_type const& itm )
302 {
303 keyval_type& item = const_cast<keyval_type&>(itm);
304 while ( true ) {
305 bool bBkoff = false;
306 {
307 std::unique_lock< typename value::lock_type> ac( item.val.m_access );
308 if ( item.val.bInitialized ) {
309 if ( m_cnt.nKeyExpected == item.val.nKey && m_cnt.nKeyExpected * 8 == item.val.nData )
310 ++m_cnt.nSuccessItem;
311 else
312 ++m_cnt.nFailedItem;
313 item.val.nData++;
314 item.val.nKey = 0;
315 break;
316 }
317 else
318 bBkoff = true;
319 }
320 if ( bBkoff )
321 cds::backoff::yield()();
322 }
323 }
324 };

389 template <class Set>
390 void run_test( Set& testSet )
391 {
392 typedef Inserter<Set> InserterThread;
393 typedef Deleter<Set> DeleterThread;
394 typedef Updater<Set> UpdaterThread;
395
396 m_pKeyArr.reset( new size_t[ s_nSetSize ] );
397 m_pKeyFirst = m_pKeyArr.get();
398 m_pKeyLast = m_pKeyFirst + s_nSetSize;
399 for ( size_t i = 0; i < s_nSetSize; ++i )
400 m_pKeyArr[i] = i;
401 shuffle( m_pKeyFirst, m_pKeyLast );
402
403 cds_test::thread_pool& pool = get_pool();
404 pool.add( new InserterThread( pool, testSet ), s_nInsertThreadCount );
405 pool.add( new DeleterThread( pool, testSet ), s_nDeleteThreadCount );
406 pool.add( new UpdaterThread( pool, testSet ), s_nUpdateThreadCount );
407
408 propout() << std::make_pair( "insert_thread_count", s_nInsertThreadCount )
409 << std::make_pair( "update_thread_count", s_nUpdateThreadCount )
410 << std::make_pair( "delete_thread_count", s_nDeleteThreadCount )
411 << std::make_pair( "thread_pass_count", s_nThreadPassCount )
412 << std::make_pair( "set_size", s_nSetSize );
413
414 std::chrono::milliseconds duration = pool.run();
415
416 propout() << std::make_pair( "duration", duration );
417
418 size_t nInsertSuccess = 0;
419 size_t nInsertFailed = 0;
420 size_t nDeleteSuccess = 0;
421 size_t nDeleteFailed = 0;
422 size_t nDelValueSuccess = 0;
423 size_t nDelValueFailed = 0;
424 size_t nUpdateFailed = 0;
425 size_t nUpdateCreated = 0;
426 size_t nUpdateModified = 0;
427 size_t nEnsFuncCreated = 0;
428 size_t nEnsFuncModified = 0;
429 size_t nTestFunctorRef = 0;
430
431 for ( size_t i = 0; i < pool.size(); ++i ) {
432 cds_test::thread& thr = pool.get( i );
433 switch ( thr.type()) {
434 case insert_thread:
435 {
436 InserterThread& inserter = static_cast<InserterThread&>( thr );
437 nInsertSuccess += inserter.m_nInsertSuccess;
438 nInsertFailed += inserter.m_nInsertFailed;
439 nTestFunctorRef += inserter.m_nTestFunctorRef;
440 }
441 break;
442 case update_thread:
443 {
444 UpdaterThread& updater = static_cast<UpdaterThread&>(thr);
445 nUpdateCreated += updater.m_nUpdateCreated;
446 nUpdateModified += updater.m_nUpdateExisted;
447 nUpdateFailed += updater.m_nUpdateFailed;
448 nEnsFuncCreated += updater.m_nFunctorCreated;
449 nEnsFuncModified += updater.m_nFunctorModified;
450 }
451 break;
452 case delete_thread:
453 {
454 DeleterThread& deleter = static_cast<DeleterThread&>(thr);
455 nDeleteSuccess += deleter.m_nDeleteSuccess;
456 nDeleteFailed += deleter.m_nDeleteFailed;
457 nDelValueSuccess += deleter.m_nValueSuccess;
458 nDelValueFailed += deleter.m_nValueFailed;
459 }
460 break;
461 }
462 }
463
464 propout()
465 << std::make_pair( "insert_success", nInsertSuccess )
466 << std::make_pair( "delete_success", nDeleteSuccess )
467 << std::make_pair( "insert_failed", nInsertFailed )
468 << std::make_pair( "delete_failed", nDeleteFailed )
469 << std::make_pair( "update_created", nUpdateCreated )
470 << std::make_pair( "update_modified", nUpdateModified )
471 << std::make_pair( "update_failed", nUpdateFailed )
472 << std::make_pair( "final_set_size", testSet.size());
473
474
475 EXPECT_EQ( nDelValueFailed, 0u );
476 EXPECT_EQ( nDelValueSuccess, nDeleteSuccess );
477
478 EXPECT_EQ( nUpdateFailed, 0u );
479 EXPECT_EQ( nUpdateCreated, nEnsFuncCreated );
480 EXPECT_EQ( nUpdateModified, nEnsFuncModified );
481
482 // nTestFunctorRef is call count of insert functor
483 EXPECT_EQ( nTestFunctorRef, nInsertSuccess );
484
485 //testSet.clear();

487 testSet.erase( *p );
488
489 EXPECT_TRUE( testSet.empty());
490 EXPECT_EQ( testSet.size(), 0u );

492 additional_check( testSet );

495 additional_cleanup( testSet );
