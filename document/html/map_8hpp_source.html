61template<typename IN_t , typename OUT_t=IN_t , typename reduceT=int>
62class ff_Map: public ff_node_t<IN_t, OUT_t> {
63 using _node = ff_node_t<IN_t,OUT_t>;
64protected:
65 ParallelForReduce<reduceT> pfr;
66protected:
67 int prepare() {
68 if (!_node::prepared) {
69 // warmup phase
70 pfr.resetskipwarmup();
71 auto r=-1;
72 if (pfr.run_then_freeze() != -1)
73 r = pfr.wait_freezing();
74 if (r<0) {
75 error("ff_Map: preparing ParallelForReduce\n");
76 return -1;
77 }
78
79 if (spinWait) {
80 if (pfr.enableSpinning() == -1) {
81 error("ParallelForReduce: enabling spinwait\n");
82 return -1;
83 }
84 }
85 _node::prepared = true;
86 }
87 return 0;
88 }
89
90 int freeze_and_run(bool=false) {
91 if (!_node::prepared) if (prepare()<0) return -1;
92 return ff_node::freeze_and_run(true);
93 }
