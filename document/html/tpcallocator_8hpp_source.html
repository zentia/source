45class ff_tpcallocator {
46 typedef std::map<const void*, tpc_handle_t> inner_map_t;
47 typedef std::map<const tpc_dev_ctx_t *, inner_map_t> outer_map_t;
48public:
49
50 tpc_handle_t createBuffer(const void *key,
51 tpc_dev_ctx_t *dev_ctx, tpc_device_alloc_flag_t flags, size_t size) {
52
53 tpc_handle_t handle = 0;
54 tpc_device_alloc(dev_ctx, &handle, size, flags);
55 if (handle) allocated[dev_ctx][key] = handle;
56 return handle;
57 }
58
59 // FIX: should be atomic !!
60 tpc_handle_t createBufferUnique(const void *key,
61 tpc_dev_ctx_t *dev_ctx, tpc_device_alloc_flag_t flags, size_t size) {
62 if (allocated.find(dev_ctx) != allocated.end()) {
63 inner_map_t::iterator it = allocated[dev_ctx].find(key);
64 if (it != allocated[dev_ctx].end()) return it->second;
65 }
66 return createBuffer(key,dev_ctx,flags,size);
67 }
68
69
70 //FIX: we need an updateKeyUnique that is atomic
71 void updateKey(const void *oldkey, const void *newkey, tpc_dev_ctx_t *dev_ctx) {
72 assert(allocated.find(dev_ctx) != allocated.end());
73 assert(allocated[dev_ctx].find(oldkey) != allocated[dev_ctx].end());
74
75 tpc_handle_t handle = allocated[dev_ctx][oldkey];
76 allocated[dev_ctx][newkey] = handle;
77 allocated[dev_ctx].erase(oldkey);
78 }
79
80
81 void releaseBuffer(const void *key, tpc_dev_ctx_t *dev_ctx, tpc_handle_t handle) {
82 assert(allocated.find(dev_ctx) != allocated.end());
83 assert(allocated[dev_ctx].find(key) != allocated[dev_ctx].end());
84 tpc_device_free(dev_ctx, handle, TPC_DEVICE_ALLOC_FLAGS_NONE);
85 allocated[dev_ctx].erase(key);
86 }
87
88 void releaseAllBuffers(tpc_dev_ctx_t *dev_ctx) {
89 outer_map_t::iterator it = allocated.find(dev_ctx);
90 if (it == allocated.end()) return;
91
92 inner_map_t::iterator b=it->second.begin();
93 inner_map_t::iterator e=it->second.end();
94 while(b != e) {
95 tpc_device_free(dev_ctx, b->second, TPC_DEVICE_ALLOC_FLAGS_NONE);
96 ++b;
97 }
98 it->second.clear();
99 }
100
101protected:
102 std::map<const tpc_dev_ctx_t *, inner_map_t > allocated;
103};
