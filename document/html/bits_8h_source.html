17// ------------------------------------------------------
18// Size of a pointer.
19// We assume that `sizeof(void*)==sizeof(intptr_t)`
20// and it holds for all platforms we know of.
21//
22// However, the C standard only requires that:
23// p == (void*)((intptr_t)p))
24// but we also need:
25// i == (intptr_t)((void*)i)
26// or otherwise one might define an intptr_t type that is larger than a pointer...
27// ------------------------------------------------------

188static inline size_t mi_popcount(size_t x) {
189 #if defined(__GNUC__) && (MI_ARCH_X64 || MI_ARCH_X86)
190 #if !defined(__BMI1__)
191 if mi_unlikely(!_mi_cpu_has_popcnt) { return _mi_popcount_generic(x); }
192 #endif
193 size_t r;
194 __asm ("popcnt\t%1,%0" : "=r"(r) : "r"(x) : "cc");
195 return r;
196 #elif defined(_MSC_VER) && (MI_ARCH_X64 || MI_ARCH_X86)
197 #if !defined(__BMI1__)
198 if mi_unlikely(!_mi_cpu_has_popcnt) { return _mi_popcount_generic(x); }
199 #endif
200 return (size_t)mi_msc_builtinz(__popcnt)(x);
201 #elif defined(_MSC_VER) && MI_ARCH_ARM64
202 return (size_t)mi_msc_builtinz(__popcnt)(x);
203 #elif mi_has_builtinz(popcount)
204 return mi_builtinz(popcount)(x);
205 #else
206 #define MI_HAS_FAST_POPCOUNT 0
207 return (x<=1 ? x : _mi_popcount_generic(x));
208 #endif
209}

220static inline size_t mi_ctz(size_t x) {
221 #if defined(__GNUC__) && MI_ARCH_X64 && defined(__BMI1__)
222 size_t r;
223 __asm ("tzcnt\t%1, %0" : "=r"(r) : "r"(x) : "cc");
224 return r;
225 #elif defined(__GNUC__) && MI_ARCH_X64
226 // tzcnt is interpreted as bsf if BMI1 is not supported (pre-haswell)
227 // if the argument is zero:
228 // - tzcnt: sets carry-flag, and returns MI_SIZE_BITS
229 // - bsf : sets zero-flag, and leaves the destination _unmodified_ (on both AMD and Intel now, see <https://github.com/llvm/llvm-project/pull/102885>)
230 // so we always initialize r to MI_SIZE_BITS to work correctly on all cpu's without branching
231 size_t r = MI_SIZE_BITS;
232 __asm ("tzcnt\t%1, %0" : "+r"(r) : "r"(x) : "cc"); // use '+r' to keep the assignment to r in case this becomes bsf on older cpu's
233 return r;
234 #elif mi_has_builtinz(ctz)
235 return (x!=0 ? (size_t)mi_builtinz(ctz)(x) : MI_SIZE_BITS);
236 #elif defined(_MSC_VER) && MI_ARCH_X64 && defined(__BMI1__)
237 return (x!=0 ? _tzcnt_u64(x) : MI_SIZE_BITS); // ensure it still works on non-BMI1 cpu's as well
238 #elif defined(_MSC_VER) && (MI_ARCH_X64 || MI_ARCH_X86 || MI_ARCH_ARM64 || MI_ARCH_ARM32)
239 unsigned long idx;
240 return (mi_msc_builtinz(_BitScanForward)(&idx, x) ? (size_t)idx : MI_SIZE_BITS);
241 #elif defined(__GNUC__) && MI_ARCH_X86
242 size_t r = MI_SIZE_BITS;
243 __asm ("bsf\t%1, %0" : "+r"(r) : "r"(x) : "cc");
244 return r;
245 #elif MI_HAS_FAST_POPCOUNT
246 return (x!=0 ? (mi_popcount(x^(x-1))-1) : MI_SIZE_BITS);
247 #else
248 #define MI_HAS_FAST_BITSCAN 0
249 return (x!=0 ? _mi_ctz_generic(x) : MI_SIZE_BITS);
250 #endif
251}

253static inline size_t mi_clz(size_t x) {
254 // we don't optimize anymore to lzcnt as there are still non BMI1 cpu's around (like Intel Celeron, see issue #1016)
255 // on pre-haswell cpu's lzcnt gets executed as bsr which is not equivalent (at it returns the bit position)
256 #if defined(__GNUC__) && MI_ARCH_X64 && defined(__BMI1__) // on x64 lzcnt is defined for 0
257 size_t r;
258 __asm ("lzcnt\t%1, %0" : "=r"(r) : "r"(x) : "cc");
259 return r;
260 #elif mi_has_builtinz(clz)
261 return (x!=0 ? (size_t)mi_builtinz(clz)(x) : MI_SIZE_BITS);
262 #elif defined(_MSC_VER) && (MI_ARCH_X64 || MI_ARCH_X86 || MI_ARCH_ARM64 || MI_ARCH_ARM32)
263 unsigned long idx;
264 return (mi_msc_builtinz(_BitScanReverse)(&idx, x) ? MI_SIZE_BITS - 1 - (size_t)idx : MI_SIZE_BITS);
265 #elif defined(__GNUC__) && (MI_ARCH_X64 || MI_ARCH_X86)
266 if (x==0) return MI_SIZE_BITS;
267 size_t r;
268 __asm ("bsr\t%1, %0" : "=r"(r) : "r"(x) : "cc");
269 return (MI_SIZE_BITS - 1 - r);
270 #else
271 #define MI_HAS_FAST_BITSCAN 0
272 return (x!=0 ? _mi_clz_generic(x) : MI_SIZE_BITS);
273 #endif
274}

284// Bit scan forward: find the least significant bit that is set (i.e. count trailing zero's)
285// return false if `x==0` (with `*idx` undefined) and true otherwise,
286// with the `idx` is set to the bit index (`0 <= *idx < MI_BFIELD_BITS`).
287static inline bool mi_bsf(size_t x, size_t* idx) {
288 // we don't optimize anymore to lzcnt so we run correctly on older cpu's as well
289 #if defined(__GNUC__) && MI_ARCH_X64 && defined(__BMI1__) && (!defined(__clang_major__) || __clang_major__ >= 9)
290 // on x64 the carry flag is set on zero which gives better codegen
291 bool is_zero;
292 __asm ( "tzcnt\t%2, %1" : "=@ccc"(is_zero), "=r"(*idx) : "r"(x) : "cc" );
293 return !is_zero;
294 #elif defined(_MSC_VER) && (MI_ARCH_X64 || MI_ARCH_X86 || MI_ARCH_ARM64 || MI_ARCH_ARM32)
295 unsigned long i;
296 return (mi_msc_builtinz(_BitScanForward)(&i, x) ? (*idx = (size_t)i, true) : false);
297 #else
298 return (x!=0 ? (*idx = mi_ctz(x), true) : false);
299 #endif
300}
301
302// Bit scan reverse: find the most significant bit that is set
303// return false if `x==0` (with `*idx` undefined) and true otherwise,
304// with the `idx` is set to the bit index (`0 <= *idx < MI_BFIELD_BITS`).
305static inline bool mi_bsr(size_t x, size_t* idx) {
306 #if defined(_MSC_VER) && (MI_ARCH_X64 || MI_ARCH_X86 || MI_ARCH_ARM64 || MI_ARCH_ARM32)
307 unsigned long i;
308 return (mi_msc_builtinz(_BitScanReverse)(&i, x) ? (*idx = (size_t)i, true) : false);
309 #else
310 return (x!=0 ? (*idx = MI_SIZE_BITS - 1 - mi_clz(x), true) : false);
311 #endif
312}

319static inline size_t mi_rotr(size_t x, size_t r) {
320 #if (mi_has_builtin(rotateright64) && MI_SIZE_BITS==64)
321 return mi_builtin(rotateright64)(x,r);
322 #elif (mi_has_builtin(rotateright32) && MI_SIZE_BITS==32)
323 return mi_builtin(rotateright32)(x,r);
324 #elif defined(_MSC_VER) && (MI_ARCH_X64 || MI_ARCH_ARM64)
325 return _rotr64(x, (int)r);
326 #elif defined(_MSC_VER) && (MI_ARCH_X86 || MI_ARCH_ARM32)
327 return _lrotr(x,(int)r);
328 #else
329 // The term `(-rshift)&(BITS-1)` is written instead of `BITS - rshift` to
330 // avoid UB when `rshift==0`. See <https://blog.regehr.org/archives/1063>
331 const unsigned int rshift = (unsigned int)(r) & (MI_SIZE_BITS-1);
332 return ((x >> rshift) | (x << ((-rshift) & (MI_SIZE_BITS-1))));
333 #endif
334}
