66 // is_null_pointer
67 //
68 // C++14 type trait. Refers only to nullptr_t and not NULL (0).
69 // eastl::is_null_pointer<nullptr>::value == true
70 // eastl::is_null_pointer<std::nullptr_t>::value == true
71 // eastl::is_null_pointer<void*>::value == false
72 // eastl::is_null_pointer<NULL>::value == [cannot compile]
73 //
75
76 #if defined(EA_COMPILER_CPP11_ENABLED) && !defined(EA_COMPILER_NO_DECLTYPE) && !defined(_MSC_VER) // VC++'s handling of decltype(nullptr) is broken.
77 #define EASTL_TYPE_TRAIT_is_null_pointer_CONFORMANCE 1
78
79 template <typename T>
80 struct is_null_pointer : public eastl::is_same<typename eastl::remove_cv<T>::type, decltype(nullptr)> {}; // A C++11 compiler defines nullptr, but you need a C++11 standard library to declare std::nullptr_t. So it's safer to compare against decltype(nullptr) than to use std::nullptr_t, because we may have a C++11 compiler but C++98 library (happens with Apple frequently).
81 #else
82 #define EASTL_TYPE_TRAIT_is_null_pointer_CONFORMANCE 1
83
84 template <typename T>
85 struct is_null_pointer : public eastl::is_same<typename eastl::remove_cv<T>::type, std::nullptr_t> {};
86 #endif
87
88 #if EASTL_VARIABLE_TEMPLATES_ENABLED
89 template <class T>
90 EA_CONSTEXPR bool is_null_pointer_v = is_null_pointer<T>::value;
91 #endif
