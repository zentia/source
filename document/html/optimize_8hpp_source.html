69static inline int remove_internal_collectors(ff_farm& farm) {
70 const svector<ff_node*>& W = farm.getWorkers();
71 for(size_t i=0;i<W.size();++i) {
72 if (W[i]->isFarm() && !W[i]->isOFarm()) {
73 ff_farm* ifarm = reinterpret_cast<ff_farm*>(W[i]);
74 if (remove_internal_collectors(*ifarm)<0) return -1;
75 if (ifarm->getCollector() == nullptr)
76 ifarm->remove_collector();
77 } else {
78 if (W[i]->isPipe()) {
79 ff_pipeline* ipipe = reinterpret_cast<ff_pipeline*>(W[i]);
80 OptLevel iopt;
81 iopt.remove_collector=true;
82 if (optimize_static(*ipipe, iopt)<0) return -1;
83 }
84 if (W[i]->isAll2All()) {
85 ff_a2a *a2a = reinterpret_cast<ff_a2a*>(W[i]);
86 const svector<ff_node*>& W1 = a2a->getFirstSet();
87 const svector<ff_node*>& W2 = a2a->getSecondSet();
88 for(size_t j=0;j<W1.size();++j) {
89 if (W1[j]->isPipe()) {
90 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W1[j]);
91 OptLevel iopt;
92 iopt.remove_collector=true;
93 if (optimize_static(*ipipe, iopt)<0) return -1;
94 }
95 }
96 for(size_t j=0;j<W2.size();++j) {
97 if (W2[j]->isPipe()) {
98 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W2[j]);
99 OptLevel iopt;
100 iopt.remove_collector=true;
101 if (optimize_static(*ipipe, iopt)<0) return -1;
102 }
103 }
104 }
105 }
106 }
107 return 0;
108}

115static inline int combine_with_emitter(ff_farm& farm, ff_node*node, bool cleanup_node=false) {
116 if (node->isFarm() || node->isPipe() || node->isAll2All()) {
117 error("combine_with_emitter: the node to combine cannot be a parallel building block\n");
118 return -1;
119 }
120 ff_node* emitter = farm.getEmitter();
121 if (!emitter) {
122 farm.add_emitter(node);
123 farm.cleanup_emitter(cleanup_node);
124 return 0;
125 }
126 ff_comb* comb;
127
128 if (!emitter->isMultiOutput()) {
129 internal_mo_transformer *mo_emitter = new internal_mo_transformer(emitter,
130 farm.isset_cleanup_emitter());
131 comb = new ff_comb(node, mo_emitter, cleanup_node, true);
132 } else {
133 comb = new ff_comb(node,emitter,
134 cleanup_node, farm.isset_cleanup_emitter());
135
136 }
137 if (farm.isset_cleanup_emitter())
138 farm.cleanup_emitter(false);
139
140 farm.change_emitter(comb, true);
141 return 0;
142}

144/*
145 * It combines the node passed as parameter with the farm's collector.
146 * The node is added at the right-hand side of the collector.
147 * This transformation is logically equivalent to the following pipeline: ff_Pipe<> pipe(farm, node);
148 */
149static inline int combine_with_collector(ff_farm& farm, ff_node*node, bool cleanup_node=false) {
150 if (!farm.getCollector()) {
151 error("combine_with_collector: the farm passed as parameter does not have a collector\n");
152 return -1;
153 }
154 if (node->isFarm() || node->isPipe() || node->isAll2All()) {
155 error("combine_with_emitter: the node to combine cannot be a parallel building block\n");
156 return -1;
157 }
158 ff_node* collector = farm.getCollector();
159 ff_comb* comb = new ff_comb(collector, node,
160 farm.isset_cleanup_collector(), cleanup_node);
161 if (farm.isset_cleanup_collector())
162 farm.cleanup_collector(false);
163 farm.remove_collector();
164 farm.add_collector(comb, true);
165 return 0;
166}

168/*
169 * It combines the node passed as parameter with the first stage of the pipeline.
170 * The node is added at the left-hand side of the first pipeline node.
171 * This transformation is logically equivalent to the following pipeline: ff_Pipe<> pipe2(node, pipe);
172 */
173template<typename T>
174static inline int combine_with_firststage(ff_pipeline& pipe, T* node, bool cleanup_node) {
175 pipe.flatten();
176 ff_node* node0 = pipe.get_node(0); // it cannot be a pipeline
177 if (!node0) {
178 error("combine_with_firststage: empty pipeline\n");
179 return -1;
180 }
181 if (node0->isAll2All()) {
182 return combine_left_with_a2a(*reinterpret_cast<ff_a2a*>(node0), node, cleanup_node);
183 }
184 if (node0->isFarm()) {
185 ff_farm &farm=*(ff_farm*)node0;
186 if (combine_with_emitter(farm, node, cleanup_node)<0) return -1;
187 pipe.remove_stage(0);
188 pipe.insert_stage(0, node0);
189 } else {
190 ff_comb* comb = new ff_comb(node, node0, cleanup_node);
191 pipe.remove_stage(0);
192 pipe.insert_stage(0, comb, true);
193 }
194
195 return 0;
196}

199template<typename T>
200static inline int combine_right_with_farm(ff_farm& farm, T* node, bool cleanup_node) {
201 if (farm.hasCollector())
202 return combine_with_collector(farm, node, cleanup_node);
203
204 // farm with no collector
205 const svector<ff_node*>& w= farm.getWorkers();
206 assert(w.size()>0);
207
208 if (w[0]->isPipe()) { // NOTE: we suppose that all workers are homogeneous
209 if (combine_with_laststage(*reinterpret_cast<ff_pipeline*>(w[0]), node, cleanup_node)<0) return -1;
210 int r=0;
211 for(size_t i=1;i<w.size();++i) {
212 ff_pipeline* pipe = reinterpret_cast<ff_pipeline*>(w[i]);
213 r+=combine_with_laststage(*pipe, new T(*node), true);
214 }
215 return (r>0?-1:0);
216 }
217 if (w[0]->isFarm()) {
218 if (combine_right_with_farm(*reinterpret_cast<ff_farm*>(w[0]), node, cleanup_node)<0) return -1;
219 int r=0;
220 for(size_t i=1;i<w.size();++i) {
221 ff_farm* farm = reinterpret_cast<ff_farm*>(w[i]);
222 r+=combine_right_with_farm(*farm, new T(*node), true);
223 }
224 return (r>0?-1:0);
225 }
226 if (w[0]->isAll2All()) {
227 if (combine_right_with_a2a(*reinterpret_cast<ff_a2a*>(w[0]), node, cleanup_node)<0) return -1;
228 int r=0;
229 for(size_t i=1;i<w.size();++i) {
230 ff_a2a* a2a = reinterpret_cast<ff_a2a*>(w[i]);
231 r+=combine_right_with_a2a(*a2a, new T(*node), true);
232 }
233 return (r>0?-1:0);
234 }
235 bool workers_cleanup = farm.isset_cleanup_workers();
236 std::vector<ff_node*> new_workers;
237
238 ff_comb* comb = new ff_comb(w[0], node, workers_cleanup, cleanup_node);
239 assert(comb);
240 new_workers.push_back(comb);
241 for(size_t i=1;i<w.size();++i) {
242 ff_comb* c = new ff_comb(w[i], new T(*node), workers_cleanup, true);
243 assert(c);
244 new_workers.push_back(c);
245 }
246 farm.change_workers(new_workers);
247 return 0;
248}
249template<typename T>
250static inline int combine_right_with_a2a(ff_a2a& a2a, T* node, bool cleanup_node) {
251 const svector<ff_node*>& w= a2a.getSecondSet();
252 if (w[0]->isPipe()) { // NOTE: we suppose that all workers are homogeneous
253 if (combine_with_laststage(*reinterpret_cast<ff_pipeline*>(w[0]), node, cleanup_node)<0) return -1;
254 int r=0;
255 for(size_t i=1;i<w.size();++i) {
256 assert(w[i]->isPipe());
257 ff_pipeline* pipe = reinterpret_cast<ff_pipeline*>(w[i]);
258 r+=combine_with_laststage(*pipe, new T(*node), true);
259 }
260 return (r>0?-1:0);
261 }
262 std::vector<ff_node*> new_secondset;
263
264 ff_comb* comb = new ff_comb(w[0], node, false, cleanup_node);
265 assert(comb);
266 new_secondset.push_back(comb);
267 for(size_t i=1;i<w.size();++i) {
268 ff_comb* c = new ff_comb(w[i], new T(*node), false, true);
269 assert(c);
270 new_secondset.push_back(c);
271 }
272 a2a.change_secondset(new_secondset, true);
273 return 0;
274}
275template<typename T>
276static inline int combine_left_with_a2a(ff_a2a& a2a, T* node, bool cleanup_node) {
277 const svector<ff_node*>& w= a2a.getFirstSet();
278 if (w[0]->isPipe()) { // NOTE: we suppose that all workers are homogeneous
279
280 if (combine_with_firststage(*reinterpret_cast<ff_pipeline*>(w[0]), node, cleanup_node)<0)
281 return -1;
282 int r=0;
283 for(size_t i=1;i<w.size();++i) {
284 assert(w[i]->isPipe());
285 ff_pipeline* pipe = reinterpret_cast<ff_pipeline*>(w[i]);
286 r+=combine_with_firststage(*pipe, new T(*node), true);
287 }
288 return (r>0?-1:0);
289 }
290 std::vector<ff_node*> new_firstset;
291 ff_comb* comb = new ff_comb(node, w[0], cleanup_node, false);
292 assert(comb);
293 new_firstset.push_back(comb);
294 for(size_t i=1;i<w.size();++i) {
295 ff_comb* c = new ff_comb(new T(*node), w[i], true, false);
296 assert(c);
297 new_firstset.push_back(c);
298 }
299 a2a.change_firstset(new_firstset, a2a.ondemand_buffer(), true);
300 return 0;
301}

303/*
304 * It combines the node passed as second parameter with the last stage of the pipeline.
305 * The node is added at the right-hand side of the last pipeline stage.
306 * This transformation is logically equivalent to the following pipeline: ff_Pipe<> pipe2(pipe, node);
307 */
308template<typename T>
309static inline int combine_with_laststage(ff_pipeline& pipe, T* node, bool cleanup_node) {
310 pipe.flatten();
311 ff_node* last = pipe.get_lastnode(); // it cannot be a pipeline
312 if (!last) {
313 error("combine_with_laststage: empty pipeline\n");
314 return -1;
315 }
316 if (last->isAll2All()) {
317 return combine_right_with_a2a(*reinterpret_cast<ff_a2a*>(last), node, cleanup_node);
318 }
319 if (last->isFarm()) {
320 return combine_right_with_farm(*reinterpret_cast<ff_farm*>(last), node, cleanup_node);
321 }
322 bool node_cleanup = pipe.isset_cleanup_nodes();
323 int nstages=static_cast<int>(pipe.nodes_list.size());
324 ff_comb* comb = new ff_comb(last, node, node_cleanup , cleanup_node);
325 pipe.remove_stage(nstages-1);
326 pipe.insert_stage((nstages-1)>0?(nstages-1):0, comb, true);
327 return 0;
328}

331/* This is farm specific.
332 * - It basically sets the threshold for enabling blocking mode.
333 * - It can remove the collector of internal farms in a farm of farms composition.
334 * - TODO: Farm of farms ---> single farm with two emitters combined (external+internal)
335 */
336static inline int optimize_static(ff_farm& farm, const OptLevel& opt=OptLevel1()) {
337 if (farm.prepared) {
338 error("optimize_static (farm) called after prepare\n");
339 return -1;
340 }
341 // optimizing internal pipelines, if any
342 OptLevel iopt(opt);
343 iopt.blocking_mode = false;
344 iopt.no_initial_barrier = false;
345 iopt.no_default_mapping = false;
346 const svector<ff_node*> &Workers = farm.getWorkers();
347 for(size_t i=0;i<Workers.size();++i) {
348 if (Workers[i]->isPipe()) {
349 opt_report(opt.verbose_level, OPT_NORMAL,
350 "OPT (farm): Looking for optimizations in the internal pipeline %ld\n",i);
351
352 ff_pipeline *ipipe = reinterpret_cast<ff_pipeline*>(Workers[i]);
353 if (optimize_static(*ipipe, iopt)) return -1;
354 }
355 }
356
357 // here it looks for internal farms with null collectors
358 if (opt.remove_collector) {
359 auto optimize_internal_farm = [opt](ff_farm& ifarm) {
360 OptLevel iopt;
361 iopt.remove_collector=true;
362 iopt.verbose_level = opt.verbose_level;
363 if (optimize_static(ifarm, iopt)<0) return -1;
364 if (ifarm.getCollector() == nullptr) {
365 opt_report(opt.verbose_level, OPT_NORMAL, "OPT (farm): REMOVE_COLLECTOR: Removed farm collector\n");
366 ifarm.remove_collector();
367 }
368 return 0;
369 };
370
371 const svector<ff_node*>& W = farm.getWorkers();
372 for(size_t i=0;i<W.size();++i) {
373 if (W[i]->isFarm() && !W[i]->isOFarm()) {
374 ff_farm* ifarm = reinterpret_cast<ff_farm*>(W[i]);
375 opt_report(opt.verbose_level, OPT_NORMAL,
376 "OPT (farm): Looking for optimizations in the internal farm %ld\n",i);
377
378 if (optimize_internal_farm(*ifarm)<0) return -1;
379 } else {
380 if (W[i]->isPipe()) {
381 ff_pipeline* ipipe = reinterpret_cast<ff_pipeline*>(W[i]);
382 OptLevel iopt;
383 iopt.remove_collector=true;
384 iopt.verbose_level = opt.verbose_level;
385 if (optimize_static(*ipipe, iopt)<0) return -1;
386
387
388#if 0
389 ff_node* last = ipipe->get_lastnode();
390 if (last->isFarm() && !last->isOFarm()) {
391 ff_farm* ifarm = reinterpret_cast<ff_farm*>(last);
392
393 opt_report(opt.verbose_level, OPT_NORMAL,
394 "OPT (farm): Looking for optimizations in the internal last farm of a pipeline %ld\n",i);
395
396 if (optimize_internal_farm(*ifarm)<0) return -1;
397 }
398#endif
399 }
400 if (W[i]->isAll2All()) {
401 ff_a2a *a2a = reinterpret_cast<ff_a2a*>(W[i]);
402 const svector<ff_node*>& W1 = a2a->getFirstSet();
403 const svector<ff_node*>& W2 = a2a->getSecondSet();
404 for(size_t j=0;j<W1.size();++j) {
405 if (W1[j]->isPipe()) {
406 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W1[j]);
407 OptLevel iopt;
408 iopt.remove_collector=true;
409 iopt.verbose_level = opt.verbose_level;
410 if (optimize_static(*ipipe, iopt)<0) return -1;
411 }
412 }
413 for(size_t j=0;j<W2.size();++j) {
414 if (W2[j]->isPipe()) {
415 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W2[j]);
416 OptLevel iopt;
417 iopt.remove_collector=true;
418 iopt.verbose_level = opt.verbose_level;
419 if (optimize_static(*ipipe, iopt)<0) return -1;
420 }
421 }
422 }
423 }
424 }
425 }
426
427 // swithing to blocking mode if the n. of threads is greater than the threshold
428 if (opt.blocking_mode) {
429 ssize_t card = farm.cardinality();
430 if (opt.max_nb_threads < card) {
431 opt_report(opt.verbose_level, OPT_NORMAL,
432 "OPT (farm): BLOCKING_MODE: Activating blocking mode, threshold=%ld, number of threads=%ld\n",opt.max_nb_threads, card);
433
434 farm.blocking_mode(true);
435 }
436 }
437
438 // turning off initial/default mapping if the n. of threads is greater than the threshold
439 if (opt.no_default_mapping) {
440 ssize_t card = farm.cardinality();
441 if (opt.max_mapped_threads < card) {
442 opt_report(opt.verbose_level, OPT_NORMAL,
443 "OPT (farm): MAPPING: Disabling mapping, threshold=%ld, number of threads=%ld\n",opt.max_mapped_threads, card);
444
445 farm.no_mapping();
446 }
447 }
448 // no initial barrier
449 if (opt.no_initial_barrier) {
450 opt_report(opt.verbose_level, OPT_NORMAL,
451 "OPT (farm): NO_INITIAL_BARRIER: Initial barrier disabled\n");
452 farm.no_barrier();
453 }
454 return 0;
455}

460static inline int optimize_static(ff_pipeline& pipe, const OptLevel& opt=OptLevel1()) {
461 if (pipe.prepared) {
462 error("optimize_static (pipeline) called after prepare\n");
463 return -1;
464 }
465
466 // flattening the pipeline
467 pipe.flatten();
468 int nstages=static_cast<int>(pipe.nodes_list.size());
469
470 // looking for farm and all-to-all because they might have pipeline inside
471 // for each nested pipeline the optimize_pipeline function is recursively
472 // called following a depth-first search
473
474 OptLevel iopt(opt);
475 iopt.blocking_mode = false;
476 iopt.no_initial_barrier = false;
477 iopt.no_default_mapping = false;
478 for(int i=0;i<nstages;++i) {
479 if (pipe.nodes_list[i]->isFarm()) {
480 ff_farm *farm = reinterpret_cast<ff_farm*>(pipe.nodes_list[i]);
481 const svector<ff_node*>& W = farm->getWorkers();
482 for(size_t j=0;j<W.size();++j) {
483 if (W[j]->isPipe()) {
484 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W[j]);
485 if (optimize_static(*ipipe, iopt)) return -1;
486 }
487 }
488 } else if (pipe.nodes_list[i]->isAll2All()) {
489 ff_a2a *a2a = reinterpret_cast<ff_a2a*>(pipe.nodes_list[i]);
490 const svector<ff_node*>& W1 = a2a->getFirstSet();
491 const svector<ff_node*>& W2 = a2a->getSecondSet();
492 for(size_t j=0;j<W1.size();++j) {
493 if (W1[j]->isPipe()) {
494 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W1[j]);
495 if (optimize_static(*ipipe, iopt)) return -1;
496 }
497 }
498 for(size_t j=0;j<W2.size();++j) {
499 if (W2[j]->isPipe()) {
500 ff_pipeline* ipipe=reinterpret_cast<ff_pipeline*>(W2[j]);
501 if (optimize_static(*ipipe, iopt)) return -1;
502 }
503 }
504 }
505 }

508 auto find_farm_with_null_collector =
509 [](const svector<ff_node*>& nodeslist, int start=0)->int {
510 for(int i=start;i<(int)(nodeslist.size());++i) {
511 if (nodeslist[i]->isFarm()) {
512 ff_farm *farm = reinterpret_cast<ff_farm*>(nodeslist[i]);
513 if (farm->getCollector() == nullptr) return i;
514 }
515 }
516 return -1;

518 auto find_farm_with_null_emitter =
519 [](const svector<ff_node*>& nodeslist, int start=0)->int {
520 for(int i=start;i<(int)(nodeslist.size());++i) {
521 if (nodeslist[i]->isFarm() &&
522 (nullptr == (reinterpret_cast<ff_farm*>(nodeslist[i]))->getEmitter())
523 )
524 return i;
525 }
526 return -1;

528 // looking for the longest sequence of farms (or ofarms) with the same number of workers
529 auto farm_sequence =
530 [&](const svector<ff_node*>& nodeslist, int& first_farm, int& last_farm) {
531 bool ofarm = nodeslist[first_farm]->isOFarm();
532 size_t nworkers = (reinterpret_cast<ff_farm*>(nodeslist[first_farm]))->getNWorkers();
533 int starting_point=first_farm+1;
534 int first = first_farm, last = last_farm;
535 while (starting_point<static_cast<int>(nodeslist.size())) {
536 bool ok=true;
537 int next = find_farm_with_null_emitter(nodeslist, starting_point);
538 if (next == -1) break;
539 else {
540 for(int i=starting_point; i<=next;) {
541 if ((ofarm?nodeslist[i]->isOFarm():!nodeslist[i]->isOFarm()) &&
542 (nworkers == (reinterpret_cast<ff_farm*>(nodeslist[i]))->getNWorkers()) &&
543 (nullptr == (reinterpret_cast<ff_farm*>(nodeslist[i]))->getCollector()) &&
544 (nullptr == (reinterpret_cast<ff_farm*>(nodeslist[i]))->getEmitter())
545 )
546 ++i;
547 else { ok = false; break; }
548 }
549 if (ok)
550 if ((reinterpret_cast<ff_farm*>(nodeslist[next]))->getEmitter() != nullptr) ok=false;
551 }
552 if (ok) {
553 last = next;
554 starting_point = next+1;
555 } else {
556 if (last==-1) {
557 first = find_farm_with_null_collector(nodeslist, first+1);
558 if (first==-1) break;
559 starting_point = first_farm+1;
560 } break;
561 }
562 }
563 if (first != -1 && last != -1) {
564 first_farm = first;
565 last_farm = last;
566 }
567 };
568 // introduces the normal-form of a sequence of farms (or ofarms)
569 auto combine_farm_sequence =
570 [](const svector<ff_node*>& nodeslist, int first_farm, int last_farm) {
571
572 svector<svector<ff_node*> > W(16);
573 W.resize(last_farm-first_farm+1);
574 for(int i=first_farm, j=0; i<=last_farm; ++i,++j) {
575 W[j]=reinterpret_cast<ff_farm*>(nodeslist[i])->getWorkers();
576 }
577 size_t nfarms = W.size();
578 size_t nworkers = W[0].size();
579
580 std::vector<ff_node*> Workers(nworkers);
581 for(size_t j=0; j<nworkers; ++j) {
582 if (nfarms==2) {
583 ff_comb *p = new ff_comb(W[0][j],W[1][j]);
584 assert(p);
585 Workers[j] = p;
586 } else {
587 const ff_comb *p = new ff_comb(W[0][j],W[1][j]);
588 for(size_t i=2;i<nfarms;++i) {
589 const ff_comb* combtmp = new ff_comb(*p, W[i][j]);
590 assert(combtmp);
591 delete p;
592 p = combtmp;
593 }
594 Workers[j] = const_cast<ff_comb*>(p);
595 }
596 }
597 ff_farm* firstfarm= reinterpret_cast<ff_farm*>(nodeslist[first_farm]);
598 ff_farm* lastfarm = reinterpret_cast<ff_farm*>(nodeslist[last_farm]);
599 ff_farm* newfarm = new ff_farm;
600 if (firstfarm->isOFarm()) {
601 assert(lastfarm->isOFarm());
602 newfarm->set_ordered();
603 }
604
605 if (firstfarm->getEmitter()) newfarm->add_emitter(firstfarm->getEmitter());
606 if (lastfarm->hasCollector())
607 newfarm->add_collector(lastfarm->getCollector());
608 newfarm->add_workers(Workers);

610 newfarm->set_scheduling_ondemand(firstfarm->ondemand_buffer());

613 };

624 ff_farm *newfarm = combine_farm_sequence(pipe.nodes_list,first_farm,last_farm);

632 }while(first_farm!=-1 && first_farm < static_cast<int>(pipe.nodes_list.size()));
633 }

637 while(first_farm != -1 && (first_farm < static_cast<int>(pipe.nodes_list.size()-1))) {

639 if (pipe.nodes_list[first_farm+1]->isFarm() && !pipe.nodes_list[first_farm+1]->isOFarm()) {
640 ff_farm *farm1 = reinterpret_cast<ff_farm*>(pipe.nodes_list[first_farm]);
641 ff_farm *farm2 = reinterpret_cast<ff_farm*>(pipe.nodes_list[first_farm+1]);
642 if (farm2->getEmitter() == nullptr) {
643 opt_report(opt.verbose_level, OPT_NORMAL,
644 "OPT (pipe): INTRODUCE_A2A: Introducing all-to-all between %d and %d stages\n", first_farm, first_farm+1);
645 const ff_farm f = combine_farms_a2a(*farm1, *farm2);
646 ff_farm* newfarm = new ff_farm(f);
647 assert(newfarm);
648 pipe.remove_stage(first_farm);
649 pipe.remove_stage(first_farm);
650 pipe.insert_stage(first_farm, newfarm, true);
651 first_farm = find_farm_with_null_collector(pipe.nodes_list, first_farm+1);
652 }
653 } else {
654 if (pipe.nodes_list[first_farm+1]->isOFarm())
655 opt_report(opt.verbose_level, OPT_INFO,
656 "OPT (pipe): INTRODUCE_A2A: cannot introduce A2A because node %d is an ordered farm\n", first_farm+1);

658 }
659 } else {
660 opt_report(opt.verbose_level, OPT_INFO,

662 first_farm = find_farm_with_null_collector(pipe.nodes_list, first_farm+1);
663 }
664 }

697 }
698 } else {
699 if (!pipe.nodes_list[first_farm+1]->isAll2All()) {

702

712 }
713 }
714 }

716 ff_farm *farm = reinterpret_cast<ff_farm*>(pipe.nodes_list[first_farm]);
717 if (!farm->isOFarm()) {

719 farm->remove_collector();

721 }
722 }
723 }
724 next=first_farm+1;
725 }
726 }
727 if (opt.merge_with_emitter) {
728 int first_farm = find_farm_with_null_emitter(pipe.nodes_list);
729 if (first_farm!=-1) {
730 if (first_farm>0) { // it is not the first one
731 bool prev_single_standard = (!pipe.nodes_list[first_farm-1]->isMultiOutput());
732 if (prev_single_standard) {

745 farm->add_emitter(collector);
746 farm_prev->remove_collector();
747 }
748 } else {

760 }
761 farm->add_emitter(node);
762 pipe.remove_stage(first_farm-1);
763 }
764 }
765 }
766 }
767 }
768
769 // activate blocking mode if the n. of threads is greater than the threshold
770 if (opt.blocking_mode) {
771 ssize_t card = pipe.cardinality();
772 if (opt.max_nb_threads < card) {
773 opt_report(opt.verbose_level, OPT_NORMAL,
774 "OPT (pipe): BLOCKING_MODE: Activating blocking mode, threshold=%ld, number of threads=%ld\n",opt.max_nb_threads, card);
775 pipe.blocking_mode(true);
776 }
777 }
778 // turning off initial/default mapping if the n. of threads is greater than the threshold
779 if (opt.no_default_mapping) {

783 "OPT (pipe): MAPPING: Disabling mapping, threshold=%ld, number of threads=%ld\n",opt.max_mapped_threads, card);
784
785 pipe.no_mapping();
786 }
787 }
788 // no initial barrier
789 if (opt.no_initial_barrier) {
790 opt_report(opt.verbose_level, OPT_NORMAL,
791 "OPT (pipe): NO_INITIAL_BARRIER: Initial barrier disabled\n");
792 pipe.no_barrier();
793 }
794 return 0;
