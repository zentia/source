40 template <typename FreeList, size_t CacheSize = 16, unsigned Padding = cds::c_nCacheLineSize >
41 class CachedFreeList
42 {
43 public:
44 typedef FreeList free_list_type;
45 typedef typename free_list_type::node node;
46
47 static size_t const c_cache_size = CacheSize;
48 static unsigned const c_padding = Padding;

50 static_assert( c_cache_size >= 4, "Cache size is too small" );
51 static_assert( (c_cache_size & (c_cache_size - 1)) == 0, "CacheSize must be power of two" );
52 static_assert( (c_padding & (c_padding - 1)) == 0, "Padding must be power-of-two");

73 void put( node* pNode )
74 {
75 // try to put into free cell of cache
76 node* expect = nullptr;
77 if ( m_cache[ get_hash() ].compare_exchange_weak( expect, pNode, atomics::memory_order_release, atomics::memory_order_relaxed ))
78 return;
79
80 // cache cell is not empty - use free-list
81 m_freeList.put( pNode );
82 }
83
85 node * get()
86 {
87 // try get from cache
88 atomics::atomic<node*>& cell = m_cache[ get_hash() ];
89 node* p = cell.load( atomics::memory_order_relaxed );
90 if ( p && cell.compare_exchange_weak( p, nullptr, atomics::memory_order_acquire, atomics::memory_order_relaxed ))
91 return p;
92
93 // try read from free-list
94 p = m_freeList.get();
95 if ( p )
96 return p;
97
98 // iterate the cache
99 for ( auto& item : m_cache ) {
100 p = item.load( atomics::memory_order_relaxed );
101 if ( p && item.compare_exchange_weak( p, nullptr, atomics::memory_order_acquire, atomics::memory_order_relaxed ))
102 return p;
103 }
104
105 return m_freeList.get();
106 }
107
109 bool empty() const
110 {
111 if ( !m_freeList.empty())
112 return false;
113
114 for ( auto& cell : m_cache ) {
115 node* p = cell.load( atomics::memory_order_relaxed );
116 if ( p )
117 return false;
118 }
119
120 return true;
121 }

136 template <typename Disposer>
137 void clear( Disposer disp )
138 {
139 m_freeList.clear( disp );
140 for ( auto& cell : m_cache ) {
141 node* p = cell.load( atomics::memory_order_relaxed );
142 if ( p ) {
143 disp( p );
144 cell.store( nullptr, atomics::memory_order_relaxed );
145 }
146 }
147 }
148
149 private:
150 //@cond
151 size_t get_hash()
152 {
153 return std::hash<std::thread::id>()( std::this_thread::get_id()) & (c_cache_size - 1);
154 }
155 //@endcond
156 private:
157 //@cond
158 typedef typename cds::details::type_padding< atomics::atomic<node*>, c_padding >::type array_item;
159 array_item m_cache[ c_cache_size ];
160 free_list_type m_freeList;
161 //@endcond
162 };
