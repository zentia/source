267 [[nodiscard]] auto operator()(const compute::ShaderOption &option,
268 uint64_t seed = hash64_default_seed) const noexcept {
269 constexpr auto enable_cache_shift = 0u;
270 constexpr auto enable_fast_math_shift = 1u;
271 constexpr auto enable_debug_info_shift = 2u;
272 constexpr auto compile_only_shift = 3u;
273 auto opt_hash = hash_value((static_cast<uint>(option.enable_cache) << enable_cache_shift) |
274 (static_cast<uint>(option.enable_fast_math) << enable_fast_math_shift) |
275 (static_cast<uint>(option.enable_debug_info) << enable_debug_info_shift) |
276 (static_cast<uint>(option.compile_only) << compile_only_shift),
277 seed);
278 auto name_hash = hash_value(option.name, seed);
279 return hash_combine({opt_hash, name_hash}, seed);
280 }
