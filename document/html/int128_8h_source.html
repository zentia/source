291inline void int128_t_base::SetBit(int nIndex, int value)
292{
293 // EA_ASSERT((nIndex >= 0) && (nIndex < 128));
294
295 const uint64_t nBitMask = ((uint64_t)1 << (nIndex % 64));
296
297 if(nIndex < 64)
298 {
299 if(value)
300 mPart0 = mPart0 | nBitMask;
301 else
302 mPart0 = mPart0 & ~nBitMask;
303 }
304 else if(nIndex < 128)
305 {
306 if(value)
307 mPart1 = mPart1 | nBitMask;
308 else
309 mPart1 = mPart1 & ~nBitMask;
310 }
311}

313inline int int128_t_base::GetBit(int nIndex) const
314{
315 // EA_ASSERT((nIndex >= 0) && (nIndex < 128));
316
317 const uint64_t nBitMask = ((uint64_t)1 << (nIndex % 64));
318
319 if(nIndex < 64)
320 return ((mPart0 & nBitMask) ? 1 : 0);
321 else if(nIndex < 128)
322 return ((mPart1 & nBitMask) ? 1 : 0);
323 return 0;
324}

363// OperatorPlus
364//
365// Returns: (value1 + value2) into result.
366// The output 'result' *is* allowed to point to the same memory as one of the inputs.
367// To consider: Fix 'defect' of this function whereby it doesn't implement overflow wraparound.
368//
369inline void int128_t_base::OperatorPlus(const int128_t_base& value1, const int128_t_base& value2, int128_t_base& result)
370{
371 uint64_t t = value1.mPart0 + value2.mPart0;
372 uint64_t nCarry = (t < value1.mPart0) && (t < value2.mPart0);
373 result.mPart0 = t;
374 result.mPart1 = value1.mPart1 + value2.mPart1 + nCarry;
375}

378// OperatorMinus
379//
380// Returns: (value1 - value2) into result.
381// The output 'result' *is* allowed to point to the same memory as one of the inputs.
382// To consider: Fix 'defect' of this function whereby it doesn't implement overflow wraparound.
383//
384inline void int128_t_base::OperatorMinus(const int128_t_base& value1, const int128_t_base& value2, int128_t_base& result)
385{
386 uint64_t t = (value1.mPart0 - value2.mPart0);
387 uint64_t nCarry = (value1.mPart0 < value2.mPart0) ? 1u : 0u;
388 result.mPart0 = t;
389 result.mPart1 = (value1.mPart1 - value2.mPart1) - nCarry;
390}

393// OperatorMul
394//
395// 64 bit systems:
396// This is how it would be able to work if we could get a 128 bit result from
397// two 64 bit values. None of the 64 bit systems that we are currently working
398// with have C language support for multiplying two 64 bit numbers and retrieving
399// the 128 bit result. However, many 64 bit platforms have support at the asm
400// level for doing such a thing.
401// Part 1 Part 0
402// 0000000000000002 0000000000000001
403// x 0000000000000002 0000000000000001
404// -------------------------------------------
405// | 0000000000000002 0000000000000001
406// + 0000000000000004 | 0000000000000002 (0000000000000000)
407// -------------------------------------------------------------------------
408//
409inline void int128_t_base::OperatorMul(const int128_t_base& a, const int128_t_base& b, int128_t_base& result)
410{
411 // To consider: Use compiler or OS-provided custom functionality here, such as
412 // Windows UnsignedMultiply128 and GCC's built-in int128_t.
413
414 #if defined(DISABLED_PLATFORM_WIN64)
415 // To do: Implement x86-64 asm here.
416
417 #else
418 // Else we are stuck doing something less efficient. In this case we
419 // fall back to doing 32 bit multiplies as with 32 bit platforms.
420 result = (a.mPart0 & 0xffffffff) * (b.mPart0 & 0xffffffff);
421 int128_t v01 = (a.mPart0 & 0xffffffff) * ((b.mPart0 >> 32) & 0xffffffff);
422 int128_t v02 = (a.mPart0 & 0xffffffff) * (b.mPart1 & 0xffffffff);
423 int128_t v03 = (a.mPart0 & 0xffffffff) * ((b.mPart1 >> 32) & 0xffffffff);
424
425 int128_t v10 = ((a.mPart0 >> 32) & 0xffffffff) * (b.mPart0 & 0xffffffff);
426 int128_t v11 = ((a.mPart0 >> 32) & 0xffffffff) * ((b.mPart0 >> 32) & 0xffffffff);
427 int128_t v12 = ((a.mPart0 >> 32) & 0xffffffff) * (b.mPart1 & 0xffffffff);
428
429 int128_t v20 = (a.mPart1 & 0xffffffff) * (b.mPart0 & 0xffffffff);
430 int128_t v21 = (a.mPart1 & 0xffffffff) * ((b.mPart0 >> 32) & 0xffffffff);
431
432 int128_t v30 = ((a.mPart1 >> 32) & 0xffffffff) * (b.mPart0 & 0xffffffff);
433
434 // Do row addition, shifting as needed.
435 OperatorPlus(result, v01 << 32, result);
436 OperatorPlus(result, v02 << 64, result);
437 OperatorPlus(result, v03 << 96, result);
438
439 OperatorPlus(result, v10 << 32, result);
440 OperatorPlus(result, v11 << 64, result);
441 OperatorPlus(result, v12 << 96, result);
442
443 OperatorPlus(result, v20 << 64, result);
444 OperatorPlus(result, v21 << 96, result);
445
446 OperatorPlus(result, v30 << 96, result);
447 #endif
448}

451// OperatorShiftRight
452//
453// Returns: value >> nShift into result
454// The output 'result' may *not* be the same as one the input.
455// With rightward shifts of negative numbers, shift in zero from the left side.
456//
457inline void int128_t_base::OperatorShiftRight(const int128_t_base& value, int nShift, int128_t_base& result)
458{
459 if(nShift >= 0)
460 {
461 if(nShift < 64)
462 { // 0 - 63
463 result.mPart1 = (value.mPart1 >> nShift);
464
465 if(nShift == 0)
466 result.mPart0 = (value.mPart0 >> nShift);
467 else
468 result.mPart0 = (value.mPart0 >> nShift) | (value.mPart1 << (64 - nShift));
469 }
470 else
471 { // 64+
472 result.mPart1 = 0;
473 result.mPart0 = (value.mPart1 >> (nShift - 64));
474 }
475 }
476 else // (nShift < 0)
477 OperatorShiftLeft(value, -nShift, result);
478}

482// OperatorShiftRight
483//
484// Returns: value << nShift into result
485// The output 'result' may *not* be the same as one the input.
486// With rightward shifts of negative numbers, shift in zero from the left side.
487//
488inline void int128_t_base::OperatorShiftLeft(const int128_t_base& value, int nShift, int128_t_base& result)
489{
490 if(nShift >= 0)
491 {
492 if(nShift < 64)
493 {
494 if(nShift) // We need to have a special case because CPUs convert a shift by 64 to a no-op.
495 {
496 // 1 - 63
497 result.mPart0 = (value.mPart0 << nShift);
498 result.mPart1 = (value.mPart1 << nShift) | (value.mPart0 >> (64 - nShift));
499 }
500 else
501 {
502 result.mPart0 = value.mPart0;
503 result.mPart1 = value.mPart1;
504 }
505 }
506 else
507 { // 64+
508 result.mPart0 = 0;
509 result.mPart1 = (value.mPart0 << (nShift - 64));
510 }
511 }
512 else // (nShift < 0)
513 OperatorShiftRight(value, -nShift, result);
514}
