96 template <class Target, class Source>

98 if constexpr (std::is_const_v<Target>) {

102 ::new (_t) Target(std::move(*_s));

104 for (size_t i = 0; i < _s->arr_.size(); ++i) {
105 move_to(&((*_t)[i]), &(_s->arr_[i]));
