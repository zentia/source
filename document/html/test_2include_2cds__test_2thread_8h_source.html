166 std::chrono::milliseconds run( std::chrono::seconds duration )
167 {
168 m_startBarrier.reset( m_workers.size() + 1 );
169 m_stopBarrier.reset( m_workers.size() + 1 );
170
171 // Create threads
172 std::vector< std::thread > threads;
173 threads.reserve( m_workers.size());
174 for ( auto w : m_workers )
175 threads.emplace_back( &thread::run, w );
176
177 // The pool is intialized
178 m_startPoint.ready();
179
180 m_bTimeElapsed.store( false, std::memory_order_release );
181
182 auto native_duration = std::chrono::duration_cast<std::chrono::steady_clock::duration>(duration);
183
184 // The pool is ready to start all workers
185 m_startBarrier.wait();
186
187 auto time_start = std::chrono::steady_clock::now();
188 auto const expected_end = time_start + native_duration;
189
190 if ( duration != std::chrono::seconds::zero()) {
191 for ( ;; ) {
192 std::this_thread::sleep_for( native_duration );
193 auto time_now = std::chrono::steady_clock::now();
194 if ( time_now >= expected_end )
195 break;
196 native_duration = expected_end - time_now;
197 }
198 }
199 m_bTimeElapsed.store( true, std::memory_order_release );
200
201 // Waiting for all workers done
202 m_stopBarrier.wait();
203
204 auto time_end = std::chrono::steady_clock::now();
205
206 for ( auto& t : threads )
207 t.join();
208
209 return m_testDuration = std::chrono::duration_cast<std::chrono::milliseconds>(time_end - time_start);
210 }
