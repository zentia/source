92 size_t const nPassCount = s_nPassCount;
93
94 Map& rMap = m_Map;
95 for ( size_t nPass = 0; nPass < nPassCount; ++nPass ) {
96 if ( id() & 1 ) {
97 auto itEnd = s_Data.cend();
98 for ( auto it = s_Data.cbegin(); it != itEnd; ++it ) {
99 auto bFound = rMap.contains( *(it->pKey));
100 if ( it->bExists ) {
101 if ( check_result(bFound, rMap))
102 ++m_KeyExists.nSuccess;
103 else
104 ++m_KeyExists.nFailed;
105 }
106 else {
107 if ( check_result(bFound, rMap))
108 ++m_KeyNotExists.nFailed;
109 else
110 ++m_KeyNotExists.nSuccess;
111 }
112 }
113 }
114 else {
115 auto itEnd = s_Data.crend();
116 for ( auto it = s_Data.crbegin(); it != itEnd; ++it ) {
117 auto bFound = rMap.contains( *(it->pKey));
118 if ( it->bExists ) {
119 if ( check_result(bFound, rMap))
120 ++m_KeyExists.nSuccess;
121 else
122 ++m_KeyExists.nFailed;
123 }
124 else {
125 if ( check_result( bFound, rMap ))
126 ++m_KeyNotExists.nFailed;
127 else
128 ++m_KeyNotExists.nSuccess;
129 }
130 }
131 }

140 template <class Map>
141 void test( Map& testMap )
142 {
143 typedef Worker<Map> worker;
144
145 // Fill the map
146 for ( size_t i = 0; i < s_Data.size(); ++i ) {
147 // All keys in arrData are unique, insert() must be successful
148 if ( s_Data[i].bExists ) {
149 EXPECT_TRUE( check_result( testMap.insert( *(s_Data[i].pKey), s_Data[i] ), testMap ));
150 }
151 }
152
153 propout() << std::make_pair( "thread_count", s_nThreadCount )
154 << std::make_pair( "map_size", s_nMapSize )
155 << std::make_pair( "percent_exist", s_nPercentExists )
156 << std::make_pair( "pass_count", s_nPassCount );
157
158 cds_test::thread_pool& pool = get_pool();
159 pool.add( new worker( pool, testMap ), s_nThreadCount );
160
161 std::chrono::milliseconds duration = pool.run();
162
163 propout() << std::make_pair( "duration", duration );
164
165 size_t nExistSuccess = 0;
166 size_t nExistFailed = 0;
167 size_t nMissingSuccess = 0;
168 size_t nMissingFailed = 0;
169
170 // Postcondition: the number of success searching == the number of map item
171 for ( size_t i = 0; i < pool.size(); ++i ) {
172 worker& w = static_cast<worker&>(pool.get( i ));
173 nExistSuccess += w.m_KeyExists.nSuccess;
174 nExistFailed += w.m_KeyExists.nFailed;
175 nMissingSuccess += w.m_KeyNotExists.nSuccess;
176 nMissingFailed += w.m_KeyNotExists.nFailed;
177
178 EXPECT_EQ( w.m_KeyExists.nSuccess, s_nMapSize * s_nPassCount ) << "thread " << i;
179 EXPECT_EQ( w.m_KeyExists.nFailed, 0u ) << "thread " << i;
180 EXPECT_EQ( w.m_KeyNotExists.nSuccess, (s_Data.size() - s_nMapSize) * s_nPassCount ) << "thread " << i;
181 EXPECT_EQ( w.m_KeyNotExists.nFailed, 0u ) << "thread " << i;
182 }
183
184 propout()
185 << std::make_pair( "exist_found", nExistSuccess )
186 << std::make_pair( "exist_not_found", nExistFailed ) // must = 0
187 << std::make_pair( "missing_not_found", nMissingSuccess )
188 << std::make_pair( "missing_found", nMissingFailed ); // must = 0
189
190 check_before_cleanup( testMap );
191
192 testMap.clear();
193 additional_check( testMap );
194 print_stat( propout(), testMap );
195 additional_cleanup( testMap );
196 }
