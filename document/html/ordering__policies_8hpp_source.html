48// -------- ondemand load balancer and gatherer -----------------
49// first is the unique id, second.first is the data element
50// second.second is used to store the sender
51using ordering_pair_t = std::pair<size_t, std::pair<void*,ssize_t> >;
52
53struct ordered_lb:ff_loadbalancer {
54 ordered_lb(int max_num_workers):ff_loadbalancer(max_num_workers) {}
55 void init(ordering_pair_t* v, const size_t size) {
56 _M=v; _M_size=size; cnt=0; idx=0;
57 }
58 inline bool schedule_task(void * task, unsigned long retry, unsigned long ticks) {
59 _M[idx].first = cnt;
60 _M[idx].second.first = task;
61 auto r = ff_loadbalancer::schedule_task(&_M[idx], retry, ticks);
62 assert(r);
63 ++cnt; ++idx %= _M_size;
64 return r;
65 }
66 inline void broadcast_task(void * task) {
67 if (task > FF_TAG_MIN) {
68 ff_loadbalancer::broadcast_task(task);
69 return;
70 }
71 _M[idx].first = cnt;
72 _M[idx].second.first = task;
73 ff_loadbalancer::broadcast_task(&_M[idx]);
74 ++cnt; ++idx %= _M_size;
75 }
76 inline bool ff_send_out_to(void *task, int id, unsigned long retry, unsigned long ticks) {
77 assert(task<FF_TAG_MIN);
78 _M[idx].first = cnt;
79 _M[idx].second.first = task;
80 auto r = ff_loadbalancer::ff_send_out_to(&_M[idx], id, retry, ticks);
81 if (r) {++cnt; ++idx %= _M_size;}
82 return r;
83 }
84 size_t idx,cnt,_M_size=0;
85 ordering_pair_t* _M=nullptr;
86};

88struct ordered_gt: ff_gatherer {
89 struct PairCmp {
90 bool operator()(const ordering_pair_t* lhs, const ordering_pair_t* rhs) const {
91 return lhs->first > rhs->first;
92 }
93 };
94 ordered_gt(int max_num_workers): ff_gatherer(max_num_workers) {}
95 void init(const size_t size) { MemSize=size; cnt =0; }
96 inline ssize_t gather_task(void ** task) {
97 if (!Q.empty()) {
98 auto next = Q.top();
99 if (cnt == next->first) {
100 ++cnt;
101 *task = next->second.first;
102 Q.pop();
103 return next->second.second;
104 }
105 }
106 ssize_t nextr= ff_gatherer::gather_task(task);
107 if (*task < FF_TAG_MIN) {
108 ordering_pair_t *in = reinterpret_cast<ordering_pair_t*>(*task);
109 if (cnt == in->first) { // it's the next to send out
110 cnt++;
111 *task = in->second.first;
112 return nextr;
113 }
114 in->second.second = nextr;
115 Q.push(in);
116 if (Q.size()>MemSize) {
117 error("FATAL ERROR: OFARM, ondemand, not enough memory, increase MemoryElements\n");
118 }
119 *task = FF_GO_ON;
120 }
121 return nextr;
122 }
123 int all_gather(void *task, void **V) {
124 ssize_t sender = ff_gatherer::get_channel_id(); // set current sender of element task
125 int r= ff_gatherer::all_gather(task,V);
126 size_t nw = getnworkers();
127 for(size_t i=0;i<nw;++i) {
128 if (V[i]) {
129 if (i!=(size_t)sender)
130 V[i] = (reinterpret_cast<ordering_pair_t*>(V[i]))->second.first;
131 }
132 }
133 return r;
134 }
135
136 size_t cnt;
137 std::priority_queue<ordering_pair_t*,std::vector<ordering_pair_t*>, PairCmp> Q;
138 size_t MemSize;
139};
140// Worker wrapper to be used when ordering_pair_t is added to the data elements
141class OrderedWorkerWrapper: public ff_node_t<ordering_pair_t> {
142public:
143 OrderedWorkerWrapper(ff_node* worker, bool cleanup=false):
144 worker(worker),cleanup(cleanup) {
145 set_barrier(worker->get_barrier());
146 worker->set_barrier(nullptr);
147 }

155 }
156 int svc_init() { return worker->svc_init();}
157 void svc_end() { worker->svc_end(); }
158 void eosnotify(ssize_t id) { worker->eosnotify(id);}
159protected:
160 ff_node* worker;
161 bool cleanup;

187// A node that removes the ordering_pair_t around the data element
188class OrderedCollectorWrapper: public ff_node_t<ordering_pair_t, void> {
189public:
190 struct PairCmp {
191 bool operator()(const ordering_pair_t* lhs, const ordering_pair_t* rhs) const {
192 return lhs->first > rhs->first;
193 }
194 };
195 OrderedCollectorWrapper(const size_t size):cnt(0),MemSize(size) {}
196 int svc_init() {
197 cnt=0;
198 return 0;
199 }
200 inline void* svc(ordering_pair_t* in) {
201 if (cnt == in->first) { // it's the next to send out
202 ff_send_out(in->second.first);
203 cnt++;
204 while(!Q.empty()) {
205 auto next = Q.top();
206 if (cnt == next->first) {
207 ff_send_out(next->second.first);
208 ++cnt;
209 Q.pop();
210 } else break;
211 }
212 return GO_ON;
213 }
214 Q.push(in);
215 if (Q.size()>MemSize) {
216 error("FATAL ERROR: OFARM, ondemand, not enough memory, increase MemoryElements\n");
217 }
218 return GO_ON;
219 }
220 size_t cnt;
221 std::priority_queue<ordering_pair_t*,std::vector<ordering_pair_t*>, PairCmp> Q;
222 size_t MemSize;
223};
