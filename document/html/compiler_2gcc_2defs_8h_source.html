78// double-width CAS support
79// note: gcc-4.8 does not support double-word atomics
80// gcc-4.9: a lot of crashes when use DCAS
81// gcc-7: 128-bit atomic is not lock-free, see https://gcc.gnu.org/ml/gcc/2017-01/msg00167.html
82// You can manually suppress wide-atomic support by defining in compiler command line:
83// for 64bit platform: -DCDS_DISABLE_128BIT_ATOMIC
84// for 32bit platform: -DCDS_DISABLE_64BIT_ATOMIC
85#if CDS_COMPILER_VERSION >= 50000
86# if CDS_BUILD_BITS == 64
87# if !defined( CDS_DISABLE_128BIT_ATOMIC ) && defined( __GCC_HAVE_SYNC_COMPARE_AND_SWAP_16 ) && CDS_COMPILER_VERSION < 70000
88# define CDS_DCAS_SUPPORT
89# endif
90# else
91# if !defined( CDS_DISABLE_64BIT_ATOMIC ) && defined( __GCC_HAVE_SYNC_COMPARE_AND_SWAP_8 )
92# define CDS_DCAS_SUPPORT
93# endif
94# endif
95#endif
