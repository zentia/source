19 template <class GC, typename List, typename Traits = cc::michael_map::traits>
20 class MichaelHashMap : public cc::MichaelHashMap< GC, List, Traits >
21 {
22 typedef cc::MichaelHashMap< GC, List, Traits > base_class;
23 public:
24 template <typename Config>
25 MichaelHashMap( Config const& cfg)
26 : base_class( cfg.s_nMapSize, cfg.s_nLoadFactor )
27 {}
28
29 template <typename Iterator>
30 typename std::enable_if< std::is_same< Iterator, typename base_class::iterator>::value, Iterator>::type
31 get_begin()
32 {
33 return base_class::begin();
34 }
35
36 template <typename Iterator>
37 typename std::enable_if< std::is_same< Iterator, typename base_class::iterator>::value, Iterator>::type
38 get_end()
39 {
40 return base_class::end();
41 }
42
43 // for testing
44 static constexpr bool const c_bExtractSupported = true;
45 static constexpr bool const c_bLoadFactorDepended = true;
46 static constexpr bool const c_bEraseExactKey = false;
47 };
