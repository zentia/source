31 template <typename T, typename Lock=std::mutex>
32 class BoostSList
33 {
34 typedef boost::intrusive::slist< T, boost::intrusive::cache_last<true> > slist_type;
35 typedef Lock lock_type;
36 typedef std::lock_guard<lock_type> lock_guard;
37
38 slist_type m_List;
39 mutable lock_type m_Lock;
40 public:
41 typedef T value_type;
42
43 public:
44 bool push( value_type& v )
45 {
46 lock_guard l( m_Lock );
47 m_List.push_back( v );
48 return true;
49 }
50
51 bool enqueue( value_type& v )
52 {
53 return push( v );
54 }
55
56 value_type * pop()
57 {
58 lock_guard l( m_Lock );
59 if ( m_List.empty())
60 return nullptr;
61 value_type& v = m_List.front();
62 m_List.pop_front();
63 return &v;
64 }
65 value_type * deque()
66 {
67 return pop();
68 }
69
70 bool empty() const
71 {
72 lock_guard l( m_Lock );
73 return m_List.empty();
74 }
75
76 size_t size() const
77 {
78 lock_guard l( m_Lock );
79 return m_List.size();
80 }
81
82 empty_stat statistics() const
83 {
84 return empty_stat();
85 }
86 };
