200 template <bool bTypesAreEqual>
201 struct iter_swap_impl
202 {
203 // Handles the false case, where *a and *b are different types.
204 template <typename ForwardIterator1, typename ForwardIterator2>
205 static void iter_swap(ForwardIterator1 a, ForwardIterator2 b)
206 {
207 typedef typename eastl::iterator_traits<ForwardIterator1>::value_type value_type_a;
208
209 value_type_a temp(EASTL_MOVE(*a)); // EASTL_MOVE uses EASTL::move when available, else is a no-op.
210 *a = EASTL_MOVE(*b);
211 *b = EASTL_MOVE(temp);
212 }
213 };

215 template <>
216 struct iter_swap_impl<true>
217 {
218 template <typename ForwardIterator1, typename ForwardIterator2>
219 static void iter_swap(ForwardIterator1 a, ForwardIterator2 b)
220 {
221 swap(*a, *b); // Don't prefix swap with eastl:: as we want to allow user-defined swaps via argument-dependent lookup.
222 }

238 template <typename ForwardIterator1, typename ForwardIterator2>
239 inline void iter_swap(ForwardIterator1 a, ForwardIterator2 b)
240 {
241 typedef typename eastl::iterator_traits<ForwardIterator1>::value_type value_type_a;
242 typedef typename eastl::iterator_traits<ForwardIterator2>::value_type value_type_b;
243 typedef typename eastl::iterator_traits<ForwardIterator1>::reference reference_a;
244 typedef typename eastl::iterator_traits<ForwardIterator2>::reference reference_b;
245
246 eastl::iter_swap_impl<eastl::conjunction<eastl::is_same<value_type_a, value_type_b>, eastl::is_same<value_type_a&, reference_a>, eastl::is_same<value_type_b&, reference_b>>::value >::iter_swap(a, b);
