215// visit a tuple with a functor at runtime
216// ----------------------------------------------------------------------------

218template <typename Func, typename Tuple, size_t N = 0>
219void visit_tuple(Func func, Tuple& tup, size_t idx) {
220 if (N == idx) {
221 std::invoke(func, std::get<N>(tup));
222 return;
223 }
224 if constexpr (N + 1 < std::tuple_size_v<Tuple>) {
225 return visit_tuple<Func, Tuple, N + 1>(func, tup, idx);
