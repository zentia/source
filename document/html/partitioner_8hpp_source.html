293 template <typename F,
294 std::enable_if_t<std::is_invocable_r_v<void, F, size_t, size_t>, void>* = nullptr
295 >
296 void loop(
297 size_t N, size_t W, std::atomic<size_t>& next, F&& func
298 ) const {
299
300 size_t chunk_size = (this->_chunk_size == 0) ? size_t{1} : this->_chunk_size;
301
302 size_t p1 = 2 * W * (chunk_size + 1);
303 float p2 = 0.5f / static_cast<float>(W);
304 size_t curr_b = next.load(std::memory_order_relaxed);
305
306 while(curr_b < N) {
307
308 size_t r = N - curr_b;
309
310 // fine-grained
311 if(r < p1) {
312 while(1) {
313 curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
314 if(curr_b >= N) {
315 return;
316 }
317 func(curr_b, (std::min)(curr_b + chunk_size, N));
318 }
319 break;
320 }
321 // coarse-grained
322 else {
323 size_t q = static_cast<size_t>(p2 * r);
324 if(q < chunk_size) {
325 q = chunk_size;
326 }
327 //size_t curr_e = (q <= r) ? curr_b + q : N;
328 size_t curr_e = (std::min)(curr_b + q, N);
329 if(next.compare_exchange_strong(curr_b, curr_e, std::memory_order_relaxed,
330 std::memory_order_relaxed)) {
331 func(curr_b, curr_e);
332 curr_b = next.load(std::memory_order_relaxed);
333 }
334 }
335 }
336 }

737 std::pair<size_t, size_t> chunk_size_range(size_t N, size_t W) const {
738
739 size_t b1 = static_cast<size_t>(_alpha * N * W);
740 size_t b2 = static_cast<size_t>(_beta * N * W);
741
742 if(b1 > b2) {
743 std::swap(b1, b2);
744 }
745
746 b1 = (std::max)(b1, size_t{1});
747 b2 = (std::max)(b2, b1 + 1);
748
749 return {b1, b2};
750 }

764 ) const {
765
766 auto [b1, b2] = chunk_size_range(N, W);
767
768 std::default_random_engine engine {std::random_device{}()};
769 std::uniform_int_distribution<size_t> dist(b1, b2);

771 size_t chunk_size = dist(engine);
772 size_t curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
