166 inline void reserve(size_t newcapacity) {
167 if(newcapacity<=cap) return;
168 if (first==NULL)
169 first=(vector_type*)::malloc(sizeof(vector_type)*newcapacity);
170 else
171 first=(vector_type*)::realloc(first,sizeof(vector_type)*newcapacity);
172 cap = newcapacity;
173 }

178 inline void resize(size_t newsize) {

244 inline iterator insert(iterator where, const vector_type & elem) {
