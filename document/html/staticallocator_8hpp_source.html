47class StaticAllocator {
48public:
49 StaticAllocator(const size_t _nslot, const size_t slotsize, const int nchannels=1):
50 ssize(slotsize), nchannels(nchannels), cnts(nchannels,0), segment(nullptr) {
51
52 assert(nchannels>0);
53 assert(slotsize>0);
54 ssize = slotsize + sizeof(long*);
55
56 // rounding up nslot to be multiple of nchannels
57 nslot = ((_nslot + nchannels -1) / nchannels) * nchannels;
58 slotsxchannel = nslot / nchannels;
59 }
60
61 ~StaticAllocator() {
62 if (segment) munmap(segment, nslot*ssize);
63 segment=nullptr;
64 }
65
66 int init() {
67 void* result = 0;
68 result = mmap(NULL, nslot*ssize, PROT_READ|PROT_WRITE, MAP_PRIVATE|MAP_ANONYMOUS, -1, 0);
69 if (result == MAP_FAILED) return -1;
70 segment = (char*)result;
71
72 // initialize the "header"
73 char* p = segment;
74 for(size_t i=0; i<nslot;++i) {
75 long** _p = (long**)p;
76 _p[0] = (long*)FF_GO_ON; // this tells me that the slot is free!
77 p+=ssize;
78 }
79 return 0;
80 }
81
82 template<typename T>
83 void alloc(T*& p, int channel=0) {
84 assert(channel>=0 && channel<nchannels) ;
85 do {
86 size_t m = cnts[channel]++ % slotsxchannel;
87 char *mp = segment + (channel*slotsxchannel + m)*ssize;
88 long** _p = (long**)mp;
89 if (_p[0] == (long*)FF_GO_ON) {
90 _p[0] = (long*)FF_EOS; // the slot is occupied
91 p = new (mp+sizeof(long*)) T();
92 return;
93 }
94 } while(1);
95 }
96
97 template<typename T>
98 static inline void dealloc(T* p) {
99 p->~T();
100 long** _p = (long**)((char*)p-sizeof(long*));
101 _p[0] = (long*)FF_GO_ON; // the slot is free and can be re-used
102 }
103
104 template<typename T, typename S>
105 void realloc(T* in, S*& out) {
106 in->~T();
107 char* mp = reinterpret_cast<char*>(in);
108 out = new (mp) S();
109 }
110
111private:
112 size_t nslot; // total number of slots in the data segment
113 size_t slotsxchannel; // how many slots for each sub data segment
114 size_t ssize; // size of a data slot (real size + sizeof(long*))
115 int nchannels; // number of sub data segments
116 std::vector<size_t> cnts; // counters
117 char *segment; // data segment
118};
119
120};
