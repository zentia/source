338// Procedure: schedule
339inline void Runtime::schedule(Task task) {
340
341 TF_RUNTIME_CHECK_CALLER("schedule must be called by a worker of runtime's executor");
342
343 auto node = task._node;
344 // need to keep the invariant: when scheduling a task, the task must have
345 // zero dependency (join counter is 0)
346 // or we can encounter bug when inserting a nested flow (e.g., module task)
347 node->_join_counter.store(0, std::memory_order_relaxed);
348
349 auto& j = node->_parent ? node->_parent->_join_counter :
350 node->_topology->_join_counter;
351 j.fetch_add(1, std::memory_order_relaxed);
352 _executor._schedule(_worker, node);
353}

422class PreemptionGuard {
423

425
426 PreemptionGuard(Runtime& runtime) : _runtime {runtime} {
427 if(_runtime._preempted == true) {
428 TF_THROW("runtime is not preemptible");
429 }
430 _runtime._parent->_nstate |= NSTATE::PREEMPTED;
431 _runtime._preempted = true;
432 _runtime._parent->_join_counter.fetch_add(1, std::memory_order_release);
433 }
434
435 ~PreemptionGuard() {
436 if(_runtime._parent->_join_counter.fetch_sub(1, std::memory_order_acq_rel) == 1) {
437 _runtime._preempted = false;
438 _runtime._parent->_nstate &= ~NSTATE::PREEMPTED;
439 }
440 }

465// Function: _invoke_runtime_task_impl
466inline bool Executor::_invoke_runtime_task_impl(
467 Worker& worker, Node* node, std::function<void(Runtime&)>& work
468) {
469 // first time
470 if((node->_nstate & NSTATE::PREEMPTED) == 0) {
471
472 Runtime rt(*this, worker, node);
473
474 _observer_prologue(worker, node);
475 TF_EXECUTOR_EXCEPTION_HANDLER(worker, node, {
476 work(rt);
477 });
478 _observer_epilogue(worker, node);
479
480 // here, we cannot check the state from node->_nstate due to data race
481 if(rt._preempted) {
482 return true;
483 }
484 }
485 // second time - previously preempted
486 else {
487 node->_nstate &= ~NSTATE::PREEMPTED;
488 }
489 return false;
490}

492// Function: _invoke_runtime_task_impl
493inline bool Executor::_invoke_runtime_task_impl(
494 Worker& worker, Node* node, std::function<void(Runtime&, bool)>& work
495) {

497 Runtime rt(*this, worker, node);
498
499 // first time
500 if((node->_nstate & NSTATE::PREEMPTED) == 0) {
501
502 _observer_prologue(worker, node);
503 TF_EXECUTOR_EXCEPTION_HANDLER(worker, node, {
504 work(rt, false);
505 });
506 _observer_epilogue(worker, node);
507
508 // here, we cannot check the state from node->_nstate due to data race

512 }

516 }
