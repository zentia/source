74 rfl::Result<T> to_basic_type(const InputVarType _var) const noexcept {
75 const auto get_value = [](const auto& _n) -> std::string {
76 using Type = std::remove_cvref_t<decltype(_n)>;
77 if constexpr (std::is_same<Type, pugi::xml_node>()) {
78 return std::string(_n.child_value());
79 } else {
80 return std::string(_n.value());
81 }
82 };
83
84 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
85 return std::visit(get_value, _var.node_or_attribute_);
86
87 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
88 return std::visit(get_value, _var.node_or_attribute_) == "true";
89
90 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
91 const auto str = std::visit(get_value, _var.node_or_attribute_);
92 try {
93 return static_cast<T>(std::stod(str));
94 } catch (std::exception& e) {
95 return error("Could not cast '" + std::string(str) +
96 "' to floating point value.");
97 }
98
99 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
100 const auto str = std::visit(get_value, _var.node_or_attribute_);
101 try {
102 return static_cast<T>(std::stoi(str));
103 } catch (std::exception& e) {
104 return error("Could not cast '" + std::string(str) + "' to integer.");
105 }
106
107 } else {
108 static_assert(rfl::always_false_v<T>, "Unsupported type.");
