87 inline bool init_input_blocking(pthread_mutex_t *&m,
88 pthread_cond_t *&c,
89 bool /*feedback*/=true) {
90 if (cons_m == nullptr) {
91 assert(cons_c == nullptr);

93 cons_c = (pthread_cond_t*)malloc(sizeof(pthread_cond_t));

98 m = cons_m, c = cons_c;

240 virtual inline bool schedule_task(void * task,
241 unsigned long retry=((unsigned long)-1),
242 unsigned long ticks=TICKS2WAIT) {
243 unsigned long cnt;
244 if (blocking_out) {
245 unsigned long r = 0;
246 do {
247 cnt=0;
248 do {
249 nextw = selectworker();
250 assert(nextw>=0);
251#if defined(LB_CALLBACK)
252 task = callback(nextw, task);
253#endif
254 bool empty=workers[nextw]->get_in_buffer()->empty();
255 if(workers[nextw]->put(task)) {
256 FFTRACE(++taskcnt);
257 if (empty) put_done(nextw);
258 return true;
259 }
260 ++cnt;
261 if (cnt == nattempts()) break;
262 } while(1);
263
264 if (++r >= retry) return false;
265
266 struct timespec tv;
267 timedwait_timeout(tv);
268 pthread_mutex_lock(prod_m);
269 pthread_cond_timedwait(prod_c, prod_m, &tv);
270 pthread_mutex_unlock(prod_m);
271 } while(1);
272 return true;
273 } // blocking
274 do {
275 cnt=0;
276 do {
277 nextw = selectworker();
278 if (nextw<0) return false;
279#if defined(LB_CALLBACK)
280 task = callback(nextw, task);
281#endif
282 if(workers[nextw]->put(task)) {
283 FFTRACE(++taskcnt);
284 return true;
285 }
286 ++cnt;
287 if (cnt>=retry) { nextw=-1; return false; }
288 if (cnt == nattempts()) break;
289 } while(1);
290 losetime_out(ticks);
291 } while(1);
292 return false;
293 }

306 virtual std::deque<ff_node *>::iterator collect_task(void ** task,
307 std::deque<ff_node *> & availworkers,
308 std::deque<ff_node *>::iterator & start) {
309 int cnt, nw= (int)(availworkers.end()-availworkers.begin());
310 const std::deque<ff_node *>::iterator & ite(availworkers.end());
311 do {
312 cnt=0;
313 do {
314 if (++start == ite) start=availworkers.begin();
315 //if (start == ite) start=availworkers.begin(); // read again next time from the same, this needs (**)
316
317 const size_t idx = (start-availworkers.begin());
318 if (filter && !filter->in_active && (idx >= multi_input_start)) continue;
319
320 if((*start)->get(task)) {
321
322 input_channelid = (idx >= multi_input_start) ? (*start)->get_my_id():-1;
323 channelid = idx;
324 if (idx >= multi_input_start) {
325 channelid = -1;
326 // NOTE: the filter can be a multi-input node so this call allows
327 // to set the proper channelid of the gatherer (gt).
328 if (filter) filter->set_input_channelid((ssize_t)(idx-multi_input_start), true);
329 }
330 else {
331 if (idx == (size_t)managerpos) {
332 channelid = (*start)->get_my_id();
333 if (filter) filter->set_input_channelid(channelid, true);
334 } else
335 if (filter) filter->set_input_channelid((ssize_t)idx, false);
336 }
337
338
339 return start;
340 }
341 else {
342 //++start; // (**)
343 if (++cnt == nw) {
344 if (filter && !filter->in_active) { *task=NULL; channelid=-2; return ite;}
345 if (buffer && buffer->pop(task)) {
346 channelid = -1;
347 return ite;
348 }
349 break;
350 }
351 }
352 } while(1);
353 if (blocking_in) {
354 struct timespec tv;
355 timedwait_timeout(tv);
356 pthread_mutex_lock(cons_m);
357 pthread_cond_timedwait(cons_c, cons_m, &tv);
358 pthread_mutex_unlock(cons_m);
359 } else losetime_in();
360 } while(1);
361 return ite;
362 }

373 bool pop(void ** task) {
374 //register int cnt = 0;
375 if (blocking_in) {
376 if (!filter) {
377 while (! buffer->pop(task)) {
378 struct timespec tv;
379 timedwait_timeout(tv);
380 pthread_mutex_lock(cons_m);
381 pthread_cond_timedwait(cons_c, cons_m, &tv);
382 pthread_mutex_unlock(cons_m);
383 } // while
384 } else {
385 if (cons_m) {
386 while (! filter->pop(task)) {
387 struct timespec tv;
388 timedwait_timeout(tv);
389 pthread_mutex_lock(cons_m);
390 pthread_cond_timedwait(cons_c, cons_m, &tv);
391 pthread_mutex_unlock(cons_m);
392 } //while
393 } else {
394 // NOTE:
395 // it may happen that the filter has been transformed
396 // into a multi-output node (e.g., by using internal_mo_transformer,
397 // see the all-to-all) and so the blocking stuff could have been
398 // initialized for the filter and not for the multi-output node
399 // because the transformation has been postponed until the very last
400 // moment (e.g., in the prepare method of the all-to-all)
401 filter->Pop(task);
402 }
403 }
404 return true;
405 }
406 if (!filter)
407 while (! buffer->pop(task)) losetime_in();
408 else
409 while (! filter->pop(task)) losetime_in();
410 return true;
411 }

735 virtual inline bool ff_send_out_to(void *task, int id,
736 unsigned long retry=((unsigned long)-1),
737 unsigned long ticks=(TICKS2WAIT)) {
738 if (blocking_out) {
739 unsigned long r=0;
740 _retry:
741 bool empty=workers[id]->get_in_buffer()->empty();
742 if (workers[id]->put(task)) {
743 FFTRACE(++taskcnt);
744 if (empty) put_done(id);
745 } else {
746 if (++r >= retry) return false;
747 struct timespec tv;
748 timedwait_timeout(tv);
749 pthread_mutex_lock(prod_m);
750 pthread_cond_timedwait(prod_c, prod_m, &tv);
751 pthread_mutex_unlock(prod_m);
752 goto _retry;
753 }
754#if defined(FF_TASK_CALLBACK)
755 callbackOut(this);
756#endif
757 return true;
758 }
759 for(unsigned long i=0;i<retry;++i) {
760 if (workers[id]->put(task)) {
761 FFTRACE(++taskcnt);
762#if defined(FF_TASK_CALLBACK)
763 callbackOut(this);
764#endif
765 return true;
766 }
767 losetime_out(ticks);
768 }
769 return false;
770 }

777 virtual inline void broadcast_task(void * task) {
778 std::vector<size_t> retry;
779 if (blocking_out) {
780 for(ssize_t i=0;i<running;++i) {
781 bool empty=workers[i]->get_in_buffer()->empty();
782 if(!workers[i]->put(task))
783 retry.push_back(i);
784 else if (empty) put_done(i);
785 }
786 while(retry.size()) {
787 bool empty=workers[retry.back()]->get_in_buffer()->empty();
788 if(workers[retry.back()]->put(task)) {
789 if (empty) put_done(retry.back());
790 retry.pop_back();
791 } else {
792 struct timespec tv;
793 timedwait_timeout(tv);
794 pthread_mutex_lock(prod_m);
795 pthread_cond_timedwait(prod_c, prod_m, &tv);
796 pthread_mutex_unlock(prod_m);
797 }
798 }
799#if defined(FF_TASK_CALLBACK)
800 callbackOut(this);
801#endif
802 return;
803 }
804 for(ssize_t i=0;i<running;++i) {
805 if(!workers[i]->put(task))
806 retry.push_back(i);
807 }
808 while(retry.size()) {
809 if(workers[retry.back()]->put(task))
810 retry.pop_back();
811 else losetime_out();
812 }
813#if defined(FF_TASK_CALLBACK)
814 callbackOut(this);
815#endif
816 }

818 int all_gather(void* task, void**V) {
819 V[input_channelid]=task;
820 if (multi_input.size()==0) return -1;
821 size_t _nw=0;
822 svector<ff_node*> _workers(MAX_NUM_THREADS);
823 for(size_t i=0;i<multi_input.size(); ++i) {
824 if (!offline[i]) {
825 ++_nw;
826 _workers.push_back(multi_input[i]);
827 }
828 else _workers.push_back(nullptr);
829 }
830 svector<size_t> retry(_nw);
831 for(size_t i=0;i<_workers.size();++i) {
832 if(i!=(size_t)input_channelid) {
833 if (_workers[i]) {
834 if (!_workers[i]->get(&V[i])) retry.push_back(i);
835 }
836 }
837 }
838 while(retry.size()) {
839 input_channelid = retry.back();
840 if(_workers[input_channelid]->get(&V[input_channelid])) {
841 retry.pop_back();
842 }
843 else {
844 if (blocking_in) {
845 struct timespec tv;
846 timedwait_timeout(tv);
847 pthread_mutex_lock(cons_m);
848 pthread_cond_timedwait(cons_c, cons_m, &tv);
849 pthread_mutex_unlock(cons_m);
850 } else losetime_in();
851 }
852 }
853 bool eos=false;
854 for(size_t i=0;i<_nw;++i) {
855 if (V[i] == FF_EOS || V[i] == FF_EOS_NOFREEZE) {
856 eos=true;
857 offline[i] = true;
858 availworkers.erase(availworkers.begin()+i);
859 V[i]=nullptr;
860 FFTRACE(taskcnt--);
861 }
862 }
863 FFTRACE(taskcnt+=_nw-1);
864 return eos?-1:0;
865 }

898 virtual void * svc(void *) {
899 void * task = NULL;
900 void * ret = FF_EOS;
901 bool inpresent = (get_in_buffer() != NULL);
902 bool skipfirstpop = skip1pop;
903
904 if (!inpresent && filter && (filter->get_in_buffer()!=NULL)) {
905 inpresent = true;
906 set_in_buffer(filter->get_in_buffer());
907 }
908
909 gettimeofday(&wtstart,NULL);
910 if (!master_worker && (multi_input.size()==0) && (inputNodesFeedback.size()==0)) {
911
912 // it is possible that the input queue has been configured as multi-producer
913 // therefore multiple node write in that queue and so the EOS has to be
914 // notified only when 'neos' EOSs have been received. By default neos = 1
915 int neos = filter?filter->neos:1;
916
917 do {
918#ifdef DFF_ENABLED
919 if (!_skipallpop && inpresent){
920#else
921 if (inpresent) {
922#endif
923 if (!skipfirstpop) pop(&task);
924 else skipfirstpop=false;
925
926 // ignoring EOSW in input
927 if (task == FF_EOSW) continue;
928 if (task == FF_EOS) {
929 if (--neos>0) continue;
930 if (filter) filter->eosnotify();
931 push_eos();
932 break;
933 } else if (task == FF_GO_OUT)
934 break;
935 else if (task == FF_EOS_NOFREEZE) {
936 if (--neos>0) continue;
937 if (filter)
938 filter->eosnotify();
939 ret = task;
940 break;
941 }
942 }
943
944 if (filter) {
945 FFTRACE(ticks t0 = getticks());
946
947#if defined(FF_TASK_CALLBACK)
948 callbackIn(this);
949#endif
950 task = filter->svc(task);
951
952
953#if defined(TRACE_FASTFLOW)
954 ticks diff=(getticks()-t0);
955 tickstot +=diff;
956 ticksmin=(std::min)(ticksmin,diff);
957 ticksmax=(std::max)(ticksmax,diff);
958#endif
959
960 if (task == FF_GO_ON) continue;
961 if ((task == FF_GO_OUT) || (task == FF_EOS_NOFREEZE)) {
962 ret = task;
963 break; // exiting from the loop without sending out the task
964 }
965 // if the filter returns NULL/EOS we exit immediatly
966 if (!task || (task==FF_EOS) || (task==FF_EOSW)) { // EOSW is propagated to workers
967 push_eos(task);
968 ret = FF_EOS;
969 break;
970 }
971 } else
972 if (!inpresent) {
973 push_goon();
974 push_eos();
975 ret = FF_EOS;
976 break;
977 }

1018 std::deque<ff_node *>::iterator victim(availworkers.begin());

1022 } else skipfirstpop=false;

1026 break;
1027 }

1035 }

1037 if ((task != FF_EOS_NOFREEZE) && channelid>0 &&
1038 (channelid == managerpos || ((size_t)channelid<workers.size() && !workers[channelid]->isfrozen()))) {
1039
1040 availworkers.erase(victim);
1041 start=availworkers.begin(); // restart iterator
1042 if (channelid == managerpos) {
1043 // the manager has been added as a worker
1044 // so when it terminates we have to decrease the
1045 // starting point of the multi-input channels
1046 multi_input_start -= 1;
1047 }
1048 }
1049 }

1051 // this conditions means that if there is a loop
1052 // we don't want to send an additional
1053 // EOS since all EOSs have already been received
1054 if (!master_worker && (task==FF_EOS) && (inputNodesFeedback.size()==0)) {
1055 push_eos();
1056 }
1057 ret = task;
1058 break; // received all EOS, exit
1059 }
1060 //}
1061 } else {
1062 if (filter) {
1063 FFTRACE(ticks t0 = getticks());
1064
1065#if defined(FF_TASK_CALLBACK)
1066 callbackIn(this);
1067#endif
1068 task = filter->svc(task);
1069
1070#if defined(TRACE_FASTFLOW)
1071 ticks diff=(getticks()-t0);
1072 tickstot +=diff;
1073 ticksmin=(std::min)(ticksmin,diff);
1074 ticksmax=(std::max)(ticksmax,diff);
1075#endif

1078 if ((task == FF_GO_OUT) || (task == FF_EOS_NOFREEZE)){
1079 ret = task;
1080 break; // exiting from the loop without sending out the task
1081 }
1082 // if the filter returns NULL we exit immediatly
1083 if (!task || (task == FF_EOS) || (task == FF_EOSW) ) {
1084 push_eos(task);
1085 // try to remove the additional EOS due to
1086 // the feedback channel
1087
1088 if (inpresent || multi_input.size()>0 || isfrozen()) {
1089 if (master_worker && !parallel_workers) absorb_eos(workers, running);
1090 if (inputNodesFeedback.size()>0) absorb_eos(inputNodesFeedback,
1091 inputNodesFeedback.size());
1092 }
1093 ret = FF_EOS;
1094 break;
1095 }
1096 }
1097 schedule_task(task);
1098#if defined(FF_TASK_CALLBACK)
1099 callbackOut(this);
1100#endif
1101 }
1102 } while(1);
1103 }
1104 gettimeofday(&wtstop,NULL);
1105 wttime+=diffmsec(wtstop,wtstart);
1106
1107 return ret;
