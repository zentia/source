253 template <class Tree>
254 void test( Tree& t )
255 {
256 // Precondition: tree is empty
257 // Postcondition: tree is empty
258
259 ASSERT_TRUE( t.empty());
260 ASSERT_CONTAINER_SIZE( t, 0 );
261 size_t const nTreeSize = kSize;
262
263 typedef typename Tree::value_type value_type;
264
265 std::vector< value_type > data;
266 std::vector< size_t > indices;
267 data.reserve( kSize );
268 indices.reserve( kSize );
269 for ( size_t key = 0; key < kSize; ++key ) {
270 data.push_back( value_type( static_cast<int>( key )));
271 indices.push_back( key );
272 }
273 shuffle( indices.begin(), indices.end());
274
275 // insert/find
276 for ( auto idx : indices ) {
277 auto& i = data[ idx ];
278
279 ASSERT_FALSE( t.contains( i.nKey ));
280 ASSERT_FALSE( t.contains( i ));
281 ASSERT_FALSE( t.contains( other_item( i.key()), other_less()));
282 ASSERT_FALSE( t.find( i.nKey, []( value_type&, int ) {} ));
283 ASSERT_FALSE( t.find_with( other_item( i.key()), other_less(), []( value_type&, other_item const& ) {} ));
284
285 std::pair<bool, bool> updResult;
286
287 updResult = t.update( i, []( bool, value_type&, value_type& )
288 {
289 ASSERT_TRUE( false );
290 }, false );
291 EXPECT_FALSE( updResult.first );
292 EXPECT_FALSE( updResult.second );
293
294 switch ( i.key() % 3 ) {
295 case 0:
296 ASSERT_TRUE( t.insert( i ));
297 ASSERT_FALSE( t.insert( i ));
298 updResult = t.update( i, []( bool bNew, value_type& val, value_type& arg)
299 {
300 EXPECT_FALSE( bNew );
301 EXPECT_EQ( &val, &arg );
302 }, false );
303 EXPECT_TRUE( updResult.first );
304 EXPECT_FALSE( updResult.second );
305 break;
306 case 1:
307 EXPECT_EQ( i.nUpdateNewCount, 0u );
308 ASSERT_TRUE( t.insert( i, []( value_type& v ) { ++v.nUpdateNewCount;} ));
309 EXPECT_EQ( i.nUpdateNewCount, 1u );
310 ASSERT_FALSE( t.insert( i, []( value_type& v ) { ++v.nUpdateNewCount;} ));
311 EXPECT_EQ( i.nUpdateNewCount, 1u );
312 i.nUpdateNewCount = 0;
313 break;
314 case 2:
315 updResult = t.update( i, []( bool, value_type&, value_type& )
316 {
317 EXPECT_TRUE( false );
318 }, false );
319 EXPECT_FALSE( updResult.first );
320 EXPECT_FALSE( updResult.second );
321
322 EXPECT_EQ( i.nUpdateNewCount, 0u );
323 updResult = t.update( i, []( bool bNew, value_type& val, value_type& arg )
324 {
325 EXPECT_TRUE( bNew );
326 EXPECT_EQ( &val, &arg );
327 ++val.nUpdateNewCount;
328 });
329 EXPECT_TRUE( updResult.first );
330 EXPECT_TRUE( updResult.second );
331 EXPECT_EQ( i.nUpdateNewCount, 1u );
332 i.nUpdateNewCount = 0;
333
334 EXPECT_EQ( i.nUpdateCount, 0u );
335 updResult = t.update( i, []( bool bNew, value_type& val, value_type& arg )
336 {
337 EXPECT_FALSE( bNew );
338 EXPECT_EQ( &val, &arg );
339 ++val.nUpdateCount;
340 }, false );
341 EXPECT_TRUE( updResult.first );
342 EXPECT_FALSE( updResult.second );
343 EXPECT_EQ( i.nUpdateCount, 1u );
344 i.nUpdateCount = 0;
345
346 break;
347 }
348
349 ASSERT_TRUE( t.contains( i.nKey ));
350 ASSERT_TRUE( t.contains( i ));
351 ASSERT_TRUE( t.contains( other_item( i.key()), other_less()));
352 EXPECT_EQ( i.nFindCount, 0u );
353 ASSERT_TRUE( t.find( i.nKey, []( value_type& v, int ) { ++v.nFindCount; } ));
354 EXPECT_EQ( i.nFindCount, 1u );
355 ASSERT_TRUE( t.find_with( other_item( i.key()), other_less(), []( value_type& v, other_item const& ) { ++v.nFindCount; } ));
356 EXPECT_EQ( i.nFindCount, 2u );
357 ASSERT_TRUE( t.find( i, []( value_type& v, value_type& ) { ++v.nFindCount; } ));
358 EXPECT_EQ( i.nFindCount, 3u );
359 }
360 ASSERT_FALSE( t.empty());
361 ASSERT_CONTAINER_SIZE( t, nTreeSize );
362
363 std::for_each( data.begin(), data.end(), []( value_type& v ) { v.clear_stat(); });
364
365 // erase
366 shuffle( indices.begin(), indices.end());
367 for ( auto idx : indices ) {
368 auto& i = data[ idx ];
369
370 ASSERT_TRUE( t.contains( i.nKey ));
371 ASSERT_TRUE( t.contains( i ));
372 ASSERT_TRUE( t.contains( other_item( i.key()), other_less()));
373 EXPECT_EQ( i.nFindCount, 0u );
374 ASSERT_TRUE( t.find( i.nKey, []( value_type& v, int ) { ++v.nFindCount; } ));
375 EXPECT_EQ( i.nFindCount, 1u );
376 ASSERT_TRUE( t.find_with( other_item( i.key()), other_less(), []( value_type& v, other_item const& ) { ++v.nFindCount; } ));
377 EXPECT_EQ( i.nFindCount, 2u );
378
379 value_type v( i );
380 switch ( i.key() % 6 ) {
381 case 0:
382 ASSERT_FALSE( t.unlink( v ));
383 ASSERT_TRUE( t.unlink( i ));
384 ASSERT_FALSE( t.unlink( i ));
385 break;
386 case 1:
387 ASSERT_TRUE( t.erase( i.key()));
388 ASSERT_FALSE( t.erase( i.key()));
389 break;
390 case 2:
391 ASSERT_TRUE( t.erase( v ));
392 ASSERT_FALSE( t.erase( v ));
393 break;
394 case 3:
395 ASSERT_TRUE( t.erase_with( other_item( i.key()), other_less()));
396 ASSERT_FALSE( t.erase_with( other_item( i.key()), other_less()));
397 break;
398 case 4:
399 EXPECT_EQ( i.nEraseCount, 0u );
400 ASSERT_TRUE( t.erase( v, []( value_type& val ) { ++val.nEraseCount; } ));
401 EXPECT_EQ( i.nEraseCount, 1u );
402 ASSERT_FALSE( t.erase( v, []( value_type& val ) { ++val.nEraseCount; } ));
403 EXPECT_EQ( i.nEraseCount, 1u );
404 break;
405 case 5:
406 EXPECT_EQ( i.nEraseCount, 0u );
407 ASSERT_TRUE( t.erase_with( other_item( i.key()), other_less(), []( value_type& val ) { ++val.nEraseCount; } ));
408 EXPECT_EQ( i.nEraseCount, 1u );
409 ASSERT_FALSE( t.erase_with( other_item( i.key()), other_less(), []( value_type& val ) { ++val.nEraseCount; } ));
410 EXPECT_EQ( i.nEraseCount, 1u );
411 break;
412 }
413
414 ASSERT_FALSE( t.contains( i.nKey ));
415 ASSERT_FALSE( t.contains( i ));
416 ASSERT_FALSE( t.contains( other_item( i.key()), other_less()));
417 ASSERT_FALSE( t.find( i.nKey, []( value_type&, int ) {} ));
418 ASSERT_FALSE( t.find( i, []( value_type&, value_type const& ) {} ));
419 ASSERT_FALSE( t.find_with( other_item( i.key()), other_less(), []( value_type&, other_item const& ) {} ));
420 }
421 ASSERT_TRUE( t.empty());
422 ASSERT_CONTAINER_SIZE( t, 0 );
423
424 // Force retiring cycle
425 Tree::gc::force_dispose();
426 for ( auto& i : data ) {
427 EXPECT_EQ( i.nDisposeCount, 1u ) << "key=" << i.key();
428 }
429
430 // clear
431 for ( auto idx : indices ) {
432 auto& i = data[idx];
433 i.clear_stat();
434 ASSERT_TRUE( t.insert( i ));
435 }
436 ASSERT_FALSE( t.empty());
437 ASSERT_CONTAINER_SIZE( t, nTreeSize );
438
439 // clear test
440 t.clear();
441
442 ASSERT_TRUE( t.empty());
443 ASSERT_CONTAINER_SIZE( t, 0u );
444
445 // Force retiring cycle
446 Tree::gc::force_dispose();
447 for ( auto& i : data ) {
448 EXPECT_EQ( i.nDisposeCount, 1u ) << "key=" << i.key();
449 }
450 }
451 };
