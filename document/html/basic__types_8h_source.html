243template<size_t N>
244struct hash<Matrix<N>> {
245 using is_avalanching = void;
246 [[nodiscard]] uint64_t operator()(Matrix<N> m, uint64_t seed = hash64_default_seed) const noexcept {
247 std::array<float, N * N> data{};
248 for (size_t i = 0u; i < N; i++) {
249 for (size_t j = 0u; j < N; j++) {
250 data[i * N + j] = m[i][j];
251 }
252 }
253 return hash64(data.data(), data.size() * sizeof(float), seed);
254 }
