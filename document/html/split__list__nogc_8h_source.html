485 iterator insert_( value_type& val )
486 {
487 size_t nHash = hash_value( val );
488 aux_node_type * pHead = get_bucket( nHash );
489 assert( pHead != nullptr );
490
491 node_traits::to_node_ptr( val )->m_nHash = split_list::regular_hash<bit_reversal>( nHash );
492
493 list_iterator it = m_List.insert_at_( pHead, val );
494 if ( it != m_List.end()) {
495 inc_item_count();
496 m_Stat.onInsertSuccess();
497 return iterator( it, m_List.end());
498 }
499 m_Stat.onInsertFailed();
500 return end();

585 size_t bucket_no( size_t nHash ) const
586 {
587 return nHash & ((1 << m_nBucketCountLog2.load( memory_model::memory_order_relaxed )) - 1);
588 }

590 static size_t parent_bucket( size_t nBucket )
591 {
592 assert( nBucket > 0 );
593 return nBucket & ~(1 << bitop::MSBnz( nBucket ));
594 }
595
596 aux_node_type * init_bucket( size_t const nBucket )
597 {
598 assert( nBucket > 0 );
599 size_t nParent = parent_bucket( nBucket );
600
601 aux_node_type * pParentBucket = m_Buckets.bucket( nParent );
602 if ( pParentBucket == nullptr ) {
603 pParentBucket = init_bucket( nParent );
604 m_Stat.onRecursiveInitBucket();
605 }
606
607 assert( pParentBucket != nullptr );
608
609 // Allocate an aux node for new bucket
610 aux_node_type * pBucket = m_Buckets.bucket( nBucket );
611
612 back_off bkoff;
613 for ( ;; pBucket = m_Buckets.bucket( nBucket )) {
614 if ( pBucket )
615 return pBucket;
616
617 pBucket = alloc_aux_node( split_list::dummy_hash<bit_reversal>( nBucket ));
618 if ( pBucket ) {
619 if ( m_List.insert_aux_node( pParentBucket, pBucket )) {
620 m_Buckets.bucket( nBucket, pBucket );
621 m_Stat.onNewBucket();
622 return pBucket;
623 }
624
625 // Another thread set the bucket. Wait while it done
626 free_aux_node( pBucket );
627 m_Stat.onBucketInitContenton();
628 break;
629 }
630
631 // There are no free buckets. It means that the bucket table is full
632 // Wait while another thread set the bucket or a free bucket will be available
633 m_Stat.onBucketsExhausted();
634 bkoff();
635 }
636
637 // Another thread set the bucket. Wait while it done
638 for ( pBucket = m_Buckets.bucket( nBucket ); pBucket == nullptr; pBucket = m_Buckets.bucket( nBucket )) {
639 bkoff();
640 m_Stat.onBusyWaitBucketInit();
641 }
642
643 return pBucket;
644 }

646 aux_node_type * get_bucket( size_t nHash )
647 {
648 size_t nBucket = bucket_no( nHash );
649
650 aux_node_type * pHead = m_Buckets.bucket( nBucket );
651 if ( pHead == nullptr )
652 pHead = init_bucket( nBucket );
653
654 assert( pHead->is_dummy());
655
656 return pHead;
657 }

670 static size_t max_item_count( size_t nBucketCount, size_t nLoadFactor )
671 {
672 return nBucketCount * nLoadFactor;
673 }
674
675 void inc_item_count()
676 {
677 size_t nMaxCount = m_nMaxItemCount.load( memory_model::memory_order_relaxed );
678 if ( ++m_ItemCounter <= nMaxCount )
679 return;
680
681 size_t sz = m_nBucketCountLog2.load( memory_model::memory_order_relaxed );
682 const size_t nBucketCount = static_cast<size_t>(1) << sz;
683 if ( nBucketCount < m_Buckets.capacity()) {
684 // we may grow the bucket table
685 const size_t nLoadFactor = m_Buckets.load_factor();
686 if ( nMaxCount < max_item_count( nBucketCount, nLoadFactor ))
687 return; // someone already have updated m_nBucketCountLog2, so stop here
688
689 m_nMaxItemCount.compare_exchange_strong( nMaxCount, max_item_count( nBucketCount << 1, nLoadFactor ),
690 memory_model::memory_order_relaxed, atomics::memory_order_relaxed );
691 m_nBucketCountLog2.compare_exchange_strong( sz, sz + 1, memory_model::memory_order_relaxed, atomics::memory_order_relaxed );
692 }
693 else
694 m_nMaxItemCount.store( std::numeric_limits<size_t>::max(), memory_model::memory_order_relaxed );
695 }
