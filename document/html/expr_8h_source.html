232template<typename T>
233struct ExprEnableStaticCast {
234 template<typename Dest>
235 requires concepts::static_convertible<
236 expr_value_t<T>, expr_value_t<Dest>>
237 [[nodiscard]] auto cast() const noexcept {
238 auto src = def(*static_cast<const T *>(this));
239 if constexpr (std::is_same_v<expr_value_t<T>,
240 expr_value_t<Dest>>) {
241 return src;
242 }
243 using TrueDest = expr_value_t<Dest>;
244 return def<TrueDest>(
245 FunctionBuilder::current()->cast(
246 Type::of<TrueDest>(),
247 CastOp::STATIC,
248 src.expression()));
249 }
250};
251
253template<typename T>
254struct ExprEnableBitwiseCast {
255 template<typename Dest>
256 requires concepts::bitwise_convertible<
257 expr_value_t<T>, expr_value_t<Dest>>
258 [[nodiscard]] auto as() const noexcept {
259 auto src = def(*static_cast<const T *>(this));
260 using TrueSrc = expr_value_t<T>;
261 using TrueDest = expr_value_t<Dest>;
262 // cast between same types is a no-op
263 if constexpr (std::is_same_v<TrueSrc, TrueDest>) {
264 return src;
265 }
266 // cast between integral types can be reduced to a static cast
267 auto op = CastOp::BITWISE;
268 if constexpr (is_scalar_v<TrueSrc> && is_integral_v<TrueSrc> &&
269 is_scalar_v<TrueDest> && is_integral_v<TrueDest>) {
270 op = CastOp::STATIC;
271 }
272 return def<TrueDest>(
273 FunctionBuilder::current()->cast(
274 Type::of<TrueDest>(),
275 op, src.expression()));
276 }
277 template<typename Dest>
278 requires concepts::bitwise_convertible<
279 expr_value_t<T>, expr_value_t<Dest>>
280 [[nodiscard]] auto bitcast() const noexcept {
281 return as();
282 }
283};

286template<typename T>
287struct ExprEnableSubscriptAccess {
288 template<typename I>
289 requires is_integral_expr_v<I>
290 [[nodiscard]] auto operator[](I &&index) const noexcept {
291 auto self = def<T>(static_cast<const T *>(this)->expression());
292 using Elem = std::remove_cvref_t<
293 decltype(std::declval<expr_value_t<T>>()[0])>;
294 return def<Elem>(FunctionBuilder::current()->access(
295 Type::of<Elem>(), self.expression(),
296 extract_expression(std::forward<I>(index))));
297 }
298};
