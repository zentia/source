118 inline void broadcast_task(void * task) {
119 const svector<ff_node*> &W = getWorkers();
120 if (blocking_out) {
121 size_t nw = getnworkers();
122 for(size_t i=victim;i<nw;++i) {
123 while (!W[i]->put(task)) {
124 pthread_mutex_lock(prod_m);
125 struct timespec tv;
126 timedwait_timeout(tv);
127 pthread_cond_timedwait(prod_c, prod_m,&tv);
128 pthread_mutex_unlock(prod_m);
129 }
130 put_done(i);
131 }
132 for(size_t i=0;i<victim;++i) {
133 while (!W[i]->put(task)) {
134 pthread_mutex_lock(prod_m);
135 struct timespec tv;
136 timedwait_timeout(tv);
137 pthread_cond_timedwait(prod_c, prod_m,&tv);
138 pthread_mutex_unlock(prod_m);
139 }
140 put_done(i);
141 }
142#if defined(FF_TASK_CALLBACK)
143 callbackOut(this);
144#endif
145 return;
146 }

149 while (!W[i]->put(task)) losetime_out();

151 for(size_t i=0;i<victim;++i) {
152 while (!W[i]->put(task)) losetime_out();

156 callbackOut(this);

172 struct ofarm_gt: ff_gatherer {
173 ofarm_gt(int max_num_workers):
174 ff_gatherer(max_num_workers),dead(max_num_workers) {
175 dead.resize(max_num_workers);
176 }
177 inline ssize_t selectworker() { return victim; }
178 void updatenextone() {
179 size_t start = victim;
180 do {
181 victim= (victim+1) % getrunning();
182 } while(dead[victim] && victim != start);
183 }
184 inline ssize_t gather_task(void ** task) {
185 auto nextr = ff_gatherer::gather_task(task);
186 assert((size_t)nextr == victim);
187 if (*task == FF_EOS || *task==FF_EOSW || *task==FF_EOS_NOFREEZE)
188 dead[victim] = true;
189 updatenextone();
190 return nextr;
191 }
192 int svc_init() {
193 for(size_t i=0;i<dead.size();++i) dead[i]=false;
194 return ff_gatherer::svc_init();
195 }
196 void thaw(bool freeze=false, ssize_t nw=-1) {
197 if (nw < (ssize_t)victim) victim = 0;
198 ff_gatherer::thaw(freeze,nw);
199 }
200 size_t victim = 0;
201 svector<bool> dead;

204 inline int cardinality(BARRIER_T * const barrier) {
205 int card=0;
206 for(size_t i=0;i<workers.size();++i)
207 card += workers[i]->cardinality(barrier);

215 inline int prepare() {
216 size_t nworkers = workers.size();
217 if (nworkers==0 || nworkers > max_nworkers) {
218 error("FARM: wrong number of workers\n");
219 return -1;
220 }
221 for(size_t i=0;i<workers.size();++i) {
222 workers[i]->set_id(int(i));
223 }
224
225 // NOTE: if the farm is in a master-worker configuration, all workers must be either
226 // sequential or parallel building block
227 if (lb->masterworker()) {
228 bool is_parallel_bb = workers[0]->isAll2All() || workers[0]->isFarm() || workers[0]->isPipe();
229 lb->parallel_workers = is_parallel_bb;
230 for(size_t i=1;i<workers.size();++i) {
231 bool tmp=workers[i]->isAll2All() || workers[i]->isFarm() || workers[i]->isPipe();
232 if (tmp != is_parallel_bb) {
233 error("FARM, prepare, farm in master-worker configuration but with non homogeneous workers\n");
234 return -1;
235 }
236 }
237 }
238
239 // ordering
240 if (ordered) {
241
242 // TODO: this constraint must be relaxed!!!!!!
243 if (workers[0]->isFarm() || workers[0]->isPipe() || workers[0]->isMultiInput()
244 || workers[0]->isMultiOutput() || workers[0]->isAll2All() || workers[0]->isComp() ) {
245 error("FARM: ordered farm is currently supported only for standard node!\n");
246 return -1;
247 }
248
249 if (ondemand) {
250 ordered_lb* _lb= new ordered_lb(nworkers);
251 ordered_gt* _gt= new ordered_gt(nworkers);
252 assert(_lb); assert(_gt);
253 ordering_Memory.resize(nworkers * (2*ff_farm::ondemand_buffer()+3)+ordering_memsize);
254 _lb->init(ordering_Memory.begin(), ordering_Memory.size());
255 _gt->init(ordering_memsize);
256 setlb(_lb, true);
257 setgt(_gt, true);
258
259 for(size_t i=0;i<nworkers;++i) {
260 workers[i] = new OrderedWorkerWrapper(workers[i], worker_cleanup);
261 assert(workers[i]);
262 }
263 worker_cleanup = true;
264 } else {
265 ofarm_lb* _lb = new ofarm_lb(nworkers);
266 ofarm_gt* _gt = new ofarm_gt(nworkers);
267 assert(lb); assert(gt);
268 setlb(_lb, true);
269 setgt(_gt, true);
270 }
271 }

273 // accelerator
274 if (has_input_channel) {
275 if (create_input_buffer(in_buffer_entries, fixedsizeIN)<0) {
276 error("FARM, creating input buffer\n");
277 return -1;
278 }
279 if (hasCollector()) {
280 // NOTE: the queue is forced to be unbounded
281 if (create_output_buffer(out_buffer_entries, false)<0) return -1;
282 }
283 }

285
286
287 for(size_t i=0;i<nworkers;++i) {
288
289 ff_a2a* a2a_first = nullptr; // the Emitter sees this all-to-all
290 ff_a2a* a2a_last = nullptr; // the Collector sees this all-to-all
291 if (workers[i]->isAll2All()) {
292 a2a_first = reinterpret_cast<ff_a2a*>(workers[i]);
293 a2a_last = a2a_first;
294 } else { // TODO: farm with workers A2A or pipeline ending with A2A
295 if (workers[i]->isPipe()) {
296 ff_pipeline* pipe=reinterpret_cast<ff_pipeline*>(workers[i]);
297 ff_node* node0 = pipe->get_node(0);
298 ff_node* nodeN = pipe->get_lastnode();
299 if (node0->isAll2All()) {
300 a2a_first = reinterpret_cast<ff_a2a*>(node0);
301 }
302 if (nodeN->isAll2All()) {
303 a2a_last = reinterpret_cast<ff_a2a*>(nodeN);
304 }
305 }
306 }
307
308 if (a2a_first) {
309 //NOTE: if the worker is an A2A or a pipe starting with an A2A, the L-Workers
310 // are transformed by the prepare, that's why the prepare must be done
311 // before adding workers to the emitter
312 if (a2a_first->prepare()<0) {
313 error("FARM, preparing worker A2A %d\n", i);
314 return -1;
315 }
316
317
318
319 if (a2a_first->create_input_buffer((int) (ondemand ? ondemand: in_buffer_entries),
320 (ondemand ? true: fixedsizeIN))<0) return -1;
321
322 const svector<ff_node*>& W1 = a2a_first->getFirstSet();
323 for(size_t i=0;i<W1.size();++i) {
324 lb->register_worker(W1[i]);
325 }
326 } else {
327 if (workers[i]->create_input_buffer((int) (ondemand ? ondemand: in_buffer_entries),
328 (ondemand ? true: fixedsizeIN))<0) return -1;
329
330 lb->register_worker(workers[i]);
331 }
332
333 if (a2a_last) {
334
335 if (a2a_first != a2a_last) {
336 //NOTE: if the worker is an A2A or a pipe ending with an A2A, the R-Workers
337 // are transformed by the prepare, that's why the prepare must be done
338 // before adding workers to the collector
339 if (a2a_last->prepare()<0) {
340 error("FARM, preparing worker A2A %d (collector side)\n", i);
341 return -1;
342 }
343 }
344
345 const svector<ff_node*>& W2 = a2a_last->getSecondSet();
346
347 // TODO: If the internal A2A has feedbacks toward the Emitter (i.e. master-worker) and there is also the collector
348 // then this case is not properly handled because R-Workers are not connected to the collector
349 if (collector && !collector_removed) {
350 if (!lb->masterworker()) {
351
352 // NOTE: the following call might fail because the buffers were already created for example by
353 // the pipeline that contains this stage
354 a2a_last->create_output_buffer(out_buffer_entries,(lb->masterworker()?false:fixedsizeOUT));
355
356 for(size_t i=0;i<W2.size();++i) {
357 svector<ff_node*> w(1);
358 W2[i]->get_out_nodes(w);
359 assert(w.size()>0);
360 for(size_t j=0;j<w.size();++j)
361 gt->register_worker(w[j]);
362 }
363 } else {
364 error("FARM feature not yet supported (1)\n");
365 abort(); // <---- FIX
366 }
367 } else { // there is no collector
368 if (outputNodes.size()) {
369 assert(W2.size() == outputNodes.size());
370 for(size_t i=0;i<W2.size();++i) {
371 if (W2[i]->set_output(outputNodes[i])<0) return -1;
372 }
373 } else {
374 // TODO: for the moment we support only feedback channels and not both feedback and forward channels
375 // for the last stages of the last all-to-all
376 if (lb->masterworker()) {
377 if (a2a_last->create_output_buffer(out_buffer_entries,(lb->masterworker()?false:fixedsizeOUT))<0) {
378 error("FARM failed to create feedback channels\n");
379 return -1;
380 }
381
382 for(size_t i=0;i<W2.size();++i) {
383 svector<ff_node*> w(1);
384 W2[i]->get_out_nodes(w);
385 assert(w.size()>0);
386 for(size_t j=0;j<w.size();++j)
387 lb->set_input_feedback(w[j]);
388 }
389 }
390 }
391 }
392
393 continue;
394 }

397 // helper function
398 auto create_feedback_buffers = [&]() {
399 static int idx=0;
400 svector<ff_node*> w(1);
401 workers[i]->get_out_nodes(w);
402 for(size_t j=0;j<w.size();++j) {
403 ff_buffernode* t = new ff_buffernode(out_buffer_entries, false, idx++);
404 assert(t);
405 internalSupportNodes.push_back(t);
406 workers[i]->set_output_feedback(t);
407
408 // REMEMBER: if the worker is not parallel (i.e., farm, a2a, pipeline) then
409 // the lb adds the workers to the 'availworkers' array directly,
410 // so set_input_feedback must be avoided to don't have duplicates.
411 if (lb->parallel_workers) {
412 lb->set_input_feedback(t);
413 } else {
414 workers[i]->set_output_buffer(t->get_out_buffer());
415 }
416 }
417 };
418
419
420 if (collector && !collector_removed) { // there is a collector
421 if (workers[i]->get_out_buffer()==NULL) {
422 if (workers[i]->isMultiOutput()) { // the worker is multi-output
423 if (lb->masterworker()) { // there is a feedback
424 create_feedback_buffers();
425 }
426
427
428 svector<ff_node*> w(MAX_NUM_THREADS);
429 workers[i]->get_out_nodes(w);
430 if (w.size()>0) {
431 static int idx=0;
432
433 for(size_t j=0;j<w.size();++j) {
434 ff_node* t = new ff_buffernode(out_buffer_entries,fixedsizeOUT, idx++);
435 assert(t);
436 internalSupportNodes.push_back(t);
437 if (w[j]->isMultiOutput()) {
438 if (w[j]->set_output(t)<0) return -1;
439 } else {
440 if (workers[i]->set_output(t)<0) return -1;
441 }
442 gt->register_worker(t);
443 }
444 } else { // single node multi-output
445 ff_node* t = new ff_buffernode(out_buffer_entries,fixedsizeOUT, i);
446 internalSupportNodes.push_back(t);
447 workers[i]->set_output(t);
448 if (!lb->masterworker()) workers[i]->set_output_buffer(t->get_out_buffer());
449 gt->register_worker(t);
450 }
451 } else { // standard worker or composition where the second stage is not multi-output
452 if (workers[i]->create_output_buffer(out_buffer_entries,(lb->masterworker()?false:fixedsizeOUT))<0)
453 return -1;
454 assert(!lb->masterworker());
455 gt->register_worker(workers[i]);
456 }
457
458 }
459
460 // this is possible only if the collector filter is a multi-output node
461 if (outputNodes.size()) {
462 assert((collector != (ff_node*)gt) && collector->isMultiOutput());
463 collector->set_output(outputNodes);
464 }
465
466 } else { // there is not a collector
467 if (workers[i]->get_out_buffer()==NULL) {
468 if (workers[i]->isMultiOutput()) {
469 if (lb->masterworker()) { // feedback channels from workers
470 create_feedback_buffers();
471 }
472
473 if (outputNodes.size()) {
474 workers[i]->set_output(outputNodes);
475 }
476
477 } else { // the worker is not multi-output
478 if (outputNodes.size()) {
479 assert(outputNodesFeedback.size()==0);
480 assert(!lb->masterworker()); // no master-worker
481 assert(outputNodes.size() == workers.size()); // same cardinality
482 workers[i]->set_output_buffer(outputNodes[i]->get_in_buffer());
483 }
484 else{
485 if (outputNodesFeedback.size()) {
486 assert(!lb->masterworker()); // no master-worker
487 assert(outputNodesFeedback.size() == workers.size()); // same cardinality
488 workers[i]->set_output_buffer(outputNodesFeedback[i]->get_in_buffer());
489 } else {
490 if (lb->masterworker()) {
491 create_feedback_buffers();
492 }
493 }
494 }
495 }
496 }
497 }
498 }
499
500 // preparing emitter
501 if (emitter) {
502 if (emitter->isMultiOutput()) {
503 // we can use a multi-output node as emitter or a composition where the
504 // last stage is a multi-output node
505 emitter->setlb(lb);
506 }
507 if (lb->set_filter(emitter)<0) {
508 error("FARM, preparing emitter filter\n");
509 return -1;
510 }
511 // if the emitter is a composition we have to call prepare
512 // and to be sure to have a consistent gt
513 if (emitter->isComp()) {
514 svector<ff_node*> w(MAX_NUM_THREADS);
515 lb->get_in_nodes(w);
516 if (w.size()) emitter->set_input(w);
517 if (emitter->prepare()<0) {
518 error("FARM, preparing emitter filter\n");
519 return -1;
520 }
521 }
522 }
523 // preparing collector
524 if (collector && !collector_removed && (collector != (ff_node*)gt) ) {
525 if (collector->isMultiInput()){
526 collector->setgt(gt);
527 }
528 if (collector->isMultiOutput())
529 if (collector->prepare()<0) {
530 error("FARM, preparing multi-output collector filter\n");
531 return -1;
532 }
533 if (collector->isComp())
534 if (collector->prepare()<0) {
535 error("FARM, preparing collector filter\n");
536 return -1;
537 }
538
539 // NOTE: if the collector has a filter and the set_output_blocking has
540 // been already executed (for example by the pipeline), then we
541 // have to set the blocking stuff also for the filter
542 if (gt->set_filter(collector)<0) {
543 error("FARM, preparing collector filter\n");
544 return -1;
545 }
546 }
547
548
549 // blocking stuff --------------------------------
550 for(size_t i=0;i<nworkers;++i) {
551 pthread_mutex_t *m = NULL;
552 pthread_cond_t *c = NULL;
553 if (!workers[i]->init_input_blocking(m,c)) {
554 error("FARM, init input blocking mode for worker %d\n", i);
555 return -1;
556 }
557 if (!workers[i]->init_output_blocking(m,c)) {
558 error("FARM, init output blocking mode for worker %d\n", i);
559 return -1;
560 }
561 }
562
563 // NOTE: if the emitter filter is a multi-output node, it shares
564 // the same lb of the farm.
565 // NOTE: if the collector filter is a multi-input node, it shares
566 // the same gt of the farm.
567 pthread_mutex_t *m = NULL;
568 pthread_cond_t *c = NULL;
569 if (!lb->init_output_blocking(m,c)) {
570 error("FARM, init output blocking mode for LB\n");
571 return -1;
572 }
573 if (collector && !collector_removed) {
574 if (!gt->init_input_blocking(m,c)) {
575 error("FARM, init output blocking mode for GT\n");
576 return -1;
577 }
578 for(size_t j=0;j<nworkers;++j) {
579 svector<ff_node*> w;
580 workers[j]->get_out_nodes(w);
581 assert(w.size()>0);
582 for(size_t i=0;i<w.size(); ++i)
583 w[i]->set_output_blocking(m,c);
584 }
585 }
586
587 if (lb->masterworker()) {
588 bool last_multioutput = [&]() {
589 // WARNING: we consider homogeneous workers!
590 if (lb->parallel_workers) {
591 svector<ff_node*> w;
592 workers[0]->get_out_nodes(w);
593 if (w[0]->isMultiOutput()) return true;
594 return false;
595 }
596 if (workers[0]->isMultiOutput()) return true;
597 return false;
598 } ();
599
600 pthread_mutex_t *m = NULL;
601 pthread_cond_t *c = NULL;
602 if (!init_input_blocking(m,c)) {
603 error("FARM, init input blocking mode for master-worker\n");
604 return -1;
605 }
606 for(size_t j=0;j<nworkers;++j) {
607 svector<ff_node*> w;
608 if (last_multioutput) {
609 workers[j]->get_out_nodes_feedback(w);
610 if (w.size() == 0)
611 workers[j]->get_out_nodes(w);
612 } else
613 workers[j]->get_out_nodes(w);
614 assert(w.size()>0);
615 // NOTE: it is possible that we have to overwrite the
616 // p_cons_* variables that could have been set in the
617 // wrap_around method. This can happen when the last
618 // stage is multi-output and it has a feedback channel
619 // and one (or more) forward channels to the next stage.
620 for(size_t i=0;i<w.size(); ++i)
621 w[i]->set_output_blocking(m,c, true);
622 }
623 }
624
625 if (has_input_channel) {
626 pthread_mutex_t *m = NULL;
627 pthread_cond_t *c = NULL;
628 if (!init_input_blocking(m,c)) {
629 error("FARM, init input blocking\n");
630 return -1;
631 }
632 // this is to notify the Emitter when the queue
633 // is not anymore empty
634 ff_node::set_output_blocking(m,c);
635 // this is to setup the condition variable where
636 // the thread (main?) that is using the accelerator
637 // will sleep if the queue fill up
638 if (!ff_node::init_output_blocking(m,c)) {
639 error("FARM, init output blocking\n");
640 return -1;
641 }
642
643 if (hasCollector()) {
644 m=NULL,c=NULL;
645 if (!init_output_blocking(m,c)) {
646 error("FARM, init output blocking\n");
647 return -1;
648 }
649
650 m=NULL,c=NULL;
651 if (!ff_node::init_input_blocking(m,c)) {
652 error("FARM, init input blocking\n");
653 return -1;
654 }
655 set_output_blocking(m,c);
656 }
657 }
658
659
660 prepared=true;
661 return 0;
662 }

685 virtual inline bool init_input_blocking(pthread_mutex_t *&m,
686 pthread_cond_t *&c,
687 bool /*feedback*/=true) {
688 bool r = lb->init_input_blocking(m,c);
689 if (!r) return false;
690 // NOTE: for all registered input node (or buffernode) we have to set the
691 // blocking stuff (see also ff_minode::init_input_blocking)
692 svector<ff_node*> w;
693 lb->get_in_nodes(w);
694 lb->get_in_nodes_feedback(w);
695 for(size_t i=0;i<w.size();++i)
696 w[i]->set_output_blocking(m,c);
697 return true;
698 }
699 // producer
700 virtual inline bool init_output_blocking(pthread_mutex_t *&m,
701 pthread_cond_t *&c,
702 bool /*feedback*/=true) {
703 if (collector && !collector_removed) {
704 if (collector == (ff_node*)gt)
705 return gt->init_output_blocking(m,c);
706 return collector->init_output_blocking(m,c);
707 }
708 for(size_t i=0;i<workers.size();++i)
709 if (!workers[i]->init_output_blocking(m,c)) return false;
710
711 return true;
712 }
713 virtual inline void set_output_blocking(pthread_mutex_t *&m,
714 pthread_cond_t *&c,
715 bool canoverwrite=false) {
716 if (collector && !collector_removed) {
717 if (collector == (ff_node*)gt)
718 gt->set_output_blocking(m,c, canoverwrite);
719 else
720 collector->set_output_blocking(m,c, canoverwrite);
721 }
