876 size_t bucket_no( size_t nHash ) const
877 {
878 return nHash & ( (1 << m_nBucketCountLog2.load(memory_model::memory_order_relaxed)) - 1 );
879 }
880
881 static size_t parent_bucket( size_t nBucket )
882 {
883 assert( nBucket > 0 );
884 return nBucket & ~( 1 << bitop::MSBnz( nBucket ));
885 }
886
887 aux_node_type * init_bucket( size_t const nBucket )
888 {
889 assert( nBucket > 0 );
890 size_t nParent = parent_bucket( nBucket );
891
892 aux_node_type * pParentBucket = m_Buckets.bucket( nParent );
893 if ( pParentBucket == nullptr ) {
894 pParentBucket = init_bucket( nParent );
895 m_Stat.onRecursiveInitBucket();
896 }
897
898 assert( pParentBucket != nullptr );
899
900 // Allocate an aux node for new bucket
901 aux_node_type * pBucket = m_Buckets.bucket( nBucket );
902
903 back_off bkoff;
904 for ( ;; pBucket = m_Buckets.bucket( nBucket )) {
905 if ( pBucket )
906 return pBucket;
907
908 pBucket = alloc_aux_node( split_list::dummy_hash<bit_reversal>( nBucket ));
909 if ( pBucket ) {
910 if ( m_List.insert_aux_node( pParentBucket, pBucket )) {
911 m_Buckets.bucket( nBucket, pBucket );
912 m_Stat.onNewBucket();
913 return pBucket;
914 }
915
916 // Another thread set the bucket. Wait while it done
917 free_aux_node( pBucket );
918 m_Stat.onBucketInitContenton();
919 break;
920 }
921
922 // There are no free buckets. It means that the bucket table is full
923 // Wait while another thread set the bucket or a free bucket will be available
924 m_Stat.onBucketsExhausted();
925 bkoff();
926 }
927
928 // Another thread set the bucket. Wait while it done
929 for ( pBucket = m_Buckets.bucket( nBucket ); pBucket == nullptr; pBucket = m_Buckets.bucket( nBucket )) {
930 bkoff();
931 m_Stat.onBusyWaitBucketInit();
932 }
933
934 return pBucket;
935 }

961 static size_t max_item_count( size_t nBucketCount, size_t nLoadFactor )
962 {
963 return nBucketCount * nLoadFactor;
964 }
965
966 void inc_item_count()
967 {
968 size_t nMaxCount = m_nMaxItemCount.load(memory_model::memory_order_relaxed);
969 if ( ++m_ItemCounter <= nMaxCount )
970 return;
971
972 size_t sz = m_nBucketCountLog2.load(memory_model::memory_order_relaxed);
973 const size_t nBucketCount = static_cast<size_t>(1) << sz;
974 if ( nBucketCount < m_Buckets.capacity()) {
975 // we may grow the bucket table
976 const size_t nLoadFactor = m_Buckets.load_factor();
977 if ( nMaxCount < max_item_count( nBucketCount, nLoadFactor ))
978 return; // someone already have updated m_nBucketCountLog2, so stop here
979
980 m_nMaxItemCount.compare_exchange_strong( nMaxCount, max_item_count( nBucketCount << 1, nLoadFactor ),
981 memory_model::memory_order_relaxed, atomics::memory_order_relaxed );
982 m_nBucketCountLog2.compare_exchange_strong( sz, sz + 1, memory_model::memory_order_relaxed, atomics::memory_order_relaxed );
983 }
984 else
985 m_nMaxItemCount.store( std::numeric_limits<size_t>::max(), memory_model::memory_order_relaxed );
986 }
