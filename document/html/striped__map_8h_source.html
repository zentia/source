621 template <typename K, typename... Args>
622 bool emplace( K&& key, Args&&... args )
623 {
624 bool bOk;
625 bool bResize;
626 size_t nHash = base_class::hashing( std::forward<K>(key));
627 bucket_type * pBucket;
628 {
629 scoped_cell_lock sl( base_class::m_MutexPolicy, nHash );
630 pBucket = base_class::bucket( nHash );
631
632 bOk = pBucket->emplace( std::forward<K>(key), std::forward<Args>(args)...);
633 bResize = bOk && base_class::m_ResizingPolicy( ++base_class::m_ItemCounter, *this, *pBucket );
634 }
635
636 if ( bResize )
637 base_class::resize();
638
639 return bOk;
640 }

663 template <typename K, typename Func>
664 std::pair<bool, bool> update( K const& key, Func func, bool bAllowInsert = true )
665 {
666 std::pair<bool, bool> result;
667 bool bResize;
668 size_t nHash = base_class::hashing( key );
669 bucket_type * pBucket;
670 {
671 scoped_cell_lock sl( base_class::m_MutexPolicy, nHash );
672 pBucket = base_class::bucket( nHash );
673
674 result = pBucket->update( key, func, bAllowInsert );
675 bResize = result.first && result.second && base_class::m_ResizingPolicy( ++base_class::m_ItemCounter, *this, *pBucket );
676 }
677
678 if ( bResize )
679 base_class::resize();
680 return result;
681 }
