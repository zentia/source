166 struct lock_array_disposer {
167 void operator()( lock_array_type * pArr )
168 {
169 // Seems, there is a false positive in std::shared_ptr deallocation in uninstrumented libc++
170 // see, for example, https://groups.google.com/forum/#!topic/thread-sanitizer/eHu4dE_z7Cc
171 // https://reviews.llvm.org/D21609
172 CDS_TSAN_ANNOTATE_IGNORE_WRITES_BEGIN;
173 lock_array_allocator().Delete( pArr );
174 CDS_TSAN_ANNOTATE_IGNORE_WRITES_END;
175 }
176 };
177
178 lock_array_ptr create_lock_array( size_t nCapacity )
179 {
180 m_nCapacity.store( nCapacity, atomics::memory_order_relaxed );
181 return lock_array_ptr( lock_array_allocator().New( nCapacity ), lock_array_disposer());
182 }
183
184 lock_type& acquire( size_t nHash )
185 {
186 owner_t me = (owner_t) cds::OS::get_current_thread_id();
187 owner_t who;
188
189 back_off bkoff;
190 while ( true ) {
191 // wait while resizing
192 while ( true ) {
193 who = m_Owner.load( atomics::memory_order_acquire );
194 if ( !( who & 1 ) || (who >> 1) == (me & c_nOwnerMask))
195 break;
196 bkoff();
197 }
198
199 lock_array_ptr pLocks;
200 {
201 scoped_spinlock sl(m_access);
202 pLocks = m_arrLocks;
203 }
204
205 lock_type& lock = pLocks->at( nHash & (pLocks->size() - 1));
206 lock.lock();
207
208 who = m_Owner.load( atomics::memory_order_acquire );
209 if ( ( !(who & 1) || (who >> 1) == (me & c_nOwnerMask)) && m_arrLocks == pLocks )
210 return lock;
211 lock.unlock();
212 }
213 }
214
215 lock_array_ptr acquire_all()
216 {
217 owner_t me = (owner_t) cds::OS::get_current_thread_id();
218 owner_t who;
219
220 back_off bkoff;
221 while ( true ) {
222 // wait while resizing
223 while ( true ) {
224 who = m_Owner.load( atomics::memory_order_acquire );
225 if ( !( who & 1 ) || (who >> 1) == (me & c_nOwnerMask))
226 break;
227 bkoff();
228 }
229
230 lock_array_ptr pLocks;
231 {
232 scoped_spinlock sl(m_access);
233 pLocks = m_arrLocks;
234 }
235
236 pLocks->lock_all();
237
238 who = m_Owner.load( atomics::memory_order_acquire );
239 if ( ( !(who & 1) || (who >> 1) == (me & c_nOwnerMask)) && m_arrLocks == pLocks )
240 return pLocks;
241
242 pLocks->unlock_all();
243 }
244 }
245
246 void release_all( lock_array_ptr p )
247 {
248 p->unlock_all();
249 }
250
251 bool acquire_resize()
252 {
253 owner_t me = (owner_t) cds::OS::get_current_thread_id();
254
255 back_off bkoff;
256 for (unsigned int nAttempts = 0; nAttempts < 32; ++nAttempts ) {
257 owner_t ownNull = 0;
258 if ( m_Owner.compare_exchange_strong( ownNull, (me << 1) | 1, atomics::memory_order_acquire, atomics::memory_order_relaxed )) {
259 lock_array_ptr pOldLocks = m_arrLocks;
260 size_t const nLockCount = pOldLocks->size();
261 for ( size_t i = 0; i < nLockCount; ++i ) {
262 typename lock_array_type::lock_type& lock = pOldLocks->at(i);
263 bkoff.reset();
264 while ( !lock.try_lock())
265 bkoff();
266 lock.unlock();
267 }
268 return true;
269 }
270 else
271 bkoff();
272 }
273 return false;
274 }

276 void release_resize()
277 {
278 m_Owner.store( 0, atomics::memory_order_release );
279 }
