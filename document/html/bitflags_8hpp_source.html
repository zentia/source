68static inline const memoryflagsVector extractFlags(const std::string &cmd, const int kernel_id) {
69 memoryflagsVector V;
70 // no command in input, we just return a vector with one (default) entry
71 if (cmd == "") {
72 V.resize(1);
73 return V;
74 }
75 const std::string kid = "kernel_"+std::to_string(kernel_id);
76 const char semicolon = ';';
77 size_t n = cmd.rfind(kid);
78 assert(n != std::string::npos);
79 n = cmd.find_first_of(semicolon, n); // first ';' after kernel_id
80 assert(n != std::string::npos);
81 n = cmd.find_first_of(semicolon, n+1); // first ';' after device_id
82 assert(n != std::string::npos);
83
84 size_t m = cmd.find_first_of('$', n+1);
85 assert(m != std::string::npos);
86 // gets just the sub-string of the command related to the memory flags
87 // starting and ending with ';' (i.e. ; UF;SF;...;)
88 const std::string &flag_string = cmd.substr(n,m-n);
89 assert(flag_string != "");
90
91 n = 0;
92 m = flag_string.find_first_of(semicolon,n+1);
93 assert(m != std::string::npos);
94 V.reserve(10);
95 do {
96 const std::string &flags = flag_string.substr(n+1, m-n-1);
97
98 struct MemoryFlags mf;
99 if (flags.find('S') != std::string::npos) mf.copy = CopyFlags::COPY;
100 else if (flags.find('R') != std::string::npos) mf.copy = CopyFlags::COPY;
101 else mf.copy = CopyFlags::DONTCOPY;
102 mf.reuse = (flags.find('U')!=std::string::npos ? ReuseFlags::REUSE : ReuseFlags::DONTREUSE);
103 mf.release = (flags.find('F')!=std::string::npos ? ReleaseFlags::RELEASE : ReleaseFlags::DONTRELEASE);
104 V.push_back(mf);
105
106 n = m;
107 m = flag_string.find_first_of(semicolon, n+1);
108 } while(m != std::string::npos);
109
110 return V;
111}
