763 void clear()
764 {
765 if( !empty()) {
766 check_deadlock_policy::check();
767
768 marked_node_ptr pHead;
769 for (;;) {
770 {
771 rcu_lock l;
772 pHead = m_pHead.load(memory_model::memory_order_acquire);
773 if ( !pHead.ptr())
774 break;
775 marked_node_ptr pNext( pHead->m_pNext.load(memory_model::memory_order_relaxed));
776 if ( cds_unlikely( !pHead->m_pNext.compare_exchange_weak( pNext, pNext | 1, memory_model::memory_order_acquire, memory_model::memory_order_relaxed )))
777 continue;
778 if ( cds_unlikely( !m_pHead.compare_exchange_weak( pHead, marked_node_ptr(pNext.ptr()), memory_model::memory_order_release, memory_model::memory_order_relaxed )))
779 continue;
780 }
781
782 --m_ItemCounter;
783 dispose_node( pHead.ptr());
784 }
785 }
786 }

829 static void dispose_chain( node_type * pChain )
830 {
831 if ( pChain ) {
832 assert( !gc::is_locked());
833
834 auto f = [&pChain]() -> cds::urcu::retired_ptr {
835 node_type * p = pChain;
836 if ( p ) {
837 pChain = p->m_pDelChain;
838 return cds::urcu::make_retired_ptr<clear_and_dispose>( node_traits::to_value_ptr( p ));
839 }
840 return cds::urcu::make_retired_ptr<clear_and_dispose>( static_cast<value_type *>(nullptr));

846 bool link_node( node_type * pNode, position& pos )
847 {
848 assert( pNode != nullptr );
849 link_checker::is_empty( pNode );
850
851 marked_node_ptr p( pos.pCur );
852 pNode->m_pNext.store( p, memory_model::memory_order_release );
853 if ( cds_likely( pos.pPrev->compare_exchange_strong( p, marked_node_ptr( pNode ), memory_model::memory_order_release, atomics::memory_order_relaxed )))
854 return true;
855
856 pNode->m_pNext.store( marked_node_ptr(), memory_model::memory_order_relaxed );
857 return false;
858 }

868 bool unlink_node( position& pos, erase_node_mask nMask )
869 {
870 assert( gc::is_locked());
871
872 // Mark the node (logical deletion)
873 marked_node_ptr next( pos.pNext, 0 );
874
875 if ( cds_likely( pos.pCur->m_pNext.compare_exchange_strong( next, next | nMask, memory_model::memory_order_release, atomics::memory_order_relaxed ))) {
876
877 // Try physical removal - fast path
878 marked_node_ptr cur( pos.pCur );
879 if ( cds_likely( pos.pPrev->compare_exchange_strong( cur, marked_node_ptr( pos.pNext ), memory_model::memory_order_acquire, atomics::memory_order_relaxed ))) {
880 if ( nMask == erase_mask )
881 link_to_remove_chain( pos, pos.pCur );
882 }
883 else {
884 // Slow path
885 search( pos.refHead, *node_traits::to_value_ptr( pos.pCur ), pos, key_comparator());
886 }
887 return true;
888 }
889 return false;
890 }

918 template <typename Func>
919 bool insert_at( atomic_node_ptr& refHead, value_type& val, Func f )
920 {
921 position pos( refHead );
922
923 {
924 rcu_lock l;
925 while ( true ) {
926 if ( search( refHead, val, pos, key_comparator())) {
927 m_Stat.onInsertFailed();
928 return false;
929 }
930
931 if ( link_node( node_traits::to_node_ptr( val ), pos )) {
932 f( val );
933 ++m_ItemCounter;
934 m_Stat.onInsertSuccess();
935 return true;
936 }
937
938 // clear next field
939 node_traits::to_node_ptr( val )->m_pNext.store( marked_node_ptr(), memory_model::memory_order_relaxed );
940 m_Stat.onInsertRetry();
941 }
942 }

975 bool unlink_at( atomic_node_ptr& refHead, value_type& val )
976 {
977 position pos( refHead );
978 back_off bkoff;
979 check_deadlock_policy::check();
980
981 for (;;) {
982 {
983 rcu_lock l;
984 if ( !search( refHead, val, pos, key_comparator()) || node_traits::to_value_ptr( *pos.pCur ) != &val ) {
985 m_Stat.onEraseFailed();
986 return false;
987 }
988 if ( !unlink_node( pos, erase_mask )) {
989 bkoff();
990 m_Stat.onEraseRetry();
991 continue;
992 }
993 }
994
995 --m_ItemCounter;
996 m_Stat.onEraseSuccess();
997 return true;
998 }

1001 template <typename Q, typename Compare, typename Func>
1002 bool erase_at( position& pos, Q const& val, Compare cmp, Func f )
1003 {
1004 back_off bkoff;
1005 check_deadlock_policy::check();
1006
1007 node_type * pDel;
1008 for (;;) {
1009 {
1010 rcu_lock l;
1011 if ( !search( pos.refHead, val, pos, cmp )) {
1012 m_Stat.onEraseFailed();
1013 return false;
1014 }
1015
1016 // store pCur since it may be changed by unlink_node() slow path
1017 pDel = pos.pCur;
1018 if ( !unlink_node( pos, erase_mask )) {
1019 bkoff();
1020 m_Stat.onEraseRetry();
1021 continue;
1022 }
1023 }
1024 assert( pDel );
1025 f( *node_traits::to_value_ptr( pDel ));
1026 --m_ItemCounter;
1027 m_Stat.onEraseSuccess();
1028 return true;
1029 }

1047 value_type * extract_at( atomic_node_ptr& refHead, Q const& val, Compare cmp )
1048 {
1049 position pos( refHead );
1050 back_off bkoff;
1051 assert( !gc::is_locked()) ; // RCU must not be locked!!!
1052
1053 node_type * pExtracted;
1054 {
1055 rcu_lock l;
1056 for (;;) {
1057 if ( !search( refHead, val, pos, cmp )) {
1058 m_Stat.onEraseFailed();

1060 }
1061
1062 // store pCur since it may be changed by unlink_node() slow path
1063 pExtracted = pos.pCur;
1064 if ( !unlink_node( pos, extract_mask )) {
1065 bkoff();
1066 m_Stat.onEraseRetry();
1067 continue;
1068 }

1070 --m_ItemCounter;
1071 value_type * pRet = node_traits::to_value_ptr( pExtracted );
1072 assert( pExtracted->m_pDelChain == nullptr );
1073 m_Stat.onEraseSuccess();
