6// Note (March 2014): shared_array is not a full implementation of an array version
7// of C++11 shared_ptr, and there currently are no plans to make it so. A future
8// version of shared_ptr would likely take on the ability to store arrays,
9// same as unique_ptr has array support. This class isn't deprecated, but it
10// is frozen until some future decision is made on what to do about arrays.
11//
13// This class implements a shared_array template. This is a class which is
14// similar to the C++ shared_ptr template, except that it works with arrays
15// instead of individual objects.
16//
17// Important notice:
18// As of this writing (9/2003), the implementation provided here has
19// limitations that you should be aware of. These limitations will be shortly
20// rectified. Most significantly, this implementation has the following
21// weaknesses:
22// - It cannot safely deal with exceptions that occur during the
23// construction of shared_ptr objects.
24// - It cannot safely deal with recursive shared_ptr objects.
25// If a shared_ptr<A> holds a pointer to an instance of A and
26// class A owns an instance of shared_ptr<A> that refers to,
27// the original instance, the memory will leak.
28// - A template of type shared_ptr<void> will not call the destructor
29// for an object that it stores. You thus must declare a shared_ptr
30// template specifically for the class type.
31// - It doesn't safely handle multiple instances of shared_ptr
32// which own the same pointer accessed from multiple threads.
33// This weakness is by design, for performance reasons. You should
34// use shared_ptr_mt for multi-thread safe access.
35//
36// The rectification of the above issues are discussed in the C++ standardization
37// documents for the next C++ standard (as of 2003):
38// http://std.dkuug.dk/jtc1/sc22/wg21/docs/papers/2003/n1450.html#Implementation-difficulty
39//
40// This current implementation will be eventually (hopefully by 1/2004) rectified
41// to be in line with the second generation C++ standard proposal.
42//
43// The intended design of this class is based somewhat on the design of the Boost
44// shared_array template. This design is also being considered for the next C++
45// standard (as of 2003). The C++ standard update proposal is currently available at:
46// http://std.dkuug.dk/jtc1/sc22/wg21/docs/papers/2003/n1450.html
47// Boost smart pointers, including shared_array are documented at:
48// http://www.boost.org/libs/smart_ptr/
49//
50// As of this writing (10/2003), this class has received approval from EA legal
51// for use. The potential issue is the similarity of the class name and class
52// interface to existing open source code.
53//

133 explicit shared_array(T* pArray = NULL, const allocator_type& allocator = EASTL_SHARED_ARRAY_DEFAULT_ALLOCATOR)
134 : mpArray(pArray),
135 mpRefCount(NULL),
136 mAllocator(allocator)
137 {
138 // Allocate memory for the reference count.
139 void* const pMemory = EASTLAlloc(mAllocator, sizeof(ref_count));
140 if(pMemory)
141 mpRefCount = ::new(pMemory) ref_count(1);
142 }

151 shared_array(const shared_array& sharedArray)
152 : mpArray(sharedArray.mpArray),
153 mpRefCount(sharedArray.mpRefCount),
154 mAllocator(sharedArray.mAllocator)
155 {
156 ++*mpRefCount;
157 }

164 ~shared_array()
165 {
166 const ref_count newRefCount(--*mpRefCount);
167 // assert(newRefCount >= 0);
168 if(newRefCount == 0)
169 {
170 EASTLFree(mAllocator, mpRefCount, sizeof(ref_count));
171 Deleter del;
172 del(mpArray);
173 }
174 }

187 shared_array& operator=(const shared_array& sharedArray)
188 {
189 if(mpArray != sharedArray.mpArray)
190 {
191 // The easiest thing to do is to create a temporary and
192 // copy ourselves ourselves into it. This is a standard
193 // method for switching pointer ownership in systems like this.
194 shared_array(sharedArray).swap(*this);
195 }
196 return *this;
197 }

217 void reset(T* pArray = NULL)

221 // The easiest thing to do is to create a temporary and

223 // method for switching pointer ownership in systems like this.

232 {
233 // We leave mAllocator as-is.

235 // eastl::swap(mpArray, sharedArray.mpArray);
236 T* const pArray = sharedArray.mpArray;
237 sharedArray.mpArray = mpArray;
238 mpArray = pArray;
