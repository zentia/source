155 bool enqueue( value_type& val )
156 {
157 auto pRec = m_FlatCombining.acquire_record();
158 pRec->pVal = &val;
159
160 constexpr_if ( c_bEliminationEnabled )
161 m_FlatCombining.batch_combine( op_enq, pRec, *this );
162 else
163 m_FlatCombining.combine( op_enq, pRec, *this );
164
165 assert( pRec->is_done());
166 m_FlatCombining.release_record( pRec );
167 m_FlatCombining.internal_statistics().onEnqueue();

291 void fc_apply( fc_record * pRec )
292 {
293 assert( pRec );
294
295 // this function is called under FC mutex, so switch TSan off
296 // All TSan warnings are false positive
297 //CDS_TSAN_ANNOTATE_IGNORE_RW_BEGIN;
298
299 switch ( pRec->op()) {
300 case op_enq:
301 assert( pRec->pVal );
302 m_Queue.push_back( *(pRec->pVal ));
303 break;
304 case op_deq:
305 pRec->bEmpty = m_Queue.empty();
306 if ( !pRec->bEmpty ) {
307 pRec->pVal = &m_Queue.front();
308 m_Queue.pop_front();
309 }
310 break;
311 case op_clear:
312 m_Queue.clear();
313 break;
314 case op_clear_and_dispose:
315 m_Queue.clear_and_dispose( disposer());
316 break;
317 default:
318 assert(false);
319 break;
320 }
321 //CDS_TSAN_ANNOTATE_IGNORE_RW_END;
322 }

325 void fc_process( typename fc_kernel::iterator itBegin, typename fc_kernel::iterator itEnd )
326 {
327 // this function is called under FC mutex, so switch TSan off
328 // All TSan warnings are false positive
329 //CDS_TSAN_ANNOTATE_IGNORE_RW_BEGIN;
330
331 typedef typename fc_kernel::iterator fc_iterator;
332 for ( fc_iterator it = itBegin, itPrev = itEnd; it != itEnd; ++it ) {
333 switch ( it->op( atomics::memory_order_acquire )) {
334 case op_enq:
335 case op_deq:
336 if ( m_Queue.empty()) {
337 if ( itPrev != itEnd && collide( *itPrev, *it ))
338 itPrev = itEnd;
339 else
340 itPrev = it;
341 }
342 break;
343 }
344 }
345 //CDS_TSAN_ANNOTATE_IGNORE_RW_END;
346 }
347 //@endcond
348
349 private:
350 //@cond
351 bool collide( fc_record& rec1, fc_record& rec2 )
352 {
353 assert( m_Queue.empty());
354
355 switch ( rec1.op()) {
356 case op_enq:
357 if ( rec2.op() == op_deq ) {
358 assert(rec1.pVal);
359 rec2.pVal = rec1.pVal;
360 rec2.bEmpty = false;
361 m_FlatCombining.operation_done( rec1 );
362 m_FlatCombining.operation_done( rec2 );
363 m_FlatCombining.internal_statistics().onCollide();
364 return true;
365 }
366 break;
367 case op_deq:
368 if ( rec2.op() == op_enq ) {
369 assert(rec2.pVal);
370 rec1.pVal = rec2.pVal;
371 rec1.bEmpty = false;
372 m_FlatCombining.operation_done( rec1 );
373 m_FlatCombining.operation_done( rec2 );
374 m_FlatCombining.internal_statistics().onCollide();
375 return true;
376 }
377 break;
378 }
379 return false;
380 }
381 //@endcond
382 };
