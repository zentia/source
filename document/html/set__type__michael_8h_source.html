22 template <class GC, typename List, typename Traits = cc::michael_set::traits>
23 class MichaelHashSet : public cc::MichaelHashSet< GC, List, Traits >
24 {
25 typedef cc::MichaelHashSet< GC, List, Traits > base_class;
26 public:
27 template <class Config>
28 MichaelHashSet( Config const& cfg )
29 : base_class( cfg.s_nSetSize, cfg.s_nLoadFactor )
30 {}
31
32 template <typename Iterator>
33 typename std::enable_if< std::is_same< Iterator, typename base_class::iterator>::value, Iterator>::type
34 get_begin()
35 {
36 return base_class::begin();
37 }
38
39 template <typename Iterator>
40 typename std::enable_if< std::is_same< Iterator, typename base_class::iterator>::value, Iterator>::type
41 get_end()
42 {
43 return base_class::end();
44 }
45
46 // for testing
47 static constexpr bool const c_bExtractSupported = true;
48 static constexpr bool const c_bLoadFactorDepended = true;
49 static constexpr bool const c_bEraseExactKey = false;
50 };
