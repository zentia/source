309 void runTest() {
310 TestsListener::theInstance().testHasRun();
311 TestsListener::theInstance().currentTestName(_name);
312 try {
313 (_parent->*_testCaseMethod)();
314 TestsListener::theInstance().testHasPassed();
315 }
316 catch (std::exception &error) {
317 TestsListener::theInstance().testHasThrown();
318 TestsListener::theInstance().errorsLog()
319 << "std::exception catched by MiniCppUnit: \n"
320 << "what() : "
321 << Assert::yellow() << error.what()
322 << Assert::normal() << "\n";
323 }
324 catch (TestFailedException &) //just for skiping current test case

326 // the assert() calls testHasFailed()
327 }
328 catch (...) {
329 TestsListener::theInstance().testHasThrown();
330 TestsListener::theInstance().errorsLog()
331 << "non standard exception catched by MiniCppUnit.\n";
332 }
333 }
