51 //
52 // Cast a enum value to its underlying type.
53 // For example:
54 //
55 // enum class MyEnum : uint8_t { Value = 0; }
56 // auto x = MyEnum::Value;
57 // std::cout << to_underlying(x); // equivalent to sts::cout << static_cast<uint8_t>(x);
59
60 #if EASTL_VARIABLE_TEMPLATES_ENABLED && !defined(EA_COMPILER_NO_TEMPLATE_ALIASES)
61 template<class T>
62 constexpr underlying_type_t<T> to_underlying(T value) noexcept
63 {
64 return static_cast<underlying_type_t<T>>(value);

381 // is_complete_type
382 //
383 // Determines if the specified type is complete
384 //
385 // Warning: Be careful when using is_complete_type since the value is fixed at first instantiation.
386 // Consider the following:
387 //
388 // struct Foo;
389 // is_complete_type_v<Foo> // false
390 // struct Foo {};
391 // is_complete_type_v<Foo> // still false
393
394 template<typename T, typename = void>
395 struct is_complete_type : public false_type {};
396
397 template<typename T>
398 struct is_complete_type<T, eastl::void_t<decltype(sizeof(T) != 0)>> : public true_type {};
399
400 template<>
401 struct is_complete_type<const volatile void> : public false_type {};
402 template<>
403 struct is_complete_type<const void> : public false_type {};
404 template<>
405 struct is_complete_type<volatile void> : public false_type {};
406 template<>
407 struct is_complete_type<void> : public false_type {};
408
409 template<typename T>
410 struct is_complete_type<T, eastl::enable_if_t<eastl::is_function_v<T>>> : public true_type {};
411
412 template <typename T>
