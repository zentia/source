595// helper function for easy traversal over the ASTs
596template<typename Enter, typename Exit>
597void traverse_subexpressions(const Expression *expr,
598 const Enter &enter,
599 const Exit &exit) noexcept {
600 enter(expr);
601 switch (expr->tag()) {
602 case Expression::Tag::UNARY: {
603 auto unary_expr = static_cast<const UnaryExpr *>(expr);
604 traverse_subexpressions(unary_expr->operand(), enter, exit);
605 break;
606 }
607 case Expression::Tag::BINARY: {
608 auto binary_expr = static_cast<const BinaryExpr *>(expr);
609 traverse_subexpressions(binary_expr->lhs(), enter, exit);
610 traverse_subexpressions(binary_expr->rhs(), enter, exit);
611 break;
612 }
613 case Expression::Tag::MEMBER: {
614 auto member_expr = static_cast<const MemberExpr *>(expr);
615 traverse_subexpressions(member_expr->self(), enter, exit);
616 break;
617 }
618 case Expression::Tag::ACCESS: {
619 auto access_expr = static_cast<const AccessExpr *>(expr);
620 traverse_subexpressions(access_expr->range(), enter, exit);
621 traverse_subexpressions(access_expr->index(), enter, exit);
622 break;
623 }
624 case Expression::Tag::LITERAL:
625 case Expression::Tag::REF:
626 case Expression::Tag::CONSTANT: break;
627 case Expression::Tag::CALL: {
628 auto call_expr = static_cast<const CallExpr *>(expr);
629 for (auto arg : call_expr->arguments()) {
630 traverse_subexpressions(arg, enter, exit);
631 }
632 break;
633 }
634 case Expression::Tag::CAST: {
635 auto cast_expr = static_cast<const CastExpr *>(expr);
636 traverse_subexpressions(cast_expr->expression(), enter, exit);
637 break;
638 }
639 case Expression::Tag::FUNC_REF:
640 case Expression::Tag::TYPE_ID:
641 case Expression::Tag::STRING_ID:
642 case Expression::Tag::CPUCUSTOM:
643 case Expression::Tag::GPUCUSTOM: break;
644 }
645 exit(expr);
646}
