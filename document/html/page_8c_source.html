140static void mi_page_thread_collect_to_local(mi_page_t* page, mi_block_t* head)
141{
142 if (head == NULL) return;
143
144 // find the last block in the list -- also to get a proper use count (without data races)
145 size_t max_count = page->capacity; // cannot collect more than capacity
146 size_t count = 1;
147 mi_block_t* last = head;
148 mi_block_t* next;
149 while ((next = mi_block_next(page, last)) != NULL && count <= max_count) {
150 count++;
151 last = next;
152 }
153
154 // if `count > max_count` there was a memory corruption (possibly infinite list due to double multi-threaded free)
155 if (count > max_count) {
156 _mi_error_message(EFAULT, "corrupted thread-free list\n");
157 return; // the thread-free items cannot be freed
158 }
159
160 // and append the current local free list
161 mi_block_set_next(page, last, page->local_free);
162 page->local_free = head;
163
164 // update counts now
165 mi_assert_internal(count <= UINT16_MAX);
166 page->used = page->used - (uint16_t)count;
167}

169// Collect the local `thread_free` list using an atomic exchange.
170static void mi_page_thread_free_collect(mi_page_t* page)
171{
172 // atomically capture the thread free list
173 mi_block_t* head;
174 mi_thread_free_t tfreex;
175 mi_thread_free_t tfree = mi_atomic_load_relaxed(&page->xthread_free);
176 do {
177 head = mi_tf_block(tfree);
178 if mi_likely(head == NULL) return; // return if the list is empty
179 tfreex = mi_tf_create(NULL,mi_tf_is_owned(tfree)); // set the thread free list to NULL
180 } while (!mi_atomic_cas_weak_acq_rel(&page->xthread_free, &tfree, tfreex)); // release is enough?
181 mi_assert_internal(head != NULL);
182
183 // and move it to the local list
184 mi_page_thread_collect_to_local(page, head);
185}

187void _mi_page_free_collect(mi_page_t* page, bool force) {
188 mi_assert_internal(page!=NULL);
189
190 // collect the thread free list
191 mi_page_thread_free_collect(page);
192
193 // and the local free list
194 if (page->local_free != NULL) {
195 if mi_likely(page->free == NULL) {
196 // usual case
197 page->free = page->local_free;
198 page->local_free = NULL;
199 page->free_is_zero = false;
200 }
201 else if (force) {
202 // append -- only on shutdown (force) as this is a linear operation
203 mi_block_t* tail = page->local_free;
204 mi_block_t* next;
205 while ((next = mi_block_next(page, tail)) != NULL) {
206 tail = next;
207 }
208 mi_block_set_next(page, tail, page->free);
209 page->free = page->local_free;
210 page->local_free = NULL;
211 page->free_is_zero = false;
212 }
213 }
214
215 mi_assert_internal(!force || page->local_free == NULL);
216}

218// Collect elements in the thread-free list starting at `head`. This is an optimized
219// version of `_mi_page_free_collect` to be used from `free.c:_mi_free_collect_mt` that avoids atomic access to `xthread_free`.
220//
221// `head` must be in the `xthread_free` list. It will not collect `head` itself
222// so the `used` count is not fully updated in general. However, if the `head` is
223// the last remaining element, it will be collected and the used count will become `0` (so `mi_page_all_free` becomes true).
224void _mi_page_free_collect_partly(mi_page_t* page, mi_block_t* head) {
225 if (head == NULL) return;
226 mi_block_t* next = mi_block_next(page,head); // we cannot collect the head element itself as `page->thread_free` may point to it (and we want to avoid atomic ops)
227 if (next != NULL) {
228 mi_block_set_next(page, head, NULL);
229 mi_page_thread_collect_to_local(page, next);
230 if (page->local_free != NULL && page->free == NULL) {
231 page->free = page->local_free;
232 page->local_free = NULL;
233 page->free_is_zero = false;
234 }
235 }
236 if (page->used == 1) {
237 // all elements are free'd since we skipped the `head` element itself
238 mi_assert_internal(mi_tf_block(mi_atomic_load_relaxed(&page->xthread_free)) == head);
239 mi_assert_internal(mi_block_next(page,head) == NULL);
240 _mi_page_free_collect(page, false); // collect the final element
241 }
242}

282void _mi_page_abandon(mi_page_t* page, mi_page_queue_t* pq) {
283 _mi_page_free_collect(page, false); // ensure used count is up to date
284 if (mi_page_all_free(page)) {
285 _mi_page_free(page, pq);
286 }
287 else {
288 mi_page_queue_remove(pq, page);
289 mi_heap_t* heap = page->heap;
290 mi_page_set_heap(page, NULL);
291 page->heap = heap; // dont set heap to NULL so we can reclaim_on_free within the same heap
292 _mi_arenas_page_abandon(page, heap->tld);
293 _mi_arenas_collect(false, false, heap->tld); // allow purging
294 }

298// allocate a fresh page from a segment
299static mi_page_t* mi_page_fresh_alloc(mi_heap_t* heap, mi_page_queue_t* pq, size_t block_size, size_t page_alignment) {
300 #if !MI_HUGE_PAGE_ABANDON
301 mi_assert_internal(pq != NULL);
302 mi_assert_internal(mi_heap_contains_queue(heap, pq));
303 mi_assert_internal(page_alignment > 0 || block_size > MI_LARGE_MAX_OBJ_SIZE || block_size == pq->block_size);
304 #endif
305 mi_page_t* page = _mi_arenas_page_alloc(heap, block_size, page_alignment);
306 if (page == NULL) {
307 // out-of-memory
308 return NULL;
309 }
310 if (mi_page_is_abandoned(page)) {
311 _mi_heap_page_reclaim(heap, page);
312 if (!mi_page_immediate_available(page)) {
313 if (mi_page_is_expandable(page)) {
314 mi_page_extend_free(heap, page);
315 }
316 else {
317 mi_assert(false); // should not happen?
318 return NULL;
319 }
320 }
321 }
322 else if (pq != NULL) {
323 mi_page_queue_push(heap, pq, page);
324 }
325 mi_heap_stat_increase(heap, pages, 1);
326 mi_assert_internal(pq!=NULL || mi_page_block_size(page) >= block_size);
327 mi_heap_stat_increase(heap, page_bins[mi_page_bin(page)], 1);
328 mi_assert_expensive(_mi_page_is_valid(page));
329 return page;
330}

363static void mi_page_to_full(mi_page_t* page, mi_page_queue_t* pq) {
364 mi_assert_internal(pq == mi_page_queue_of(page));
365 mi_assert_internal(!mi_page_immediate_available(page));
366 mi_assert_internal(!mi_page_is_in_full(page));
367
368 mi_heap_t* heap = mi_page_heap(page);
369 if (heap->allow_page_abandon) {
370 // abandon full pages (this is the usual case in order to allow for sharing of memory between heaps)
371 _mi_page_abandon(page, pq);
372 }
373 else if (!mi_page_is_in_full(page)) {
374 // put full pages in a heap local queue (this is for heaps that cannot abandon, for example, if the heap can be destroyed)
375 mi_page_queue_enqueue_from(&mi_page_heap(page)->pages[MI_BIN_FULL], pq, page);
376 _mi_page_free_collect(page, false); // try to collect right away in case another thread freed just before MI_USE_DELAYED_FREE was set
377 }

407// Retire a page with no more used blocks
408// Important to not retire too quickly though as new
409// allocations might coming.
410// Note: called from `mi_free` and benchmarks often
411// trigger this due to freeing everything and then
412// allocating again so careful when changing this.
413void _mi_page_retire(mi_page_t* page) mi_attr_noexcept {
414 mi_assert_internal(page != NULL);
415 mi_assert_expensive(_mi_page_is_valid(page));
416 mi_assert_internal(mi_page_all_free(page));
417
418 mi_page_set_has_aligned(page, false);
419
420 // don't retire too often..
421 // (or we end up retiring and re-allocating most of the time)
422 // NOTE: refine this more: we should not retire if this
423 // is the only page left with free blocks. It is not clear
424 // how to check this efficiently though...
425 // for now, we don't retire if it is the only page left of this size class.
426 mi_page_queue_t* pq = mi_page_queue_of(page);
427 #if MI_RETIRE_CYCLES > 0
428 const size_t bsize = mi_page_block_size(page);
429 if mi_likely( /* bsize < MI_MAX_RETIRE_SIZE && */ !mi_page_queue_is_special(pq)) { // not full or huge queue?
430 if (pq->last==page && pq->first==page) { // the only page in the queue?
431 mi_heap_t* heap = mi_page_heap(page);
432 mi_debug_heap_stat_counter_increase(heap, pages_retire, 1);
433 page->retire_expire = (bsize <= MI_SMALL_MAX_OBJ_SIZE ? MI_RETIRE_CYCLES : MI_RETIRE_CYCLES/4);
434 mi_assert_internal(pq >= heap->pages);
435 const size_t index = pq - heap->pages;
436 mi_assert_internal(index < MI_BIN_FULL && index < MI_BIN_HUGE);
437 if (index < heap->page_retired_min) heap->page_retired_min = index;
438 if (index > heap->page_retired_max) heap->page_retired_max = index;
439 mi_assert_internal(mi_page_all_free(page));
440 return; // don't free after all
441 }
442 }
443 #endif
444 _mi_page_free(page, pq);
445}

447// free retired pages: we don't need to look at the entire queues
448// since we only retire pages that are at the head position in a queue.
449void _mi_heap_collect_retired(mi_heap_t* heap, bool force) {
450 size_t min = MI_BIN_FULL;
451 size_t max = 0;
452 for(size_t bin = heap->page_retired_min; bin <= heap->page_retired_max; bin++) {
453 mi_page_queue_t* pq = &heap->pages[bin];
454 mi_page_t* page = pq->first;
455 if (page != NULL && page->retire_expire != 0) {
456 if (mi_page_all_free(page)) {
457 page->retire_expire--;
458 if (force || page->retire_expire == 0) {
459 _mi_page_free(pq->first, pq);
460 }
461 else {
462 // keep retired, update min/max
463 if (bin < min) min = bin;
464 if (bin > max) max = bin;
465 }
466 }
467 else {
468 page->retire_expire = 0;
469 }
470 }
471 }
472 heap->page_retired_min = min;
473 heap->page_retired_max = max;
474}

500/* -----------------------------------------------------------
501 Initialize the initial free list in a page.
502 In secure mode we initialize a randomized list by
503 alternating between slices.
504----------------------------------------------------------- */
505
506#define MI_MAX_SLICE_SHIFT (6) // at most 64 slices
507#define MI_MAX_SLICES (1UL << MI_MAX_SLICE_SHIFT)
508#define MI_MIN_SLICES (2)
509
510static void mi_page_free_list_extend_secure(mi_heap_t* const heap, mi_page_t* const page, const size_t bsize, const size_t extend, mi_stats_t* const stats) {
511 MI_UNUSED(stats);
512 #if (MI_SECURE<3)
513 mi_assert_internal(page->free == NULL);
514 mi_assert_internal(page->local_free == NULL);
515 #endif
516 mi_assert_internal(page->capacity + extend <= page->reserved);
517 mi_assert_internal(bsize == mi_page_block_size(page));
518 void* const page_area = mi_page_start(page);
519
520 // initialize a randomized free list
521 // set up `slice_count` slices to alternate between
522 size_t shift = MI_MAX_SLICE_SHIFT;
523 while ((extend >> shift) == 0) {
524 shift--;
525 }
526 const size_t slice_count = (size_t)1U << shift;
527 const size_t slice_extend = extend / slice_count;
528 mi_assert_internal(slice_extend >= 1);
529 mi_block_t* blocks[MI_MAX_SLICES]; // current start of the slice
530 size_t counts[MI_MAX_SLICES]; // available objects in the slice
531 for (size_t i = 0; i < slice_count; i++) {
532 blocks[i] = mi_page_block_at(page, page_area, bsize, page->capacity + i*slice_extend);
533 counts[i] = slice_extend;
534 }
535 counts[slice_count-1] += (extend % slice_count); // final slice holds the modulus too (todo: distribute evenly?)
536
537 // and initialize the free list by randomly threading through them
538 // set up first element
539 const uintptr_t r = _mi_heap_random_next(heap);
540 size_t current = r % slice_count;
541 counts[current]--;
542 mi_block_t* const free_start = blocks[current];
543 // and iterate through the rest; use `random_shuffle` for performance
544 uintptr_t rnd = _mi_random_shuffle(r|1); // ensure not 0
545 for (size_t i = 1; i < extend; i++) {
546 // call random_shuffle only every INTPTR_SIZE rounds
547 const size_t round = i%MI_INTPTR_SIZE;
548 if (round == 0) rnd = _mi_random_shuffle(rnd);
549 // select a random next slice index
550 size_t next = ((rnd >> 8*round) & (slice_count-1));
551 while (counts[next]==0) { // ensure it still has space
552 next++;
553 if (next==slice_count) next = 0;
554 }
555 // and link the current block to it
556 counts[next]--;
557 mi_block_t* const block = blocks[current];
558 blocks[current] = (mi_block_t*)((uint8_t*)block + bsize); // bump to the following block
559 mi_block_set_next(page, block, blocks[next]); // and set next; note: we may have `current == next`
560 current = next;
561 }
562 // prepend to the free list (usually NULL)

567static mi_decl_noinline void mi_page_free_list_extend( mi_page_t* const page, const size_t bsize, const size_t extend, mi_stats_t* const stats)
568{
569 MI_UNUSED(stats);
570 #if (MI_SECURE<3)
571 mi_assert_internal(page->free == NULL);
572 mi_assert_internal(page->local_free == NULL);
573 #endif

593/* -----------------------------------------------------------
594 Page initialize and extend the capacity
595----------------------------------------------------------- */
596
597#define MI_MAX_EXTEND_SIZE (4*1024) // heuristic, one OS page seems to work well.
598#if (MI_SECURE>=3)
599#define MI_MIN_EXTEND (8*MI_SECURE) // extend at least by this many
600#else
601#define MI_MIN_EXTEND (1)
602#endif
603
604// Extend the capacity (up to reserved) by initializing a free list
605// We do at most `MI_MAX_EXTEND` to avoid touching too much memory
606// Note: we also experimented with "bump" allocation on the first
607// allocations but this did not speed up any benchmark (due to an
608// extra test in malloc? or cache effects?)
609static void mi_page_extend_free(mi_heap_t* heap, mi_page_t* page) {
610 mi_assert_expensive(mi_page_is_valid_init(page));
611 #if (MI_SECURE<3)
612 mi_assert(page->free == NULL);
613 mi_assert(page->local_free == NULL);
614 if (page->free != NULL) return;
615 #endif
616 if (page->capacity >= page->reserved) return;
617
618 size_t page_size;
619 //uint8_t* page_start =
620 mi_page_area(page, &page_size);
621 mi_debug_heap_stat_counter_increase(heap, pages_extended, 1);
622
623 // calculate the extend count
624 const size_t bsize = mi_page_block_size(page);
625 size_t extend = (size_t)page->reserved - page->capacity;
626 mi_assert_internal(extend > 0);
627
628 size_t max_extend = (bsize >= MI_MAX_EXTEND_SIZE ? MI_MIN_EXTEND : MI_MAX_EXTEND_SIZE/bsize);
629 if (max_extend < MI_MIN_EXTEND) { max_extend = MI_MIN_EXTEND; }
630 mi_assert_internal(max_extend > 0);
631
632 if (extend > max_extend) {
633 // ensure we don't touch memory beyond the page to reduce page commit.
634 // the `lean` benchmark tests this. Going from 1 to 8 increases rss by 50%.
635 extend = max_extend;
636 }
637
638 mi_assert_internal(extend > 0 && extend + page->capacity <= page->reserved);
639 mi_assert_internal(extend < (1UL<<16));
640
641 // commit on demand?
642 if (page->slice_committed > 0) {
643 const size_t needed_size = (page->capacity + extend)*bsize;
644 const size_t needed_commit = _mi_align_up( mi_page_slice_offset_of(page, needed_size), MI_PAGE_MIN_COMMIT_SIZE );
645 if (needed_commit > page->slice_committed) {
646 mi_assert_internal(((needed_commit - page->slice_committed) % _mi_os_page_size()) == 0);
647 _mi_os_commit(mi_page_slice_start(page) + page->slice_committed, needed_commit - page->slice_committed, NULL);
648 page->slice_committed = needed_commit;
649 }
650 }
651
652 // and append the extend the free list
653 if (extend < MI_MIN_SLICES || MI_SECURE<3) {
654 mi_page_free_list_extend(page, bsize, extend, &heap->tld->stats );
655 }
656 else {
657 mi_page_free_list_extend_secure(heap, page, bsize, extend, &heap->tld->stats);
658 }
659 // enable the new free list
660 page->capacity += (uint16_t)extend;
661 mi_debug_heap_stat_increase(heap, page_committed, extend * bsize);
662 mi_assert_expensive(mi_page_is_valid_init(page));
663}

712// Find a page with free blocks of `page->block_size`.
713static mi_decl_noinline mi_page_t* mi_page_queue_find_free_ex(mi_heap_t* heap, mi_page_queue_t* pq, bool first_try)
714{
715 // search through the pages in "next fit" order
716 size_t count = 0;
717 long candidate_limit = 0; // we reset this on the first candidate to limit the search
718 long page_full_retain = (pq->block_size > MI_SMALL_MAX_OBJ_SIZE ? 0 : heap->page_full_retain); // only retain small pages
719 mi_page_t* page_candidate = NULL; // a page with free space
720 mi_page_t* page = pq->first;
721
722 while (page != NULL)
723 {
724 mi_page_t* next = page->next; // remember next (as this page can move to another queue)
725 count++;
726 candidate_limit--;
727
728 // search up to N pages for a best candidate
729
730 // is the local free list non-empty?
731 bool immediate_available = mi_page_immediate_available(page);
732 if (!immediate_available) {
733 // collect freed blocks by us and other threads to we get a proper use count
734 _mi_page_free_collect(page, false);
735 immediate_available = mi_page_immediate_available(page);
736 }
737
738 // if the page is completely full, move it to the `mi_pages_full`
739 // queue so we don't visit long-lived pages too often.
740 if (!immediate_available && !mi_page_is_expandable(page)) {
741 page_full_retain--;
742 if (page_full_retain < 0) {
743 mi_assert_internal(!mi_page_is_in_full(page) && !mi_page_immediate_available(page));
744 mi_page_to_full(page, pq);
745 }
746 }
747 else {
748 // the page has free space, make it a candidate
749 // we prefer non-expandable pages with high usage as candidates (to reduce commit, and increase chances of free-ing up pages)
750 if (page_candidate == NULL) {
751 page_candidate = page;
752 candidate_limit = _mi_option_get_fast(mi_option_page_max_candidates);
753 }
754 else if (mi_page_all_free(page_candidate)) {
755 _mi_page_free(page_candidate, pq);
756 page_candidate = page;
757 }
758 // prefer to reuse fuller pages (in the hope the less used page gets freed)
759 else if (page->used >= page_candidate->used && !mi_page_is_mostly_used(page)) { // && !mi_page_is_expandable(page)) {
760 page_candidate = page;
761 }
762 // if we find a non-expandable candidate, or searched for N pages, return with the best candidate
763 if (immediate_available || candidate_limit <= 0) {
764 mi_assert_internal(page_candidate!=NULL);
765 break;
766 }
767 }
768
769 #if 0
770 // first-fit algorithm without candidates
771 // If the page contains free blocks, we are done
772 if (mi_page_immediate_available(page) || mi_page_is_expandable(page)) {
773 break; // pick this one
774 }
775
776 // If the page is completely full, move it to the `mi_pages_full`
777 // queue so we don't visit long-lived pages too often.
778 mi_assert_internal(!mi_page_is_in_full(page) && !mi_page_immediate_available(page));
779 mi_page_to_full(page, pq);
780 #endif
781
782 page = next;
783 } // for each page
784
785 mi_heap_stat_counter_increase(heap, page_searches, count);
786
787 // set the page to the best candidate
788 if (page_candidate != NULL) {
789 page = page_candidate;
790 }
791 if (page != NULL) {
792 if (!mi_page_immediate_available(page)) {
793 mi_assert_internal(mi_page_is_expandable(page));
794 mi_page_extend_free(heap, page);
795 }
796 mi_assert_internal(mi_page_immediate_available(page));
797 }
798
799 if (page == NULL) {
800 _mi_heap_collect_retired(heap, false); // perhaps make a page available
801 page = mi_page_fresh(heap, pq);
802 mi_assert_internal(page == NULL || mi_page_immediate_available(page));
803 if (page == NULL && first_try) {
804 // out-of-memory _or_ an abandoned page with free blocks was reclaimed, try once again
805 page = mi_page_queue_find_free_ex(heap, pq, false);
806 mi_assert_internal(page == NULL || mi_page_immediate_available(page));
807 }
808 }
809 else {
810 mi_assert_internal(page == NULL || mi_page_immediate_available(page));
811 // move the page to the front of the queue
812 mi_page_queue_move_to_front(heap, pq, page);
813 page->retire_expire = 0;
814 // _mi_heap_collect_retired(heap, false); // update retire counts; note: increases rss on MemoryLoad bench so don't do this
815 }
816 mi_assert_internal(page == NULL || mi_page_immediate_available(page));
817
818
819 return page;
820}

928// Generic allocation routine if the fast path (`alloc.c:mi_page_malloc`) does not succeed.
929// Note: in debug mode the size includes MI_PADDING_SIZE and might have overflowed.
930// The `huge_alignment` is normally 0 but is set to a multiple of MI_SLICE_SIZE for
931// very large requested alignments in which case we use a huge singleton page.
932void* _mi_malloc_generic(mi_heap_t* heap, size_t size, bool zero, size_t huge_alignment) mi_attr_noexcept
933{
934 mi_assert_internal(heap != NULL);
935
936 // initialize if necessary
937 if mi_unlikely(!mi_heap_is_initialized(heap)) {
938 heap = mi_heap_get_default(); // calls mi_thread_init
939 if mi_unlikely(!mi_heap_is_initialized(heap)) { return NULL; }
940 }
941 mi_assert_internal(mi_heap_is_initialized(heap));
942
943 // do administrative tasks every N generic mallocs
944 if mi_unlikely(++heap->generic_count >= 1000) {
945 heap->generic_collect_count += heap->generic_count;
946 heap->generic_count = 0;
947 // call potential deferred free routines
948 _mi_deferred_free(heap, false);
949
950 // collect every once in a while (10000 by default)
951 const long generic_collect = mi_option_get_clamp(mi_option_generic_collect, 1, 1000000L);
952 if (heap->generic_collect_count >= generic_collect) {
953 heap->generic_collect_count = 0;
954 mi_heap_collect(heap, false /* force? */);
955 }
956 }
957
958 // find (or allocate) a page of the right size
959 mi_page_t* page = mi_find_page(heap, size, huge_alignment);
960 if mi_unlikely(page == NULL) { // first time out of memory, try to collect and retry the allocation once more
961 mi_heap_collect(heap, true /* force? */);
962 page = mi_find_page(heap, size, huge_alignment);
963 }
964
965 if mi_unlikely(page == NULL) { // out of memory
966 const size_t req_size = size - MI_PADDING_SIZE; // correct for padding_size in case of an overflow on `size`
967 _mi_error_message(ENOMEM, "unable to allocate memory (%zu bytes)\n", req_size);
968 return NULL;
969 }
970
971 mi_assert_internal(mi_page_immediate_available(page));
972 mi_assert_internal(mi_page_block_size(page) >= size);
973 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
974 mi_assert_internal(_mi_ptr_page(page)==page);
975
976 // and try again, this time succeeding! (i.e. this should never recurse through _mi_page_malloc)
977 void* p;
978 if mi_unlikely(zero && mi_page_is_huge(page)) {
979 // note: we cannot call _mi_page_malloc with zeroing for huge blocks; we zero it afterwards in that case.
980 p = _mi_page_malloc(heap, page, size);
981 mi_assert_internal(p != NULL);
982 _mi_memzero_aligned(p, mi_page_usable_block_size(page));
983 }
984 else {
985 p = _mi_page_malloc_zero(heap, page, size, zero);
986 mi_assert_internal(p != NULL);
987 }
988 // move singleton pages to the full queue
989 if (page->reserved == page->used) {
990 mi_page_to_full(page, mi_page_queue_of(page));
991 }
992 return p;
993}
