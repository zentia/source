469 bool do_dequeue( dequeue_result& res, bool bDeque )
470 {
471 // Note:
472 // If bDeque == false then the function is called from empty method and no real dequeuing operation is performed
473
474 back_off bkoff;
475
476 marked_ptr h;
477 marked_ptr t;
478 marked_ptr pNext;
479
480 while ( true ) {
481 h = res.guards.protect( 0, m_pHead, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
482 t = res.guards.protect( 1, m_pTail, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
483 pNext = res.guards.protect( 2, h->m_pNext, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
484
485 if ( h == m_pHead.load( memory_model::memory_order_acquire )) {
486 if ( h.ptr() == t.ptr()) {
487 if ( !pNext.ptr()) {
488 m_Stat.onEmptyDequeue();
489 return false;
490 }
491
492 {
493 typename gc::Guard g;
494 while ( pNext->m_pNext.load(memory_model::memory_order_relaxed).ptr() && m_pTail.load(memory_model::memory_order_relaxed) == t ) {
495 pNext = g.protect( pNext->m_pNext, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
496 res.guards.copy( 2, g );
497 }
498 }
499
500 m_pTail.compare_exchange_weak( t, marked_ptr(pNext.ptr()), memory_model::memory_order_acquire, atomics::memory_order_relaxed );
501 }
502 else {
503 marked_ptr iter( h );
504 size_t hops = 0;
505
506 typename gc::Guard g;
507
508 while ( pNext.ptr() && pNext.bits() && iter.ptr() != t.ptr() && m_pHead.load(memory_model::memory_order_relaxed) == h ) {
509 iter = pNext;
510 g.assign( res.guards.template get<value_type>(2));
511 pNext = res.guards.protect( 2, pNext->m_pNext, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
512 ++hops;
513 }
514
515 if ( m_pHead.load(memory_model::memory_order_relaxed) != h )
516 continue;
517
518 if ( iter.ptr() == t.ptr())
519 free_chain( h, iter );
520 else if ( bDeque ) {
521 res.pNext = pNext.ptr();
522
523 if ( iter->m_pNext.compare_exchange_weak( pNext, marked_ptr( pNext.ptr(), 1 ), memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
524 if ( hops >= m_nMaxHops )
525 free_chain( h, pNext );
526 break;
527 }
528 }
529 else
530 return true;
531 }
532 }
533
534 if ( bDeque )
535 m_Stat.onDequeueRace();
536 bkoff();
537 }
538
539 if ( bDeque ) {
540 --m_ItemCounter;
541 m_Stat.onDequeue();
542 }
543
544 return true;
545 }

547 void free_chain( marked_ptr head, marked_ptr newHead )
548 {
549 // "head" and "newHead" are guarded
550
551 if ( m_pHead.compare_exchange_strong( head, marked_ptr(newHead.ptr()), memory_model::memory_order_release, atomics::memory_order_relaxed ))
552 {
553 typename gc::template GuardArray<2> guards;
554 guards.assign( 0, node_traits::to_value_ptr(head.ptr()));
555 while ( head.ptr() != newHead.ptr()) {
556 marked_ptr pNext = guards.protect( 1, head->m_pNext, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
557 assert( pNext.bits() != 0 );
558 dispose_node( head.ptr());
559 guards.copy( 0, 1 );
560 head = pNext;
561 }
562 }

629 bool enqueue( value_type& val )
630 {
631 node_type * pNew = node_traits::to_node_ptr( val );
632 link_checker::is_empty( pNew );
633
634 typename gc::Guard guard;
635 typename gc::Guard gNext;
636 back_off bkoff;
637
638 marked_ptr t;
639 while ( true ) {
640 t = guard.protect( m_pTail, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
641
642 marked_ptr pNext = t->m_pNext.load(memory_model::memory_order_relaxed );
643
644 if ( pNext.ptr() == nullptr ) {
645 pNew->m_pNext.store( marked_ptr(), memory_model::memory_order_relaxed );
646 if ( t->m_pNext.compare_exchange_weak( pNext, marked_ptr(pNew), memory_model::memory_order_release, atomics::memory_order_relaxed )) {
647 if ( !m_pTail.compare_exchange_strong( t, marked_ptr(pNew), memory_model::memory_order_release, atomics::memory_order_relaxed ))
648 m_Stat.onAdvanceTailFailed();
649 break;
650 }
651
652 // Try adding to basket
653 m_Stat.onTryAddBasket();
654
655 // Reread tail next
656 try_again:
657 pNext = gNext.protect( t->m_pNext, []( marked_ptr p ) -> value_type * { return node_traits::to_value_ptr( p.ptr());});
658
659 // add to the basket
660 if ( m_pTail.load( memory_model::memory_order_relaxed ) == t
661 && t->m_pNext.load( memory_model::memory_order_relaxed) == pNext
662 && !pNext.bits())
663 {
664 bkoff();
665 pNew->m_pNext.store( pNext, memory_model::memory_order_relaxed );
666 if ( t->m_pNext.compare_exchange_weak( pNext, marked_ptr( pNew ), memory_model::memory_order_release, atomics::memory_order_relaxed )) {
667 m_Stat.onAddBasket();
668 break;
669 }
670 goto try_again;
671 }
672 }
673 else {
674 // Tail is misplaced, advance it
675
676 typename gc::template GuardArray<2> g;
677 g.assign( 0, node_traits::to_value_ptr( pNext.ptr()));
678 if ( m_pTail.load( memory_model::memory_order_acquire ) != t
679 || t->m_pNext.load( memory_model::memory_order_relaxed ) != pNext )
680 {
681 m_Stat.onEnqueueRace();
682 bkoff();
683 continue;
684 }
685
686 marked_ptr p;
687 bool bTailOk = true;
688 while ( (p = pNext->m_pNext.load( memory_model::memory_order_acquire )).ptr() != nullptr )
689 {
690 bTailOk = m_pTail.load( memory_model::memory_order_relaxed ) == t;
691 if ( !bTailOk )
692 break;
693
694 g.assign( 1, node_traits::to_value_ptr( p.ptr()));
695 if ( pNext->m_pNext.load( memory_model::memory_order_relaxed ) != p )
696 continue;
697 pNext = p;
698 g.assign( 0, g.template get<value_type>( 1 ));
699 }
700 if ( !bTailOk || !m_pTail.compare_exchange_weak( t, marked_ptr( pNext.ptr()), memory_model::memory_order_release, atomics::memory_order_relaxed ))
701 m_Stat.onAdvanceTailFailed();
702
703 m_Stat.onBadTail();
704 }
705
706 m_Stat.onEnqueueRace();
707 }
708
709 ++m_ItemCounter;
710 m_Stat.onEnqueue();
711
712 return true;
713 }
