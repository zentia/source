80 {
81 friend class smr;
82 public:
83 thread_hp_storage( guard* arr, size_t nSize ) noexcept
84 : free_head_( arr )
85 , array_( arr )
86 , initial_capacity_( nSize )
87# ifdef CDS_ENABLE_HPSTAT
88 , alloc_guard_count_( 0 )
89 , free_guard_count_( 0 )
90 , extend_call_count_( 0 )
91# endif
92 {
93 // Initialize guards
94 new( arr ) guard[nSize];
95 extended_list_.store( nullptr, atomics::memory_order_release );
96 }
97
98 thread_hp_storage() = delete;
99 thread_hp_storage( thread_hp_storage const& ) = delete;
100 thread_hp_storage( thread_hp_storage&& ) = delete;
101
102 ~thread_hp_storage()
103 {
104 clear();
105 }
106
107 guard* alloc()
108 {
109 if ( cds_unlikely( free_head_ == nullptr )) {
110 extend();
111 assert( free_head_ != nullptr );
112 }
113
114 guard* g = free_head_;
115 free_head_ = g->next_;
116 CDS_HPSTAT( ++alloc_guard_count_ );
117 return g;
118 }
119
120 void free( guard* g ) noexcept
121 {
122 if ( g ) {
123 g->clear();
124 g->next_ = free_head_;
125 free_head_ = g;
126 CDS_HPSTAT( ++free_guard_count_ );
127 }
128 }
129
130 template< size_t Capacity>
131 size_t alloc( guard_array<Capacity>& arr )
132 {
133 for ( size_t i = 0; i < Capacity; ++i ) {
134 if ( cds_unlikely( free_head_ == nullptr ))
135 extend();
136 arr.reset( i, free_head_ );
137 free_head_ = free_head_->next_;
138 }
139 CDS_HPSTAT( alloc_guard_count_ += Capacity );
140 return Capacity;
141 }
142
143 template <size_t Capacity>
144 void free( guard_array<Capacity>& arr ) noexcept
145 {
146 guard* gList = free_head_;
147 for ( size_t i = 0; i < Capacity; ++i ) {
148 guard* g = arr[i];
149 if ( g ) {
150 g->clear();
151 g->next_ = gList;
152 gList = g;
153 CDS_HPSTAT( ++free_guard_count_ );
154 }
155 }
156 free_head_ = gList;
157 }
158
159 void clear()
160 {
161 // clear array_
162 for ( guard* cur = array_, *last = array_ + initial_capacity_; cur < last; ++cur )
163 cur->clear();
164
165 // free all extended blocks
166 hp_allocator& a = hp_allocator::instance();
167 for ( guard_block* p = extended_list_.load( atomics::memory_order_relaxed ); p; ) {
168 guard_block* next = p->next_block_;
169 a.free( p );
170 p = next;
171 }
172
173 extended_list_.store( nullptr, atomics::memory_order_release );
174 }
175
176 void init()
177 {
178 assert( extended_list_.load(atomics::memory_order_relaxed) == nullptr );
179
180 guard* p = array_;
181 for ( guard* pEnd = p + initial_capacity_ - 1; p != pEnd; ++p )
182 p->next_ = p + 1;
183 p->next_ = nullptr;
184 free_head_ = array_;
185 }
186
187 private:
188 void extend()
189 {
190 assert( free_head_ == nullptr );
191
192 guard_block* block = hp_allocator::instance().alloc();
193 block->next_block_ = extended_list_.load( atomics::memory_order_relaxed );
194 extended_list_.store( block, atomics::memory_order_release );
195 free_head_ = block->first();
196 CDS_HPSTAT( ++extend_call_count_ );
197 }
198
199 private:
200 guard* free_head_;
201 atomics::atomic<guard_block*> extended_list_;
202 guard* const array_;
203 size_t const initial_capacity_;
204# ifdef CDS_ENABLE_HPSTAT
205 public:
206 size_t alloc_guard_count_;
207 size_t free_guard_count_;
208 size_t extend_call_count_;
209# endif
210 };

269 class retired_array
270 {
271 friend class smr;
272 public:
273 retired_array() noexcept
274 : current_block_( nullptr )
275 , current_cell_( nullptr )
276 , list_head_( nullptr )
277 , list_tail_( nullptr )
278 , block_count_(0)
279# ifdef CDS_ENABLE_HPSTAT
280 , retire_call_count_( 0 )
281 , extend_call_count_( 0 )
282# endif
283 {}
284
285 retired_array( retired_array const& ) = delete;
286 retired_array( retired_array&& ) = delete;
287
288 ~retired_array()
289 {
290 assert( empty());
291 fini();
292 }
293
294 bool push( retired_ptr const& p ) noexcept
295 {
296 assert( current_block_ != nullptr );
297 assert( current_block_->first() <= current_cell_ );
298 assert( current_cell_ < current_block_->last());
299 //assert( &p != current_cell_ );
300
301 *current_cell_ = p;
302 CDS_HPSTAT( ++retire_call_count_ );
303
304 if ( ++current_cell_ == current_block_->last()) {
305 // goto next block if exists
306 if ( current_block_->next_ ) {
307 current_block_ = current_block_->next_;
308 current_cell_ = current_block_->first();
309 return true;
310 }
311
312 // no free block
313 // basic_smr::scan() extend retired_array if needed
314 return false;
315 }
316
317 return true;
318 }
319
320 bool repush( retired_ptr* p ) noexcept
321 {
322 bool ret = push( *p );
323 CDS_HPSTAT( --retire_call_count_ );
324 assert( ret );
325 return ret;
326 }
327
328 private: // called by basic_smr
329 void init()
330 {
331 if ( list_head_ == nullptr ) {
332 retired_block* block = retired_allocator::instance().alloc();
333 assert( block->next_ == nullptr );
334
335 current_block_ =
336 list_head_ =
337 list_tail_ = block;
338 current_cell_ = block->first();
339
340 block_count_ = 1;
341 }
342 }
343
344 void fini()
345 {
346 retired_allocator& alloc = retired_allocator::instance();
347 for ( retired_block* p = list_head_; p; ) {
348 retired_block* next = p->next_;
349 alloc.free( p );
350 p = next;
351 }
352
353 current_block_ =
354 list_head_ =
355 list_tail_ = nullptr;
356 current_cell_ = nullptr;
357
358 block_count_ = 0;
359 }
360
361 void extend()
362 {
363 assert( list_head_ != nullptr );
364 assert( current_block_ == list_tail_ );
365 assert( current_cell_ == current_block_->last());
366
367 retired_block* block = retired_allocator::instance().alloc();
368 assert( block->next_ == nullptr );
369
370 current_block_ = list_tail_ = list_tail_->next_ = block;
371 current_cell_ = block->first();
372 ++block_count_;
373 CDS_HPSTAT( ++extend_call_count_ );
374 }
375
376 bool empty() const
377 {
378 return current_block_ == nullptr
379 || ( current_block_ == list_head_ && current_cell_ == current_block_->first());
380 }
381
382 private:
383 retired_block* current_block_;
384 retired_ptr* current_cell_; // in current_block_
385
386 retired_block* list_head_;
387 retired_block* list_tail_;
388 size_t block_count_;
389# ifdef CDS_ENABLE_HPSTAT
390 public:
391 size_t retire_call_count_;
392 size_t extend_call_count_;
393# endif
394 };
