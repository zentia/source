251 static bool link_node( node_type * pNode, position& pos )
252 {
253 assert( pNode != nullptr );
254 link_checker::is_empty( pNode );
255
256 marked_node_ptr cur(pos.pCur);
257 pNode->m_pNext.store( cur, memory_model::memory_order_release );
258 if ( cds_likely( pos.pPrev->compare_exchange_strong( cur, marked_node_ptr(pNode), memory_model::memory_order_release, atomics::memory_order_relaxed )))
259 return true;
260
261 pNode->m_pNext.store( marked_node_ptr(), memory_model::memory_order_relaxed );
262 return false;
263 }
264
265 static bool unlink_node( position& pos )
266 {
267 assert( pos.pPrev != nullptr );
268 assert( pos.pCur != nullptr );
269
270 // Mark the node (logical deleting)
271 marked_node_ptr next(pos.pNext, 0);
272 if ( cds_likely( pos.pCur->m_pNext.compare_exchange_strong( next, marked_node_ptr(pos.pNext, 1), memory_model::memory_order_release, atomics::memory_order_relaxed ))) {
273 // physical deletion may be performed by search function if it detects that a node is logically deleted (marked)
274 // CAS may be successful here or in other thread that searching something
275 marked_node_ptr cur(pos.pCur);
276 if ( cds_likely( pos.pPrev->compare_exchange_strong( cur, marked_node_ptr( pos.pNext ), memory_model::memory_order_acquire, atomics::memory_order_relaxed )))
277 retire_node( pos.pCur );
278 return true;
279 }
280 return false;
281 }

295 void next()
296 {
297 if ( m_pNode ) {
298 typename gc::Guard g;
299 node_type * pCur = node_traits::to_node_ptr( *m_pNode );
300
301 marked_node_ptr pNext;
302 do {
303 pNext = pCur->m_pNext.load(memory_model::memory_order_relaxed);
304 g.assign( node_traits::to_value_ptr( pNext.ptr()));
305 } while ( cds_unlikely( pNext != pCur->m_pNext.load(memory_model::memory_order_acquire)));
306
307 if ( pNext.ptr())
308 m_pNode = m_Guard.assign( g.template get<value_type>());
309 else {
310 m_pNode = nullptr;
311 m_Guard.clear();
312 }
313 }
314 }

316 iterator_type( atomic_node_ptr const& pNode )
317 {
318 for (;;) {
319 marked_node_ptr p = pNode.load(memory_model::memory_order_relaxed);
320 if ( p.ptr()) {
321 m_pNode = m_Guard.assign( node_traits::to_value_ptr( p.ptr()));
322 }
323 else {
324 m_pNode = nullptr;
325 m_Guard.clear();
326 }
327 if ( cds_likely( p == pNode.load(memory_model::memory_order_acquire)))
328 break;
329 }

880 void clear()
881 {
882 typename gc::Guard guard;
883 marked_node_ptr head;
884 while ( true ) {
885 head = m_pHead.load(memory_model::memory_order_relaxed);
886 if ( head.ptr())
887 guard.assign( node_traits::to_value_ptr( *head.ptr()));
888 if ( cds_likely( m_pHead.load(memory_model::memory_order_acquire) == head )) {
889 if ( head.ptr() == nullptr )
890 break;
891 value_type& val = *node_traits::to_value_ptr( *head.ptr());
892 unlink( val );
893 }

987 template <typename Func>
988 std::pair<bool, bool> update_at( atomic_node_ptr& refHead, value_type& val, Func func, bool bInsert )
989 {
990 position pos;
991
992 node_type * pNode = node_traits::to_node_ptr( val );
993 while ( true ) {
994 if ( search( refHead, val, pos, key_comparator())) {
995 if ( cds_unlikely( pos.pCur->m_pNext.load(memory_model::memory_order_acquire).bits())) {
996 back_off()();
997 m_Stat.onUpdateMarked();
998 continue; // the node found is marked as deleted
999 }
1000 assert( key_comparator()( val, *node_traits::to_value_ptr( *pos.pCur )) == 0 );
1001
1002 func( false, *node_traits::to_value_ptr( *pos.pCur ) , val );
1003 m_Stat.onUpdateExisting();
1004 return std::make_pair( true, false );
1005 }
1006 else {
1007 if ( !bInsert ) {
1008 m_Stat.onUpdateFailed();
1009 return std::make_pair( false, false );
1010 }
1011
1012 typename gc::Guard guard;
1013 guard.assign( &val );
1014 if ( link_node( pNode, pos )) {
1015 ++m_ItemCounter;
1016 func( true, val, val );
1017 m_Stat.onUpdateNew();
1018 return std::make_pair( true, true );
1019 }
1020 }
1021
1022 m_Stat.onUpdateRetry();
1023 }

1026 bool unlink_at( atomic_node_ptr& refHead, value_type& val )
1027 {
1028 position pos;
1029
1030 back_off bkoff;
1031 while ( search( refHead, val, pos, key_comparator())) {
1032 if ( node_traits::to_value_ptr( *pos.pCur ) == &val ) {
1033 if ( unlink_node( pos )) {
1034 --m_ItemCounter;
1035 m_Stat.onEraseSuccess();
1036 return true;
1037 }

1039 bkoff();
1040 }
1041 else {
1042 m_Stat.onUpdateFailed();
1043 break;

1045
1046 m_Stat.onEraseRetry();
1047 }

1049 m_Stat.onEraseFailed();

1160 template <typename Q, typename Compare >
1161 bool search( atomic_node_ptr& refHead, const Q& val, position& pos, Compare cmp )
1162 {
1163 atomic_node_ptr * pPrev;
1164 marked_node_ptr pNext;
1165 marked_node_ptr pCur;
1166
1167 back_off bkoff;
1168
1169 try_again:
1170 pPrev = &refHead;
1171 pNext = nullptr;
1172
1173 pCur = pos.guards.protect( position::guard_current_item, *pPrev,
1174 [](marked_node_ptr p) -> value_type *
1175 {
1176 return node_traits::to_value_ptr( p.ptr());
1177 });
1178
1179 while ( true ) {
1180 if ( pCur.ptr() == nullptr ) {
1181 pos.pPrev = pPrev;
1182 pos.pCur = nullptr;
1183 pos.pNext = nullptr;
1184 return false;
1185 }
1186
1187 pNext = pos.guards.protect( position::guard_next_item, pCur->m_pNext,
1188 [](marked_node_ptr p ) -> value_type *
1189 {
1190 return node_traits::to_value_ptr( p.ptr());
1191 });
1192 if ( cds_unlikely( pPrev->load(memory_model::memory_order_acquire).all() != pCur.ptr())) {
1193 bkoff();
1194 goto try_again;
1195 }
1196
1197 // pNext contains deletion mark for pCur
1198 if ( pNext.bits() == 1 ) {
1199 // pCur marked i.e. logically deleted. Help the erase/unlink function to unlink pCur node
1200 marked_node_ptr cur( pCur.ptr());
1201 if ( cds_unlikely( pPrev->compare_exchange_strong( cur, marked_node_ptr( pNext.ptr()), memory_model::memory_order_acquire, atomics::memory_order_relaxed ))) {
1202 retire_node( pCur.ptr());
1203 m_Stat.onHelpingSuccess();
1204 }
1205 else {
1206 bkoff();
1207 m_Stat.onHelpingFailed();
1208 goto try_again;
1209 }
1210 }
1211 else {
1212 assert( pCur.ptr() != nullptr );
1213 int nCmp = cmp( *node_traits::to_value_ptr( pCur.ptr()), val );
1214 if ( nCmp >= 0 ) {
1215 pos.pPrev = pPrev;
1216 pos.pCur = pCur.ptr();
1217 pos.pNext = pNext.ptr();
1218 return nCmp == 0;
1219 }
1220 pPrev = &( pCur->m_pNext );
1221 pos.guards.copy( position::guard_prev_item, position::guard_current_item );
1222 }
1223 pCur = pNext;
1224 pos.guards.copy( position::guard_current_item, position::guard_next_item );
1225 }
1226 }
