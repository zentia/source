224template<typename T, bool has_step>
225class ForRange {
226
227 static_assert(is_integral_v<T>);
228
229public:
230 struct ForRangeEnd {};
231
233 class ForRangeIter {
234 private:
235 Expr<T> _begin;
236 Expr<T> _end;
237 Expr<T> _step;
238 const Expression *_var{nullptr};
239 const Expression *_cond{nullptr};
240 ForStmt *_stmt{nullptr};
241 uint _time{0u};
242
243 public:
245 explicit ForRangeIter(Expr<T> begin, Expr<T> end, Expr<T> step) noexcept
246 : _begin{begin}, _end{end}, _step{step} {}
247
249 [[nodiscard]] auto operator*() noexcept {
250 assert(_time == 0u && "Can only call operator* once");
251 auto f = FunctionBuilder::current();
252 Var var{_begin};
253 _var = var.expression();
254 auto bool_type = Type::of<bool>();
255 _cond = f->binary(bool_type, BinaryOp::LESS, _var, _end.expression());
256 if constexpr (has_step) {
257 // step < 0
258 auto neg_step = f->binary(bool_type, BinaryOp::LESS, _step.expression(), f->literal(Type::of<T>(), T{0}));
259 // ((step < 0) && !(var < end)) || (!(step < 0) && (var < end))
260 _cond = f->binary(bool_type, BinaryOp::BIT_XOR, _cond, neg_step);
261 }
262 _stmt = f->for_(_var, _cond, _step.expression());
263 f->push_scope(_stmt->body());
264 return Var{std::move(var)};// to guarantee rvo
265 }
266
267 auto &operator++() noexcept {
268 _time++;
269 FunctionBuilder::current()->pop_scope(_stmt->body());
270 return *this;
271 }
272
273 [[nodiscard]] auto operator!=(ForRangeEnd) const noexcept {
274 return _time == 0u;
275 }
276 };
277
278private:
279 Expr<T> _begin;
280 Expr<T> _end;
281 Expr<T> _step;
282
283public:
284 explicit ForRange(Expr<T> begin, Expr<T> end, Expr<T> step) noexcept
285 : _begin{begin}, _end{end}, _step{step} {}
286 [[nodiscard]] auto begin() const noexcept { return ForRangeIter{_begin, _end, _step}; }
287 [[nodiscard]] auto end() const noexcept { return ForRangeEnd{}; }
288};

306template<typename Lhs, typename Rhs>
307void assign(Lhs &&lhs, Rhs &&rhs) noexcept {
308 static_assert(
309 std::tuple_size_v<linear_layout_t<expr_value_t<Lhs>>> ==
310 std::tuple_size_v<linear_layout_t<expr_value_t<Rhs>>>);
311 if constexpr (concepts::assignable<expr_value_t<Lhs>, expr_value_t<Rhs>>) {
312 detail::FunctionBuilder::current()->assign(
313 detail::extract_expression(std::forward<Lhs>(lhs)),
314 detail::extract_expression(std::forward<Rhs>(rhs)));
315 } else if (is_tuple_v<std::remove_cvref_t<Rhs>>) {
316 assign(
317 detail::Ref{std::forward<Lhs>(lhs)},
318 compose(std::forward<Rhs>(rhs)));
319 } else {
320 detail::assign_impl(
321 detail::Ref{std::forward<Lhs>(lhs)},
322 Expr{std::forward<Rhs>(rhs)});
323 }
