112 template <typename Key, typename Val>
113 struct set_type< tag_FeldmanHashSet, Key, Val >: public set_type_base< Key, Val >
114 {
115 typedef set_type_base< Key, Val > base_class;
116 typedef typename base_class::compare compare;
117 typedef typename base_class::less less;
118 typedef typename base_class::hash hash;
119 typedef typename base_class::key_type key_type;
120 typedef typename base_class::value_type value_type;
121
122 template <typename Hasher>
123 struct hash_type
124 {
125 typedef Hasher hasher;
126 typedef typename hasher::hash_type type;
127 };
128
129 template <typename TH>
130 struct hash_type<std::hash<TH>>
131 {
132 typedef std::hash<TH> hasher;
133 typedef size_t type;
134 };
135
136 template <typename Hasher>
137 struct key_val: base_class::key_val
138 {
139 typedef typename base_class::key_val base;
140 typedef Hasher hasher;
141 typedef typename hash_type<hasher>::type hash_type;
142
143 hash_type hash;
144
145 explicit key_val( key_type const& k ): base(k), hash( hasher()( k )) {}
146 key_val( key_type const& k, value_type const& v ): base(k, v), hash( hasher()( k )) {}
147
148 template <typename K>
149 explicit key_val( K const& k ): base(k), hash( hasher()( k )) {}
150
151 template <typename K, typename T>
152 key_val( K const& k, T const& v ): base(k, v), hash( hasher()( k )) {}
153 };
