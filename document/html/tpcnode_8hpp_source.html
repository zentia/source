429 // NOTE: this implementation supposes that the number of arguments does not
430 // change between two different tasks ! Only the size does change.
431 void *svc(void *task) {
432 if (task) {
433 Task.resetTask();
434 Task.setTask((IN_t*)task);
435 }
436
437#if defined(FF_REPARA)
438 rpr_sizeIn = rpr_sizeOut = 0;
439#endif
440
441 const tpc_func_id_t f_id = Task.getKernelId();
442 if (tpc_device_func_instance_count(dev_ctx, f_id) == 0) {
443 error("No instances for function with id %d\n", (int)f_id);
444 return GO_ON;
445 }
446 tpc_res_t res;
447 tpc_job_id_t j_id = tpc_device_acquire_job_id(dev_ctx, f_id, TPC_DEVICE_ACQUIRE_JOB_ID_BLOCKING);
448
449 const std::vector<internal::Arg_t> &inV = Task.getInputArgs();
450 const std::vector<internal::Arg_t> &outV = Task.getOutputArgs();
451 size_t ret_val_size=0;
452 void *ret_val = Task.getReturnVar(ret_val_size);
453
454 if (inHandles.size() == 0) inHandles.resize(inV.size());
455 if (outHandles.size() == 0) outHandles.resize(outV.size());
456
457
458 uint32_t arg_idx = 0;
459
460 for(size_t i=0;i<inV.size();++i) {
461
462 if ( ! inV[i].byvalue ) {
463 tpc_handle_t handle = inHandles[i].handle; // previous handle
464 if (inHandles[i].size < inV[i].size) {
465
466 if (inHandles[i].handle) { // not first time
467 assert(!inV[i].reuse);
468 allocator->releaseBuffer(inHandles[i].ptr, dev_ctx, inHandles[i].handle);
469 }
470 if (inV[i].reuse) {
471 handle = allocator->createBufferUnique(inV[i].ptr, dev_ctx, TPC_DEVICE_ALLOC_FLAGS_NONE, inV[i].size);
472 } else {
473 handle = allocator->createBuffer(inV[i].ptr, dev_ctx, TPC_DEVICE_ALLOC_FLAGS_NONE, inV[i].size);
474 }
475
476 if (!handle) {
477 error("ff_tpcNode::svc unable to allocate memory on the TPC device (IN)");
478 inHandles[i].size = 0, inHandles[i].ptr = nullptr, inHandles[i].handle = 0;
479 return (oneshot?NULL:GO_ON);
480 }
481 inHandles[i].size = inV[i].size, inHandles[i].ptr = inV[i].ptr, inHandles[i].handle = handle;
482 }
483
484 if (inV[i].copy) {
485 res = tpc_device_copy_to(dev_ctx, inV[i].ptr, handle, inV[i].size, TPC_DEVICE_COPY_BLOCKING);
486 if (res != TPC_SUCCESS) {
487 error("ff_tpcNode::svc unable to copy data into the device\n");
488 return (oneshot?NULL:GO_ON);
489 }
490#if defined(FF_REPARA)
491 rpr_sizeIn += inV[i].size;
492#endif
493 }
494
495 res = tpc_device_job_set_arg(dev_ctx, j_id, arg_idx, sizeof(handle), &handle);
496
497 } else { // argument passed by-value
498 res = tpc_device_job_set_arg(dev_ctx, j_id, arg_idx, inV[i].size, inV[i].ptr);
499 }
500
501 if (res != TPC_SUCCESS) {
502 error("ff_tpcNode::svc unable to set argument for the device (IN)\n");
503 return (oneshot?NULL:GO_ON);
504 }
505 ++arg_idx;
506 }
507
508 for(size_t i=0;i<outV.size();++i) {
509 void *ptr = outV[i].ptr;
510
511 // check if ptr is an in/out parameter
512 typename std::vector<handle_t>::iterator it;
513 it = std::find_if(inHandles.begin(), inHandles.end(), [ptr](const handle_t &arg) { return arg.ptr == ptr; } );
514
515 tpc_handle_t handle;
516 if (it == inHandles.end()) { // not found
517 handle = outHandles[i].handle; // previous handle
518 if (outHandles[i].size < outV[i].size) {
519
520 if (outHandles[i].handle) { // not first time
521 assert(!outV[i].reuse);
522 allocator->releaseBuffer(outHandles[i].ptr, dev_ctx, outHandles[i].handle);
523 }
524
525 if (outV[i].reuse) {
526 handle = allocator->createBufferUnique(ptr, dev_ctx, TPC_DEVICE_ALLOC_FLAGS_NONE, outV[i].size);
527 } else {
528 handle = allocator->createBuffer(ptr, dev_ctx, TPC_DEVICE_ALLOC_FLAGS_NONE, outV[i].size);
529 }
530
531 if (!handle) {
532 error("ff_tpcNode::svc unable to allocate memory on the TPC device (OUT)");
533 outHandles[i].size = 0, outHandles[i].ptr = nullptr, outHandles[i].handle = 0;
534 return (oneshot?NULL:GO_ON);
535 }
536 outHandles[i].size = outV[i].size, outHandles[i].ptr = ptr, outHandles[i].handle = handle;
537 }
538 } else handle= (*it).handle; // in/out parameter
539
540 res = tpc_device_job_set_arg(dev_ctx, j_id, arg_idx, sizeof(handle), &handle);
541 if (res != TPC_SUCCESS) {
542 error("ff_tpcNode::svc unable to set argument for the device (OUT)\n");
543 return (oneshot?NULL:GO_ON);
544 }
545 ++arg_idx;
546 }
547 res = tpc_device_job_launch(dev_ctx, j_id, TPC_DEVICE_JOB_LAUNCH_BLOCKING);
548 if (res != TPC_SUCCESS) {
549 error("ff_tpcNode::svc error launching kernel on TPC device\n");
550 return (oneshot?NULL:GO_ON);
551 }
552
553 // reset the arg_idx to the initial value for the out parameters
554 arg_idx=inV.size();
555
556 // TODO: async data transfer
557 for(size_t i=0;i<outV.size();++i) {
558 tpc_handle_t handle;
559 res = tpc_device_job_get_arg(dev_ctx, j_id, arg_idx, sizeof(handle), &handle);
560 if (res != TPC_SUCCESS) {
561 error("ff_tpcNode::svc unable to get argument handle\n");
562 return (oneshot?NULL:GO_ON);
563 }
564 if (outV[i].copy) {
565 res = tpc_device_copy_from(dev_ctx, handle, outV[i].ptr, outV[i].size, TPC_DEVICE_COPY_BLOCKING);
566 if (res != TPC_SUCCESS) {
567 error("ff_tpcNode::svc unable to copy data back from the device\n");
568 return (oneshot?NULL:GO_ON);
569 }
570#if defined(FF_REPARA)
571 rpr_sizeOut += outV[i].size;
572#endif
573 }
574
575 // By default the buffers are not released !
576 if (outV[i].release) {
577 assert(outHandles[i].handle == handle);
578 allocator->releaseBuffer(outHandles[i].ptr, dev_ctx, handle);
579 outHandles[i].size = 0, outHandles[i].ptr = nullptr, outHandles[i].handle = 0;
580 }
581
582 ++arg_idx;
583 }
584
585 // now check if we have to get back the return value (if the kernel is a function)
586 if (ret_val && ret_val_size>0) {
587 res = tpc_device_job_get_return(dev_ctx, j_id, ret_val_size, ret_val);
588 if (res != TPC_SUCCESS) {
589 error("ff_tpcNode::svc error getting back the return value of the kernel function\n");
590 return (oneshot?NULL:GO_ON);
591 }
592 }
593
594 // By default the buffers are not released !
595 for(size_t i=0;i<inV.size();++i) {
596 if (inV[i].release) {
597 allocator->releaseBuffer(inHandles[i].ptr, dev_ctx, inHandles[i].handle);
598 inHandles[i].size = 0, inHandles[i].ptr = nullptr, inHandles[i].handle = 0;
599 }
600 }
601
602 // release job id
603 tpc_device_release_job_id(dev_ctx, j_id);
604
605 // per task host memory cleanup phase
606 OUT_t * task_out;
607 if (task) task_out = Task.releaseTask((IN_t*)task);
608 else task_out = Task.releaseTask(const_cast<IN_t*>(oneshotTask));
609
610 return (oneshot ? NULL : task_out);
611 }
612
613protected:
614 int tpcId;
615 tpc_dev_id_t deviceId;
616 tpc_dev_ctx_t *dev_ctx;
617
618 TPC_t Task;
619
620 bool my_own_allocator;
621 const bool oneshot;
622 const IN_t *const oneshotTask;
623 ff_tpcallocator *allocator;
624
625 std::vector<handle_t> inHandles;
626 std::vector<handle_t> outHandles;
627};
