29 // Most significant bit number (1..N)
30 // Return 0 if x == 0
31 //
32#ifndef cds_bitop_msb32_DEFINED
33 // Return number (1..32) of most significant bit
34 // Return 0 if x == 0
35 // Source: Linux kernel
36 static inline int msb32( uint32_t x )
37 {
38 int r = 32;
39
40 if (!x)
41 return 0;
42 if (!(x & 0xffff0000u)) {
43 x <<= 16;
44 r -= 16;
45 }
46 if (!(x & 0xff000000u)) {
47 x <<= 8;
48 r -= 8;
49 }
50 if (!(x & 0xf0000000u)) {
51 x <<= 4;
52 r -= 4;
53 }
54 if (!(x & 0xc0000000u)) {
55 x <<= 2;
56 r -= 2;
57 }
58 if (!(x & 0x80000000u)) {
59 //x <<= 1;
60 r -= 1;
61 }
62 return r;
63 }
64#endif

67 static inline int msb32nz( uint32_t x )

69 return msb32( x ) - 1;

94#ifndef cds_bitop_lsb32_DEFINED
95 // Return number (1..32) of least significant bit
96 // Return 0 if x == 0
97 // Source: Linux kernel
98 static inline int lsb32( uint32_t x )
99 {
100 int r = 1;
101
102 if (!x)
103 return 0;
104 if (!(x & 0xffff)) {
105 x >>= 16;
106 r += 16;
107 }
108 if (!(x & 0xff)) {
109 x >>= 8;
110 r += 8;
111 }
112 if (!(x & 0xf)) {
113 x >>= 4;
114 r += 4;
115 }
116 if (!(x & 3)) {
117 x >>= 2;
118 r += 2;
119 }
120 if (!(x & 1)) {
121 //x >>= 1;
122 r += 1;
123 }
124 return r;
125 }
126#endif

153 //******************************************************
154 // Reverse bit order
155 //******************************************************
156#ifndef cds_bitop_rbo32_DEFINED
157 static inline uint32_t rbo32( uint32_t x )
158 {
159 // swap odd and even bits
160 x = ((x >> 1) & 0x55555555) | ((x & 0x55555555) << 1);
161 // swap consecutive pairs
162 x = ((x >> 2) & 0x33333333) | ((x & 0x33333333) << 2);
163 // swap nibbles ...
164 x = ((x >> 4) & 0x0F0F0F0F) | ((x & 0x0F0F0F0F) << 4);
165 // swap bytes
166 x = ((x >> 8) & 0x00FF00FF) | ((x & 0x00FF00FF) << 8);
167 // swap 2-byte long pairs
168 return ( x >> 16 ) | ( x << 16 );
169 }
170#endif

172#ifndef cds_bitop_rbo64_DEFINED
173 static inline uint64_t rbo64( uint64_t x )
174 {
175 // Low 32bit Hight 32bit
176 return ( static_cast<uint64_t>(rbo32( (uint32_t) x )) << 32 ) | ( static_cast<uint64_t>( rbo32( (uint32_t) (x >> 32))));
177 }
178#endif

180 //******************************************************
181 // Set bit count. Return count of non-zero bits in word
182 //******************************************************
183#ifndef cds_bitop_sbc32_DEFINED
184 static inline int sbc32( uint32_t x )
185 {
186# ifdef cds_beans_zbc32_DEFINED
187 return 32 - zbc32( x );
188# else
189 // Algorithm from Sean Eron Anderson's great collection
190 x = x - ((x >> 1) & 0x55555555);
191 x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
192 return (((x + (x >> 4)) & 0xF0F0F0F) * 0x1010101) >> 24;
193# endif
194 }
195#endif

204# endif
