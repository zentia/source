91INLINE static int usleep(unsigned long microsecs) {
92 if (microsecs > 100000)
93 /* At least 100 mS. Typical best resolution is ~ 15ms */
94 Sleep (microsecs/ 1000);
95 else
96 {
97 /* Use Sleep for the largest part, and busy-loop for the rest. */
98 static double frequency;
99 if (frequency == 0)
100 {
101 LARGE_INTEGER freq;
102 if (!QueryPerformanceFrequency (&freq))
103 {
104 /* Cannot use QueryPerformanceCounter. */
105 Sleep (microsecs / 1000);
106 return 0;
107 }
108 frequency = (double) freq.QuadPart / 1000000000.0;
109 }
110 long long expected_counter_difference = 1000 * microsecs * (long long) frequency;
111 int sleep_part = (int) (microsecs) / 1000 - 10;
112 LARGE_INTEGER before;
113 QueryPerformanceCounter (&before);
114 long long expected_counter = before.QuadPart +
115expected_counter_difference;
116 if (sleep_part > 0)
117 Sleep (sleep_part);
118 for (;;)
119 {
120 LARGE_INTEGER after;
121 QueryPerformanceCounter (&after);
122 if (after.QuadPart >= expected_counter)
123 break;
124 }
125 }
126 return(0);
127}

156INLINE static int gettimeofday(struct timeval *tv, struct timezone *tz)
157{
158 FILETIME ft;
159 unsigned __int64 tmpres = 0;
160 static int tzflag;
161
162 if (NULL != tv)
163 {
164 GetSystemTimeAsFileTime(&ft);
165
166 tmpres |= ft.dwHighDateTime;
167 tmpres <<= 32;
168 tmpres |= ft.dwLowDateTime;
169
170 /*converting file time to unix epoch*/
171 tmpres -= DELTA_EPOCH_IN_MICROSECS;
172 tmpres /= 10; /*convert into microseconds*/
173 tv->tv_sec = (long)(tmpres / 1000000UL);
174 tv->tv_usec = (long)(tmpres % 1000000UL);
175 }
176
177 if (NULL != tz)
178 {
179 if (!tzflag)
180 {
181 _tzset();
182 tzflag++;
183 }
184 tz->tz_minuteswest = _timezone / 60;
185 tz->tz_dsttime = _daylight;
186 }
187
188 return 0;
189}
