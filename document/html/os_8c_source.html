71// round to a good OS allocation size (bounded by max 12.5% waste)
72size_t _mi_os_good_alloc_size(size_t size) {
73 size_t align_size;
74 if (size < 512*MI_KiB) align_size = _mi_os_page_size();
75 else if (size < 2*MI_MiB) align_size = 64*MI_KiB;
76 else if (size < 8*MI_MiB) align_size = 256*MI_KiB;
77 else if (size < 32*MI_MiB) align_size = 1*MI_MiB;
78 else align_size = 4*MI_MiB;
79 if mi_unlikely(size >= (SIZE_MAX - align_size)) return size; // possible overflow?
80 return _mi_align_up(size, align_size);
81}

149static void mi_os_prim_free(void* addr, size_t size, size_t commit_size) {
150 mi_assert_internal((size % _mi_os_page_size()) == 0);
151 if (addr == NULL || size == 0) return; // || _mi_os_is_huge_reserved(addr)
152 int err = _mi_prim_free(addr, size);
153 if (err != 0) {
154 _mi_warning_message("unable to free OS memory (error: %d (0x%x), size: 0x%zx bytes, address: %p)\n", err, err, size, addr);
155 }
156 if (commit_size > 0) {
157 mi_os_stat_decrease(committed, commit_size);
158 }
159 mi_os_stat_decrease(reserved, size);
160}

162void _mi_os_free_ex(void* addr, size_t size, bool still_committed, mi_memid_t memid) {
163 if (mi_memkind_is_os(memid.memkind)) {
164 size_t csize = memid.mem.os.size;
165 if (csize==0) { _mi_os_good_alloc_size(size); }
166 size_t commit_size = (still_committed ? csize : 0);
167 void* base = addr;
168 // different base? (due to alignment)
169 if (memid.mem.os.base != base) {
170 mi_assert(memid.mem.os.base <= addr);
171 base = memid.mem.os.base;
172 const size_t diff = (uint8_t*)addr - (uint8_t*)memid.mem.os.base;
173 if (memid.mem.os.size==0) {
174 csize += diff;
175 }
176 if (still_committed) {
177 commit_size -= diff; // the (addr-base) part was already un-committed
178 }
179 }
180 // free it
181 if (memid.memkind == MI_MEM_OS_HUGE) {
182 mi_assert(memid.is_pinned);
183 mi_os_free_huge_os_pages(base, csize);
184 }
185 else {
186 mi_os_prim_free(base, csize, (still_committed ? commit_size : 0));
187 }
188 }
189 else {
190 // nothing to do
191 mi_assert(memid.memkind < MI_MEM_OS);
192 }
193}

204// Note: the `try_alignment` is just a hint and the returned pointer is not guaranteed to be aligned.
205// Also `hint_addr` is a hint and may be ignored.
206static void* mi_os_prim_alloc_at(void* hint_addr, size_t size, size_t try_alignment, bool commit, bool allow_large, bool* is_large, bool* is_zero) {
207 mi_assert_internal(size > 0 && (size % _mi_os_page_size()) == 0);
208 mi_assert_internal(is_zero != NULL);
209 mi_assert_internal(is_large != NULL);
210 if (size == 0) return NULL;
211 if (!commit) { allow_large = false; }
212 if (try_alignment == 0) { try_alignment = 1; } // avoid 0 to ensure there will be no divide by zero when aligning
213 *is_zero = false;
214 void* p = NULL;
215 int err = _mi_prim_alloc(hint_addr, size, try_alignment, commit, allow_large, is_large, is_zero, &p);
216 if (err != 0) {
217 _mi_warning_message("unable to allocate OS memory (error: %d (0x%x), addr: %p, size: 0x%zx bytes, align: 0x%zx, commit: %d, allow large: %d)\n", err, err, hint_addr, size, try_alignment, commit, allow_large);
218 }
219
220 mi_os_stat_counter_increase(mmap_calls, 1);
221 if (p != NULL) {
222 mi_os_stat_increase(reserved, size);
223 if (commit) {
224 mi_os_stat_increase(committed, size);
225 // seems needed for asan (or `mimalloc-test-api` fails)
226 #ifdef MI_TRACK_ASAN
227 if (*is_zero) { mi_track_mem_defined(p,size); }
228 else { mi_track_mem_undefined(p,size); }
229 #endif
230 }
231 }
232 return p;
233}

240// Primitive aligned allocation from the OS.
241// This function guarantees the allocated memory is aligned.
242static void* mi_os_prim_alloc_aligned(size_t size, size_t alignment, bool commit, bool allow_large, bool* is_large, bool* is_zero, void** base) {
243 mi_assert_internal(alignment >= _mi_os_page_size() && ((alignment & (alignment - 1)) == 0));
244 mi_assert_internal(size > 0 && (size % _mi_os_page_size()) == 0);
245 mi_assert_internal(is_large != NULL);
246 mi_assert_internal(is_zero != NULL);
247 mi_assert_internal(base != NULL);
248 if (!commit) allow_large = false;
249 if (!(alignment >= _mi_os_page_size() && ((alignment & (alignment - 1)) == 0))) return NULL;
250 size = _mi_align_up(size, _mi_os_page_size());
251
252 // try a direct allocation if the alignment is below the default, or if larger than 1/8 fraction of the size.
253 const bool try_direct_alloc = (alignment <= mi_os_mem_config.alloc_granularity || alignment > size/8);
254
255 void* p = NULL;
256 if (try_direct_alloc) {
257 p = mi_os_prim_alloc(size, alignment, commit, allow_large, is_large, is_zero);
258 }
259
260 // aligned already?
261 if (p != NULL && ((uintptr_t)p % alignment) == 0) {
262 *base = p;
263 }
264 else {
265 // if not aligned, free it, overallocate, and unmap around it
266 #if !MI_TRACK_ASAN
267 if (try_direct_alloc) {
268 _mi_warning_message("unable to allocate aligned OS memory directly, fall back to over-allocation (size: 0x%zx bytes, address: %p, alignment: 0x%zx, commit: %d)\n", size, p, alignment, commit);
269 }
270 #endif
271 if (p != NULL) { mi_os_prim_free(p, size, (commit ? size : 0)); }
272 if (size >= (SIZE_MAX - alignment)) return NULL; // overflow
273 const size_t over_size = size + alignment;
274
275 if (!mi_os_mem_config.has_partial_free) { // win32 virtualAlloc cannot free parts of an allocated block
276 // over-allocate uncommitted (virtual) memory
277 p = mi_os_prim_alloc(over_size, 1 /*alignment*/, false /* commit? */, false /* allow_large */, is_large, is_zero);
278 if (p == NULL) return NULL;
279
280 // set p to the aligned part in the full region
281 // note: on Windows VirtualFree needs the actual base pointer
282 // this is handledby having the `base` field in the memid.
283 *base = p; // remember the base
284 p = _mi_align_up_ptr(p, alignment);
285
286 // explicitly commit only the aligned part
287 if (commit) {
288 _mi_os_commit(p, size, NULL);
289 }
290 }
291 else { // mmap can free inside an allocation
292 // overallocate...
293 p = mi_os_prim_alloc(over_size, 1, commit, false, is_large, is_zero);
294 if (p == NULL) return NULL;
295
296 // and selectively unmap parts around the over-allocated area.
297 void* aligned_p = _mi_align_up_ptr(p, alignment);
298 size_t pre_size = (uint8_t*)aligned_p - (uint8_t*)p;
299 size_t mid_size = _mi_align_up(size, _mi_os_page_size());
300 size_t post_size = over_size - pre_size - mid_size;
301 mi_assert_internal(pre_size < over_size&& post_size < over_size&& mid_size >= size);
302 if (pre_size > 0) { mi_os_prim_free(p, pre_size, (commit ? pre_size : 0)); }
303 if (post_size > 0) { mi_os_prim_free((uint8_t*)aligned_p + mid_size, post_size, (commit ? post_size : 0)); }
304 // we can return the aligned pointer on `mmap` systems
305 p = aligned_p;
306 *base = aligned_p; // since we freed the pre part, `*base == p`.
307 }
308 }
309
310 mi_assert_internal(p == NULL || (p != NULL && *base != NULL && ((uintptr_t)p % alignment) == 0));
311 return p;
312}

319void* _mi_os_alloc(size_t size, mi_memid_t* memid) {
320 *memid = _mi_memid_none();

322 size = _mi_os_good_alloc_size(size);

326 if (p != NULL) {
327 *memid = _mi_memid_create_os(p, size, true, os_is_zero, os_is_large);
328 }

332void* _mi_os_alloc_aligned(size_t size, size_t alignment, bool commit, bool allow_large, mi_memid_t* memid)
333{
334 MI_UNUSED(&_mi_os_get_aligned_hint); // suppress unused warnings
335 *memid = _mi_memid_none();
336 if (size == 0) return NULL;
337 size = _mi_os_good_alloc_size(size);
338 alignment = _mi_align_up(alignment, _mi_os_page_size());
339
340 bool os_is_large = false;
341 bool os_is_zero = false;
342 void* os_base = NULL;
343 void* p = mi_os_prim_alloc_aligned(size, alignment, commit, allow_large, &os_is_large, &os_is_zero, &os_base);
344 if (p != NULL) {
345 *memid = _mi_memid_create_os(p, size, commit, os_is_zero, os_is_large);
346 memid->mem.os.base = os_base;
347 // memid->mem.os.alignment = alignment;
348 memid->mem.os.size += ((uint8_t*)p - (uint8_t*)os_base); // todo: return from prim_alloc_aligned
349 }
350 return p;
351}

365/* -----------------------------------------------------------
366 OS aligned allocation with an offset. This is used
367 for large alignments > MI_BLOCK_ALIGNMENT_MAX. We use a large mimalloc
368 page where the object can be aligned at an offset from the start of the segment.
369 As we may need to overallocate, we need to free such pointers using `mi_free_aligned`
370 to use the actual start of the memory region.
371----------------------------------------------------------- */
372
373void* _mi_os_alloc_aligned_at_offset(size_t size, size_t alignment, size_t offset, bool commit, bool allow_large, mi_memid_t* memid) {
374 mi_assert(offset <= size);
375 mi_assert((alignment % _mi_os_page_size()) == 0);
376 *memid = _mi_memid_none();
377 if (offset == 0) {
378 // regular aligned allocation
379 return _mi_os_alloc_aligned(size, alignment, commit, allow_large, memid);
380 }
381 else {
382 // overallocate to align at an offset
383 const size_t extra = _mi_align_up(offset, alignment) - offset;
384 const size_t oversize = size + extra;
385 void* const start = _mi_os_alloc_aligned(oversize, alignment, commit, allow_large, memid);
386 if (start == NULL) return NULL;
387
388 void* const p = (uint8_t*)start + extra;
389 mi_assert(_mi_is_aligned((uint8_t*)p + offset, alignment));
390 // decommit the overallocation at the start
391 if (commit && extra > _mi_os_page_size()) {
392 _mi_os_decommit(start, extra);
393 }
394 return p;
395 }
396}

402// OS page align within a given area, either conservative (pages inside the area only),
403// or not (straddling pages outside the area is possible)
404static void* mi_os_page_align_areax(bool conservative, void* addr, size_t size, size_t* newsize) {
405 mi_assert(addr != NULL && size > 0);
406 if (newsize != NULL) *newsize = 0;
407 if (size == 0 || addr == NULL) return NULL;
408
409 // page align conservatively within the range, or liberally straddling pages outside the range
410 void* start = (conservative ? _mi_align_up_ptr(addr, _mi_os_page_size())
411 : mi_align_down_ptr(addr, _mi_os_page_size()));
412 void* end = (conservative ? mi_align_down_ptr((uint8_t*)addr + size, _mi_os_page_size())
413 : _mi_align_up_ptr((uint8_t*)addr + size, _mi_os_page_size()));
414 ptrdiff_t diff = (uint8_t*)end - (uint8_t*)start;
415 if (diff <= 0) return NULL;
416
417 mi_assert_internal((conservative && (size_t)diff <= size) || (!conservative && (size_t)diff >= size));
418 if (newsize != NULL) *newsize = (size_t)diff;
419 return start;
420}

422static void* mi_os_page_align_area_conservative(void* addr, size_t size, size_t* newsize) {
423 return mi_os_page_align_areax(true, addr, size, newsize);
424}

426bool _mi_os_commit_ex(void* addr, size_t size, bool* is_zero, size_t stat_size) {
427 if (is_zero != NULL) { *is_zero = false; }
428 mi_os_stat_increase(committed, stat_size); // use size for precise commit vs. decommit
429 mi_os_stat_counter_increase(commit_calls, 1);
430
431 // page align range
432 size_t csize;
433 void* start = mi_os_page_align_areax(false /* conservative? */, addr, size, &csize);
434 if (csize == 0) return true;
435
436 // commit
437 bool os_is_zero = false;
438 int err = _mi_prim_commit(start, csize, &os_is_zero);
439 if (err != 0) {
440 _mi_warning_message("cannot commit OS memory (error: %d (0x%x), address: %p, size: 0x%zx bytes)\n", err, err, start, csize);
441 return false;
442 }
443 if (os_is_zero && is_zero != NULL) {
444 *is_zero = true;
445 mi_assert_expensive(mi_mem_is_zero(start, csize));
446 }
447 // note: the following seems required for asan (otherwise `mimalloc-test-stress` fails)
448 #ifdef MI_TRACK_ASAN
449 if (os_is_zero) { mi_track_mem_defined(start,csize); }
450 else { mi_track_mem_undefined(start,csize); }
451 #endif
452 return true;
453}

459static bool mi_os_decommit_ex(void* addr, size_t size, bool* needs_recommit, size_t stat_size) {
460 mi_assert_internal(needs_recommit!=NULL);
461 mi_os_stat_decrease(committed, stat_size);
462
463 // page align
464 size_t csize;
465 void* start = mi_os_page_align_area_conservative(addr, size, &csize);
466 if (csize == 0) return true;
467
468 // decommit
469 *needs_recommit = true;
470 int err = _mi_prim_decommit(start,csize,needs_recommit);
471 if (err != 0) {
472 _mi_warning_message("cannot decommit OS memory (error: %d (0x%x), address: %p, size: 0x%zx bytes)\n", err, err, start, csize);
473 }
474 mi_assert_internal(err == 0);
475 return (err == 0);
476}

478bool _mi_os_decommit(void* addr, size_t size) {

484// Signal to the OS that the address range is no longer in use
485// but may be used later again. This will release physical memory
486// pages and reduce swapping while keeping the memory committed.
487// We page align to a conservative area inside the range to reset.
488bool _mi_os_reset(void* addr, size_t size) {
489 // page align conservatively within the range
490 size_t csize;
491 void* start = mi_os_page_align_area_conservative(addr, size, &csize);
492 if (csize == 0) return true; // || _mi_os_is_huge_reserved(addr)
493 mi_os_stat_increase(reset, csize);
494 mi_os_stat_counter_increase(reset_calls, 1);
495
496 #if (MI_DEBUG>1) && !MI_SECURE && !MI_TRACK_ENABLED // && !MI_TSAN
497 memset(start, 0, csize); // pretend it is eagerly reset
498 #endif
499
500 int err = _mi_prim_reset(start, csize);
501 if (err != 0) {
502 _mi_warning_message("cannot reset OS memory (error: %d (0x%x), address: %p, size: 0x%zx bytes)\n", err, err, start, csize);
503 }
504 return (err == 0);
505}

508// either resets or decommits memory, returns true if the memory needs
509// to be recommitted if it is to be re-used later on.
510bool _mi_os_purge_ex(void* p, size_t size, bool allow_reset, size_t stat_size)
511{
512 if (mi_option_get(mi_option_purge_delay) < 0) return false; // is purging allowed?
513 mi_os_stat_counter_increase(purge_calls, 1);
514 mi_os_stat_increase(purged, size);
515
516 if (mi_option_is_enabled(mi_option_purge_decommits) && // should decommit?
517 !_mi_preloading()) // don't decommit during preloading (unsafe)
518 {
519 bool needs_recommit = true;
520 mi_os_decommit_ex(p, size, &needs_recommit, stat_size);
521 return needs_recommit;
522 }
523 else {
524 if (allow_reset) { // this can sometimes be not allowed if the range is not fully committed (on Windows, we cannot reset uncommitted memory)
525 _mi_os_reset(p, size);
526 }
527 return false; // needs no recommit
528 }
529}

538// Protect a region in memory to be not accessible.
539static bool mi_os_protectx(void* addr, size_t size, bool protect) {
540 // page align conservatively within the range
541 size_t csize = 0;
542 void* start = mi_os_page_align_area_conservative(addr, size, &csize);
543 if (csize == 0) return false;
544 /*
545 if (_mi_os_is_huge_reserved(addr)) {
546 _mi_warning_message("cannot mprotect memory allocated in huge OS pages\n");
547 }
548 */
549 int err = _mi_prim_protect(start,csize,protect);

566/* ----------------------------------------------------------------------------
567Support for allocating huge OS pages (1Gib) that are reserved up-front
568and possibly associated with a specific NUMA node. (use `numa_node>=0`)
569-----------------------------------------------------------------------------*/
570#define MI_HUGE_OS_PAGE_SIZE (MI_GiB)
571
572
573#if (MI_INTPTR_SIZE >= 8)
574// To ensure proper alignment, use our own area for huge OS pages
575static mi_decl_cache_align _Atomic(uintptr_t) mi_huge_start; // = 0
576
577// Claim an aligned address range for huge pages
578static uint8_t* mi_os_claim_huge_pages(size_t pages, size_t* total_size) {
579 if (total_size != NULL) *total_size = 0;
580 const size_t size = pages * MI_HUGE_OS_PAGE_SIZE;
581
582 uintptr_t start = 0;
583 uintptr_t end = 0;
584 uintptr_t huge_start = mi_atomic_load_relaxed(&mi_huge_start);
585 do {
586 start = huge_start;
587 if (start == 0) {
588 // Initialize the start address after the 32TiB area
589 start = ((uintptr_t)8 << 40); // 8TiB virtual start address
590 #if (MI_SECURE>0 || MI_DEBUG==0) // security: randomize start of huge pages unless in debug mode
591 uintptr_t r = _mi_heap_random_next(mi_prim_get_default_heap());
592 start = start + ((uintptr_t)MI_HUGE_OS_PAGE_SIZE * ((r>>17) & 0x0FFF)); // (randomly 12bits)*1GiB == between 0 to 4TiB
593 #endif
594 }
595 end = start + size;
596 } while (!mi_atomic_cas_weak_acq_rel(&mi_huge_start, &huge_start, end));
597
598 if (total_size != NULL) *total_size = size;
599 return (uint8_t*)start;
600}
601#else
602static uint8_t* mi_os_claim_huge_pages(size_t pages, size_t* total_size) {
603 MI_UNUSED(pages);
604 if (total_size != NULL) *total_size = 0;
605 return NULL;
606}
607#endif
608
609// Allocate MI_ARENA_SLICE_ALIGN aligned huge pages
610void* _mi_os_alloc_huge_os_pages(size_t pages, int numa_node, mi_msecs_t max_msecs, size_t* pages_reserved, size_t* psize, mi_memid_t* memid) {
611 *memid = _mi_memid_none();
612 if (psize != NULL) *psize = 0;
613 if (pages_reserved != NULL) *pages_reserved = 0;
614 size_t size = 0;
615 uint8_t* start = mi_os_claim_huge_pages(pages, &size);
616 if (start == NULL) return NULL; // or 32-bit systems
617
618 // Allocate one page at the time but try to place them contiguously
619 // We allocate one page at the time to be able to abort if it takes too long
620 // or to at least allocate as many as available on the system.
621 mi_msecs_t start_t = _mi_clock_start();
622 size_t page = 0;
623 bool all_zero = true;
624 while (page < pages) {
625 // allocate a page
626 bool is_zero = false;
627 void* addr = start + (page * MI_HUGE_OS_PAGE_SIZE);
628 void* p = NULL;
629 int err = _mi_prim_alloc_huge_os_pages(addr, MI_HUGE_OS_PAGE_SIZE, numa_node, &is_zero, &p);
630 if (!is_zero) { all_zero = false; }
631 if (err != 0) {
632 _mi_warning_message("unable to allocate huge OS page (error: %d (0x%x), address: %p, size: %zx bytes)\n", err, err, addr, MI_HUGE_OS_PAGE_SIZE);
633 break;
634 }
635
636 // Did we succeed at a contiguous address?
637 if (p != addr) {
638 // no success, issue a warning and break
639 if (p != NULL) {
640 _mi_warning_message("could not allocate contiguous huge OS page %zu at %p\n", page, addr);
641 mi_os_prim_free(p, MI_HUGE_OS_PAGE_SIZE, MI_HUGE_OS_PAGE_SIZE);
642 }
643 break;
644 }
645
646 // success, record it
647 page++; // increase before timeout check (see issue #711)
648 mi_os_stat_increase(committed, MI_HUGE_OS_PAGE_SIZE);
649 mi_os_stat_increase(reserved, MI_HUGE_OS_PAGE_SIZE);
650
651 // check for timeout
652 if (max_msecs > 0) {
653 mi_msecs_t elapsed = _mi_clock_end(start_t);
654 if (page >= 1) {
655 mi_msecs_t estimate = ((elapsed / (page+1)) * pages);
656 if (estimate > 2*max_msecs) { // seems like we are going to timeout, break
657 elapsed = max_msecs + 1;
658 }
659 }
660 if (elapsed > max_msecs) {
661 _mi_warning_message("huge OS page allocation timed out (after allocating %zu page(s))\n", page);
662 break;
663 }
664 }
665 }
666 mi_assert_internal(page*MI_HUGE_OS_PAGE_SIZE <= size);
667 if (pages_reserved != NULL) { *pages_reserved = page; }
668 if (psize != NULL) { *psize = page * MI_HUGE_OS_PAGE_SIZE; }
669 if (page != 0) {
670 mi_assert(start != NULL);
671 *memid = _mi_memid_create_os(start, *psize, true /* is committed */, all_zero, true /* is_large */);
672 memid->memkind = MI_MEM_OS_HUGE;
673 mi_assert(memid->is_pinned);
674 #ifdef MI_TRACK_ASAN
675 if (all_zero) { mi_track_mem_defined(start,size); }
676 #endif
677 }
678 return (page == 0 ? NULL : start);
679}
680
681// free every huge page in a range individually (as we allocated per page)
682// note: needed with VirtualAlloc but could potentially be done in one go on mmap'd systems.
683static void mi_os_free_huge_os_pages(void* p, size_t size) {
684 if (p==NULL || size==0) return;
685 uint8_t* base = (uint8_t*)p;
686 while (size >= MI_HUGE_OS_PAGE_SIZE) {
687 mi_os_prim_free(base, MI_HUGE_OS_PAGE_SIZE, MI_HUGE_OS_PAGE_SIZE);
688 size -= MI_HUGE_OS_PAGE_SIZE;
689 base += MI_HUGE_OS_PAGE_SIZE;
690 }
691}
