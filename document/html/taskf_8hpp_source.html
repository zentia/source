112 // NOTE: by default the scheduling is round-robin (pseudo round-robin indeed).
113 // In order to select the ondemand scheduling policy, set the ondemand_buffer to a
114 // value grather than 1.
115 ff_taskf(int maxnw=ff_realNumCores(),
116 const size_t maxTasks=DEFAULT_OUTSTANDING_TASKS,
117 const int ondemand_buffer=0):
118 ff_farm(true,
119 (std::max)(maxTasks, (size_t)(MAX_NUM_THREADS*8)),
120 (std::max)(maxTasks, (size_t)(MAX_NUM_THREADS*8)),
121 true, maxnw, true),
122 farmworkers(maxnw),ntasks(0),
123 outstandingTasks((std::max)(maxTasks, (size_t)(MAX_NUM_THREADS*8))),taskscounter(0) {
124
125 TASKS.resize(outstandingTasks);
126 std::vector<ff_node *> w;
127 // NOTE: Worker objects are going to be destroyed by the farm destructor
128 for(int i=0;i<maxnw;++i) w.push_back(new Worker);
129 ff_farm::add_workers(w);
130 ff_farm::add_emitter(sched = new Scheduler(ff_farm::getlb(), maxnw));
131 ff_farm::wrap_around();
132 ff_farm::set_scheduling_ondemand(ondemand_buffer);
133
134 // needed to avoid the initial barrier
135 if (ff_farm::prepare() < 0)
136 error("ff_taskf: running farm (1)\n");
137
138 auto r=-1;
139 getlb()->freeze();
140 ff_farm::offload(GO_OUT);
141 if (getlb()->run() != -1) {
142 getlb()->broadcast_task(GO_OUT);
143 r = getlb()->wait_freezing();
144 }
145 if (r<0) error("ff_taskf: running farm (2)\n");
146 }
147 virtual ~ff_taskf() {
148 if (sched) { delete sched; sched=nullptr;}
149 }

152 inline task_f_t* AddTask(const F_t F, Param... args) {
153 // FIX: use ff_allocator here !
154 ff_task_f_t<F_t, Param...> *wtask = new ff_task_f_t<F_t, Param...>(F, args...);
155 std::vector<param_info> useless;
156 task_f_t *task = alloc_task(useless,wtask);
157 while(!ff_farm::offload(task, 1)) ff_relax(1);
158 ++taskscounter;
159 return task;
160 }

162 virtual inline int run_and_wait_end() {
163 while(!ff_farm::offload(EOS, 1)) ff_relax(1);
164 sched->thaw(true,farmworkers);
165 sched->wait_freezing();
166 return sched->wait();
167 }
168 virtual int run_then_freeze(ssize_t nw=-1) {
169 while(!ff_farm::offload(EOS, 1)) ff_relax(1);
170 sched->thaw(true,(nw>0) ? nw:taskscounter);
171 int r=sched->wait_freezing();
172 taskscounter=0;
173 return r;
174 }
175
176 // it starts all workers
177 virtual inline int run(bool=false) {
178 sched->thaw(true,farmworkers);
179 return 0;
180 }
181 virtual inline int wait() {
182 while(!ff_farm::offload(EOS, 1)) ff_relax(1);
183 int r=sched->wait_freezing();
184 taskscounter=0;
185 return r;
186 }
