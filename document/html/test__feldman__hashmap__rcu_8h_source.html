22 template <class Map>
23 void test( Map& m )
24 {
25 // Precondition: map is empty
26 // Postcondition: map is empty
27
28 base_class::test( m );
29
30 ASSERT_TRUE( m.empty());
31 ASSERT_CONTAINER_SIZE( m, 0 );
32
33 typedef typename Map::key_type key_type;
34 typedef typename Map::mapped_type value_type;
35 typedef typename Map::value_type map_pair;
36 typedef typename Map::rcu_lock rcu_lock;
37 typedef typename Map::exempt_ptr exempt_ptr;
38
39 size_t const kkSize = base_class::kSize;
40
41 std::vector<key_type> arrKeys;
42 for ( int i = 0; i < static_cast<int>(kkSize); ++i )
43 arrKeys.push_back( key_type( i ));
44 shuffle( arrKeys.begin(), arrKeys.end());
45
46 std::vector< value_type > arrVals;
47 for ( size_t i = 0; i < kkSize; ++i ) {
48 value_type val;
49 val.nVal = static_cast<int>( i );
50 val.strVal = std::to_string( i );
51 arrVals.push_back( val );
52 }
53
54 for ( auto const& i : arrKeys )
55 ASSERT_TRUE( m.insert( i ));
56 ASSERT_FALSE( m.empty());
57 ASSERT_CONTAINER_SIZE( m, kkSize );
58
59 // iterators
60 {
61 rcu_lock l;
62 size_t nCount = 0;
63 for ( auto it = m.begin(); it != m.end(); ++it ) {
64 EXPECT_EQ( it->second.nVal, 0 );
65 it->second.nVal = it->first.nKey * 2;
66 ++nCount;
67 }
68 EXPECT_EQ( nCount, kkSize );
69
70 nCount = 0;
71 for ( auto it = m.cbegin(); it != m.cend(); ++it ) {
72 EXPECT_EQ( it->second.nVal, it->first.nKey * 2 );
73 ++nCount;
74 }
75 EXPECT_EQ( nCount, kkSize );
76
77 nCount = 0;
78 for ( auto it = m.rbegin(); it != m.rend(); ++it ) {
79 EXPECT_EQ( it->second.nVal, it->first.nKey * 2 );
80 it->second.nVal = it->first.nKey * 4;
81 ++nCount;
82 }
83 EXPECT_EQ( nCount, kkSize );
84
85 nCount = 0;
86 for ( auto it = m.crbegin(); it != m.crend(); ++it ) {
87 EXPECT_EQ( it->second.nVal, it->first.nKey * 4 );
88 ++nCount;
89 }
90 EXPECT_EQ( nCount, kkSize );
91 }
92
93 // get/extract
94 exempt_ptr xp;
95
96 for ( auto const& i : arrKeys ) {
97 value_type const& val = arrVals.at( i.nKey );
98
99 {
100 rcu_lock l;
101 map_pair * p = m.get( i.nKey );
102 ASSERT_FALSE( p == nullptr );
103 EXPECT_EQ( p->first.nKey, i.nKey );
104
105 p = m.get( i );
106 ASSERT_FALSE( p == nullptr );
107 EXPECT_EQ( p->first.nKey, i.nKey );
108
109 p = m.get( val.strVal );
110 ASSERT_FALSE( p == nullptr );
111 EXPECT_EQ( p->first.nKey, i.nKey );
112 }
113
114 switch ( i.nKey % 3 ) {
115 case 0:
116 xp = m.extract( i.nKey );
117 break;
118 case 1:
119 xp = m.extract( i );
120 break;
121 case 2:
122 xp = m.extract( val.strVal );
123 break;
124 }
125 ASSERT_FALSE( !xp );
126 ASSERT_EQ( xp->first.nKey, i.nKey );
127
128 {
129 rcu_lock l;
130
131 map_pair * p = m.get( i.nKey );
132 EXPECT_TRUE( p == nullptr );
133 p = m.get( i );
134 EXPECT_TRUE( p == nullptr );
135 }
136 }
137 ASSERT_TRUE( m.empty());
138 ASSERT_CONTAINER_SIZE( m, 0 );
139 }
