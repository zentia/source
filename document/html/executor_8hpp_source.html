1229 //_latch.arrive_and_wait();

1233 std::hash<std::thread::id>()(std::this_thread::get_id()))
1234 );

1320inline bool Executor::_explore_task(Worker& w, Node*& t) {
1321
1322 //assert(!t);
1323
1324 const size_t MAX_STEALS = ((num_queues() + 1) << 1);
1325
1326 size_t num_steals = 0;
1327 size_t num_empty_steals = 0;
1328
1329 // Make the worker steal immediately from the assigned victim.
1330 while(true) {
1331 // If the worker's victim thread (w._vtm) is within the worker pool, steal from the worker's queue.
1332 // Otherwise, steal from the buffer, adjusting the victim index based on the worker pool size.
1333 t = (w._vtm < _workers.size())
1334 ? _workers[w._vtm]._wsq.steal_with_hint(num_empty_steals)
1335 : _buffers.steal_with_hint(w._vtm - _workers.size(), num_empty_steals);
1336
1337 if(t) {
1338 break;
1339 }
1340
1341 // Increment the steal count, and if it exceeds MAX_STEALS, yield the thread.
1342 // If the number of *consecutive* empty steals reaches MAX_STEALS, exit the loop.
1343 if (++num_steals > MAX_STEALS) {
1344 std::this_thread::yield();
1345 if(num_empty_steals == MAX_STEALS) {
1346 break;
1347 }
1348 }
1349
1350 #if __cplusplus >= TF_CPP20
1351 if(w._done.test(std::memory_order_relaxed)) {
1352 #else
1353 if(w._done.load(std::memory_order_relaxed)) {
1354 #endif
1355 return false;
1356 }
1357
1358 // Randomely generate a next victim.
1359 w._vtm = w._rdvtm();
1360 }
1361
1362 return true;
1363}

1373// Function: _wait_for_task
1374inline bool Executor::_wait_for_task(Worker& w, Node*& t) {
1375
1376 explore_task:
1377
1378 if(_explore_task(w, t) == false) {
1379 return false;
1380 }
1381
1382 // Go exploit the task if we successfully steal one.
1383 if(t) {
1384 return true;
1385 }
1386
1387 // Entering the 2PC guard as all queues should be empty after many stealing attempts.
1388 _notifier.prepare_wait(w._waiter);
1389
1390 // Condition #1: buffers should be empty
1391 if(!_buffers.empty(w._vtm)) {
1392 w._vtm += _workers.size();
1393 _notifier.cancel_wait(w._waiter);
1394 goto explore_task;
1395 }
1396
1397 // Condition #2: worker queues should be empty
1398 // Note: We need to use index-based looping to avoid data race with _spawan
1399 // which initializes other worker data structure at the same time
1400 for(size_t vtm=0; vtm<w._id; ++vtm) {
1401 if(!_workers[vtm]._wsq.empty()) {
1402 w._vtm = vtm;
1403 _notifier.cancel_wait(w._waiter);
1404 goto explore_task;
1405 }
1406 }
1407
1408 for(size_t vtm=w._id+1; vtm<_workers.size(); vtm++) {
1409 if(!_workers[vtm]._wsq.empty()) {
1410 w._vtm = vtm;
1411 _notifier.cancel_wait(w._waiter);
1412 goto explore_task;
1413 }
1414 }
1415
1416 // Condition #3: worker should be alive
1417#if __cplusplus >= TF_CPP20
1418 if(w._done.test(std::memory_order_relaxed)) {
1419#else
1420 if(w._done.load(std::memory_order_relaxed)) {
1421#endif
1422 _notifier.cancel_wait(w._waiter);
1423 return false;
1424 }
1425
1426 // Now I really need to relinquish my self to others.
1427 _notifier.commit_wait(w._waiter);
1428 w._vtm = w._id;
1429 goto explore_task;
1430
1431}

1493template <typename I>
1494void Executor::_schedule(Worker& worker, I first, I last) {
1495
1496 size_t num_nodes = last - first;
1497
1498 if(num_nodes == 0) {
1499 return;
1500 }
1501
1502 // NOTE: We cannot use first/last as the for-loop condition
1503 // (e.g., for(; first != last; ++first)) since when a node is inserted
1504 // into the queue the node can run and finish immediately.
1505 // If this is the last node in the graph, it will tear down the parent
1506 // container which cause the last ++first to fail.
1507 // This problem is specific to MSVC which has strict iterator arithmetics.
1508 if(worker._executor == this) {
1509 for(size_t i=0; i<num_nodes; i++) {
1510 auto node = detail::get_node_ptr(first[i]);
1511 worker._wsq.push(node, [&](){ _buffers.push(node); });
1512 _notifier.notify_one();
1513 }
1514 return;
1515 }
1516
1517 for(size_t i=0; i<num_nodes; i++) {
1518 _buffers.push(detail::get_node_ptr(first[i]));
1519 }
1520 _notifier.notify_n(num_nodes);
1521}

1556inline void Executor::_invoke(Worker& worker, Node* node) {
1557
1558 #define TF_INVOKE_CONTINUATION() \
1559 if (cache) { \
1560 node = cache; \
1561 goto begin_invoke; \
1562 }
1563
1564 begin_invoke:
1565
1566 Node* cache {nullptr};
1567
1568 // if this is the second invoke due to preemption, directly jump to invoke task
1569 if(node->_nstate & NSTATE::PREEMPTED) {
1570 goto invoke_task;
1571 }
1572
1573 // if the work has been cancelled, there is no need to continue
1574 if(node->_is_cancelled()) {
1575 _tear_down_invoke(worker, node, cache);
1576 TF_INVOKE_CONTINUATION();
1577 return;
1578 }
1579
1580 // if acquiring semaphore(s) exists, acquire them first
1581 if(node->_semaphores && !node->_semaphores->to_acquire.empty()) {
1582 SmallVector<Node*> waiters;
1583 if(!node->_acquire_all(waiters)) {
1584 _schedule(worker, waiters.begin(), waiters.end());
1585 return;
1586 }
1587 }
1588
1589 invoke_task:
1590
1591 SmallVector<int> conds;
1592
1593 // switch is faster than nested if-else due to jump table
1594 switch(node->_handle.index()) {
1595 // static task
1596 case Node::STATIC:{
1597 _invoke_static_task(worker, node);
1598 }
1599 break;
1600
1601 // runtime task
1602 case Node::RUNTIME:{
1603 if(_invoke_runtime_task(worker, node)) {
1604 return;
1605 }
1606 }
1607 break;
1608
1609 // subflow task
1610 case Node::SUBFLOW: {
1611 if(_invoke_subflow_task(worker, node)) {
1612 return;
1613 }
1614 }
1615 break;
1616
1617 // condition task
1618 case Node::CONDITION: {
1619 _invoke_condition_task(worker, node, conds);
1620 }
1621 break;
1622
1623 // multi-condition task
1624 case Node::MULTI_CONDITION: {
1625 _invoke_multi_condition_task(worker, node, conds);
1626 }
1627 break;
1628
1629 // module task
1630 case Node::MODULE: {
1631 if(_invoke_module_task(worker, node)) {
1632 return;
1633 }
1634 }
1635 break;
1636
1637 // async task
1638 case Node::ASYNC: {
1639 if(_invoke_async_task(worker, node)) {
1640 return;
1641 }
1642 _tear_down_async(worker, node, cache);
1643 TF_INVOKE_CONTINUATION();
1644 return;
1645 }
1646 break;
1647
1648 // dependent async task
1649 case Node::DEPENDENT_ASYNC: {
1650 if(_invoke_dependent_async_task(worker, node)) {
1651 return;
1652 }
1653 _tear_down_dependent_async(worker, node, cache);
1654 TF_INVOKE_CONTINUATION();
1655 return;
1656 }
1657 break;
1658
1659 // monostate (placeholder)
1660 default:
1661 break;
1662 }
1663
1664 // if releasing semaphores exist, release them
1665 if(node->_semaphores && !node->_semaphores->to_release.empty()) {
1666 SmallVector<Node*> waiters;
1667 node->_release_all(waiters);
1668 _schedule(worker, waiters.begin(), waiters.end());
1669 }
1670
1671 // Reset the join counter with strong dependencies to support cycles.
1672 // + We must do this before scheduling the successors to avoid race
1673 // condition on _predecessors.
1674 // + We must use fetch_add instead of direct assigning
1675 // because the user-space call on "invoke" may explicitly schedule
1676 // this task again (e.g., pipeline) which can access the join_counter.
1677 node->_join_counter.fetch_add(
1678 node->num_predecessors() - (node->_nstate & ~NSTATE::MASK), std::memory_order_relaxed
1679 );
1680
1681 // acquire the parent flow counter
1682 auto& join_counter = (node->_parent) ? node->_parent->_join_counter :
1683 node->_topology->_join_counter;
1684
1685 // Invoke the task based on the corresponding type
1686 switch(node->_handle.index()) {
1687
1688 // condition and multi-condition tasks
1689 case Node::CONDITION:
1690 case Node::MULTI_CONDITION: {
1691 for(auto cond : conds) {
1692 if(cond >= 0 && static_cast<size_t>(cond) < node->_num_successors) {
1693 auto s = node->_edges[cond];
1694 // zeroing the join counter for invariant
1695 s->_join_counter.store(0, std::memory_order_relaxed);
1696 join_counter.fetch_add(1, std::memory_order_relaxed);
1697 _update_cache(worker, cache, s);
1698 }
1699 }
1700 }
1701 break;
1702
1703 // non-condition task
1704 default: {
1705 for(size_t i=0; i<node->_num_successors; ++i) {
1706 //if(auto s = node->_successors[i]; --(s->_join_counter) == 0) {
1707 if(auto s = node->_edges[i]; s->_join_counter.fetch_sub(1, std::memory_order_acq_rel) == 1) {
1708 join_counter.fetch_add(1, std::memory_order_relaxed);
1709 _update_cache(worker, cache, s);
1710 }
1711 }
1712 }
1713 break;
1714 }
1715
1716 // clean up the node after execution
1717 _tear_down_invoke(worker, node, cache);
1718 TF_INVOKE_CONTINUATION();
1719}

1757// Procedure: _process_exception
1758inline void Executor::_process_exception(Worker&, Node* node) {
1759
1760 constexpr static auto flag = ESTATE::EXCEPTION | ESTATE::CANCELLED;
1761
1762 // find the anchor and mark the entire path with exception so recursive
1763 // or nested tasks can be cancelled properly
1764 // since exception can come from asynchronous task (with runtime), the node
1765 // itself can be anchored
1766 auto anchor = node;
1767 while(anchor && (anchor->_estate.load(std::memory_order_relaxed) & ESTATE::ANCHORED) == 0) {
1768 anchor->_estate.fetch_or(flag, std::memory_order_relaxed);
1769 anchor = anchor->_parent;
1770 }
1771
1772 // the exception occurs under a blocking call (e.g., corun, join)
1773 if(anchor) {
1774 // multiple tasks may throw, and we only take the first thrown exception
1775 if((anchor->_estate.fetch_or(flag, std::memory_order_relaxed) & ESTATE::EXCEPTION) == 0) {
1776 anchor->_exception_ptr = std::current_exception();
1777 return;
1778 }
1779 }
1780 // otherwise, we simply store the exception in the topology and cancel it
1781 else if(auto tpg = node->_topology; tpg) {
1782 // multiple tasks may throw, and we only take the first thrown exception
1783 if((tpg->_estate.fetch_or(flag, std::memory_order_relaxed) & ESTATE::EXCEPTION) == 0) {
1784 tpg->_exception_ptr = std::current_exception();
1785 return;
1786 }
1787 }
1788
1789 // for now, we simply store the exception in this node; this can happen in an
1790 // execution that does not have any external control to capture the exception,
1791 // such as silent async task
1792 node->_exception_ptr = std::current_exception();
1793}
