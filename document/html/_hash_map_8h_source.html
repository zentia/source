111 void put(const K &key, const V &value) {
112 Entry *entry = find(key);
113 if (entry) {
114 entry->_key = key;
115 entry->_value = value;
116 } else {
117 entry = new(__FILE__, __LINE__) Entry();
118 entry->_key = key;

150 Entry *prev = entry->prev;
151 Entry *next = entry->next;
152

154 else _head = next;
155 if (next) next->prev = entry->prev;

157 delete entry;
