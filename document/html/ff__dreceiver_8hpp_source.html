108 virtual int handleBatch(int sck){
109 int requestSize;
110 switch(readn(sck, reinterpret_cast<char*>(&requestSize), sizeof(requestSize))) {
111 case -1: {
112 perror("readn");
113 error("Something went wrong in receiving the number of tasks!\n");
114 return -1;
115 } break;
116 case 0: return -1;
117 }
118 // always sending back the acknowledgement
119 if (writen(sck, reinterpret_cast<char*>(&ACK), sizeof(ack_t)) < 0){
120 if (errno != ECONNRESET && errno != EPIPE) {
121 error("Error sending back ACK to the sender (errno=%d)\n",errno);
122 return -1;
123 }
124 }
125 ChannelType t = sck2ChannelType[sck];
126 requestSize = ntohl(requestSize);
127 for(int i = 0; i < requestSize; i++)
128 if (handleRequest(sck, t)<0) return -1;
129
130 return 0;

133 virtual int handleRequest(int sck, ChannelType t){
134 int sender;
135 int chid;
136 size_t sz;
137 struct iovec iov[3];
138 iov[0].iov_base = &sender;
139 iov[0].iov_len = sizeof(sender);
140 iov[1].iov_base = &chid;
141 iov[1].iov_len = sizeof(chid);
142 iov[2].iov_base = &sz;
143 iov[2].iov_len = sizeof(sz);
144
145 switch (readvn(sck, iov, 3)) {
146 case -1: error("Error reading from socket errno=%d\n",errno); // fatal error
147 case 0: return -1; // connection close
148 }
149
150 // convert values to host byte order
151 sender = ntohl(sender);
152 chid = ntohl(chid);
153 sz = be64toh(sz);
154
155 if (sz > 0){
156 char* buff = new char [sz];
157 assert(buff);
158 if(readn(sck, buff, sz) < 0){
159 error("Error reading from socket in handleRequest\n");
160 delete [] buff;
161 return -1;
162 }
163 message_t* out = new message_t(buff, sz, true);
164 assert(out);
165 out->feedback = t == ChannelType::FBK;
166 out->sender = sender;
167 out->chid = chid;
168 this->forward(out, sck);
169
170 return 0;
171 }
172 //logical EOS
173 if (chid == -2){
174 registerLogicalEOS(sender);
175 return 0;
176 }
177
178 //pyshical EOS
179 registerEOS(sck);
180 return -1;

244 message_t *svc(message_t* task) {
245 fd_set set, tmpset;
246 // intialize both sets (master, temp)
247 FD_ZERO(&set);
248 FD_ZERO(&tmpset);
249
250 // add the listen socket to the master set
251 FD_SET(this->listen_sck, &set);
252
253 // hold the greater descriptor
254 int fdmax = this->listen_sck;
255
256 while(neos < input_channels){
257 // copy the master set to the temporary
258 tmpset = set;
259
260 switch(select(fdmax+1, &tmpset, NULL, NULL, NULL)){
261 case -1: error("Error on selecting socket\n"); return EOS;
262 case 0: continue;
263 }
264
265 for(int idx=0; idx <= fdmax; idx++){
266 if (FD_ISSET(idx, &tmpset)){
267 if (idx == this->listen_sck) {
268 int connfd = accept(this->listen_sck, (struct sockaddr*)NULL ,NULL);
269 if (connfd == -1){
270 error("Error accepting client\n");
271 } else {
272 FD_SET(connfd, &set);
273 if(connfd > fdmax) fdmax = connfd;
274
275 this->handshakeHandler(connfd);
276 }
277 continue;
278 }
279
280 if (this->handleBatch(idx) < 0){
281 close(idx);
282 FD_CLR(idx, &set);
283
284 // update the maximum file descriptor
285 if (idx == fdmax)
286 for(int ii=(fdmax-1);ii>=0;--ii)
287 if (FD_ISSET(ii, &set)){
288 fdmax = ii;
289 break;
290 }
291
292 }
293
294 }
295 }
296 }
297
298 return this->EOS;
299 }
