96 * In the following we implement two kinds of queues:
97 * - the MPMC_Ptr_Queue is an implementation of the ** bounded **
98 * Multi-Producer/Multi-Consumer queue algorithm by Dmitry Vyukov
99 * (www.1024cores.net). It stores pointers.

101 * - the uMPMC_Ptr_Queue implements an ** unbounded **
102 * Multi-Producer/Multi-Consumer queue which does not require

104 * The implementation blends together the MPMC_Ptr_Queue and the

148 /* | data | seq | | data | seq | | data | seq |
149 * | NULL | 0 | ------ | NULL | 1 | ------ | NULL | ... |
150 * |||||||||||||||| |||||||||||||||| ||||||||||||||||
151 * |
152 * |
153 * |
154 * pwrite pread

160 inline bool init(size_t size) {
161 if (size<2) size=2;
162 // we need a size that is a power 2 in order to set the mask
163 if (!isPowerOf2(size)) size = nextPowerOf2(size);
164 mask = size-1;
165
166 buf = new element_t[size];
167 if (!buf) return false;
168 for(size_t i=0;i<size;++i) {
169 buf[i].data = NULL;
170 buf[i].seq.store(i,std::memory_order_relaxed);
171
172 // store method
173 // Atomically stores the value 'i'.
174 //
175 // Memory is affected according to the value of memory_order:
176 // memory_order must be one of
177 // std::memory_order_relaxed
178 // std::memory_order_release
179 // std::memory_order_seq_cst.
180 // Otherwise the behavior is undefined.
181 }
182 pwrite.store(0,std::memory_order_relaxed);
183 pread.store(0,std::memory_order_relaxed);
184 return true;
185 }
186
192 inline bool push(void *const data) {
193 unsigned long pw, seq;
194 element_t * node;
195 unsigned long bk = BACKOFF_MIN;
196 do {
197 pw = pwrite.load(std::memory_order_relaxed);
198 node = &buf[pw & mask];
199 seq = node->seq.load(std::memory_order_acquire);
200
201 // load method
202 // Atomically loads and returns the current value of the atomic variable.
203 // Memory is affected according to the value of memory_order.
204
205 if (pw == seq) { // CAS
206 if (pwrite.compare_exchange_weak(pw, pw+1, std::memory_order_relaxed))
207 break;
208
209 // exponential delay with max value
210 for(volatile unsigned i=0;i<bk;++i) ;
211 bk <<= 1;
212 bk &= BACKOFF_MAX;
213 } else
214 if (pw > seq) return false; // queue full
215 } while(1);
216 node->data = data;
217 node->seq.store(seq+1,std::memory_order_release);
218 return true;
219 }
220
227 inline bool pop(void** data) {
228 unsigned long pr, seq;
229 element_t * node;
230 unsigned long bk = BACKOFF_MIN;
231
232 do {
233 pr = pread.load(std::memory_order_relaxed);
234 node = &buf[pr & mask];
235 seq = node->seq.load(std::memory_order_acquire);
236
237 long diff = seq - (pr+1);
238 if (diff == 0) { // CAS
239 if (pread.compare_exchange_weak(pr, (pr+1), std::memory_order_relaxed))
240 break;
241
242 // exponential delay with max value
243 for(volatile unsigned i=0;i<bk;++i) ;
244 bk <<= 1;
245 bk &= BACKOFF_MAX;
246 } else {
247 if (diff < 0) return false; // queue empty
248 }
249 } while(1);
250 *data = node->data;
251 node->seq.store((pr+mask+1), std::memory_order_release);
252 return true;
253 }
254
255private:
256 union {
257 std::atomic<unsigned long> pwrite;
258 char padding1[CACHE_LINE_SIZE];
259 };
260 union {
261 std::atomic<unsigned long> pread;
262 char padding2[CACHE_LINE_SIZE];
263 };
264 element_t * buf;
265 unsigned long mask;
266};

333 inline bool push(void *const data) {
334 unsigned long pw, seq;
335 element_t * node;
336 unsigned long bk = BACKOFF_MIN;
337
338 do {
339 pw = atomic_long_read(&pwrite);
340 node = &buf[pw & mask];
341 seq = atomic_long_read(&node->seq);
342
343 if (pw == seq) {
344 if (abstraction_cas((volatile atom_t*)&pwrite, (atom_t)(pw+1), (atom_t)pw)==(atom_t)pw)
345 break;
346
347 // exponential delay with max value
348 for(volatile unsigned i=0;i<bk;++i) ;
349 bk <<= 1;
350 bk &= BACKOFF_MAX;
351 } else
352 if (pw > seq) return false;
353
354 } while(1);
355 node->data = data;
356 //atomic_long_inc(&node->seq);
357 atomic_long_set(&node->seq, (seq+1));
358 return true;
359 }

367 inline bool pop(void** data) {
368 unsigned long pr , seq;
369 element_t * node;
370 unsigned long bk = BACKOFF_MIN;
371
372 do {
373 pr = atomic_long_read(&pread);
374 node = &buf[pr & mask];
375 seq = atomic_long_read(&node->seq);
376 long diff = seq - (pr+1);
377 if (diff == 0) {
378 if (abstraction_cas((volatile atom_t*)&pread, (atom_t)(pr+1), (atom_t)pr)==(atom_t)pr)
379 break;
380
381 // exponential delay with max value
382 for(volatile unsigned i=0;i<bk;++i) ;
383 bk <<= 1;
384 bk &= BACKOFF_MAX;
385 } else {
386 if (diff < 0) return false;
387 }
388
389 } while(1);
390 *data = node->data;
391 atomic_long_set(&node->seq,(pr+mask+1));
392 return true;
393 }

427class uMPMC_Ptr_Queue {
428protected:
429 enum {DEFAULT_NUM_QUEUES=4, DEFAULT_uSPSC_SIZE=2048};

431 typedef void * data_element_t;
432 typedef atomic_long_t sequenceP_t;
433 typedef atomic_long_t sequenceC_t;
434
435public:
439 uMPMC_Ptr_Queue() {}
440
444 ~uMPMC_Ptr_Queue() {
445 if (buf) {
446 for(size_t i=0;i<(mask+1);++i) {
447 if (buf[i]) delete (uSWSR_Ptr_Buffer*)(buf[i]);
448 }
449 freeAlignedMemory(buf);

459 inline bool init(unsigned long nqueues=DEFAULT_NUM_QUEUES, size_t size=DEFAULT_uSPSC_SIZE) {
460 if (nqueues<2) nqueues=2;
461 if (!isPowerOf2(nqueues)) nqueues = nextPowerOf2(nqueues);
462 mask = nqueues-1;
463
464 buf=(data_element_t*)getAlignedMemory(longxCacheLine*sizeof(long),nqueues*sizeof(data_element_t));
465 seqP=(sequenceP_t*)getAlignedMemory(longxCacheLine*sizeof(long),nqueues*sizeof(sequenceP_t));
466 seqC=(sequenceP_t*)getAlignedMemory(longxCacheLine*sizeof(long),nqueues*sizeof(sequenceC_t));
467
468 for(size_t i=0;i<nqueues;++i) {
469 buf[i]= new uSWSR_Ptr_Buffer(size);
470 ((uSWSR_Ptr_Buffer*)(buf[i]))->init();
471 atomic_long_set(&(seqP[i]),long(i));
472 atomic_long_set(&(seqC[i]),long(i));
473 }
474 atomic_long_set(&preadP,0);
475 atomic_long_set(&preadC,0);
476 return true;
477 }
478
484 inline bool push(void *const data) {
485 unsigned long pw,seq,idx;
486 unsigned long bk = BACKOFF_MIN;
487 do {
488 pw = atomic_long_read(&preadP);
489 idx = pw & mask;
490 seq = atomic_long_read(&seqP[idx]);
491 if (pw == seq) {
492 if (abstraction_cas((volatile atom_t*)&preadP, (atom_t)(pw+1), (atom_t)pw)==(atom_t)pw)
493 break;
494
495 // exponential delay with max value
496 for(volatile unsigned i=0;i<bk;++i) ;
497 bk <<= 1;
498 bk &= BACKOFF_MAX;
499 }
500 } while(1);
501 ((uSWSR_Ptr_Buffer*)(buf[idx]))->push(data); // cannot fail
502 atomic_long_set(&seqP[idx],(pw+mask+1));
503 return true;
504 }
505
510 inline bool pop(void ** data) {
511 unsigned long pr,idx;
512 long seq;
513 unsigned long bk = BACKOFF_MIN;
514
515 do {
516 pr = atomic_long_read(&preadC);
517 idx = pr & mask;
518 seq = atomic_long_read(&seqC[idx]);
519 if (pr == (unsigned long)seq) {
520 if (atomic_long_read(&seqP[idx]) <= (unsigned long)seq) return false; // queue
521 if (abstraction_cas((volatile atom_t*)&preadC, (atom_t)(pr+1), (atom_t)pr)==(atom_t)pr)
522 break;
523
524 // exponential delay with max value
525 for(volatile unsigned i=0;i<bk;++i) ;
526 bk <<= 1;
527 bk &= BACKOFF_MAX;
528 }
529 } while(1);
530 ((uSWSR_Ptr_Buffer*)(buf[idx]))->pop(data);
531 atomic_long_set(&seqC[idx],(pr+mask+1));
532 return true;
533 }

541 atomic_long_t preadC;
542 char padding2[CACHE_LINE_SIZE];
543 };
544protected:
545 data_element_t * buf;
546 sequenceP_t * seqP;
547 sequenceC_t * seqC;
548 unsigned long mask;

550};

583 struct Pointer {
584 Pointer() { ptr[MSQUEUE_PTR]=0;}
585
586 inline bool operator !() {
587 return (ptr[MSQUEUE_PTR]==0);
588 }
589 inline Pointer& operator=(const Pointer & p) {

594 inline Pointer& operator=(Node & node) {
595 ptr[MSQUEUE_PTR]=&node;
596 return *this;
597 }
598
599 inline Pointer & getNodeNext() {
600 return ptr[MSQUEUE_PTR]->next;
601 }
602 inline Node * getNode() { return ptr[MSQUEUE_PTR]; }
603
604 inline bool operator==( const Pointer& r ) const {
605 return ((ptr[MSQUEUE_PTR]==r.ptr[MSQUEUE_PTR]));
606 }
607
608 inline operator volatile atom_t * () const {
609 union { Node* const volatile* p1; volatile atom_t * p2;} pn;
610 pn.p1 = ptr;
611 return pn.p2;
612 }
613 inline operator atom_t * () const {
614 union { Node* const volatile* p1; atom_t * p2;} pn;
615 pn.p1 = ptr;
616 return pn.p2;
617 }
618
619 inline operator atom_t () const {
620 union { Node* volatile p1; atom_t p2;} pn;
621 pn.p1 = ptr[MSQUEUE_PTR];
622 return pn.p2;
623 }
624
625 inline void set(Node & node) {
626 ptr[MSQUEUE_PTR]=&node;
627 }
628
629 inline void * getData() const { return ptr[MSQUEUE_PTR]->getData(); }
630
631 Node * volatile ptr[1];
632 } ALIGN_TO_POST(ALIGN_SINGLE_POINTER);
633
634 struct Node {
635 Node():data(0) { next.ptr[MSQUEUE_PTR]=0;}
636 Node(void * data):data(data) {
637 next.ptr[MSQUEUE_PTR]=0;
638 }
639
640 inline operator atom_t * () const { return (atom_t *)next; }
641
642 inline void setData(void * const d) { data=d;}
643 inline void * getData() const { return data; }
644
645 Pointer next;
646 void * data;
647 } ALIGN_TO_POST(ALIGN_DOUBLE_POINTER);
648
649 Pointer head;
650 long padding1[longxCacheLine-1];
651 Pointer tail;
652 long padding2[longxCacheLine-1];;
653 FFAllocator *delayedAllocator;
654
655private:
656 inline void allocnode(Pointer & p, void * data) {
657 union { Node * p1; void * p2;} pn;
658
659 if (delayedAllocator->posix_memalign((void**)&pn.p2,ALIGN_DOUBLE_POINTER,sizeof(Node))!=0) {
660 abort();
661 }
662 new (pn.p2) Node(data);
663 p.set(*pn.p1);
664 }

666 inline void deallocnode( Node * n) {
667 n->~Node();
668 delayedAllocator->free(n);
669 }

671public:
672 MSqueue(): delayedAllocator(NULL) { }
673
674 ~MSqueue() {
675 if (delayedAllocator) {
676 delete delayedAllocator;
677 delayedAllocator = NULL;
678 }
679 }

681 MSqueue& operator=(const MSqueue& v) {
682 head=v.head;
683 tail=v.tail;
684 return *this;
685 }

688 int init() {
689 if (delayedAllocator) return 0;
690 delayedAllocator = new FFAllocator(2);
691 if (!delayedAllocator) {
692 error("MSqueue::init, cannot allocate FFAllocator\n");
693 return -1;
694 }
695
696 // create the first NULL node
697 // so the queue is never really empty
698 Pointer dummy;
699 allocnode(dummy,NULL);
700
701 head = dummy;
702 tail = dummy;
703 return 1;
704 }
705
706 // insert method, it never fails
707 inline bool push(void * const data) {
708 bool done = false;
709
710 Pointer tailptr ALIGN_TO_POST(ALIGN_SINGLE_POINTER);
711 Pointer next ALIGN_TO_POST(ALIGN_SINGLE_POINTER);
712 Pointer node ALIGN_TO_POST(ALIGN_SINGLE_POINTER);
713 allocnode(node,data);
714
715 do {
716 tailptr = tail;
717 next = tailptr.getNodeNext();
718
719 if (tailptr == tail) {
720 if (!next) { // tail was pointing to the last node
721 done = (CAS((volatile atom_t *)(tailptr.getNodeNext()),
722 (atom_t)node,
723 (atom_t)next) == (atom_t)next);
724 } else { // tail was not pointing to the last node
725 CAS((volatile atom_t *)tail, (atom_t)next, (atom_t)tailptr);
726 }
727 }
728 } while(!done);
729 CAS((volatile atom_t *)tail, (atom_t)node, (atom_t) tailptr);
730 return true;
731 }
732
733 // extract method, it returns false if the queue is empty
734 inline bool pop(void ** data) {
735 bool done = false;
736
737 ALIGN_TO_PRE(ALIGN_SINGLE_POINTER) Pointer headptr;
738 ALIGN_TO_PRE(ALIGN_SINGLE_POINTER) Pointer tailptr;
739 ALIGN_TO_PRE(ALIGN_SINGLE_POINTER) Pointer next;
740
741 do {
742 headptr = head;
743 tailptr = tail;
744 next = headptr.getNodeNext();
745
746 if (head == headptr) {
747 if (headptr.getNode() == tailptr.getNode()) {
748 if (!next) return false; // empty
749 CAS((volatile atom_t *)tail, (atom_t)next, (atom_t)tailptr);
750 } else {
751 *data = next.getData();
752 done = (CAS((volatile atom_t *)head, (atom_t)next, (atom_t)headptr) == (atom_t)headptr);
753 }
754 }
755 } while(!done);
756
757 deallocnode(headptr.getNode());
758 return true;
759 }
760
761 // return true if the queue is empty
762 inline bool empty() {
763 if ((head.getNode() == tail.getNode()) && !(head.getNodeNext()))
764 return true;
765 return false;
766 }
767};

800 CLock=(CLHSpinLock*)getAlignedMemory(CACHE_LINE_SIZE,nqueues*sizeof(CLHSpinLock));
801
802 for(size_t i=0;i<nqueues;++i) {
803 buf[i]= new uSWSR_Ptr_Buffer(size);
804 buf[i]->init();
805 PLock[i].init();
806 CLock[i].init();
807 }
808 atomic_long_set(&count, 0);
809 atomic_long_set(&enqueue,0);
810 atomic_long_set(&dequeue,0);
811 return true;
812 }
813
814 // it always returns true
815 inline bool push(void *const data, int tid) {
816 long q = atomic_long_inc_return(&enqueue) & mask;
817 PLock[q].spin_lock(tid);
818 buf[q]->push(data);
819 PLock[q].spin_unlock(tid);
820 atomic_long_inc(&count);
821 return true;
822 }
823
824 // non-blocking pop
825 inline bool pop(void ** data, int tid) {
826 if (!atomic_long_read(&count)) return false; // empty
827
828 long q = atomic_long_inc_return(&dequeue) & mask;
829 CLock[q].spin_lock(tid);
830 bool r = buf[q]->pop(data);
831 CLock[q].spin_unlock(tid);
832 if (r) { atomic_long_dec(&count); return true;}
833 return false;
834 }
