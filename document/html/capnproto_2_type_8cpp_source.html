39void handle_fields_in_structs_or_unions(const auto& _struct_or_union,
40 const size_t _indent,
41 const std::string& _namespace,
42 std::ostream* _os, size_t* _ix) {
43 for (const auto& pair : _struct_or_union.fields) {
44 const auto name = pair.first;
45 // Because of the way Cap'n proto handles unions, we need a special case for
46 // them.
47 pair.second.reflection().visit([&](const auto& _r) {
48 using R = std::remove_cvref_t<decltype(_r)>;
49 if constexpr (std::is_same<R, Type::Union>()) {
50 // Special case: Union field.
51 *_os << std::string(_indent * 2, ' ') << name << " :union {"
52 << std::endl;
53 for (size_t i = 0; i < _r.fields.size(); ++i) {
54 _r.fields[i].second.reflection().visit([&](const auto& _union_field) {
55 using U = std::remove_cvref_t<decltype(_union_field)>;
56 if constexpr (std::is_same<U, Type::Union>()) {
57 *_os << std::string(_indent * 2 + 2, ' ')
58 << internal::strings::to_camel_case(_namespace + name + "_" +
59 _r.fields[i].first)
60 << " :union {" << std::endl;
61 handle_fields_in_structs_or_unions(
62 _union_field, _indent + 2,
63 _namespace + name + "_" + _r.fields[i].first + "_", _os, _ix);
64 *_os << std::string(_indent * 2 + 2, ' ') << "}" << std::endl;
65
66 } else {
67 *_os << std::string(_indent * 2 + 2, ' ')
68 << internal::strings::to_camel_case(_namespace + name + "_" +
69 _r.fields[i].first)
70 << " @" << (*_ix)++ << " :" << _union_field << ";"
71 << std::endl;
72 }
73 });
74 }
75 *_os << std::string(_indent * 2, ' ') << "}" << std::endl;
76
77 } else {
78 // Standard case: Non-union field.
79 *_os << std::string(_indent * 2, ' ')
80 << internal::strings::to_camel_case(_namespace + name) << " @"
81 << (*_ix)++ << " :" << _r << ";" << std::endl;
82 }
83 });
84 }
85}

87Type Type::with_name(const std::string& _name) const {
88 const auto set_name = [&](const auto& _v) -> ReflectionType {
89 using T = std::remove_cvref_t<decltype(_v)>;
90 if constexpr (std::is_same<T, Struct>() || std::is_same<T, Union>() ||
91 std::is_same<T, Enum>()) {
92 auto v_with_name = _v;
93 v_with_name.name = _name;
94 return v_with_name;
95 } else {
96 return _v;
97 }
98 };
99 return Type{.value = value.visit(set_name)};
100}
