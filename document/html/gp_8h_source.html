37 template <typename RCUtag>
38 inline void gp_thread_gc<RCUtag>::access_lock()
39 {
40 thread_record * pRec = get_thread_record();
41 assert( pRec != nullptr );
42
43 uint32_t tmp = pRec->m_nAccessControl.load( atomics::memory_order_relaxed );
44 if ( (tmp & rcu_class::c_nNestMask) == 0 ) {
45 pRec->m_nAccessControl.store( gp_singleton<RCUtag>::instance()->global_control_word(atomics::memory_order_relaxed),
46 atomics::memory_order_relaxed );
47
48 atomics::atomic_thread_fence( atomics::memory_order_seq_cst );
49 }
50 else {
51 // nested lock
52 pRec->m_nAccessControl.store( tmp + 1, atomics::memory_order_relaxed );
53 }
54 }

56 template <typename RCUtag>
57 inline void gp_thread_gc<RCUtag>::access_unlock()
58 {
59 thread_record * pRec = get_thread_record();
60 assert( pRec != nullptr );
61
62 uint32_t tmp = pRec->m_nAccessControl.load( atomics::memory_order_relaxed );
63 assert( (tmp & rcu_class::c_nNestMask) > 0 );
64
65#if CDS_COMPILER == CDS_COMPILER_CLANG && CDS_COMPILER_VERSION < 30800
66 // CLang 3.6-3.7: some tests of intrusive::FeldmanHashSet based on general-purpose RCU
67 // are failed even in single-threaded mode (unit tests) without magic compiler barrier below
68 CDS_COMPILER_RW_BARRIER;
69#endif
70 pRec->m_nAccessControl.store( tmp - 1, atomics::memory_order_release );
71 }

84 template <typename RCUtag>
85 inline bool gp_singleton<RCUtag>::check_grace_period( typename gp_singleton<RCUtag>::thread_record * pRec ) const
86 {
87 uint32_t const v = pRec->m_nAccessControl.load( atomics::memory_order_acquire );
88 return (v & general_purpose_rcu::c_nNestMask)
89 && (( v ^ m_nGlobalControl.load( atomics::memory_order_relaxed )) & ~general_purpose_rcu::c_nNestMask );
90 }

92 template <typename RCUtag>
93 template <class Backoff>
94 inline void gp_singleton<RCUtag>::flip_and_wait( Backoff& bkoff )
95 {
96 OS::ThreadId const nullThreadId = OS::c_NullThreadId;
97 m_nGlobalControl.fetch_xor( general_purpose_rcu::c_nControlBit, atomics::memory_order_seq_cst );
98
99 for ( thread_record * pRec = m_ThreadList.head( atomics::memory_order_acquire ); pRec; pRec = pRec->m_list.next_ ) {
100 while ( pRec->m_list.thread_id_.load( atomics::memory_order_acquire ) != nullThreadId && check_grace_period( pRec )) {
101 bkoff();
102 CDS_COMPILER_RW_BARRIER;
103 }
104 bkoff.reset();
105 }
106 }
