199public:
200 BufferView(void *native_handle, uint64_t handle,
201 size_t element_stride, size_t offset_bytes,
202 size_t size, size_t total_size) noexcept
203 : _native_handle{native_handle}, _handle{handle}, _offset_bytes{offset_bytes},
204 _element_stride{element_stride}, _size{size}, _total_size{total_size} {

228 [[nodiscard]] auto original() const noexcept {

232 }
233 [[nodiscard]] auto subview(size_t offset_elements, size_t size_elements) const noexcept {
234 if (size_elements + offset_elements > _size) [[unlikely]] {
235 detail::error_buffer_subview_overflow(offset_elements, size_elements, _size);
236 }

241 // reinterpret cast buffer to another type U
242 template<typename U>
243 requires(!is_custom_struct_v<U>)
244 [[nodiscard]] auto as() const noexcept {

248 auto total_size_bytes = _total_size * _element_stride;
249 return BufferView<U>{_native_handle, _handle, sizeof(U), _offset_bytes,
250 this->size_bytes() / sizeof(U), total_size_bytes / sizeof(U)};
251 }

266 }
