31Type Type::with_name(const std::string& _name) const {
32 const auto set_name = [&](const auto& _v) -> ReflectionType {
33 using T = std::remove_cvref_t<decltype(_v)>;
34 if constexpr (std::is_same<T, Record>() || std::is_same<T, Enum>()) {
35 auto v_with_name = _v;
36 v_with_name.name = _name;
37 return v_with_name;
38 } else {
39 return _v;
40 }
41 };
42 return Type{.value = value.visit(set_name)};
43}
