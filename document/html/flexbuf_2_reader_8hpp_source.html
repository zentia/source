71 rfl::Result<T> to_basic_type(const InputVarType& _var) const noexcept {
72 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
73 if (!_var.IsString()) {
74 return error("Could not cast to a string.");
75 }
76 return std::string(_var.AsString().c_str());
77
78 } else if constexpr (std::is_same<std::remove_cvref_t<T>,
79 rfl::Bytestring>()) {
80 if (!_var.IsBlob()) {
81 return error("Could not cast to a bytestring.");
82 }
83 const auto blob = _var.AsBlob();
84 const auto data = internal::ptr_cast<const std::byte*>(blob.data());
85 return rfl::Bytestring(data, data + blob.size());
86
87 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
88 if (!_var.IsBool()) {
89 return error("Could not cast to boolean.");
90 }
91 return _var.AsBool();
92
93 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
94 if (!_var.IsFloat()) {
95 return error("Could not cast to double.");
96 }
97 return static_cast<T>(_var.AsDouble());
98
99 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
100 if (!_var.IsIntOrUint()) {
101 return error("Could not cast to int.");
102 }
103 return static_cast<T>(_var.AsInt64());
104
105 } else {
106 static_assert(rfl::always_false_v<T>, "Unsupported type.");
107 }
