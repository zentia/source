113 void put( node * pNode )
114 {
115 assert( m_Head.is_lock_free());
116
117 tagged_ptr currentHead = m_Head.load( atomics::memory_order_relaxed );
118 tagged_ptr newHead = { pNode };
119 do {
120 newHead.tag = currentHead.tag + 1;
121 pNode->m_freeListNext.store( currentHead.ptr, atomics::memory_order_relaxed );
122 CDS_TSAN_ANNOTATE_HAPPENS_BEFORE( &pNode->m_freeListNext );
123 } while ( cds_unlikely( !m_Head.compare_exchange_weak( currentHead, newHead, atomics::memory_order_release, atomics::memory_order_acquire )));
124 }

127 node * get()
128 {
129 tagged_ptr currentHead = m_Head.load( atomics::memory_order_acquire );
130 tagged_ptr newHead;
131 while ( currentHead.ptr != nullptr ) {
132 CDS_TSAN_ANNOTATE_HAPPENS_AFTER( &currentHead.ptr->m_freeListNext );
133 newHead.ptr = currentHead.ptr->m_freeListNext.load( atomics::memory_order_relaxed );
134 newHead.tag = currentHead.tag + 1;
135 if ( cds_likely( m_Head.compare_exchange_weak( currentHead, newHead, atomics::memory_order_release, atomics::memory_order_acquire )))
136 break;
137 }
138 return currentHead.ptr;
139 }
