134 template <typename Value>
135 struct intrusive_hashtable_iterator_base
136 {
137 public:
138 typedef Value value_type;
139
140 protected:
141 template <typename, typename, typename, typename, size_t, bool, bool>
142 friend class intrusive_hashtable;
143
144 template <typename, bool>
145 friend struct intrusive_hashtable_iterator;
146
147 template <typename V>
148 friend bool operator==(const intrusive_hashtable_iterator_base<V>&, const intrusive_hashtable_iterator_base<V>&);
149
150 template <typename V>
151 friend bool operator!=(const intrusive_hashtable_iterator_base<V>&, const intrusive_hashtable_iterator_base<V>&);
152
153 value_type* mpNode; // Current node within current bucket.
154 value_type** mpBucket; // Current bucket.
155
156 public:
157 intrusive_hashtable_iterator_base(value_type* pNode, value_type** pBucket)
158 : mpNode(pNode), mpBucket(pBucket) { }
159
160 void increment_bucket()
161 {
162 ++mpBucket;
163 while(*mpBucket == NULL) // We store an extra bucket with some non-NULL value at the end
164 ++mpBucket; // of the bucket array so that finding the end of the bucket
165 mpNode = *mpBucket; // array is quick and simple.
166 }
167
168 void increment()
169 {
170 mpNode = static_cast<value_type*>(mpNode->mpNext);
171
172 while(mpNode == NULL)
173 mpNode = *++mpBucket;
174 }
175
176 }; // intrusive_hashtable_iterator_base

497 void intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::swap(this_type& x)
498 {
499 for(size_t i = 0; i < kBucketCount; i++)

594 template <typename K, typename V, typename H, typename Eq, size_t bC, bool bM, bool bU>
595 typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::size_type
596 intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::count(const key_type& k) const
597 {
598 const size_type n = (size_type)(mHash(k) % kBucketCount);
599 size_type result = 0;
600 extract_key extractKey; // extract_key is empty and thus this ctor is a no-op.
601
602 // To do: Make a specialization for bU (unique keys) == true and take
603 // advantage of the fact that the count will always be zero or one in that case.
604 for(node_type* pNode = mBucketArray[n]; pNode; pNode = static_cast<node_type*>(pNode->mpNext))
605 {
606 if(mEqual(k, extractKey(*pNode)))
607 ++result;
608 }
609 return result;
610 }

613 template <typename K, typename V, typename H, typename Eq, size_t bC, bool bM, bool bU>
614 eastl::pair<typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::iterator,
615 typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::iterator>
616 intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::equal_range(const key_type& k)
617 {
618 const size_type n = (size_type)(mHash(k) % kBucketCount);
619 node_type** head = mBucketArray + n;
620 node_type* pNode = DoFindNode(*head, k);
621 extract_key extractKey; // extract_key is empty and thus this ctor is a no-op.
622
623 if(pNode)
624 {
625 node_type* p1 = static_cast<node_type*>(pNode->mpNext);
626
627 for(; p1; p1 = static_cast<node_type*>(p1->mpNext))
628 {
629 if(!mEqual(k, extractKey(*p1)))
630 break;
631 }
632
633 iterator first(pNode, head);
634 iterator last(p1, head);
635
636 if(!p1)
637 last.increment_bucket();
638
639 return eastl::pair<iterator, iterator>(first, last);
640 }
641
642 return eastl::pair<iterator, iterator>(iterator(mBucketArray + kBucketCount),
643 iterator(mBucketArray + kBucketCount));
644 }

699 template <typename U, typename BinaryPredicate>
700 inline typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::node_type*
701 intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::DoFindNode(node_type* pNode, const U& other, BinaryPredicate predicate) const
702 {
703 extract_key extractKey; // extract_key is empty and thus this ctor is a no-op.
704
705 for(; pNode; pNode = static_cast<node_type*>(pNode->mpNext))

714 template <typename K, typename V, typename H, typename Eq, size_t bC, bool bM, bool bU>
715 eastl::pair<typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::iterator, bool>
716 intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::DoInsertValue(value_type& value, true_type) // true_type means bUniqueKeys is true.
717 {
718 // For sets (as opposed to maps), one could argue that all insertions are successful,
719 // as all elements are unique. However, the equal function might not think so.
720 extract_key extractKey; // extract_key is empty and thus this ctor is a no-op.
721 const size_type n = (size_type)(mHash(extractKey(value)) % kBucketCount);
722 node_type* const pNode = DoFindNode(mBucketArray[n], extractKey(value));
723
724 if(pNode == NULL)

727 mBucketArray[n] = &value;
728 ++mnElementCount;
729
730 return eastl::pair<iterator, bool>(iterator(&value, mBucketArray + n), true);
731 }
732
733 return eastl::pair<iterator, bool>(iterator(pNode, mBucketArray + n), false);

772 template <typename K, typename V, typename H, typename Eq, size_t bC, bool bM, bool bU>
773 typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::iterator
774 intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::erase(const_iterator i)
775 {
776 iterator iNext(i.mpNode, i.mpBucket);
777 ++iNext;
778
779 node_type* pNode = i.mpNode;
780 node_type* pNodeCurrent = *i.mpBucket;
781
782 if(pNodeCurrent == pNode)
783 *i.mpBucket = static_cast<node_type*>(pNodeCurrent->mpNext);
784 else
785 {
786 // We have a singly-linked list, so we have no choice but to
787 // walk down it till we find the node before the node at 'i'.
788 node_type* pNodeNext = static_cast<node_type*>(pNodeCurrent->mpNext);
789
790 while(pNodeNext != pNode)
791 {
792 pNodeCurrent = pNodeNext;
793 pNodeNext = static_cast<node_type*>(pNodeCurrent->mpNext);
794 }
795
796 pNodeCurrent->mpNext = static_cast<node_type*>(pNodeNext->mpNext);
797 }
798
799 // To consider: In debug builds set the node mpNext to NULL.
800 --mnElementCount;
801
802 return iNext;
803 }

816 template <typename K, typename V, typename H, typename Eq, size_t bC, bool bM, bool bU>
817 typename intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::size_type
818 intrusive_hashtable<K, V, H, Eq, bC, bM, bU>::erase(const key_type& k)
819 {
820 const size_type n = (size_type)(mHash(k) % kBucketCount);
821 const size_type nElementCountSaved = mnElementCount;
822 node_type*& pNodeBase = mBucketArray[n];
823 extract_key extractKey; // extract_key is empty and thus this ctor is a no-op.
824
825 // Note by Paul Pedriana:
826 // We have two loops here, and I'm not finding any easy way to having just one
827 // loop without changing the requirements of the hashtable node definition.
828 // It's a problem of taking an address of a variable and converting it to the
829 // address of another type without knowing what that type is. Perhaps I'm a
830 // little overly tired, so if there is a simple solution I am probably missing it.
831
832 while(pNodeBase && mEqual(k, extractKey(*pNodeBase)))
833 {
834 pNodeBase = static_cast<node_type*>(pNodeBase->mpNext);
835 --mnElementCount;
836 }
837
838 node_type* pNodePrev = pNodeBase;
839
840 if(pNodePrev)
841 {
842 node_type* pNodeCur;
843
844 while((pNodeCur = static_cast<node_type*>(pNodePrev->mpNext)) != NULL)
845 {
846 if(mEqual(k, extractKey(*pNodeCur)))
847 {
848 pNodePrev->mpNext = static_cast<node_type*>(pNodeCur->mpNext);
849 --mnElementCount; // To consider: In debug builds set the node mpNext to NULL.
850 }
851 else
852 pNodePrev = static_cast<node_type*>(pNodePrev->mpNext);
853 }
854 }
855
856 return nElementCountSaved - mnElementCount;
857 }
