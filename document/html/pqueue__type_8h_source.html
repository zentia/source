45 template <typename T, typename Container, typename Lock, typename Less = std::less<typename Container::value_type> >
46 class StdPQueue
47 {
48 public:
49 typedef T value_type;
50 typedef std::priority_queue<value_type, Container, Less> pqueue_type;
51
52 private:
53 pqueue_type m_PQueue;
54 mutable Lock m_Lock;
55
56 typedef std::unique_lock<Lock> scoped_lock;
57
58 public:
59 bool push( value_type const& val )
60 {
61 scoped_lock l( m_Lock );
62 m_PQueue.push( val );
63 return true;
64 }
65
66 bool pop( value_type& dest )
67 {
68 scoped_lock l( m_Lock );
69 if ( !m_PQueue.empty()) {
70 dest = m_PQueue.top();
71 m_PQueue.pop();
72 return true;
73 }
74 return false;
75 }
76
77 template <typename Q, typename MoveFunc>
78 bool pop_with( Q& dest, MoveFunc f )
79 {
80 scoped_lock l( m_Lock );
81 if ( !m_PQueue.empty()) {
82 f( dest, m_PQueue.top());
83 m_PQueue.pop();
84 return true;
85 }
86 return false;
87 }
88
89 void clear()
90 {
91 scoped_lock l( m_Lock );
92 while ( !m_PQueue.empty())
93 m_PQueue.pop();
94 }
95
96 template <typename Func>
97 void clear_with( Func f )
98 {
99 scoped_lock l( m_Lock );
100 while ( !m_PQueue.empty()) {
101 f( m_PQueue.top());
102 m_PQueue.pop();
103 }
104 }
105
106 bool empty() const
107 {
108 scoped_lock l( m_Lock );
109 return m_PQueue.empty();
110 }
111
112 size_t size() const
113 {
114 scoped_lock l( m_Lock );
115 return m_PQueue.size();
116 }
117
118 cds::opt::none statistics() const
119 {
120 return cds::opt::none();
121 }
122 };
