31/*
32 * - Author:
33 * Massimo Torquati <torquati@di.unipi.it>
34 *
35 *
36 * This file contains the ParallelFor and the ParallelForReduce classes
37 * (and also some static functions).
38 *
39 * Iterations scheduling:
40 *
41 * As a general rule, the scheduling strategy is selected according to the chunk value:
42 * - chunk == 0 means default static scheduling, that is, ~(#iteration_space/num_workers)
43 * iterations per thread assigned in one single shot at the beginning.
44 * - chunk > 0 means dynamic scheduling with grain equal to the chunk size, that is,
45 * no more than chunk iterations at a time is assigned to one Worker, the
46 * chunk is assigned to Workers dynamically
47 * - chunk < 0 means static scheduling with grain equal to the chunk size, that is,
48 * the iteration space is divided into chunks each one of no more
49 * than chunk iterations. Then chunks are assigned to the Workers statically
50 * and in a round-robin fashion.
51 *
52 * If you want to use the static scheduling policy (either default or with a given grain),
53 * please use the **parallel_for_static** method.
54 *
55 * To use or not to use a scheduler thread ?
56 * As always, it depends on the application, scheduling strategy, platform at hand,
57 * parallelism degree, ...etc....
58 *
59 * The general rule is: a scheduler thread is started if:
60 * 1. the dynamic scheduling policy is used (chunk>0);
61 * 2. there are enough cores for hosting both worker threads and the scheduler thread;
62 * 3. the number of tasks per thread is greater than 1.
63 *
64 * In case of static scheduling (chunk <= 0), the scheduler thread is never started.
65 * It is possible to explicitly disable/enable the presence of the scheduler thread
66 * both at compile time and at run-time by using the disableScheduler method and the
67 * two defines NO_PARFOR_SCHEDULER_THREAD and PARFOR_SCHEDULER_THREAD.
68 *
69 *
70 * How to use the ParallelFor (in a nutshell) :
71 * ParallelForReduce<long> pfr;
72 * for(long i=0;i<N;i++) pfr.parallel_for(0,N,[&](const long i) {
73 * A[i]=f(i); A[i]=f(i);
74 * long sum=0; ---> });
75 * for(long i=0; i<N;++i) long sum=0;
76 * sum+=g(A[i]); pfr.parallel_reduce(sum,0,0,N,[&](const long i,long &sum) {
77 * sum+=g(A[i]);
78 * }, [](long &v, const long elem) {v+=elem;});
79 *
80 *
81 *
82 * For just a single parallel loop, it is better to use the one-shot version (see at the end of
83 * this file). Useful when there is just
84 * a single parallel loop.
85 * This version should not be used if the parallel loop is called many
86 * times (e.g., within a sequential loop) or if there are several loops
87 * that can be parallelized by using the same ParallelFor* object.
88 * If this is the case, the version with the object instance is more
89 * efficient because the Worker threads are created once and then
90 * re-used many times.
91 * On the contrary, the one-shot version has a lower setup overhead but
92 * Worker threads are destroyed at the end of the loop.
93
94 *
95 */

706template<typename task_t>
707class ParallelForPipeReduce: public ff_pipeline {
708protected:
709 ff_forall_farm<forallpipereduce_W> pfr;
710 struct reduceStage: ff_minode {
711 typedef std::function<void(const task_t &)> F_t;
712 inline void *svc(void *t) {
713 const task_t& task=reinterpret_cast<task_t>(t);
714 F(task);
715 return GO_ON;
716 }
717 inline int wait() { return ff_minode::wait(); }
718 inline void setF(F_t f) { F = f; }
719
720 F_t F;
721 } reduce;
722
723public:
724 explicit ParallelForPipeReduce(const long maxnw=FF_AUTO, bool spinwait=false, bool /*spinbarrier*/=false):
725 pfr(maxnw,false,true,false) // skip loop warmup and disable spinwait/spinbarrier
726 {
727 ff_pipeline::add_stage(&pfr);
728 ff_pipeline::add_stage(&reduce);
729
730 // required to avoid error
731 pfr.remove_collector();
732
733 // avoiding initial barrier
734 if (ff_pipeline::dryrun()<0) // preparing all connections
735 error("ParallelForPipeReduce: preparing pipe\n");
736
737 // warmup phase
738 pfr.resetskipwarmup();
739 auto r=-1;
740 if (pfr.run_then_freeze() != -1)
741 if (reduce.run_then_freeze() != -1)
742 r = ff_pipeline::wait_freezing();
743 if (r<0) error("ParallelForPipeReduce: running pipe\n");
744
745
746 if (spinwait) { // NOTE: spinning is enabled only for the Map part and not for the Reduce part
747 if (pfr.enableSpinning() == -1)
748 error("ParallelForPipeReduce: enabling spinwait\n");
749 }
750 }
751
752 ~ParallelForPipeReduce() {
753 pfr.stop(); pfr.wait();
754 reduce.wait();
755 }
756
757 // By calling this method with 'true' the scheduler will be disabled,
758 // to restore the usage of the scheduler thread just pass 'false' as
759 // parameter
760 inline void disableScheduler(bool onoff=true) {
761 pfr.disableScheduler(onoff);
762 }
763
764 // It puts all spinning threads to sleep. It does not disable the spinWait flag
765 // so at the next call, threads start spinning again.
766 inline int threadPause() {
767 return pfr.stopSpinning();

775 inline void parallel_for_idx(long first, long last, long step, long grain,
776 const Function& Map, const long nw=FF_AUTO) {
777
778 // the setloop decides the real number of worker threads that will be started
779 // the n. maybe different from nw !
780 pfr.setloop(first,last,step,grain,nw);
781 pfr.setF(Map);
782 auto donothing=[](task_t) { };
783 reduce.setF(donothing);
784 auto r=-1;
785 if (pfr.run_then_freeze(pfr.getnw()) != -1)
786 if (reduce.run_then_freeze(pfr.getnw()) != -1)
787 r = ff_pipeline::wait_freezing();
