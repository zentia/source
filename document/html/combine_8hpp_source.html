110 ff_comb(const ff_comb& c) : ff_minode(c) {
111 for (auto s: c.comp_nodes) {
112 if (s->isComp()) {
113 comp_nodes.push_back(new ff_comb(*(ff_comb*)s));
114 assert(comp_nodes.back());
115 cleanup_stages.push_back(comp_nodes.back());
116 } else {
117 comp_nodes.push_back(s);
118 }
119 }
120 // this is a dirty part, we modify a const object.....
121 ff_comb *dirty= const_cast<ff_comb*>(&c);
122 for (size_t i=0;i<dirty->cleanup_stages.size();++i) {
123 cleanup_stages.push_back(dirty->cleanup_stages[i]);
124 dirty->cleanup_stages[i]=nullptr;
125 }
126 }

134 int run(bool skip_init=false) {
135 if (!skip_init) {
136 if (getFirst()->get_in_buffer() == nullptr)
137 getFirst()->skipfirstpop(true);
138 }
139
140 if (!prepared) if (prepare()<0) return -1;
141
142 // set blocking mode for the last node of the composition
143 getLast()->blocking_mode(blocking_in);
144 if (comp_nodes[0]->isMultiInput()) {
145 svector<ff_node*> w(1);
146 getFirst()->get_in_nodes(w);
147 if (w.size() == 0) getFirst()->skipfirstpop(true);
148 return ff_minode::run();
149 }
150
151 if (ff_node::run(true)<0) return -1;
152 return 0;
153 }

215 ff_node* replace_first(ff_node* n, bool cleanup=false, bool remove_from_cleanuplist=true) {
216 if (comp_nodes[0]->isComp()) return nullptr;
217 ff_node* first = comp_nodes[0];
218 comp_nodes[0] = n;
219
220 if (remove_from_cleanuplist) {
221 ssize_t pos=-1;
222 for(size_t i=0;i<cleanup_stages.size();++i)
223 if (cleanup_stages[i] == first) { pos=i; break;}
224 if (pos>=0)
225 cleanup_stages.erase(cleanup_stages.begin()+pos);
226 }
227 if (cleanup)
228 cleanup_stages.push_back(n);
229 return first;
230 }

445 int prepare() {
446 if (prepared) return 0;
447 connectCallback();
448
449 // checking if the first node is a multi-input node
450 ff_node *n1 = getFirst();
451 if (n1->isMultiInput()) {
452 // here we substitute the gt
453 ((ff_minode*)n1)->setgt(ff_minode::getgt());
454 }
455 // dryrun should be executed here because the gt of the
456 // first node might have been substituted
457 ff_comb::dryrun();
458
459 // registering a special callback if the last stage does
460 // not have an output channel
461 ff_node *n2 = getLast();
462 if (n2->isMultiOutput()) {
463 svector<ff_node*> w(1);
464 n2->get_out_nodes(w);
465 if ((w.size()==0) && (n2->callback == nullptr))
466 n2->registerCallback(devnull, nullptr); // devnull callback
467 } else
468 if ((n2->get_out_buffer() == nullptr) && (n2->callback == nullptr))
469 n2->registerCallback(devnull, nullptr); // devnull callback
470
471
472 prepared = true;
473 return 0;
474 }

879static inline const ff_pipeline combine_nodes_in_pipeline(ff_node& node1, ff_node& node2, bool cleanup1=false, bool cleanup2=false) {
880 if (node1.isAll2All() || node2.isAll2All()) {
881 error("combine_nodes_in_pipeline, cannot be used if one of the nodes is A2A\n");
882 return ff_pipeline();
883 }
884 if (node1.isOFarm()) {
885 if (node2.isFarm()) {
886 ff_farm *farm1 = reinterpret_cast<ff_farm*>(&node1);
887 ff_farm *farm2 = reinterpret_cast<ff_farm*>(&node2);
888 if (cleanup1) farm1->cleanup_all();
889 if (cleanup2) farm2->cleanup_all();
890 return combine_ofarm_farm(*farm1, *farm2);
891 }
892 error("combine_nodes_in_pipeline, FEATURE NOT YET SUPPORTED (node1 ordered farm and node2 standard or combine node\n");
893 abort(); // FIX: TODO <---------
894 return ff_pipeline();
895 }
896 if (!node1.isFarm() && !node2.isFarm()) { // two sequential nodes
897 ff_pipeline pipe;
898 pipe.add_stage(&node1, cleanup1);
899 pipe.add_stage(&node2, cleanup2);
900 return pipe;
901 } else if (!node1.isFarm() && node2.isFarm()) { // seq with farm's emitter
902 ff_pipeline pipe;
903 ff_farm* farm = reinterpret_cast<ff_farm*>(&node2);
904 ff_node *e = farm->getEmitter();
905 if (!e) farm->add_emitter(&node1);
906 else {
907 ff_comb *p;
908 if (!e->isMultiOutput()) { // we have to transform the emitter node into a multi-output
909 if (e->isMultiInput()) { // this is a "strange" case: the emitter is multi-input without being also multi-output (through a combine node)
910 struct hnode:ff_monode {
911 void* svc(void*in) {return in;}
912 };
913
914 // c is a multi-input AND multi-output node
915 ff_comb *c = new ff_comb(e, new hnode,
916 farm->isset_cleanup_emitter(), true);
917 assert(c);
918 p = new ff_comb(&node1, c, cleanup1, true);
919 } else {
920 auto mo = new internal_mo_transformer(e, farm->isset_cleanup_emitter());
921 assert(mo);
922 p = new ff_comb(&node1, mo, cleanup1, true);
923 }
924 } else {
925 p = new ff_comb(&node1, e, cleanup1, farm->isset_cleanup_emitter());
926 }
927 assert(p);
928 if (farm->isset_cleanup_emitter()) farm->cleanup_emitter(false);
929 farm->change_emitter(p, true);
930 }
931 pipe.add_stage(farm, cleanup2);
932 return pipe;
933 } else if (node1.isFarm() && !node2.isFarm()) { // first farm and seq
934 ff_pipeline pipe;
935 ff_farm* farm = reinterpret_cast<ff_farm*>(&node1);
936 ff_node *c = farm->getCollector();
937 if (!c) farm->add_collector(&node2, cleanup2);
938 else {
939 ff_comb *p = new ff_comb(c, &node2, farm->isset_cleanup_collector(), cleanup2);
940 if (farm->isset_cleanup_collector()) farm->cleanup_collector(false);
941 farm->remove_collector();
942 farm->add_collector(p, true);
943 }
944 pipe.add_stage(farm, cleanup1);
945 return pipe;
946 }
947 assert(node1.isFarm() && node2.isFarm());
948 ff_farm* farm1 = reinterpret_cast<ff_farm*>(&node1);
949 ff_farm* farm2 = reinterpret_cast<ff_farm*>(&node2);
950
951 ff_node *e = farm2->getEmitter();
952 ff_node *c = farm1->getCollector();
953 ff_pipeline pipe;
954 if (c) {
955 ff_comb *p = new ff_comb(c,e,
956 farm1->isset_cleanup_collector(),
957 farm2->isset_cleanup_emitter());
958 if (farm1->isset_cleanup_collector()) farm1->cleanup_collector(false);
959 if (farm2->isset_cleanup_emitter()) farm2->cleanup_emitter(false);
960 farm2->change_emitter(p, true);
961 }
962 farm1->remove_collector();
963 pipe.add_stage(farm1, cleanup1);
964 pipe.add_stage(farm2, cleanup2);
965 return pipe;
966}

1275 * depending on the parameter passed to the function (node1,node2 and mergeCE):
1276 *
1277 * case1 - node1 and node2 are both null:
1278 * 1. mergeCE==false: it produces a pipeline of a single farm
1279 * whose worker is an all-to-all building block (equivalent behavior
1280 * of calling combine_farms_a2a(farm1,farm2)
1281 * 2. mergeCE==true: if the parallelism degree of the two farms is
1282 * the same, it produces a pipeline of a single farm
1283 * whose workers are a composition of both farm1 and farm2 workers (normal form).
1284 * If the parallelism degree of the two farms is different, we fall back
1285 * to the case1.1
1286 *
1287 * case2 - node1 is null and node2 is not null
1288 * 1. mergeCE==false: it produces a pipeline of a single farm
1289 * whose worker is an all-to-all building block where the nodes of the second set
1290 * is a composition of node2 and farm2's workers
1291 * 2. mergeCE==true: this produces a pipeline of two farms where the first
1292 * farm has no collector while the second farm has as emitter the node2.
1293 *
1294 * case3 - node1 is not null and node2 is null
1295 * 1. mergeCE==false: it produces a pipeline of a single farm
1296 * whose worker is an all-to-all building block where the nodes of the first set
1297 * is a composition of farm1's workers and node1
1298 * 2. mergeCE==true: this produces a pipeline of two farms where the first
1299 * farm has no collector while the second farm has as emitter the node1.

1302 * 1. mergeCE==false: it produces a pipeline of a single farm
1303 * whose worker is an all-to-all building block where the nodes of the first set
1304 * is a composition of farm1's workers and node1 whereas the nodes of the second set
1305 * is a composition of nodes2 and farm2's workers.
