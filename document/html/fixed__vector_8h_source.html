236 template <typename T, size_t nodeCount, bool bEnableOverflow, typename OverflowAllocator>
237 inline fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::fixed_vector(this_type&& x)
238 : base_type(fixed_allocator_type(mBuffer.buffer))
239 {
240 // Since we are a fixed_vector, we can't swap pointers. We can possibly so something like fixed_swap or
241 // we can just do an assignment from x. If we want to do the former then we need to have some complicated
242 // code to deal with overflow or no overflow, and whether the memory is in the fixed-size buffer or in
243 // the overflow allocator. 90% of the time the memory should be in the fixed buffer, in which case
244 // a simple assignment is no worse than the fancy pathway.
245
246 // Since we are a fixed_list, we can't normally swap pointers unless both this and
247 // x are using using overflow and the overflow allocators are equal. To do:
248 //if(has_overflowed() && x.has_overflowed() && (get_overflow_allocator() == x.get_overflow_allocator()))
249 //{
250 // We can swap contents and may need to swap the allocators as well.
251 //}
252
253 // The following is currently identical to the fixed_vector(const this_type& x) code above. If it stays that
254 // way then we may want to make a shared implementation.
255 if(has_overflowed() && x.has_overflowed() && (get_overflow_allocator() == x.get_overflow_allocator())) {
256 mpBegin = x.mpBegin;
257 mpEnd = x.mpEnd;
258 } else{
259 get_allocator().copy_overflow_allocator(x.get_allocator());
260
261 #if EASTL_NAME_ENABLED
262 get_allocator().set_name(x.get_allocator().get_name());
263 #endif
264
265 mpBegin = mpEnd = (value_type*)&mBuffer.buffer[0];
266 internalCapacityPtr() = mpBegin + nodeCount;
267 base_type::template DoAssign<move_iterator<iterator>, true>(eastl::make_move_iterator(x.begin()), eastl::make_move_iterator(x.end()), false_type());
268 if constexpr (!std::is_trivially_destructible_v<T>){
269 for(auto ptr = x.mpBegin; ptr != x.mpEnd; ++ptr){
270 ptr->~T();
271 }
272 }
273 }
274 new (std::launder(&x)) this_type();
275 }

364 inline typename fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::this_type&
365 fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::operator=(this_type&& x)
366 {
367 // Since we are a fixed_vector, we can't swap pointers. We can possibly do something like fixed_swap or
368 // we can just do an assignment from x. If we want to do the former then we need to have some complicated
369 // code to deal with overflow or no overflow, and whether the memory is in the fixed-size buffer or in
370 // the overflow allocator. 90% of the time the memory should be in the fixed buffer, in which case
371 // a simple assignment is no worse than the fancy pathway.
372 if (this != &x) [[likely]]
373 {
374 if(has_overflowed() && x.has_overflowed() && (get_overflow_allocator() == x.get_overflow_allocator())) {
375 mpBegin = x.mpBegin;
376 mpEnd = x.mpEnd;
377 } else{
378 clear();
379

382 #endif
383
384 base_type::template DoAssign<move_iterator<iterator>, true>(eastl::make_move_iterator(x.begin()), eastl::make_move_iterator(x.end()), false_type()); // Shorter route.
385 if constexpr (!std::is_trivially_destructible_v<T>){
386 for(auto ptr = x.mpBegin; ptr != x.mpEnd; ++ptr){
387 ptr->~T();
388 }
389 }
390 }
391 new (std::launder(&x)) this_type();
392 }

397 template <typename T, size_t nodeCount, bool bEnableOverflow, typename OverflowAllocator>
398 inline void fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::swap(this_type& x)
399 {
400 if((has_overflowed() && x.has_overflowed()) && (get_overflow_allocator() == x.get_overflow_allocator())) // If both containers are using the heap instead of local memory
401 { // then we can do a fast pointer swap instead of content swap.
402 eastl::swap(mpBegin, x.mpBegin);
403 eastl::swap(mpEnd, x.mpEnd);
404 eastl::swap(internalCapacityPtr(), x.internalCapacityPtr());
405 }
406 else
407 {
408 // Fixed containers use a special swap that can deal with excessively large buffers.
409 eastl::fixed_swap(*this, x);
410 }
411 }

414 template <typename T, size_t nodeCount, bool bEnableOverflow, typename OverflowAllocator>
415 inline void fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::set_capacity(size_type n)
416 {
417 const size_type nPrevSize = (size_type)(mpEnd - mpBegin);
418 const size_type nPrevCapacity = (size_type)(internalCapacityPtr() - mpBegin);
419
420 if(n == npos) // If the user means to set the capacity so that it equals the size (i.e. free excess capacity)...
421 n = nPrevSize;
422
423 if(n != nPrevCapacity) // If the request results in a capacity change...
424 {
425 if(can_overflow() && (((uintptr_t)mpBegin != (uintptr_t)mBuffer.buffer) || (n > kMaxSize))) // If we are or would be using dynamically allocated memory instead of our fixed-size member buffer...
426 {
427 T* const pNewData = (n <= kMaxSize) ? (T*)&mBuffer.buffer[0] : DoAllocate(n);
428 T* const pCopyEnd = (n < nPrevSize) ? (mpBegin + n) : mpEnd;
429 eastl::uninitialized_move_ptr(mpBegin, pCopyEnd, pNewData); // Move [mpBegin, pCopyEnd) to p.
430 eastl::destruct(mpBegin, mpEnd);
431 if((uintptr_t)mpBegin != (uintptr_t)mBuffer.buffer)
432 DoFree(mpBegin, (size_type)(internalCapacityPtr() - mpBegin));
433
434 mpEnd = pNewData + (pCopyEnd - mpBegin);
435 mpBegin = pNewData;
436 internalCapacityPtr() = mpBegin + n;
437 } // Else the new capacity would be within our fixed buffer.
438 else if(n < nPrevSize) // If the newly requested capacity is less than our size, we do what vector::set_capacity does and resize, even though we actually aren't reducing the capacity.
439 resize(n);
440 }
441 }

464 template <typename T, size_t nodeCount, bool bEnableOverflow, typename OverflowAllocator>
465 inline bool fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::full() const
466 {
467 // If size >= capacity, then we are definitely full.
468 // Also, if our size is smaller but we've switched away from mBuffer due to a previous overflow, then we are considered full.
469 return ((size_t)(mpEnd - mpBegin) >= kMaxSize) || ((void*)mpBegin != (void*)mBuffer.buffer);
470 }

473 template <typename T, size_t nodeCount, bool bEnableOverflow, typename OverflowAllocator>
474 inline bool fixed_vector<T, nodeCount, bEnableOverflow, OverflowAllocator>::has_overflowed() const
475 {
476 // This will be incorrect for the case that bOverflowEnabled is true and the container was resized
477 // down to a small size where the fixed buffer could take over ownership of the data again.
478 // The only simple fix for this is to take on another member variable which tracks whether this overflow
479 // has occurred at some point in the past.
480 return ((void*)mpBegin != (void*)mBuffer.buffer);
481 }
