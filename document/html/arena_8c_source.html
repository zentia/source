8/* ----------------------------------------------------------------------------
9"Arenas" are fixed area's of OS memory from which we can allocate
10large blocks (>= MI_ARENA_MIN_BLOCK_SIZE, 4MiB).
11In contrast to the rest of mimalloc, the arenas are shared between
12threads and need to be accessed using atomic operations.
13
14Arenas are also used to for huge OS page (1GiB) reservations or for reserving
15OS memory upfront which can be improve performance or is sometimes needed
16on embedded devices. We can also employ this with WASI or `sbrk` systems
17to reserve large arenas upfront and be able to reuse the memory more effectively.
18
19The arena allocation needs to be thread safe and we use an atomic bitmap to allocate.
20-----------------------------------------------------------------------------*/

35// A memory arena descriptor
36typedef struct mi_arena_s {
37 mi_memid_t memid; // memid of the memory area
38 mi_subproc_t* subproc; // subprocess this arena belongs to (`this 'in' this->subproc->arenas`)
39
40 size_t slice_count; // total size of the area in arena slices (of `MI_ARENA_SLICE_SIZE`)
41 size_t info_slices; // initial slices reserved for the arena bitmaps
42 int numa_node; // associated NUMA node
43 bool is_exclusive; // only allow allocations if specifically for this arena
44 _Atomic(mi_msecs_t) purge_expire; // expiration time when slices can be purged from `slices_purge`.
45
46 mi_bbitmap_t* slices_free; // is the slice free? (a binned bitmap with size classes)
47 mi_bitmap_t* slices_committed; // is the slice committed? (i.e. accessible)
48 mi_bitmap_t* slices_dirty; // is the slice potentially non-zero?
49 mi_bitmap_t* slices_purge; // slices that can be purged
50 mi_bitmap_t* pages; // all registered pages (abandoned and owned)
51 mi_bitmap_t* pages_abandoned[MI_BIN_COUNT]; // abandoned pages per size bin (a set bit means the start of the page)
52 // the full queue contains abandoned full pages
53 // followed by the bitmaps (whose sizes depend on the arena size)
54 // note: when adding bitmaps revise `mi_arena_info_slices_needed`
55} mi_arena_t;

182static mi_decl_noinline void* mi_arena_try_alloc_at(
183 mi_arena_t* arena, size_t slice_count, bool commit, size_t tseq, mi_memid_t* memid)
184{
185 size_t slice_index;
186 if (!mi_bbitmap_try_find_and_clearN(arena->slices_free, slice_count, tseq, &slice_index)) return NULL;
187
188 // claimed it!
189 void* p = mi_arena_slice_start(arena, slice_index);
190 *memid = mi_memid_create_arena(arena, slice_index, slice_count);
191 memid->is_pinned = arena->memid.is_pinned;
192
193 // set the dirty bits and track which slices become accessible
194 size_t touched_slices = slice_count;
195 if (arena->memid.initially_zero) {
196 size_t already_dirty = 0;
197 memid->initially_zero = mi_bitmap_setN(arena->slices_dirty, slice_index, slice_count, &already_dirty);
198 mi_assert_internal(already_dirty <= touched_slices);
199 touched_slices -= already_dirty;
200 }
201
202 // set commit state
203 if (commit) {
204 memid->initially_committed = true;
205
206 // commit requested, but the range may not be committed as a whole: ensure it is committed now
207 if (!mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count)) {
208 // not fully committed: commit the full range and set the commit bits
209 // we set the bits first since we own these slices (they are no longer free)
210 size_t already_committed_count = 0;
211 mi_bitmap_setN(arena->slices_committed, slice_index, slice_count, &already_committed_count);
212 // now actually commit
213 bool commit_zero = false;
214 if (!_mi_os_commit_ex(p, mi_size_of_slices(slice_count), &commit_zero, mi_size_of_slices(slice_count - already_committed_count))) {
215 memid->initially_committed = false;
216 }
217 else {
218 // committed
219 if (commit_zero) { memid->initially_zero = true; }
220 #if MI_DEBUG > 1
221 if (memid->initially_zero) {
222 if (!mi_mem_is_zero(p, mi_size_of_slices(slice_count))) {
223 _mi_error_message(EFAULT, "interal error: arena allocation was not zero-initialized!\n");
224 memid->initially_zero = false;
225 }
226 }
227 #endif
228 }
229 }
230 else {
231 // already fully commited.
232 // if the OS has overcommit, and this is the first time we access these pages, then
233 // count the commit now (as at arena reserve we didn't count those commits as these are on-demand)
234 if (_mi_os_has_overcommit() && touched_slices > 0) {
235 mi_subproc_stat_increase( arena->subproc, committed, mi_size_of_slices(touched_slices));
236 }
237 }
238 // tool support
239 if (memid->initially_zero) {
240 mi_track_mem_defined(p, slice_count * MI_ARENA_SLICE_SIZE);
241 }
242 else {
243 mi_track_mem_undefined(p, slice_count * MI_ARENA_SLICE_SIZE);
244 }
245 }
246 else {
247 // no need to commit, but check if already fully committed
248 // commit requested, but the range may not be committed as a whole: ensure it is committed now
249 memid->initially_committed = mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count);
250 if (!memid->initially_committed) {
251 // partly committed.. adjust stats
252 size_t already_committed_count = 0;
253 mi_bitmap_setN(arena->slices_committed, slice_index, slice_count, &already_committed_count);
254 mi_bitmap_clearN(arena->slices_committed, slice_index, slice_count);
255 mi_os_stat_decrease(committed, mi_size_of_slices(already_committed_count));
256 }
257 }
258
259 mi_assert_internal(mi_bbitmap_is_clearN(arena->slices_free, slice_index, slice_count));
260 if (commit) { mi_assert_internal(mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count)); }
261 mi_assert_internal(mi_bitmap_is_setN(arena->slices_dirty, slice_index, slice_count));
262
263 return p;
264}

269// try to reserve a fresh arena space
270static bool mi_arena_reserve(mi_subproc_t* subproc, size_t req_size, bool allow_large, mi_arena_id_t* arena_id)
271{
272 const size_t arena_count = mi_arenas_get_count(subproc);
273 if (arena_count > (MI_MAX_ARENAS - 4)) return false;
274
275 // calc reserve
276 size_t arena_reserve = mi_option_get_size(mi_option_arena_reserve);
277 if (arena_reserve == 0) return false;
278
279 if (!_mi_os_has_virtual_reserve()) {
280 arena_reserve = arena_reserve/4; // be conservative if virtual reserve is not supported (for WASM for example)
281 }
282 arena_reserve = _mi_align_up(arena_reserve, MI_ARENA_SLICE_SIZE);
283
284 if (arena_count >= 1 && arena_count <= 128) {
285 // scale up the arena sizes exponentially every 4 entries
286 const size_t multiplier = (size_t)1 << _mi_clamp(arena_count/4, 0, 16);
287 size_t reserve = 0;
288 if (!mi_mul_overflow(multiplier, arena_reserve, &reserve)) {
289 arena_reserve = reserve;
290 }
291 }
292
293 // check arena bounds
294 const size_t min_reserve = MI_ARENA_MIN_SIZE;
295 const size_t max_reserve = MI_ARENA_MAX_SIZE; // 16 GiB
296 if (arena_reserve < min_reserve) {
297 arena_reserve = min_reserve;
298 }
299 else if (arena_reserve > max_reserve) {
300 arena_reserve = max_reserve;
301 }
302
303 if (arena_reserve < req_size) return false; // should be able to at least handle the current allocation size
304
305 // commit eagerly?
306 bool arena_commit = false;
307 const bool overcommit = _mi_os_has_overcommit();
308 if (mi_option_get(mi_option_arena_eager_commit) == 2) { arena_commit = overcommit; }
309 else if (mi_option_get(mi_option_arena_eager_commit) == 1) { arena_commit = true; }
310
311 // on an OS with overcommit (Linux) we don't count the commit yet as it is on-demand. Once a slice
312 // is actually allocated for the first time it will be counted.
313 const bool adjust = (overcommit && arena_commit);
314 if (adjust) { mi_subproc_stat_adjust_decrease( subproc, committed, arena_reserve); }
315 // and try to reserve the arena
316 int err = mi_reserve_os_memory_ex2(subproc, arena_reserve, arena_commit, allow_large, false /* exclusive? */, arena_id);
317 if (err != 0) {
318 if (adjust) { mi_subproc_stat_adjust_increase( subproc, committed, arena_reserve); } // roll back
319 // failed, try a smaller size?
320 const size_t small_arena_reserve = (MI_SIZE_BITS == 32 ? 128*MI_MiB : 1*MI_GiB);
321 if (adjust) { mi_subproc_stat_adjust_decrease( subproc, committed, arena_reserve); }
322 if (arena_reserve > small_arena_reserve) {
323 // try again
324 err = mi_reserve_os_memory_ex(small_arena_reserve, arena_commit, allow_large, false /* exclusive? */, arena_id);
325 if (err != 0 && adjust) { mi_subproc_stat_adjust_increase( subproc, committed, arena_reserve); } // roll back
326 }
327 }
328 return (err==0);
329}

338static inline bool mi_arena_is_suitable(mi_arena_t* arena, mi_arena_t* req_arena, bool match_numa, int numa_node, bool allow_pinned) {
339 if (!allow_pinned && arena->memid.is_pinned) return false;
340 if (!mi_arena_id_is_suitable(arena, req_arena)) return false;
341 if (req_arena == NULL) { // if not specific, check numa affinity
342 const bool numa_suitable = (numa_node < 0 || arena->numa_node < 0 || arena->numa_node == numa_node);
343 if (match_numa) { if (!numa_suitable) return false; }
344 else { if (numa_suitable) return false; }
345 }
346 return true;
347}

350 const size_t _arena_count = mi_arenas_get_count(subproc); \
351 const size_t _arena_cycle = (_arena_count == 0 ? 0 : _arena_count - 1); /* first search the arenas below the last one */ \
352 /* always start searching in the arena's below the max */ \
353 size_t _start = (_arena_cycle <= 1 ? 0 : (tseq % _arena_cycle)); \
354 for (size_t _i = 0; _i < _arena_count; _i++) { \
355 mi_arena_t* name_arena; \
356 if (req_arena != NULL) { \
357 name_arena = req_arena; /* if there is a specific req_arena, only search that one */\
358 if (_i > 0) break; /* only once */ \
359 } \
360 else { \
361 size_t _idx; \
362 if (_i < _arena_cycle) { \
363 _idx = _i + _start; \
364 if (_idx >= _arena_cycle) { _idx -= _arena_cycle; } /* adjust so we rotate through the cycle */ \
365 } \
366 else { \
367 _idx = _i; /* remaining arena's */ \
368 } \
369 name_arena = mi_arena_from_index(subproc,_idx); \
370 } \
371 if (name_arena != NULL) \
372 {
373
374#define mi_forall_arenas_end() \
375 } \
376 } \
377 }

391// allocate slices from the arenas
392static mi_decl_noinline void* mi_arenas_try_find_free(
393 mi_subproc_t* subproc, size_t slice_count, size_t alignment,
394 bool commit, bool allow_large, mi_arena_t* req_arena, size_t tseq, int numa_node, mi_memid_t* memid)
395{
396 mi_assert_internal(slice_count <= mi_slice_count_of_size(MI_ARENA_MAX_OBJ_SIZE));
397 mi_assert(alignment <= MI_ARENA_SLICE_ALIGN);
398 if (alignment > MI_ARENA_SLICE_ALIGN) return NULL;
399
400 // search arena's
401 mi_forall_suitable_arenas(subproc, req_arena, tseq, true /* only numa matching */, numa_node, allow_large, arena)
402 {
403 void* p = mi_arena_try_alloc_at(arena, slice_count, commit, tseq, memid);
404 if (p != NULL) return p;
405 }
406 mi_forall_suitable_arenas_end();
407 if (numa_node < 0) return NULL;
408
409 // search again but now regardless of preferred numa affinity
410 mi_forall_suitable_arenas(subproc, req_arena, tseq, false /* numa non-matching now */, numa_node, allow_large, arena)
411 {
412 void* p = mi_arena_try_alloc_at(arena, slice_count, commit, tseq, memid);
413 if (p != NULL) return p;
414 }
415 mi_forall_suitable_arenas_end();
416 return NULL;
417}

419// Allocate slices from the arena's -- potentially allocating a fresh arena
420static mi_decl_noinline void* mi_arenas_try_alloc(
421 mi_subproc_t* subproc,
422 size_t slice_count, size_t alignment,
423 bool commit, bool allow_large,
424 mi_arena_t* req_arena, size_t tseq, int numa_node, mi_memid_t* memid)
425{
426 mi_assert(slice_count <= MI_ARENA_MAX_OBJ_SLICES);
427 mi_assert(alignment <= MI_ARENA_SLICE_ALIGN);
428 void* p;
429
430 // try to find free slices in the arena's
431 p = mi_arenas_try_find_free(subproc, slice_count, alignment, commit, allow_large, req_arena, tseq, numa_node, memid);
432 if (p != NULL) return p;
433
434 // did we need a specific arena?
435 if (req_arena != NULL) return NULL;
436
437 // don't create arena's while preloading (todo: or should we?)
438 if (_mi_preloading()) return NULL;
439
440 // otherwise, try to reserve a new arena -- but one thread at a time.. (todo: allow 2 or 4 to reduce contention?)
441 const size_t arena_count = mi_arenas_get_count(subproc);
442 mi_lock(&subproc->arena_reserve_lock) {
443 if (arena_count == mi_arenas_get_count(subproc)) {
444 // we are the first to enter the lock, reserve a fresh arena
445 mi_arena_id_t arena_id = 0;
446 mi_arena_reserve(subproc, mi_size_of_slices(slice_count), allow_large, &arena_id);
447 }
448 else {
449 // another thread already reserved a new arena
450 }
451 }
452 // try once more to allocate in the new arena
453 mi_assert_internal(req_arena == NULL);
454 p = mi_arenas_try_find_free(subproc, slice_count, alignment, commit, allow_large, req_arena, tseq, numa_node, memid);
455 if (p != NULL) return p;
456
457 return NULL;
458}

481// Allocate large sized memory
482void* _mi_arenas_alloc_aligned( mi_subproc_t* subproc,
483 size_t size, size_t alignment, size_t align_offset,
484 bool commit, bool allow_large,
485 mi_arena_t* req_arena, size_t tseq, int numa_node, mi_memid_t* memid)
486{
487 mi_assert_internal(memid != NULL);
488 mi_assert_internal(size > 0);
489
490 // try to allocate in an arena if the alignment is small enough and the object is not too small (as for heap meta data)
491 if (!mi_option_is_enabled(mi_option_disallow_arena_alloc) && // is arena allocation allowed?
492 size >= MI_ARENA_MIN_OBJ_SIZE && size <= MI_ARENA_MAX_OBJ_SIZE && // and not too small/large
493 alignment <= MI_ARENA_SLICE_ALIGN && align_offset == 0) // and good alignment
494 {
495 const size_t slice_count = mi_slice_count_of_size(size);
496 void* p = mi_arenas_try_alloc(subproc,slice_count, alignment, commit, allow_large, req_arena, tseq, numa_node, memid);
497 if (p != NULL) return p;
498 }
499
500 // fall back to the OS
501 void* p = mi_arena_os_alloc_aligned(size, alignment, align_offset, commit, allow_large, req_arena, memid);
502 return p;
503}

516static bool mi_arena_try_claim_abandoned(size_t slice_index, mi_arena_t* arena, mi_heaptag_t heap_tag, bool* keep_abandoned) {
517 // found an abandoned page of the right size
518 mi_page_t* const page = (mi_page_t*)mi_arena_slice_start(arena, slice_index);
519 // can we claim ownership?
520 if (!mi_page_try_claim_ownership(page)) {
521 // there was a concurrent free ..
522 // we need to keep it in the abandoned map as the free will call `mi_arena_page_unabandon`,
523 // and wait for readers (us!) to finish. This is why it is very important to set the abandoned
524 // bit again (or otherwise the unabandon will never stop waiting).
525 *keep_abandoned = true;
526 return false;
527 }
528 if (heap_tag != page->heap_tag) {
529 // wrong heap_tag.. we need to unown again
530 // note: this normally never happens unless heaptags are actually used.
531 // (an unown might free the page, and depending on that we can keep it in the abandoned map or not)
532 // note: a minor wrinkle: the page will still be mapped but the abandoned map entry is (temporarily) clear at this point.
533 // so we cannot check in `mi_arenas_free` for this invariant to hold.
534 const bool freed = _mi_page_unown(page);
535 *keep_abandoned = !freed;
536 return false;
537 }
538 // yes, we can reclaim it, keep the abandoned map entry clear
539 *keep_abandoned = false;
540 return true;
541}
542
543static mi_page_t* mi_arenas_page_try_find_abandoned(mi_subproc_t* subproc, size_t slice_count, size_t block_size, mi_arena_t* req_arena, mi_heaptag_t heaptag, size_t tseq)
544{
545 MI_UNUSED(slice_count);
546 const size_t bin = _mi_bin(block_size);
547 mi_assert_internal(bin < MI_BIN_COUNT);
548
549 // any abandoned in our size class?
550 mi_assert_internal(subproc != NULL);
551 if (mi_atomic_load_relaxed(&subproc->abandoned_count[bin]) == 0) {
552 return NULL;
553 }
554
555 // search arena's
556 const bool allow_large = true;
557 const int any_numa = -1;
558 const bool match_numa = true;
559 mi_forall_suitable_arenas(subproc, req_arena, tseq, match_numa, any_numa, allow_large, arena)
560 {
561 size_t slice_index;
562 mi_bitmap_t* const bitmap = arena->pages_abandoned[bin];
563
564 if (mi_bitmap_try_find_and_claim(bitmap, tseq, &slice_index, &mi_arena_try_claim_abandoned, arena, heaptag)) {
565 // found an abandoned page of the right size
566 // and claimed ownership.
567 mi_page_t* page = (mi_page_t*)mi_arena_slice_start(arena, slice_index);
568 mi_assert_internal(mi_page_is_owned(page));
569 mi_assert_internal(mi_page_is_abandoned(page));
570 mi_assert_internal(mi_arena_has_page(arena,page));
571 mi_atomic_decrement_relaxed(&subproc->abandoned_count[bin]);
572 mi_tld_t* tld = _mi_thread_tld();
573 mi_tld_stat_decrease( tld, pages_abandoned, 1);
574 mi_tld_stat_counter_increase( tld, pages_reclaim_on_alloc, 1);
575
576 _mi_page_free_collect(page, false); // update `used` count
577 mi_assert_internal(mi_bbitmap_is_clearN(arena->slices_free, slice_index, slice_count));
578 mi_assert_internal(page->slice_committed > 0 || mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count));
579 mi_assert_internal(mi_bitmap_is_setN(arena->slices_dirty, slice_index, slice_count));
580 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
581 mi_assert_internal(_mi_ptr_page(page)==page);
582 mi_assert_internal(_mi_ptr_page(mi_page_start(page))==page);
583 mi_assert_internal(mi_page_block_size(page) == block_size);
584 mi_assert_internal(!mi_page_is_full(page));
585 return page;
586 }
587 }
588 mi_forall_suitable_arenas_end();
589 return NULL;
590}

592// Allocate a fresh page
593static mi_page_t* mi_arenas_page_alloc_fresh(mi_subproc_t* subproc, size_t slice_count, size_t block_size, size_t block_alignment,
594 mi_arena_t* req_arena, size_t tseq, int numa_node, bool commit)
595{
596 const bool allow_large = (MI_SECURE < 2); // 2 = guard page at end of each arena page
597 const bool os_align = (block_alignment > MI_PAGE_MAX_OVERALLOC_ALIGN);
598 const size_t page_alignment = MI_ARENA_SLICE_ALIGN;
599
600 // try to allocate from free space in arena's
601 mi_memid_t memid = _mi_memid_none();
602 mi_page_t* page = NULL;
603 const size_t alloc_size = mi_size_of_slices(slice_count);
604 if (!mi_option_is_enabled(mi_option_disallow_arena_alloc) && // allowed to allocate from arena's?
605 !os_align && // not large alignment
606 slice_count <= MI_ARENA_MAX_OBJ_SLICES) // and not too large
607 {
608 page = (mi_page_t*)mi_arenas_try_alloc(subproc, slice_count, page_alignment, commit, allow_large, req_arena, tseq, numa_node, &memid);
609 if (page != NULL) {
610 mi_assert_internal(mi_bitmap_is_clearN(memid.mem.arena.arena->pages, memid.mem.arena.slice_index, memid.mem.arena.slice_count));
611 mi_bitmap_set(memid.mem.arena.arena->pages, memid.mem.arena.slice_index);
612 }
613 }
614
615 // otherwise fall back to the OS
616 if (page == NULL) {
617 if (os_align) {
618 // note: slice_count already includes the page
619 mi_assert_internal(slice_count >= mi_slice_count_of_size(block_size) + mi_slice_count_of_size(page_alignment));
620 page = (mi_page_t*)mi_arena_os_alloc_aligned(alloc_size, block_alignment, page_alignment /* align offset */, commit, allow_large, req_arena, &memid);
621 }
622 else {
623 page = (mi_page_t*)mi_arena_os_alloc_aligned(alloc_size, page_alignment, 0 /* align offset */, commit, allow_large, req_arena, &memid);
624 }
625 }
626
627 if (page == NULL) return NULL;
628 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
629 mi_assert_internal(!os_align || _mi_is_aligned((uint8_t*)page + page_alignment, block_alignment));
630
631 // guard page at the end of mimalloc page?
632 #if MI_SECURE < 2
633 const size_t page_noguard_size = alloc_size;
634 #else
635 mi_assert(alloc_size > _mi_os_secure_guard_page_size());
636 const size_t page_noguard_size = alloc_size - _mi_os_secure_guard_page_size();
637 if (memid.initially_committed) {
638 _mi_os_secure_guard_page_set_at((uint8_t*)page + page_noguard_size, memid.is_pinned);
639 }
640 #endif
641
642 // claimed free slices: initialize the page partly
643 if (!memid.initially_zero && memid.initially_committed) {
644 mi_track_mem_undefined(page, slice_count * MI_ARENA_SLICE_SIZE);
645 _mi_memzero_aligned(page, sizeof(*page));
646 }
647 else if (memid.initially_committed) {
648 mi_track_mem_defined(page, slice_count * MI_ARENA_SLICE_SIZE);
649 }
650 #if MI_DEBUG > 1
651 if (memid.initially_zero && memid.initially_committed) {
652 if (!mi_mem_is_zero(page, page_noguard_size)) {
653 _mi_error_message(EFAULT, "internal error: page memory was not zero initialized.\n");
654 memid.initially_zero = false;
655 _mi_memzero_aligned(page, sizeof(*page));
656 }
657 }
658 #endif
659 mi_assert(MI_PAGE_INFO_SIZE >= mi_page_info_size());
660
661 size_t block_start;
662 #if MI_GUARDED
663 // in a guarded build, we align pages with blocks a multiple of an OS page size, to the OS page size
664 // this ensures that all blocks in such pages are OS page size aligned (which is needed for the guard pages)
665 const size_t os_page_size = _mi_os_page_size();
666 mi_assert_internal(MI_PAGE_ALIGN >= os_page_size);
667 if (!os_align && block_size % os_page_size == 0 && block_size > os_page_size /* at least 2 or more */ ) {
668 block_start = _mi_align_up(mi_page_info_size(), os_page_size);
669 }
670 else
671 #endif
672 if (os_align) {
673 block_start = MI_PAGE_ALIGN;
674 }
675 else if (_mi_is_power_of_two(block_size) && block_size <= MI_PAGE_MAX_START_BLOCK_ALIGN2) {
676 // naturally align all power-of-2 blocks
677 block_start = _mi_align_up(mi_page_info_size(), block_size);
678 }
679 else {
680 // otherwise start after the info
681 block_start = mi_page_info_size();
682 }
683 const size_t reserved = (os_align ? 1 : (page_noguard_size - block_start) / block_size);
684 mi_assert_internal(reserved > 0 && reserved <= UINT16_MAX);
685
686 // commit first block?
687 size_t commit_size = 0;
688 if (!memid.initially_committed) {
689 commit_size = _mi_align_up(block_start + block_size, MI_PAGE_MIN_COMMIT_SIZE);
690 if (commit_size > page_noguard_size) { commit_size = page_noguard_size; }
691 bool is_zero;
692 _mi_os_commit(page, commit_size, &is_zero);
693 if (!memid.initially_zero && !is_zero) {
694 _mi_memzero_aligned(page, commit_size);
695 }
696 }
697
698 // initialize
699 page->reserved = (uint16_t)reserved;
700 page->page_start = (uint8_t*)page + block_start;
701 page->block_size = block_size;
702 page->slice_committed = commit_size;
703 page->memid = memid;
704 page->free_is_zero = memid.initially_zero;
705 if (block_size > 0 && _mi_is_power_of_two(block_size)) {
706 page->block_size_shift = (uint8_t)mi_ctz(block_size);
707 }
708 else {
709 page->block_size_shift = 0;
710 }
711 // and own it
712 mi_page_try_claim_ownership(page);
713
714 // register in the page map
715 _mi_page_map_register(page);
716 mi_assert_internal(_mi_ptr_page(page)==page);
717 mi_assert_internal(_mi_ptr_page(mi_page_start(page))==page);
718 mi_assert_internal(mi_page_block_size(page) == block_size);
719 mi_assert_internal(mi_page_is_abandoned(page));
720 mi_assert_internal(mi_page_is_owned(page));
721 return page;
722}

724// Allocate a regular small/medium/large page.
725static mi_page_t* mi_arenas_page_regular_alloc(mi_heap_t* heap, size_t slice_count, size_t block_size) {
726 mi_arena_t* req_arena = heap->exclusive_arena;
727 mi_tld_t* const tld = heap->tld;
728
729 // 1. look for an abandoned page
730 mi_page_t* page = mi_arenas_page_try_find_abandoned(tld->subproc, slice_count, block_size, req_arena, heap->tag, tld->thread_seq);
731 if (page != NULL) {
732 return page; // return as abandoned
733 }
734
735 // 2. find a free block, potentially allocating a new arena
736 const long commit_on_demand = mi_option_get(mi_option_page_commit_on_demand);
737 const bool commit = (slice_count <= mi_slice_count_of_size(MI_PAGE_MIN_COMMIT_SIZE) || // always commit small pages
738 (commit_on_demand == 2 && _mi_os_has_overcommit()) || (commit_on_demand == 0));
739 page = mi_arenas_page_alloc_fresh(tld->subproc, slice_count, block_size, 1, req_arena, tld->thread_seq, heap->numa_node, commit);
740 if (page != NULL) {
741 mi_assert_internal(page->memid.memkind != MI_MEM_ARENA || page->memid.mem.arena.slice_count == slice_count);
742 _mi_page_init(heap, page);
743 return page;
744 }
745
746 return NULL;
747}

749// Allocate a page containing one block (very large, or with large alignment)
750static mi_page_t* mi_arenas_page_singleton_alloc(mi_heap_t* heap, size_t block_size, size_t block_alignment) {
751 mi_arena_t* req_arena = heap->exclusive_arena;
752 mi_tld_t* const tld = heap->tld;
753 const bool os_align = (block_alignment > MI_PAGE_MAX_OVERALLOC_ALIGN);
754 const size_t info_size = (os_align ? MI_PAGE_ALIGN : mi_page_info_size());
755 #if MI_SECURE < 2
756 const size_t slice_count = mi_slice_count_of_size(info_size + block_size);
757 #else
758 const size_t slice_count = mi_slice_count_of_size(_mi_align_up(info_size + block_size, _mi_os_secure_guard_page_size()) + _mi_os_secure_guard_page_size());
759 #endif
760
761 mi_page_t* page = mi_arenas_page_alloc_fresh(tld->subproc, slice_count, block_size, block_alignment, req_arena, tld->thread_seq, heap->numa_node, true /* commit singletons always */);
762 if (page == NULL) return NULL;
763
764 mi_assert(page->reserved == 1);
765 _mi_page_init(heap, page);
766
767 return page;
768}

786 }

789 page = mi_arenas_page_singleton_alloc(heap, block_size, block_alignment);
790 }
791 // mi_assert_internal(page == NULL || _mi_page_segment(page)->subproc == tld->subproc);
792 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));

800void _mi_arenas_page_free(mi_page_t* page) {
801 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
802 mi_assert_internal(_mi_ptr_page(page)==page);
803 mi_assert_internal(mi_page_is_owned(page));
804 mi_assert_internal(mi_page_all_free(page));
805 mi_assert_internal(mi_page_is_abandoned(page));
806 mi_assert_internal(page->next==NULL && page->prev==NULL);
807
808 #if MI_DEBUG>1
809 if (page->memid.memkind==MI_MEM_ARENA && !mi_page_is_full(page)) {
810 size_t bin = _mi_bin(mi_page_block_size(page));
811 size_t slice_index;
812 size_t slice_count;
813 mi_arena_t* arena = mi_page_arena(page, &slice_index, &slice_count);
814
815 mi_assert_internal(mi_bbitmap_is_clearN(arena->slices_free, slice_index, slice_count));
816 mi_assert_internal(page->slice_committed > 0 || mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count));
817 mi_assert_internal(mi_bitmap_is_clearN(arena->pages_abandoned[bin], slice_index, 1));
818 mi_assert_internal(mi_bitmap_is_setN(page->memid.mem.arena.arena->pages, page->memid.mem.arena.slice_index, 1));
819 // note: we cannot check for `!mi_page_is_abandoned_and_mapped` since that may
820 // be (temporarily) not true if the free happens while trying to reclaim
821 // see `mi_arana_try_claim_abandoned`
822 }
823 #endif
824
825 // recommit guard page at the end?
826 // we must do this since we may later allocate large spans over this page and cannot have a guard page in between
827 #if MI_SECURE >= 2
828 if (!page->memid.is_pinned) {
829 _mi_os_secure_guard_page_reset_before((uint8_t*)page + mi_page_full_size(page));
830 }
831 #endif
832
833 // unregister page
834 _mi_page_map_unregister(page);
835 if (page->memid.memkind == MI_MEM_ARENA) {
836 mi_arena_t* arena = page->memid.mem.arena.arena;
837 mi_bitmap_clear(arena->pages, page->memid.mem.arena.slice_index);
838 if (page->slice_committed > 0) {
839 // if committed on-demand, set the commit bits to account commit properly
840 mi_assert_internal(mi_page_full_size(page) >= page->slice_committed);
841 const size_t total_slices = page->slice_committed / MI_ARENA_SLICE_SIZE; // conservative
842 //mi_assert_internal(mi_bitmap_is_clearN(arena->slices_committed, page->memid.mem.arena.slice_index, total_slices));
843 mi_assert_internal(page->memid.mem.arena.slice_count >= total_slices);
844 if (total_slices > 0) {
845 mi_bitmap_setN(arena->slices_committed, page->memid.mem.arena.slice_index, total_slices, NULL);
846 }
847 // any left over?
848 const size_t extra = page->slice_committed % MI_ARENA_SLICE_SIZE;
849 if (extra > 0) {
850 // pretend it was decommitted already
851 mi_os_stat_decrease(committed, extra);
852 }
853 }
854 else {
855 mi_assert_internal(mi_bitmap_is_setN(arena->slices_committed, page->memid.mem.arena.slice_index, page->memid.mem.arena.slice_count));
856 }
857 }
858 _mi_arenas_free(page, mi_page_full_size(page), page->memid);
859}

864
865void _mi_arenas_page_abandon(mi_page_t* page, mi_tld_t* tld) {
866 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
867 mi_assert_internal(_mi_ptr_page(page)==page);
868 mi_assert_internal(mi_page_is_owned(page));
869 mi_assert_internal(mi_page_is_abandoned(page));
870 mi_assert_internal(!mi_page_all_free(page));
871 mi_assert_internal(page->next==NULL && page->prev == NULL);
872
873 if (page->memid.memkind==MI_MEM_ARENA && !mi_page_is_full(page)) {
874 // make available for allocations
875 size_t bin = _mi_bin(mi_page_block_size(page));
876 size_t slice_index;
877 size_t slice_count;
878 mi_arena_t* arena = mi_page_arena(page, &slice_index, &slice_count);
879 mi_assert_internal(!mi_page_is_singleton(page));
880 mi_assert_internal(mi_bbitmap_is_clearN(arena->slices_free, slice_index, slice_count));
881 mi_assert_internal(page->slice_committed > 0 || mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count));
882 mi_assert_internal(mi_bitmap_is_setN(arena->slices_dirty, slice_index, slice_count));
883
884 mi_page_set_abandoned_mapped(page);
885 const bool wasclear = mi_bitmap_set(arena->pages_abandoned[bin], slice_index);
886 MI_UNUSED(wasclear); mi_assert_internal(wasclear);
887 mi_atomic_increment_relaxed(&arena->subproc->abandoned_count[bin]);
888 mi_tld_stat_increase(tld, pages_abandoned, 1);
889 }
890 else {
891 // page is full (or a singleton), or the page is OS/externally allocated
892 // leave as is; it will be reclaimed when an object is free'd in the page
893 mi_subproc_t* subproc = _mi_subproc();
894 // but for non-arena pages, add to the subproc list so these can be visited
895 if (page->memid.memkind != MI_MEM_ARENA && mi_option_is_enabled(mi_option_visit_abandoned)) {
896 mi_lock(&subproc->os_abandoned_pages_lock) {
897 // push in front
898 page->prev = NULL;
899 page->next = subproc->os_abandoned_pages;
900 if (page->next != NULL) { page->next->prev = page; }
901 subproc->os_abandoned_pages = page;
902 }
903 }
904 mi_tld_stat_increase(tld, pages_abandoned, 1);
905 }
906 _mi_page_unown(page);
907}

909bool _mi_arenas_page_try_reabandon_to_mapped(mi_page_t* page) {
910 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
911 mi_assert_internal(_mi_ptr_page(page)==page);
912 mi_assert_internal(mi_page_is_owned(page));
913 mi_assert_internal(mi_page_is_abandoned(page));
914 mi_assert_internal(!mi_page_is_abandoned_mapped(page));
915 mi_assert_internal(!mi_page_is_full(page));
916 mi_assert_internal(!mi_page_all_free(page));
917 mi_assert_internal(!mi_page_is_singleton(page));
918 if (mi_page_is_full(page) || mi_page_is_abandoned_mapped(page) || page->memid.memkind != MI_MEM_ARENA) {

930// called from `mi_free` if trying to unabandon an abandoned page
931void _mi_arenas_page_unabandon(mi_page_t* page) {
932 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
933 mi_assert_internal(_mi_ptr_page(page)==page);
934 mi_assert_internal(mi_page_is_owned(page));
935 mi_assert_internal(mi_page_is_abandoned(page));
936
937 if (mi_page_is_abandoned_mapped(page)) {
938 mi_assert_internal(page->memid.memkind==MI_MEM_ARENA);
939 // remove from the abandoned map
940 size_t bin = _mi_bin(mi_page_block_size(page));
941 size_t slice_index;
942 size_t slice_count;
943 mi_arena_t* arena = mi_page_arena(page, &slice_index, &slice_count);
944
945 mi_assert_internal(mi_bbitmap_is_clearN(arena->slices_free, slice_index, slice_count));
946 mi_assert_internal(page->slice_committed > 0 || mi_bitmap_is_setN(arena->slices_committed, slice_index, slice_count));
947
948 // this busy waits until a concurrent reader (from alloc_abandoned) is done
949 mi_bitmap_clear_once_set(arena->pages_abandoned[bin], slice_index);
950 mi_page_clear_abandoned_mapped(page);
951 mi_atomic_decrement_relaxed(&arena->subproc->abandoned_count[bin]);
952 mi_tld_stat_decrease(_mi_thread_tld(), pages_abandoned, 1);
953 }
954 else {
955 // page is full (or a singleton), page is OS allocated
956 mi_tld_stat_decrease(_mi_thread_tld(), pages_abandoned, 1);
957 // if not an arena page, remove from the subproc os pages list
958 if (page->memid.memkind != MI_MEM_ARENA && mi_option_is_enabled(mi_option_visit_abandoned)) {
959 mi_subproc_t* subproc = _mi_subproc();
960 mi_lock(&subproc->os_abandoned_pages_lock) {
961 if (page->prev != NULL) { page->prev->next = page->next; }
962 if (page->next != NULL) { page->next->prev = page->prev; }
963 if (subproc->os_abandoned_pages == page) { subproc->os_abandoned_pages = page->next; }
964 page->next = NULL;
965 page->prev = NULL;
966 }
967 }
968 }

978void _mi_arenas_free(void* p, size_t size, mi_memid_t memid) {
979 if (p==NULL) return;
980 if (size==0) return;
981
982 // need to set all memory to undefined as some parts may still be marked as no_access (like padding etc.)
983 mi_track_mem_undefined(p, size);
984
985 if (mi_memkind_is_os(memid.memkind)) {
986 // was a direct OS allocation, pass through
987 _mi_os_free(p, size, memid);
988 }
989 else if (memid.memkind == MI_MEM_ARENA) {
990 // allocated in an arena
991 size_t slice_count;
992 size_t slice_index;
993 mi_arena_t* arena = mi_arena_from_memid(memid, &slice_index, &slice_count);
994 mi_assert_internal((size%MI_ARENA_SLICE_SIZE)==0);
995 mi_assert_internal((slice_count*MI_ARENA_SLICE_SIZE)==size);
996 mi_assert_internal(mi_arena_slice_start(arena,slice_index) <= (uint8_t*)p);
997 mi_assert_internal(mi_arena_slice_start(arena,slice_index) + mi_size_of_slices(slice_count) > (uint8_t*)p);
998 // checks
999 if (arena == NULL) {
1000 _mi_error_message(EINVAL, "trying to free from an invalid arena: %p, size %zu, memid: 0x%zx\n", p, size, memid);
1001 return;
1002 }
1003 mi_assert_internal(slice_index < arena->slice_count);
1004 mi_assert_internal(slice_index >= mi_arena_info_slices(arena));
1005 if (slice_index < mi_arena_info_slices(arena) || slice_index > arena->slice_count) {
1006 _mi_error_message(EINVAL, "trying to free from an invalid arena block: %p, size %zu, memid: 0x%zx\n", p, size, memid);
1007 return;
1008 }
1009
1010 // potentially decommit
1011 if (!arena->memid.is_pinned /* && !arena->memid.initially_committed */) { // todo: allow decommit even if initially committed?
1012 // (delay) purge the page
1013 mi_arena_schedule_purge(arena, slice_index, slice_count);
1014 }
1015
1016 // and make it available to others again
1017 bool all_inuse = mi_bbitmap_setN(arena->slices_free, slice_index, slice_count);
1018 if (!all_inuse) {
1019 _mi_error_message(EAGAIN, "trying to free an already freed arena block: %p, size %zu\n", mi_arena_slice_start(arena,slice_index), mi_size_of_slices(slice_count));
1020 return;
1021 };
1022 }
1023 else if (memid.memkind == MI_MEM_META) {
1024 _mi_meta_free(p, size, memid);
1025 }
1026 else {
1027 // arena was none, external, or static; nothing to do
1028 mi_assert_internal(mi_memid_needs_no_free(memid));
1029 }
1030
1031 // try to purge expired decommits
1032 // mi_arenas_try_purge(false, false, NULL);
1033}

1113 // success
1114 if (arena_id != NULL) { *arena_id = arena; }
1115 return true;
1116 }
1117 }
1118 }
1119
1120 // otherwise increase the max
1121 i = mi_atomic_increment_acq_rel(&subproc->arena_count);
1122 if (i >= MI_MAX_ARENAS) {
1123 mi_atomic_decrement_acq_rel(&subproc->arena_count);
1124 arena->subproc = NULL;
1125 return false;
1126 }
1127
1128 mi_subproc_stat_counter_increase(arena->subproc, arena_count, 1);
1129 mi_atomic_store_ptr_release(mi_arena_t,&subproc->arenas[i], arena);
1130 if (arena_id != NULL) { *arena_id = arena; }
1131 return true;
1132}

1163static bool mi_manage_os_memory_ex2(mi_subproc_t* subproc, void* start, size_t size, int numa_node, bool exclusive, mi_memid_t memid, mi_arena_id_t* arena_id) mi_attr_noexcept
1164{
1165 mi_assert(_mi_is_aligned(start,MI_ARENA_SLICE_SIZE));
1166 mi_assert(start!=NULL);
1167 if (arena_id != NULL) { *arena_id = _mi_arena_id_none(); }
1168 if (start==NULL) return false;
1169 if (!_mi_is_aligned(start,MI_ARENA_SLICE_SIZE)) {
1170 // we can align the start since the memid tracks the real base of the memory.
1171 void* const aligned_start = _mi_align_up_ptr(start, MI_ARENA_SLICE_SIZE);
1172 const size_t diff = (uint8_t*)aligned_start - (uint8_t*)start;
1173 if (diff >= size || (size - diff) < MI_ARENA_SLICE_SIZE) {
1174 _mi_warning_message("after alignment, the size of the arena becomes too small (memory at %p with size %zu)\n", start, size);
1175 return false;
1176 }
1177 start = aligned_start;
1178 size = size - diff;
1179 }
1180
1181 const size_t slice_count = _mi_align_down(size / MI_ARENA_SLICE_SIZE, MI_BCHUNK_BITS);
1182 if (slice_count > MI_BITMAP_MAX_BIT_COUNT) { // 16 GiB for now
1183 // todo: allow larger areas (either by splitting it up in arena's or having larger arena's)
1184 _mi_warning_message("cannot use OS memory since it is too large (size %zu MiB, maximum is %zu MiB)", size/MI_MiB, mi_size_of_slices(MI_BITMAP_MAX_BIT_COUNT)/MI_MiB);
1185 return false;
1186 }
1187 size_t bitmap_base;
1188 const size_t info_slices = mi_arena_info_slices_needed(slice_count, &bitmap_base);
1189 if (slice_count < info_slices+1) {
1190 _mi_warning_message("cannot use OS memory since it is not large enough (size %zu KiB, minimum required is %zu KiB)", size/MI_KiB, mi_size_of_slices(info_slices+1)/MI_KiB);
1191 return false;
1192 }
1193
1194 mi_arena_t* arena = (mi_arena_t*)start;
1195
1196 // commit & zero if needed
1197 if (!memid.initially_committed) {
1198 // leave a guard OS page decommitted at the end
1199 _mi_os_commit(arena, mi_size_of_slices(info_slices) - _mi_os_secure_guard_page_size(), NULL);
1200 }
1201 else {
1202 // if MI_SECURE, set a guard page at the end
1203 _mi_os_secure_guard_page_set_before((uint8_t*)arena + mi_size_of_slices(info_slices), memid.is_pinned);
1204 }
1205 if (!memid.initially_zero) {
1206 _mi_memzero(arena, mi_size_of_slices(info_slices) - _mi_os_secure_guard_page_size());
1207 }
1208
1209 // init
1210 arena->subproc = subproc;
1211 arena->memid = memid;
1212 arena->is_exclusive = exclusive;
1213 arena->slice_count = slice_count;
1214 arena->info_slices = info_slices;
1215 arena->numa_node = numa_node; // TODO: or get the current numa node if -1? (now it allows anyone to allocate on -1)
1216 arena->purge_expire = 0;
1217 // mi_lock_init(&arena->abandoned_visit_lock);
1218
1219 // init bitmaps
1220 uint8_t* base = mi_arena_start(arena) + bitmap_base;
1221 arena->slices_free = mi_arena_bbitmap_init(slice_count,&base);
1222 arena->slices_committed = mi_arena_bitmap_init(slice_count,&base);
1223 arena->slices_dirty = mi_arena_bitmap_init(slice_count,&base);
1224 arena->slices_purge = mi_arena_bitmap_init(slice_count, &base);
1225 arena->pages = mi_arena_bitmap_init(slice_count, &base);
1226 for( size_t i = 0; i < MI_ARENA_BIN_COUNT; i++) {
1227 arena->pages_abandoned[i] = mi_arena_bitmap_init(slice_count,&base);
1228 }
1229 mi_assert_internal(mi_size_of_slices(info_slices) >= (size_t)(base - mi_arena_start(arena)));
1230
1231 // reserve our meta info (and reserve slices outside the memory area)
1232 mi_bbitmap_unsafe_setN(arena->slices_free, info_slices /* start */, arena->slice_count - info_slices);
1233 if (memid.initially_committed) {
1234 mi_bitmap_unsafe_setN(arena->slices_committed, 0, arena->slice_count);
1235 }
1236 else {
1237 mi_bitmap_setN(arena->slices_committed, 0, info_slices, NULL);
1238 }
1239 if (!memid.initially_zero) {
1240 mi_bitmap_unsafe_setN(arena->slices_dirty, 0, arena->slice_count);
1241 }
1242 else {
1243 mi_bitmap_setN(arena->slices_dirty, 0, info_slices, NULL);
1244 }
1245
1246 return mi_arenas_add(subproc, arena, arena_id);
1247}

1605// Schedule a purge. This is usually delayed to avoid repeated decommit/commit calls.
1606// Note: assumes we (still) own the area as we may purge immediately
1607static void mi_arena_schedule_purge(mi_arena_t* arena, size_t slice_index, size_t slice_count) {
1608 const long delay = mi_arena_purge_delay();
1609 if (arena->memid.is_pinned || delay < 0 || _mi_preloading()) return; // is purging allowed at all?
1610
1611 mi_assert_internal(mi_bbitmap_is_clearN(arena->slices_free, slice_index, slice_count));
1612 if (delay == 0) {
1613 // purge directly
1614 mi_arena_purge(arena, slice_index, slice_count);
1615 }
1616 else {
1617 // schedule purge
1618 const mi_msecs_t expire = _mi_clock_now() + delay;
1619 mi_msecs_t expire0 = 0;
1620 if (mi_atomic_casi64_strong_acq_rel(&arena->purge_expire, &expire0, expire)) {
1621 // expiration was not yet set
1622 // maybe set the global arenas expire as well (if it wasn't set already)
1623 mi_assert_internal(expire0==0);
1624 mi_atomic_casi64_strong_acq_rel(&arena->subproc->purge_expire, &expire0, expire);
1625 }
1626 else {
1627 // already an expiration was set
1628 }
1629 mi_bitmap_setN(arena->slices_purge, slice_index, slice_count, NULL);
1630 }

1676// returns true if anything was purged
1677static bool mi_arena_try_purge(mi_arena_t* arena, mi_msecs_t now, bool force)
1678{
1679 // check pre-conditions
1680 if (arena->memid.is_pinned) return false;
1681
1682 // expired yet?
1683 mi_msecs_t expire = mi_atomic_loadi64_relaxed(&arena->purge_expire);
1684 if (!force && (expire == 0 || expire > now)) return false;
1685
1686 // reset expire
1687 mi_atomic_store_release(&arena->purge_expire, (mi_msecs_t)0);
1688 mi_subproc_stat_counter_increase(arena->subproc, arena_purges, 1);

1691 // this also clears those ranges atomically (so any newly freed blocks will get purged next
1692 // time around)
1693 mi_purge_visit_info_t vinfo = { now, mi_arena_purge_delay(), true /*all?*/, false /*any?*/};
1694 _mi_bitmap_forall_setc_ranges(arena->slices_purge, &mi_arena_try_purge_visitor, arena, &vinfo);
1695
1696 return vinfo.any_purged;
1697}

1700static void mi_arenas_try_purge(bool force, bool visit_all, mi_tld_t* tld)
1701{
1702 // try purge can be called often so try to only run when needed
1703 const long delay = mi_arena_purge_delay();
1704 if (_mi_preloading() || delay <= 0) return; // nothing will be scheduled
1705
1706 // check if any arena needs purging?
1707 mi_subproc_t* subproc = tld->subproc;
1708 const mi_msecs_t now = _mi_clock_now();
1709 const mi_msecs_t arenas_expire = mi_atomic_load_acquire(&subproc->purge_expire);
1710 if (!visit_all && !force && (arenas_expire == 0 || arenas_expire > now)) return;
1711
1712 const size_t max_arena = mi_arenas_get_count(subproc);
1713 if (max_arena == 0) return;
1714
1715 // allow only one thread to purge at a time (todo: allow concurrent purging?)
1716 static mi_atomic_guard_t purge_guard;
1717 mi_atomic_guard(&purge_guard)
1718 {
1719 // increase global expire: at most one purge per delay cycle
1720 if (arenas_expire > now) { mi_atomic_store_release(&subproc->purge_expire, now + (delay/10)); }
1721 const size_t arena_start = tld->thread_seq % max_arena;
1722 size_t max_purge_count = (visit_all ? max_arena : (max_arena/4)+1);
1723 bool all_visited = true;
1724 bool any_purged = false;
1725 for (size_t _i = 0; _i < max_arena; _i++) {
1726 size_t i = _i + arena_start;
1727 if (i >= max_arena) { i -= max_arena; }
1728 mi_arena_t* arena = mi_arena_from_index(subproc,i);
1729 if (arena != NULL) {
1730 if (mi_arena_try_purge(arena, now, force)) {
1731 any_purged = true;
1732 if (max_purge_count <= 1) {
1733 all_visited = false;
1734 break;
1735 }
1736 max_purge_count--;
1737 }
1738 }
1739 }
1740 if (all_visited && !any_purged) {
1741 mi_atomic_store_release(&subproc->purge_expire, 0);
1742 }
1743 }
