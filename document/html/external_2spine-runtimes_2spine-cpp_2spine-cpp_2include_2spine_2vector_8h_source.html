75 inline void setSize(size_t newSize, const T &defaultValue) {
76 assert(newSize >= 0);
77 size_t oldSize = _size;
78 _size = newSize;
79 if (_capacity < newSize) {
80 _capacity = (int) (_size * 1.75f);
81 if (_capacity < 8) _capacity = 8;
82 _buffer = spine::SpineExtension::realloc<T>(_buffer, _capacity, __FILE__, __LINE__);
83 }

85 for (size_t i = oldSize; i < _size; i++) {
86 construct(_buffer + i, defaultValue);

88 }

91 inline void ensureCapacity(size_t newCapacity = 0) {
92 if (_capacity >= newCapacity) return;
93 _capacity = newCapacity;
94 _buffer = SpineExtension::realloc<T>(_buffer, newCapacity, __FILE__, __LINE__);
95 }

97 inline void add(const T &inValue) {
98 if (_size == _capacity) {
99 // inValue might reference an element in this buffer
100 // When we reallocate, the reference becomes invalid.
101 // We thus need to create a defensive copy before
102 // reallocating.
103 T valueCopy = inValue;
104 _capacity = (int) (_size * 1.75f);
105 if (_capacity < 8) _capacity = 8;
106 _buffer = spine::SpineExtension::realloc<T>(_buffer, _capacity, __FILE__, __LINE__);
107 construct(_buffer + _size++, valueCopy);
108 } else {
109 construct(_buffer + _size++, inValue);
110 }
111 }
112
113 inline void addAll(Vector<T> &inValue) {
114 ensureCapacity(this->size() + inValue.size());
115 for (size_t i = 0; i < inValue.size(); i++) {
116 add(inValue[i]);
117 }
118 }
119
120 inline void clearAndAddAll(Vector<T> &inValue) {
121 this->clear();
122 this->addAll(inValue);
123 }
124
125 inline void removeAt(size_t inIndex) {
126 assert(inIndex < _size);
127
128 --_size;
129
130 if (inIndex != _size) {
131 for (size_t i = inIndex; i < _size; ++i) {
132 T tmp(_buffer[i]);
133 _buffer[i] = _buffer[i + 1];
134 _buffer[i + 1] = tmp;
135 }
136 }
137
138 destroy(_buffer + _size);
139 }
140
141 inline bool contains(const T &inValue) {
142 for (size_t i = 0; i < _size; ++i) {
143 if (_buffer[i] == inValue) {
144 return true;
145 }
146 }
147
148 return false;
149 }
150
151 inline int indexOf(const T &inValue) {
152 for (size_t i = 0; i < _size; ++i) {
153 if (_buffer[i] == inValue) {
154 return (int) i;
155 }
156 }
157
158 return -1;
159 }

167 inline friend bool operator==(Vector<T> &lhs, Vector<T> &rhs) {
168 if (lhs.size() != rhs.size()) {
169 return false;
170 }
171
172 for (size_t i = 0, n = lhs.size(); i < n; ++i) {
173 if (lhs[i] != rhs[i]) {
174 return false;
175 }
176 }
177
178 return true;
179 }
180
181 inline friend bool operator!=(Vector<T> &lhs, Vector<T> &rhs) {
182 return !(lhs == rhs);
183 }
184
185 inline T *buffer() {
186 return _buffer;
187 }
