131#define EASTL_ATOMIC_POINTER_FUNC_IMPL(op, bits) \
132 T* retVal; \
133 { \
134 ptr_integral_type retType; \
135 ptr_integral_type addend = static_cast<ptr_integral_type>(arg) * static_cast<ptr_integral_type>(sizeof(T)); \
136 \
137 EA_PREPROCESSOR_JOIN(op, bits)(ptr_integral_type, retType, EASTL_ATOMIC_INTEGRAL_CAST(ptr_integral_type, this->GetAtomicAddress()), addend); \
138 \
139 retVal = reinterpret_cast<T*>(retType); \
140 } \
141 return retVal;
