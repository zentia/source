188void _mi_options_init(void) {
189 // called on process load
190 mi_add_stderr_output(); // now it safe to use stderr for output
191 for(int i = 0; i < _mi_option_last; i++ ) {
192 mi_option_t option = (mi_option_t)i;
193 long l = mi_option_get(option); MI_UNUSED(l); // initialize
194 }
195 mi_max_error_count = mi_option_get(mi_option_max_errors);
196 mi_max_warning_count = mi_option_get(mi_option_max_warnings);
197 #if MI_GUARDED
198 if (mi_option_get(mi_option_guarded_sample_rate) > 0) {
199 if (mi_option_is_enabled(mi_option_allow_large_os_pages)) {
200 mi_option_disable(mi_option_allow_large_os_pages);
201 _mi_warning_message("option 'allow_large_os_pages' is disabled to allow for guarded objects\n");
202 }
203 }
204 #endif
205 if (mi_option_is_enabled(mi_option_verbose)) { mi_options_print(); }
206}

260mi_decl_nodiscard long mi_option_get(mi_option_t option) {
261 mi_assert(option >= 0 && option < _mi_option_last);
262 if (option < 0 || option >= _mi_option_last) return 0;
263 mi_option_desc_t* desc = &options[option];
264 mi_assert(desc->option == option); // index should match the option
265 if mi_unlikely(desc->init == UNINIT) {
266 mi_option_init(desc);
267 }
268 return desc->value;
269}

271mi_decl_nodiscard long mi_option_get_clamp(mi_option_t option, long min, long max) {
272 long x = mi_option_get(option);
273 return (x < min ? min : (x > max ? max : x));
274}

276mi_decl_nodiscard size_t mi_option_get_size(mi_option_t option) {
277 const long x = mi_option_get(option);
278 size_t size = (x < 0 ? 0 : (size_t)x);
279 if (mi_option_has_size_in_kib(option)) {
280 size *= MI_KiB;
281 }
282 return size;

285void mi_option_set(mi_option_t option, long value) {
286 mi_assert(option >= 0 && option < _mi_option_last);
287 if (option < 0 || option >= _mi_option_last) return;
288 mi_option_desc_t* desc = &options[option];
289 mi_assert(desc->option == option); // index should match the option
290 desc->value = value;
291 desc->init = INITIALIZED;
292 // ensure min/max range; be careful to not recurse.
293 if (desc->option == mi_option_guarded_min && _mi_option_get_fast(mi_option_guarded_max) < value) {
294 mi_option_set(mi_option_guarded_max, value);
295 }
296 else if (desc->option == mi_option_guarded_max && _mi_option_get_fast(mi_option_guarded_min) > value) {
297 mi_option_set(mi_option_guarded_min, value);
298 }
299}

337// Since an output function can be registered earliest in the `main`
338// function we also buffer output that happens earlier. When
339// an output function is registered it is called immediately with
340// the output up to that point.
341#ifndef MI_MAX_DELAY_OUTPUT
342#define MI_MAX_DELAY_OUTPUT ((size_t)(16*1024))
343#endif
344static char out_buf[MI_MAX_DELAY_OUTPUT+1];
345static _Atomic(size_t) out_len;
346
347static void mi_cdecl mi_out_buf(const char* msg, void* arg) {
348 MI_UNUSED(arg);
349 if (msg==NULL) return;
350 if (mi_atomic_load_relaxed(&out_len)>=MI_MAX_DELAY_OUTPUT) return;
351 size_t n = _mi_strlen(msg);
352 if (n==0) return;
353 // claim space
354 size_t start = mi_atomic_add_acq_rel(&out_len, n);
355 if (start >= MI_MAX_DELAY_OUTPUT) return;
356 // check bound
357 if (start+n >= MI_MAX_DELAY_OUTPUT) {
358 n = MI_MAX_DELAY_OUTPUT-start-1;
359 }
360 _mi_memcpy(&out_buf[start], msg, n);
361}
362
363static void mi_out_buf_flush(mi_output_fun* out, bool no_more_buf, void* arg) {
364 if (out==NULL) return;
365 // claim (if `no_more_buf == true`, no more output will be added after this point)
366 size_t count = mi_atomic_add_acq_rel(&out_len, (no_more_buf ? MI_MAX_DELAY_OUTPUT : 1));
367 // and output the current contents
368 if (count>MI_MAX_DELAY_OUTPUT) count = MI_MAX_DELAY_OUTPUT;
369 out_buf[count] = 0;
370 out(out_buf,arg);
371 if (!no_more_buf) {
372 out_buf[count] = '\n'; // if continue with the buffer, insert a newline
373 }
374}

386// --------------------------------------------------------
387// Default output handler
388// --------------------------------------------------------
389
390// Should be atomic but gives errors on many platforms as generally we cannot cast a function pointer to a uintptr_t.
391// For now, don't register output from multiple threads.
392static mi_output_fun* volatile mi_out_default; // = NULL
393static _Atomic(void*) mi_out_arg; // = NULL
394
395static mi_output_fun* mi_out_get_default(void** parg) {
396 if (parg != NULL) { *parg = mi_atomic_load_ptr_acquire(void,&mi_out_arg); }
397 mi_output_fun* out = mi_out_default;
398 return (out == NULL ? &mi_out_buf : out);
399}
400
401void mi_register_output(mi_output_fun* out, void* arg) mi_attr_noexcept {
402 mi_out_default = (out == NULL ? &mi_out_stderr : out); // stop using the delayed output buffer
403 mi_atomic_store_ptr_release(void,&mi_out_arg, arg);
404 if (out!=NULL) mi_out_buf_flush(out,true,arg); // output all the delayed output now
405}

407// add stderr to the delayed output after the module is loaded
408static void mi_add_stderr_output(void) {
409 mi_assert_internal(mi_out_default == NULL);
410 mi_out_buf_flush(&mi_out_stderr, false, NULL); // flush current contents to stderr
411 mi_out_default = &mi_out_buf_stderr; // and add stderr to the delayed output
412}

420// When overriding malloc, we may recurse into mi_vfprintf if an allocation
421// inside the C runtime causes another message.
422// In some cases (like on macOS) the loader already allocates which
423// calls into mimalloc; if we then access thread locals (like `recurse`)
424// this may crash as the access may call _tlv_bootstrap that tries to
425// (recursively) invoke malloc again to allocate space for the thread local
426// variables on demand. This is why we use a _mi_preloading test on such
427// platforms. However, C code generator may move the initial thread local address
428// load before the `if` and we therefore split it out in a separate function.
429static mi_decl_thread bool recurse = false;
430
431static mi_decl_noinline bool mi_recurse_enter_prim(void) {
432 if (recurse) return false;
433 recurse = true;
434 return true;
435}
436
437static mi_decl_noinline void mi_recurse_exit_prim(void) {
438 recurse = false;
439}
440
441static bool mi_recurse_enter(void) {
442 #if defined(__APPLE__) || defined(MI_TLS_RECURSE_GUARD)
443 if (_mi_preloading()) return false;
444 #endif
445 return mi_recurse_enter_prim();
446}
447
448static void mi_recurse_exit(void) {
449 #if defined(__APPLE__) || defined(MI_TLS_RECURSE_GUARD)
450 if (_mi_preloading()) return;
451 #endif
452 mi_recurse_exit_prim();
453}

455void _mi_fputs(mi_output_fun* out, void* arg, const char* prefix, const char* message) {
456 if (out==NULL || (void*)out==(void*)stdout || (void*)out==(void*)stderr) { // TODO: use mi_out_stderr for stderr?
457 if (!mi_recurse_enter()) return;
458 out = mi_out_get_default(&arg);
459 if (prefix != NULL) out(prefix, arg);
460 out(message, arg);
461 mi_recurse_exit();
462 }
463 else {
464 if (prefix != NULL) out(prefix, arg);
465 out(message, arg);
466 }
467}

469// Define our own limited `fprintf` that avoids memory allocation.
470// We do this using `_mi_vsnprintf` with a limited buffer.
471static void mi_vfprintf( mi_output_fun* out, void* arg, const char* prefix, const char* fmt, va_list args ) {
472 char buf[992];
473 if (fmt==NULL) return;
474 if (!mi_recurse_enter()) return;
475 _mi_vsnprintf(buf, sizeof(buf)-1, fmt, args);
476 mi_recurse_exit();
477 _mi_fputs(out,arg,prefix,buf);
478}

480void _mi_fprintf( mi_output_fun* out, void* arg, const char* fmt, ... ) {
481 va_list args;
482 va_start(args,fmt);
483 mi_vfprintf(out,arg,NULL,fmt,args);
484 va_end(args);
485}

487static void mi_vfprintf_thread(mi_output_fun* out, void* arg, const char* prefix, const char* fmt, va_list args) {
488 if (prefix != NULL && _mi_strnlen(prefix,33) <= 32 && !_mi_is_main_thread()) {
489 char tprefix[64];
490 _mi_snprintf(tprefix, sizeof(tprefix), "%sthread 0x%tx: ", prefix, (uintptr_t)_mi_thread_id());
491 mi_vfprintf(out, arg, tprefix, fmt, args);
492 }
493 else {
494 mi_vfprintf(out, arg, prefix, fmt, args);
495 }
496}

528static void mi_show_error_message(const char* fmt, va_list args) {
529 if (!mi_option_is_enabled(mi_option_verbose)) {
530 if (!mi_option_is_enabled(mi_option_show_errors)) return;
531 if (mi_max_error_count >= 0 && (long)mi_atomic_increment_acq_rel(&error_count) > mi_max_error_count) return;

609#include <stdlib.h> // strtol
610#include <string.h> // strstr

612
613static void mi_option_init(mi_option_desc_t* desc) {
614 // Read option value from the environment
615 char s[64 + 1];
616 char buf[64+1];
617 _mi_strlcpy(buf, "mimalloc_", sizeof(buf));
618 _mi_strlcat(buf, desc->name, sizeof(buf));
619 bool found = _mi_getenv(buf, s, sizeof(s));
620 if (!found && desc->legacy_name != NULL) {
621 _mi_strlcpy(buf, "mimalloc_", sizeof(buf));
622 _mi_strlcat(buf, desc->legacy_name, sizeof(buf));
623 found = _mi_getenv(buf, s, sizeof(s));
624 if (found) {
625 _mi_warning_message("environment option \"mimalloc_%s\" is deprecated -- use \"mimalloc_%s\" instead.\n", desc->legacy_name, desc->name);
626 }
627 }
628
629 if (found) {
630 size_t len = _mi_strnlen(s, sizeof(buf) - 1);
631 for (size_t i = 0; i < len; i++) {
632 buf[i] = _mi_toupper(s[i]);
633 }
634 buf[len] = 0;
635 if (buf[0] == 0 || strstr("1;TRUE;YES;ON", buf) != NULL) {
636 desc->value = 1;
637 desc->init = INITIALIZED;
638 }
639 else if (strstr("0;FALSE;NO;OFF", buf) != NULL) {
640 desc->value = 0;
641 desc->init = INITIALIZED;
642 }
643 else {
644 char* end = buf;
645 long value = strtol(buf, &end, 10);
646 if (mi_option_has_size_in_kib(desc->option)) {
647 // this option is interpreted in KiB to prevent overflow of `long` for large allocations
648 // (long is 32-bit on 64-bit windows, which allows for 4TiB max.)
649 size_t size = (value < 0 ? 0 : (size_t)value);
650 bool overflow = false;
651 if (*end == 'K') { end++; }
652 else if (*end == 'M') { overflow = mi_mul_overflow(size,MI_KiB,&size); end++; }
653 else if (*end == 'G') { overflow = mi_mul_overflow(size,MI_MiB,&size); end++; }
654 else if (*end == 'T') { overflow = mi_mul_overflow(size,MI_GiB,&size); end++; }
655 else { size = (size + MI_KiB - 1) / MI_KiB; }
656 if (end[0] == 'I' && end[1] == 'B') { end += 2; } // KiB, MiB, GiB, TiB
657 else if (*end == 'B') { end++; } // Kb, Mb, Gb, Tb
658 if (overflow || size > MI_MAX_ALLOC_SIZE) { size = (MI_MAX_ALLOC_SIZE / MI_KiB); }
659 value = (size > LONG_MAX ? LONG_MAX : (long)size);
660 }
661 if (*end == 0) {
662 mi_option_set(desc->option, value);
663 }
664 else {
665 // set `init` first to avoid recursion through _mi_warning_message on mimalloc_verbose.
666 desc->init = DEFAULTED;
667 if (desc->option == mi_option_verbose && desc->value == 0) {
668 // if the 'mimalloc_verbose' env var has a bogus value we'd never know
669 // (since the value defaults to 'off') so in that case briefly enable verbose
670 desc->value = 1;
671 _mi_warning_message("environment option mimalloc_%s has an invalid value.\n", desc->name);
672 desc->value = 0;
673 }
674 else {
675 _mi_warning_message("environment option mimalloc_%s has an invalid value.\n", desc->name);
676 }
677 }
678 }
679 mi_assert_internal(desc->init != UNINIT);
680 }
681 else if (!_mi_preloading()) {
682 desc->init = DEFAULTED;
683 }
684}
