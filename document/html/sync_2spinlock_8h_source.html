141 void lock() noexcept(noexcept( backoff_strategy()()))
142 {
143 backoff_strategy backoff;
144
145 // Deadlock detected
146 CDS_TSAN_ANNOTATE_IGNORE_READS_BEGIN;
147 assert( m_dbgOwnerId != OS::get_current_thread_id());
148 CDS_TSAN_ANNOTATE_IGNORE_READS_END;
149
150 // TATAS algorithm
151 while ( !try_lock()) {
152 while ( m_spin.load( atomics::memory_order_acquire ))
153 backoff();
154 }
155
156 assert( m_dbgOwnerId == OS::get_current_thread_id());
157 }
158
160 void unlock() noexcept
161 {
162 assert( m_spin.load( atomics::memory_order_relaxed ));
163 assert( m_dbgOwnerId == OS::get_current_thread_id());
164 CDS_DEBUG_ONLY( m_dbgOwnerId = OS::c_NullThreadId; )
165
166 CDS_TSAN_ANNOTATE_MUTEX_RELEASED( this );
167 m_spin.store( false, atomics::memory_order_release );
168 }

182 template <typename Integral, class Backoff>
183 class reentrant_spin_lock
184 {
185 typedef OS::ThreadId thread_id;
186
187 public:
188 typedef Integral integral_type;
189 typedef Backoff backoff_strategy;
190
191 private:
192 //@cond
193 atomics::atomic<integral_type> m_spin;
194 atomics::atomic<thread_id> m_OwnerId;
195 //@endcond
196
197 private:
198 //@cond
199 void take( thread_id tid ) noexcept
200 {
201 m_OwnerId.store( tid, atomics::memory_order_relaxed );
202 }
203
204 void free() noexcept
205 {
206 m_OwnerId.store( OS::c_NullThreadId, atomics::memory_order_relaxed );
207 }
208
209 bool is_taken( thread_id tid ) const noexcept
210 {
211 return m_OwnerId.load( atomics::memory_order_relaxed ) == tid;
212 }
213
214 bool try_taken_lock( thread_id tid ) noexcept
215 {
216 if ( is_taken( tid )) {
217 m_spin.fetch_add( 1, atomics::memory_order_relaxed );
218 return true;
219 }
220 return false;
221 }
222
223 bool try_acquire() noexcept
224 {
225 integral_type nCurrent = 0;
226 bool bRet = m_spin.compare_exchange_weak( nCurrent, 1, atomics::memory_order_acquire, atomics::memory_order_acquire );
227
228# ifdef CDS_THREAD_SANITIZER_ENABLED
229 if ( bRet )
230 CDS_TSAN_ANNOTATE_MUTEX_ACQUIRED( this );
231# endif
232
233 return bRet;
234 }
235
236 bool try_acquire( unsigned int nTryCount ) noexcept( noexcept( backoff_strategy()()))
237 {
238 backoff_strategy bkoff;
239
240 while ( nTryCount-- ) {
241 if ( try_acquire())
242 return true;
243 bkoff();
244 }
245 return false;
246 }
247
248 void acquire() noexcept( noexcept( backoff_strategy()()))
249 {
250 // TATAS algorithm
251 backoff_strategy bkoff;
252 while ( !try_acquire()) {
253 while ( m_spin.load( atomics::memory_order_acquire ))
254 bkoff();
255 }
256 }
257 //@endcond
258
259 public:
261 reentrant_spin_lock() noexcept
262 : m_spin(0)
263 , m_OwnerId( OS::c_NullThreadId )
264 {
265 CDS_TSAN_ANNOTATE_MUTEX_CREATE( this );
266 }
267

274 reentrant_spin_lock( const reentrant_spin_lock<Integral, Backoff>& ) noexcept
275 : m_spin(0)
276 , m_OwnerId( OS::c_NullThreadId )
277 {
278 CDS_TSAN_ANNOTATE_MUTEX_CREATE( this );
279 }

282 explicit reentrant_spin_lock( bool bLocked )
283 : m_spin(0)
284 , m_OwnerId( OS::c_NullThreadId )
285 {
286 CDS_TSAN_ANNOTATE_MUTEX_CREATE( this );
287 if ( bLocked )
288 lock();
289 }
290
292 ~reentrant_spin_lock()
293 {
294 assert( m_spin.load( atomics::memory_order_acquire ) == 0 );
295 assert( m_OwnerId.load( atomics::memory_order_relaxed ) == OS::c_NullThreadId );
296
297 CDS_TSAN_ANNOTATE_MUTEX_DESTROY( this );
298 }

305 bool is_locked() const noexcept
306 {
307 return !( m_spin.load( atomics::memory_order_relaxed ) == 0 || is_taken( cds::OS::get_current_thread_id()));
308 }

311 bool try_lock() noexcept( noexcept( std::declval<reentrant_spin_lock>().try_acquire()))
312 {
313 thread_id tid = OS::get_current_thread_id();
314 if ( try_taken_lock( tid ))
315 return true;
316 if ( try_acquire()) {
317 take( tid );
318 return true;
319 }
320 return false;
321 }
322
324 bool try_lock( unsigned int nTryCount ) noexcept( noexcept( std::declval<reentrant_spin_lock>().try_acquire( nTryCount )))
325 {
326 thread_id tid = OS::get_current_thread_id();
327 if ( try_taken_lock( tid ))
328 return true;
329 if ( try_acquire( nTryCount )) {
330 take( tid );
331 return true;
332 }
333 return false;
334 }
335
337 void lock() noexcept( noexcept( std::declval<reentrant_spin_lock>().acquire()))
338 {
339 thread_id tid = OS::get_current_thread_id();
340 if ( !try_taken_lock( tid )) {
341 acquire();
342 take( tid );
343 }
344 }
345
347 void unlock() noexcept
348 {
349 assert( is_taken( OS::get_current_thread_id()));
350
351 integral_type n = m_spin.load( atomics::memory_order_relaxed );
352 if ( n > 1 )
353 m_spin.store( n - 1, atomics::memory_order_relaxed );
354 else {
355 free();
356 CDS_TSAN_ANNOTATE_MUTEX_RELEASED( this );
357 m_spin.store( 0, atomics::memory_order_release );
358 }
359 }
360
362 void change_owner( OS::ThreadId newOwnerId ) noexcept
363 {
364 assert( is_taken( OS::get_current_thread_id()));
365 assert( newOwnerId != OS::c_NullThreadId );
366
367 m_OwnerId.store( newOwnerId, atomics::memory_order_relaxed );
368 }
