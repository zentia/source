17 void test( Queue& q )
18 {
19 typedef typename Queue::value_type value_type;
20 value_type it;
21
22 const size_t nSize = 100;
23
24 ASSERT_TRUE( q.empty());
25 ASSERT_CONTAINER_SIZE( q, 0 );
26
27 // enqueue/dequeue
28 for ( size_t i = 0; i < nSize; ++i ) {
29 it = static_cast<value_type>(i);
30 ASSERT_TRUE( q.enqueue( it ));
31 ASSERT_CONTAINER_SIZE( q, i + 1 );
32 }
33 ASSERT_FALSE( q.empty());
34 ASSERT_CONTAINER_SIZE( q, nSize );
35
36 for ( size_t i = 0; i < nSize; ++i ) {
37 it = -1;
38 ASSERT_TRUE( q.dequeue( it ));
39 ASSERT_EQ( it, static_cast<value_type>( i ));
40 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
41 }
42 ASSERT_TRUE( q.empty());
43 ASSERT_CONTAINER_SIZE( q, 0 );
44
45 // push/pop
46 for ( size_t i = 0; i < nSize; ++i ) {
47 it = static_cast<value_type>(i);
48 ASSERT_TRUE( q.push( it ));
49 ASSERT_CONTAINER_SIZE( q, i + 1 );
50 }
51 ASSERT_FALSE( q.empty());
52 ASSERT_CONTAINER_SIZE( q, nSize );
53
54 for ( size_t i = 0; i < nSize; ++i ) {
55 it = -1;
56 ASSERT_TRUE( q.pop( it ));
57 ASSERT_EQ( it, static_cast<value_type>( i ));
58 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
59 }
60 ASSERT_TRUE( q.empty());
61 ASSERT_CONTAINER_SIZE( q, 0 );
62
63 // push/pop with lambda
64 for ( size_t i = 0; i < nSize; ++i ) {
65 it = static_cast<value_type>(i);
66 ASSERT_NE( it, -1 );
67 auto f = [&it]( value_type& dest ) { dest = it; it = -1; };
68 if ( i & 1 )
69 ASSERT_TRUE( q.enqueue_with( f ));
70 else
71 ASSERT_TRUE( q.push_with( f ));
72 ASSERT_EQ( it, -1 );
73 ASSERT_CONTAINER_SIZE( q, i + 1 );
74 }
75 ASSERT_FALSE( q.empty());
76 ASSERT_CONTAINER_SIZE( q, nSize );
77
78 for ( size_t i = 0; i < nSize; ++i ) {
79 it = -1;
80 auto f = [&it]( value_type& src ) { it = src; src = -1; };
81 if ( i & 1 )
82 ASSERT_TRUE( q.pop_with( f ));
83 else
84 ASSERT_TRUE( q.dequeue_with( f ));
85 ASSERT_EQ( it, static_cast<value_type>( i ));
86 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );

88 ASSERT_TRUE( q.empty());
89 ASSERT_CONTAINER_SIZE( q, 0 );

93 ASSERT_TRUE( q.push( static_cast<value_type>(i)));

115 template <class Queue>
116 void test_string( Queue& q )
117 {
118 std::string str[3];
119 str[0] = "one";
120 str[1] = "two";
121 str[2] = "three";
122 const size_t nSize = sizeof( str ) / sizeof( str[0] );
123
124 // emplace
125 for ( size_t i = 0; i < nSize; ++i ) {
126 ASSERT_TRUE( q.emplace( str[i].c_str()));
127 ASSERT_CONTAINER_SIZE( q, i + 1 );
128 }
129 ASSERT_FALSE( q.empty());
130 ASSERT_CONTAINER_SIZE( q, nSize );
131
132 {
133 std::string s;
134 auto f = [&s]( std::string& src ) {
135 ASSERT_FALSE( src.empty());
136 s = std::move( src );
137 ASSERT_NE( s, src );
138 };
139 for ( size_t i = 0; i < nSize; ++i ) {
140 if ( i & 1 )
141 ASSERT_TRUE( q.pop_with( f ));
142 else
143 ASSERT_TRUE( q.dequeue_with( f ));
144
145 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
146 ASSERT_EQ( s, str[i] );
147 }
148 }
149 ASSERT_TRUE( q.empty());
150 ASSERT_CONTAINER_SIZE( q, 0 );
151
152
153 // move push
154 for ( size_t i = 0; i < nSize; ++i ) {
155 std::string s = str[i];
156 ASSERT_FALSE( s.empty());
157 if ( i & 1 )
158 ASSERT_TRUE( q.enqueue( std::move( s )));
159 else
160 ASSERT_TRUE( q.push( std::move( s )));
161 ASSERT_TRUE( s.empty());
162 ASSERT_CONTAINER_SIZE( q, i + 1 );
163 }
164 ASSERT_FALSE( q.empty());
165 ASSERT_CONTAINER_SIZE( q, nSize );
166
167 for ( size_t i = 0; i < nSize; ++i ) {
168 std::string s;
169 ASSERT_TRUE( q.pop( s ));
170 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
171 ASSERT_EQ( s, str[i] );
172 }
173 ASSERT_TRUE( q.empty());
174 ASSERT_CONTAINER_SIZE( q, 0 );
175 }
