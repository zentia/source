86 template<typename I>
87 [[nodiscard]] auto read(I &&index) const noexcept {
88 if constexpr (element_stride == 1u) {
89 auto i = soa_offset() + std::forward<I>(index) + element_offset();
90 auto x = buffer().read(i);
91 if constexpr (sizeof(T) == sizeof(uint)) {
92 return x.template as<T>();
93 } else if constexpr (sizeof(T) * 2u == sizeof(uint)) {// 16bit
94 if constexpr (is_scalar_v<T>) {
95 auto u = x.template as<Vector<T, 2u>>();
96 return u.x;
97 } else {
98 static_assert(std::is_same_v<T, bool2>);
99 return x.template as<bool4>().xy();
100 }
101 } else if constexpr (sizeof(T) * 4u == sizeof(uint)) {// 8bit
102 static_assert(is_scalar_v<T>);
103 auto u = x.template as<Vector<T, 4u>>();
104 return u.x;
105 } else {// unreachable
106 static_assert(sizeof(T) == sizeof(uint));
107 }
108 } else if constexpr (element_stride == 2u) {
109 auto i = soa_offset() + (std::forward<I>(index) + element_offset()) * 2u;
110 auto u = dsl::make_uint2(buffer().read(i),
111 buffer().read(i + 1u));
112 return u.template as<T>();
113 } else {
114 static_assert(element_stride == 4u);
115 auto i = soa_offset() + (std::forward<I>(index) + element_offset()) * 4u;
116 auto u = dsl::make_uint4(buffer().read(i),
117 buffer().read(i + 1u),
118 buffer().read(i + 2u),
119 buffer().read(i + 3u));
120 return u.template as<T>();
121 }
122 }

124 template<typename I>
125 void write(I &&index, Expr<T> value) const noexcept {
126 if constexpr (element_stride == 1u) {
127 auto i = soa_offset() + std::forward<I>(index) + element_offset();
128 if constexpr (sizeof(T) == sizeof(uint)) {
129 buffer().write(i, value.template as<uint>());
130 } else if constexpr (sizeof(T) * 2u == sizeof(uint)) {// 16bit
131 if constexpr (is_scalar_v<T>) {
132 auto u = def<Vector<T, 2u>>();
133 u.x = value;
134 buffer().write(i, u.template as<uint>());
135 } else {
136 static_assert(std::is_same_v<T, bool2>);
137 auto u = make_bool4(value, make_bool2());
138 buffer().write(i, u.template as<uint>());
139 }
140 } else if constexpr (sizeof(T) * 4u == sizeof(uint)) {
141 static_assert(is_scalar_v<T>);
142 auto u = def<Vector<T, 4u>>();
143 u.x = value;
144 buffer().write(i, u.template as<uint>());
145 } else {// unreachable
146 static_assert(sizeof(T) == sizeof(uint));
147 }
148 } else if constexpr (element_stride == 2u) {
149 auto i = soa_offset() + (std::forward<I>(index) + element_offset()) * 2u;
150 auto u = value.template as<Vector<uint, 2>>();
151 buffer().write(i, u.x);
152 buffer().write(i + 1u, u.y);
153 } else {
154 static_assert(element_stride == 4u);
155 auto i = soa_offset() + (std::forward<I>(index) + element_offset()) * 4u;
156 auto u = value.template as<Vector<uint, 4>>();
157 buffer().write(i, u.x);
158 buffer().write(i + 1u, u.y);
159 buffer().write(i + 2u, u.z);
160 buffer().write(i + 3u, u.w);
161 }
162 }

566template<typename T>
567class SOAView : public detail::SOAViewBase<T> {
568
569public:
570 static constexpr auto element_stride =
571 static_cast<uint>((sizeof(T) + sizeof(uint) - 1u) / sizeof(uint));
572
573public:
574 [[nodiscard]] static auto compute_soa_size(auto n) noexcept {
575 return align_to_soa_cache_line(n * element_stride);
576 }
577
578public:
579 SOAView() noexcept = default;
580 SOAView(BufferView<uint> buffer,
581 size_t soa_offset, size_t soa_size,
582 size_t elem_offset, size_t elem_size) noexcept
583 : detail::SOAViewBase<T>{buffer, soa_offset, soa_size, elem_offset, elem_size} {
584 auto buffer_end = this->buffer().offset() + soa_offset +
585 (elem_offset + elem_size) * element_stride;
586 if (!(buffer_end <= (std::numeric_limits<uint>::max)())) [[unlikely]] {
587 detail::error_soa_view_exceeds_uint_max();
588 }
589 }
