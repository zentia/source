492#define BZ_GET_FAST(cccc) \
493 /* c_tPos is unsigned, hence test < 0 is pointless. */ \
494 if (s->tPos >= (UInt32)100000 * (UInt32)s->blockSize100k) return True; \
495 s->tPos = s->tt[s->tPos]; \
496 cccc = (UChar)(s->tPos & 0xff); \
497 s->tPos >>= 8;
498
499#define BZ_GET_FAST_C(cccc) \
500 /* c_tPos is unsigned, hence test < 0 is pointless. */ \
501 if (c_tPos >= (UInt32)100000 * (UInt32)ro_blockSize100k) return True; \
502 c_tPos = c_tt[c_tPos]; \
503 cccc = (UChar)(c_tPos & 0xff); \
504 c_tPos >>= 8;
505
506#define SET_LL4(i,n) \
507 { if (((i) & 0x1) == 0) \
508 s->ll4[(i) >> 1] = (s->ll4[(i) >> 1] & 0xf0) | (n); else \
509 s->ll4[(i) >> 1] = (s->ll4[(i) >> 1] & 0x0f) | ((n) << 4); \
510 }
511
512#define GET_LL4(i) \
513 ((((UInt32)(s->ll4[(i) >> 1])) >> (((i) << 2) & 0x4)) & 0xF)
514
515#define SET_LL(i,n) \
516 { s->ll16[i] = (UInt16)(n & 0x0000ffff); \
517 SET_LL4(i, n >> 16); \
518 }
519
520#define GET_LL(i) \
521 (((UInt32)s->ll16[i]) | (GET_LL4(i) << 16))
522
523#define BZ_GET_SMALL(cccc) \
524 /* c_tPos is unsigned, hence test < 0 is pointless. */ \
525 if (s->tPos >= (UInt32)100000 * (UInt32)s->blockSize100k) return True; \
526 cccc = BZ2_indexIntoF ( s->tPos, s->cftab ); \
527 s->tPos = GET_LL(s->tPos);
