43struct FibTask: public tbb::task {
44 long n;
45 long * sum;
46 FibTask( const long n_, long * const sum_ ) :
47 n(n_), sum(sum_)
48 {}
49 tbb::task* execute() {
50 if( n<CutOff ) {
51 *sum = SerialFib(n);
52 return NULL;
53 } else {
54 FibContinuation& c =
55 *new( allocate_continuation() ) FibContinuation(sum);
56 FibTask& b = *new( c.allocate_child() ) FibTask(n-1,&c.y);
57 recycle_as_child_of(c);
58 n -= 2;
59 sum = &c.x;
60 // Set ref_count to "two children".
61 c.set_ref_count(2);
62 c.spawn( b );
63 return this;
64 }
65 }
66};
