100// EASTL_LIMITS macros
101// These apply to integral types only.
103
104// true or false.
105#define EASTL_LIMITS_IS_SIGNED(T) ((T)(-1) < 0)
106
107// The min possible value of T.
108#define EASTL_LIMITS_MIN_S(T) ((T)((T)1 << EASTL_LIMITS_DIGITS_S(T)))
109#define EASTL_LIMITS_MIN_U(T) ((T)0)
110#define EASTL_LIMITS_MIN(T) ((EASTL_LIMITS_IS_SIGNED(T) ? EASTL_LIMITS_MIN_S(T) : EASTL_LIMITS_MIN_U(T)))
111
112// The max possible value of T.
113#define EASTL_LIMITS_MAX_S(T) ((T)(((((T)1 << (EASTL_LIMITS_DIGITS(T) - 1)) - 1) << 1) + 1))
114#define EASTL_LIMITS_MAX_U(T) ((T)~(T)0)
115#define EASTL_LIMITS_MAX(T) ((EASTL_LIMITS_IS_SIGNED(T) ? EASTL_LIMITS_MAX_S(T) : EASTL_LIMITS_MAX_U(T)))
116
117// The number of bits in the representation of T.
118#define EASTL_LIMITS_DIGITS_S(T) ((sizeof(T) * CHAR_BIT) - 1)
119#define EASTL_LIMITS_DIGITS_U(T) ((sizeof(T) * CHAR_BIT))
120#define EASTL_LIMITS_DIGITS(T) ((EASTL_LIMITS_IS_SIGNED(T) ? EASTL_LIMITS_DIGITS_S(T) : EASTL_LIMITS_DIGITS_U(T)))
121
122// The number of decimal digits that can be represented by T.
123#define EASTL_LIMITS_DIGITS10_S(T) ((EASTL_LIMITS_DIGITS_S(T) * 643L) / 2136) // (643 / 2136) ~= log10(2).
124#define EASTL_LIMITS_DIGITS10_U(T) ((EASTL_LIMITS_DIGITS_U(T) * 643L) / 2136)
125#define EASTL_LIMITS_DIGITS10(T) ((EASTL_LIMITS_IS_SIGNED(T) ? EASTL_LIMITS_DIGITS10_S(T) : EASTL_LIMITS_DIGITS10_U(T)))

155 // Defines default values for numeric_limits, which can be overridden by class specializations.
156 // See C++11 18.3.2.3
157 struct numeric_limits_base
158 {
159 // true if the type has an explicit specialization defined in the template class; false if not.
160 static EA_CONSTEXPR_OR_CONST bool is_specialized = false;
161
162 // Integer types: the number of *bits* in the representation of T.
163 // Floating types: the number of digits in the mantissa of T (same as FLT_MANT_DIG, DBL_MANT_DIG or LDBL_MANT_DIG).
164 static EA_CONSTEXPR_OR_CONST int digits = 0;
165
166 // The number of decimal digits that can be represented by T.
167 // Equivalent to FLT_DIG, DBL_DIG or LDBL_DIG for floating types.
168 static EA_CONSTEXPR_OR_CONST int digits10 = 0;
169
170 // The number of decimal digits required to make sure that two distinct values of the type have distinct decimal representations.
171 static EA_CONSTEXPR_OR_CONST int max_digits10 = 0;
172
173 // True if the type is signed.
174 static EA_CONSTEXPR_OR_CONST bool is_signed = false;
175
176 // True if the type is integral.
177 static EA_CONSTEXPR_OR_CONST bool is_integer = false;
178
179 // True if the type uses an exact representation. All integral types are
180 // exact, but other types can be exact as well.
181 static EA_CONSTEXPR_OR_CONST bool is_exact = false;
182
183 // Integer types: the base of the representation. Always 2 for integers.
184 // Floating types: the base of the exponent representation. Always FLT_RADIX (typically 2) for float.
185 static EA_CONSTEXPR_OR_CONST int radix = 0;
186
187 // The minimum integral radix-based exponent representable by the type.
188 static EA_CONSTEXPR_OR_CONST int min_exponent = 0;
189
190 // The minimum integral base 10 exponent representable by the type.
191 static EA_CONSTEXPR_OR_CONST int min_exponent10 = 0;
192
193 // The maximum integral radix-based exponent representable by the type.
194 static EA_CONSTEXPR_OR_CONST int max_exponent = 0;
195
196 // The maximum integral base 10 exponent representable by the type.
197 static EA_CONSTEXPR_OR_CONST int max_exponent10 = 0;
198
199 // True if the type has a representation for positive infinity.
200 static EA_CONSTEXPR_OR_CONST bool has_infinity = false;
201
202 // True if the type has a representation for a quiet (non-signaling) NaN.
203 static EA_CONSTEXPR_OR_CONST bool has_quiet_NaN = false;
204
205 // True if the type has a representation for a signaling NaN.
206 static EA_CONSTEXPR_OR_CONST bool has_signaling_NaN = false;
207
208 // An enumeration which identifies denormalization behavior.
209 // In practice the application can change this at runtime via hardware-specific commands.
210 static EA_CONSTEXPR_OR_CONST float_denorm_style has_denorm = denorm_absent;
211
212 // True if the loss of accuracy is detected as a denormalization loss.
213 // Typically false for integer types and true for floating point types.
214 static EA_CONSTEXPR_OR_CONST bool has_denorm_loss = false;
215
216 // True if the type has a bounded set of representable values. Typically true for
217 // all built-in numerial types (integer and floating point).
218 static EA_CONSTEXPR_OR_CONST bool is_bounded = false;
219
220 // True if the type has a modulo representation (if it's possible to add two
221 // positive numbers and have a result that wraps around to a third number
222 // that is less. Typically true for integers and false for floating types.
223 static EA_CONSTEXPR_OR_CONST bool is_modulo = false;
224
225 // True if trapping (arithmetic exception generation) is implemented for this type.
226 // Typically true for integer types (div by zero) and false for floating point types,
227 // though in practice the application may be able to change floating point to trap at runtime.
228 static EA_CONSTEXPR_OR_CONST bool traps = false;
229
230 // True if tiny-ness is detected before rounding.
231 static EA_CONSTEXPR_OR_CONST bool tinyness_before = false;
232
233 // An enumeration which identifies default rounding behavior.
234 // In practice the application can change this at runtime via hardware-specific commands.
235 static EA_CONSTEXPR_OR_CONST float_round_style round_style = round_toward_zero;
236
237 // True if the type is floating point and follows the IEC 559 standard (IEEE 754).
238 // In practice the application or OS can change this at runtime via hardware-specific commands or via compiler optimizations.
239 static EA_CONSTEXPR_OR_CONST bool is_iec559 = false;
240 };
