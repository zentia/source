22// --------------------------------------------------------------------------------------------
23// Atomics
24// We need to be portable between C, C++, and MSVC.
25// We base the primitives on the C/C++ atomics and create a minimal wrapper for MSVC in C compilation mode.
26// This is why we try to use only `uintptr_t` and `<type>*` as atomic types.
27// To gain better insight in the range of used atomics, we use explicitly named memory order operations
28// instead of passing the memory order as a parameter.
29// -----------------------------------------------------------------------------------------------

195static inline bool mi_atomic_compare_exchange_strong_explicit(_Atomic(uintptr_t)*p, uintptr_t* expected, uintptr_t desired, mi_memory_order mo1, mi_memory_order mo2) {
196 (void)(mo1); (void)(mo2);
197 uintptr_t read = (uintptr_t)MI_64(_InterlockedCompareExchange)((volatile msc_intptr_t*)p, (msc_intptr_t)desired, (msc_intptr_t)(*expected));
198 if (read == *expected) {
199 return true;
200 }
201 else {
202 *expected = read;
203 return false;
204 }
205}

218static inline uintptr_t mi_atomic_load_explicit(_Atomic(uintptr_t) const* p, mi_memory_order mo) {
219 (void)(mo);
220#if defined(_M_IX86) || defined(_M_X64)
221 return *p;
222#else
223 uintptr_t x = *p;
224 if (mo > mi_memory_order_relaxed) {
225 while (!mi_atomic_compare_exchange_weak_explicit((_Atomic(uintptr_t)*)p, &x, x, mo, mi_memory_order_relaxed)) { /* nothing */ };
226 }
227 return x;
228#endif
229}
230static inline void mi_atomic_store_explicit(_Atomic(uintptr_t)*p, uintptr_t x, mi_memory_order mo) {
231 (void)(mo);
232#if defined(_M_IX86) || defined(_M_X64)
233 *p = x;
234#else
235 mi_atomic_exchange_explicit(p, x, mo);
236#endif
237}
238static inline int64_t mi_atomic_loadi64_explicit(_Atomic(int64_t)*p, mi_memory_order mo) {
239 (void)(mo);
240#if defined(_M_X64)
241 return *p;
242#else
243 int64_t old = *p;
244 int64_t x = old;
245 while ((old = InterlockedCompareExchange64(p, x, old)) != x) {
246 x = old;
247 }
248 return x;
249#endif
250}
251static inline void mi_atomic_storei64_explicit(_Atomic(int64_t)*p, int64_t x, mi_memory_order mo) {
252 (void)(mo);
253#if defined(x_M_IX86) || defined(_M_X64)
254 *p = x;
255#else
256 InterlockedExchange64(p, x);
257#endif
258}

261static inline int64_t mi_atomic_addi64_relaxed(volatile _Atomic(int64_t)*p, int64_t add) {
262#ifdef _WIN64
263 return (int64_t)mi_atomic_addi((int64_t*)p, add);
264#else
265 int64_t current;
266 int64_t sum;
267 do {
268 current = *p;
269 sum = current + add;
270 } while (_InterlockedCompareExchange64(p, sum, current) != current);
271 return current;
272#endif
273}

275 int64_t current;
276 do {
277 current = *p;
278 } while (current < x && _InterlockedCompareExchange64(p, x, current) != current);
279}
