52#define THREE_QUARTERS ((int) ((BITS_IN_int * 3) / 4))
53#define ONE_EIGHTH ((int) (BITS_IN_int / 8))
54#define HIGH_BITS ( ~((unsigned int)(~0) >> ONE_EIGHTH ))

69static inline unsigned int
70hash_pjw(void* key)
71{
72 char *datum = (char *)key;
73 unsigned int hash_value, i;
74
75 if(!datum) return 0;
76
77 for (hash_value = 0; *datum; ++datum) {
78 hash_value = (hash_value << ONE_EIGHTH) + *datum;
79 if ((i = hash_value & HIGH_BITS) != 0)
80 hash_value = (hash_value ^ (i >> THREE_QUARTERS)) & ~HIGH_BITS;
81 }
82 return (hash_value);
83}

101static inline icl_hash_t *
102icl_hash_create( int nbuckets, unsigned int (*hash_function)(void*), int (*hash_key_compare)(void*, void*) )
103{
104 icl_hash_t *ht;
105 int i;
106
107 ht = (icl_hash_t*) malloc(sizeof(icl_hash_t));
108 assert(ht!=NULL);
109 if(!ht) return NULL;
110
111 ht->nentries = 0;
112 ht->buckets = (icl_entry_t**)malloc(nbuckets * sizeof(icl_entry_t*));
113 assert(ht->buckets!=NULL);
114 if(!ht->buckets) return NULL;
115
116 ht->nbuckets = nbuckets;
117 for(i=0;i<ht->nbuckets;i++)
118 ht->buckets[i] = NULL;
119
120 ht->hash_function = hash_function ? hash_function : hash_pjw;
121 ht->hash_key_compare = hash_key_compare ? hash_key_compare : string_compare;
122
123 return ht;
124}

136static inline void *
137icl_hash_find(icl_hash_t *ht, void* key)
138{
139 icl_entry_t* curr;
140 unsigned int hash_val;
141
142 if(!ht || !key) return NULL;
143
144 hash_val = (* ht->hash_function)(key) % ht->nbuckets;
145
146 for (curr=ht->buckets[hash_val]; curr != NULL; curr=curr->next)
147 if ( ht->hash_key_compare(curr->key, key))
148 return(curr->data);
149
150 return NULL;
151}

163static inline icl_entry_t *
164icl_hash_insert(icl_hash_t *ht, void* key, void *data)
165{
166 icl_entry_t *curr;
167 unsigned int hash_val;
168
169 if(!ht || !key) return NULL;
170
171 hash_val = (* ht->hash_function)(key) % ht->nbuckets;
172
173 for (curr=ht->buckets[hash_val]; curr != NULL; curr=curr->next)
174 if ( ht->hash_key_compare(curr->key, key))
175 return(NULL); /* key already exists */
176
177 /* if key was not found */
178 curr = (icl_entry_t*)malloc(sizeof(icl_entry_t));
179 assert(curr != NULL);
180 if(!curr) return NULL;
181
182 curr->key = key;
183 curr->data = data;
184 curr->next = ht->buckets[hash_val]; /* add at start */
185
186 ht->buckets[hash_val] = curr;
187 ht->nentries++;
188
189 return curr;
190}

256static inline int icl_hash_delete(icl_hash_t *ht, void* key, void (*free_key)(void*), void (*free_data)(void*))
257{
258 icl_entry_t *curr, *prev;
259 unsigned int hash_val;
260
261 if(!ht || !key) return -1;
262 hash_val = (* ht->hash_function)(key) % ht->nbuckets;
263
264 prev = NULL;
265 for (curr=ht->buckets[hash_val]; curr != NULL; ) {
266 if ( ht->hash_key_compare(curr->key, key)) {
267 if (prev == NULL) {
268 ht->buckets[hash_val] = curr->next;
269 } else {
270 prev->next = curr->next;
271 }
272 if (*free_key && curr->key) (*free_key)(curr->key);
273 if (*free_data && curr->data) (*free_data)(curr->data);
274 ht->nentries++;
275 free(curr);
276 return 0;
277 }
278 prev = curr;
279 curr = curr->next;
280 }
281 return -1;
282}

293static inline int
294icl_hash_destroy(icl_hash_t *ht, void (*free_key)(void*), void (*free_data)(void*))
295{
296 icl_entry_t *bucket, *curr, *next;
297 int i;
298
299 if(!ht) return -1;
300
301 for (i=0; i<ht->nbuckets; i++) {
302 bucket = ht->buckets[i];
303 for (curr=bucket; curr!=NULL; ) {
304 next=curr->next;
305 if (*free_key && curr->key) (*free_key)(curr->key);
306 if (*free_data && curr->data) (*free_data)(curr->data);
307 free(curr);
308 curr=next;
309 }
310 }
311
312 if(ht->buckets) free(ht->buckets);
313 if(ht) free(ht);
314
315 return 0;
316}
