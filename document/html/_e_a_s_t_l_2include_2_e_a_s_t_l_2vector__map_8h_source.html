6// This file implements vector_map. It acts much like std::map, except its
7// underlying representation is a random access container such as vector.
8// These containers are sometimes also known as "sorted vectors."
9// vector_maps have an advantage over conventional maps in that their memory
10// is contiguous and node-less. The result is that lookups are faster, more
11// cache friendly (which potentially more so benefits speed), and the container
12// uses less memory. The downside is that inserting new items into the container
13// is slower if they are inserted in random order instead of in sorted order.
14// This tradeoff is well-worth it for many cases. Note that vector_map allows
15// you to use a deque or other random access container which may perform
16// better for you than vector.
17//
18// Note that with vector_set, vector_multiset, vector_map, vector_multimap
19// that the modification of the container potentially invalidates all
20// existing iterators into the container, unlike what happens with conventional
21// sets and maps.
22//
23// This type could conceptually use a eastl::array as its underlying container,
24// however the current design requires an allocator aware container.
25// Consider using a fixed_vector instead.

257 // Note: vector_map operator[] returns a reference to the mapped_type, same as map does.
258 // But there's an important difference: This reference can be invalidated by -any- changes
259 // to the vector_map that cause it to change capacity. This is unlike map, with which
260 // mapped_type references are invalidated only if that mapped_type element itself is removed
261 // from the map. This is because vector is array-based and map is node-based. As a result
262 // the following code that is safe for map is unsafe for vector_map for the case that
263 // the vMap[100] doesn't already exist in the vector_map:
264 // vMap[100] = vMap[0]
265 mapped_type& operator[](const key_type& k);
266 mapped_type& operator[](key_type&& k);
267
268 // aka. the standard's at() member function.
269 mapped_type& at_key(const key_type& k);
270 const mapped_type& at_key(const key_type& k) const;

279 // NOTE(rparolin): It is undefined behaviour if user code fails to ensure the container
280 // invariants are respected by performing an explicit call to 'sort' before any other
281 // operations on the container are performed that do not clear the elements.
282 //
283 // 'push_back_unsorted' and 'emplace_back_unsorted' do not satisfy container invariants
284 // for being sorted. We provide these overloads explicitly labelled as '_unsorted' as an
285 // optimization opportunity when batch inserting elements so users can defer the cost of
286 // sorting the container once when all elements are contained. This was done to clarify
287 // the intent of code by leaving a trace that a manual call to sort is required.
288 //
289 template <typename... Args> decltype(auto) push_back_unsorted(Args&&... args)
290 { return base_type::push_back(eastl::forward<Args>(args)...); }
291 template <typename... Args> decltype(auto) emplace_back_unsorted(Args&&... args)
292 { return base_type::emplace_back(eastl::forward<Args>(args)...); }

536 template <typename K, typename T, typename C, typename A, typename RAC>
537 typename vector_map<K, T, C, A, RAC>::iterator
538 vector_map<K, T, C, A, RAC>::insert(const_iterator position, const value_type& value)
539 {
540 // We assume that the user knows what he is doing and has supplied us with
541 // a position that is right where value should be inserted (put in front of).
542 // We do a test to see if the position is correct. If so then we insert,
543 // if not then we ignore the input position.
544
545 if((position == end()) || value_compare::operator()(value, *position)) // If the element at position is greater than value...
546 {
547 if((position == begin()) || value_compare::operator()(*(position - 1), value)) // If the element before position is less than value...
548 return base_type::insert(position, value);
549 }
550
551 // In this case we either have an incorrect position or value is already present.
552 // We fall back to the regular insert function. An optimization would be to detect
553 // that the element is already present, but that's only useful if the user supplied
554 // a good position but a present element.
555 const eastl::pair<typename vector_map<K, T, C, A, RAC>::iterator, bool> result = insert(value);
556
557 return result.first;
558 }

732 {
733 // The resulting range will either be empty or have one element,
734 // so instead of doing two tree searches (one for lower_bound and
735 // one for upper_bound), we do just lower_bound and see if the
736 // result is a range of size zero or one.
737 const iterator itLower(lower_bound(k));
738

740 return eastl::pair<iterator, iterator>(itLower, itLower);
741
742 iterator itUpper(itLower);

744 }
