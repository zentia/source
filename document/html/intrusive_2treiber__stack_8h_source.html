400 template <typename Stat>
401 bool backoff( operation_desc& op, Stat& stat )
402 {
403 elimination_backoff_type bkoff;
404 op.nStatus.store( op_waiting, atomics::memory_order_relaxed );
405
406 elimination_rec * myRec = cds::algo::elimination::init_record<elimination_storage>( op );
407
408 collision_array_record& slot = m_Elimination.collisions[ slot_index() ];
409 {
410 slot.lock.lock();
411 elimination_rec * himRec = slot.pRec;
412 if ( himRec ) {
413 operation_desc * himOp = static_cast<operation_desc *>( himRec->pOp );
414 assert( himOp );
415 if ( himOp->idOp != op.idOp ) {
416
417 if ( op.idOp == treiber_stack::op_push )
418 himOp->pVal = op.pVal;
419 else
420 op.pVal = himOp->pVal;
421
422 slot.pRec = nullptr;
423 himOp->nStatus.store( op_collided, atomics::memory_order_release );
424 slot.lock.unlock();
425
426 cds::algo::elimination::clear_record<elimination_storage>();
427 stat.onActiveCollision( op.idOp );
428 return true;
429 }
430 //himOp->nStatus.store( op_free, atomics::memory_order_release );
431 }
432 slot.pRec = myRec;
433 slot.lock.unlock();
434 }
435
436 // Wait for colliding operation
437 bkoff( [&op]() noexcept -> bool { return op.nStatus.load( atomics::memory_order_acquire ) != op_waiting; } );
438
439 {
440 slot_scoped_lock l( slot.lock );
441 if ( slot.pRec == myRec )
442 slot.pRec = nullptr;
443 }
444
445 bool bCollided = op.nStatus.load( atomics::memory_order_relaxed ) == op_collided;
446
447 if ( !bCollided )
448 stat.onEliminationFailed();
449 else
450 stat.onPassiveCollision( op.idOp );
451
452 cds::algo::elimination::clear_record<elimination_storage>();
453 return bCollided;
454 }
455 };

726 bool push( value_type& val )
727 {
728 node_type * pNew = node_traits::to_node_ptr( val );
729 link_checker::is_empty( pNew );
730
731 typename elimination_backoff::type bkoff = m_Backoff.init();
732
733 operation_desc op;
734 constexpr_if( enable_elimination ) {
735 op.idOp = treiber_stack::op_push;
736 op.pVal = &val;
737 }
738
739 node_type * t = m_Top.load( memory_model::memory_order_relaxed );
740 while ( true ) {
741 pNew->m_pNext.store( t, memory_model::memory_order_relaxed );
742 if ( m_Top.compare_exchange_weak( t, pNew, memory_model::memory_order_release, atomics::memory_order_acquire )) {
743 ++m_ItemCounter;
744 m_stat.onPush();
745 return true;
746 }
747 m_stat.onPushRace();
748
749 if ( bkoff.backoff( op, m_stat ))
750 return true;
751 }
752 }
753
755
760 value_type * pop()
761 {
762 typename elimination_backoff::type bkoff = m_Backoff.init();
763 typename gc::Guard guard;
764
765 operation_desc op;
766 constexpr_if( enable_elimination ) {
767 op.idOp = treiber_stack::op_pop;
768 }
769
770 while ( true ) {
771 node_type * t = guard.protect( m_Top,
772 []( node_type * p ) -> value_type * {
773 return node_traits::to_value_ptr( p );
774 });
775 if ( t == nullptr )
776 return nullptr; // stack is empty
777
778 node_type * pNext = t->m_pNext.load(memory_model::memory_order_relaxed);
779 if ( m_Top.compare_exchange_weak( t, pNext, memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
780 clear_links( t );
781 --m_ItemCounter;
782 m_stat.onPop();
783 return node_traits::to_value_ptr( *t );
784 }
785
786 m_stat.onPopRace();
787 if ( bkoff.backoff( op, m_stat )) {
788 // may return nullptr if stack is empty
789 return op.pVal;
790 }
791 }
792 }
793
795 bool empty() const
796 {
797 return m_Top.load( memory_model::memory_order_relaxed ) == nullptr;
798 }

801
807 void clear()
808 {
809 back_off bkoff;
810 node_type * pTop;
811 while ( true ) {
812 pTop = m_Top.load( memory_model::memory_order_relaxed );
813 if ( pTop == nullptr )
814 return;
815 if ( m_Top.compare_exchange_weak( pTop, nullptr, memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
816 m_ItemCounter.reset();
817 break;
818 }
819 bkoff();
820 }
821
822 while( pTop ) {
823 node_type * p = pTop;
824 pTop = p->m_pNext.load(memory_model::memory_order_relaxed);
825 clear_links( p );
826 gc::template retire<disposer>( node_traits::to_value_ptr( *p ));
827 }
828 }
