63 static Result<MapType> make_map(const R& _r, const auto& _obj_or_map) {
64 MapType map;
65 std::vector<Error> errors;
66 const auto map_reader =
67 MapReader<R, W, MapType, ProcessorsType>(&_r, &map, &errors);
68 if constexpr (schemaful::IsSchemafulReader<R>) {
69 const auto err = _r.read_map(map_reader, _obj_or_map);
70 if (err) {
71 return error(*err);
72 }
73 } else {

75 if (err) {
76 return error(*err);
77 }
78 }
79 if (errors.size() != 0) {
80 return error(to_single_error_message(errors));
