333 template <typename Func>
334 bool insert( value_type& val, Func f )
335 {
336 typename gc::Guard guardInsert;
337 guardInsert.assign( &val );
338
339 unique_internal_node_ptr pNewInternal;
340 search_result res;
341 back_off bkoff;
342
343 for ( ;; ) {
344 if ( search( res, val, node_compare())) {
345 if ( pNewInternal.get())
346 m_Stat.onInternalNodeDeleted() ; // unique_internal_node_ptr deletes internal node
347 m_Stat.onInsertFailed();
348 return false;
349 }
350
351 if ( res.updGrandParent.bits() == update_desc::Clean && res.updParent.bits() == update_desc::Clean ) {
352
353 if ( !pNewInternal.get())
354 pNewInternal.reset( alloc_internal_node());
355
356 if ( try_insert( val, pNewInternal.get(), res )) {
357 f( val );
358 pNewInternal.release(); // internal node is linked into the tree and should not be deleted
359 break;
360 }
361 }
362
363 bkoff();
364 m_Stat.onInsertRetry();
365 }
366
367 ++m_ItemCounter;
368 m_Stat.onInsertSuccess();
369 return true;
370 }

400 template <typename Func>
401 std::pair<bool, bool> update( value_type& val, Func func, bool bAllowInsert = true )
402 {
403 typename gc::Guard guardInsert;
404 guardInsert.assign( &val );
405
406 unique_internal_node_ptr pNewInternal;
407 search_result res;
408 back_off bkoff;
409
410 for ( ;; ) {
411 if ( search( res, val, node_compare())) {
412 func( false, *node_traits::to_value_ptr( res.pLeaf ), val );
413 if ( pNewInternal.get())
414 m_Stat.onInternalNodeDeleted() ; // unique_internal_node_ptr deletes internal node
415 m_Stat.onUpdateExist();
416 return std::make_pair( true, false );
417 }
418
419 if ( res.updGrandParent.bits() == update_desc::Clean && res.updParent.bits() == update_desc::Clean ) {
420 if ( !bAllowInsert )
421 return std::make_pair( false, false );
422
423 if ( !pNewInternal.get())
424 pNewInternal.reset( alloc_internal_node());
425
426 if ( try_insert( val, pNewInternal.get(), res )) {
427 func( true, val, val );
428 pNewInternal.release() ; // internal node has been linked into the tree and should not be deleted
429 break;
430 }
431 }
432
433 bkoff();
434 m_Stat.onUpdateRetry();
435 }
436
437 ++m_ItemCounter;
438 m_Stat.onUpdateNew();
439 return std::make_pair( true, true );
440 }

804 void unsafe_clear()
805 {
806 while ( true ) {
807 internal_node * pParent = nullptr;
808 internal_node * pGrandParent = nullptr;
809 tree_node * pLeaf = const_cast<internal_node *>( &m_Root );
810
811 // Get leftmost leaf
812 while ( pLeaf->is_internal()) {
813 pGrandParent = pParent;
814 pParent = static_cast<internal_node *>( pLeaf );
815 pLeaf = pParent->m_pLeft.load( memory_model::memory_order_relaxed );
816 }
817
818 if ( pLeaf->infinite_key()) {
819 // The tree is empty
820 return;
821 }
822
823 // Remove leftmost leaf and its parent node
824 assert( pGrandParent );
825 assert( pParent );
826 assert( pLeaf->is_leaf());
827
828 pGrandParent->m_pLeft.store( pParent->m_pRight.load( memory_model::memory_order_relaxed ), memory_model::memory_order_relaxed );
829 free_leaf_node( node_traits::to_value_ptr( static_cast<leaf_node *>( pLeaf )));
830 free_internal_node( pParent );
831 }
832 }

866 bool check_consistency( internal_node const * pRoot ) const
867 {
868 tree_node * pLeft = pRoot->m_pLeft.load( atomics::memory_order_relaxed );
869 tree_node * pRight = pRoot->m_pRight.load( atomics::memory_order_relaxed );
870 assert( pLeft );
871 assert( pRight );
872
873 if ( node_compare()( *pLeft, *pRoot ) < 0
874 && node_compare()( *pRoot, *pRight ) <= 0
875 && node_compare()( *pLeft, *pRight ) < 0 )
876 {
877 bool bRet = true;
878 if ( pLeft->is_internal())
879 bRet = check_consistency( static_cast<internal_node *>( pLeft ));
880 assert( bRet );
881
882 if ( bRet && pRight->is_internal())
883 bRet = bRet && check_consistency( static_cast<internal_node *>( pRight ));
884 assert( bRet );
885
886 return bRet;
887 }
888 return false;
889 }

891 tree_node * protect_child_node( search_result& res, internal_node * pParent, bool bRight, update_ptr updParent ) const
892 {
893 retry:
894 tree_node * p = bRight
895 ? res.guards.protect( search_result::Guard_Leaf, pParent->m_pRight,
896 []( tree_node * pn ) -> internal_node* { return static_cast<internal_node *>(pn);})
897 : res.guards.protect( search_result::Guard_Leaf, pParent->m_pLeft,
898 []( tree_node * pn ) -> internal_node* { return static_cast<internal_node *>(pn);});
899
900 // If we use member hook, data node pointer != internal node pointer
901 // So, we need protect the child twice: as internal node and as data node
902 // and then analyze what kind of node we have
903 tree_node * pVal = bRight
904 ? res.guards.protect( search_result::Guard_temporary, pParent->m_pRight,
905 []( tree_node * pn ) -> value_type* { return node_traits::to_value_ptr( static_cast<leaf_node *>(pn));} )
906 : res.guards.protect( search_result::Guard_temporary, pParent->m_pLeft,
907 []( tree_node * pn ) -> value_type* { return node_traits::to_value_ptr( static_cast<leaf_node *>(pn));} );
908
909 // child node is guarded
910 // See whether pParent->m_pUpdate has not been changed
911 if ( pParent->m_pUpdate.load( memory_model::memory_order_acquire ) != updParent ) {
912 // update has been changed - returns nullptr as a flag to search retry
913 return nullptr;
914 }
915
916 if ( p != pVal )
917 goto retry;
918
919 if ( p && p->is_leaf())
920 res.guards.assign( search_result::Guard_Leaf, node_traits::to_value_ptr( static_cast<leaf_node *>( p )));
921
922 res.guards.clear( search_result::Guard_temporary );
923
924 return p;
925 }

932 template <typename KeyValue, typename Compare>
933 bool search( search_result& res, KeyValue const& key, Compare cmp ) const
934 {
935 internal_node * pParent;
936 internal_node * pGrandParent = nullptr;
937 update_ptr updParent;
938 update_ptr updGrandParent;
939 bool bRightLeaf;
940 bool bRightParent = false;
941
942 int nCmp = 0;
943
944 retry:
945 pParent = nullptr;
946 //pGrandParent = nullptr;
947 updParent = nullptr;
948 bRightLeaf = false;
949 tree_node * pLeaf = const_cast<internal_node *>( &m_Root );
950 while ( pLeaf->is_internal()) {
951 res.guards.copy( search_result::Guard_GrandParent, search_result::Guard_Parent );
952 pGrandParent = pParent;
953 res.guards.copy( search_result::Guard_Parent, search_result::Guard_Leaf );
954 pParent = static_cast<internal_node *>( pLeaf );
955 bRightParent = bRightLeaf;
956 res.guards.copy( search_result::Guard_updGrandParent, search_result::Guard_updParent );
957 updGrandParent = updParent;
958
959 updParent = search_protect_update( res, pParent->m_pUpdate );
960
961 switch ( updParent.bits()) {
962 case update_desc::DFlag:
963 case update_desc::Mark:
964 m_Stat.onSearchRetry();
965 goto retry;
966 }
967
968 nCmp = cmp( key, *pParent );
969 bRightLeaf = nCmp >= 0;
970
971 pLeaf = protect_child_node( res, pParent, bRightLeaf, updParent );
972 if ( !pLeaf ) {
973 m_Stat.onSearchRetry();
974 goto retry;
975 }
976 }
977
978 assert( pLeaf->is_leaf());
979 nCmp = cmp( key, *static_cast<leaf_node *>(pLeaf));
980
981 res.pGrandParent = pGrandParent;
982 res.pParent = pParent;
983 res.pLeaf = static_cast<leaf_node *>( pLeaf );
984 res.updParent = updParent;
985 res.updGrandParent = updGrandParent;
986 res.bRightParent = bRightParent;
987 res.bRightLeaf = bRightLeaf;
988
989 return nCmp == 0;
990 }

992 bool search_min( search_result& res ) const
993 {
994 internal_node * pParent;
995 internal_node * pGrandParent;
996 update_ptr updParent;
997 update_ptr updGrandParent;
998
999 retry:
1000 pParent = nullptr;
1001 pGrandParent = nullptr;
1002 updParent = nullptr;
1003 tree_node * pLeaf = const_cast<internal_node *>( &m_Root );
1004 while ( pLeaf->is_internal()) {
1005 res.guards.copy( search_result::Guard_GrandParent, search_result::Guard_Parent );
1006 pGrandParent = pParent;
1007 res.guards.copy( search_result::Guard_Parent, search_result::Guard_Leaf );
1008 pParent = static_cast<internal_node *>( pLeaf );
1009 res.guards.copy( search_result::Guard_updGrandParent, search_result::Guard_updParent );
1010 updGrandParent = updParent;
1011
1012 updParent = search_protect_update( res, pParent->m_pUpdate );
1013
1014 switch ( updParent.bits()) {
1015 case update_desc::DFlag:
1016 case update_desc::Mark:
1017 m_Stat.onSearchRetry();
1018 goto retry;
1019 }
1020
1021 pLeaf = protect_child_node( res, pParent, false, updParent );
1022 if ( !pLeaf ) {
1023 m_Stat.onSearchRetry();
1024 goto retry;
1025 }
1026 }
1027
1028 if ( pLeaf->infinite_key())
1029 return false;
1030
1031 res.pGrandParent = pGrandParent;
1032 res.pParent = pParent;
1033 assert( pLeaf->is_leaf());
1034 res.pLeaf = static_cast<leaf_node *>( pLeaf );
1035 res.updParent = updParent;
1036 res.updGrandParent = updGrandParent;
1037 res.bRightParent = false;
1038 res.bRightLeaf = false;
1039
1040 return true;
1041 }

1043 bool search_max( search_result& res ) const
1044 {
1045 internal_node * pParent;
1046 internal_node * pGrandParent;
1047 update_ptr updParent;
1048 update_ptr updGrandParent;
1049 bool bRightLeaf;
1050 bool bRightParent = false;
1051
1052 retry:
1053 pParent = nullptr;
1054 pGrandParent = nullptr;
1055 updParent = nullptr;
1056 bRightLeaf = false;
1057 tree_node * pLeaf = const_cast<internal_node *>( &m_Root );
1058 while ( pLeaf->is_internal()) {
1059 res.guards.copy( search_result::Guard_GrandParent, search_result::Guard_Parent );
1060 pGrandParent = pParent;
1061 res.guards.copy( search_result::Guard_Parent, search_result::Guard_Leaf );
1062 pParent = static_cast<internal_node *>( pLeaf );
1063 bRightParent = bRightLeaf;
1064 res.guards.copy( search_result::Guard_updGrandParent, search_result::Guard_updParent );
1065 updGrandParent = updParent;
1066
1067 updParent = search_protect_update( res, pParent->m_pUpdate );
1068
1069 switch ( updParent.bits()) {
1070 case update_desc::DFlag:
1071 case update_desc::Mark:
1072 m_Stat.onSearchRetry();
1073 goto retry;
1074 }
1075
1076 bRightLeaf = !pParent->infinite_key();
1077 pLeaf = protect_child_node( res, pParent, bRightLeaf, updParent );
1078 if ( !pLeaf ) {
1079 m_Stat.onSearchRetry();
