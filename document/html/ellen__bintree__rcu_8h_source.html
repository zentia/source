743 template <typename Func>
744 bool insert( value_type& val, Func f )
745 {
746 check_deadlock_policy::check();
747
748 unique_internal_node_ptr pNewInternal;
749 retired_list updRetire;
750 back_off bkoff;
751
752 {
753 rcu_lock l;
754
755 search_result res;
756 for ( ;; ) {
757 if ( search( res, val, node_compare())) {
758 if ( pNewInternal.get())
759 m_Stat.onInternalNodeDeleted() ; // unique_internal_node_ptr deletes internal node
760 m_Stat.onInsertFailed();
761 return false;
762 }
763
764 if ( res.updGrandParent.bits() == update_desc::Clean && res.updParent.bits() == update_desc::Clean ) {
765 if ( !pNewInternal.get())
766 pNewInternal.reset( alloc_internal_node());
767
768 if ( try_insert( val, pNewInternal.get(), res, updRetire )) {
769 f( val );
770 pNewInternal.release() ; // internal node is linked into the tree and should not be deleted
771 break;
772 }
773 }
774 else
775 help( res.updParent, updRetire );
776
777 bkoff();
778 m_Stat.onInsertRetry();
779 }
780 }
781
782 ++m_ItemCounter;
783 m_Stat.onInsertSuccess();
784
785 return true;
786 }

818 template <typename Func>
819 std::pair<bool, bool> update( value_type& val, Func func, bool bAllowInsert = true )
820 {
821 check_deadlock_policy::check();
822
823 unique_internal_node_ptr pNewInternal;
824 retired_list updRetire;
825 back_off bkoff;
826
827 {
828 rcu_lock l;
829
830 search_result res;
831 for ( ;; ) {
832 if ( search( res, val, node_compare())) {
833 func( false, *node_traits::to_value_ptr( res.pLeaf ), val );
834 if ( pNewInternal.get())
835 m_Stat.onInternalNodeDeleted() ; // unique_internal_node_ptr deletes internal node
836 m_Stat.onUpdateExist();
837 return std::make_pair( true, false );
838 }
839
840 if ( res.updGrandParent.bits() == update_desc::Clean && res.updParent.bits() == update_desc::Clean ) {
841 if ( !bAllowInsert )
842 return std::make_pair( false, false );
843
844 if ( !pNewInternal.get())
845 pNewInternal.reset( alloc_internal_node());
846
847 if ( try_insert( val, pNewInternal.get(), res, updRetire )) {
848 func( true, val, val );
849 pNewInternal.release() ; // internal node is linked into the tree and should not be deleted
850 break;
851 }
852 }
853 else
854 help( res.updParent, updRetire );
855
856 bkoff();
857 m_Stat.onUpdateRetry();
858 }
859 }
860
861 ++m_ItemCounter;
862 m_Stat.onUpdateNew();
863
864 return std::make_pair( true, true );
865 }

1262 internal_node * pGrandParent = nullptr;
