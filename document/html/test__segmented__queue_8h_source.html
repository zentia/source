16 template <typename Queue>
17 void test( Queue& q )
18 {
19 typedef typename Queue::value_type value_type;
20 value_type it;
21
22 const size_t nSize = 100;
23
24 ASSERT_TRUE( q.empty());
25 ASSERT_CONTAINER_SIZE( q, 0 );
26
27 // enqueue/dequeue
28 for ( size_t i = 0; i < nSize; ++i ) {
29 it = static_cast<value_type>(i);
30 ASSERT_TRUE( q.enqueue( it ));
31 ASSERT_CONTAINER_SIZE( q, i + 1 );
32 }
33 ASSERT_FALSE( q.empty());
34 ASSERT_CONTAINER_SIZE( q, nSize );
35
36 for ( size_t i = 0; i < nSize; ++i ) {
37 it = -1;
38 ASSERT_TRUE( q.dequeue( it ));
39 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
40
41 int nSegment = int( i / q.quasi_factor());
42 int nMin = nSegment * int( q.quasi_factor());
43 int nMax = nMin + int( q.quasi_factor()) - 1;
44 EXPECT_LE( nMin, it );
45 EXPECT_LE( it, nMax );
46 }
47 ASSERT_TRUE( q.empty());
48 ASSERT_CONTAINER_SIZE( q, 0 );
49
50 // push/pop
51 for ( size_t i = 0; i < nSize; ++i ) {
52 it = static_cast<value_type>(i);
53 ASSERT_TRUE( q.push( it ));
54 ASSERT_CONTAINER_SIZE( q, i + 1 );
55 }
56 ASSERT_FALSE( q.empty());
57 ASSERT_CONTAINER_SIZE( q, nSize );
58
59 size_t nPushed = nSize;
60 size_t nStartSegment = nPushed / q.quasi_factor();
61 size_t nOffset = nPushed % q.quasi_factor();
62 for ( size_t i = 0; i < nSize; ++i ) {
63 it = -1;
64 ASSERT_TRUE( q.pop( it ));
65 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
66
67 int nSegment = static_cast<int>((i + nPushed) / q.quasi_factor() - nStartSegment );
68 int nMin = nSegment * static_cast<int>( q.quasi_factor());
69 if ( nSegment )
70 nMin -= static_cast<int>( nOffset );
71 int nMax = nMin + static_cast<int>( q.quasi_factor()) - 1;
72 EXPECT_LE( nMin, it );
73 EXPECT_LE( it, nMax );
74 }
75 ASSERT_TRUE( q.empty());
76 ASSERT_CONTAINER_SIZE( q, 0 );
77
78 // push/pop with lambda
79 for ( size_t i = 0; i < nSize; ++i ) {
80 it = static_cast<value_type>(i);
81 EXPECT_NE( it, -1 );
82 auto f = [&it]( value_type& dest ) { dest = it; it = -1; };
83 if ( i & 1 )
84 ASSERT_TRUE( q.enqueue_with( f ));
85 else
86 ASSERT_TRUE( q.push_with( f ));
87 ASSERT_EQ( it, -1 );
88 ASSERT_CONTAINER_SIZE( q, i + 1 );
89 }
90 ASSERT_FALSE( q.empty());
91 ASSERT_CONTAINER_SIZE( q, nSize );
92
93 nPushed += nSize;
94 nStartSegment = nPushed / q.quasi_factor();
95 nOffset = nPushed % q.quasi_factor();
96 for ( size_t i = 0; i < nSize; ++i ) {
97 it = -1;
98 auto f = [&it]( value_type& src ) { it = src; src = -1; };
99 if ( i & 1 )
100 ASSERT_TRUE( q.pop_with( f ));
101 else
102 ASSERT_TRUE( q.dequeue_with( f ));
103 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
104
105 int nSegment = static_cast<int>((i + nPushed) / q.quasi_factor() - nStartSegment);
106 int nMin = nSegment * static_cast<int>(q.quasi_factor());
107 if ( nSegment )
108 nMin -= static_cast<int>(nOffset);
109 int nMax = nMin + static_cast<int>(q.quasi_factor()) - 1;
110 EXPECT_LE( nMin, it );
111 EXPECT_LE( it, nMax );
112 }
113 ASSERT_TRUE( q.empty());
114 ASSERT_CONTAINER_SIZE( q, 0 );
115
116 // clear
117 for ( size_t i = 0; i < nSize; ++i ) {
118 ASSERT_TRUE( q.push( static_cast<value_type>(i)));
119 }
120 ASSERT_FALSE( q.empty());
121 ASSERT_CONTAINER_SIZE( q, nSize );
122
123 q.clear();
124 ASSERT_TRUE( q.empty());
125 ASSERT_CONTAINER_SIZE( q, 0 );
126
127 // pop from empty queue
128 it = nSize * 2;
129 ASSERT_FALSE( q.pop( it ));
130 ASSERT_EQ( it, static_cast<value_type>( nSize * 2 ));
131 ASSERT_TRUE( q.empty());
132 ASSERT_CONTAINER_SIZE( q, 0 );
133
134 ASSERT_FALSE( q.dequeue( it ));
135 ASSERT_EQ( it, static_cast<value_type>( nSize * 2 ));
136 ASSERT_TRUE( q.empty());
137 ASSERT_CONTAINER_SIZE( q, 0 );
138 }

140 template <class Queue>
141 void test_string( Queue& q )
142 {
143 std::string str[3];
144 str[0] = "one";
145 str[1] = "two";
146 str[2] = "three";
147 const size_t nSize = sizeof( str ) / sizeof( str[0] );
148
149 // emplace
150 for ( size_t i = 0; i < nSize; ++i ) {
151 ASSERT_TRUE( q.emplace( str[i].c_str()));
152 ASSERT_CONTAINER_SIZE( q, i + 1 );
153 }
154 ASSERT_FALSE( q.empty());
155 ASSERT_CONTAINER_SIZE( q, nSize );
156
157 {
158 std::string s;
159 auto f = [&s]( std::string& src ) {
160 ASSERT_FALSE( src.empty());
161 s = std::move( src );
162 ASSERT_NE( s, src );
163 };
164 for ( size_t i = 0; i < nSize; ++i ) {
165 if ( i & 1 )
166 ASSERT_TRUE( q.pop_with( f ));
167 else
168 ASSERT_TRUE( q.dequeue_with( f ));
169
170 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
171 ASSERT_TRUE( s == str[0] || s == str[1] || s == str[2] );
172 }
173 }
174 ASSERT_TRUE( q.empty());
175 ASSERT_CONTAINER_SIZE( q, 0 );
176
177
178 // move push
179 for ( size_t i = 0; i < nSize; ++i ) {
180 std::string s = str[i];
181 ASSERT_FALSE( s.empty());
182 if ( i & 1 )
183 ASSERT_TRUE( q.enqueue( std::move( s )));
184 else
185 ASSERT_TRUE( q.push( std::move( s )));
186 ASSERT_TRUE( s.empty());
187 ASSERT_CONTAINER_SIZE( q, i + 1 );
188 }
189 ASSERT_FALSE( q.empty());
190 ASSERT_CONTAINER_SIZE( q, nSize );
191
192 for ( size_t i = 0; i < nSize; ++i ) {
193 std::string s;
194 ASSERT_TRUE( q.pop( s ));
195 ASSERT_CONTAINER_SIZE( q, nSize - i - 1 );
196 ASSERT_TRUE( s == str[0] || s == str[1] || s == str[2] );
197 }
198 ASSERT_TRUE( q.empty());
199 ASSERT_CONTAINER_SIZE( q, 0 );
200 }

202 };
