1178 static size_t parent_bucket( size_t nBucket )
1179 {
1180 assert( nBucket > 0 );
1181 return nBucket & ~(1 << bitop::MSBnz( nBucket ));
1182 }
1183
1184 aux_node_type * init_bucket( size_t const nBucket )
1185 {
1186 assert( nBucket > 0 );
1187 size_t nParent = parent_bucket( nBucket );
1188
1189 aux_node_type * pParentBucket = m_Buckets.bucket( nParent );
1190 if ( pParentBucket == nullptr ) {
1191 pParentBucket = init_bucket( nParent );
1192 m_Stat.onRecursiveInitBucket();
1193 }
1194
1195 assert( pParentBucket != nullptr );
1196
1197 // Allocate an aux node for new bucket
1198 aux_node_type * pBucket = m_Buckets.bucket( nBucket );
1199
1200 back_off bkoff;
1201 for ( ;; pBucket = m_Buckets.bucket( nBucket )) {
1202 if ( pBucket )
1203 return pBucket;
1204
1205 pBucket = alloc_aux_node( split_list::dummy_hash<bit_reversal>( nBucket ));
1206 if ( pBucket ) {
1207 if ( m_List.insert_aux_node( pParentBucket, pBucket )) {
1208 m_Buckets.bucket( nBucket, pBucket );
1209 m_Stat.onNewBucket();
1210 return pBucket;
1211 }
1212
1213 // Another thread set the bucket. Wait while it done
1214 free_aux_node( pBucket );
1215 m_Stat.onBucketInitContenton();
1216 break;
1217 }
1218
1219 // There are no free buckets. It means that the bucket table is full
1220 // Wait while another thread set the bucket or a free bucket will be available
1221 m_Stat.onBucketsExhausted();
1222 bkoff();
1223 }
1224
1225 // Another thread set the bucket. Wait while it done
1226 for ( pBucket = m_Buckets.bucket( nBucket ); pBucket == nullptr; pBucket = m_Buckets.bucket( nBucket )) {
1227 bkoff();
1228 m_Stat.onBusyWaitBucketInit();
1229 }
1230
1231 return pBucket;
1232 }

1247 void init()
1248 {
1249 // GC and OrderedList::gc must be the same
1250 static_assert(std::is_same<gc, typename ordered_list::gc>::value, "GC and OrderedList::gc must be the same");
1251
1252 // atomicity::empty_item_counter is not allowed as a item counter
1253 static_assert(!std::is_same<item_counter, cds::atomicity::empty_item_counter>::value,
1254 "cds::atomicity::empty_item_counter is not allowed as a item counter");
1255
1256 // Initialize bucket 0
1257 aux_node_type * pNode = alloc_aux_node( 0 /*split_list::dummy_hash<bit_reversal>(0)*/ );
1258 assert( pNode != nullptr );
1259
1260 // insert_aux_node cannot return false for empty list
1261 CDS_VERIFY( m_List.insert_aux_node( pNode ));
1262
1263 m_Buckets.bucket( 0, pNode );
1264 }

1266 static size_t max_item_count( size_t nBucketCount, size_t nLoadFactor )
1267 {
1268 return nBucketCount * nLoadFactor;
1269 }
1270
1271 void inc_item_count()
1272 {
1273 size_t nMaxCount = m_nMaxItemCount.load( memory_model::memory_order_relaxed );
1274 if ( ++m_ItemCounter <= nMaxCount )
1275 return;
1276
1277 size_t sz = m_nBucketCountLog2.load( memory_model::memory_order_relaxed );
1278 const size_t nBucketCount = static_cast<size_t>(1) << sz;
1279 if ( nBucketCount < m_Buckets.capacity()) {
1280 // we may grow the bucket table
1281 const size_t nLoadFactor = m_Buckets.load_factor();
1282 if ( nMaxCount < max_item_count( nBucketCount, nLoadFactor ))
1283 return; // someone already have updated m_nBucketCountLog2, so stop here
1284
