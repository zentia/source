102// double-width CAS support - only for libc++
103// You can manually suppress wide-atomic support by defining in compiler command line:
104// for 64bit platform: -DCDS_DISABLE_128BIT_ATOMIC
105// for 32bit platform: -DCDS_DISABLE_64BIT_ATOMIC
106#ifdef _LIBCPP_VERSION
107# if CDS_BUILD_BITS == 64
108# if !defined( CDS_DISABLE_128BIT_ATOMIC ) && defined( __GCC_HAVE_SYNC_COMPARE_AND_SWAP_16 )
109# define CDS_DCAS_SUPPORT
110# endif
111# else
112# if !defined( CDS_DISABLE_64BIT_ATOMIC ) && defined( __GCC_HAVE_SYNC_COMPARE_AND_SWAP_8 )
113# define CDS_DCAS_SUPPORT
114# endif
115# endif
