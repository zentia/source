190/*
191 * Atomic compare and exchange. Compare OLD with MEM, if identical,
192 * store NEW in MEM. Return the initial value in MEM. Success is
193 * indicated by comparing RETURN with OLD.
194 */
195
196#define __xg(x) ((volatile unsigned long *)(x))
197static __inline__ unsigned long cmpxchg8(volatile void *ptr, unsigned long old,
198 unsigned long New)
199{
200 unsigned long prev;
201 asm volatile(LOCK_PREFIX "cmpxchgq %1,%2"
202 : "=a"(prev)
203 : "r"(New), "m"(*__xg(ptr)), "0"(old)
204 : "memory");
205 return prev;
206}
