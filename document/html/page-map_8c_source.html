14// The page-map contains a byte for each 64kb slice in the address space.
15// For an address `a` where `ofs = _mi_page_map[a >> 16]`:
16// 0 = unused
17// 1 = the slice at `a & ~0xFFFF` is a mimalloc page.
18// 1 < ofs <= 127 = the slice is part of a page, starting at `(((a>>16) - ofs - 1) << 16)`.
19//
20// 1 byte per slice => 1 TiB address space needs a 2^14 * 2^16 = 16 MiB page map.
21// A full 256 TiB address space (48 bit) needs a 4 GiB page map.
22// A full 4 GiB address space (32 bit) needs only a 64 KiB page map.
23
24mi_decl_cache_align uint8_t* _mi_page_map = NULL;
25static void* mi_page_map_max_address = NULL;
26static mi_memid_t mi_page_map_memid;
27
28#define MI_PAGE_MAP_ENTRIES_PER_COMMIT_BIT MI_ARENA_SLICE_SIZE
29static mi_bitmap_t* mi_page_map_commit; // one bit per committed 64 KiB entries
30
31static void mi_page_map_ensure_committed(size_t idx, size_t slice_count);

33bool _mi_page_map_init(void) {
34 size_t vbits = (size_t)mi_option_get_clamp(mi_option_max_vabits, 0, MI_SIZE_BITS);
35 if (vbits == 0) {
36 vbits = _mi_os_virtual_address_bits();
37 #if MI_ARCH_X64 // canonical address is limited to the first 128 TiB
38 if (vbits >= 48) { vbits = 47; }
39 #endif
40 }
41
42 // Allocate the page map and commit bits
43 mi_page_map_max_address = (void*)(vbits >= MI_SIZE_BITS ? (SIZE_MAX - MI_ARENA_SLICE_SIZE + 1) : (MI_PU(1) << vbits));
44 const size_t page_map_size = (MI_ZU(1) << (vbits - MI_ARENA_SLICE_SHIFT));
45 const bool commit = (page_map_size <= 1*MI_MiB || mi_option_is_enabled(mi_option_pagemap_commit)); // _mi_os_has_overcommit(); // commit on-access on Linux systems?
46 const size_t commit_bits = _mi_divide_up(page_map_size, MI_PAGE_MAP_ENTRIES_PER_COMMIT_BIT);
47 const size_t bitmap_size = (commit ? 0 : mi_bitmap_size(commit_bits, NULL));
48 const size_t reserve_size = bitmap_size + page_map_size;
49 uint8_t* const base = (uint8_t*)_mi_os_alloc_aligned(reserve_size, 1, commit, true /* allow large */, &mi_page_map_memid);
50 if (base==NULL) {
51 _mi_error_message(ENOMEM, "unable to reserve virtual memory for the page map (%zu KiB)\n", page_map_size / MI_KiB);
52 return false;
53 }
54 if (mi_page_map_memid.initially_committed && !mi_page_map_memid.initially_zero) {
55 _mi_warning_message("internal: the page map was committed but not zero initialized!\n");
56 _mi_memzero_aligned(base, reserve_size);
57 }
58 if (bitmap_size > 0) {
59 mi_page_map_commit = (mi_bitmap_t*)base;
60 _mi_os_commit(mi_page_map_commit, bitmap_size, NULL);
61 mi_bitmap_init(mi_page_map_commit, commit_bits, true);
62 }
63 _mi_page_map = base + bitmap_size;
64
65 // commit the first part so NULL pointers get resolved without an access violation
66 if (!commit) {
67 mi_page_map_ensure_committed(0, 1);
68 }
69 _mi_page_map[0] = 1; // so _mi_ptr_page(NULL) == NULL
70 mi_assert_internal(_mi_ptr_page(NULL)==NULL);
71 return true;
72}

74static void mi_page_map_ensure_committed(size_t idx, size_t slice_count) {
75 // is the page map area that contains the page address committed?
76 // we always set the commit bits so we can track what ranges are in-use.
77 // we only actually commit if the map wasn't committed fully already.
78 if (mi_page_map_commit != NULL) {
79 const size_t commit_idx = idx / MI_PAGE_MAP_ENTRIES_PER_COMMIT_BIT;
80 const size_t commit_idx_hi = (idx + slice_count - 1) / MI_PAGE_MAP_ENTRIES_PER_COMMIT_BIT;
81 for (size_t i = commit_idx; i <= commit_idx_hi; i++) { // per bit to avoid crossing over bitmap chunks
82 if (mi_bitmap_is_clear(mi_page_map_commit, i)) {
83 // this may race, in which case we do multiple commits (which is ok)
84 bool is_zero;
85 uint8_t* const start = _mi_page_map + (i * MI_PAGE_MAP_ENTRIES_PER_COMMIT_BIT);
86 const size_t size = MI_PAGE_MAP_ENTRIES_PER_COMMIT_BIT;
87 _mi_os_commit(start, size, &is_zero);
88 if (!is_zero && !mi_page_map_memid.initially_zero) { _mi_memzero(start, size); }
89 mi_bitmap_set(mi_page_map_commit, i);
90 }
91 }
92 }
93 #if MI_DEBUG > 0
94 _mi_page_map[idx] = 0;
95 _mi_page_map[idx+slice_count-1] = 0;
96 #endif
97}

108void _mi_page_map_register(mi_page_t* page) {
109 mi_assert_internal(page != NULL);
110 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
111 mi_assert_internal(_mi_page_map != NULL); // should be initialized before multi-thread access!
112 if mi_unlikely(_mi_page_map == NULL) {
113 if (!_mi_page_map_init()) return;
114 }
115 mi_assert(_mi_page_map!=NULL);
116 uint8_t* page_start;
117 size_t slice_count;
118 const size_t idx = mi_page_map_get_idx(page, &page_start, &slice_count);
119
120 mi_page_map_ensure_committed(idx, slice_count);
121
122 // set the offsets
123 for (size_t i = 0; i < slice_count; i++) {
124 mi_assert_internal(i < 128);
125 _mi_page_map[idx + i] = (uint8_t)(i+1);
126 }
127}

147mi_page_t* _mi_safe_ptr_page(const void* p) {
148 if mi_unlikely(p >= mi_page_map_max_address) return NULL;
149 const uintptr_t idx = _mi_page_map_index(p);

154}

175bool _mi_page_map_init(void) {
176 size_t vbits = (size_t)mi_option_get_clamp(mi_option_max_vabits, 0, MI_SIZE_BITS);
177 if (vbits == 0) {
178 vbits = _mi_os_virtual_address_bits();
179 #if MI_ARCH_X64 // canonical address is limited to the first 128 TiB
180 if (vbits >= 48) { vbits = 47; }
181 #endif
182 }
183
184 // Allocate the page map and commit bits
185 mi_assert(MI_MAX_VABITS >= vbits);
186 mi_page_map_max_address = (void*)(vbits >= MI_SIZE_BITS ? (SIZE_MAX - MI_ARENA_SLICE_SIZE + 1) : (MI_PU(1) << vbits));
187 const size_t page_map_count = (MI_ZU(1) << (vbits - MI_PAGE_MAP_SUB_SHIFT - MI_ARENA_SLICE_SHIFT));
188 mi_assert(page_map_count <= MI_PAGE_MAP_COUNT);
189 const size_t os_page_size = _mi_os_page_size();
190 const size_t page_map_size = _mi_align_up( page_map_count * sizeof(mi_page_t**), os_page_size);
191 const size_t reserve_size = page_map_size + os_page_size;
192 const bool commit = page_map_size <= 64*MI_KiB ||
193 mi_option_is_enabled(mi_option_pagemap_commit) || _mi_os_has_overcommit();
194 _mi_page_map = (mi_page_t***)_mi_os_alloc_aligned(reserve_size, 1, commit, true /* allow large */, &mi_page_map_memid);
195 if (_mi_page_map==NULL) {
196 _mi_error_message(ENOMEM, "unable to reserve virtual memory for the page map (%zu KiB)\n", page_map_size / MI_KiB);
197 return false;
198 }
199 if (mi_page_map_memid.initially_committed && !mi_page_map_memid.initially_zero) {
200 _mi_warning_message("internal: the page map was committed but not zero initialized!\n");
201 _mi_memzero_aligned(_mi_page_map, page_map_size);
202 }
203 mi_atomic_store_release(&mi_page_map_commit, (commit ? ~MI_ZU(0) : MI_ZU(0)));
204
205 // note: for the NULL range we only commit one OS page (in the map and sub)
206 if (!mi_page_map_memid.initially_committed) {
207 _mi_os_commit(&_mi_page_map[0], os_page_size, NULL); // commit first part of the map
208 }
209 _mi_page_map[0] = (mi_page_t**)((uint8_t*)_mi_page_map + page_map_size); // we reserved 2 sub maps at the end already
210 if (!mi_page_map_memid.initially_committed) {
211 _mi_os_commit(_mi_page_map[0], os_page_size, NULL); // only first OS page
212 }
213 _mi_page_map[0][0] = (mi_page_t*)&_mi_page_empty; // caught in `mi_free`
214
215 mi_assert_internal(_mi_ptr_page(NULL)==&_mi_page_empty);
216 return true;
217}

222static inline bool mi_page_map_is_committed(size_t idx, size_t* pbit_idx) {
223 mi_bfield_t commit = mi_atomic_load_relaxed(&mi_page_map_commit);
224 const size_t bit_idx = idx/MI_PAGE_MAP_ENTRIES_PER_CBIT;
225 mi_assert_internal(bit_idx < MI_BFIELD_BITS);
226 if (pbit_idx != NULL) { *pbit_idx = bit_idx; }
227 return ((commit & (MI_ZU(1) << bit_idx)) != 0);
228}

230static mi_page_t** mi_page_map_ensure_committed(size_t idx) {
231 size_t bit_idx;
232 if mi_unlikely(!mi_page_map_is_committed(idx, &bit_idx)) {
233 uint8_t* start = (uint8_t*)&_mi_page_map[bit_idx * MI_PAGE_MAP_ENTRIES_PER_CBIT];
234 _mi_os_commit(start, MI_PAGE_MAP_ENTRIES_PER_CBIT * sizeof(mi_page_t**), NULL);
235 mi_atomic_or_acq_rel(&mi_page_map_commit, MI_ZU(1) << bit_idx);
236 }
237 return _mi_page_map[idx];
238}

240static mi_page_t** mi_page_map_ensure_at(size_t idx) {
241 mi_page_t** sub = mi_page_map_ensure_committed(idx);
242 if mi_unlikely(sub == NULL) {
243 // sub map not yet allocated, alloc now
244 mi_memid_t memid;
245 sub = (mi_page_t**)_mi_os_alloc(MI_PAGE_MAP_SUB_COUNT * sizeof(mi_page_t*), &memid);
246 mi_page_t** expect = NULL;
247 if (!mi_atomic_cas_strong_acq_rel(((_Atomic(mi_page_t**)*)&_mi_page_map[idx]), &expect, sub)) {
248 // another thread already allocated it.. free and continue
249 _mi_os_free(sub, MI_PAGE_MAP_SUB_COUNT * sizeof(mi_page_t*), memid);
250 sub = expect;
251 mi_assert_internal(sub!=NULL);
252 }
253 if (sub == NULL) {
254 _mi_error_message(EFAULT, "internal error: unable to extend the page map\n");
255 }
256 }
257 return sub;
258}

260static void mi_page_map_set_range(mi_page_t* page, size_t idx, size_t sub_idx, size_t slice_count) {
261 // is the page map area that contains the page address committed?
262 while (slice_count > 0) {
263 mi_page_t** sub = mi_page_map_ensure_at(idx);
264 // set the offsets for the page
265 while (sub_idx < MI_PAGE_MAP_SUB_COUNT) {
266 sub[sub_idx] = page;
267 slice_count--; if (slice_count == 0) return;
268 sub_idx++;
269 }
270 idx++; // potentially wrap around to the next idx
271 sub_idx = 0;
272 }
273}
