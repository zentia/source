139 static void CopyFunctor(FunctorStorageType& to, const FunctorStorageType& from)

145 }

152 static void* Manager(void* to,
153 void* from,
154 typename function_base_detail::ManagerOperations ops) EA_NOEXCEPT
155 {
156 if constexpr (isCopyable)
157 {
158 switch (ops)
159 {
160 case MGROPS_DESTRUCT_FUNCTOR:
161 {
162 DestructFunctor(*static_cast<FunctorStorageType*>(to));
163 }
164 break;
165 case MGROPS_COPY_FUNCTOR:
166 {
167
168 CopyFunctor(*static_cast<FunctorStorageType*>(to),
169 *static_cast<const FunctorStorageType*>(from));
170 }
171 break;
172
173 case MGROPS_MOVE_FUNCTOR:
174 {
175 MoveFunctor(*static_cast<FunctorStorageType*>(to),
176 *static_cast<FunctorStorageType*>(from));
177 DestructFunctor(*static_cast<FunctorStorageType*>(from));
178 }
179 break;
180 default:
181 break;
182 }
183 }
184 else
185 {
186 switch (ops)
187 {
188 case MGROPS_DESTRUCT_FUNCTOR:
189 {
190 DestructFunctor(*static_cast<FunctorStorageType*>(to));
191 }
192 break;
193 case MGROPS_MOVE_FUNCTOR:
194 {
195 MoveFunctor(*static_cast<FunctorStorageType*>(to),
196 *static_cast<FunctorStorageType*>(from));
197 DestructFunctor(*static_cast<FunctorStorageType*>(from));
198 }
199 break;
200 default:
201 break;
202 }
203 }
204
205 return nullptr;

227 template <typename T>
228 static void CreateFunctor(FunctorStorageType& storage, T&& functor)
229 {
230 auto& allocator = *EASTLAllocatorDefault();
231 Functor* func = static_cast<Functor*>(allocator.allocate(sizeof(Functor), alignof(Functor), 0));
232
233#if EASTL_EXCEPTIONS_ENABLED
234 if (!func)
235 {
236 throw std::bad_alloc();
237 }
238#else
239 EASTL_ASSERT_MSG(func != nullptr, "Allocation failed!");
240#endif
241
242 ::new (static_cast<void*>(func)) Functor(eastl::forward<T>(functor));
243 GetFunctorPtrRef(storage) = func;
244 }
245
246 static void DestructFunctor(FunctorStorageType& storage)
247 {
248 Functor* func = GetFunctorPtr(storage);
249 if (func)
250 {
251 auto& allocator = *EASTLAllocatorDefault();
252 func->~Functor();
253 allocator.deallocate(static_cast<void*>(func), sizeof(Functor));
254 }
255 }
256
257 static void CopyFunctor(FunctorStorageType& to, const FunctorStorageType& from)
258 {
259 if constexpr (isCopyable)
260 {
261 auto& allocator = *EASTLAllocatorDefault();
262 Functor* func = static_cast<Functor*>(allocator.allocate(sizeof(Functor), alignof(Functor), 0));
263#if EASTL_EXCEPTIONS_ENABLED
264 if (!func)
265 {
266 throw std::bad_alloc();
267 }
268#else
269 EASTL_ASSERT_MSG(func != nullptr, "Allocation failed!");
270#endif
271 ::new (static_cast<void*>(func)) Functor(*GetFunctorPtr(from));
272 GetFunctorPtrRef(to) = func;
273 }
274 }

276 static void MoveFunctor(FunctorStorageType& to, FunctorStorageType& from) EA_NOEXCEPT
277 {
278 Functor* func = GetFunctorPtr(from);
279 GetFunctorPtrRef(to) = func;
280 GetFunctorPtrRef(from) = nullptr;
281 }

283 static void* Manager(void* to,
284 void* from,
285 typename function_base_detail::ManagerOperations ops) EA_NOEXCEPT
286 {
287 switch (ops)
288 {
289 case MGROPS_DESTRUCT_FUNCTOR:
290 {
291 DestructFunctor(*static_cast<FunctorStorageType*>(to));
292 }
293 break;
294 case MGROPS_COPY_FUNCTOR:
295 {
296 if constexpr (isCopyable)
297 {
298 CopyFunctor(*static_cast<FunctorStorageType*>(to),
299 *static_cast<const FunctorStorageType*>(from));
300 }
301 }
302 break;
303 case MGROPS_MOVE_FUNCTOR:
304 {
305 MoveFunctor(*static_cast<FunctorStorageType*>(to), *static_cast<FunctorStorageType*>(from));
306 // Moved ptr, no need to destruct ourselves
307 }
308 break;
309 default:
310 break;
311 }
312 return nullptr;
313 }

663 template <typename Functor>
664 void CreateForwardFunctor(Functor&& functor)
665 {
666 using DecayedFunctorType = typename eastl::decay<Functor>::type;
667 using FunctionManagerType =
668 typename Base::template function_manager<DecayedFunctorType, isCopyable, R, Args...>;
669
670 if (internal::is_null(functor))
671 {
672 mMgrFuncPtr = nullptr;
673 mInvokeFuncPtr = &DefaultInvoker;
674 }
675 else
676 {
677 mMgrFuncPtr = &FunctionManagerType::Manager;
678 mInvokeFuncPtr = &FunctionManagerType::Invoker;
679 FunctionManagerType::CreateFunctor(mStorage, eastl::forward<Functor>(functor));
680 }
681 }
