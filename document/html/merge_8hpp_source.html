73template<
74 cudaMergeBoundType bounds = cudaMergeBoundType::LOWER,
75 typename a_keys_it, typename b_keys_it, typename comp_t
76>
77__device__ auto cuda_merge_path(
78 a_keys_it a_keys, unsigned a_count,
79 b_keys_it b_keys, unsigned b_count,
80 unsigned diag, comp_t comp
81) {
82
83 unsigned beg = (diag > b_count) ? diag - b_count : 0;
84 unsigned end = diag < a_count ? diag : a_count;
85
86 while(beg < end) {
87 auto mid = (beg + end) / 2;
88 auto a_key = a_keys[mid];
89 auto b_key = b_keys[diag - 1 - mid];
90 bool pred = (cudaMergeBoundType::UPPER == bounds) ?
91 comp(a_key, b_key) :
92 !comp(b_key, a_key);
93
94 if(pred) beg = mid + 1;
95 else end = mid;
96 }
97 return beg;
98}

113template<cudaMergeBoundType bounds, bool range_check, typename T, typename comp_t>
114__device__ bool cuda_merge_predicate(
115 T a_key, T b_key, cudaMergeRange range, comp_t comp
116) {
117
118 bool p;
119 if(range_check && !range.a_valid()) {
120 p = false;
121 }
122 else if(range_check && !range.b_valid()) {
123 p = true;
124 }
125 else {
126 p = (cudaMergeBoundType::UPPER == bounds) ? comp(a_key, b_key) :
127 !comp(b_key, a_key);

252template<cudaMergeBoundType bounds, unsigned vt, typename T, typename comp_t>
253__device__ auto cuda_serial_merge(
254 const T* keys_shared, cudaMergeRange range, comp_t comp, bool sync = true
255) {
256
257 auto a_key = keys_shared[range.a_begin];
258 auto b_key = keys_shared[range.b_begin];
259
260 cudaMergePair<T, vt> merge_pair;
261 cuda_iterate<vt>([&](auto i) {
262 bool p = cuda_merge_predicate<bounds, true>(a_key, b_key, range, comp);
263 auto index = p ? range.a_begin : range.b_begin;
264
265 merge_pair.keys[i] = p ? a_key : b_key;
266 merge_pair.indices[i] = index;
267
268 T c_key = keys_shared[++index];
269 if(p) a_key = c_key, range.a_begin = index;
270 else b_key = c_key, range.b_begin = index;
271 });
272
273 if(sync) __syncthreads();
274 return merge_pair;
275}

282template<cudaMergeBoundType bounds,
283 unsigned nt, unsigned vt,
284 typename a_it, typename b_it, typename T, typename comp_t, unsigned S
285>
286__device__ auto block_merge_from_mem(
287 a_it a, b_it b, cudaMergeRange range_mem, unsigned tid, comp_t comp, T (&keys_shared)[S]
288) {
289
290 static_assert(S >= nt * vt + 1,
291 "block_merge_from_mem requires temporary storage of at "
292 "least nt * vt + 1 items");
293
294 // Load the data into shared memory.
295 cuda_load_two_streams_shared<nt, vt>(
296 a + range_mem.a_begin, range_mem.a_count(),
297 b + range_mem.b_begin, range_mem.b_count(),
298 tid, keys_shared, true
299 );
300
301 // Run a merge path to find the start of the serial merge for each thread.
302 auto range_local = range_mem.to_local();
303 auto diag = vt * tid;
304 auto mp = cuda_merge_path<bounds>(keys_shared, range_local, diag, comp);
305
306 // Compute the ranges of the sources in shared memory. The end iterators
307 // of the range are inaccurate, but still facilitate exact merging, because
308 // only vt elements will be merged.
309 auto merged = cuda_serial_merge<bounds, vt>(
310 keys_shared, range_local.partition(mp, diag), comp
311 );
312
313 return merged;
314};

374template<
375 typename P,
376 typename a_keys_it, typename a_vals_it,
377 typename b_keys_it, typename b_vals_it,
378 typename c_keys_it, typename c_vals_it,
379 typename comp_t
380>
381void cuda_merge_loop(
382 P&& p,
383 a_keys_it a_keys, a_vals_it a_vals, unsigned a_count,
384 b_keys_it b_keys, b_vals_it b_vals, unsigned b_count,
385 c_keys_it c_keys, c_vals_it c_vals,
386 comp_t comp,
387 void* ptr
388) {
389
390 using E = std::decay_t<P>;
391 using T = typename std::iterator_traits<a_keys_it>::value_type;
392 using V = typename std::iterator_traits<a_vals_it>::value_type;
393
394 auto buf = static_cast<unsigned*>(ptr);
395
396 auto has_values = !std::is_same<V, cudaEmpty>::value;
397
398 cuda_merge_path_partitions<cudaMergeBoundType::LOWER>(
399 p, a_keys, a_count, b_keys, b_count, E::nv, comp, buf
400 );
401
402 unsigned B = p.num_blocks(a_count + b_count);
403
404 // we use small kernel
405 cuda_kernel<<<B, E::nt, 0, p.stream()>>>([=] __device__ (auto tid, auto bid) {
406
407 __shared__ union {
408 T keys[E::nv + 1];
409 unsigned indices[E::nv];
410 } shared;
411
412 // Load the range for this CTA and merge the values into register.
413 auto mp0 = buf[bid + 0];
414 auto mp1 = buf[bid + 1];
415 auto range = cuda_compute_merge_range(a_count, b_count, bid, E::nv, mp0, mp1);
416
417 auto merge = block_merge_from_mem<cudaMergeBoundType::LOWER, E::nt, E::vt>(
418 a_keys, b_keys, range, tid, comp, shared.keys
419 );
420
421 auto dest_offset = E::nv * bid;
422 cuda_reg_to_mem_thread<E::nt>(
423 merge.keys, tid, range.total(), c_keys + dest_offset, shared.keys
424 );
425
426 if(has_values) {
427 // Transpose the indices from thread order to strided order.
428 auto indices = cuda_reg_thread_to_strided<E::nt>(
429 merge.indices, tid, shared.indices
430 );
431
432 // Gather the input values and merge into the output values.
433 cuda_transfer_two_streams_strided<E::nt>(
434 a_vals + range.a_begin, range.a_count(),
435 b_vals + range.b_begin, range.b_count(), indices, tid,
436 c_vals + dest_offset
437 );
438 }
439 });
440}
