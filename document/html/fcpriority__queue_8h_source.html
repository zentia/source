152 bool push(
153 value_type const& val
154 )
155 {
156 auto pRec = m_FlatCombining.acquire_record();
157 pRec->pValPush = &val;
158
159 m_FlatCombining.combine( op_push, pRec, *this );
160
161 assert( pRec->is_done());

163 m_FlatCombining.internal_statistics().onPush();

278 public: // flat combining cooperation, not for direct use!
279 //@cond
280 /*
281 The function is called by \ref cds::algo::flat_combining::kernel "flat combining kernel"
282 object if the current thread becomes a combiner. Invocation of the function means that
283 the priority queue should perform an action recorded in \p pRec.
284 */
285 void fc_apply( fc_record * pRec )
286 {
287 assert( pRec );
288
289 // this function is called under FC mutex, so switch TSan off
290 //CDS_TSAN_ANNOTATE_IGNORE_RW_BEGIN;
291
292 switch ( pRec->op()) {
293 case op_push:
294 assert( pRec->pValPush );
295 m_PQueue.push( *(pRec->pValPush));
296 break;
297 case op_push_move:
298 assert( pRec->pValPush );
299 m_PQueue.push( std::move( *(pRec->pValPush )));
300 break;
301 case op_pop:
302 assert( pRec->pValPop );
303 pRec->bEmpty = m_PQueue.empty();
304 if ( !pRec->bEmpty ) {
305 *(pRec->pValPop) = std::move( m_PQueue.top());
306 m_PQueue.pop();
307 }
308 break;
309 case op_clear:
310 while ( !m_PQueue.empty())
311 m_PQueue.pop();
312 break;
313 default:
314 assert(false);
315 break;
316 }
317
318 //CDS_TSAN_ANNOTATE_IGNORE_RW_END;
319 }
320 //@endcond
321 };
