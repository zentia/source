330 bool is_power_of_2( T val ) {
331 size_t intval = size_t(val);
332 return (intval&(intval-1)) == size_t(0);
333 }
334 int step_function_plus(int previous, double step){
335 return static_cast<int>(previous+step);
336 }
337 int step_function_multiply(int previous, double multiply){
338 return static_cast<int>(previous*multiply);
339 }
340 // "Power-of-2 ladder": nsteps is the desired number of steps between any subsequent powers of 2.
341 // The actual step is the quotient of the nearest smaller power of 2 divided by that number (but at least 1).
342 // E.g., '1:32:#4' means 1,2,3,4,5,6,7,8,10,12,14,16,20,24,28,32
343 int step_function_power2_ladder(int previous, double nsteps){
344 int steps = int(nsteps);
345 assert( is_power_of_2(steps) ); // must be a power of 2
346 // The actual step is 1 until the value is twice as big as nsteps
347 if( previous < 2*steps )
348 return previous+1;
349 // calculate the previous power of 2
350 int prev_power2 = previous/2; // start with half the given value
351 int rshift = 1; // and with the shift of 1;
352 while( int shifted = prev_power2>>rshift ) { // shift the value right; while the result is non-zero,
353 prev_power2 |= shifted; // add the bits set in 'shifted';
354 rshift <<= 1; // double the shift, as twice as many top bits are set;
355 } // repeat.
356 ++prev_power2; // all low bits set; now it's just one less than the desired power of 2
357 assert( is_power_of_2(prev_power2) );
358 assert( (prev_power2<=previous)&&(2*prev_power2>previous) );
359 // The actual step value is the previous power of 2 divided by steps
360 return previous + (prev_power2/steps);
361 }
