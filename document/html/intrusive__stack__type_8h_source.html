30 template < typename T, typename Stack, typename Lock>
31 class StdStack
32 {
33 Stack m_Impl;
34 mutable Lock m_Lock;
35 cds::intrusive::treiber_stack::empty_stat m_stat;
36
37 typedef std::unique_lock<Lock> unique_lock;
38
39 public:
40 typedef T value_type;
41
42 bool push( T& v )
43 {
44 unique_lock l( m_Lock );
45 m_Impl.push( &v );
46 return true;
47 }
48
49 T * pop()
50 {
51 unique_lock l( m_Lock );
52 if ( !m_Impl.empty()) {
53 T * v = m_Impl.top();
54 m_Impl.pop();
55 return v;
56 }
57 return nullptr;
58 }
59
60 bool empty() const
61 {
62 unique_lock l( m_Lock );
63 return m_Impl.empty();
64 }
65
66 cds::intrusive::treiber_stack::empty_stat const& statistics() const
67 {
68 return m_stat;
69 }
70 };
71 }
