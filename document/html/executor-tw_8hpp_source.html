1617// Procedure: _invoke
1618inline void Executor::_invoke(Worker& worker, Node* node) {
1619
1620 // synchronize all outstanding memory operations caused by reordering
1621 while(!(node->_state.load(std::memory_order_acquire) & Node::READY));
1622
1623 begin_invoke:
1624
1625 SmallVector<int> conds;
1626
1627 // no need to do other things if the topology is cancelled
1628 if(node->_is_cancelled()) {
1629 _tear_down_invoke(worker, node);
1630 return;
1631 }
1632
1633 // if acquiring semaphore(s) exists, acquire them first
1634 if(node->_semaphores && !node->_semaphores->to_acquire.empty()) {
1635 SmallVector<Node*> nodes;
1636 if(!node->_acquire_all(nodes)) {
1637 _schedule(worker, nodes);
1638 return;
1639 }
1640 node->_state.fetch_or(Node::ACQUIRED, std::memory_order_release);
1641 }
1642
1643 // condition task
1644 //int cond = -1;
1645
1646 // switch is faster than nested if-else due to jump table
1647 switch(node->_handle.index()) {
1648 // static task
1649 case Node::STATIC:{
1650 _invoke_static_task(worker, node);
1651 }
1652 break;
1653
1654 // subflow task
1655 case Node::SUBFLOW: {
1656 _invoke_subflow_task(worker, node);
1657 }
1658 break;
1659
1660 // condition task
1661 case Node::CONDITION: {
1662 _invoke_condition_task(worker, node, conds);
1663 }
1664 break;
1665
1666 // multi-condition task
1667 case Node::MULTI_CONDITION: {
1668 _invoke_multi_condition_task(worker, node, conds);
1669 }
1670 break;
1671
1672 // module task
1673 case Node::MODULE: {
1674 _invoke_module_task(worker, node);
1675 }
1676 break;
1677
1678 // async task
1679 case Node::ASYNC: {
1680 _invoke_async_task(worker, node);
1681 _tear_down_async(node);
1682 return ;
1683 }
1684 break;
1685
1686 // dependent async task
1687 case Node::DEPENDENT_ASYNC: {
1688 _invoke_dependent_async_task(worker, node);
1689 _tear_down_dependent_async(worker, node);
1690 if(worker._cache) {
1691 node = worker._cache;
1692 goto begin_invoke;
1693 }
1694 return;
1695 }
1696 break;
1697
1698 // monostate (placeholder)
1699 default:
1700 break;
1701 }
1702
1703 //invoke_successors:
1704
1705 // if releasing semaphores exist, release them
1706 if(node->_semaphores && !node->_semaphores->to_release.empty()) {
1707 _schedule(worker, node->_release_all());
1708 }
1709
1710 // Reset the join counter to support the cyclic control flow.
1711 // + We must do this before scheduling the successors to avoid race
1712 // condition on _dependents.
1713 // + We must use fetch_add instead of direct assigning
1714 // because the user-space call on "invoke" may explicitly schedule
1715 // this task again (e.g., pipeline) which can access the join_counter.
1716 if((node->_state.load(std::memory_order_relaxed) & Node::CONDITIONED)) {
1717 node->_join_counter.fetch_add(node->num_strong_dependents(), std::memory_order_relaxed);
1718 }
1719 else {
1720 node->_join_counter.fetch_add(node->num_dependents(), std::memory_order_relaxed);
1721 }
1722
1723 // acquire the parent flow counter
1724 auto& j = (node->_parent) ? node->_parent->_join_counter :
1725 node->_topology->_join_counter;
1726
1727 // Here, we want to cache the latest successor with the highest priority
1728 worker._cache = nullptr;
1729 auto max_p = static_cast<unsigned>(TaskPriority::MAX);
1730
1731 // Invoke the task based on the corresponding type
1732 switch(node->_handle.index()) {
1733
1734 // condition and multi-condition tasks
1735 case Node::CONDITION:
1736 case Node::MULTI_CONDITION: {
1737 for(auto cond : conds) {
1738 if(cond >= 0 && static_cast<size_t>(cond) < node->_successors.size()) {
1739 auto s = node->_successors[cond];
1740 // zeroing the join counter for invariant
1741 s->_join_counter.store(0, std::memory_order_relaxed);
1742 j.fetch_add(1, std::memory_order_relaxed);
1743 if(s->_priority <= max_p) {
1744 if(worker._cache) {
1745 _schedule(worker, worker._cache);
1746 }
1747 worker._cache = s;
1748 max_p = s->_priority;
1749 }
1750 else {
1751 _schedule(worker, s);
1752 }
1753 }
1754 }
1755 }
1756 break;
1757
1758 // non-condition task
1759 default: {
1760 for(size_t i=0; i<node->_successors.size(); ++i) {
1761 //if(auto s = node->_successors[i]; --(s->_join_counter) == 0) {
1762 if(auto s = node->_successors[i];
1763 s->_join_counter.fetch_sub(1, std::memory_order_acq_rel) == 1) {
1764 j.fetch_add(1, std::memory_order_relaxed);
1765 if(s->_priority <= max_p) {
1766 if(worker._cache) {
1767 _schedule(worker, worker._cache);
1768 }
1769 worker._cache = s;
1770 max_p = s->_priority;
1771 }
1772 else {
1773 _schedule(worker, s);
1774 }
1775 }
1776 }
1777 }
1778 break;
1779 }
1780
1781 // tear_down the invoke
1782 _tear_down_invoke(worker, node);
1783
1784 // perform tail recursion elimination for the right-most child to reduce
1785 // the number of expensive pop/push operations through the task queue
1786 if(worker._cache) {
1787 node = worker._cache;
1788 //node->_state.fetch_or(Node::READY, std::memory_order_release);
1789 goto begin_invoke;
1790 }
1791}
