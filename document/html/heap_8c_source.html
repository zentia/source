23// Visit all pages in a heap; returns `false` if break was called.
24static bool mi_heap_visit_pages(mi_heap_t* heap, heap_page_visitor_fun* fn, void* arg1, void* arg2)
25{
26 if (heap==NULL || heap->page_count==0) return 0;
27
28 // visit all pages
29 #if MI_DEBUG>1
30 size_t total = heap->page_count;
31 size_t count = 0;
32 #endif
33
34 for (size_t i = 0; i <= MI_BIN_FULL; i++) {
35 mi_page_queue_t* pq = &heap->pages[i];
36 mi_page_t* page = pq->first;
37 while(page != NULL) {
38 mi_page_t* next = page->next; // save next in case the page gets removed from the queue
39 mi_assert_internal(mi_page_heap(page) == heap);
40 #if MI_DEBUG>1
41 count++;
42 #endif
43 if (!fn(heap, pq, page, arg1, arg2)) return false;
44 page = next; // and continue
45 }
46 }
47 mi_assert_internal(count == total);
48 return true;
49}

77 "Collect" pages by migrating `local_free` and `thread_free`
78 lists and freeing empty pages. This is done when a thread
79 stops (and in that case abandons pages if there are still
80 blocks alive)

83typedef enum mi_collect_e {
84 MI_NORMAL,
85 MI_FORCE,
86 MI_ABANDON
87} mi_collect_t;

90static bool mi_heap_page_collect(mi_heap_t* heap, mi_page_queue_t* pq, mi_page_t* page, void* arg_collect, void* arg2 ) {
91 MI_UNUSED(arg2);
92 MI_UNUSED(heap);
93 mi_assert_internal(mi_heap_page_is_valid(heap, pq, page, NULL, NULL));
94 mi_collect_t collect = *((mi_collect_t*)arg_collect);
95 _mi_page_free_collect(page, collect >= MI_FORCE);
96 if (mi_page_all_free(page)) {
97 // no more used blocks, free the page.
98 // note: this will free retired pages as well.
99 _mi_page_free(page, pq);
100 }
101 else if (collect == MI_ABANDON) {
102 // still used blocks but the thread is done; abandon the page
103 _mi_page_abandon(page, pq);
104 }
105 return true; // don't break
106}

109static void mi_heap_collect_ex(mi_heap_t* heap, mi_collect_t collect)
110{
111 if (heap==NULL || !mi_heap_is_initialized(heap)) return;
112 mi_assert_expensive(mi_heap_is_valid(heap));
113
114 const bool force = (collect >= MI_FORCE);
115 _mi_deferred_free(heap, force);
116
117 // python/cpython#112532: we may be called from a thread that is not the owner of the heap
118 // const bool is_main_thread = (_mi_is_main_thread() && heap->thread_id == _mi_thread_id());
119
120 // collect retired pages
121 _mi_heap_collect_retired(heap, force);
122
123 // if (_mi_is_main_thread()) { mi_debug_show_arenas(true, false, false); }
124
125 // collect all pages owned by this thread
126 mi_heap_visit_pages(heap, &mi_heap_page_collect, &collect, NULL);
127
128 // collect arenas (this is program wide so don't force purges on abandonment of threads)
129 //mi_atomic_storei64_release(&heap->tld->subproc->purge_expire, 1);
130 _mi_arenas_collect(collect == MI_FORCE /* force purge? */, collect >= MI_FORCE /* visit all? */, heap->tld);
131
132 // merge statistics
133 if (collect <= MI_FORCE) {
134 mi_stats_merge();
135 }
136}

178// todo: make order of parameters consistent (but would that break compat with CPython?)
179void _mi_heap_init(mi_heap_t* heap, mi_arena_id_t arena_id, bool allow_destroy, uint8_t heap_tag, mi_tld_t* tld)
180{
181 mi_assert_internal(heap!=NULL);
182 mi_memid_t memid = heap->memid;
183 _mi_memcpy_aligned(heap, &_mi_heap_empty, sizeof(mi_heap_t));
184 heap->memid = memid;
185 heap->tld = tld; // avoid reading the thread-local tld during initialization
186 heap->tag = heap_tag;
187 heap->numa_node = tld->numa_node;
188 heap->exclusive_arena = _mi_arena_from_id(arena_id);
189 heap->allow_page_reclaim = (!allow_destroy && mi_option_get(mi_option_page_reclaim_on_free) >= 0);
190 heap->allow_page_abandon = (!allow_destroy && mi_option_get(mi_option_page_full_retain) >= 0);
191 heap->page_full_retain = mi_option_get_clamp(mi_option_page_full_retain, -1, 32);
192 if (heap->tld->is_in_threadpool) {
193 // if we run as part of a thread pool it is better to not arbitrarily reclaim abandoned pages into our heap.
194 // this is checked in `free.c:mi_free_try_collect_mt`
195 // .. but abandoning is good in this case: halve the full page retain (possibly to 0)
196 // (so blocked threads do not hold on to too much memory)
197 if (heap->page_full_retain > 0) {
198 heap->page_full_retain = heap->page_full_retain / 4;
199 }
200 }
201
202 if (heap->tld->heap_backing == NULL) {
203 heap->tld->heap_backing = heap; // first heap becomes the backing heap
204 _mi_random_init(&heap->random);
205 }
206 else {
207 _mi_random_split(&heap->tld->heap_backing->random, &heap->random);
208 }
209 heap->cookie = _mi_heap_random_next(heap) | 1;
210 //heap->keys[0] = _mi_heap_random_next(heap);
211 //heap->keys[1] = _mi_heap_random_next(heap);*/
212 _mi_heap_guarded_init(heap);
213
214 // push on the thread local heaps list
215 heap->next = heap->tld->heaps;
216 heap->tld->heaps = heap;
217}

219mi_heap_t* _mi_heap_create(int heap_tag, bool allow_destroy, mi_arena_id_t arena_id, mi_tld_t* tld) {
220 mi_assert_internal(tld!=NULL);
221 mi_assert(heap_tag >= 0 && heap_tag < 256);
222 // allocate and initialize a heap
223 mi_memid_t memid;
224 mi_heap_t* heap;
225 if (arena_id == _mi_arena_id_none()) {
226 heap = (mi_heap_t*)_mi_meta_zalloc(sizeof(mi_heap_t), &memid);
227 }
228 else {
229 // heaps associated wita a specific arena are allocated in that arena
230 // note: takes up at least one slice which is quite wasteful...
231 heap = (mi_heap_t*)_mi_arenas_alloc(_mi_subproc(), _mi_align_up(sizeof(mi_heap_t),MI_ARENA_MIN_OBJ_SIZE), true, true, _mi_arena_from_id(arena_id), tld->thread_seq, tld->numa_node, &memid);
232 }
233 if (heap==NULL) {
234 _mi_error_message(ENOMEM, "unable to allocate heap meta-data\n");
235 return NULL;
236 }
237 heap->memid = memid;
238 _mi_heap_init(heap, arena_id, allow_destroy, (uint8_t)heap_tag, tld);
239 return heap;
240}

281// called from `mi_heap_destroy` and `mi_heap_delete` to free the internal heap resources.
282static void mi_heap_free(mi_heap_t* heap, bool do_free_mem) {
283 mi_assert(heap != NULL);
284 mi_assert_internal(mi_heap_is_initialized(heap));
285 if (heap==NULL || !mi_heap_is_initialized(heap)) return;
286 if (mi_heap_is_backing(heap)) return; // dont free the backing heap
287
288 // reset default
289 if (mi_heap_is_default(heap)) {
290 _mi_heap_set_default_direct(heap->tld->heap_backing);
291 }
292
293 // remove ourselves from the thread local heaps list
294 // linear search but we expect the number of heaps to be relatively small
295 mi_heap_t* prev = NULL;
296 mi_heap_t* curr = heap->tld->heaps;
297 while (curr != heap && curr != NULL) {
298 prev = curr;
299 curr = curr->next;
300 }
301 mi_assert_internal(curr == heap);
302 if (curr == heap) {
303 if (prev != NULL) { prev->next = heap->next; }
304 else { heap->tld->heaps = heap->next; }
305 }
306 mi_assert_internal(heap->tld->heaps != NULL);
307
308 // and free the used memory
309 if (do_free_mem) {
310 _mi_meta_free(heap, sizeof(*heap), heap->memid);
311 }
312}

332 MI_UNUSED(arg1);
333 MI_UNUSED(arg2);
334 MI_UNUSED(heap);

337 // ensure no more thread_delayed_free will be added
338 //_mi_page_use_delayed_free(page, MI_NEVER_DELAYED_FREE, false);

341 const size_t bsize = mi_page_block_size(page);
342 if (bsize > MI_LARGE_MAX_OBJ_SIZE) {
343 mi_heap_stat_decrease(heap, malloc_huge, bsize);
344 }
345 #if (MI_STAT)
346 _mi_page_free_collect(page, false); // update used count
347 const size_t inuse = page->used;
348 if (bsize <= MI_LARGE_MAX_OBJ_SIZE) {
349 mi_heap_stat_decrease(heap, malloc_normal, bsize * inuse);
350 #if (MI_STAT>1)

355 #endif
356
358 mi_assert_internal(mi_page_thread_free(page) == NULL);
359 page->used = 0;
360
361 // and free the page
362 // mi_page_free(page,false);
363 page->next = NULL;
364 page->prev = NULL;
365 mi_page_set_heap(page, NULL);
366 _mi_arenas_page_free(page);
367
368 return true; // keep going
369}

384void mi_heap_destroy(mi_heap_t* heap) {
385 mi_assert(heap != NULL);
386 mi_assert(mi_heap_is_initialized(heap));
387 mi_assert(!heap->allow_page_reclaim);
388 mi_assert(!heap->allow_page_abandon);
389 mi_assert_expensive(mi_heap_is_valid(heap));
390 if (heap==NULL || !mi_heap_is_initialized(heap)) return;
391 #if MI_GUARDED
392 // _mi_warning_message("'mi_heap_destroy' called but MI_GUARDED is enabled -- using `mi_heap_delete` instead (heap at %p)\n", heap);
393 mi_heap_delete(heap);
394 return;
395 #else
396 if (heap->allow_page_reclaim) {
397 _mi_warning_message("'mi_heap_destroy' called but ignored as the heap was not created with 'allow_destroy' (heap at %p)\n", heap);
398 // don't free in case it may contain reclaimed pages,
399 mi_heap_delete(heap);
400 }
401 else {
402 // track all blocks as freed
403 #if MI_TRACK_HEAP_DESTROY
404 mi_heap_visit_blocks(heap, true, mi_heap_track_block_free, NULL);
405 #endif
406 // free all pages
407 _mi_heap_destroy_pages(heap);
408 mi_heap_free(heap,true);
409 }
410 #endif
411}

489void mi_heap_unload(mi_heap_t* heap) {
490 mi_assert(mi_heap_is_initialized(heap));
491 mi_assert_expensive(mi_heap_is_valid(heap));
492 if (heap==NULL || !mi_heap_is_initialized(heap)) return;
493 if (heap->exclusive_arena == NULL) {
494 _mi_warning_message("cannot unload heaps that are not associated with an exclusive arena\n");
495 return;
496 }
497
498 // abandon all pages so all thread'id in the pages are cleared
499 _mi_heap_collect_abandon(heap);
500 mi_assert_internal(heap->page_count==0);
501
502 // remove from heap list
503 mi_heap_free(heap, false /* but don't actually free the memory */);
504
505 // disassociate from the current thread-local and static state
506 heap->tld = NULL;
507 return;
508}

510bool mi_heap_reload(mi_heap_t* heap, mi_arena_id_t arena_id) {
511 mi_assert(mi_heap_is_initialized(heap));
512 if (heap==NULL || !mi_heap_is_initialized(heap)) return false;
513 if (heap->exclusive_arena == NULL) {
514 _mi_warning_message("cannot reload heaps that were not associated with an exclusive arena\n");
515 return false;
516 }
517 if (heap->tld != NULL) {
518 _mi_warning_message("cannot reload heaps that were not unloaded first\n");
519 return false;
520 }
521 mi_arena_t* arena = _mi_arena_from_id(arena_id);
522 if (heap->exclusive_arena != arena) {
523 _mi_warning_message("trying to reload a heap at a different arena address: %p vs %p\n", heap->exclusive_arena, arena);
524 return false;
525 }
526
527 mi_assert_internal(heap->page_count==0);
528
529 // re-associate with the current thread-local and static state
530 heap->tld = mi_heap_get_default()->tld;
531
532 // reinit direct pages (as we may be in a different process)
533 mi_assert_internal(heap->page_count == 0);
534 for (size_t i = 0; i < MI_PAGES_DIRECT; i++) {
535 heap->pages_free_direct[i] = (mi_page_t*)&_mi_page_empty;
536 }
537
538 // push on the thread local heaps list
539 heap->next = heap->tld->heaps;
540 heap->tld->heaps = heap;
541 return true;
542}

562static bool mi_heap_page_check_owned(mi_heap_t* heap, mi_page_queue_t* pq, mi_page_t* page, void* p, void* vfound) {
563 MI_UNUSED(heap);
564 MI_UNUSED(pq);
565 bool* found = (bool*)vfound;
566 void* start = mi_page_start(page);
567 void* end = (uint8_t*)start + (page->capacity * mi_page_block_size(page));
568 *found = (p >= start && p < end);
569 return (!*found); // continue if not found
570}

591void _mi_heap_area_init(mi_heap_area_t* area, mi_page_t* page) {
592 const size_t bsize = mi_page_block_size(page);
593 const size_t ubsize = mi_page_usable_block_size(page);
594 area->reserved = page->reserved * bsize;
595 area->committed = page->capacity * bsize;
596 area->blocks = mi_page_start(page);

604static void mi_get_fast_divisor(size_t divisor, uint64_t* magic, size_t* shift) {
605 mi_assert_internal(divisor > 0 && divisor <= UINT32_MAX);
606 *shift = MI_SIZE_BITS - mi_clz(divisor - 1);
607 *magic = ((((uint64_t)1 << 32) * (((uint64_t)1 << *shift) - divisor)) / divisor + 1);
608}
609
610static size_t mi_fast_divide(size_t n, uint64_t magic, size_t shift) {
611 mi_assert_internal(n <= UINT32_MAX);
612 const uint64_t hi = ((uint64_t)n * magic) >> 32;
613 return (size_t)((hi + n) >> shift);
614}

616bool _mi_heap_area_visit_blocks(const mi_heap_area_t* area, mi_page_t* page, mi_block_visit_fun* visitor, void* arg) {
617 mi_assert(area != NULL);
618 if (area==NULL) return true;
619 mi_assert(page != NULL);
620 if (page == NULL) return true;
621
622 _mi_page_free_collect(page,true); // collect both thread_delayed and local_free
623 mi_assert_internal(page->local_free == NULL);
624 if (page->used == 0) return true;
625
626 size_t psize;
627 uint8_t* const pstart = mi_page_area(page, &psize);
628 mi_heap_t* const heap = mi_page_heap(page);
629 const size_t bsize = mi_page_block_size(page);
630 const size_t ubsize = mi_page_usable_block_size(page); // without padding
631
632 // optimize page with one block
633 if (page->capacity == 1) {
634 mi_assert_internal(page->used == 1 && page->free == NULL);
635 return visitor(mi_page_heap(page), area, pstart, ubsize, arg);
636 }
637 mi_assert(bsize <= UINT32_MAX);
638
639 // optimize full pages
640 if (page->used == page->capacity) {
641 uint8_t* block = pstart;
642 for (size_t i = 0; i < page->capacity; i++) {
643 if (!visitor(heap, area, block, ubsize, arg)) return false;
644 block += bsize;
645 }
646 return true;
647 }
648
649 // create a bitmap of free blocks.
650 #define MI_MAX_BLOCKS (MI_SMALL_PAGE_SIZE / sizeof(void*))
651 uintptr_t free_map[MI_MAX_BLOCKS / MI_INTPTR_BITS];
652 const uintptr_t bmapsize = _mi_divide_up(page->capacity, MI_INTPTR_BITS);
653 memset(free_map, 0, bmapsize * sizeof(intptr_t));
654 if (page->capacity % MI_INTPTR_BITS != 0) {
655 // mark left-over bits at the end as free
656 size_t shift = (page->capacity % MI_INTPTR_BITS);
657 uintptr_t mask = (UINTPTR_MAX << shift);
658 free_map[bmapsize - 1] = mask;
659 }
660
661 // fast repeated division by the block size
662 uint64_t magic;
663 size_t shift;
664 mi_get_fast_divisor(bsize, &magic, &shift);
665
666 #if MI_DEBUG>1
667 size_t free_count = 0;
668 #endif
669 for (mi_block_t* block = page->free; block != NULL; block = mi_block_next(page, block)) {
670 #if MI_DEBUG>1
671 free_count++;
672 #endif
673 mi_assert_internal((uint8_t*)block >= pstart && (uint8_t*)block < (pstart + psize));
674 size_t offset = (uint8_t*)block - pstart;
675 mi_assert_internal(offset % bsize == 0);
676 mi_assert_internal(offset <= UINT32_MAX);
677 size_t blockidx = mi_fast_divide(offset, magic, shift);
678 mi_assert_internal(blockidx == offset / bsize);
679 mi_assert_internal(blockidx < MI_MAX_BLOCKS);
680 size_t bitidx = (blockidx / MI_INTPTR_BITS);
681 size_t bit = blockidx - (bitidx * MI_INTPTR_BITS);
682 free_map[bitidx] |= ((uintptr_t)1 << bit);
683 }
684 mi_assert_internal(page->capacity == (free_count + page->used));
685
686 // walk through all blocks skipping the free ones
687 #if MI_DEBUG>1
688 size_t used_count = 0;
689 #endif
690 uint8_t* block = pstart;
691 for (size_t i = 0; i < bmapsize; i++) {
692 if (free_map[i] == 0) {
693 // every block is in use
694 for (size_t j = 0; j < MI_INTPTR_BITS; j++) {
695 #if MI_DEBUG>1
696 used_count++;
697 #endif
698 if (!visitor(heap, area, block, ubsize, arg)) return false;
699 block += bsize;
700 }
701 }
702 else {
703 // visit the used blocks in the mask
704 uintptr_t m = ~free_map[i];
705 while (m != 0) {
706 #if MI_DEBUG>1
707 used_count++;
708 #endif
709 size_t bitidx = mi_ctz(m);
710 if (!visitor(heap, area, block + (bitidx * bsize), ubsize, arg)) return false;
711 m &= m - 1; // clear least significant bit
712 }
713 block += bsize * MI_INTPTR_BITS;
714 }
715 }
716 mi_assert_internal(page->used == used_count);
717 return true;
718}
