65static inline int32_t __TBB_machine_cmpswp4(volatile void *ptr, int32_t value, int32_t comparand )
66{
67 int32_t oldval, res;
68
69 __TBB_full_memory_fence();
70
71 do {
72 __asm__ __volatile__(
73 "ldrex %1, [%3]\n"
74 "mov %0, #0\n"
75 "cmp %1, %4\n"
76 "it eq\n"
77 "strexeq %0, %5, [%3]\n"
78 : "=&r" (res), "=&r" (oldval), "+Qo" (*(volatile int32_t*)ptr)
79 : "r" ((volatile int32_t *)ptr), "Ir" (comparand), "r" (value)
80 : "cc");
81 } while (res);
82
83 __TBB_full_memory_fence();
84
85 return oldval;
86}

98 int32_t res;
99
100 __TBB_full_memory_fence();
101
102 do {
103 __asm__ __volatile__(
104 "mov %0, #0\n"
105 "ldrexd %1, %H1, [%3]\n"
106 "cmp %1, %4\n"

121static inline int32_t __TBB_machine_fetchadd4(volatile void* ptr, int32_t addend)
122{
123 unsigned long tmp;
124 int32_t result, tmp2;
125
126 __TBB_full_memory_fence();
127
128 __asm__ __volatile__(
129"1: ldrex %0, [%4]\n"
130" add %3, %0, %5\n"
131" strex %1, %3, [%4]\n"
132" cmp %1, #0\n"
133" bne 1b\n"
134 : "=&r" (result), "=&r" (tmp), "+Qo" (*(volatile int32_t*)ptr), "=&r"(tmp2)
135 : "r" ((volatile int32_t *)ptr), "Ir" (addend)
136 : "cc");
137
138 __TBB_full_memory_fence();
139
140 return result;
141}
