24 // Tells if the Deleter type has a typedef for pointer to T. If so then return it,
25 // else return T*. The large majority of the time the pointer type will be T*.
26 // The C++11 Standard requires that scoped_ptr let the deleter define the pointer type.
27 //
28 // Example usage:
29 // typedef typename unique_pointer_type<int, SomeDeleter>::type pointer
30 //
31 template <typename T, typename Deleter>
32 class unique_pointer_type
33 {
34 template <typename U>
35 static typename U::pointer test(typename U::pointer*);
36
37 template <typename U>
38 static T* test(...);
39
40 public:
41 typedef decltype(test<typename eastl::remove_reference<Deleter>::type>(0)) type;
42 };

46 // is_array_cv_convertible
47 //
48 // Tells if the array pointer P1 is cv-convertible to array pointer P2.
49 // The two types have two be equivalent pointer types and be convertible
50 // when you consider const/volatile properties of them.
51 //
52 // Example usage:
53 // is_array_cv_convertible<int, Base*>::value => false
54 // is_array_cv_convertible<Base, Base*>::value => false
55 // is_array_cv_convertible<double*, bool*>::value => false
56 // is_array_cv_convertible<Subclass*, Base*>::value => false
57 // is_array_cv_convertible<const Base*, Base*>::value => false
58 // is_array_cv_convertible<Base*, Base*>::value => true
59 // is_array_cv_convertible<Base*, const Base*>::value => true
60 // is_array_cv_convertible<Base*, volatile Base*>::value => true
62
63#define EASTL_TYPE_TRAIT_is_array_cv_convertible_CONFORMANCE 1
64
65 template <typename P1,
66 typename P2,
67 bool = eastl::is_same_v<eastl::remove_cv_t<typename pointer_traits<P1>::element_type>,
68 eastl::remove_cv_t<typename pointer_traits<P2>::element_type>>>
69 struct is_array_cv_convertible_impl : public eastl::is_convertible<P1, P2>
70 {
71 }; // Return true if P1 is convertible to P2.
72
73 template <typename P1, typename P2>
74 struct is_array_cv_convertible_impl<P1, P2, false> : public eastl::false_type
75 {
76 }; // P1's underlying type is not the same as P2's, so it can't be converted, even if P2 refers to a subclass of
77 // P1. Parent == Child, but Parent[] != Child[]
78
79 template <typename P1, typename P2, bool = eastl::is_scalar_v<P1> && !eastl::is_pointer_v<P1>>
80 struct is_array_cv_convertible : public is_array_cv_convertible_impl<P1, P2>
81 {
82 };

85 struct is_array_cv_convertible<P1, P2, true> : public eastl::false_type
86 {
87 }; // P1 is scalar not a pointer, so it can't be converted to a pointer.

91 // is_derived
92 //
93 // Given two (possibly identical) types Base and Derived, is_base_of<Base, Derived>::value == true
94 // if and only if Base is a direct or indirect base class of Derived. This is like is_base_of<Base, Derived>
95 // but returns false if Derived is the same as Base. So is_derived is true only if Derived is actually a
96 // subclass of Base and not Base itself.
97 //
98 // is_derived may only be applied to complete types.
99 //
100 // Example usage:
101 // is_derived<int, int>::value => false
102 // is_derived<int, bool>::value => false
103 // is_derived<Parent, Child>::value => true
104 // is_derived<Child, Parent>::value => false
106
107#if EASTL_TYPE_TRAIT_is_base_of_CONFORMANCE
108#define EASTL_TYPE_TRAIT_is_derived_CONFORMANCE 1
109
110 template <typename Base, typename Derived>
111 struct is_derived
112 : public eastl::integral_constant<bool,
113 eastl::is_base_of<Base, Derived>::value &&
114 !eastl::is_same<typename eastl::remove_cv<Base>::type,
115 typename eastl::remove_cv<Derived>::type>::value>
116 {
117 };
118#else
119#define EASTL_TYPE_TRAIT_is_derived_CONFORMANCE 0
120
121 template <typename Base,
122 typename Derived> // This returns true if Derived is unrelated to Base. That's a wrong answer, but is
123 // better for us than returning false for compilers that don't support is_base_of.
124 struct is_derived
125 : public eastl::integral_constant<bool,
126 !eastl::is_same<typename eastl::remove_cv<Base>::type,
127 typename eastl::remove_cv<Derived>::type>::value>
128 {
129 };
130#endif

134 // is_safe_array_conversion
135 //
136 // Say you have two array types: T* t and U* u. You want to assign the u to t but only if
137 // that's a safe thing to do. As shown in the logic below, the array conversion
138 // is safe if U* and T* are convertible, if U is an array, and if either U or T is not
139 // a pointer or U is not derived from T.
140 //
141 // Note: Usage of this class could be replaced with is_array_cv_convertible usage.
142 // To do: Do this replacement and test it.
143 //
145
146 template <typename T, typename T_pointer, typename U, typename U_pointer>
147 struct is_safe_array_conversion
148 : public eastl::integral_constant<
149 bool,
150 eastl::is_convertible<U_pointer, T_pointer>::value && eastl::is_array<U>::value &&
151 (!eastl::is_pointer<U_pointer>::value || !is_pointer<T_pointer>::value ||
152 !Internal::is_derived<T, typename eastl::remove_extent<U>::type>::value)>
153 {
154 };

215 void operator()(T* p) const EA_NOEXCEPT
216 {
217 auto header_size = max<size_t>(alignof(T), sizeof(size_t));
218 auto header_ptr = reinterpret_cast<size_t*>(reinterpret_cast<size_t>(p) - header_size);
219 auto ele_size = *header_ptr;
220 auto alloc_size = header_size * ele_size * sizeof(T);
221 if constexpr (!std::is_trivially_constructible_v<T>)
222 {
223 auto end = p + ele_size;
224 for (auto ptr = p; ptr != end; ++ptr)
225 {
226 ptr->~T();
227 }
228 }
229 eastl::GetDefaultAllocator()->deallocate(header_ptr, alloc_size);
230 }
