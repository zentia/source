1314 // is_assignable
1315 //
1316 // The expression declval<T>() = declval<U>() is well-formed when treated as an unevaluated operand.
1317 // Access checking is performed as if in a context unrelated to T and U. Only the validity of
1318 // the immediate context of the assignment expression is considered. The compilation of the expression
1319 // can result in side effects such as the instantiation of class template specializations and function
1320 // template specializations, the generation of implicitly-defined functions, and so on. Such side
1321 // effects are not in the "immediate context" and can result in the program being ill-formed.
1322 //
1323 // Note:
1324 // This type trait has a misleading and counter-intuitive name. It does not indicate whether an instance
1325 // of U can be assigned to an instance of T (e.g. t = u). Instead it indicates whether the assignment can be
1326 // done after adding rvalue references to both, as in add_rvalue_reference<T>::type = add_rvalue_reference<U>::type.
1327 // A counterintuitive result of this is that is_assignable<int, int>::value == false. The is_copy_assignable
1328 // trait indicates if a type can be assigned to its own type, though there isn't a standard C++ way to tell
1329 // if an arbitrary type is assignable to another type.
1330 // http://stackoverflow.com/questions/19920213/why-is-stdis-assignable-counter-intuitive
1331 //
1332 // Note:
1333 // A true is_assignable value doesn't guarantee that the expression is compile-able, the compiler checks
1334 // only that the assignment matches before compilation. In particular, if you have templated operator=
1335 // for a class, the compiler will always say is_assignable is true, regardless of what's being tested
1336 // on the right hand side of the expression. It may actually turn out during compilation that the
1337 // templated operator= fails to compile because in practice it doesn't accept every possible type for
1338 // the right hand side of the expression.
1339 //
1340 // Expected results:
1341 // is_assignable<void, void>::value == false
1342 // is_assignable<int&, int>::value == true
1343 // is_assignable<int, int>::value == false
1344 // is_assignable<int, int&>::value == false
1345 // is_assignable<bool, bool>::value == false
1346 // is_assignable<int, float>::value == false
1347 // is_assignable<int[], int[]>::value == false
1348 // is_assignable<char*, int*>::value == false
1349 // is_assignable<char*, const char*>::value == false
1350 // is_assignable<const char*, char*>::value == false
1351 // is_assignable<PodA, PodB*>::value == false
1352 // is_assignable<Assignable, Assignable>::value == true
1353 // is_assignable<Assignable, Unrelated>::value == false
1354 //
1355 // Note:
1356 // Our implementation here yields different results than does the std::is_assignable from Dinkumware-based Standard
1357 // Libraries, but yields similar results to the std::is_assignable from GCC's libstdc++ and clang's libc++. It may
1358 // possibly be that the Dinkumware results are intentionally different for some practical purpose or because they
1359 // represent the spirit or the Standard but not the letter of the Standard.
1360 //
1362 #define EASTL_TYPE_TRAIT_is_assignable_CONFORMANCE 1
1363
1364 template<typename T, typename U>
1365 struct is_assignable_helper
1366 {
1367 template<typename, typename>
1368 static eastl::no_type is(...);
1369
1370 template<typename T1, typename U1>
1371 static decltype(eastl::declval<T1>() = eastl::declval<U1>(), eastl::yes_type()) is(int);
1372
1373 static const bool value = (sizeof(is<T, U>(0)) == sizeof(eastl::yes_type));
1374 };

1377 struct is_assignable :
1378 public eastl::integral_constant<bool, eastl::is_assignable_helper<T, U>::value> {};
