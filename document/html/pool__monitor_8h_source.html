164 template <typename Node>
165 void lock( Node const& p ) const
166 {
167 lock_type * pLock;
168
169 m_Stat.onLock();
170
171 // try lock spin and increment reference counter
172 refspin_type cur = p.m_SyncMonitorInjection.m_RefSpin.load( atomics::memory_order_relaxed ) & ~c_nSpinBit;
173 if ( !p.m_SyncMonitorInjection.m_RefSpin.compare_exchange_weak( cur, cur + c_nRefIncrement + c_nSpinBit,
174 atomics::memory_order_acq_rel, atomics::memory_order_acquire ))
175 {
176 back_off bkoff;
177 do {
178 m_Stat.onLockContention();
179 bkoff();
180 cur &= ~c_nSpinBit;
181 } while ( !p.m_SyncMonitorInjection.m_RefSpin.compare_exchange_weak( cur, cur + c_nRefIncrement + c_nSpinBit,
182 atomics::memory_order_acq_rel, atomics::memory_order_acquire ));
183 }
184
185 // spin locked
186 // If the node has no lock, allocate it from pool
187 pLock = p.m_SyncMonitorInjection.m_pLock;
188 if ( !pLock ) {
189 assert( cur == 0 );
190 pLock = p.m_SyncMonitorInjection.m_pLock = m_Pool.allocate( 1 );
191 assert( pLock != nullptr );
192 m_Stat.onLockAllocation();
193 }
194
195 // unlock spin
196 p.m_SyncMonitorInjection.m_RefSpin.store( cur + c_nRefIncrement, atomics::memory_order_release );
197
198 // lock the node
199 pLock->lock();
200 }
201
203 template <typename Node>
204 void unlock( Node const& p ) const
205 {
206 lock_type * pLock = nullptr;
207
208 m_Stat.onUnlock();
209
210 assert( p.m_SyncMonitorInjection.m_pLock != nullptr );
211 p.m_SyncMonitorInjection.m_pLock->unlock();
212
213 // try lock spin
214 refspin_type cur = p.m_SyncMonitorInjection.m_RefSpin.load( atomics::memory_order_relaxed ) & ~c_nSpinBit;
215 if ( !p.m_SyncMonitorInjection.m_RefSpin.compare_exchange_weak( cur, cur | c_nSpinBit,
216 atomics::memory_order_acquire, atomics::memory_order_acquire ))
217 {
218 back_off bkoff;
219 do {
220 m_Stat.onUnlockContention();
221 bkoff();
222 cur &= ~c_nSpinBit;
223 } while ( !p.m_SyncMonitorInjection.m_RefSpin.compare_exchange_weak( cur, cur | c_nSpinBit,
224 atomics::memory_order_acquire, atomics::memory_order_acquire ));
225 }
226
227 // spin locked now
228
229 // If we are the unique owner - deallocate lock
230 if ( cur == c_nRefIncrement ) {
231 pLock = p.m_SyncMonitorInjection.m_pLock;
232 p.m_SyncMonitorInjection.m_pLock = nullptr;
233 }
234
235 // unlock spin
236 p.m_SyncMonitorInjection.m_RefSpin.store( cur - c_nRefIncrement, atomics::memory_order_release );
237
238 // free pLock
239 if ( pLock ) {
240 m_Pool.deallocate( pLock, 1 );
241 m_Stat.onLockDeallocation();
242 }
243 }
