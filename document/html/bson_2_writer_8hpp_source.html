97 template <class T>
98 OutputVarType add_value_to_array(const T& _var,
99 OutputArrayType* _parent) const noexcept {
100 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
101 bson_array_builder_append_utf8(_parent->val_, _var.c_str(),
102 static_cast<int>(_var.size()));
103 } else if constexpr (std::is_same<std::remove_cvref_t<T>,
104 rfl::Bytestring>()) {
105 bson_array_builder_append_binary(
106 _parent->val_, BSON_SUBTYPE_BINARY,
107 internal::ptr_cast<const uint8_t*>(_var.c_str()),
108 static_cast<uint32_t>(_var.size()));
109 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
110 bson_array_builder_append_bool(_parent->val_, _var);
111 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
112 bson_array_builder_append_double(_parent->val_,
113 static_cast<double>(_var));
114 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
115 bson_array_builder_append_int64(_parent->val_,
116 static_cast<std::int64_t>(_var));
117 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bson_oid_t>()) {
118 bson_array_builder_append_oid(_parent->val_, &_var);
119 } else {
120 static_assert(rfl::always_false_v<T>, "Unsupported type.");
121 }
122 return OutputVarType{};
123 }

126 OutputVarType add_value_to_object(const std::string_view& _name,
127 const T& _var,
128 OutputObjectType* _parent) const noexcept {
129 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
130 bson_append_utf8(_parent->val_, _name.data(),

132 static_cast<int>(_var.size()));

135 bson_append_binary(_parent->val_, _name.data(),
136 static_cast<int>(_name.size()), BSON_SUBTYPE_BINARY,
137 internal::ptr_cast<const uint8_t*>(_var.c_str()),
138 static_cast<uint32_t>(_var.size()));
139 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
140 bson_append_bool(_parent->val_, _name.data(),
141 static_cast<int>(_name.size()), _var);
142 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
143 bson_append_double(_parent->val_, _name.data(),
144 static_cast<int>(_name.size()),
145 static_cast<double>(_var));
146 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
147 bson_append_int64(_parent->val_, _name.data(),
148 static_cast<int>(_name.size()),

151 bson_append_oid(_parent->val_, _name.data(),

153 } else {
154 static_assert(rfl::always_false_v<T>, "Unsupported type.");
155 }
