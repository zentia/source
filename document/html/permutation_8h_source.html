198 template <typename Int=int, class RandomGenerator = std::mt19937, class RandomDevice = std::random_device >
199 class random_shuffle_permutation
200 {
201 public:
202 typedef Int integer_type;
203 typedef RandomGenerator random_generator;
204 typedef RandomDevice random_device;
205
206 protected:
207 //@cond
208 integer_type * m_pCur;
209 integer_type * m_pFirst;
210 integer_type * m_pLast;
211 //@endcond
212
213 public:
215 random_shuffle_permutation( size_t nLength )
216 : m_pCur( nullptr )
217 {
218 m_pFirst = new integer_type[nLength];
219 m_pLast = m_pFirst + nLength;
220 std::iota(m_pFirst, m_pLast, integer_type{0} );
221 reset();
222 }
223
224 ~random_shuffle_permutation()
225 {
226 delete [] m_pFirst;
227 }
228
230 operator integer_type() const
231 {
232 return *m_pCur;
233 }
234
236 bool next()
237 {
238 return ++m_pCur < m_pLast;
239 }
240
242 void reset()
243 {
244 std::shuffle( m_pFirst, m_pLast, random_generator{ random_device{}() });
245 m_pCur = m_pFirst;
246 }
