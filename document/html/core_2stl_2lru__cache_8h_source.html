21class lru_cache {
22
23private:
24 using List = std::list<Key>;
25 using ListIterator = typename List::iterator;
26 struct Node {
27 ListIterator it;
28 Value value;
29 };
30 using Cache = std::unordered_map<Key, Node, Hash>;

32private:
33 Cache _cache;

36 size_t _capacity;

38private:

40 _list.splice(_list.begin(), _list, node.it);
41 }
42
43 void _make_space() noexcept {
44 while (_cache.size() >= _capacity) {
45 auto iter = _cache.find(_list.back());
46 if (_delete_callback) { _delete_callback(iter->second.value); }
47 _cache.erase(iter);
48 _list.pop_back();
49 }
50 }

57 }
58 [[nodiscard]] auto at(const Key &key) noexcept -> luisa::optional<Value> {
59 auto iter = _cache.find(key);
60 return iter == _cache.end() ? luisa::nullopt : luisa::make_optional(iter->second.value);
61 }
62 auto touch(const Key &key) noexcept -> bool {
63 if (auto iter = _cache.find(key); iter != _cache.end()) {
64 _touch(iter->second);
65 return true;
66 }
67 return false;
68 }
69 template<typename... Args>
70 void emplace(const Key &key, Args &&...args) noexcept {
71 if (auto iter = _cache.find(key); iter != _cache.end()) {
72 _touch(iter->second);
73 iter->second.value = Value{std::forward<Args>(args)...};
74 } else {
75 _make_space();
76 _list.push_front(key);
77 _cache.emplace(key, Node{_list.begin(), Value{std::forward<Args>(args)...}});
78 }
79 }
