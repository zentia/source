464template <typename T,
465 std::enable_if_t<is_std_basic_string_v<std::decay_t<T>>, void>*

467SizeType Serializer<Stream, SizeType>::_save(T&& t) {
468 using U = std::decay_t<T>;

470 _stream.write(

479template <typename T,
480 std::enable_if_t<is_std_vector_v<std::decay_t<T>>, void>*

632

635 static_assert(std::tuple_size<U>::value > 0, "Array size can't be zero");
636
637 SizeType sz;
638
639 if constexpr(std::is_arithmetic_v<typename U::value_type>) {
640 _stream.write(reinterpret_cast<const char*>(t.data()), sizeof(t));
641 sz = sizeof(t);
642 }
643 else {
644 sz = 0;
645 for(auto&& item : t) {
646 sz += _save(item);
647 }
648 }
649
650 return sz;
651}
