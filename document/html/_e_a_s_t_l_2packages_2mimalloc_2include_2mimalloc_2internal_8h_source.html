319// Align upwards
320static inline uintptr_t _mi_align_up(uintptr_t sz, size_t alignment) {
321 mi_assert_internal(alignment != 0);
322 uintptr_t mask = alignment - 1;
323 if ((alignment & mask) == 0) { // power of two?
324 return ((sz + mask) & ~mask);
325 }
326 else {
327 return (((sz + mask)/alignment)*alignment);
328 }
329}

338static inline uintptr_t _mi_align_down(uintptr_t sz, size_t alignment) {
339 mi_assert_internal(alignment != 0);
340 uintptr_t mask = alignment - 1;
341 if ((alignment & mask) == 0) { // power of two?
342 return (sz & ~mask);

382// Overflow detecting multiply
383#if __has_builtin(__builtin_umul_overflow) || (defined(__GNUC__) && (__GNUC__ >= 5))
384#include <limits.h> // UINT_MAX, ULONG_MAX
385#if defined(_CLOCK_T) // for Illumos
386#undef _CLOCK_T
387#endif
388static inline bool mi_mul_overflow(size_t count, size_t size, size_t* total) {
389 #if (SIZE_MAX == ULONG_MAX)
390 return __builtin_umull_overflow(count, size, (unsigned long *)total);
391 #elif (SIZE_MAX == UINT_MAX)
392 return __builtin_umul_overflow(count, size, (unsigned int *)total);
393 #else
394 return __builtin_umulll_overflow(count, size, (unsigned long long *)total);
395 #endif
396}
397#else /* __builtin_umul_overflow is unavailable */
398static inline bool mi_mul_overflow(size_t count, size_t size, size_t* total) {
399 #define MI_MUL_COULD_OVERFLOW ((size_t)1 << (4*sizeof(size_t))) // sqrt(SIZE_MAX)
400 *total = count * size;
401 // note: gcc/clang optimize this to directly check the overflow flag
402 return ((size >= MI_MUL_COULD_OVERFLOW || count >= MI_MUL_COULD_OVERFLOW) && size > 0 && (SIZE_MAX / size) < count);
403}
404#endif
405
406// Safe multiply `count*size` into `total`; return `true` on overflow.
407static inline bool mi_count_size_overflow(size_t count, size_t size, size_t* total) {
408 if (count==1) { // quick check for the case where count is one (common for C++ allocators)
409 *total = size;
410 return false;
411 }
412 else if mi_unlikely(mi_mul_overflow(count, size, total)) {
413 #if MI_DEBUG > 0
414 _mi_error_message(EOVERFLOW, "allocation request is too large (%zu * %zu bytes)\n", count, size);
415 #endif
416 *total = SIZE_MAX;
417 return true;
418 }
419 else return false;
420}

464static inline size_t _mi_page_map_index(const void* p) {
465 return (size_t)((uintptr_t)p >> MI_ARENA_SLICE_SHIFT);
466}

468static inline mi_page_t* _mi_ptr_page_ex(const void* p, bool* valid) {
469 const size_t idx = _mi_page_map_index(p);
470 const size_t ofs = _mi_page_map[idx];
471 if (valid != NULL) { *valid = (ofs != 0); }
472 return (mi_page_t*)((((uintptr_t)p >> MI_ARENA_SLICE_SHIFT) + 1 - ofs) << MI_ARENA_SLICE_SHIFT);
473}

487// 2-level page map:
488// double indirection, but low commit and low virtual reserve.
489//
490// the page-map is usually 4 MiB (for 48 bits virtual addresses) and points to sub maps of 64 KiB.
491// the page-map is committed on-demand (in 64 KiB parts) (and sub-maps are committed on-demand as well)
492// one sub page-map = 64 KiB => covers 2^(16-3) * 2^16 = 2^29 = 512 MiB address space
493// the page-map needs 48-(16+13) = 19 bits => 2^19 sub map pointers = 4 MiB size.

771// release ownership of a page. This may free the page if all blocks were concurrently
772// freed in the meantime. Returns true if the page was freed.
773static inline bool _mi_page_unown(mi_page_t* page) {
774 mi_assert_internal(mi_page_is_owned(page));
775 mi_assert_internal(mi_page_is_abandoned(page));
776 mi_thread_free_t tf_new;
777 mi_thread_free_t tf_old = mi_atomic_load_relaxed(&page->xthread_free);
778 do {
779 mi_assert_internal(mi_tf_is_owned(tf_old));
780 while mi_unlikely(mi_tf_block(tf_old) != NULL) {
781 _mi_page_free_collect(page, false); // update used
782 if (mi_page_all_free(page)) { // it may become free just before unowning it
783 _mi_arenas_page_unabandon(page);
784 _mi_arenas_page_free(page);
785 return true;
786 }
787 tf_old = mi_atomic_load_relaxed(&page->xthread_free);
788 }
789 mi_assert_internal(mi_tf_block(tf_old)==NULL);
790 tf_new = mi_tf_create(NULL, false);
791 } while (!mi_atomic_cas_weak_acq_rel(&page->xthread_free, &tf_old, tf_new));
792 return false;
793}

836/* -------------------------------------------------------------------
837Encoding/Decoding the free list next pointers
838
839This is to protect against buffer overflow exploits where the
840free list is mutated. Many hardened allocators xor the next pointer `p`
841with a secret key `k1`, as `p^k1`. This prevents overwriting with known
842values but might be still too weak: if the attacker can guess
843the pointer `p` this can reveal `k1` (since `p^k1^p == k1`).
844Moreover, if multiple blocks can be read as well, the attacker can
845xor both as `(p1^k1) ^ (p2^k1) == p1^p2` which may reveal a lot
846about the pointers (and subsequently `k1`).
847
848Instead mimalloc uses an extra key `k2` and encodes as `((p^k2)<<<k1)+k1`.
849Since these operations are not associative, the above approaches do not
850work so well any more even if the `p` can be guesstimated. For example,
851for the read case we can subtract two entries to discard the `+k1` term,
852but that leads to `((p1^k2)<<<k1) - ((p2^k2)<<<k1)` at best.
853We include the left-rotation since xor and addition are otherwise linear
854in the lowest bit. Finally, both keys are unique per page which reduces
855the re-use of keys by a large factor.
856
857We also pass a separate `null` value to be used as `NULL` or otherwise
858`(k2<<<k1)+k1` would appear (too) often as a sentinel value.
859------------------------------------------------------------------- */
860
861static inline bool mi_is_in_same_page(const void* p, const void* q) {
862 mi_page_t* page = _mi_ptr_page(p);
863 return mi_page_contains_address(page,q);
864 // return (_mi_ptr_page(p) == _mi_ptr_page(q));
865}

867static inline void* mi_ptr_decode(const void* null, const mi_encoded_t x, const uintptr_t* keys) {
868 void* p = (void*)(mi_rotr(x - keys[0], keys[0]) ^ keys[1]);
869 return (p==null ? NULL : p);
870}

872static inline mi_encoded_t mi_ptr_encode(const void* null, const void* p, const uintptr_t* keys) {
873 uintptr_t x = (uintptr_t)(p==NULL ? null : p);
874 return mi_rotl(x ^ keys[1], keys[0]) + keys[0];
875}

877static inline uint32_t mi_ptr_encode_canary(const void* null, const void* p, const uintptr_t* keys) {
878 const uint32_t x = (uint32_t)(mi_ptr_encode(null,p,keys));
879 // make the lowest byte 0 to prevent spurious read overflows which could be a security issue (issue #951)
880 #if MI_BIG_ENDIAN
881 return (x & 0x00FFFFFF);
882 #else
883 return (x & 0xFFFFFF00);
884 #endif
885}

994#if (MI_INTPTR_SIZE>=8)

996 x ^= x >> 30;
997 x *= 0xbf58476d1ce4e5b9UL;
998 x ^= x >> 27;

1000 x ^= x >> 31;
1001#elif (MI_INTPTR_SIZE==4)

1003 x ^= x >> 16;
1004 x *= 0x7feb352dUL;
1005 x ^= x >> 15;
1006 x *= 0x846ca68bUL;
1007 x ^= x >> 16;
1008#endif
1009 return x;
