71// Sorts [begin, end) using insertion sort with the given comparison function.
72template<typename RandItr, typename Compare>
73void insertion_sort(RandItr begin, RandItr end, Compare comp) {
74
75 using T = typename std::iterator_traits<RandItr>::value_type;
76
77 if (begin == end) {
78 return;
79 }
80
81 for (RandItr cur = begin + 1; cur != end; ++cur) {
82
83 RandItr shift = cur;
84 RandItr shift_1 = cur - 1;
85
86 // Compare first to avoid 2 moves for an element
87 // already positioned correctly.
88 if (comp(*shift, *shift_1)) {
89 T tmp = std::move(*shift);
90 do {
91 *shift-- = std::move(*shift_1);
92 }while (shift != begin && comp(tmp, *--shift_1));
93 *shift = std::move(tmp);
94 }
95 }
96}

128// Attempts to use insertion sort on [begin, end).
129// Will return false if more than
130// partial_insertion_sort_limit elements were moved,
131// and abort sorting. Otherwise it will successfully sort and return true.
132template<typename RandItr, typename Compare>
133bool partial_insertion_sort(RandItr begin, RandItr end, Compare comp) {
134
135 using T = typename std::iterator_traits<RandItr>::value_type;
136 using D = typename std::iterator_traits<RandItr>::difference_type;
137
138 // When we detect an already sorted partition, attempt an insertion sort
139 // that allows this amount of element moves before giving up.
140 constexpr auto partial_insertion_sort_limit = D{8};
141
142 if (begin == end) return true;
143
144 auto limit = D{0};
145
146 for (RandItr cur = begin + 1; cur != end; ++cur) {
147
148 if (limit > partial_insertion_sort_limit) {
149 return false;
150 }
151
152 RandItr shift = cur;
153 RandItr shift_1 = cur - 1;
154
155 // Compare first so we can avoid 2 moves
156 // for an element already positioned correctly.
157 if (comp(*shift, *shift_1)) {
158 T tmp = std::move(*shift);
159
160 do {
161 *shift-- = std::move(*shift_1);
162 }while (shift != begin && comp(tmp, *--shift_1));
163
164 *shift = std::move(tmp);
165 limit += cur - shift;
166 }
167 }
168
169 return true;
170}

172// Partitions [begin, end) around pivot *begin using comparison function comp. Elements equal
173// to the pivot are put in the right-hand partition. Returns the position of the pivot after
174// partitioning and whether the passed sequence already was correctly partitioned. Assumes the
175// pivot is a median of at least 3 elements and that [begin, end) is at least
176// insertion_sort_threshold long. Uses branchless partitioning.

191 // this exists).

193
194 // Find the first element strictly smaller than the pivot. We have to guard this search if
195 // there was no element before *first.
196 if (first - 1 == begin) while (first < last && !comp(*--last, pivot));
197 else while ( !comp(*--last, pivot));

199 // If the first pair of elements that should be swapped to partition are the same element,
200 // the passed in sequence already was correctly partitioned.
201 bool already_partitioned = first >= last;
202 if (!already_partitioned) {
203 std::iter_swap(first, last);
204 ++first;

206 // The following branchless partitioning is derived from "BlockQuicksort: How Branch
207 // Mispredictions don't affect Quicksort" by Stefan Edelkamp and Armin Weiss, but
208 // heavily micro-optimized.
209 unsigned char offsets_l_storage[block_size + cacheline_size];
210 unsigned char offsets_r_storage[block_size + cacheline_size];
211 unsigned char* offsets_l = align_cacheline(offsets_l_storage);
212 unsigned char* offsets_r = align_cacheline(offsets_r_storage);

215 Iter offsets_r_base = last;

217 num_l = num_r = start_l = start_r = 0;

220 // Fill up offset blocks with elements that are on the wrong side.
221 // First we determine how much elements are considered for each offset block.
222 size_t num_unknown = last - first;
223 size_t left_split = num_l == 0 ? (num_r == 0 ? num_unknown / 2 : num_unknown) : 0;
224 size_t right_split = num_r == 0 ? (num_unknown - left_split) : 0;

229 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
230 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
231 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
232 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
233 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
234 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
235 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
236 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
237 }
238 } else {
239 for (unsigned char i = 0; i < left_split;) {
240 offsets_l[num_l] = i++; num_l += !comp(*first, pivot); ++first;
241 }
242 }
243
244 if (right_split >= block_size) {
245 for (unsigned char i = 0; i < block_size;) {
246 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
247 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
248 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
249 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
250 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
251 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
252 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
253 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
254 }
255 } else {
256 for (unsigned char i = 0; i < right_split;) {
257 offsets_r[num_r] = ++i; num_r += comp(*--last, pivot);
258 }
259 }

261 // Swap elements and update block sizes and first/last boundaries.
262 size_t num = std::min(num_l, num_r);
263 swap_offsets(
264 offsets_l_base, offsets_r_base,
265 offsets_l + start_l, offsets_r + start_r,
266 num, num_l == num_r
267 );

274 }
275
276 if (num_r == 0) {

278 offsets_r_base = last;
279 }
280 }

282 // We have now fully identified [first, last)'s proper position. Swap the last elements.
283 if (num_l) {
284 offsets_l += start_l;

286 first = last;
287 }
288 if (num_r) {
289 offsets_r += start_r;
290 while (num_r--) std::iter_swap(offsets_r_base - offsets_r[num_r], first), ++first;
291 last = first;
292 }
293 }

296 Iter pivot_pos = first - 1;
297 *begin = std::move(*pivot_pos);
298 *pivot_pos = std::move(pivot);
299
300 return std::make_pair(pivot_pos, already_partitioned);

303// Partitions [begin, end) around pivot *begin using comparison function comp.
304// Elements equal to the pivot are put in the right-hand partition.
305// Returns the position of the pivot after partitioning and whether the passed
306// sequence already was correctly partitioned.
307// Assumes the pivot is a median of at least 3 elements and that [begin, end)
308// is at least insertion_sort_threshold long.
309template<typename Iter, typename Compare>
310std::pair<Iter, bool> partition_right(Iter begin, Iter end, Compare comp) {
311
312 using T = typename std::iterator_traits<Iter>::value_type;
313
314 // Move pivot into local for speed.
315 T pivot(std::move(*begin));
316
317 Iter first = begin;
318 Iter last = end;
319
320 // Find the first element greater than or equal than the pivot
321 // (the median of 3 guarantees/ this exists).
322 while (comp(*++first, pivot));
323
324 // Find the first element strictly smaller than the pivot.
325 // We have to guard this search if there was no element before *first.
326 if (first - 1 == begin) while (first < last && !comp(*--last, pivot));
327 else while (!comp(*--last, pivot));
328
329 // If the first pair of elements that should be swapped to partition
330 // are the same element, the passed in sequence already was correctly
331 // partitioned.
332 bool already_partitioned = first >= last;
333
334 // Keep swapping pairs of elements that are on the wrong side of the pivot.
335 // Previously swapped pairs guard the searches,
336 // which is why the first iteration is special-cased above.
337 while (first < last) {
338 std::iter_swap(first, last);
339 while (comp(*++first, pivot));
340 while (!comp(*--last, pivot));
341 }
342
343 // Put the pivot in the right place.
344 Iter pivot_pos = first - 1;
345 *begin = std::move(*pivot_pos);
346 *pivot_pos = std::move(pivot);
347
348 return std::make_pair(pivot_pos, already_partitioned);
349}

351// Similar function to the one above, except elements equal to the pivot
352// are put to the left of the pivot and it doesn't check or return
353// if the passed sequence already was partitioned.
354// Since this is rarely used (the many equal case),
355// and in that case pdqsort already has O(n) performance,
356// no block quicksort is applied here for simplicity.
357template<typename RandItr, typename Compare>
358RandItr partition_left(RandItr begin, RandItr end, Compare comp) {
359
360 using T = typename std::iterator_traits<RandItr>::value_type;
361
362 T pivot(std::move(*begin));
363
364 RandItr first = begin;
365 RandItr last = end;
366
367 while (comp(pivot, *--last));
368
369 if (last + 1 == end) {
370 while (first < last && !comp(pivot, *++first));
371 }
372 else {
373 while (!comp(pivot, *++first));
374 }
375
376 while (first < last) {
377 std::iter_swap(first, last);
378 while (comp(pivot, *--last));
379 while (!comp(pivot, *++first));
380 }
381
382 RandItr pivot_pos = last;
383 *begin = std::move(*pivot_pos);
384 *pivot_pos = std::move(pivot);
385
386 return pivot_pos;
387}

389template<typename Iter, typename Compare, bool Branchless>
390void parallel_pdqsort(
391 Runtime& rt,
392 Iter begin, Iter end, Compare comp,
393 int bad_allowed, bool leftmost = true
394) {
395
396 // Partitions below this size are sorted sequentially
397 constexpr auto cutoff = parallel_sort_cutoff<Iter>();
398
399 // Partitions below this size are sorted using insertion sort
400 constexpr auto insertion_sort_threshold = 24;
401
402 // Partitions above this size use Tukey's ninther to select the pivot.
403 constexpr auto ninther_threshold = 128;
404
405 //using diff_t = typename std::iterator_traits<Iter>::difference_type;
406
407 // Use a while loop for tail recursion elimination.
408 while (true) {
409
410 //diff_t size = end - begin;
411 size_t size = end - begin;
412
413 // Insertion sort is faster for small arrays.
414 if (size < insertion_sort_threshold) {
415 if (leftmost) {
416 insertion_sort(begin, end, comp);
417 }
418 else {
419 unguarded_insertion_sort(begin, end, comp);
420 }
421 return;
422 }
423
424 if(size <= cutoff) {
425 std::sort(begin, end, comp);
426 return;
427 }
428
429 // Choose pivot as median of 3 or pseudomedian of 9.
430 //diff_t s2 = size / 2;
431 size_t s2 = size >> 1;
432 if (size > ninther_threshold) {
433 sort3(begin, begin + s2, end - 1, comp);
434 sort3(begin + 1, begin + (s2 - 1), end - 2, comp);
435 sort3(begin + 2, begin + (s2 + 1), end - 3, comp);
436 sort3(begin + (s2 - 1), begin + s2, begin + (s2 + 1), comp);
437 std::iter_swap(begin, begin + s2);
438 }
439 else {
440 sort3(begin + s2, begin, end - 1, comp);
441 }
442
443 // If *(begin - 1) is the end of the right partition
444 // of a previous partition operation, there is no element in [begin, end)
445 // that is smaller than *(begin - 1).
446 // Then if our pivot compares equal to *(begin - 1) we change strategy,
447 // putting equal elements in the left partition,
448 // greater elements in the right partition.
449 // We do not have to recurse on the left partition,
450 // since it's sorted (all equal).
451 if (!leftmost && !comp(*(begin - 1), *begin)) {
452 begin = partition_left(begin, end, comp) + 1;
453 continue;
454 }
455
456 // Partition and get results.
457 const auto pair = Branchless ? partition_right_branchless(begin, end, comp) :
458 partition_right(begin, end, comp);
459
460 const auto pivot_pos = pair.first;
461 const auto already_partitioned = pair.second;
462
463 // Check for a highly unbalanced partition.
464 //diff_t l_size = pivot_pos - begin;
465 //diff_t r_size = end - (pivot_pos + 1);
466 const size_t l_size = pivot_pos - begin;
467 const size_t r_size = end - (pivot_pos + 1);
468 const bool highly_unbalanced = l_size < size / 8 || r_size < size / 8;
469
470 // If we got a highly unbalanced partition we shuffle elements
471 // to break many patterns.
472 if (highly_unbalanced) {
473 // If we had too many bad partitions, switch to heapsort
474 // to guarantee O(n log n).
475 if (--bad_allowed == 0) {
476 std::make_heap(begin, end, comp);
477 std::sort_heap(begin, end, comp);
478 return;
479 }
480
481 if (l_size >= insertion_sort_threshold) {
482 std::iter_swap(begin, begin + l_size / 4);
483 std::iter_swap(pivot_pos - 1, pivot_pos - l_size / 4);
484 if (l_size > ninther_threshold) {
485 std::iter_swap(begin + 1, begin + (l_size / 4 + 1));
486 std::iter_swap(begin + 2, begin + (l_size / 4 + 2));
487 std::iter_swap(pivot_pos - 2, pivot_pos - (l_size / 4 + 1));
488 std::iter_swap(pivot_pos - 3, pivot_pos - (l_size / 4 + 2));
489 }
490 }
491
492 if (r_size >= insertion_sort_threshold) {
493 std::iter_swap(pivot_pos + 1, pivot_pos + (1 + r_size / 4));
494 std::iter_swap(end - 1, end - r_size / 4);
495 if (r_size > ninther_threshold) {
496 std::iter_swap(pivot_pos + 2, pivot_pos + (2 + r_size / 4));
497 std::iter_swap(pivot_pos + 3, pivot_pos + (3 + r_size / 4));
498 std::iter_swap(end - 2, end - (1 + r_size / 4));
499 std::iter_swap(end - 3, end - (2 + r_size / 4));
500 }
501 }
502 }
503 // decently balanced
504 else {
505 // sequence try to use insertion sort.
506 if (already_partitioned &&
507 partial_insertion_sort(begin, pivot_pos, comp) &&
508 partial_insertion_sort(pivot_pos + 1, end, comp)
509 ) {
510 return;
511 }
512 }
513
514 // Sort the left partition first using recursion and
515 // do tail recursion elimination for the right-hand partition.
516 // here we need to copy runtime so it stays alive during the sort recursion
517 rt.silent_async([=] () mutable {
518 parallel_pdqsort<Iter, Compare, Branchless>(
519 rt, begin, pivot_pos, comp, bad_allowed, leftmost
520 );
521 });
522 begin = pivot_pos + 1;
523 leftmost = false;
524 }
525}

532template <typename RandItr, typename C>
533void parallel_3wqsort(Runtime& rt, RandItr first, RandItr last, C compare) {
534
535 using namespace std::string_literals;
536
537 constexpr auto cutoff = parallel_sort_cutoff<RandItr>();
538
539 sort_partition:
540
541 if(static_cast<size_t>(last - first) < cutoff) {
542 std::sort(first, last+1, compare);
543 return;
544 }
545
546 auto m = pseudo_median_of_nine(first, last, compare);
547
548 if(m != first) {
549 std::iter_swap(first, m);
550 }
551
552 auto l = first;
553 auto r = last;
554 auto f = std::next(first, 1);
555 bool is_swapped_l = false;
556 bool is_swapped_r = false;
557
558 while(f <= r) {
559 if(compare(*f, *l)) {
560 is_swapped_l = true;
561 std::iter_swap(l, f);
562 l++;
563 f++;
564 }
565 else if(compare(*l, *f)) {
566 is_swapped_r = true;
567 std::iter_swap(r, f);
568 r--;
569 }
570 else {
571 f++;
572 }
573 }
574
575 if(l - first > 1 && is_swapped_l) {
576 rt.silent_async([=] () mutable {
577 parallel_3wqsort(rt, first, l-1, compare);
578 });
579 }
580
581 if(last - r > 1 && is_swapped_r) {
582 first = r+1;
583 goto sort_partition;
584 }
585}
