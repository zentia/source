97 virtual void test()
98 {
99 Map& rMap = m_Map;
100
101 unsigned int i = 0;
102 size_t const nNormalize = size_t(-1) / ( s_nMapSize * 2 );
103
104 size_t nRand = 0;
105 while ( !time_elapsed()) {
106 nRand = cds::bitop::RandXorShift( nRand );
107 size_t n = nRand / nNormalize;
108 switch ( s_arrShuffle[i] ) {
109 case do_find:
110 if ( rMap.contains( n ))
111 ++m_nFindSuccess;
112 else
113 ++m_nFindFailed;
114 break;
115 case do_insert:
116 if ( n % 2 ) {
117 if ( rMap.insert( n, n ))
118 ++m_nInsertSuccess;
119 else
120 ++m_nInsertFailed;
121 }
122 else {
123 if ( rMap.update( n, update_functor(), true ).first )
124 ++m_nInsertSuccess;
125 else
126 ++m_nInsertFailed;
127 }
128 break;
129 case do_delete:
130 if ( rMap.erase( n ))
131 ++m_nDeleteSuccess;
132 else
133 ++m_nDeleteFailed;
134 break;
135 }
136
137 if ( ++i >= c_nShuffleSize )
138 i = 0;
139 }
140 }
