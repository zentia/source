26// regular free of a (thread local) block pointer
27// fast path written carefully to prevent spilling on the stack
28static inline void mi_free_block_local(mi_page_t* page, mi_block_t* block, bool track_stats, bool check_full)
29{
30 // checks
31 if mi_unlikely(mi_check_is_double_free(page, block)) return;
32 mi_check_padding(page, block);
33 if (track_stats) { mi_stat_free(page, block); }
34 #if (MI_DEBUG>0) && !MI_TRACK_ENABLED && !MI_TSAN && !MI_GUARDED
35 memset(block, MI_DEBUG_FREED, mi_page_block_size(page));
36 #endif
37 if (track_stats) { mi_track_free_size(block, mi_page_usable_size_of(page, block)); } // faster then mi_usable_size as we already know the page and that p is unaligned
38
39 // actual free: push on the local free list
40 mi_block_set_next(page, block, page->local_free);
41 page->local_free = block;
42 if mi_unlikely(--page->used == 0) {
43 _mi_page_retire(page);
44 }
45 else if mi_unlikely(check_full && mi_page_is_in_full(page)) {
46 _mi_page_unfull(page);
47 }
48}

53// Free a block multi-threaded
54static inline void mi_free_block_mt(mi_page_t* page, mi_block_t* block) mi_attr_noexcept
55{
56 // adjust stats (after padding check and potentially recursive `mi_free` above)
57 mi_stat_free(page, block); // stat_free may access the padding
58 mi_track_free_size(block, mi_page_usable_size_of(page, block));
59
60 // _mi_padding_shrink(page, block, sizeof(mi_block_t));
61#if (MI_DEBUG>0) && !MI_TRACK_ENABLED && !MI_TSAN // note: when tracking, cannot use mi_usable_size with multi-threading
62 size_t dbgsize = mi_usable_size(block);
63 if (dbgsize > MI_MiB) { dbgsize = MI_MiB; }
64 _mi_memset_aligned(block, MI_DEBUG_FREED, dbgsize);
65#endif
66
67 // push atomically on the page thread free list
68 mi_thread_free_t tf_new;
69 mi_thread_free_t tf_old = mi_atomic_load_relaxed(&page->xthread_free);
70 do {
71 mi_block_set_next(page, block, mi_tf_block(tf_old));
72 tf_new = mi_tf_create(block, true /* always use owned: try to claim it if the page is abandoned */);
73 } while (!mi_atomic_cas_weak_acq_rel(&page->xthread_free, &tf_old, tf_new)); // todo: release is enough?
74
75 // and atomically try to collect the page if it was abandoned
76 const bool is_owned_now = !mi_tf_is_owned(tf_old);
77 if (is_owned_now) {
78 mi_assert_internal(mi_page_is_abandoned(page));
79 mi_free_try_collect_mt(page,block);
80 }
81}

84// Adjust a block that was allocated aligned, to the actual start of the block in the page.
85// note: this can be called from `mi_free_generic_mt` where a non-owning thread accesses the
86// `page_start` and `block_size` fields; however these are constant and the page won't be
87// deallocated (as the block we are freeing keeps it alive) and thus safe to read concurrently.
88mi_block_t* _mi_page_ptr_unalign(const mi_page_t* page, const void* p) {
89 mi_assert_internal(page!=NULL && p!=NULL);
90
91 size_t diff = (uint8_t*)p - mi_page_start(page);
92 size_t adjust;
93 if mi_likely(page->block_size_shift != 0) {
94 adjust = diff & (((size_t)1 << page->block_size_shift) - 1);
95 }
96 else {
97 adjust = diff % mi_page_block_size(page);
98 }
99
100 return (mi_block_t*)((uintptr_t)p - adjust);
101}

116// free a local pointer (page parameter comes first for better codegen)
117static void mi_decl_noinline mi_free_generic_local(mi_page_t* page, void* p) mi_attr_noexcept {
118 mi_block_t* const block = (mi_page_has_aligned(page) ? _mi_page_ptr_unalign(page, p) : (mi_block_t*)p);
119 mi_block_check_unguard(page, block, p);

123// free a pointer owned by another thread (page parameter comes first for better codegen)
124static void mi_decl_noinline mi_free_generic_mt(mi_page_t* page, void* p) mi_attr_noexcept {
125 if (p==NULL) return; // a NULL pointer is seen as abandoned (tid==0) with a full flag set
126 #if !MI_PAGE_MAP_FLAT
127 if (page==&_mi_page_empty) return; // an invalid pointer may lead to using the empty page
128 #endif
129 mi_assert_internal(p!=NULL && page != NULL && page != &_mi_page_empty);
130 mi_block_t* const block = _mi_page_ptr_unalign(page, p); // don't check `has_aligned` flag to avoid a race (issue #865)
131 mi_block_check_unguard(page, block, p);
132 mi_free_block_mt(page, block);
133}

167// Free a block
168// Fast path written carefully to prevent register spilling on the stack
169void mi_free(void* p) mi_attr_noexcept
170{
171 mi_page_t* const page = mi_validate_ptr_page(p,"mi_free");
172
173 #if MI_PAGE_MAP_FLAT // if not flat, p==NULL leads to `_mi_page_empty` which leads to `mi_free_generic_mt`
174 if mi_unlikely(page==NULL) return;
175 #endif
176 mi_assert_internal(page!=NULL);
177
178 const mi_threadid_t xtid = (_mi_prim_thread_id() ^ mi_page_xthread_id(page));
179 if mi_likely(xtid == 0) { // `tid == mi_page_thread_id(page) && mi_page_flags(page) == 0`
180 // thread-local, aligned, and not a full page
181 mi_block_t* const block = (mi_block_t*)p;
182 mi_free_block_local(page, block, true /* track stats */, false /* no need to check if the page is full */);
183 }
184 else if (xtid <= MI_PAGE_FLAG_MASK) { // `tid == mi_page_thread_id(page) && mi_page_flags(page) != 0`
185 // page is local, but is full or contains (inner) aligned blocks; use generic path
186 mi_free_generic_local(page, p);
187 }
188 // free-ing in a page owned by a heap in another thread, or an abandoned page (not belonging to a heap)
189 else if ((xtid & MI_PAGE_FLAG_MASK) == 0) { // `tid != mi_page_thread_id(page) && mi_page_flags(page) == 0`
190 // blocks are aligned (and not a full page); push on the thread_free list
191 mi_block_t* const block = (mi_block_t*)p;
192 mi_free_block_mt(page,block);
193 }
194 else {
195 // page is full or contains (inner) aligned blocks; use generic multi-thread path
196 mi_free_generic_mt(page, p);
197 }
198}

217static void mi_decl_noinline mi_free_try_collect_mt(mi_page_t* page, mi_block_t* mt_free) mi_attr_noexcept {
218 mi_assert_internal(mi_page_is_owned(page));
219 mi_assert_internal(mi_page_is_abandoned(page));
220
221 // we own the page now..
222 // safe to collect the thread atomic free list
223 // use the `_partly` version to avoid atomic operations since we already have the `mt_free` pointing into the thread free list
224 _mi_page_free_collect_partly(page, mt_free);
225
226 #if MI_DEBUG > 1
227 if (mi_page_is_singleton(page)) { mi_assert_internal(mi_page_all_free(page)); }
228 #endif
229
230 // 1. free if the page is free now (this is updated by `_mi_page_free_collect_partly`)
231 if (mi_page_all_free(page))
232 {
233 // first remove it from the abandoned pages in the arena (if mapped, this waits for any readers to finish)
234 _mi_arenas_page_unabandon(page);
235 // we can free the page directly
236 _mi_arenas_page_free(page);
237 return;
238 }
239
240 // 2. we can try to reclaim the page for ourselves
241 // note: we only reclaim if the page originated from our heap (the heap field is preserved on abandonment)
242 // to avoid claiming arbitrary object sizes and limit indefinite expansion. This helps benchmarks like `larson`
243 if (page->block_size <= MI_SMALL_MAX_OBJ_SIZE) // only for small sized blocks
244 {
245 const long reclaim_on_free = _mi_option_get_fast(mi_option_page_reclaim_on_free);
246 if (reclaim_on_free >= 0) { // and reclaiming is allowed
247 // get our heap (with the right tag)
248 // note: don't use `mi_heap_get_default()` as we may just have terminated this thread and we should
249 // not reinitialize the heap for this thread. (can happen due to thread-local destructors for example -- issue #944)
250 mi_heap_t* heap = mi_prim_get_default_heap();
251 if (heap != page->heap) {
252 if (mi_heap_is_initialized(heap)) {
253 heap = _mi_heap_by_tag(heap, page->heap_tag);
254 }
255 }
256 // can we reclaim into this heap?
257 if (heap != NULL && heap->allow_page_reclaim) {
258 const long reclaim_max = _mi_option_get_fast(mi_option_page_reclaim_max);
259 if ((heap == page->heap && mi_page_queue_len_is_atmost(heap, page->block_size, reclaim_max)) || // only reclaim if we were the originating heap, and we have at most N pages already
260 (reclaim_on_free == 1 && // OR if the reclaim across heaps is allowed
261 !mi_page_is_used_at_frac(page, 8) && // and the page is not too full
262 !heap->tld->is_in_threadpool && // and not part of a threadpool
263 _mi_arena_memid_is_suitable(page->memid, heap->exclusive_arena)) // and the memory is suitable
264 )
265 {
266 // first remove it from the abandoned pages in the arena -- this waits for any readers to finish
267 _mi_arenas_page_unabandon(page);
268 _mi_heap_page_reclaim(heap, page);
269 mi_heap_stat_counter_increase(heap, pages_reclaim_on_free, 1);
270 return;
271 }
272 }
273 }
274 }
275
276 // 3. if the page is unmapped, try to reabandon so it can possibly be mapped and found for allocations
277 if (!mi_page_is_used_at_frac(page, 8) && // only reabandon if a full page starts to have enough blocks available to prevent immediate re-abandon of a full page
278 !mi_page_is_abandoned_mapped(page) && page->memid.memkind == MI_MEM_ARENA &&
279 _mi_arenas_page_try_reabandon_to_mapped(page))
280 {
281 return;
282 }
283
284
285 // not reclaimed or free'd, unown again
286 // _mi_page_unown(page);
287 mi_page_unown_from_free(page, mt_free);
288}

291// release ownership of a page. This may free the page if all (other) blocks were concurrently
292// freed in the meantime. Returns true if the page was freed.
293// This is a specialized version of `mi_page_unown` to (try to) avoid calling `mi_page_free_collect` again.
294static bool mi_page_unown_from_free(mi_page_t* page, mi_block_t* mt_free) {
295 mi_assert_internal(mi_page_is_owned(page));
296 mi_assert_internal(mi_page_is_abandoned(page));
297 mi_assert_internal(mt_free != NULL);
298 mi_assert_internal(page->used > 1);
299 mi_thread_free_t tf_expect = mi_tf_create(mt_free, true);
300 mi_thread_free_t tf_new = mi_tf_create(mt_free, false);
301 while mi_unlikely(!mi_atomic_cas_weak_acq_rel(&page->xthread_free, &tf_expect, tf_new)) {
302 mi_assert_internal(mi_tf_is_owned(tf_expect));
303 while (mi_tf_block(tf_expect) != NULL) {
304 _mi_page_free_collect(page,false); // update used
305 if (mi_page_all_free(page)) { // it may become free just before unowning it
306 _mi_arenas_page_unabandon(page);
307 _mi_arenas_page_free(page);
308 return true;
309 }
310 tf_expect = mi_atomic_load_relaxed(&page->xthread_free);
311 }
312 mi_assert_internal(mi_tf_block(tf_expect)==NULL);
313 tf_new = mi_tf_create(NULL, false);
314 }
315 return false;
316}

323// Bytes available in a block
324static size_t mi_decl_noinline mi_page_usable_aligned_size_of(const mi_page_t* page, const void* p) mi_attr_noexcept {
325 const mi_block_t* block = _mi_page_ptr_unalign(page, p);
326 const size_t size = mi_page_usable_size_of(page, block);
327 const ptrdiff_t adjust = (uint8_t*)p - (uint8_t*)block;
328 mi_assert_internal(adjust >= 0 && (size_t)adjust <= size);
329 const size_t aligned_size = (size - adjust);
330 #if MI_GUARDED
331 if (mi_block_ptr_is_guarded(block, p)) {
332 return aligned_size - _mi_os_page_size();
333 }
334 #endif
335 return aligned_size;
336}

394static mi_decl_noinline bool mi_check_is_double_freex(const mi_page_t* page, const mi_block_t* block) {
395 // The decoded value is in the same page (or NULL).
396 // Walk the free lists to verify positively if it is already freed
397 if (mi_list_contains(page, page->free, block) ||
398 mi_list_contains(page, page->local_free, block) ||
399 mi_list_contains(page, mi_page_thread_free(page), block))
400 {
401 _mi_error_message(EAGAIN, "double free detected of block %p with size %zu\n", block, mi_page_block_size(page));
402 return true;
403 }
404 return false;
405}

407#define mi_track_page(page,access) { size_t psize; void* pstart = _mi_page_start(_mi_page_segment(page),page,&psize); mi_track_mem_##access( pstart, psize); }

409static inline bool mi_check_is_double_free(const mi_page_t* page, const mi_block_t* block) {
410 bool is_double_free = false;
411 mi_block_t* n = mi_block_nextx(page, block, page->keys); // pretend it is freed, and get the decoded first field
412 if (((uintptr_t)n & (MI_INTPTR_SIZE-1))==0 && // quick check: aligned pointer?
413 (n==NULL || mi_is_in_same_page(block, n))) // quick check: in same page or NULL?
414 {
415 // Suspicious: decoded value a in block is in the same page (or NULL) -- maybe a double free?
416 // (continue in separate function to improve code generation)
417 is_double_free = mi_check_is_double_freex(page, block);
418 }
419 return is_double_free;
420}

430// ---------------------------------------------------------------------------
431// Check for heap block overflow by setting up padding at the end of the block
432// ---------------------------------------------------------------------------
433
434#if MI_PADDING // && !MI_TRACK_ENABLED
435static bool mi_page_decode_padding(const mi_page_t* page, const mi_block_t* block, size_t* delta, size_t* bsize) {
436 *bsize = mi_page_usable_block_size(page);
437 const mi_padding_t* const padding = (mi_padding_t*)((uint8_t*)block + *bsize);
438 mi_track_mem_defined(padding,sizeof(mi_padding_t));
439 *delta = padding->delta;
440 uint32_t canary = padding->canary;
441 uintptr_t keys[2];
442 keys[0] = page->keys[0];
443 keys[1] = page->keys[1];
444 bool ok = (mi_ptr_encode_canary(page,block,keys) == canary && *delta <= *bsize);
445 mi_track_mem_noaccess(padding,sizeof(mi_padding_t));
446 return ok;
447}

449// Return the exact usable size of a block.
450static size_t mi_page_usable_size_of(const mi_page_t* page, const mi_block_t* block) {
451 size_t bsize;
452 size_t delta;
453 bool ok = mi_page_decode_padding(page, block, &delta, &bsize);
454 mi_assert_internal(ok); mi_assert_internal(delta <= bsize);
455 return (ok ? bsize - delta : 0);
456}

458// When a non-thread-local block is freed, it becomes part of the thread delayed free
459// list that is freed later by the owning heap. If the exact usable size is too small to
460// contain the pointer for the delayed list, then shrink the padding (by decreasing delta)
461// so it will later not trigger an overflow error in `mi_free_block`.
462void _mi_padding_shrink(const mi_page_t* page, const mi_block_t* block, const size_t min_size) {
463 size_t bsize;
464 size_t delta;
465 bool ok = mi_page_decode_padding(page, block, &delta, &bsize);
466 mi_assert_internal(ok);
467 if (!ok || (bsize - delta) >= min_size) return; // usually already enough space
468 mi_assert_internal(bsize >= min_size);
469 if (bsize < min_size) return; // should never happen
470 size_t new_delta = (bsize - min_size);
471 mi_assert_internal(new_delta < bsize);
472 mi_padding_t* padding = (mi_padding_t*)((uint8_t*)block + bsize);
473 mi_track_mem_defined(padding,sizeof(mi_padding_t));
474 padding->delta = (uint32_t)new_delta;
475 mi_track_mem_noaccess(padding,sizeof(mi_padding_t));
476}
477#else

479 MI_UNUSED(block);
480 return mi_page_usable_block_size(page);
481}

491
492static bool mi_verify_padding(const mi_page_t* page, const mi_block_t* block, size_t* size, size_t* wrong) {
493 size_t bsize;
494 size_t delta;
495 bool ok = mi_page_decode_padding(page, block, &delta, &bsize);
496 *size = *wrong = bsize;
497 if (!ok) return false;
498 mi_assert_internal(bsize >= delta);
499 *size = bsize - delta;
500 if (!mi_page_is_huge(page)) {
501 uint8_t* fill = (uint8_t*)block + bsize - delta;
502 const size_t maxpad = (delta > MI_MAX_ALIGN_SIZE ? MI_MAX_ALIGN_SIZE : delta); // check at most the first N padding bytes
503 mi_track_mem_defined(fill, maxpad);
504 for (size_t i = 0; i < maxpad; i++) {
505 if (fill[i] != MI_DEBUG_PADDING) {
506 *wrong = bsize - delta + i;
507 ok = false;
508 break;
509 }
510 }
511 mi_track_mem_noaccess(fill, maxpad);
512 }
513 return ok;
514}

516static void mi_check_padding(const mi_page_t* page, const mi_block_t* block) {
517 size_t size;
