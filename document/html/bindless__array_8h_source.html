38 struct ModSlotHash {

49 struct ModSlotEqual {
50 using is_transparent = void;
51
52 template<typename T>
53 [[nodiscard]] static auto slot(const T &m) noexcept -> size_t {
54 if constexpr (requires(T m) { m.slot; }) {
55 return m.slot;
56 } else {
57 return m;
58 }
59 }
60
61 template<typename Lhs, typename Rhs>
62 [[nodiscard]] auto operator()(const Lhs &lhs, const Rhs &rhs) const noexcept -> bool {
63 return slot(lhs) == slot(rhs);
64 }
65 };
