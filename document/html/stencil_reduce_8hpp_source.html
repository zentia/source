257 inline void *svc(void *task) {
258 if (task != NULL) Task = *((stencilTask<T>*)task);
259 T *Min = Task.getInPtr();
260 T *Mout = Task.getOutPtr();
261
262 if (oneShot && (initInF1 || initInF2)) {
263 const size_t& Xsize = Task.X_size();
264 const size_t& Ysize = Task.Y_size();
265
266 if (initInF1) {
267 ploop.parallel_for(0,Xsize,1,chunkSize,[&](const long i) {
268 for(long j=0; j< Ysize; ++j)
269 Min[i*Xsize+j] = initInF1(i,j, extraInitInParam);
270 }, nw);
271 } else {
272 initInF2(ploop, Min, Xsize, Ysize);
273 }
274 }
275 if (oneShot && (initOutF1 || initOutF2)) {
276 const size_t& Xsize = Task.X_size();
277 const size_t& Ysize = Task.Y_size();
278
279 if (initOutF1) {
280 ploop.parallel_for(0,Xsize,1,chunkSize, [&](const long i) {
281 for(long j=0; j< Ysize; ++j)
282 Mout[i*Xsize+j] = initOutF1(i,j, extraInitOutParam);
283 }, nw);
284 } else {
285 initOutF2(ploop, Mout, Xsize, Ysize);
286 }
287 }
288
289 {
290 const size_t& Xsize = Task.X_size();
291 const size_t& Ysize = Task.Y_size();
292 const size_t& Xstart = Task.X_start();
293 const size_t& Xstop = Task.X_stop();
294 const size_t& Xstep = Task.X_step();
295 const size_t& Ystart = Task.Y_start();
296 const size_t& Ystop = Task.Y_stop();
297 const size_t& Ystep = Task.Y_step();
298
299 T rVar = reduceVar;
300 rVar = identityValue;
301 iter = 0;
302 swap(); // because of the next swap op
303 if (computeFReduce1) {
304 do {
305 swap();
306 Min = Task.getInPtr(); Mout = Task.getOutPtr();
307
308 if (beforeFor) beforeFor(Min,Xsize, Ysize, rVar);
309 ploop.parallel_reduce(rVar,identityValue, Xstart,Xstop, Xstep, chunkSize,
310 [&](const long i,T &rVar) {
311 for(long j=Ystart; j< Ystop; j+=Ystep) {
312 Mout[i*Xsize+j] = computeFReduce1(i,j,Min,Xsize,Ysize,rVar);
313 }
314 }, reduceOp, nw);
315
316
317 if (afterFor) afterFor(Mout,Xsize, Ysize, rVar);
318
319 } while(++iter<maxIter && iterCondition(rVar, iter));
320 } else {
321 do {
322 swap();
323 Min = Task.getInPtr(); Mout = Task.getOutPtr();
324
325 if (beforeFor) beforeFor(Min,Xsize, Ysize, rVar);
326 computeFReduce2(ploop, Min,Mout,Xsize, Xstart,Xstop, Ysize,Ystart,Ystop,rVar);
327 if (afterFor) afterFor(Mout,Xsize, Ysize, rVar);
328
329 } while(++iter<maxIter && iterCondition(rVar, iter));
330 }
331 reduceVar = rVar;
332 }
333
334 return (oneShot?NULL:task);
335 }
