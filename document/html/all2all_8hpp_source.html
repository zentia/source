49 inline int cardinality(BARRIER_T * const barrier) {
50 int card=0;
51 for(size_t i=0;i<workers1.size();++i)
52 card += workers1[i]->cardinality(barrier);
53 for(size_t i=0;i<workers2.size();++i)
54 card += workers2[i]->cardinality(barrier);
55
56 return card;
57 }

59 inline int prepare() {
60 /* ----------------------- */
61 if (wraparound) {
62 if (workers2[0]->isMultiOutput()) { // NOTE: we suppose that all others are the same
63 if (workers1[0]->isMultiInput()) { // NOTE: we suppose that all others are the same
64 for(size_t i=0;i<workers2.size(); ++i) {
65 for(size_t j=0;j<workers1.size();++j) {
66 ff_node* t = new ff_buffernode(in_buffer_entries,false);
67 t->set_id(i);
68 internalSupportNodes.push_back(t);
69 workers2[i]->set_output_feedback(t);
70 workers1[j]->set_input_feedback(t);
71 }
72 }
73 } else {
74 // the cardinatlity of the first and second set of workers must be the same
75 if (workers1.size() != workers2.size()) {
76 error("A2A, wrap_around, the workers of the second set are not multi-output nodes so the cardinatlity of the first and second set must be the same\n");
77 return -1;
78 }
79
80 if (create_input_buffer(in_buffer_entries, false) <0) {
81 error("A2A, error creating input buffers\n");
82 return -1;
83 }
84
85 for(size_t i=0;i<workers2.size(); ++i)
86 workers2[i]->set_output_feedback(workers1[i]);
87
88 }
89 } else {
90 // the cardinatlity of the first and second set of workers must be the same
91 if (workers1.size() != workers2.size()) {
92 error("A2A, wrap_around, the workers of the second set are not multi-output nodes so the cardinatlity of the first and second set must be the same\n");
93 return -1;
94 }
95 if (!workers1[0]->isMultiInput()) { // we suppose that all others are the same
96 if (create_input_buffer(in_buffer_entries, false) <0) {
97 error("A2A, error creating input buffers\n");
98 return -1;
99 }
100
101 for(size_t i=0;i<workers2.size(); ++i)
102 workers2[i]->set_output_buffer(workers1[i]->get_in_buffer());
103
104 } else {
105 if (create_output_buffer(out_buffer_entries, false) <0) {
106 error("A2A, error creating output buffers\n");
107 return -1;
108 }
109
110 for(size_t i=0;i<workers1.size(); ++i)
111 if (workers1[i]->set_input_feedback(workers2[i])<0) {
112 error("A2A, wrap_around, the nodes of the first set are not multi-input\n");
113 return -1;
114 }
115 }
116 }
117
118 // blocking stuff --------------------------------------------
119 pthread_mutex_t *m = NULL;
120 pthread_cond_t *c = NULL;
121 for(size_t i=0;i<workers2.size(); ++i) {
122 if (!workers2[i]->init_output_blocking(m,c)) return -1;
123 }
124 if (!workers2[0]->isMultiOutput()) {
125 assert(workers1.size() == workers2.size());
126 }
127 if (workers1[0]->isMultiInput()) {
128 for(size_t i=0;i<workers1.size();++i) {
129 if (!workers1[i]->init_input_blocking(m,c)) return -1;
130 }
131 } else {
132 assert(workers1.size() == workers2.size());
133 for(size_t i=0;i<workers1.size();++i) {
134 if (!workers1[i]->init_input_blocking(m,c)) return -1;
135 workers2[i]->set_output_blocking(m,c);
136 }
137 }
138 // -----------------------------------------------------------
139 } // wraparound
140
141
142 if (workers1[0]->isFarm() || workers1[0]->isAll2All()) {
143 error("A2A, nodes of the first set cannot be farm or all-to-all\n");
144 return -1;
145 }
146 if (workers2[0]->isFarm() || workers2[0]->isAll2All()) {
147 error("A2A, nodes of the second set cannot be farm or all-to-all\n");
148 return -1;
149 }
150
151 if (workers1[0]->isPipe()) {
152 // all other Workers must be pipe
153 for(size_t i=1;i<workers1.size();++i) {
154 if (!workers1[i]->isPipe()) {
155 error("A2A, workers of the first set are not homogeneous, all of them must be of the same kind of building-block (e.g., all pipelines)\n");
156 return -1;
157 }
158 }
159
160 if (!workers1[0]->isMultiOutput()) {
161 error("A2A, workers of the first set can be pipelines but only if they are multi-output (automatic transformation NOT YET SUPPORTED)\n");
162 return -1;
163 }
164
165 ff_node* last = ispipe_getlast(workers1[0]); // NOTE: we suppose homogeneous first set
166 assert(last);
167 if (last->isFarm() && !last->isOFarm()) { // standard farm ...
168 if (!isfarm_withcollector(last)) { // ... with no collector
169 svector<ff_node*> w1;
170 last->get_out_nodes(w1);
171 if (!w1[0]->isMultiOutput()) { // NOTE: we suppose homogeneous workers
172 error("A2A, workers (farm/ofarm) of the first set are pipelines but their last stage is not multi-output (automatic transformation NOT YET SUPPORTED)\n");
173 return -1;
174 }
175 }
176 }
177 // since by default the a2a is multi-output, we have to check if its workers
178 // are multi-output
179 if (last->isAll2All()) {
180 svector<ff_node*> w1;
181 last->get_out_nodes(w1);
182 if (!w1[0]->isMultiOutput()) { // NOTE: we suppose homogeneous second set
183 error("A2A, workers (a2a) of the first set are pipelines but their last stage is not multi-output (automatic transformation NOT YET SUPPORTED)\n");
184 return -1;
185 }
186 }
187 }
188 if (workers2[0]->isPipe()) {
189 // all other Workers must be pipe
190 for(size_t i=1;i<workers2.size();++i) {
191 if (!workers2[i]->isPipe()) {
192 error("A2A, workers of the second set are not homogeneous, all of them must be of the same kind (e.g., all pipelines)\n");
193 return -1;
194 }
195 }
196
197 if (!workers2[0]->isMultiInput()) {
198 error("A2A, workers of the second set can be pipelines but only if they are multi-input (automatic transformation NOT YET SUPPORTED)\n");
199 return -1;
200 }
201 // since by default the a2a is multi-input, we have to check if its workers
202 // are multi-input
203 svector<ff_node*> w1;
204 workers2[0]->get_in_nodes(w1);
205 if (!w1[0]->isMultiInput()) { // NOTE: we suppose homogeneous second set
206 error("A2A, workers of the second set are pipelines but their first stage is not multi-input (automatic transformation NOT YET SUPPORTED)\n");
207 return -1;
208 }
209 }
210
211 // checking L-Workers
212 if (!workers1[0]->isMultiOutput()) { // NOTE: we suppose all others to be the same
213 // the nodes in the first set cannot be multi-input nodes without being
214 // also multi-output
215 if (workers1[0]->isMultiInput()) { // NOTE: we suppose all others to be the same
216 error("A2A, the nodes of the first set cannot be multi-input nodes without being also multi-output (i.e., a composition of nodes). The node must be either standard node or multi-output node or compositions where the second stage is a multi-output node\n");
217 return -1;
218 }
219 // it is a standard node or a pipeline with a standard node as last stage, so we transform it to a multi-output node
220 for(size_t i=0;i<workers1.size();++i) {
221 internal_mo_transformer *mo = new internal_mo_transformer(workers1[i], false);
222 if (!mo) {
223 error("A2A, FATAL ERROR not enough memory\n");
224 return -1;
225 }
226 BARRIER_T *bar =workers1[i]->get_barrier();
227 if (bar) mo->set_barrier(bar);
228 workers1[i] = mo; // replacing old node
229 internalSupportNodes.push_back(mo);
230 }
231 }
232 for(size_t i=0;i<workers1.size();++i) {
233 if (ondemand_chunk && (workers1[i]->ondemand_buffer()==0)) {
234 svector<ff_node*> w;
235 workers1[i]->get_out_nodes(w);
236 for(size_t k=0;k<w.size(); ++k)
237 w[k]->set_scheduling_ondemand(ondemand_chunk);
238 //workers1[i]->set_scheduling_ondemand(ondemand_chunk);
239 }
240 workers1[i]->set_id(int(i));
241 }
242 // checking R-Workers
243 if (!workers2[0]->isMultiInput()) { // NOTE: we suppose that all others are the same
244 if (workers2[0]->isMultiOutput()) {
245 error("A2A, the nodes of the second set cannot be multi-output nodes without being also multi-input (i.e., a composition of nodes). The node must be either standard node or multi-input node or compositions where the first stage is a multi-input node\n");
246 return -1;
247 }
248
249 // here we have to transform the standard node into a multi-input node
250 for(size_t i=0;i<workers2.size();++i) {
251 internal_mi_transformer *mi = new internal_mi_transformer(workers2[i], false);
252 if (!mi) {
253 error("A2A, FATAL ERROR not enough memory\n");
254 return -1;
255 }
256 BARRIER_T *bar =workers2[i]->get_barrier();
257 if (bar) mi->set_barrier(bar);
258 workers2[i] = mi; // replacing old node
259 internalSupportNodes.push_back(mi);
260 }
261 }
262 for(size_t i=0;i<workers2.size();++i) {
263 workers2[i]->set_id(int(i));
264 }
265
266 size_t nworkers1 = workers1.size();
267 size_t nworkers2 = workers2.size();
268
269 {
270 int ondemand = workers1[0]->ondemand_buffer(); // NOTE: here we suppose that all nodes in workers1 are homogeneous!
271
272 svector<ff_node*> L;
273 for(size_t i=0;i<nworkers1;++i)
274 workers1[i]->get_out_nodes(L);
275 if (L.size()==0) L=workers1;
276 svector<ff_node*> R;
277 for(size_t i=0;i<nworkers2;++i)
278 workers2[i]->get_in_nodes(R);
279 if (R.size()==0) R=workers2;
280
281 for(size_t i=0;i<R.size(); ++i) {
282 for(size_t j=0;j<L.size();++j) {
283 ff_node* t = new ff_buffernode(ondemand?ondemand:in_buffer_entries,ondemand?true:(fixedsizeIN|fixedsizeOUT), j);
284 assert(t);
285 internalSupportNodes.push_back(t);
286 L[j]->set_output(t);
287 R[i]->set_input(t);
288 }
289 }
290 }
291 if (outputNodes.size()) {
292
293 svector<ff_node*> w;
294 for(size_t i=0;i<workers2.size();++i)
295 workers2[i]->get_out_nodes(w);
296
297 if (outputNodes.size() != w.size()) {
298 error("A2A, prepare, invalid state detected\n");
299 return -1;
300 }
301
302 for(size_t i=0;i<w.size(); ++i) {
303 if (w[i]->isMultiOutput()) {
304 error("A2A, prepare, invalid state, unexpected multi-output node\n");
305 return -1;
306 }
307
308 assert(outputNodes[i]->get_in_buffer() != nullptr);
309 if (w[i]->set_output_buffer(outputNodes[i]->get_in_buffer()) < 0) {
310 error("A2A, prepare, invalid state, setting output buffer\n");
311 return -1;
312 }
313 }
314 }
315
316
317
318 // blocking stuff --------------------------------------------
319 pthread_mutex_t *m = NULL;
320 pthread_cond_t *c = NULL;
321 for(size_t i=0;i<nworkers2;++i) {
322 // initialize worker2 local cons_* stuff and sets all p_cons_*
323 if (!workers2[i]->init_input_blocking(m,c)) return -1;
324 }
325 for(size_t i=0;i<nworkers1;++i) {
326 // initialize worker1 local prod_* stuff and sets all p_prod_*
327 if (!workers1[i]->init_output_blocking(m,c)) return -1;
328 }
329 // ------------------------------------------------------------
330 prepared = true;
331 return 0;
332 }

461 bool change_node(ff_node* old, ff_node* n, bool cleanup=false, bool remove_from_cleanuplist=false) {
462 if (prepared) {
463 error("A2A, change_node cannot be called because the A2A has already been prepared\n");
464 return false;
465 }
466 for(size_t i=0; i<workers1.size();++i) {
467 if (workers1[i] == old) {
468 if (remove_from_cleanuplist) {
469 int pos=-1;
470 for(size_t i=0;i<internalSupportNodes.size();++i)
471 if (internalSupportNodes[i] == old) { pos = i; break; }
472 if (pos>=0) internalSupportNodes.erase(internalSupportNodes.begin()+pos);
473 }
474 workers1[i] = n;
475 if (cleanup) internalSupportNodes.push_back(n);
476 return true;
477 }
478 }
479 for(size_t i=0; i<workers2.size();++i) {
480 if (workers2[i] == old) {
481 if (remove_from_cleanuplist) {
482 int pos=-1;
483 for(size_t i=0;i<internalSupportNodes.size();++i)
484 if (internalSupportNodes[i] == old) { pos = i; break; }
485 if (pos>=0) internalSupportNodes.erase(internalSupportNodes.begin()+pos);
486 }
487 workers2[i] = n;
488 if (cleanup) internalSupportNodes.push_back(n);
489 return true;
490 }
491 }
492
493 return false;

530 int run(bool skip_init=false) {
531 if (!skip_init) {
532#if defined(FF_INITIAL_BARRIER)
533 if (initial_barrier) {
534 // set the initial value for the barrier
535 if (!barrier) barrier = new BARRIER_T;
536 const int nthreads = cardinality(barrier);
537 if (nthreads > MAX_NUM_THREADS) {
538 error("PIPE, too much threads, increase MAX_NUM_THREADS !\n");
539 return -1;
540 }
541 barrier->barrierSetup(nthreads);
542 }
543#endif
544 skipfirstpop(true);
545 }
546 if (!prepared) if (prepare()<0) return -1;
547
548 const size_t nworkers1 = workers1.size();
549 const size_t nworkers2 = workers2.size();
550
551 for(size_t i=0;i<nworkers1; ++i) {
552 workers1[i]->blocking_mode(blocking_in);
553 if (!default_mapping) workers1[i]->no_mapping();
554 if (workers1[i]->run(true)<0) {
555 error("ERROR: A2A, running worker (first set) %d\n", i);
556 return -1;
557 }
558 }
559 for(size_t i=0;i<nworkers2; ++i) {
560 workers2[i]->blocking_mode(blocking_in);
561 if (!default_mapping) workers2[i]->no_mapping();
562 if (workers2[i]->run(true)<0) {
563 error("ERROR: A2A, running worker (second set) %d\n", i);
564 return -1;
565 }
566 }
567 return 0;
568 }

570 int wait() {
571 int ret=0;
572 const size_t nworkers1 = workers1.size();
573 const size_t nworkers2 = workers2.size();
574 for(size_t i=0;i<nworkers1; ++i)
575 if (workers1[i]->wait()<0) {
576 error("A2A, waiting Worker1 thread, id = %d\n",workers1[i]->get_my_id());
577 ret = -1;
578 }
579 for(size_t i=0;i<nworkers2; ++i)
580 if (workers2[i]->wait()<0) {
581 error("A2A, waiting Worker2 thread, id = %d\n",workers2[i]->get_my_id());
582 ret = -1;
583 }
584
585 return ret;
586 }

793 }
