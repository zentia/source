325 private:
326 template <class K, class... Args>
327 insert_return_type try_emplace_forwarding(K&& k, Args&&... args)
328 {
329 const auto key_data = base_type::DoFindKeyData(k);
330 if (key_data.node)
331 { // Node exists, no insertion needed.
332 return eastl::pair<iterator, bool>(
333 iterator(key_data.node, base_type::mpBucketArray + key_data.bucket_index), false);
334 }
335 else
336 {
337 node_type* const pNodeNew =
338 base_type::DoAllocateNode(piecewise_construct, eastl::forward_as_tuple(eastl::forward<K>(k)),
339 forward_as_tuple(eastl::forward<Args>(args)...));
340 // the key might have been moved from above, so we can't use `k` anymore.
341 const auto& key = base_type::mExtractKey(pNodeNew->mValue);
342 return base_type::template DoInsertUniqueNode<true>(key, key_data.code, key_data.bucket_index, pNodeNew);
343 }

542 template <typename Key, typename T, typename Hash, typename Predicate, typename Allocator, bool bCacheHashCode>
543 inline bool operator==(const hash_map<Key, T, Hash, Predicate, Allocator, bCacheHashCode>& a,
544 const hash_map<Key, T, Hash, Predicate, Allocator, bCacheHashCode>& b)
545 {
546 typedef typename hash_map<Key, T, Hash, Predicate, Allocator, bCacheHashCode>::const_iterator const_iterator;
547
548 // We implement branching with the assumption that the return value is usually false.
549 if(a.size() != b.size())
550 return false;
551
552 // For map (with its unique keys), we need only test that each element in a can be found in b,
553 // as there can be only one such pairing per element. multimap needs to do a something more elaborate.
554 for(const_iterator ai = a.begin(), aiEnd = a.end(), biEnd = b.end(); ai != aiEnd; ++ai)
555 {
556 const_iterator bi = b.find(ai->first);
557
558 if((bi == biEnd) || !(*ai == *bi)) // We have to compare the values, because lookups are done by keys alone but the full value_type of a map is a key/value pair.
559 return false; // It's possible that two elements in the two containers have identical keys but different values.
560 }
561
562 return true;
563 }

574 template <typename Key, typename T, typename Hash, typename Predicate, typename Allocator, bool bCacheHashCode>
575 inline bool operator==(const hash_multimap<Key, T, Hash, Predicate, Allocator, bCacheHashCode>& a,
576 const hash_multimap<Key, T, Hash, Predicate, Allocator, bCacheHashCode>& b)
577 {
578 typedef typename hash_multimap<Key, T, Hash, Predicate, Allocator, bCacheHashCode>::const_iterator const_iterator;
579 typedef typename eastl::iterator_traits<const_iterator>::difference_type difference_type;
580
581 // We implement branching with the assumption that the return value is usually false.
582 if(a.size() != b.size())
583 return false;
584
585 // We can't simply search for each element of a in b, as it may be that the bucket for
586 // two elements in a has those same two elements in b but in different order (which should
587 // still result in equality). Also it's possible that one bucket in a has two elements which
588 // both match a solitary element in the equivalent bucket in b (which shouldn't result in equality).
589 eastl::pair<const_iterator, const_iterator> aRange;
590 eastl::pair<const_iterator, const_iterator> bRange;
591
592 for(const_iterator ai = a.begin(), aiEnd = a.end(); ai != aiEnd; ai = aRange.second) // For each element in a...
593 {
594 aRange = a.equal_range(ai->first); // Get the range of elements in a that are equal to ai.
595 bRange = b.equal_range(ai->first); // Get the range of elements in b that are equal to ai.
596
597 // We need to verify that aRange == bRange. First make sure the range sizes are equivalent...
598 const difference_type aDistance = eastl::distance(aRange.first, aRange.second);
599 const difference_type bDistance = eastl::distance(bRange.first, bRange.second);
600
601 if(aDistance != bDistance)
602 return false;
603
604 // At this point, aDistance > 0 and aDistance == bDistance.
605 // Implement a fast pathway for the case that there's just a single element.
606 if(aDistance == 1)
607 {
608 if(!(*aRange.first == *bRange.first)) // We have to compare the values, because lookups are done by keys alone but the full value_type of a map is a key/value pair.
609 return false; // It's possible that two elements in the two containers have identical keys but different values. Ditto for the permutation case below.
610 }
611 else
612 {
613 // Check to see if these aRange and bRange are any permutation of each other.
614 // This check gets slower as there are more elements in the range.
615 if(!eastl::is_permutation(aRange.first, aRange.second, bRange.first))
616 return false;
617 }
618 }
619
620 return true;
621 }
