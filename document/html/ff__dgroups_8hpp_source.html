73 void parseConfig(std::string configFile){
74 std::ifstream is(configFile);
75
76 if (!is) throw FF_Exception("Unable to open configuration file for the program!");
77
78 try {
79 cereal::JSONInputArchive ari(is);
80 ari(cereal::make_nvp("groups", this->parsedGroups));
81
82 // get the protocol to be used from the configuration file
83 try {
84 std::string tmpProtocol;
85 ari(cereal::make_nvp("protocol", tmpProtocol));
86 if (tmpProtocol == "MPI"){
87 #ifdef DFF_MPI
88 this->usedProtocol = Proto::MPI;
89 #else
90 std::cout << "NO MPI support! Falling back to TCP\n";
91 this->usedProtocol = Proto::TCP;
92 #endif
93
94 } else this->usedProtocol = Proto::TCP;
95 } catch (cereal::Exception&) {
96 ari.setNextName(nullptr);
97 this->usedProtocol = Proto::TCP;
98 }
99
100 } catch (const cereal::Exception& e){
101 std::cerr << "Error parsing the JSON config file. Check syntax and structure of the file and retry!" << std::endl;
102 exit(EXIT_FAILURE);
103 }

135 int run_and_wait_end(ff_pipeline* parent){
136 if (annotatedGroups.find(runningGroup) == annotatedGroups.end()){
137 ff::error("The group %s is not found nor implemented!\n", runningGroup.c_str());
138 return -1;
139 }
140
141 bool allDeriveFromParent = true;
142 for(auto& [name, ir]: annotatedGroups)
143 if (ir.parentBB != parent) { allDeriveFromParent = false; break; }
144
145 if (allDeriveFromParent) {
146 ff_pipeline *mypipe = new ff_pipeline;
147 mypipe->add_stage(parent);
148 parent = mypipe;
149 }
150
151 // qui dovrei creare la rappresentazione intermedia di tutti
152 this->prepareIR(parent);
153
154#ifdef PRINT_IR
155 this->annotatedGroups[this->runningGroup].print();
156#endif
157
158 // buildare il farm dalla rappresentazione intermedia del gruppo che devo rannare
159 dGroup _grp(this->annotatedGroups[this->runningGroup]);
160 // rannere il farm come sotto!
161 if (_grp.run() < 0){
162 std::cerr << "Error running the group!" << std::endl;
163 return -1;
164 }
165
166 if (_grp.wait() < 0){
167 std::cerr << "Error waiting the group!" << std::endl;
168 return -1;
169 }
170
171 #ifdef DFF_MPI
172 if (usedProtocol == Proto::MPI)
173 if (MPI_Finalize() != MPI_SUCCESS) abort();
174 #endif
175
176 return 0;
177 }

200 void load( Archive & ar ){
201 ar(cereal::make_nvp("name", name));
202
203 try {
204 std::string endpoint;
205 ar(cereal::make_nvp("endpoint", endpoint)); std::vector endp(split(endpoint, ':'));
206 address = endp[0]; port = std::stoi(endp[1]);
207 } catch (cereal::Exception&) {ar.setNextName(nullptr);}
208
209 try {
210 ar(cereal::make_nvp("batchSize", batchSize));
211 } catch (cereal::Exception&) {ar.setNextName(nullptr);}
212
213 try {
214 ar(cereal::make_nvp("internalMessageOTF", internalMessageOTF));
215 } catch (cereal::Exception&) {ar.setNextName(nullptr);}
216
217 try {
218 ar(cereal::make_nvp("messageOTF", messageOTF));

220
221 try {
222 ar(cereal::make_nvp("threadMapping", threadMapping));
223 } catch (cereal::Exception&) {ar.setNextName(nullptr);}

236 result.push_back (item);

242 void prepareIR(ff_pipeline* parentPipe){
243 ff::ff_IR& runningGroup_IR = annotatedGroups[this->runningGroup];
244 runningGroup_IR.protocol = this->usedProtocol; // set the protocol
245 ff_node* previousStage = getPreviousStage(parentPipe, runningGroup_IR.parentBB);
246 ff_node* nextStage = getNextStage(parentPipe, runningGroup_IR.parentBB);
247 // TODO: check coverage all 1st level
248
249 for(size_t i = 0; i < parsedGroups.size(); i++){
250 const G & g = parsedGroups[i];
251
252 // throw an error if a group in the configuration has not been annotated in the current program
253 if (!annotatedGroups.contains(g.name)) throw FF_Exception("present in the configuration file has not been implemented! :(");
254 auto endpoint = this->usedProtocol == Proto::TCP ? ff_endpoint(g.address, g.port) : ff_endpoint(i);
255 endpoint.groupName = g.name;
256
257 // annotate the listen endpoint for the specified group
258 annotatedGroups[g.name].listenEndpoint = endpoint;
259 // set the batch size for each group
260 if (g.batchSize > 1) annotatedGroups[g.name].outBatchSize = g.batchSize;
261 if (g.messageOTF) annotatedGroups[g.name].messageOTF = g.messageOTF;
262 if (g.internalMessageOTF) annotatedGroups[g.name].internalMessageOTF = g.internalMessageOTF;
263 }
264
265 // TODO check first level pipeline before strting building the groups.
266
267 // build the map parentBB -> <groups names>
268 std::map<ff_node*, std::set<std::string>> parentBB2GroupsName;
269 for(auto& p: annotatedGroups) parentBB2GroupsName[p.second.parentBB].insert(p.first);
270
271 // iterate over the 1st level building blocks and the list of create groups
272 for(const auto& pair : parentBB2GroupsName){
273
274 //just build the current previous the current and the next stage of the parentbuilding block i'm going to exeecute //// TODO: reprhase this comment!
275 //if (!(pair.first == previousStage || pair.first == runningGroup_IR.parentBB || pair.first == nextStage))
276 // continue;
277
278 bool isSrc = isSource(pair.first, parentPipe);
279 bool isSnk = isSink(pair.first, parentPipe);
280 // check if from the under analysis 1st level building block it has been created just one group
281 if (pair.second.size() == 1){
282 // if the unique group is not the one i'm going to run just skip this 1st level building block
283 if ((isSrc || pair.first->isDeserializable()) && (isSnk || pair.first->isSerializable())){
284 auto& ir_ = annotatedGroups[*pair.second.begin()];
285 ir_.insertInList(std::make_pair(pair.first, SetEnum::L), true);
286 ir_.isSink = isSnk; ir_.isSource = isSrc;
287 }
288 continue; // skip anyway
289 }
290
291 // if i'm here it means that from this 1st level building block, multiple groups have been created! (An Error or an A2A or a Pipe BB)
292
293
294 // multiple groups created from a pipeline!
295 if (pair.first->isPipe()){
296 bool isMainPipe = pair.first == parentPipe;
297 ff_pipeline* originalPipe = reinterpret_cast<ff_pipeline*>(pair.first);
298
299 // if the pipe coincide with the main Pipe, just ignore the fact that is wrapped around, since it will be handled later on!
300 bool iswrappedAround = isMainPipe ? false : originalPipe->isset_wraparound();
301
302 // check that all stages were annotated
303 for(ff_node* child : originalPipe->getStages())
304 if (!annotated.contains(child)){
305 error("When create a group from a pipeline, all the stages must be annotated on a group!");
306 abort();
307 }
308
309 for(auto& gName: pair.second){
310 ff_pipeline* mypipe = new ff_pipeline;
311
312 for(ff_node* child : originalPipe->getStages()){
313 if (annotated[child] == gName){
314 // if the current builiding pipe has a stages, and the last one is not the previous i'm currently including there is a problem!
315 if (mypipe->getStages().size() != 0 && originalPipe->get_stageindex(mypipe->get_laststage())+1 != originalPipe->get_stageindex(child)) {
316 error("There are some stages missing in the annottation!\n");
317 abort();
318 } else
319 mypipe->add_stage(child);
320 }
321 }
322
323 bool head = mypipe->get_firststage() == originalPipe->get_firststage();
324 bool tail = mypipe->get_laststage() == originalPipe->get_laststage();
325
326 if (((head && isSrc && !iswrappedAround) || mypipe->isDeserializable()) && ((tail && isSnk && !iswrappedAround) || mypipe->isSerializable()))
327 annotatedGroups[gName].insertInList(std::make_pair(mypipe, SetEnum::L));
328 else {
329 error("The group cannot serialize something!\n");
330 abort();
331 }
332
333 if (head && isSrc) annotatedGroups[gName].isSource = true;
334 if (tail && isSnk) annotatedGroups[gName].isSink = true;
335
336
337 if (!tail)
338 annotatedGroups[gName].destinationEndpoints.push_back({ChannelType::FWD, annotatedGroups[annotated[originalPipe->get_nextstage(mypipe->get_laststage())]].listenEndpoint});
339 else if (iswrappedAround)
340 // if this is the last stage & the pipe is wrapper around i connect tot he "head" groups of this pipeline
341 annotatedGroups[gName].destinationEndpoints.push_back({ChannelType::FBK, annotatedGroups[annotated[originalPipe->get_firststage()]].listenEndpoint});
342
343 // add a new expected connection if i'm not the head or i'm the head and the pipeline is wrapped around
344 if (!head || iswrappedAround)
345 annotatedGroups[gName].expectedEOS = 1;
346
347 }
348
349 } else { // multiple groups created from an all2all!
350
351 std::set<std::pair<ff_node*, SetEnum>> children = getChildBB(pair.first);
352
353 std::erase_if(children, [&](auto& p){
354 if (!annotated.contains(p.first)) return false;
355 std::string& groupName = annotated[p.first];
356 if (((isSrc && p.second == SetEnum::L) || p.first->isDeserializable()) && (isSnk || p.first->isSerializable())){
357 annotatedGroups[groupName].insertInList(std::make_pair(p.first, p.second)); return true;
358 }
359 return false;
360 });
361
362 if (!children.empty()){
363 // seconda passata per verificare se ce da qualche parte c'è copertura completa altrimenti errore
364 for(const std::string& gName : pair.second){
365 ff_IR& ir = annotatedGroups[gName];
366 ir.computeCoverage();
367 if (ir.coverageL)
368 std::erase_if(children, [&](auto& p){
369 if (p.second == SetEnum::R && (isSnk || p.first->isSerializable()) && p.first->isDeserializable()){
370 ir.insertInList(std::make_pair(p.first, SetEnum::R)); return true;
371 }
372 return false;
373 });
374
375 if (ir.coverageR)
376 std::erase_if(children, [&](auto& p){
377 if (p.second == SetEnum::L && (isSrc || p.first->isDeserializable()) && p.first->isSerializable()){
378 ir.insertInList(std::make_pair(p.first, SetEnum::L)); return true;
379 }
380 return false;
381 });
382 }
383
384 // ancora dei building block figli non aggiunti a nessun gruppo, lancio errore e abortisco l'esecuzione
385 if (!children.empty()){
386 std::cerr << "Some building block has not been annotated and no coverage found! You missed something. Aborting now" << std::endl;
387 abort();
388 }
389 } else
390 for(const std::string& gName : pair.second) {
391 annotatedGroups[gName].computeCoverage();
392 // compute the coverage anyway
393 }
394
395 for(const std::string& _gName : pair.second){
396 auto& _ir = annotatedGroups[_gName];
397 // set the isSrc and isSink fields
398 _ir.isSink = isSnk; _ir.isSource = isSrc;
399 // populate the set with the names of other groups created from this 1st level BB
400 _ir.otherGroupsFromSameParentBB = pair.second;
401 }
402
403 }
404 //runningGroup_IR.isSink = isSnk; runningGroup_IR.isSource = isSrc;
405
406 //runningGroup_IR.otherGroupsFromSameParentBB = pair.second;
407
408 }
409
410
411 // compute routing for horizzontally splitted A2A
412 // this is meaningful only if the group is horizontal and made of an a2a
413 if (!runningGroup_IR.isVertical()){
414 assert(runningGroup_IR.parentBB->isAll2All());
415 ff_a2a* parentA2A = reinterpret_cast<ff_a2a*>(runningGroup_IR.parentBB);
416 {
417 ff::svector<ff_node*> inputs;
418 for(ff_node* child : parentA2A->getSecondSet()) child->get_in_nodes(inputs);
419 runningGroup_IR.rightTotalInputs = inputs.size();
420 }
421 {
422 ff::svector<ff_node*> outputs;
423 for(ff_node* child : parentA2A->getFirstSet()) child->get_out_nodes(outputs);
424 runningGroup_IR.leftTotalOuputs = outputs.size();
425 }
426 }
427
428
429 //############# compute the number of excpected input connections
430
431 // handle horizontal groups
432 if (runningGroup_IR.hasRightChildren()){
433 auto& currentGroups = parentBB2GroupsName[runningGroup_IR.parentBB];
434 runningGroup_IR.expectedEOS = std::count_if(currentGroups.cbegin(), currentGroups.cend(), [&](auto& gName){return (!annotatedGroups[gName].isVertical() || annotatedGroups[gName].hasLeftChildren());});
435 // if the current group is horizontal count out itsleft from the all horizontals
436 if (!runningGroup_IR.isVertical()) runningGroup_IR.expectedEOS -= 1;
437 }
438
439 // if the previousStage exists, count all the ouput groups pointing to the one i'm going to run
440 // if the runningGroup comes from a pipe, make sure i'm the head
441 if (previousStage && runningGroup_IR.hasLeftChildren() && (!runningGroup_IR.parentBB->isPipe() || inputGroups(parentBB2GroupsName[runningGroup_IR.parentBB]).contains(runningGroup)))
442 runningGroup_IR.expectedEOS += outputGroups(parentBB2GroupsName[previousStage]).size();
443
444 // FEEDBACK RELATED (wrap around of the main pipe!)
445 // if the main pipe is wrapped-around i take all the outputgroups of the last stage of the pipeline and the cardinality must set to the expected input connections
446 if (!previousStage && parentPipe->isset_wraparound())
447 runningGroup_IR.expectedEOS += outputGroups(parentBB2GroupsName[parentPipe->getStages().back()]).size();
448
449
450 if (runningGroup_IR.expectedEOS > 0) runningGroup_IR.hasReceiver = true;
451
452 //############ compute the name of the outgoing connection groups
453 if (runningGroup_IR.parentBB->isAll2All() && runningGroup_IR.isVertical() && runningGroup_IR.hasLeftChildren() && !runningGroup_IR.wholeParent){
454 // inserisci tutte i gruppi di questo bb a destra
455 for(const auto& gName: parentBB2GroupsName[runningGroup_IR.parentBB])
456 if (!annotatedGroups[gName].isVertical() || annotatedGroups[gName].hasRightChildren())
457 runningGroup_IR.destinationEndpoints.push_back({ChannelType::FWD, annotatedGroups[gName].listenEndpoint});
458 }
459 else if (runningGroup_IR.parentBB->isPipe() && runningGroup_IR.parentBB != parentPipe){
460 if (nextStage && outputGroups(parentBB2GroupsName[runningGroup_IR.parentBB]).contains(runningGroup))
461 for(const auto& gName : inputGroups(parentBB2GroupsName[nextStage]))
462 runningGroup_IR.destinationEndpoints.push_back({ChannelType::FWD, annotatedGroups[gName].listenEndpoint});
463 }
464 else {
465 if (!runningGroup_IR.isVertical()){
466 // inserisci tutti i gruppi come sopra
467 for(const auto& gName: parentBB2GroupsName[runningGroup_IR.parentBB])
468 if ((!annotatedGroups[gName].isVertical() || annotatedGroups[gName].hasRightChildren()) && gName != runningGroup)
469 runningGroup_IR.destinationEndpoints.push_back({ChannelType::INT, annotatedGroups[gName].listenEndpoint});
470 }
471
472 if (nextStage)
473 for(const auto& gName : inputGroups(parentBB2GroupsName[nextStage]))
474 runningGroup_IR.destinationEndpoints.push_back({ChannelType::FWD, annotatedGroups[gName].listenEndpoint});
475 else
476 // FEEDBACK RELATED (wrap around of the main pipe!)
477 if (parentPipe->isset_wraparound()){
478 for(const auto& gName : inputGroups(parentBB2GroupsName[parentPipe->getStages().front()]))
479 runningGroup_IR.destinationEndpoints.push_back({ChannelType::FBK, annotatedGroups[gName].listenEndpoint});
480 }
481 }
482
483
484 runningGroup_IR.buildIndexes();
485
486 if (!runningGroup_IR.destinationEndpoints.empty()) runningGroup_IR.hasSender = true;
487
488 // experimental building the expected routing table for the running group offline (i.e., statically)
489 if (runningGroup_IR.hasSender)
490 for(auto& [ct, ep] : runningGroup_IR.destinationEndpoints){
491 auto& destIR = annotatedGroups[ep.groupName];
492 destIR.buildIndexes();
493 bool internalConnection = ct == ChannelType::INT || (ct == ChannelType::FWD && runningGroup_IR.parentBB == destIR.parentBB); //runningGroup_IR.parentBB == destIR.parentBB;
494 runningGroup_IR.routingTable[ep.groupName] = std::make_pair(destIR.getInputIndexes(internalConnection), ct);
495 }
496
497 }
498
499 std::set<std::string> outputGroups(std::set<std::string> groupNames){
500 if (groupNames.size() > 1)
501 std::erase_if(groupNames, [this](const auto& gName){
502 ff_IR& ir = annotatedGroups[gName];
503 return ((ir.parentBB->isAll2All() && ir.isVertical() && ir.hasLeftChildren()) || (ir.parentBB->isPipe() && annotated[reinterpret_cast<ff_pipeline*>(ir.parentBB)->get_laststage()] != gName));
504 });
505 return groupNames;
506 }
507
508 std::set<std::string> inputGroups(std::set<std::string> groupNames){
509 if (groupNames.size() > 1)
510 std::erase_if(groupNames,[this](const auto& gName){
511 ff_IR& ir = annotatedGroups[gName];
512 return ((ir.parentBB->isAll2All() && ir.isVertical() && annotatedGroups[gName].hasRightChildren()) || (ir.parentBB->isPipe() && annotated[reinterpret_cast<ff_pipeline*>(ir.parentBB)->get_firststage()] != gName));
513 });
514 return groupNames;
515 }

521static inline int DFF_Init(int& argc, char**& argv){
522 struct sigaction s;
523 memset(&s,0,sizeof(s));
524 s.sa_handler=SIG_IGN;
525 if ( (sigaction(SIGPIPE,&s,NULL) ) == -1 ) {
526 perror("sigaction");
527 return -1;
528 }
529
530
531 std::string configFile, groupName;
532
533 for(int i = 0; i < argc; i++){
534 if (strstr(argv[i], "--DFF_Config") != NULL){
535 char * equalPosition = strchr(argv[i], '=');
536 if (equalPosition == NULL){
537 // the option is in the next argument array position
538 configFile = std::string(argv[i+1]);
539 argv[i] = argv[i+1] = NULL;
540 i++;
541 } else {
542 // the option is in the next position of this string
543 configFile = std::string(++equalPosition);
544 argv[i] = NULL;
545 }
546 continue;
547 }
548
549
550 if (strstr(argv[i], "--DFF_GName") != NULL){
551 char * equalPosition = strchr(argv[i], '=');
552 if (equalPosition == NULL){
553 // the option is in the next argument array position
554 groupName = std::string(argv[i+1]);
555 argv[i] = argv[i+1] = NULL;
556 i++;
557 } else {
558 // the option is in the next position of this string
559 groupName = std::string(++equalPosition);
560 argv[i] = NULL;
561 }
562 continue;
563 }
564 }
565
566 if (configFile.empty()){
567 ff::error("Config file not passed as argument!\nUse option --DFF_Config=\"config-file-name\"\n");
568 return -1;
569 }
570
571 dGroups::Instance()->parseConfig(configFile);
572
573 if (!groupName.empty())
574 dGroups::Instance()->forceProtocol(Proto::TCP);
575 #ifdef DFF_MPI
576 else
577 dGroups::Instance()->forceProtocol(Proto::MPI);
578 #endif
579
580
581 if (dGroups::Instance()->usedProtocol == Proto::TCP){
582 if (groupName.empty()){
583 ff::error("Group not passed as argument!\nUse option --DFF_GName=\"group-name\"\n");
584 return -1;
585 }
586 dGroups::Instance()->setRunningGroup(groupName);
587 }
588
589 #ifdef DFF_MPI
590 if (dGroups::Instance()->usedProtocol == Proto::MPI){
591 //MPI_Init(&argc, &argv);
592 int provided;
593
594 if (MPI_Init_thread(&argc, &argv, MPI_THREAD_MULTIPLE, &provided) != MPI_SUCCESS)
595 return -1;
596
597
598 // no thread support
599 if (provided < MPI_THREAD_MULTIPLE){
600 error("No thread support by MPI\n");
601 return -1;
602 }
603
604 int myrank;
605 MPI_Comm_rank(MPI_COMM_WORLD, &myrank);
606 dGroups::Instance()->setRunningGroupByRank(myrank);
607
608 std::cout << "Running group: " << dGroups::Instance()->getRunningGroup() << " on rank: " << myrank << "\n";
609 }
610
611 #endif
612
613 // set the mapping if specified
614 dGroups::Instance()->setThreadMapping();
615
616 // set the name for the printer
617 ff::cout.setPrefix(dGroups::Instance()->getRunningGroup());
618
619 // recompact the argv array
620 int j = 0;
621 for(int i = 0; i < argc; i++)
622 if (argv[i] != NULL)
623 argv[j++] = argv[i];
624
625 // update the argc value
626 argc = j;
627
628 return 0;
629}
