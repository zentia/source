40 class SP_API DebugExtension : public SpineExtension {
41 struct Allocation {
42 void *address;
43 size_t size;
44 const char *fileName;
45 int line;
46
47 Allocation() : address(NULL), size(0), fileName(NULL), line(0) {
48 }
49
50 Allocation(void *a, size_t s, const char *f, int l) : address(a), size(s), fileName(f), line(l) {
51 }
52 };
53
54 public:
55 DebugExtension(SpineExtension *extension) : _extension(extension), _allocations(0), _reallocations(0),
56 _frees(0) {
57 }
58
59 void reportLeaks() {
60 for (std::map<void *, Allocation>::iterator it = _allocated.begin(); it != _allocated.end(); it++) {
61 printf("\"%s:%i (%zu bytes at %p)\n", it->second.fileName, it->second.line, it->second.size,
62 it->second.address);
63 }
64 printf("allocations: %zu, reallocations: %zu, frees: %zu\n", _allocations, _reallocations, _frees);
65 if (_allocated.empty()) printf("No leaks detected\n");
66 }
67
68 void clearAllocations() {
69 _allocated.clear();
70 _usedMemory = 0;
71 }
72
73 virtual void *_alloc(size_t size, const char *file, int line) {
74 void *result = _extension->_alloc(size, file, line);
75 _allocated[result] = Allocation(result, size, file, line);
76 _allocations++;
77 _usedMemory += size;
78 return result;
79 }
80
81 virtual void *_calloc(size_t size, const char *file, int line) {
82 void *result = _extension->_calloc(size, file, line);
83 _allocated[result] = Allocation(result, size, file, line);
84 _allocations++;
85 _usedMemory += size;
86 return result;
87 }
88
89 virtual void *_realloc(void *ptr, size_t size, const char *file, int line) {
90 if (_allocated.count(ptr)) _usedMemory -= _allocated[ptr].size;
91 _allocated.erase(ptr);
92 void *result = _extension->_realloc(ptr, size, file, line);
93 _reallocations++;
94 _allocated[result] = Allocation(result, size, file, line);
95 _usedMemory += size;
96 return result;
97 }
98
99 virtual void _free(void *mem, const char *file, int line) {
100 if (_allocated.count(mem)) {
101 _extension->_free(mem, file, line);
102 _frees++;
103 _usedMemory -= _allocated[mem].size;
104 _allocated.erase(mem);
105 return;
106 }
107
108 printf("%s:%i (address %p): Double free or not allocated through SpineExtension\n", file, line, mem);
109 _extension->_free(mem, file, line);
110 }
111
112 virtual char *_readFile(const String &path, int *length) {
113 auto data = _extension->_readFile(path, length);
114
115 if (_allocated.count(data) == 0) {
116 _allocated[data] = Allocation(data, sizeof(char) * (*length), nullptr, 0);
117 _allocations++;
118 _usedMemory += sizeof(char) * (*length);
119 }
120
121 return data;
122 }
123
124 size_t getUsedMemory() {
125 return _usedMemory;
126 }
127
128 private:
129 SpineExtension *_extension;
130 std::map<void *, Allocation> _allocated;
131 size_t _allocations;
132 size_t _reallocations;
133 size_t _frees;
134 size_t _usedMemory;
135 };
136}
