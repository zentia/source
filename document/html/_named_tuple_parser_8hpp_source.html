82 static Result<NamedTuple<FieldTypes...>> read(
83 const R& _r, const InputVarType& _var) noexcept {
84 static_assert(
85 internal::no_duplicate_field_names<typename NamedTupleType::Fields>());
86 alignas(NamedTuple<FieldTypes...>) unsigned char
87 buf[sizeof(NamedTuple<FieldTypes...>)];
88 auto ptr = internal::ptr_cast<NamedTuple<FieldTypes...>*>(&buf);
89 auto view = rfl::to_view(*ptr);
90 using ViewType = std::remove_cvref_t<decltype(view)>;
91 const auto [set, err] =
92 Parser<R, W, ViewType, ProcessorsType>::read_view(_r, _var, &view);
93 if (err) [[unlikely]] {
94 call_destructors_where_necessary(set, &view);
95 return error(*err);
96 }
97 auto res = Result<NamedTuple<FieldTypes...>>(std::move(*ptr));
98 call_destructors_where_necessary(set, &view);
99 return res;

171 static void add_field_to_object(const W& _w,
172 const NamedTuple<FieldTypes...>& _tup,
173 OutputObjectOrArrayType* _ptr) noexcept {
174 using FieldType = internal::nth_element_t<_i, FieldTypes...>;
175 using ValueType = std::remove_cvref_t<typename FieldType::Type>;
176 const auto value = rfl::get<_i>(_tup);
177 if constexpr (internal::is_extra_fields_v<ValueType>) {
178 for (const auto& [k, v] : *value) {
179 const auto new_parent = make_parent(k, _ptr);
180 Parser<R, W, std::remove_cvref_t<decltype(v)>, ProcessorsType>::write(
181 _w, v, new_parent);
182 }
183 } else if constexpr (!_all_required && !_no_field_names &&
184 !is_required<ValueType, _ignore_empty_containers>()) {
185 constexpr auto name = FieldType::name_.string_view();
186 const auto new_parent = make_parent(name, _ptr);
187 if (!is_empty(value)) {
188 if constexpr (internal::is_attribute_v<ValueType>) {
189 Parser<R, W, ValueType, ProcessorsType>::write(
190 _w, value, new_parent.as_attribute());

192 Parser<R, W, ValueType, ProcessorsType>::write(_w, value, new_parent);
193 }
194 }

247 template <int _i>
248 static void handle_one_missing_field(const std::array<bool, size_>& _found,
249 const NamedTupleType& _view,
250 std::array<bool, size_>* _set,
251 std::vector<Error>* _errors) noexcept {
252 using FieldType = internal::nth_element_t<_i, FieldTypes...>;
253 using ValueType = std::remove_reference_t<
254 std::remove_pointer_t<typename FieldType::Type>>;
255
256 if (!std::get<_i>(_found)) {
257 constexpr bool is_required_field =
258 !internal::is_extra_fields_v<ValueType> &&
259 (_all_required || is_required<ValueType, _ignore_empty_containers>());
260 if constexpr (is_required_field) {
261 constexpr auto current_name =
262 internal::nth_element_t<_i, FieldTypes...>::name();
263 std::stringstream stream;
264 stream << "Field named '" << std::string(current_name)
265 << "' not found.";
266 _errors->emplace_back(Error(stream.str()));
267 } else {
268 if constexpr (!std::is_const_v<ValueType>) {
269 ::new (rfl::get<_i>(_view)) ValueType();
270 } else {
271 using NonConstT = std::remove_const_t<ValueType>;
272 ::new (const_cast<NonConstT*>(rfl::get<_i>(_view))) NonConstT();
273 }
274 std::get<_i>(*_set) = true;
275 }
276 }

297 static std::pair<std::array<bool, NamedTupleType::size()>,
298 std::optional<Error>>
299 read_object_or_array(const R& _r, const InputObjectOrArrayType& _obj_or_arr,
300 NamedTupleType* _view) noexcept {
301 auto found = std::array<bool, NamedTupleType::size()>();
302 found.fill(false);
303 auto set = std::array<bool, NamedTupleType::size()>();
304 set.fill(false);
305 std::vector<Error> errors;
306 const auto reader = ViewReaderType(&_r, _view, &found, &set, &errors);
307 std::optional<Error> err;
308 if constexpr (_no_field_names) {
309 err = _r.read_array(reader, _obj_or_arr);
310 } else {
311 err = _r.read_object(reader, _obj_or_arr);
312 }
313 if (err) {
314 return std::make_pair(set, err);
315 }
316 handle_missing_fields(found, *_view, &set, &errors,
317 std::make_integer_sequence<int, size_>());
318 if (errors.size() != 0) {
319 return std::make_pair(set, to_single_error_message(errors));
320 }
321 return std::make_pair(set, std::optional<Error>());
322 }
