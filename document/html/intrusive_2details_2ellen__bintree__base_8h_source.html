83 struct alignas( void* ) basic_node
84 {
85 enum flags {
86 internal = 1,
87 key_infinite1 = 2,
88 key_infinite2 = 4,
89
90 key_infinite = key_infinite1 | key_infinite2
91 };
92
93 atomics::atomic<unsigned int> m_nFlags;
94
96 explicit basic_node( bool bInternal )
97 {
98 m_nFlags.store( bInternal ? internal: 0, atomics::memory_order_release );
99 }
100
102 bool is_leaf() const
103 {
104 return !is_internal();
105 }
106
108 bool is_internal() const
109 {
110 return (m_nFlags.load(atomics::memory_order_acquire) & internal) != 0;
111 }
112
114 unsigned int infinite_key() const
115 {
116 return m_nFlags.load(atomics::memory_order_acquire) & key_infinite;
117 }
118
120 void infinite_key( int nInf )
121 {
122 unsigned int nFlags = m_nFlags.load(atomics::memory_order_relaxed);
123 nFlags &= ~key_infinite;
124 switch ( nInf ) {
125 case 1:
126 nFlags |= key_infinite1;
127 break;
128 case 2:
129 nFlags |= key_infinite2;
130 break;
131 case 0:
132 break;
133 default:
134 assert( false );
135 break;
136 }
137 m_nFlags.store( nFlags, atomics::memory_order_release );
138 }
