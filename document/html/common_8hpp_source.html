48inline float CNDF( float InputX ) {
49 int sign;
50
51 float OutputX;
52 float xInput;
53 float xNPrimeofX;
54 float expValues;
55 float xK2;
56 float xK2_2, xK2_3;
57 float xK2_4, xK2_5;
58 float xLocal, xLocal_1;
59 float xLocal_2, xLocal_3;
60
61 // Check for negative value of InputX
62 if (InputX < 0.0) {
63 InputX = -InputX;
64 sign = 1;
65 }
66 else {
67 sign = 0;
68 }
69
70 xInput = InputX;
71
72 // Compute NPrimeX term common to both four & six decimal accuracy calcs
73 expValues = std::exp(-0.5f * InputX * InputX);
74 xNPrimeofX = expValues;
75 xNPrimeofX = xNPrimeofX * inv_sqrt_2xPI;
76
77 xK2 = 0.2316419 * xInput;
78 xK2 = 1.0 + xK2;
79 xK2 = 1.0 / xK2;
80 xK2_2 = xK2 * xK2;
81 xK2_3 = xK2_2 * xK2;
82 xK2_4 = xK2_3 * xK2;
83 xK2_5 = xK2_4 * xK2;
84
85 xLocal_1 = xK2 * 0.319381530;
86 xLocal_2 = xK2_2 * (-0.356563782);
87 xLocal_3 = xK2_3 * 1.781477937;
88 xLocal_2 = xLocal_2 + xLocal_3;
89 xLocal_3 = xK2_4 * (-1.821255978);
90 xLocal_2 = xLocal_2 + xLocal_3;
91 xLocal_3 = xK2_5 * 1.330274429;
92 xLocal_2 = xLocal_2 + xLocal_3;
93
94 xLocal_1 = xLocal_2 + xLocal_1;
95 xLocal = xLocal_1 * xNPrimeofX;
96 xLocal = 1.0 - xLocal;
97
98 OutputX = xLocal;
99
100 if (sign) {
101 OutputX = 1.0 - OutputX;
102 }
103
104 return OutputX;
105}

109inline float BlkSchlsEqEuroNoDiv(
110 float sptprice, float strike, float rate,
111 float volatility, float time, int otype, float) {
112
113 float OptionPrice;
114
115 // local private working variables for the calculation
116 //float xStockPrice; // These two variables are not used
117 //float xStrikePrice;
118 float xRiskFreeRate;
119 float xVolatility;
120 float xTime;
121 float xSqrtTime;
122
123 float logValues;
124 float xLogTerm;
125 float xD1;
126 float xD2;
127 float xPowerTerm;
128 float xDen;
129 float d1;
130 float d2;
131 float FutureValueX;
132 float NofXd1;
133 float NofXd2;
134 float NegNofXd1;
135 float NegNofXd2;
136
137 //xStockPrice = sptprice;
138 //xStrikePrice = strike;
139 xRiskFreeRate = rate;
140 xVolatility = volatility;
141
142 xTime = time;
143 xSqrtTime = std::sqrt(xTime);
144
145 logValues = std::log(sptprice/strike);
146
147 xLogTerm = logValues;
148
149 xPowerTerm = xVolatility * xVolatility;
150 xPowerTerm = xPowerTerm * 0.5;
151
152 xD1 = xRiskFreeRate + xPowerTerm;
153 xD1 = xD1 * xTime;
154 xD1 = xD1 + xLogTerm;
155
156 xDen = xVolatility * xSqrtTime;
157 xD1 = xD1 / xDen;
158 xD2 = xD1 - xDen;
159
160 d1 = xD1;
161 d2 = xD2;
162
163 NofXd1 = CNDF( d1 );
164 NofXd2 = CNDF( d2 );
165
166 FutureValueX = strike * ( std::exp( -(rate)*(time) ) );
167 if (otype == 0) {
168 OptionPrice = (sptprice * NofXd1) - (FutureValueX * NofXd2);
169 }
170 else {
171 NegNofXd1 = (1.0 - NofXd1);
172 NegNofXd2 = (1.0 - NofXd2);
173 OptionPrice = (FutureValueX * NegNofXd2) - (sptprice * NegNofXd1);
174 }
175
176 return OptionPrice;
177}
