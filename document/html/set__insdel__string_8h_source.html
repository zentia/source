316
317 cds_test::thread_pool& pool = get_pool();

329

333 size_t nDeleteFailed = 0;
334 for ( size_t i = 0; i < pool.size(); ++i ) {
335 cds_test::thread& thr = pool.get( i );
336 switch ( thr.type()) {

343 break;
