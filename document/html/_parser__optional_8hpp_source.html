28 static Result<std::optional<T>> read(const R& _r,

32 ProcessorsType>;

34 return _r.template read_union<std::optional<T>, O>(_u);
35 };
36 return _r.to_union(_var).and_then(to_optional);
37 } else {
38 if (_r.is_empty(_var)) {
39 return std::optional<T>();
40 }

42 return Parser<R, W, std::remove_cvref_t<T>, ProcessorsType>::read(_r,
43 _var)
44 .transform(to_opt);
45 }

49 static void write(const W& _w, const std::optional<T>& _o,
50 const P& _parent) noexcept {
51 if constexpr (schemaful::IsSchemafulWriter<W>) {
52 auto u = ParentType::add_union(_w, _parent);
53 using UnionType = typename ParentType::template Union<decltype(u)>;

55 UnionType{.index_ = static_cast<size_t>(_o ? 0 : 1), .union_ = &u};
56 if (_o) {

58 } else {
59 ParentType::add_null(_w, p);

63 Parser<R, W, std::remove_cvref_t<T>, ProcessorsType>::write(_w, *_o,
64 _parent);
65 } else {

67 }
