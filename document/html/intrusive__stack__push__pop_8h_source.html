224 template <typename Stack>
225 void do_test( Stack& stack, value_array<typename Stack::value_type>& arrValue )
226 {
227 cds_test::thread_pool& pool = get_pool();
228
229 s_nWorkingProducers.store( s_nPushThreadCount, atomics::memory_order_release );
230 size_t const nPushCount = s_nStackSize / s_nPushThreadCount;
231
232 typename Stack::value_type * pValStart = arrValue.get();
233 typename Stack::value_type * pValEnd = pValStart + s_nStackSize;
234
235 pool.add( new Producer<Stack>( pool, stack ), s_nPushThreadCount );
236 {
237 for ( typename Stack::value_type * it = pValStart; it != pValEnd; ++it )
238 it->nConsumer = c_nBadConsumer;
239
240 typename Stack::value_type * pStart = pValStart;
241 for ( size_t thread_no = 0; thread_no < pool.size(); ++thread_no ) {
242 static_cast<Producer<Stack>&>(pool.get( thread_no )).m_pStart = pStart;
243 pStart += nPushCount;
244 static_cast<Producer<Stack>&>(pool.get( thread_no )).m_pEnd = pStart;
245 }
246 }
247 pool.add( new Consumer<Stack>( pool, stack ), s_nPopThreadCount );
248
249 propout() << std::make_pair( "producer_thread_count", s_nPushThreadCount )
250 << std::make_pair( "consumer_thread_count", s_nPopThreadCount )
251 << std::make_pair( "push_count", nPushCount * s_nPushThreadCount )
252;
253
254 std::chrono::milliseconds duration = pool.run();
255
256 propout() << std::make_pair( "duration", duration );
257
258 s_nStackSize = nPushCount * s_nPushThreadCount;
259
260 {
261 typename Stack::value_type * pEnd = pValStart + s_nStackSize;
262 size_t const nBadConsumer = c_nBadConsumer;
263 for ( typename Stack::value_type * it = pValStart; it != pEnd; ++it )
264 EXPECT_NE( it->nConsumer, nBadConsumer );
265 }
266
267 analyze( stack );
268
269 propout() << stack.statistics();
270 }
