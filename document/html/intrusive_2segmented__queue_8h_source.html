338 segment * create_tail( segment * pTail, typename gc::Guard& guard )
339 {
340 // pTail is guarded by GC
341
342 m_Stat.onCreateSegmentReq();
343
344 scoped_lock l( m_Lock );
345
346 if ( !m_List.empty() && ( pTail != &m_List.back() || get_version(pTail) != m_List.back().version )) {
347 m_pTail.store( &m_List.back(), memory_model::memory_order_relaxed );
348
349 return guard.assign( &m_List.back());
350 }
351
352# ifdef _DEBUG
353 assert( m_List.empty() || populated( m_List.back()));
354# endif
355
356 segment * pNew = allocate_segment();
357 m_Stat.onSegmentCreated();
358
359 if ( m_List.empty())
360 m_pHead.store( pNew, memory_model::memory_order_release );
361 m_List.push_back( *pNew );
362 m_pTail.store( pNew, memory_model::memory_order_release );
363 return guard.assign( pNew );
364 }
365
366 segment * remove_head( segment * pHead, typename gc::Guard& guard )
367 {
368 // pHead is guarded by GC
369 m_Stat.onDeleteSegmentReq();
370
371 segment * pRet;
372 {
373 scoped_lock l( m_Lock );
374
375 if ( m_List.empty()) {
376 m_pTail.store( nullptr, memory_model::memory_order_relaxed );
377 m_pHead.store( nullptr, memory_model::memory_order_relaxed );
378 return guard.assign( nullptr );
379 }
380
381 if ( pHead != &m_List.front() || get_version(pHead) != m_List.front().version ) {
382 m_pHead.store( &m_List.front(), memory_model::memory_order_relaxed );
383 return guard.assign( &m_List.front());
384 }
385
386# ifdef _DEBUG
387 assert( exhausted( m_List.front()));
388# endif
389
390 m_List.pop_front();
391 if ( m_List.empty()) {
392 pRet = guard.assign( nullptr );
393 m_pTail.store( nullptr, memory_model::memory_order_relaxed );
394 }
395 else
396 pRet = guard.assign( &m_List.front());
397 m_pHead.store( pRet, memory_model::memory_order_release );
398 }
399
400 retire_segment( pHead );
401 m_Stat.onSegmentDeleted();
402
403 return pRet;
404 }

462 bool enqueue( value_type& val )
463 {
464 // LSB is used as a flag in marked pointer
465 assert( (reinterpret_cast<uintptr_t>( &val ) & 1) == 0 );
466
467 typename gc::Guard segmentGuard;
468 segment * pTailSegment = m_SegmentList.tail( segmentGuard );
469 if ( !pTailSegment ) {
470 // no segments, create the new one
471 pTailSegment = m_SegmentList.create_tail( pTailSegment, segmentGuard );
472 assert( pTailSegment );
473 }
474
475 permutation_generator gen( quasi_factor());
476
477 // First, increment item counter.
478 // We sure that the item will be enqueued
479 // but if we increment the counter after inserting we can get a negative counter value
480 // if dequeuing occurs before incrementing (enqueue/dequeue race)
481 ++m_ItemCounter;
482
483 while ( true ) {
484 CDS_DEBUG_ONLY( size_t nLoopCount = 0);
485 do {
486 typename permutation_generator::integer_type i = gen;
487 CDS_DEBUG_ONLY( ++nLoopCount );
488 if ( pTailSegment->cells[i].data.load(memory_model::memory_order_relaxed).all()) {
489 // Cell is not empty, go next
490 m_Stat.onPushPopulated();
491 }
492 else {
493 // Empty cell found, try to enqueue here
494 regular_cell nullCell;
495 if ( pTailSegment->cells[i].data.compare_exchange_strong( nullCell, regular_cell( &val ),
496 memory_model::memory_order_release, atomics::memory_order_relaxed ))
497 {
498 // Ok to push item
499 m_Stat.onPush();
500 return true;
501 }
502 assert( nullCell.ptr());
503 m_Stat.onPushContended();
504 }
505 } while ( gen.next());
506
507 assert( nLoopCount == quasi_factor());
508
509 // No available position, create a new segment
510 pTailSegment = m_SegmentList.create_tail( pTailSegment, segmentGuard );
511
512 // Get new permutation
513 gen.reset();
514 }
515 }

626 protected:
627 //@cond
628 bool do_dequeue( typename gc::Guard& itemGuard )
629 {
630 typename gc::Guard segmentGuard;
631 segment * pHeadSegment = m_SegmentList.head( segmentGuard );
632
633 permutation_generator gen( quasi_factor());
634 while ( true ) {
635 if ( !pHeadSegment ) {
636 // Queue is empty
637 m_Stat.onPopEmpty();
638 return false;
639 }
640
641 bool bHadNullValue = false;
642 regular_cell item;
643 CDS_DEBUG_ONLY( size_t nLoopCount = 0 );
644 do {
645 typename permutation_generator::integer_type i = gen;
646 CDS_DEBUG_ONLY( ++nLoopCount );
647
648 // Guard the item
649 // In segmented queue the cell cannot be reused
650 // So no loop is needed here to protect the cell
651 item = pHeadSegment->cells[i].data.load( memory_model::memory_order_relaxed );
652 itemGuard.assign( item.ptr());
653
654 // Check if this cell is empty, which means an element
655 // can be enqueued to this cell in the future
656 if ( !item.ptr())
657 bHadNullValue = true;
658 else {
659 // If the item is not deleted yet
660 if ( !item.bits()) {
661 // Try to mark the cell as deleted
662 if ( pHeadSegment->cells[i].data.compare_exchange_strong( item, item | 1,
663 memory_model::memory_order_acquire, atomics::memory_order_relaxed ))
664 {
665 --m_ItemCounter;
666 m_Stat.onPop();
667
668 return true;
669 }
670 assert( item.bits());
671 m_Stat.onPopContended();
672 }
673 }
674 } while ( gen.next());
675
676 assert( nLoopCount == quasi_factor());
677
678 // scanning the entire segment without finding a candidate to dequeue
679 // If there was an empty cell, the queue is considered empty
680 if ( bHadNullValue ) {
681 m_Stat.onPopEmpty();
682 return false;
683 }
684
685 // All nodes have been dequeued, we can safely remove the first segment
686 pHeadSegment = m_SegmentList.remove_head( pHeadSegment, segmentGuard );
687
688 // Get new permutation
689 gen.reset();
690 }
691 }
692 //@endcond
