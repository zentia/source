192[[nodiscard]] constexpr size_t pixel_storage_align(PixelStorage storage) noexcept {
193 if (is_block_compressed(storage)) {
194 switch (storage) {
195 case PixelStorage::BC1:

197 case PixelStorage::BC2:
198 case PixelStorage::BC3:
199 case PixelStorage::BC5:
200 case PixelStorage::BC6:

205 }
206 switch (storage) {
207 case PixelStorage::BYTE1: return alignof(std::byte) * 1u;
208 case PixelStorage::BYTE2: return alignof(std::byte) * 2u;
209 case PixelStorage::BYTE4: return alignof(std::byte) * 4u;
210 case PixelStorage::SHORT1: return alignof(short) * 1u;

212 case PixelStorage::SHORT4: return alignof(short) * 4u;
213 case PixelStorage::INT1: return alignof(int) * 1u;
214 case PixelStorage::INT2: return alignof(int) * 2u;
215 case PixelStorage::INT4: return alignof(int) * 4u;

218 case PixelStorage::HALF4: return alignof(short) * 4u;
219 case PixelStorage::FLOAT1: return alignof(float) * 1u;

230[[nodiscard]] constexpr size_t pixel_storage_size(PixelStorage storage, uint3 size) noexcept {
231 if (is_block_compressed(storage)) {
232 auto block_width = (size.x + 3u) / 4u;
233 auto block_height = (size.y + 3u) / 4u;
234 auto block_count = block_width * block_height * (std::max)(size.z, 1u);
235 switch (storage) {
236 case PixelStorage::BC1:
237 case PixelStorage::BC4: return block_count * 8u;
238 case PixelStorage::BC2:
239 case PixelStorage::BC3:
240 case PixelStorage::BC5:
241 case PixelStorage::BC6:
242 case PixelStorage::BC7: return block_count * 16u;
243 default: break;
244 }
245 detail::error_pixel_invalid_format("unknown.");
246 }
247 auto pixel_count = size.x * size.y * size.z;
248 switch (storage) {
249 case PixelStorage::BYTE1: return pixel_count * sizeof(std::byte) * 1u;
250 case PixelStorage::BYTE2: return pixel_count * sizeof(std::byte) * 2u;
251 case PixelStorage::BYTE4: return pixel_count * sizeof(std::byte) * 4u;
252 case PixelStorage::SHORT1: return pixel_count * sizeof(short) * 1u;
253 case PixelStorage::SHORT2: return pixel_count * sizeof(short) * 2u;
254 case PixelStorage::SHORT4: return pixel_count * sizeof(short) * 4u;
255 case PixelStorage::INT1: return pixel_count * sizeof(int) * 1u;
256 case PixelStorage::INT2: return pixel_count * sizeof(int) * 2u;
257 case PixelStorage::INT4: return pixel_count * sizeof(int) * 4u;
258 case PixelStorage::HALF1: return pixel_count * sizeof(short) * 1u;
259 case PixelStorage::HALF2: return pixel_count * sizeof(short) * 2u;
260 case PixelStorage::HALF4: return pixel_count * sizeof(short) * 4u;
261 case PixelStorage::FLOAT1: return pixel_count * sizeof(float) * 1u;
262 case PixelStorage::FLOAT2: return pixel_count * sizeof(float) * 2u;
263 case PixelStorage::FLOAT4: return pixel_count * sizeof(float) * 4u;
264 case PixelStorage::R10G10B10A2:
265 case PixelStorage::R11G11B10:
266 return pixel_count * 4;
267 default: break;
268 }
269 detail::error_pixel_invalid_format("unknown");
270}

310[[nodiscard]] constexpr auto pixel_storage_to_format(PixelStorage storage) noexcept {
311 if constexpr (std::is_same_v<T, float>) {
312 switch (storage) {
313 case PixelStorage::BYTE1: return PixelFormat::R8UNorm;
314 case PixelStorage::BYTE2: return PixelFormat::RG8UNorm;
315 case PixelStorage::BYTE4: return PixelFormat::RGBA8UNorm;
316 case PixelStorage::SHORT1: return PixelFormat::R16UNorm;
317 case PixelStorage::SHORT2: return PixelFormat::RG16UNorm;
318 case PixelStorage::SHORT4: return PixelFormat::RGBA16UNorm;
319 case PixelStorage::HALF1: return PixelFormat::R16F;
320 case PixelStorage::HALF2: return PixelFormat::RG16F;
321 case PixelStorage::HALF4: return PixelFormat::RGBA16F;
322 case PixelStorage::FLOAT1: return PixelFormat::R32F;
323 case PixelStorage::FLOAT2: return PixelFormat::RG32F;
324 case PixelStorage::FLOAT4: return PixelFormat::RGBA32F;
325 case PixelStorage::R11G11B10: return PixelFormat::R11G11B10F;
326 case PixelStorage::R10G10B10A2: return PixelFormat::R10G10B10A2UNorm;
327 case PixelStorage::BC1: return PixelFormat ::BC1UNorm;
328 case PixelStorage::BC2: return PixelFormat ::BC2UNorm;
329 case PixelStorage::BC3: return PixelFormat ::BC3UNorm;
330 case PixelStorage::BC4: return PixelFormat ::BC4UNorm;
331 case PixelStorage::BC5: return PixelFormat ::BC5UNorm;
332 case PixelStorage::BC6: return PixelFormat ::BC6HUF16;
333 case PixelStorage::BC7: return PixelFormat ::BC7UNorm;
334 default: detail::error_pixel_invalid_format("float");
335 }
336 } else if constexpr (std::is_same_v<T, int>) {
337 switch (storage) {
338 case PixelStorage::BYTE1: return PixelFormat::R8SInt;
339 case PixelStorage::BYTE2: return PixelFormat::RG8SInt;
340 case PixelStorage::BYTE4: return PixelFormat::RGBA8SInt;
341 case PixelStorage::SHORT1: return PixelFormat::R16SInt;
342 case PixelStorage::SHORT2: return PixelFormat::RG16SInt;
343 case PixelStorage::SHORT4: return PixelFormat::RGBA16SInt;
344 case PixelStorage::INT1: return PixelFormat::R32SInt;
345 case PixelStorage::INT2: return PixelFormat::RG32SInt;
346 case PixelStorage::INT4: return PixelFormat::RGBA32SInt;
347 default: detail::error_pixel_invalid_format("int");
348 }
349 } else if constexpr (std::is_same_v<T, uint>) {
350 switch (storage) {
351 case PixelStorage::BYTE1: return PixelFormat::R8UInt;
352 case PixelStorage::BYTE2: return PixelFormat::RG8UInt;
353 case PixelStorage::BYTE4: return PixelFormat::RGBA8UInt;
354 case PixelStorage::SHORT1: return PixelFormat::R16UInt;
355 case PixelStorage::SHORT2: return PixelFormat::RG16UInt;
356 case PixelStorage::SHORT4: return PixelFormat::RGBA16UInt;
357 case PixelStorage::INT1: return PixelFormat::R32UInt;
358 case PixelStorage::INT2: return PixelFormat::RG32UInt;
359 case PixelStorage::INT4: return PixelFormat::RGBA32UInt;
360 case PixelStorage::R10G10B10A2: return PixelFormat::R10G10B10A2UInt;
361 default: detail::error_pixel_invalid_format("uint");
362 }
363 } else {
364 static_assert(always_false_v<T>);
365 }
366 // unreachable
367 return PixelFormat{};
368}
