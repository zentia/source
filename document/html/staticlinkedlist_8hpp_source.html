135 staticlinkedlist(int cachesize=DEFAULT_CACHE_SIZE, bool /*fillcache*/=false){
136 // avoid unused field warning for padding
137 if (longxCacheLine>1)
138 padding1[0]=padding2[0];
139 // end
140 cache_mem =::malloc((sizeof(Node))*(cachesize+5));
141 unsigned int CPU_cachesize = longxCacheLine*sizeof(void*);
142 if(CAST_TO_UL(cache_mem)%CPU_cachesize){
143 min_cache = (Node *)(
144 ((CAST_TO_UL(cache_mem)
145 /CPU_cachesize)+1
146 )*CPU_cachesize
147 );
148 }else{
149 min_cache = (Node *)(cache_mem);
150 }
151 min_cache_size = cachesize;
152 int i;
153 #if POINTERS_VERSION
154
155 for(i=0; i<cachesize; i++){
156 min_cache[i].next = &min_cache[(i+1)%(cachesize)];
157 min_cache[i].previous = &min_cache[(cachesize + i-1)%(cachesize)];
158 min_cache[i].data = 0;
159 }
160 head = &min_cache[0];
161 tail = &min_cache[1];
162 #else
163 for(i=0; i<cachesize; i++){
164 min_cache[i].next = (i+1)%(cachesize);
165 min_cache[i].previous = (cachesize + i-1)%(cachesize);
166 min_cache[i].data = 0;
167 }
168 head = 0;
169 tail = 1;
170 tail_previous = 0;
171#endif
172 }
