330 T* detach()
331 {
332 T* const pValue = mpValue;
333
334 linked_ptr_base* p = this;
335 do
336 {
337 linked_ptr_base* const pNext = p->mpNext;
338 static_cast<this_type*>(p)->mpValue = NULL;
339 p->mpNext = p->mpPrev = p;
340 p = pNext;
341 }
342 while(p != this);
