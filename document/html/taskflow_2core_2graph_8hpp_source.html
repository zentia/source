548// Procedure: _precede
549/*
550u successor layout: s1, s2, s3, p1, p2 (num_successors = 3)
551v predecessor layout: s1, p1, p2
552
553add a new successor: u->v
554u successor layout:
555 s1, s2, s3, p1, p2, v (push_back v)
556 s1, s2, s3, v, p2, p1 (swap adj[num_successors] with adj[n-1])
557v predecessor layout:
558 s1, p1, p2, u (push_back u)
559*/
560inline void Node::_precede(Node* v) {
561 _edges.push_back(v);
562 std::swap(_edges[_num_successors++], _edges[_edges.size() - 1]);
563 v->_edges.push_back(this);
564
565 //_successors.push_back(v);
566 //v->_predecessors.push_back(this);
567}

569// Function: _remove_successors
570inline void Node::_remove_successors(Node* node) {
571 auto sit = std::remove(_edges.begin(), _edges.begin() + _num_successors, node);
572 size_t new_num_successors = std::distance(_edges.begin(), sit);
573 std::move(_edges.begin() + _num_successors, _edges.end(), sit);
574 _edges.resize(_edges.size() - (_num_successors - new_num_successors));
575 _num_successors = new_num_successors;
576}

582 );

686// Function: _acquire_all
687inline bool Node::_acquire_all(SmallVector<Node*>& nodes) {
688 // assert(_semaphores != nullptr);
689 auto& to_acquire = _semaphores->to_acquire;
690 for(size_t i = 0; i < to_acquire.size(); ++i) {
691 if(!to_acquire[i]->_try_acquire_or_wait(this)) {
692 for(size_t j = 1; j <= i; ++j) {
693 to_acquire[i-j]->_release(nodes);
694 }
695 return false;
696 }
697 }
698 return true;
699}
