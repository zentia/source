108 bool forward(void* task, int destination){
109
110 if (destination == -1) {
111 message_t* msg = nullptr;
112 bool datacopied = true;
113 do {
114 for(size_t i = 0; i < localWorkersMap.size(); i++){
115 long actualDestination = (nextDestination + 1 + i) % localWorkersMap.size();
116 if (ff_send_out_to(task, actualDestination, 1)){ // non blcking ff_send_out_to, we try just once
117 nextDestination = actualDestination;
118 if (msg) {
119 if (!datacopied) msg->data.doNotCleanup();
120 delete msg;
121 }
122 return true;
123 }
124 }
125 if (!msg) {
126 msg = new message_t(index, destination);
127 datacopied = this->n->serializeF(task, msg->data);
128 if (!datacopied) {
129 msg->data.freetaskF = this->n->freetaskF;
130 }
131 }
132 if (ff_send_out_to(msg, localWorkersMap.size(), 1)) {
133 if (datacopied) this->n->freetaskF(task);
134 return true;
135 }
136 } while(1);
137 }
138
139 auto pyshicalDestination = localWorkersMap.find(destination);
140 if (pyshicalDestination != localWorkersMap.end()) {
141 return ff_send_out_to(task, pyshicalDestination->second);
142 } else {
143 message_t* msg = new message_t(index, destination);
144 bool datacopied = this->n->serializeF(task, msg->data);
145 if (!datacopied) {
146 msg->data.freetaskF = this->n->freetaskF;
147 }
148 ff_send_out_to(msg, localWorkersMap.size());
149 if (datacopied) this->n->freetaskF(task);
