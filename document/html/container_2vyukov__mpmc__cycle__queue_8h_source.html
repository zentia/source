207 VyukovMPMCCycleQueue(
208 size_t nCapacity = 0
209 )
210 : m_buffer( nCapacity )
211 , m_nBufferMask( m_buffer.capacity() - 1 )
212 {
213 nCapacity = m_buffer.capacity();
214
215 // Buffer capacity must be power of 2
216 assert( nCapacity >= 2 && (nCapacity & (nCapacity - 1)) == 0 );
217
218 for (size_t i = 0; i != nCapacity; ++i )
219 m_buffer[i].sequence.store(i, memory_model::memory_order_relaxed);
220
221 m_posEnqueue.store(0, memory_model::memory_order_relaxed);
222 m_posDequeue.store(0, memory_model::memory_order_relaxed);
223 }

240 template <typename Func>
241 bool enqueue_with(Func f)
242 {
243 cell_type* cell;
244 back_off bkoff;
245
246 size_t pos = m_posEnqueue.load(memory_model::memory_order_relaxed);
247 for (;;)
248 {
249 cell = &m_buffer[pos & m_nBufferMask];
250 size_t seq = cell->sequence.load(memory_model::memory_order_acquire);
251
252 intptr_t dif = static_cast<intptr_t>(seq) - static_cast<intptr_t>(pos);
253
254 if (dif == 0) {
255 if ( m_posEnqueue.compare_exchange_weak(pos, pos + 1, memory_model::memory_order_relaxed, atomics::memory_order_relaxed ))
256 break;
257 }
258 else if (dif < 0) {
259 // Queue full?
260 if ( pos - m_posDequeue.load( memory_model::memory_order_relaxed ) == capacity())
261 return false; // queue full
262 bkoff();
263 pos = m_posEnqueue.load( memory_model::memory_order_relaxed );
264 }
265 else
266 pos = m_posEnqueue.load(memory_model::memory_order_relaxed);
267 }
268
269 f( cell->data );
270
271 cell->sequence.store(pos + 1, memory_model::memory_order_release);
272 ++m_ItemCounter;
273
274 return true;
275 }

336 template <typename Func>
337 bool dequeue_with( Func f )
338 {
339 cell_type * cell;
340 back_off bkoff;
341
342 size_t pos = m_posDequeue.load( memory_model::memory_order_relaxed );
343 for (;;)
344 {
345 cell = &m_buffer[pos & m_nBufferMask];
346 size_t seq = cell->sequence.load(memory_model::memory_order_acquire);
347 intptr_t dif = static_cast<intptr_t>(seq) - static_cast<intptr_t>(pos + 1);
348
349 if (dif == 0) {
350 if ( m_posDequeue.compare_exchange_weak(pos, pos + 1, memory_model::memory_order_relaxed, atomics::memory_order_relaxed))
351 break;
352 }
353 else if (dif < 0) {
354 // Queue empty?
355 if ( pos - m_posEnqueue.load( memory_model::memory_order_relaxed ) == 0 )
356 return false; // queue empty
357 bkoff();
358 pos = m_posDequeue.load( memory_model::memory_order_relaxed );
359 }
360 else
361 pos = m_posDequeue.load(memory_model::memory_order_relaxed);
362 }
363
364 f( cell->data );
365 value_cleaner()( cell->data );
366 cell->sequence.store( pos + m_nBufferMask + 1, memory_model::memory_order_release );
367 --m_ItemCounter;
368
369 return true;
370 }

397 template <bool SC = c_single_consumer >
398 typename std::enable_if<SC, value_type *>::type front()
399 {
400 static_assert( c_single_consumer, "front() is enabled only if traits::single_consumer is true");
401
402 cell_type * cell;
403 back_off bkoff;
404
405 size_t pos = m_posDequeue.load( memory_model::memory_order_relaxed );
406 for ( ;;)
407 {
408 cell = &m_buffer[pos & m_nBufferMask];
409 size_t seq = cell->sequence.load( memory_model::memory_order_acquire );
410 intptr_t dif = static_cast<intptr_t>(seq) - static_cast<intptr_t>(pos + 1);
411
412 if ( dif == 0 )
413 return &cell->data;
414 else if ( dif < 0 ) {
415 // Queue empty?
416 if ( pos - m_posEnqueue.load( memory_model::memory_order_relaxed ) == 0 )
417 return nullptr; // queue empty
418 bkoff();
419 pos = m_posDequeue.load( memory_model::memory_order_relaxed );
420 }
421 else
422 pos = m_posDequeue.load( memory_model::memory_order_relaxed );
423 }
424 }
