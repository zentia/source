6// This file implements the class template optional that represents optional objects.
7//
8// An optional object is an object that contains the storage for another object and
9// manages the lifetime of this contained object, if any. The contained object may be
10// initialized after the optional object has been initialized, and may be destroyed before
11// the optional object has been destroyed.
12//
13// Any instance of optional<T> at any given time either contains a value or does not
14// contain a value. When an instance of optional<T> contains a value, it means that an
15// object of type T, referred to as the optional object's contained value, is allocated
16// within the storage of the optional object. Implementations are not permitted to use
17// additional storage, such as dynamic memory, to allocate its contained value.
18//
19// The contained value is allocated in the optional<T> storage suitably
20// aligned for the type T. When an object of type optional<T> is contextually converted to
21// bool, the conversion returns true if the object contains a value; otherwise the
22// conversion returns false.
23//
24// T shall be an object type and satisfy the requirements of Destructible.

505 inline EA_CONSTEXPR bool operator==(const optional<T>& lhs, const optional<T>& rhs)
506 {
507 // NOTE:
508 //
509 // Code collapsed onto a single line to satisfy requirements for constexpr expressions
510 // being a single line return statement.
511 //
512 // if(bool(lhs) != bool(rhs))
513 // return false;
514
515 // if(bool(lhs) == false)
516 // return true;
517
518 // return *lhs == *rhs;
519
520 return (bool(lhs) != bool(rhs)) ? false : (bool(lhs) == false) ? true : *lhs == *rhs;
521 }
