219template <typename T>
220void Tensor<T>::_make_chunks(size_t M) {
221
222 size_t P = 1;
223 size_t N = 1;
224
225 for(int i=_shape.size()-1; i>=0; i--) {
226 if(M >= _shape[i]) {
227 _chunk_shape[i] = _shape[i];
228 _chunk_grid[i] = 1;
229 N *= _chunk_shape[i];
230 M /= _shape[i];
231 }
232 else {
233 _chunk_shape[i] = M;
234 _chunk_grid[i] = (_shape[i] + _chunk_shape[i] - 1) / _chunk_shape[i];
235 P *= _chunk_grid[i];
236 N *= _chunk_shape[i];
237 for(i--; i>=0; i--) {
238 _chunk_shape[i] = 1;
239 _chunk_grid[i] = _shape[i];
240 P *= _chunk_grid[i];
241 }
242 break;
243 }
244 }
245
246 _chunks.resize(P);
247
248 // we allocate the first data in memory
249 _chunks[0].data.resize(N);
250
251 // TODO: the rest sits in the disk
252 for(size_t i=1; i<_chunks.size(); ++i) {
253 }
254
255 // assign the storage level
256 _storage_level = (_chunks.size() <= 1) ? MEMORY : MEMORY_AND_DISK;
257}
