188 template <class Set>
189 void test( Set& s )
190 {
191 // Precondition: set is empty
192 // Postcondition: set is empty
193
194 ASSERT_TRUE( s.empty());
195 ASSERT_CONTAINER_SIZE( s, 0 );
196 size_t const nSetSize = std::max( s.head_size() * 2, static_cast<size_t>(100));
197
198 typedef typename Set::value_type value_type;
199
200 std::vector< value_type > data;
201 std::vector< size_t> indices;
202 data.reserve( nSetSize );
203 indices.reserve( nSetSize );
204 for ( size_t key = 0; key < nSetSize; ++key ) {
205 data.push_back( value_type( static_cast<int>( key )));
206 indices.push_back( key );
207 }
208 shuffle( indices.begin(), indices.end());
209
210 // insert/find
211 for ( auto idx : indices ) {
212 auto& i = data[ idx ];
213
214 ASSERT_FALSE( s.contains( i.nKey ));
215 ASSERT_FALSE( s.find( i.nKey, []( value_type& ) {} ));
216
217 std::pair<bool, bool> updResult;
218
219 updResult = s.update( i, false );
220 EXPECT_FALSE( updResult.first );
221 EXPECT_FALSE( updResult.second );
222
223 switch ( i.key() % 3 ) {
224 case 0:
225 ASSERT_TRUE( s.insert( i ));
226 ASSERT_FALSE( s.insert( i ));
227 updResult = s.update( i, false );
228 EXPECT_TRUE( updResult.first );
229 EXPECT_FALSE( updResult.second );
230 break;
231 case 1:
232 EXPECT_EQ( i.nInsertCount, 0u );
233 ASSERT_TRUE( s.insert( i, []( value_type& v ) { ++v.nInsertCount;} ));
234 EXPECT_EQ( i.nInsertCount, 1u );
235 ASSERT_FALSE( s.insert( i, []( value_type& v ) { ++v.nInsertCount;} ));
236 EXPECT_EQ( i.nInsertCount, 1u );
237 i.nInsertCount = 0;
238 break;
239 case 2:
240 updResult = s.update( i );
241 EXPECT_TRUE( updResult.first );
242 EXPECT_TRUE( updResult.second );
243 break;
244 }
245
246 ASSERT_TRUE( s.contains( i.nKey ));
247 EXPECT_EQ( i.nFindCount, 0u );
248 ASSERT_TRUE( s.find( i.nKey, []( value_type& v ) { ++v.nFindCount; } ));
249 EXPECT_EQ( i.nFindCount, 1u );
250 }
251 ASSERT_FALSE( s.empty());
252 ASSERT_CONTAINER_SIZE( s, nSetSize );
253
254 std::for_each( data.begin(), data.end(), []( value_type& v ) { v.clear_stat(); });
255
256 // get_level_statistics
257 {
258 std::vector< typename Set::level_statistics > level_stat;
259 s.get_level_statistics( level_stat );
260 EXPECT_GT( level_stat.size(), 0u );
261 }
262
263 // erase
264 shuffle( indices.begin(), indices.end());
265 for ( auto idx : indices ) {
266 auto& i = data[ idx ];
267
268 ASSERT_TRUE( s.contains( i.nKey ));
269 EXPECT_EQ( i.nFindCount, 0u );
270 ASSERT_TRUE( s.find( i.nKey, []( value_type& v ) { ++v.nFindCount; } ));
271 EXPECT_EQ( i.nFindCount, 1u );
272
273 value_type v( i );
274 switch ( i.key() % 3 ) {
275 case 0:
276 ASSERT_FALSE( s.unlink( v ));
277 ASSERT_TRUE( s.unlink( i ));
278 ASSERT_FALSE( s.unlink( i ));
279 break;
280 case 1:
281 ASSERT_TRUE( s.erase( i.key()));
282 ASSERT_FALSE( s.erase( i.key()));
283 break;
284 case 2:
285 EXPECT_EQ( i.nEraseCount, 0u );
286 ASSERT_TRUE( s.erase( v.key(), []( value_type& val ) { ++val.nEraseCount; } ));
287 EXPECT_EQ( i.nEraseCount, 1u );
288 ASSERT_FALSE( s.erase( v.key(), []( value_type& val ) { ++val.nEraseCount; } ));
289 EXPECT_EQ( i.nEraseCount, 1u );
290 break;
291 }
292
293 ASSERT_FALSE( s.contains( i.nKey ));
294 ASSERT_FALSE( s.find( i.nKey, []( value_type const& ) {} ));
295 }
296 ASSERT_TRUE( s.empty());
297 ASSERT_CONTAINER_SIZE( s, 0 );
298
299 // Force retiring cycle
300 Set::gc::force_dispose();
301 for ( auto& i : data ) {
302 EXPECT_EQ( i.nDisposeCount, 1u );
303 }
304
305 // clear
306 for ( auto& i : data ) {
307 i.clear_stat();
308 ASSERT_TRUE( s.insert( i ));
309 }
310 ASSERT_FALSE( s.empty());
311 ASSERT_CONTAINER_SIZE( s, nSetSize );
312
313 // Forward iterator test
314 for ( auto it = s.begin(); it != s.end(); ++it ) {
315 ++it->nFindCount;
316 }
317 for ( auto it = s.cbegin(); it != s.cend(); ++it ) {
318 EXPECT_EQ( it->nFindCount, 1u );
319 }
320
321 // Reverse iterator test
322 for ( auto it = s.rbegin(); it != s.rend(); ++it ) {
323 ++it->nFindCount;
324 }
325 for ( auto it = s.crbegin(); it != s.crend(); ++it ) {
326 EXPECT_EQ( it->nFindCount, 2u );
327 }
328
329 for ( auto& i : data ) {
330 EXPECT_EQ( i.nFindCount, 2u );
331 }
332
333 // clear test
334 s.clear();
335
336 ASSERT_TRUE( s.empty());
337 ASSERT_CONTAINER_SIZE( s, 0u );
338 ASSERT_TRUE( s.begin() == s.end());
339 ASSERT_TRUE( s.cbegin() == s.cend());
340
341 // Force retiring cycle
342 Set::gc::force_dispose();
343 for ( auto& i : data ) {
344 EXPECT_EQ( i.nDisposeCount, 1u );
345 }
346 }
