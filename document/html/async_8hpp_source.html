58// Procedure: _tear_down_async
59inline void Executor::_tear_down_async(Worker& worker, Node* node, Node*& cache) {
60
61 // from executor
62 if(auto parent = node->_parent; parent == nullptr) {
63 _decrement_topology();
64 }
65 // from runtime
66 else {
67 auto state = parent->_nstate;
68 if(parent->_join_counter.fetch_sub(1, std::memory_order_acq_rel) == 1) {
69 if(state & NSTATE::PREEMPTED) {
70 _update_cache(worker, cache, parent);
71 }
72 }
73 }
74 recycle(node);
75}

172template <typename P, typename F, typename I,
173 std::enable_if_t<is_task_params_v<P> && !std::is_same_v<std::decay_t<I>, AsyncTask>, void>*
174>
175auto Executor::dependent_async(P&& params, F&& func, I first, I last) {
176
177 _increment_topology();
178
179 size_t num_dependents = std::distance(first, last);
180
181 // async with runtime: [] (tf::Runtime&) {}
182 if constexpr (is_runtime_task_v<F>) {
183
184 std::promise<void> p;
185 auto fu{p.get_future()};
186
187 AsyncTask task(animate(
188 NSTATE::NONE, ESTATE::ANCHORED, std::forward<P>(params), nullptr, nullptr, num_dependents,
189 std::in_place_type_t<Node::DependentAsync>{},
190 [p=MoC{std::move(p)}, f=std::forward<F>(func)] (tf::Runtime& rt, bool reentered) mutable {
191 if(!reentered) {
192 f(rt);
193 }
194 else {
195 auto& eptr = rt._parent->_exception_ptr;
196 eptr ? p.object.set_exception(eptr) : p.object.set_value();
197 }
198 }
199 ));
200
201 for(; first != last; first++) {
202 _process_async_dependent(task._node, *first, num_dependents);
203 }
204
205 if(num_dependents == 0) {
206 _schedule_async_task(task._node);
207 }
208
209 return std::make_pair(std::move(task), std::move(fu));
210 }
211 // async without runtime: [] () {}
212 else if constexpr(std::is_invocable_v<F>) {
213
214 using R = std::invoke_result_t<F>;
215 std::packaged_task<R()> p(std::forward<F>(func));
216 auto fu{p.get_future()};
217
218 AsyncTask task(animate(
219 NSTATE::NONE, ESTATE::NONE, std::forward<P>(params), nullptr, nullptr, num_dependents,
220 std::in_place_type_t<Node::DependentAsync>{},
221 [p=make_moc(std::move(p))] () mutable { p.object(); }
222 ));
223
224 for(; first != last; first++) {
225 _process_async_dependent(task._node, *first, num_dependents);
226 }
227
228 if(num_dependents == 0) {
229 _schedule_async_task(task._node);
230 }
231
232 return std::make_pair(std::move(task), std::move(fu));
233 }
234 else {
235 static_assert(dependent_false_v<F>, "invalid async callable");
236 }
237}

294// Procedure: _tear_down_dependent_async
295inline void Executor::_tear_down_dependent_async(Worker& worker, Node* node, Node*& cache) {
296
297 auto handle = std::get_if<Node::DependentAsync>(&(node->_handle));
298
299 // this async task comes from Executor
300 auto target = ASTATE::UNFINISHED;
301
302 while(!handle->state.compare_exchange_weak(target, ASTATE::FINISHED,
303 std::memory_order_acq_rel,
304 std::memory_order_relaxed)) {
305 target = ASTATE::UNFINISHED;
306 }
307
308 // spawn successors whenever their dependencies are resolved
309 for(size_t i=0; i<node->_edges.size(); ++i) {
310 if(auto s = node->_edges[i];
311 s->_join_counter.fetch_sub(1, std::memory_order_acq_rel) == 1
312 ) {
313 _update_cache(worker, cache, s);
314 }
315 }
316
317 // now the executor no longer needs to retain ownership
318 if(handle->use_count.fetch_sub(1, std::memory_order_acq_rel) == 1) {
319 recycle(node);
320 }
321
322 _decrement_topology();
323}
