29template <class R, class W, class VecType, class ProcessorsType>
30 requires AreReaderAndWriter<R, W, VecType>
31struct VectorParser {
32 public:
33 using InputArrayType = typename R::InputArrayType;
34 using InputVarType = typename R::InputVarType;
35
36 using ParentType = Parent<W>;
37
38 using T = typename VecType::value_type;

40 static Result<VecType> read(const R& _r, const InputVarType& _var) noexcept {
41 if constexpr (treat_as_map()) {
42 return MapParser<R, W, VecType, ProcessorsType>::read(_r, _var);
43 } else if constexpr (is_forward_list<VecType>()) {
44 const auto to_forward_list = [](auto&& vec) -> std::forward_list<T> {
45 std::forward_list<T> list;
46 for (auto it = vec.rbegin(); it != vec.rend(); ++it) {
47 list.emplace_front(std::move(*it));
48 }
49 return list;
50 };
51 return Parser<R, W, std::vector<T>, ProcessorsType>::read(_r, _var)
52 .transform(to_forward_list);
53 } else {
54 const auto parse = [&](const InputArrayType& _arr) -> Result<VecType> {
55 VecType vec;
56 auto vector_reader =
57 VectorReader<R, W, VecType, ProcessorsType>(&_r, &vec);
58 const auto err = _r.read_array(vector_reader, _arr);
59 if (err) {
60 return error(*err);
61 }
62 return vec;
63 };
64 return _r.to_array(_var).and_then(parse);
65 }
66 }
67
68 template <class P>
69 static void write(const W& _w, const VecType& _vec,
70 const P& _parent) noexcept {
71 if constexpr (treat_as_map()) {
72 MapParser<R, W, VecType, ProcessorsType>::write(_w, _vec, _parent);
73 } else {
74 auto arr = ParentType::add_array(
75 _w, std::distance(_vec.begin(), _vec.end()), _parent);
76 const auto new_parent = typename ParentType::Array{&arr};
77 for (const auto& v : _vec) {
78 Parser<R, W, std::remove_cvref_t<T>, ProcessorsType>::write(_w, v,
79 new_parent);
80 }
81 _w.end_array(&arr);
82 }
83 }
84
86 static schema::Type to_schema(
87 std::map<std::string, schema::Type>* _definitions) {
88 if constexpr (treat_as_map()) {
89 return MapParser<R, W, VecType, ProcessorsType>::to_schema(_definitions);
90 } else {
91 using Type = schema::Type;
92 return Type{Type::TypedArray{
93 .type_ = Ref<Type>::make(
94 Parser<R, W, T, ProcessorsType>::to_schema(_definitions))}};
95 }
96 }
97
98 private:
99 static constexpr bool treat_as_map() {
100 if constexpr (is_map_like_not_multimap<VecType>()) {
101 if constexpr (internal::has_reflection_type_v<typename T::first_type>) {
102 using U = std::remove_cvref_t<typename T::first_type::ReflectionType>;
103 return std::is_same<U, std::string>() || std::is_integral_v<U> ||
104 std::is_floating_point_v<U>;
105
106 // We do not need std::string here, it is already caught by the template
107 // specialization.
108 } else if constexpr (std::is_integral_v<typename T::first_type> ||
109 std::is_floating_point_v<typename T::first_type>) {
110 return true;
111 } else {
112 return false;
113 }
114 } else {
115 return false;
116 }
117 }
118};
