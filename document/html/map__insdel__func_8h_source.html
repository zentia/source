311 struct erase_functor {
312 value_container m_cnt;
313
314 template <typename Key, typename Val>
315 void operator()( Key const& /*key*/, Val& v )
316 {
317 while ( true ) {
318 if ( v.bInitialized.load( atomics::memory_order_relaxed )) {
319 std::unique_lock< typename value_type::lock_type> ac( v.m_access );
320
321 if ( m_cnt.nKeyExpected == v.nKey && m_cnt.nKeyExpected * 8 == v.nData )
322 ++m_cnt.nSuccessItem;
323 else
324 ++m_cnt.nFailedItem;
325 v.nData++;
326 v.nKey = 0;
327 break;
328 }
329 else
330 cds::backoff::yield()();
331 }
332 }

400 template <class Map>
401 void do_test( Map& testMap )
402 {
403 typedef Inserter<Map> inserter;
404 typedef Deleter<Map> deleter;
405 typedef Updater<Map> updater;
406
407 cds_test::thread_pool& pool = get_pool();
408 pool.add( new inserter( pool, testMap ), s_nInsertThreadCount );
409 pool.add( new deleter( pool, testMap ), s_nDeleteThreadCount );
410 if ( s_nUpdateThreadCount )
411 pool.add( new updater( pool, testMap ), s_nUpdateThreadCount );
412
413 propout() << std::make_pair( "insert_thread_count", s_nInsertThreadCount )
414 << std::make_pair( "delete_thread_count", s_nDeleteThreadCount )
415 << std::make_pair( "update_thread_count", s_nUpdateThreadCount )
416 << std::make_pair( "pass_count", s_nThreadPassCount )
417 << std::make_pair( "map_size", s_nMapSize );
418
419 std::chrono::milliseconds duration = pool.run();
420
421 propout() << std::make_pair( "duration", duration );
422
423 size_t nInsertSuccess = 0;
424 size_t nInsertFailed = 0;
425 size_t nDeleteSuccess = 0;
426 size_t nDeleteFailed = 0;
427 size_t nDelValueSuccess = 0;
428 size_t nDelValueFailed = 0;
429 size_t nUpdateFailed = 0;
430 size_t nUpdateCreated = 0;
431 size_t nUpdateModified = 0;
432 size_t nEnsFuncCreated = 0;
433 size_t nEnsFuncModified = 0;
434 size_t nInsFuncCalled = 0;
435
436 for ( size_t i = 0; i < pool.size(); ++i ) {
437 cds_test::thread& thr = pool.get( i );
438 switch ( thr.type()) {
439 case insert_thread:
440 {
441 inserter& t = static_cast<inserter&>( thr );
442 nInsertSuccess += t.m_nInsertSuccess;
443 nInsertFailed += t.m_nInsertFailed;
444 nInsFuncCalled += t.m_nTestFunctorRef;
445 }
446 break;
447 case delete_thread:
448 {
449 deleter& t = static_cast<deleter&>(thr);
450 nDeleteSuccess += t.m_nDeleteSuccess;
451 nDeleteFailed += t.m_nDeleteFailed;
452 nDelValueSuccess += t.m_nValueSuccess;
453 nDelValueFailed += t.m_nValueFailed;
454 }
455 break;
456 case update_thread:
457 {
458 updater& t = static_cast<updater&>(thr);
459 nUpdateCreated += t.m_nUpdateCreated;
460 nUpdateModified += t.m_nUpdateExisted;
461 nUpdateFailed += t.m_nUpdateFailed;
462 nEnsFuncCreated += t.m_nFunctorCreated;
463 nEnsFuncModified += t.m_nFunctorModified;
464 }
465 break;
466 default:
467 assert( false );
468 }
469 }
470
471 propout()
472 << std::make_pair( "insert_success", nInsertSuccess )
473 << std::make_pair( "insert_failed", nInsertFailed )
474 << std::make_pair( "delete_success", nDeleteSuccess )
475 << std::make_pair( "delete_failed", nDeleteFailed )
476 << std::make_pair( "update_success", nUpdateCreated + nUpdateModified )
477 << std::make_pair( "update_failed", nUpdateFailed )
478 << std::make_pair( "update_functor_create", nEnsFuncCreated )
479 << std::make_pair( "update_functor_modify", nEnsFuncModified )
480 << std::make_pair( "finish_map_size", testMap.size());
481
482 EXPECT_EQ( nDelValueFailed, 0u );
483 EXPECT_EQ( nDelValueSuccess, nDeleteSuccess );
484 EXPECT_EQ( nUpdateFailed, 0u );
485 EXPECT_EQ( nUpdateCreated + nUpdateModified, nEnsFuncCreated + nEnsFuncModified );
486
487 // nInsFuncCalled is call count of insert functor
488 EXPECT_EQ( nInsFuncCalled, nInsertSuccess );
489
490 check_before_cleanup( testMap );
491
492 for ( size_t nItem = 0; nItem < s_nMapSize; ++nItem )
493 testMap.erase( nItem );
494
495 EXPECT_TRUE( testMap.empty());
496 EXPECT_EQ( testMap.size(), 0u );
497
498 additional_check( testMap );
499 print_stat( propout(), testMap );
500 additional_cleanup( testMap );
501 }
