51 // Implementation moving copying both trivial and non-trivial data via a lesser iterator than random-access.
52 template <typename /*InputIteratorCategory*/, bool /*isMove*/, bool /*canMemmove*/>
53 struct move_and_copy_helper
54 {

58 for(; first != last; ++result, ++first)

64 // Specialization for copying non-trivial data via a random-access iterator. It's theoretically faster because the compiler can see the count when its a compile-time const.
65 // This specialization converts the random access InputIterator last-first to an integral type. There's simple way for us to take advantage of a random access output iterator,
66 // as the range is specified by the input instead of the output, and distance(first, last) for a non-random-access iterator is potentially slow.
67 template <>
68 struct move_and_copy_helper<EASTL_ITC_NS::random_access_iterator_tag, false, false>
69 {
70 template <typename InputIterator, typename OutputIterator>
71 static OutputIterator move_or_copy(InputIterator first, InputIterator last, OutputIterator result)
72 {
73 typedef typename eastl::iterator_traits<InputIterator>::difference_type difference_type;
74
75 for(difference_type n = (last - first); n > 0; --n, ++first, ++result)
76 *result = *first;
77
78 return result;
79 }
80 };

111 // Specialization for when we can use memmove/memcpy. See the notes above for what conditions allow this.
112 template <bool isMove>
113 struct move_and_copy_helper<EASTL_ITC_NS::random_access_iterator_tag, isMove, true>
114 {
115 template <typename T>
116 static T* move_or_copy(const T* first, const T* last, T* result)
117 {
118 if (EASTL_UNLIKELY(first == last))
119 return result;
120
121 // We could use memcpy here if there's no range overlap, but memcpy is rarely much faster than memmove.
122 return (T*)memmove(result, first, (size_t)((uintptr_t)last - (uintptr_t)first)) + (last - first);
123 }
124 };

140 template <typename InputIterator, typename OutputIterator>
141 struct can_be_memmoved_helper {
142 using IIC = typename eastl::iterator_traits<InputIterator>::iterator_category;
143 using OIC = typename eastl::iterator_traits<OutputIterator>::iterator_category;
144 using value_type_input = typename eastl::iterator_traits<InputIterator>::value_type;
145 using value_type_output = typename eastl::iterator_traits<OutputIterator>::value_type;
146
147 static constexpr bool value = eastl::is_trivially_copyable<value_type_output>::value &&
148 eastl::is_same<value_type_input, value_type_output>::value &&
149 (eastl::is_pointer<InputIterator>::value || is_contiguous_iterator<IIC>::value) &&
150 (eastl::is_pointer<OutputIterator>::value || is_contiguous_iterator<OIC>::value);
151
152 };
153 }

155 template <bool isMove, typename InputIterator, typename OutputIterator>
156 inline OutputIterator move_and_copy_chooser(InputIterator first, InputIterator last, OutputIterator result)
157 {
158 typedef typename eastl::iterator_traits<InputIterator>::iterator_category IIC;
159
160 const bool canBeMemmoved = internal::can_be_memmoved_helper<InputIterator, OutputIterator>::value;
161
162 return eastl::move_and_copy_helper<IIC, isMove, canBeMemmoved>::move_or_copy(first, last, result); // Need to chose based on the input iterator tag and not the output iterator tag, because containers accept input ranges of iterator types different than self.
163 }

166 // We have a second layer of unwrap_iterator calls because the original iterator might be something like move_iterator<generic_iterator<int*> > (i.e. doubly-wrapped).
167 template <bool isMove, typename InputIterator, typename OutputIterator>
168 inline OutputIterator move_and_copy_unwrapper(InputIterator first, InputIterator last, OutputIterator result)
169 {
170 return OutputIterator(eastl::move_and_copy_chooser<isMove>(eastl::unwrap_iterator(first), eastl::unwrap_iterator(last), eastl::unwrap_iterator(result))); // Have to convert to OutputIterator because unwrap_iterator(result) could be a T*
171 }
