67 template <class T>
68 rfl::Result<T> to_basic_type(const InputVarType& _var) const noexcept {
69 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
70 if (!_var.val_->is_string()) {
71 return error("Could not cast to string.");
72 }
73 return _var.val_->as<std::string>();
74
75 } else if constexpr (std::is_same<std::remove_cvref_t<T>,
76 rfl::Bytestring>()) {
77 if (!_var.val_->is_byte_string()) {
78 return error("Could not cast to bytestring.");
79 }
80 const auto vec = _var.val_->as<std::vector<uint8_t>>();
81 const auto data = internal::ptr_cast<const std::byte*>(vec.data());
82 return rfl::Bytestring(data, data + vec.size());
83
84 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
85 if (!_var.val_->is_bool()) {
86 return error("Could not cast to boolean.");
87 }
88 return _var.val_->as<bool>();
89
90 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
91 if (_var.val_->is_double()) {
92 return static_cast<T>(_var.val_->as<double>());
93 }
94 return error("Could not cast to double.");
95
96 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
97 if (_var.val_->is_int64()) {
98 return static_cast<T>(_var.val_->as<int64_t>());
99 }
100 if (_var.val_->is_uint64()) {
101 return static_cast<T>(_var.val_->as<uint64_t>());
102 }
103 return error("Could not cast to integer.");
104
105 } else {
106 static_assert(rfl::always_false_v<T>, "Unsupported type.");
107 }
108 }
