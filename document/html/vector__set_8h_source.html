6// This file implements vector_set. It acts much like std::set, except its
7// underlying representation is a random access container such as vector.
8// These containers are sometimes also known as "sorted vectors."
9// vector_sets have an advantage over conventional sets in that their memory
10// is contiguous and node-less. The result is that lookups are faster, more
11// cache friendly (which potentially more so benefits speed), and the container
12// uses less memory. The downside is that inserting new items into the container
13// is slower if they are inserted in random order instead of in sorted order.
14// This tradeoff is well-worth it for many cases. Note that vector_set allows
15// you to use a deque or other random access container which may perform
16// better for you than vector.
17//
18// Note that with vector_set, vector_multiset, vector_map, vector_multimap
19// that the modification of the container potentially invalidates all
20// existing iterators into the container, unlike what happens with conventional
21// sets and maps.
22//
23// This type could conceptually use a eastl::array as its underlying container,
24// however the current design requires an allocator aware container.
25// Consider using a fixed_vector instead.

235 // NOTE(rparolin): It is undefined behaviour if user code fails to ensure the container
236 // invariants are respected by performing an explicit call to 'sort' before any other
237 // operations on the container are performed that do not clear the elements.
238 //
239 // 'push_back_unsorted' and 'emplace_back_unsorted' do not satisfy container invariants
240 // for being sorted. We provide these overloads explicitly labelled as '_unsorted' as an
241 // optimization opportunity when batch inserting elements so users can defer the cost of
242 // sorting the container once when all elements are contained. This was done to clarify
243 // the intent of code by leaving a trace that a manual call to sort is required.
244 //
245 template <typename... Args> decltype(auto) push_back_unsorted(Args&&... args)
246 { return base_type::push_back(eastl::forward<Args>(args)...); }
247 template <typename... Args> decltype(auto) emplace_back_unsorted(Args&&... args)
248 { return base_type::emplace_back(eastl::forward<Args>(args)...); }

439 vector_set<K, C, A, RAC>::insert(const value_type& value)

463 template <typename K, typename C, typename A, typename RAC>
464 inline typename vector_set<K, C, A, RAC>::iterator
465 vector_set<K, C, A, RAC>::insert(const_iterator position, const value_type& value)
466 {
467 // We assume that the user knows what he is doing and has supplied us with
468 // a position that is right where value should be inserted (put in front of).
469 // We do a test to see if the position is correct. If so then we insert,
470 // if not then we ignore the input position.
471
472 if((position == end()) || value_compare::operator()(value, *position)) // If the element at position is greater than value...
473 {
474 if((position == begin()) || value_compare::operator()(*(position - 1), value)) // If the element before position is less than value...
475 return base_type::insert(position, value);
476 }
477
478 // In this case we either have an incorrect position or value is already present.
479 // We fall back to the regular insert function. An optimization would be to detect
480 // that the element is already present, but that's only useful if the user supplied
481 // a good position but a present element.
482 const eastl::pair<iterator, bool> result = insert(value);
483
484 return result.first;
485 }

657 template <typename K, typename C, typename A, typename RAC>
658 inline eastl::pair<typename vector_set<K, C, A, RAC>::iterator, typename vector_set<K, C, A, RAC>::iterator>
659 vector_set<K, C, A, RAC>::equal_range(const key_type& k)
660 {
