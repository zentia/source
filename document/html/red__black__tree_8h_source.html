279 // rb_base
280 //
281 // This class allows us to use a generic rbtree as the basis of map, multimap,
282 // set, and multiset transparently. The vital template parameters for this are
283 // the ExtractKey and the bUniqueKeys parameters.
284 //
285 // If the rbtree has a value type of the form pair<T1, T2> (i.e. it is a map or
286 // multimap and not a set or multiset) and a key extraction policy that returns
287 // the first part of the pair, the rbtree gets a mapped_type typedef.
288 // If it satisfies those criteria and also has unique keys, then it also gets an
289 // operator[] (which only map and set have and multimap and multiset don't have).
290 //

1059 template <typename K, typename V, typename C, typename A, typename E, bool bM, bool bU>
1060 void rbtree<K, V, C, A, E, bM, bU>::swap(this_type& x)
1061 {
1062 #if EASTL_RBTREE_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1063 if(mAllocator == x.mAllocator) // If allocators are equivalent...
1064 #endif
1065 {
1066 // Most of our members can be exchaged by a basic swap:
1067 // We leave mAllocator as-is.
1068 eastl::swap(mnSize, x.mnSize);
1069 eastl::swap(get_compare(), x.get_compare());
1070 #if !EASTL_RBTREE_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1071 eastl::swap(mAllocator, x.mAllocator);
1072 #endif
1073
1074
1075 // However, because our anchor node is a part of our class instance and not
1076 // dynamically allocated, we can't do a swap of it but must do a more elaborate
1077 // procedure. This is the downside to having the mAnchor be like this, but
1078 // otherwise we consider it a good idea to avoid allocating memory for a
1079 // nominal container instance.
1080
1081 // We optimize for the expected most common case: both pointers being non-null.
1082 if(mAnchor.mpNodeParent && x.mAnchor.mpNodeParent) // If both pointers are non-null...
1083 {
1084 eastl::swap(mAnchor.mpNodeRight, x.mAnchor.mpNodeRight);
1085 eastl::swap(mAnchor.mpNodeLeft, x.mAnchor.mpNodeLeft);
1086 eastl::swap(mAnchor.mpNodeParent, x.mAnchor.mpNodeParent);
1087
1088 // We need to fix up the anchors to point to themselves (we can't just swap them).
1089 mAnchor.mpNodeParent->mpNodeParent = &mAnchor;
1090 x.mAnchor.mpNodeParent->mpNodeParent = &x.mAnchor;
1091 }
1092 else if(mAnchor.mpNodeParent)
1093 {
1094 x.mAnchor.mpNodeRight = mAnchor.mpNodeRight;
1095 x.mAnchor.mpNodeLeft = mAnchor.mpNodeLeft;
1096 x.mAnchor.mpNodeParent = mAnchor.mpNodeParent;
1097 x.mAnchor.mpNodeParent->mpNodeParent = &x.mAnchor;
1098
1099 // We need to fix up our anchor to point it itself (we can't have it swap with x).
1100 mAnchor.mpNodeRight = &mAnchor;
1101 mAnchor.mpNodeLeft = &mAnchor;
1102 mAnchor.mpNodeParent = NULL;
1103 }
1104 else if(x.mAnchor.mpNodeParent)
1105 {
1106 mAnchor.mpNodeRight = x.mAnchor.mpNodeRight;
1107 mAnchor.mpNodeLeft = x.mAnchor.mpNodeLeft;
1108 mAnchor.mpNodeParent = x.mAnchor.mpNodeParent;
1109 mAnchor.mpNodeParent->mpNodeParent = &mAnchor;
1110
1111 // We need to fix up x's anchor to point it itself (we can't have it swap with us).
1112 x.mAnchor.mpNodeRight = &x.mAnchor;
1113 x.mAnchor.mpNodeLeft = &x.mAnchor;
1114 x.mAnchor.mpNodeParent = NULL;
1115 } // Else both are NULL and there is nothing to do.
1116 }
1117 #if EASTL_RBTREE_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1118 else
1119 {
1120 const this_type temp(*this); // Can't call eastl::swap because that would
1121 *this = x; // itself call this member swap function.
1122 x = temp;
1123 }
1124 #endif
1125 }

1250 template <typename K, typename V, typename C, typename A, typename E, bool bM, bool bU>
1251 rbtree_node_base*
1252 rbtree<K, V, C, A, E, bM, bU>::DoGetKeyInsertionPositionUniqueKeys(bool& canInsert, const key_type& key)
1253 {
1254 // This code is essentially a slightly modified copy of the the rbtree::insert
1255 // function whereby this version takes a key and not a full value_type.
1256 extract_key extractKey;
1257
1258 rbtree_node_base* pCurrent = mAnchor.mpNodeParent; // Start with the root node.
1259 rbtree_node_base* pLowerBound = &mAnchor; // Set it to the container end for now.
1260 rbtree_node_base* pParent; // This will be where we insert the new node.
1261
1262 bool bValueLessThanNode = true; // If the tree is empty, this will result in an insertion at the front.
1263
1264 // Find insertion position of the value. This will either be a position which
1265 // already contains the value, a position which is greater than the value or
1266 // end(), which we treat like a position which is greater than the value.
1267 while(EASTL_LIKELY(pCurrent)) // Do a walk down the tree.
1268 {
1269 bValueLessThanNode = compare(key, extractKey(static_cast<node_type*>(pCurrent)->mValue));
1270 pLowerBound = pCurrent;
1271
1272 if(bValueLessThanNode)
1273 {
1274 EASTL_VALIDATE_COMPARE(!compare(extractKey(static_cast<node_type*>(pCurrent)->mValue), key)); // Validate that the compare function is sane.
1275 pCurrent = pCurrent->mpNodeLeft;
1276 }
1277 else
1278 pCurrent = pCurrent->mpNodeRight;
1279 }
1280
1281 pParent = pLowerBound; // pLowerBound is actually upper bound right now (i.e. it is > value instead of <=), but we will make it the lower bound below.
1282
1283 if(bValueLessThanNode) // If we ended up on the left side of the last parent node...
1284 {
1285 if(EASTL_LIKELY(pLowerBound != mAnchor.mpNodeLeft)) // If the tree was empty or if we otherwise need to insert at the very front of the tree...
1286 {
1287 // At this point, pLowerBound points to a node which is > than value.
1288 // Move it back by one, so that it points to a node which is <= value.
1289 pLowerBound = RBTreeDecrement(pLowerBound);
1290 }
1291 else
1292 {
1293 canInsert = true;
1294 return pLowerBound;
1295 }
1296 }
1297
1298 // Since here we require values to be unique, we will do nothing if the value already exists.
1299 node_type* const pLowerBoundFullNode = static_cast<node_type*>(pLowerBound);

2005 template <typename K, typename V, typename C, typename A, typename E, bool bM, bool bU>
2006 bool rbtree<K, V, C, A, E, bM, bU>::validate() const
2007 {
2008 // Red-black trees have the following canonical properties which we validate here:
2009 // 1 Every node is either red or black.
2010 // 2 Every leaf (NULL) is black by defintion. Any number of black nodes may appear in a sequence.
2011 // 3 If a node is red, then both its children are black. Thus, on any path from
2012 // the root to a leaf, red nodes must not be adjacent.
2013 // 4 Every simple path from a node to a descendant leaf contains the same number of black nodes.
2014 // 5 The mnSize member of the tree must equal the number of nodes in the tree.
2015 // 6 The tree is sorted as per a conventional binary tree.
2016 // 7 The comparison function is sane; it obeys strict weak ordering. If compare(a,b) is true, then compare(b,a) must be false. Both cannot be true.
2017
2018 extract_key extractKey;
2019
2020 if(mnSize)
2021 {
2022 // Verify basic integrity.
2023 //if(!mAnchor.mpNodeParent || (mAnchor.mpNodeLeft == mAnchor.mpNodeRight))
2024 // return false; // Fix this for case of empty tree.
2025
2026 if(mAnchor.mpNodeLeft != RBTreeGetMinChild(mAnchor.mpNodeParent))
2027 return false;
2028
2029 if(mAnchor.mpNodeRight != RBTreeGetMaxChild(mAnchor.mpNodeParent))
2030 return false;
2031
2032 const size_t nBlackCount = RBTreeGetBlackCount(mAnchor.mpNodeParent, mAnchor.mpNodeLeft);
2033 size_type nIteratedSize = 0;
2034
2035 for(const_iterator it = begin(); it != end(); ++it, ++nIteratedSize)
2036 {
2037 const node_type* const pNode = (const node_type*)it.mpNode;
2038 const node_type* const pNodeRight = (const node_type*)pNode->mpNodeRight;
2039 const node_type* const pNodeLeft = (const node_type*)pNode->mpNodeLeft;
2040
2041 // Verify #7 above.
2042 if(pNodeRight && compare(extractKey(pNodeRight->mValue), extractKey(pNode->mValue)) && compare(extractKey(pNode->mValue), extractKey(pNodeRight->mValue))) // Validate that the compare function is sane.
2043 return false;
2044
2045 // Verify #7 above.
2046 if(pNodeLeft && compare(extractKey(pNodeLeft->mValue), extractKey(pNode->mValue)) && compare(extractKey(pNode->mValue), extractKey(pNodeLeft->mValue))) // Validate that the compare function is sane.
2047 return false;
2048
2049 // Verify item #1 above.
2050 if((pNode->mColor != kRBTreeColorRed) && (pNode->mColor != kRBTreeColorBlack))
2051 return false;
2052
2053 // Verify item #3 above.
2054 if(pNode->mColor == kRBTreeColorRed)
2055 {
2056 if((pNodeRight && (pNodeRight->mColor == kRBTreeColorRed)) ||
2057 (pNodeLeft && (pNodeLeft->mColor == kRBTreeColorRed)))
2058 return false;
2059 }
2060
2061 // Verify item #6 above.
2062 if(pNodeRight && compare(extractKey(pNodeRight->mValue), extractKey(pNode->mValue)))
2063 return false;
2064
2065 if(pNodeLeft && compare(extractKey(pNode->mValue), extractKey(pNodeLeft->mValue)))
2066 return false;
2067
2068 if(!pNodeRight && !pNodeLeft) // If we are at a bottom node of the tree...
2069 {
2070 // Verify item #4 above.
2071 if(RBTreeGetBlackCount(mAnchor.mpNodeParent, pNode) != nBlackCount)
2072 return false;
2073 }
2074 }
2075
2076 // Verify item #5 above.
2077 if(nIteratedSize != mnSize)
2078 return false;
2079
2080 return true;
2081 }
2082 else
2083 {
2084 if((mAnchor.mpNodeLeft != &mAnchor) || (mAnchor.mpNodeRight != &mAnchor))
2085 return false;
2086 }
2087
2088 return true;
2089 }
