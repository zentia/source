38template<typename __Mvec>
39class ClassWithVectorType {
40 static const int n = 16;
41 static const int F = sizeof(__Mvec)/sizeof(float);
42 __Mvec field[n];
43 void init( int start );
44public:
45 ClassWithVectorType() {init(-n);}
46 ClassWithVectorType( int i ) {init(i);}
47 ClassWithVectorType( const ClassWithVectorType& src ) {
48 for( int i=0; i<n; ++i ) {
49 field[i] = src.field[i];
50 }
51 }
52 void operator=( const ClassWithVectorType& src ) {
53 __Mvec stack[n];
54 for( int i=0; i<n; ++i )
55 stack[i^5] = src.field[i];
56 for( int i=0; i<n; ++i )
57 field[i^5] = stack[i];
58 }
59 ~ClassWithVectorType() {init(-2*n);}
60 friend bool operator==( const ClassWithVectorType& x, const ClassWithVectorType& y ) {
61 for( int i=0; i<F*n; ++i )
62 if( ((const float*)x.field)[i]!=((const float*)y.field)[i] )
63 return false;
64 return true;
65 }
66 friend bool operator!=( const ClassWithVectorType& x, const ClassWithVectorType& y ) {
67 return !(x==y);
68 }
69};

71template<typename __Mvec>
72void ClassWithVectorType<__Mvec>::init( int start ) {
73 __Mvec stack[n];
74 for( int i=0; i<n; ++i ) {
75 // Declaring value as a one-element array instead of a scalar quites
76 // gratuitous warnings about possible use of "value" before it was set.
77 __Mvec value[1];
78 for( int j=0; j<F; ++j )
79 ((float*)value)[j] = float(n*start+F*i+j);
80 stack[i^5] = value[0];
81 }
82 for( int i=0; i<n; ++i )
83 field[i^5] = stack[i];
84}
