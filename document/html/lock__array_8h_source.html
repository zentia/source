214 template <typename Q>
215 size_t lock( Q const& hint )
216 {
217 size_t nCell = m_SelectCellPolicy( hint, size());
218 assert( nCell < size());
219 m_arrLocks[nCell].lock();

221 }

230 template <typename Q>
231 size_t try_lock( Q const& hint )
232 {
233 size_t nCell = m_SelectCellPolicy( hint, size());
234 assert( nCell < size());
235 if ( m_arrLocks[nCell].try_lock())
236 return nCell;
237 return c_nUnspecifiedCell;
238 }

241 void unlock( size_t nCell )
242 {
243 assert( nCell < size());
244 m_arrLocks[nCell].unlock();
245 }

248 void lock_all()
249 {
250 lock_type * pLock = m_arrLocks;
251 for ( lock_type * pEnd = m_arrLocks + size(); pLock != pEnd; ++pLock )
252 pLock->lock();
253 }
254
256 void unlock_all()

258 lock_type * pLock = m_arrLocks;
259 for ( lock_type * pEnd = m_arrLocks + size(); pLock != pEnd; ++pLock )
260 pLock->unlock();
