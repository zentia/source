6// This file implements a vector (array-like container), much like the C++
7// std::vector class.
8// The primary distinctions between this vector and std::vector are:
9// - vector has a couple extension functions that increase performance.
10// - vector can contain objects with alignment requirements. std::vector
11// cannot do so without a bit of tedious non-portable effort.
12// - vector supports debug memory naming natively.
13// - vector is easier to read, debug, and visualize.
14// - vector is savvy to an environment that doesn't have exception handling,
15// as is sometimes the case with console or embedded environments.
16// - vector has less deeply nested function calls and allows the user to
17// enable forced inlining in debug builds in order to reduce bloat.
18// - vector<bool> is a vector of boolean values and not a bit vector.
19// - vector guarantees that memory is contiguous and that vector::iterator
20// is nothing more than a pointer to T.
21// - vector has an explicit data() method for obtaining a pointer to storage
22// which is safe to call even if the block is empty. This avoids the
23// common &v[0], &v.front(), and &*v.begin() constructs that trigger false
24// asserts in STL debugging modes.
25// - vector data is guaranteed to be contiguous.
26// - vector has a set_capacity() function which frees excess capacity.
27// The only way to do this with std::vector is via the cryptic non-obvious
28// trick of using: vector<SomeClass>(x).swap(x);

102 ///
103 /// The reason we have a VectorBase class is that it makes exception handling
104 /// simpler to implement because memory allocation is implemented entirely
105 /// in this class. If a user creates a vector which needs to allocate
106 /// memory in the constructor, VectorBase handles it. If an exception is thrown
107 /// by the allocator then the exception throw jumps back to the user code and
108 /// no try/catch code need be written in the vector or VectorBase constructor.
109 /// If an exception is thrown in the vector (not VectorBase) constructor, the
110 /// destructor for VectorBase will be called automatically (and free the allocated
111 /// memory) before the execution jumps back to the user code.
112 /// However, if the vector class were to handle both allocation and initialization
113 /// then it would have no choice but to implement an explicit try/catch statement
114 /// for all pathways that allocate memory. This increases code size and decreases
115 /// performance and makes the code a little harder read and maintain.
116 ///
117 /// The C++ standard (15.2 paragraph 2) states:
118 /// "An object that is partially constructed or partially destroyed will
119 /// have destructors executed for all its fully constructed subobjects,
120 /// that is, for subobjects for which the constructor has been completed
121 /// execution and the destructor has not yet begun execution."
122 ///
123 /// The C++ standard (15.3 paragraph 11) states:
124 /// "The fully constructed base classes and members of an object shall
125 /// be destroyed before entering the handler of a function-try-block
126 /// of a constructor or destructor for that block."
127 ///

490 template <typename T, typename Allocator>
491 inline T* VectorBase<T, Allocator>::DoAllocate(size_type n)
492 {
493#if EASTL_ASSERT_ENABLED
494 if (EASTL_UNLIKELY(n >= 0x80000000))
495 EASTL_FAIL_MSG("vector::DoAllocate -- improbably large request.");
496#endif
497
498 // If n is zero, then we allocate no memory and just return nullptr.
499 // This is fine, as our default ctor initializes with NULL pointers.
500 if (EASTL_LIKELY(n))
501 {
502 auto* p = (T*)allocate_memory(internalAllocator(), n * sizeof(T), EASTL_ALIGN_OF(T), 0);
503 EASTL_ASSERT_MSG(p != nullptr, "the behaviour of eastl::allocators that return nullptr is not defined.");
504 return p;
505 }
506 else
507 {
508 return nullptr;
509 }
510 }

627 template <typename T, typename Allocator>
628 typename vector<T, Allocator>::this_type& vector<T, Allocator>::operator=(const this_type& x)
629 {
630 if (this != &x) // If not assigning to self...
631 {
632 // If (EASTL_ALLOCATOR_COPY_ENABLED == 1) and the current contents are allocated by an
633 // allocator that's unequal to x's allocator, we need to reallocate our elements with
634 // our current allocator and reallocate it with x's allocator. If the allocators are
635 // equal then we can use a more optimal algorithm that doesn't reallocate our elements
636 // but instead can copy them in place.
637
638#if EASTL_ALLOCATOR_COPY_ENABLED
639 bool bSlowerPathwayRequired = (internalAllocator() != x.internalAllocator());
640#else
641 bool bSlowerPathwayRequired = false;
642#endif
643
644 if (bSlowerPathwayRequired)
645 {
646 DoClearCapacity(); // Must clear the capacity instead of clear because set_capacity frees our memory,
647 // unlike clear.
648
649#if EASTL_ALLOCATOR_COPY_ENABLED
650 internalAllocator() = x.internalAllocator();
651#endif
652 }
653
654 DoAssign<const_iterator, false>(x.begin(), x.end(), eastl::false_type());
655 }
656
657 return *this;
658 }

860 template <typename T, typename Allocator>
861 void vector<T, Allocator>::set_capacity(size_type n)
862 {
863 if ((n == npos) || (n <= (size_type)(mpEnd - mpBegin))) // If new capacity <= size...
864 {
865 if (n == 0) // Very often n will be 0, and clear will be faster than resize and use less stack space.
866 clear();
867 else if (n < (size_type)(mpEnd - mpBegin))
868 resize(n);
869
870 shrink_to_fit();
871 }
872 else // Else new capacity > size.
873 {
874 pointer const pNewData = DoRealloc(n, mpBegin, mpEnd, should_move_tag());
875 eastl::destruct(mpBegin, mpEnd);
876 DoFree(mpBegin, (size_type)(internalCapacityPtr() - mpBegin));
877
878 const ptrdiff_t nPrevSize = mpEnd - mpBegin;
879 mpBegin = pNewData;
880 mpEnd = pNewData + nPrevSize;
881 internalCapacityPtr() = mpBegin + n;
882 }

885 template <typename T, typename Allocator>
886 inline void vector<T, Allocator>::shrink_to_fit()
887 {
888 // This is the simplest way to accomplish this, and it is as efficient as any other.
889 this_type temp =
890 this_type(move_iterator<iterator>(begin()), move_iterator<iterator>(end()), internalAllocator());
891
892 // Call DoSwap() rather than swap() as we know our allocators match and we don't want to invoke the code path
893 // handling non matching allocators as it imposes additional restrictions on the type of T to be copyable
894 DoSwap(temp);
895 }

1082 template <typename T, typename Allocator>
1083 inline void* vector<T, Allocator>::push_back_uninitialized(size_t n)
1084 {
1085 if (n > size_type(internalCapacityPtr() - mpEnd))
1086 {
1087 const size_type nPrevSize = size_type(mpEnd - mpBegin);
1088 const size_type nGrowSize = GetNewCapacity(nPrevSize);
1089 const size_type nNewSize = eastl::max(nGrowSize, nPrevSize + n);
1090 pointer const pNewData = DoAllocate(nNewSize);
1091
1092#if EASTL_EXCEPTIONS_ENABLED
1093 pointer pNewEnd = pNewData; // Assign pNewEnd a value here in case the copy throws.
1094 try
1095 {
1096 pNewEnd = eastl::uninitialized_move_ptr_if_noexcept(mpBegin, mpEnd, pNewData);
1097 }
1098 catch (...)
1099 {
1100 eastl::destruct(pNewData, pNewEnd);
1101 DoFree(pNewData, nNewSize);
1102 throw;
1103 }
1104#else
1105 pointer pNewEnd = eastl::uninitialized_move_ptr_if_noexcept(mpBegin, mpEnd, pNewData);
1106#endif
1107
1108 eastl::destruct(mpBegin, mpEnd);
1109 DoFree(mpBegin, (size_type)(internalCapacityPtr() - mpBegin));
1110
1111 mpBegin = pNewData;
1112 mpEnd = pNewEnd;
1113 internalCapacityPtr() = pNewData + nNewSize;
1114 }
1115 auto ptr = mpEnd;
1116 mpEnd += n;
1117 return ptr;
1118 }

1398 inline void vector<T, Allocator>::reset_lose_memory() EA_NOEXCEPT
1399 {

1408 // swap exchanges the contents of two containers. With respect to the containers allocators,
1409 // the C11++ Standard (23.2.1/7) states that the behavior of a call to a container's swap function
1410 // is undefined unless the objects being swapped have allocators that compare equal or
1411 // allocator_traits<allocator_type>::propagate_on_container_swap::value is true (propagate_on_container_swap
1412 // is false by default). EASTL doesn't have allocator_traits and so this doesn't directly apply,
1413 // but EASTL has the effective behavior of propagate_on_container_swap = false for all allocators.
1414 template <typename T, typename Allocator>
1415 inline void vector<T, Allocator>::swap(this_type& x) EA_NOEXCEPT
1416 {
1417#if defined(EASTL_VECTOR_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR) && \
1418 EASTL_VECTOR_LEGACY_SWAP_BEHAVIOUR_REQUIRES_COPY_CTOR
1419 if (internalAllocator() == x.internalAllocator()) // If allocators are equivalent...
1420 DoSwap(x);
1421 else // else swap the contents.
1422 {
1423 const this_type temp(*this); // Can't call eastl::swap because that would
1424 *this = x; // itself call this member swap function.
1425 x = temp;
1426 }
1427#else
1428 // NOTE(rparolin): The previous implementation required T to be copy-constructible in the fall-back case where
1429 // allocators with unique instances copied elements. This was an unnecessary restriction and prevented the
1430 // common usage of vector with non-copyable types (eg. eastl::vector<non_copyable> or
1431 // eastl::vector<unique_ptr>).
1432 //
1433 // The previous implementation violated the following requirements of vector::swap so the fall-back code has
1434 // been removed. EASTL implicitly defines 'propagate_on_container_swap = false' therefore the fall-back case is
1435 // undefined behaviour. We simply swap the contents and the allocator as that is the common expectation of
1436 // users and does not put the container into an invalid state since it can not free its memory via its current
1437 // allocator instance.
1438 //
1439 // http://en.cppreference.com/w/cpp/container/vector/swap
1440 // "Exchanges the contents of the container with those of other. Does not invoke any move, copy, or swap
1441 // operations on individual elements."
1442 //
1443 // http://en.cppreference.com/w/cpp/concept/AllocatorAwareContainer
1444 // "Swapping two containers with unequal allocators if propagate_on_container_swap is false is undefined
1445 // behavior."
1446
1447 DoSwap(x);
1448#endif
1449 }

1679 const size_type n =
1680 (size_type)eastl::distance(first, last); // n is the number of elements we are inserting.

1719 }

1725 }
1726#else
1727 pointer pNewEnd = eastl::uninitialized_move_ptr_if_noexcept(mpBegin, destPosition, pNewData);
