34 template <typename Q>
35 node_type( unsigned int nHeight, atomic_marked_ptr * pTower, Q&& key )

41 template <typename Q, typename... Args>

43 : m_Value( std::forward<Q>(key), mapped_type( std::forward<Args>(args)... ))

50 private:

53 if ( nHeight > 1 ) {

55 base_class::make_tower( nHeight, pTower );
