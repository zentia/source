242private:
243 struct Node {

248 Node * volatile head;
249 volatile unsigned long pwrite;
250 long padding1[longxCacheLine-((sizeof(Node *)+sizeof(unsigned long))/sizeof(long))];
251 Node * volatile tail;
252 volatile unsigned long pread;
253 long padding2[longxCacheLine-((sizeof(Node*)+sizeof(unsigned long))/sizeof(long))];
254
255 const size_t cachesize;
256 void ** cache;
257
258private:
259
260 inline bool cachepush(void * const data) {
261
262 if (!cache[pwrite]) {
263 /* Write Memory Barrier: ensure all previous memory write
264 * are visible to the other processors before any later
265 * writes are executed. This is an "expensive" memory fence
266 * operation needed in all the architectures with a weak-ordering
267 * memory model where stores can be executed out-or-order
268 * (e.g. Powerpc). This is a no-op on Intel x86/x86-64 CPUs.
269 */
270 WMB();
271 cache[pwrite] = data;
272 pwrite += (pwrite+1 >= cachesize) ? (1-cachesize): 1;
273 return true;
274 }
275 return false;
276 }
277
278 inline bool cachepop(void ** data) {
279 if (!cache[pread]) return false;
280
281 *data = cache[pread];
282 cache[pread]=NULL;
283 pread += (pread+1 >= cachesize) ? (1-cachesize): 1;
284 return true;
285 }
286
287public:
288 enum {DEFAULT_CACHE_SIZE=1024};
289
290 dynqueue(int cachesize=DEFAULT_CACHE_SIZE, bool fillcache=false):cachesize(cachesize) {
291 Node * n = (Node *)::malloc(sizeof(Node));
292 n->data = NULL; n->next = NULL;
293 head=n;
294 tail=n;
295
296 cache=(void**)getAlignedMemory(longxCacheLine*sizeof(long),cachesize*sizeof(void*));
297 if (!cache) {
298 error("FATAL ERROR: dynqueue no memory available!\n");
299 abort();
300 }
301
302 if (fillcache) {
303 for(int i=0;i<cachesize;++i) {
304 n = (Node *)::malloc(sizeof(Node));
305 if (n) cachepush(n);
306 }
307 }
308 }
309
310 ~dynqueue() {
311 union { Node * n; void * n2; } p;
312 while(cachepop(&p.n2)) free(p.n);
313 while(head != tail) {
314 p.n = (Node*)head;
315 head = head->next;
316 free(p.n);
317 }
318 if (head) free((void*)head);
319 if (cache) freeAlignedMemory(cache);
320 }
321
322 inline bool push(void * const data) {
323 assert(data != NULL);
324
325 union { Node * n; void * n2; } p;
326 if (!cachepop(&p.n2))
327 p.n = (Node *)::malloc(sizeof(Node));
328
329 p.n->data = data; p.n->next = NULL;
330 WMB();
331 tail->next = p.n;
332 tail = p.n;
333
334 return true;
335 }
336
337 inline bool pop(void ** data) {
338 assert(data != NULL);
339 if (head->next) {
340 Node * n = (Node *)head;
341 *data = (head->next)->data;
342 head = head->next;
343
344 if (!cachepush(n)) free(n);
345 return true;
346 }
347 return false;
348 }
