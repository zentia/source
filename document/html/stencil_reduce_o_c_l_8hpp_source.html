364 void setSizingHeuristics() {
365 cl_int status;
366 //get device-dependent max wg size
367 size_t max_device_wgsize;
368 status = clGetDeviceInfo(deviceId,CL_DEVICE_MAX_WORK_GROUP_SIZE,sizeof(max_device_wgsize),&max_device_wgsize, NULL);
369 checkResult(status, "clGetDeviceInfo (map)");
370 if(kernel_map) { //map kernel
371 //get kernel-dependent max wg size
372 size_t max_kernel_wgsize;
373 status = clGetKernelWorkGroupInfo(kernel_map, deviceId, CL_KERNEL_WORK_GROUP_SIZE, sizeof(size_t), &max_kernel_wgsize, 0);
374 checkResult(status, "GetKernelWorkGroupInfo (map)");
375 wgsize_map_max = std::min<size_t>(max_device_wgsize,max_kernel_wgsize);
376 //get size of the atomic scheduling unit (analogous to CUDA wrap size)
377 //typical values are 16 or 32
378 size_t wg_multiple;
379 status = clGetKernelWorkGroupInfo(kernel_map, deviceId, CL_KERNEL_PREFERRED_WORK_GROUP_SIZE_MULTIPLE, sizeof(size_t), &wg_multiple, 0);
380 wgsize_map_static = std::max<size_t>(64, wg_multiple * 4); //64 or 128
381 wgsize_map_static = std::min<size_t>(wgsize_map_static,wgsize_map_max);
382 }
383 if(kernel_reduce) { //reduce kernel
384 //get kernel-dependent max wg size
385 size_t max_kernel_wgsize;
386 status = clGetKernelWorkGroupInfo(kernel_reduce, deviceId, CL_KERNEL_WORK_GROUP_SIZE, sizeof(size_t), &max_kernel_wgsize, 0);
387 checkResult(status, "GetKernelWorkGroupInfo (reduce)");
388 wgsize_reduce_max = std::min<size_t>(max_device_wgsize,max_kernel_wgsize);
389 //get size of the atomic scheduling unit (analogous to CUDA wrap size)
390 //typical values are 16 or 32
391 size_t wg_multiple;
392 status = clGetKernelWorkGroupInfo(kernel_reduce, deviceId, CL_KERNEL_PREFERRED_WORK_GROUP_SIZE_MULTIPLE, sizeof(size_t), &wg_multiple, 0);
393 wgsize_reduce_static = std::max<size_t>(64, wg_multiple * 4); //64 or 128
394 wgsize_reduce_static = std::min<size_t>(wgsize_reduce_static,wgsize_reduce_max);
395 }
396#ifdef FF_OPENCL_LOG
397 std::cerr << "[virtual accelerator @"<<this<<"]\n";
398 std::cerr << "+ static heuristics for kernel sizing parameters:\n";
399 std::cerr << "- MAP workgroup-size = " <<wgsize_map_static<< "\n";
400 std::cerr << "- MAP max workgroup-size = " <<wgsize_map_max<< "\n";
401 std::cerr << "- RED workgroup-size = " <<wgsize_reduce_static<< " \n";
402 std::cerr << "- RED max workgroup-size = " <<wgsize_reduce_max<< " \n";
403#endif
404 }

419 void relocateInputBuffer(const Tin *inPtr, const bool reuseIn, const Tout *reducePtr) {
420 cl_int status;
421
422 if (reuseIn) {
423 inputBuffer = allocator->createBufferUnique(inPtr, context,
424 CL_MEM_READ_WRITE, sizeInput_padded, &status);
425 checkResult(status, "CreateBuffer(Unique) input");
426 } else {
427 if (inputBuffer) allocator->releaseBuffer(inPtr, context, inputBuffer);
428 //allocate input-size + pre/post-windows
429 inputBuffer = allocator->createBuffer(inPtr, context,
430 CL_MEM_READ_WRITE, sizeInput_padded, &status);
431 checkResult(status, "CreateBuffer input");
432 }
433
434 //set workgroup size and nthreads for map
435 // MA patch - map not defined => workgroup_size_map
436 //size_t wgsize = wgsize_map_static==0?wgsize_reduce_static:wgsize_map_static;
437 if(kernel_map) {
438
439 // if the output buffer is bigger than input buffer then we use the size of the
440 // output buffer to compute the number of threads
441 if (lenInput < lenOutput) {
442 wgsize_map = std::min<size_t>(lenOutput, wgsize_map_static);
443 nthreads_map = wgsize_map * ((lenOutput + wgsize_map - 1) / wgsize_map); //round up
444 }
445 else {
446 wgsize_map = std::min<size_t>(lenInput, wgsize_map_static);
447 nthreads_map = wgsize_map * ((lenInput + wgsize_map - 1) / wgsize_map); //round up
448 }
449#ifdef FF_OPENCL_LOG
450 std::cerr << "[virtual accelerator @"<<this<<"]\n";
451 std::cerr << "+ computed MAP kernel sizing parameters:\n";
452 std::cerr << "- MAP workgroup-size = " << wgsize_map << "\n";
453 std::cerr << "- MAP n. threads = " << nthreads_map << " \n";
454#endif
455 }
456 //set workgroup size and nthreads for reduce
457 if (kernel_reduce && reduce_mode == REDUCE_INPUT) {
458 resetReduce(lenInput, sizeof(Tin), (void *)reducePtr);
459 }
460 }

904 inline void getBlocksAndThreads(const size_t size, const size_t maxBlocks,
905 const size_t maxThreads, size_t & blocks, size_t &threads) {
906 const size_t half = (size + 1) / 2;
907 threads =
908 (size < maxThreads * 2) ?
909 (isPowerOf2(half) ? nextPowerOf2(half + 1) : nextPowerOf2(half)) :
910 maxThreads;
911 blocks = (size + (threads * 2 - 1)) / (threads * 2);
912 blocks = std::min(maxBlocks, blocks);
913 }
