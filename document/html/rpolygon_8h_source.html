133inline bool PolygonsOverlap(RPolygon *p1, RPolygon *p2, int &xl, int &yl, int &xh, int &yh) {
134 int xl1, yl1, xh1, yh1, xl2, yl2, xh2, yh2;
135#if _DEBUG
136 rt_sleep(1); // slow down the process so we can see it.
137#endif
138 p1->get(&xl1, &yl1, &xh1, &yh1);
139 p2->get(&xl2, &yl2, &xh2, &yh2);
140 if(xl1 > xh2) return false;
141 if(xh1 < xl2) return false;
142 if(yl1 > yh2) return false;
143 if(yh1 < yl2) return false;
144 xl = (xl1 < xl2) ? xl2 : xl1;
145 xh = (xh1 < xh2) ? xh1 : xh2;
146 yl = (yl1 < yl2) ? yl2 : yl1;
147 yh = (yh1 < yh2) ? yh1 : yh2;
148 return true;
149}
