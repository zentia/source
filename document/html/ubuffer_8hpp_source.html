88class BufferPool {
89public:
90 BufferPool(int cachesize, const bool fillcache=false, unsigned long size=-1)
91 :inuse(cachesize),bufcache(cachesize) {
92 bufcache.init(); // initialise the internal buffer and allocates memory
93
94 if (fillcache) {
95 assert(size>0);
96 union { INTERNAL_BUFFER_T * buf; void * buf2;} p;
97 for(int i=0;i<cachesize;++i) {
98 p.buf = (INTERNAL_BUFFER_T*)malloc(sizeof(INTERNAL_BUFFER_T));
99 new (p.buf) INTERNAL_BUFFER_T(size);
100#if defined(uSWSR_MULTIPUSH)
101 p.buf->init(true);
102#else
103 p.buf->init();
104#endif
105 bufcache.push(p.buf);
106 }
107 }
108
109#if defined(UBUFFER_STATS)
110 miss=0;hit=0;
111 (void)padding1;
112#endif
113 }
114
115 ~BufferPool() {
116 union { INTERNAL_BUFFER_T * b1; void * b2;} p;
117
118 while(inuse.pop(&p.b2)) {
119 p.b1->~INTERNAL_BUFFER_T();
120 free(p.b2);
121 }
122 while(bufcache.pop(&p.b2)) {
123 p.b1->~INTERNAL_BUFFER_T();
124 free(p.b2);
125 }
126 }
127
128 inline INTERNAL_BUFFER_T * next_w(unsigned long size) {
129 union { INTERNAL_BUFFER_T * buf; void * buf2;} p;
130 if (!bufcache.pop(&p.buf2)) {
131#if defined(UBUFFER_STATS)
132 ++miss;
133#endif
134 p.buf = (INTERNAL_BUFFER_T*)malloc(sizeof(INTERNAL_BUFFER_T));
135 new (p.buf) INTERNAL_BUFFER_T(size);
136#if defined(uSWSR_MULTIPUSH)
137 if (!p.buf->init(true)) return NULL;
138#else
139 if (!p.buf->init()) return NULL;
140#endif
141 }
142#if defined(UBUFFER_STATS)
143 else ++hit;
144#endif
145 inuse.push(p.buf);
146 return p.buf;
147 }
148
149
150 inline INTERNAL_BUFFER_T * next_r() {
151 union { INTERNAL_BUFFER_T * buf; void * buf2;} p;
152 return (inuse.pop(&p.buf2)? p.buf : NULL);
153 }
154
155
156 inline void release(INTERNAL_BUFFER_T * const buf) {
157 buf->reset();
158 if (!bufcache.push(buf)) {
159 buf->~INTERNAL_BUFFER_T();
160 free(buf);
161 }
162 }
163
164#if defined(UBUFFER_STATS)
165
166 inline unsigned long readPoolMiss() {
167 unsigned long m = miss;
168 miss = 0;
169 return m;
170 }
171
172
173 inline unsigned long readPoolHit() {
174 unsigned long h = hit;
175 hit = 0;
176 return h;
177 }
178#endif
179
180 // just empties the inuse bucket putting data in the cache
181 void reset() {
182 union { INTERNAL_BUFFER_T * b1; void * b2;} p;
183 while(inuse.pop(&p.b2)) release(p.b1);
184 }
185
186 //
187 // WARNING: using this call in the wrong place is very dangerous!!!!
188 //
189 void changesize(size_t newsz) {
190 // the inuse queue should be empty
191 union { INTERNAL_BUFFER_T * b1; void * b2;} p;
192 while (inuse.pop(&p.b2))
193 assert(1==0);
194 dynqueue tmp;
195 while(bufcache.pop(&p.b2)) {
196 p.b1->changesize(newsz);
197 tmp.push(p.b2);
198 }
199 while(tmp.pop(&p.b2)) {
200 bufcache.push(p.b2);
201 }
202 }
203
204
205private:
206#if defined(UBUFFER_STATS)
207 unsigned long miss,hit;
208 long padding1[longxCacheLine-2];
209#endif
210
211 dynqueue inuse; // of type dynqueue, that is a Dynamic (list-based)
212 // SWSR unbounded queue.
213 // No lock is needed around pop and push methods.
214 INTERNAL_BUFFER_T bufcache; // This is a bounded buffer
215};

240 // Multipush: push a bach of items.
241 inline bool multipush() {
242 if (buf_w->multipush(multipush_buf,MULTIPUSH_BUFFER_SIZE)) {
243 mcnt=0;
244 return true;
245 }
246
247 if (fixedsize) return false;
248 // try to get a new buffer
249 INTERNAL_BUFFER_T * t = pool.next_w(size);
250 assert(t); // if (!t) return false; // EWOULDBLOCK
251 buf_w = t;
252 in_use_buffers++;
253 buf_w->multipush(multipush_buf,MULTIPUSH_BUFFER_SIZE);
254 mcnt=0;
255#if defined(UBUFFER_STATS)
256 ++numBuffers
257 //atomic_long_inc(&numBuffers);
258#endif
259 return true;
260 }
261#endif

268 uSWSR_Ptr_Buffer(unsigned long n,
269 const bool fixedsize=false,
270 const bool fillcache=false):
271 buf_r(0),buf_w(0),in_use_buffers(1),size(n),fixedsize(fixedsize),
272 pool(CACHE_SIZE,fillcache,size) {
273 init_unlocked(P_lock); init_unlocked(C_lock);
274 pushPMF=&uSWSR_Ptr_Buffer::push;
275 popPMF =&uSWSR_Ptr_Buffer::pop;
276#if defined(UBUFFER_STATS)
277 numBuffers = 0;

279#endif

297 bool init() {
298 if (buf_w || buf_r) return false;
299#if defined(uSWSR_MULTIPUSH)
300 if (size<=MULTIPUSH_BUFFER_SIZE) return false;
301#endif
302 buf_r = (INTERNAL_BUFFER_T*)::malloc(sizeof(INTERNAL_BUFFER_T));
303 assert(buf_r);
304 new ((void *)buf_r) INTERNAL_BUFFER_T(size);
305#if defined(uSWSR_MULTIPUSH)
306 if (!buf_r->init(true)) return false;
307#else
308 if (!buf_r->init()) return false;
309#endif
310 buf_w = buf_r;
311
312 return true;
313 }

344 inline bool push(void * const data) {
345 /* NULL values cannot be pushed in the queue */
346 assert(data != NULL);
347
348 // If fixedsize has been set to \p true, this method may
349 // return false. This means EWOULDBLOCK
350 if (!available()) {
351
352 if (fixedsize) return false;
353
354 // try to get a new buffer
355 INTERNAL_BUFFER_T * t = pool.next_w(size);
356 assert(t); //if (!t) return false; // EWOULDBLOCK
357 buf_w = t;
358 in_use_buffers++;
359#if defined(UBUFFER_STATS)
360 ++numBuffers;
361 //atomic_long_inc(&numBuffers);
362#endif
363 }
364 //DBG(assert(buf_w->push(data)); return true;);
365 buf_w->push(data);
366 return true;
367 }

372 spin_unlock(P_lock);

376#if defined(uSWSR_MULTIPUSH)
387 inline bool mpush(void * const data) {
388 assert(data != NULL);
389
390 if (mcnt==MULTIPUSH_BUFFER_SIZE)
391 return multipush();
392
393 multipush_buf[mcnt++]=data;
394
395 if (mcnt==MULTIPUSH_BUFFER_SIZE) return multipush();

397 return true;

400 inline bool flush() {
401 return (mcnt ? multipush() : true);

403#endif /* uSWSR_MULTIPUSH */

411 inline bool pop(void ** data) {
412 assert(data != NULL);
413
414 if (buf_r->empty()) { // current buffer is empty
415 if (buf_r == buf_w) return false;
416 if (buf_r->empty()) { // we have to check again
417 INTERNAL_BUFFER_T * tmp = pool.next_r();
418 if (tmp) {
419 // there is another buffer, release the current one
420 pool.release(buf_r);
421 in_use_buffers--;
422 buf_r = tmp;
423
424#if defined(UBUFFER_STATS)
425 --numBuffers;
426 //atomic_long_dec(&numBuffers);
427#endif
428 }
429 }
430 }
431 //DBG(assert(buf_r->pop(data)); return true;);
432 return buf_r->pop(data);
433 }
