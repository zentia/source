42 if constexpr (schemaful::IsSchemafulReader<R>) {

99 template <int... _is>
100 static ResultType find_matching_alternative(
101 const R& _r, const std::string& _disc_value, const InputVarType& _var,
102 std::integer_sequence<int, _is...>) noexcept {
103 using PossibleTags =
104 possible_tags_t<TaggedUnion<_discriminator, AlternativeTypes...>>;
105 static_assert(!PossibleTags::has_duplicates(),
106 "Duplicate tags are not allowed inside tagged unions.");
107 ResultType res = error("");
108 bool match_found = false;
109 (set_if_disc_value_matches<_is>(_r, _disc_value, _var, &res, &match_found),
110 ...);
111 if (match_found) [[likely]] {
112 return res;
113 } else {
114 const auto names = PossibleTags::names();
115 std::stringstream stream;
116 stream << "Could not parse tagged union, could not match "
117 << _discriminator.str() << " '" << _disc_value
118 << "'. The following tags are allowed: "
119 << internal::strings::join(", ", names);
120 return error(stream.str());
121 }
122 }

124 template <int _i>
125 static void set_if_disc_value_matches(const R& _r,
126 const std::string& _disc_value,
127 const InputVarType& _var,
128 ResultType* _res,
129 bool* _match_found) noexcept {
130 using AlternativeType = std::remove_cvref_t<
131 std::variant_alternative_t<_i, std::variant<AlternativeTypes...>>>;
132
133 if (!*_match_found && contains_disc_value<AlternativeType>(_disc_value)) {
134 const auto get_fields = [](auto&& _val) -> AlternativeType {
135 if constexpr (is_tagged_union_wrapper_v<decltype(_val)>) {
136 return std::move(_val.fields());
137 } else {
138 return std::move(_val);
139 }
140 };
141
142 const auto to_tagged_union = [](auto&& _val) {
143 return TaggedUnion<_discriminator, AlternativeTypes...>(
144 std::move(_val));
145 };
146
147 const auto embellish_error = [&](auto&& _e) {
148 std::stringstream stream;
149 stream << "Could not parse tagged union with "
150 "discrimininator "
151 << _discriminator.str() << " '" << _disc_value
152 << "': " << _e.what();
153 return Error(stream.str());
154 };
155
156 if constexpr (no_field_names_) {
157 using T = tagged_union_wrapper_no_ptr_t<std::invoke_result_t<
158 decltype(wrap_if_necessary<AlternativeType>), AlternativeType>>;
159 *_res = Parser<R, W, T, ProcessorsType>::read(_r, _var)
160 .transform(get_fields)
161 .transform(to_tagged_union)
162 .transform_error(embellish_error);
163 } else {
164 *_res = Parser<R, W, AlternativeType, ProcessorsType>::read(_r, _var)
165 .transform(to_tagged_union)
166 .transform_error(embellish_error);
167 }
168
169 *_match_found = true;
170 }
171 }

174 static Result<std::string> get_discriminator(
175 const R& _r, const InputObjectOrArrayType& _obj_or_arr) noexcept {
176 const auto to_type = [&_r](auto _var) {
177 return _r.template to_basic_type<std::string>(_var);
178 };
179
180 const auto embellish_error = [](const auto&) {
181 std::stringstream stream;
182 stream << "Could not parse tagged union: Could not find field '"
183 << _discriminator.str() << "' or type of field was not a string.";
184 return Error(stream.str());
185 };
186
187 if constexpr (no_field_names_) {
188 return _r.get_field_from_array(0, _obj_or_arr)
189 .and_then(to_type)
190 .transform_error(embellish_error);
191 } else {
192 return _r.get_field_from_object(_discriminator.str(), _obj_or_arr)
193 .and_then(to_type)
194 .transform_error(embellish_error);
195 }

217 template <class T>
218 static auto wrap_if_necessary(const T& _val) noexcept {
219 if constexpr (named_tuple_t<T>::Names::template contains<
220 _discriminator>()) {
221 return _val;
222 } else {
223 const auto tag = internal::make_tag<_discriminator, T>(_val);
224 using TagType = std::remove_cvref_t<decltype(tag)>;
225 if constexpr (internal::has_fields<std::remove_cvref_t<T>>()) {
226 using WrapperType =
227 TaggedUnionWrapperWithFields<T, TagType, _discriminator>;
228 return WrapperType{.tag = tag, .fields = &_val};
229 } else {
230 using WrapperType =
231 TaggedUnionWrapperNoFields<T, TagType, _discriminator>;
232 return WrapperType{.tag = tag, .fields = &_val};
233 }
