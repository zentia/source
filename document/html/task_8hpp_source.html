588// Function: work
589template <typename C>
590Task& Task::work(C&& c) {
591
592 if constexpr(is_static_task_v<C>) {
593 _node->_handle.emplace<Node::Static>(std::forward<C>(c));
594 }
595 else if constexpr(is_runtime_task_v<C>) {
596 _node->_handle.emplace<Node::Runtime>(std::forward<C>(c));
597 }
598 else if constexpr(is_subflow_task_v<C>) {
599 _node->_handle.emplace<Node::Subflow>(std::forward<C>(c));
600 }
601 else if constexpr(is_condition_task_v<C>) {
602 _node->_handle.emplace<Node::Condition>(std::forward<C>(c));
603 }
604 else if constexpr(is_multi_condition_task_v<C>) {
605 _node->_handle.emplace<Node::MultiCondition>(std::forward<C>(c));
606 }
607 else {
608 static_assert(dependent_false_v<C>, "invalid task callable");
609 }
610 return *this;
611}
