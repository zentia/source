20 template <class GC, typename Key, typename T, typename Traits = cc::skip_list::traits >
21 class SkipListMap : public cc::SkipListMap< GC, Key, T, Traits >
22 {
23 typedef cc::SkipListMap< GC, Key, T, Traits > base_class;
24 public:
25 template <typename Config>
26 SkipListMap( Config const& /*cfg*/)
27 : base_class()
28 {}
29
30 std::pair<Key, bool> extract_min_key()
31 {
32 auto xp = base_class::extract_min();
33 if ( xp )
34 return std::make_pair( xp->first, true );
35 return std::make_pair( Key(), false );
36 }
37
38 std::pair<Key, bool> extract_max_key()
39 {
40 auto xp = base_class::extract_max();
41 if ( xp )
42 return std::make_pair( xp->first, true );
43 return std::make_pair( Key(), false );
44 }
45
46 // for testing
47 static constexpr bool const c_bExtractSupported = true;
48 static constexpr bool const c_bLoadFactorDepended = false;
49 static constexpr bool const c_bEraseExactKey = false;
50 };
