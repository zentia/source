116 template<typename compare_t=CompareTask_Par>
117 class Scheduler: public TaskFScheduler<task_f_t, compare_t> {
118 using baseSched = TaskFScheduler<task_f_t, compare_t>;
119 using baseSched::lb;
120 using baseSched::insertTask;
121 using baseSched::schedule_task;
122 using baseSched::handleCompletedTask;
123 enum { RELAX_MIN_BACKOFF=1, RELAX_MAX_BACKOFF=32};
124 public:
125 Scheduler(ff_loadbalancer* lb, const int maxnw, void (*schedRelaxF)(unsigned long)):
126 TaskFScheduler<task_f_t, compare_t>(lb,maxnw),
127 task_numb(0),task_completed(0),bk_count(0),schedRelaxF(schedRelaxF),
128 gd_ended(false) {
129 }
130 virtual ~Scheduler() {}
131
132 int svc_init() {
133 if (baseSched::svc_init()<0) return -1;
134 ff_node::input_active(true);
135
136 task_numb = task_completed = 0, bk_count = 0;
137 m=0; gd_ended = false;
138
139 return 0;
140 }
141
142 task_f_t* svc(task_f_t* task) {
143 if (!task) {
144 if (!gd_ended && (task_numb-task_completed)<(unsigned long)baseSched::LOWER_TH)
145 ff_node::input_active(true); // start receiveing from input channel again
146 else if (schedRelaxF) schedRelaxF(++bk_count);
147 return baseSched::GO_ON;
148 }
149 bk_count = 0;
150 if (baseSched::fromInput()) {
151 task_f_t *const msg = task;
152 if (msg->wtask == nullptr) {
153 gd_ended = true;
154 ff_node::input_active(false); // we don't want to read FF_EOS
155 return ((task_numb!=task_completed) ?
156 baseSched::GO_ON:
157 baseSched::EOS);
158 }
159 ++task_numb;
160 insertTask(msg);
161 schedule_task(0);
162 if ((task_numb-task_completed)>(unsigned long)baseSched::LOWER_TH) {
163 ff_node::input_active(false); // stop receiving from input channel
164 }
165 return baseSched::GO_ON;
166 }
167 hash_task_t * t = (hash_task_t *)task;
168 ++task_completed;
169 handleCompletedTask(t,lb->get_channel_id());
170 schedule_task(1); // try once more
171
172 if(task_numb==task_completed && gd_ended) return baseSched::EOS;
173 return baseSched::GO_ON;
174 }
175
176 void eosnotify(ssize_t /*id*/=-1) { lb->broadcast_task(FF_EOS); }
177 int wait_freezing() { return lb->wait_lb_freezing(); }
178
179 private:
180 size_t task_numb, task_completed, bk_count,m;
181 void (*schedRelaxF)(unsigned long);
182 bool gd_ended;
183 };

217 template<typename T1, typename compare_t = CompareTask_Par>
218 ff_mdf(void (*F)(T1*const), T1*const args, size_t outstandingTasks=DEFAULT_OUTSTANDING_TASKS,
219 int maxnw=ff_realNumCores(), void (*schedRelaxF)(unsigned long)=NULL):
220 ff_pipeline(false,outstandingTasks), farmworkers(maxnw) { //NOTE: pipe has fixed size queue by default
221 GD<T1> *_gd = new GD<T1>(F,args);
222 _gd->setMaxTasks(outstandingTasks+16); // NOTE: TASKS must be greater than pipe's queue!
223 farm = new ff_farm(false,640*maxnw,1024*maxnw,true,maxnw,true);
224
225 std::vector<ff_node *> w;
226 // NOTE: Worker objects are going to be destroyed by the farm destructor
227 for(int i=0;i<maxnw;++i) w.push_back(new TaskFWorker);
228 farm->add_workers(w);
229 farm->add_emitter(sched = new Scheduler<compare_t>(farm->getlb(), maxnw, schedRelaxF));
230 farm->wrap_around();
231
232 ff_pipeline::add_stage(_gd);
233 ff_pipeline::add_stage(farm);
234 if (ff_pipeline::run_then_freeze()<0) {
235 error("ff_mdf: running pipeline\n");
236 } else {
237 ff_pipeline::wait_freezing();
238 _gd->activate(true);
239 gd = _gd;
240 reset();
241 }
242 }
243 virtual ~ff_mdf() {
244 if (gd) delete gd;
245 if (sched) delete sched;
246 if (farm) delete farm;
247 }
