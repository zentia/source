50schema::Type any_of_to_capnproto_schema_type(
51 const parsing::schema::Type::AnyOf& _any_of,
52 const std::map<std::string, parsing::schema::Type>& _definitions,
53 const Parent _parent, schema::CapnProtoTypes* _cnp_types) {
54 auto value = schema::Type::Union{};
55 size_t i = 1;
56 for (const auto& type : _any_of.types_) {
57 value.fields.push_back(
58 std::make_pair(std::string("Opt" + std::to_string(i++)),
59 type_to_capnproto_schema_type(type, _definitions,
60 _parent, _cnp_types)));
61 }
62 if (_parent == Parent::is_a_struct) {
63 return schema::Type{.value = value};
64 } else {
65 const auto name =
66 std::string("Union") + std::to_string(_cnp_types->unions_.size() + 1);
67 _cnp_types->unions_[name] = schema::Type{.value = value};
68 return schema::Type{.value = schema::Type::Reference{name}};
69 }
70}

87schema::Type object_to_capnproto_schema_type(
88 const parsing::schema::Type::Object& _obj,
89 const std::map<std::string, parsing::schema::Type>& _definitions,
90 const Parent _parent, schema::CapnProtoTypes* _cnp_types) {
91 schema::Type::Struct struct_schema;
92 for (const auto& [k, v] : _obj.types_) {
93 struct_schema.fields.push_back(std::make_pair(
94 k, type_to_capnproto_schema_type(v, _definitions, Parent::is_a_struct,
95 _cnp_types)));
96 }
97 if (_parent == Parent::is_top_level) {
98 return schema::Type{.value = struct_schema};
99 } else {
100 const auto name =
101 std::string("Tuple") + std::to_string(_cnp_types->tuples_.size() + 1);
102 _cnp_types->tuples_[name] = schema::Type{.value = struct_schema};
103 return schema::Type{.value = schema::Type::Reference{name}};
104 }

129schema::Type reference_to_capnproto_schema_type(
130 const parsing::schema::Type::Reference& _reference,
131 const std::map<std::string, parsing::schema::Type>& _definitions,
132 const Parent _parent, schema::CapnProtoTypes* _cnp_types) {
133 const auto it = _definitions.find(_reference.name_);
134 if (it == _definitions.end() || is_named_type(it->second)) {
135 return schema::Type{
136 .value = schema::Type::Reference{.type_name = _reference.name_}};
137 } else {
138 return type_to_capnproto_schema_type(it->second, _definitions, _parent,
139 _cnp_types);
140 }

143schema::Type type_to_capnproto_schema_type(
144 const parsing::schema::Type& _type,
145 const std::map<std::string, parsing::schema::Type>& _definitions,
146 const Parent _parent, schema::CapnProtoTypes* _cnp_types) {
147 auto handle_variant = [&](const auto& _t) -> schema::Type {
148 using T = std::remove_cvref_t<decltype(_t)>;
149 using Type = parsing::schema::Type;
150 if constexpr (std::is_same<T, Type::Boolean>()) {
151 return schema::Type{.value = schema::Type::Bool{}};
152
153 } else if constexpr (std::is_same<T, Type::Bytestring>()) {
154 return schema::Type{.value = schema::Type::Data{}};
155
156 } else if constexpr (std::is_same<T, Type::Int32>() ||
157 std::is_same<T, Type::Integer>()) {
158 return schema::Type{.value = schema::Type::Int32{}};
159
160 } else if constexpr (std::is_same<T, Type::Int64>()) {
161 return schema::Type{.value = schema::Type::Int64{}};
162
163 } else if constexpr (std::is_same<T, Type::UInt32>()) {
164 return schema::Type{.value = schema::Type::UInt32{}};
165
166 } else if constexpr (std::is_same<T, Type::UInt64>()) {
167 return schema::Type{.value = schema::Type::UInt64{}};
168
169 } else if constexpr (std::is_same<T, Type::Float>()) {
170 return schema::Type{.value = schema::Type::Float32{}};
171
172 } else if constexpr (std::is_same<T, Type::Double>()) {
173 return schema::Type{.value = schema::Type::Float64{}};
174
175 } else if constexpr (std::is_same<T, Type::String>()) {
176 return schema::Type{.value = schema::Type::Text{}};
177
178 } else if constexpr (std::is_same<T, Type::AnyOf>()) {
179 return any_of_to_capnproto_schema_type(_t, _definitions, _parent,
180 _cnp_types);
181
182 } else if constexpr (std::is_same<T, Type::Description>()) {
183 return type_to_capnproto_schema_type(*_t.type_, _definitions, _parent,
184 _cnp_types);
185
186 } else if constexpr (std::is_same<T, Type::FixedSizeTypedArray>()) {
187 return schema::Type{
188 .value = schema::Type::List{
189 .type = Ref<schema::Type>::make(type_to_capnproto_schema_type(
190 *_t.type_, _definitions, Parent::is_not_a_struct,
191 _cnp_types))}};
192
193 } else if constexpr (std::is_same<T, Type::Literal>()) {
194 return literal_to_capnproto_schema_type(_t, _definitions, _parent,
195 _cnp_types);
196
197 } else if constexpr (std::is_same<T, Type::Object>()) {
198 return object_to_capnproto_schema_type(_t, _definitions, _parent,
199 _cnp_types);
200
201 } else if constexpr (std::is_same<T, Type::Optional>()) {
202 return optional_to_capnproto_schema_type(_t, _definitions, _parent,
203 _cnp_types);
204
205 } else if constexpr (std::is_same<T, Type::Reference>()) {
206 return reference_to_capnproto_schema_type(_t, _definitions, _parent,
207 _cnp_types);
208
209 } else if constexpr (std::is_same<T, Type::StringMap>()) {
210 _cnp_types->has_maps_ = true;
211 return schema::Type{
212 .value = schema::Type::Map{
213 .type = Ref<schema::Type>::make(type_to_capnproto_schema_type(
214 *_t.value_type_, _definitions, Parent::is_not_a_struct,
215 _cnp_types))}};
216
217 } else if constexpr (std::is_same<T, Type::Tuple>()) {
218 return type_to_capnproto_schema_type(
219 Type{parsing::schemaful::tuple_to_object(_t)}, _definitions, _parent,
220 _cnp_types);
221
222 } else if constexpr (std::is_same<T, Type::TypedArray>()) {
223 return schema::Type{
224 .value = schema::Type::List{
225 .type = Ref<schema::Type>::make(type_to_capnproto_schema_type(
226 *_t.type_, _definitions, Parent::is_not_a_struct,
227 _cnp_types))}};
228
229 } else if constexpr (std::is_same<T, Type::Validated>()) {
230 // Cap'n Proto knows no validation.
231 return type_to_capnproto_schema_type(*_t.type_, _definitions, _parent,
232 _cnp_types);
233 } else {
234 static_assert(rfl::always_false_v<T>, "Not all cases were covered.");
235 }
236 };
237
238 return rfl::visit(handle_variant, _type.variant_);
239}

241std::string to_string_representation(
242 const parsing::schema::Definition& _internal_schema) {
243 schema::CapnProtoTypes cnp_types;
244 for (const auto& [name, def] : _internal_schema.definitions_) {
245 if (!is_named_type(def)) {
246 continue;
247 }
248 cnp_types.structs_[name] = type_to_capnproto_schema_type(
249 def, _internal_schema.definitions_, Parent::is_top_level, &cnp_types);
250 }
251
252 std::stringstream sstream;
253 sstream << cnp_types;
254 return sstream.str();
255}
