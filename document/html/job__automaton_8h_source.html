49class job_automaton: no_copy {
50private:
51 tbb::atomic<intptr_t> my_job;
52public:
54 job_automaton() {
55 my_job = 0;
56 }
57
58 ~job_automaton() {
59 __TBB_ASSERT( my_job==-1, "must plug before destroying" );
60 }
61
63
64 bool try_acquire() {
65 intptr_t snapshot = my_job;
66 if( snapshot==-1 ) {
67 return false;
68 } else {
69 __TBB_ASSERT( (snapshot&1)==0, "already marked that way" );
70 intptr_t old = my_job.compare_and_swap( snapshot|1, snapshot );
71 __TBB_ASSERT( old==snapshot || old==-1, "unexpected interference" );
72 return old==snapshot;
73 }
74 }
75
76
77 void release() {
78 intptr_t snapshot = my_job;
79 __TBB_ASSERT( snapshot&1, NULL );
80 // Atomic store suffices here.
81 my_job = snapshot&~1;
82 }
83
85
86 void set_and_release( rml::job* job ) {
87 intptr_t value = reinterpret_cast<intptr_t>(job);
88 __TBB_ASSERT( (value&1)==0, "job misaligned" );
89 __TBB_ASSERT( value!=0, "null job" );
90 __TBB_ASSERT( my_job==1, "already set, or not marked busy?" );
91 // Atomic store suffices here.
92 my_job = value;
93 }
94
96
97 bool try_plug_null() {
98 return my_job.compare_and_swap( -1, 0 )==0;
99 }
100
102
103 bool try_plug( rml::job*&j ) {
104 for(;;) {
105 intptr_t snapshot = my_job;
106 if( snapshot&1 ) {
107 j = NULL;
108 return false;
109 }
110 // Not busy
111 if( my_job.compare_and_swap( -1, snapshot )==snapshot ) {
112 j = reinterpret_cast<rml::job*>(snapshot);
113 return true;
114 }
115 // Need to retry, because current thread may be non-owner that read a 0, and owner might have
116 // caused transition 0->1->ptr after we took our snapshot.
117 }
118 }
119
121 rml::job* wait_for_job() const {
122 intptr_t snapshot;
123 for(;;) {
124 snapshot = my_job;
125 if( snapshot&~1 ) break;
126 __TBB_Yield();
127 }
128 __TBB_ASSERT( snapshot!=-1, "wait on plugged job_automaton" );
129 return reinterpret_cast<rml::job*>(snapshot&~1);
130 }
131};
