95 int svc_init() {
96
97 if (oclId < 0) oclId = clEnvironment::instance()->getOCLID();
98
99 // the user has set a specific device
100 if (deviceId != NULL) return 0;
101
102 switch (dtype) {
103 case CL_DEVICE_TYPE_ALL: {
104 // no user choice, a static greedy algorithm is used to allocate openCL components
105 ssize_t GPUdevId =clEnvironment::instance()->getGPUDeviceRR();
106 if( (GPUdevId !=-1) && ( oclId < clEnvironment::instance()->getNumGPU())) {
107 printf("%d: Allocated a GPU device, the id is %ld\n", oclId, GPUdevId);
108 deviceId=clEnvironment::instance()->getDevice(GPUdevId);
109 return 0;
110 }
111 // fall back to CPU either GPU has reached its max or there is no GPU available
112 ssize_t CPUdevId =clEnvironment::instance()->getCPUDevice();
113 if (CPUdevId != -1) {
114 printf("%d: Allocated a CPU device as either no GPU device is available or no GPU slot is available (cpuId=%ld)\n",oclId, CPUdevId);
115 deviceId=clEnvironment::instance()->getDevice(CPUdevId);
116 return 0;
117 }
118 printf("%d: cannot allocate neither a GPU nor a CPU device\n", oclId);
119 return -1;
120 } break;
121 case CL_DEVICE_TYPE_GPU: {
122 ssize_t GPUdevId =clEnvironment::instance()->getGPUDeviceRR();
123 if( (GPUdevId !=-1) && ( oclId < clEnvironment::instance()->getNumGPU())) {
124 printf("%d: Allocated a GPU device, the id is %ld\n", oclId, GPUdevId);
125 deviceId=clEnvironment::instance()->getDevice(GPUdevId);
126 return 0;
127 }
128 printf("%d: cannot allocate a GPU device\n", oclId);
129 return -1;
130 } break;
131 case CL_DEVICE_TYPE_CPU: {
132 ssize_t CPUdevId =clEnvironment::instance()->getCPUDevice();
133 if (CPUdevId != -1) {
134 printf("%d: Allocated a CPU device (cpuId=%ld)\n",oclId, CPUdevId);
135 deviceId=clEnvironment::instance()->getDevice(CPUdevId);
136 return 0;
137 }
138 printf("%d: cannot allocate a CPU device\n", oclId);
139 return -1;
140 } break;
141 default : std::cerr << "Unknown/not supported device type\n";
142 return -1;
143 }
144 return 0;
145 }

147 void svc_end() {}

150 int oclId; // the OpenCL node id
151 cl_device_id deviceId; // is the id which is provided for user
