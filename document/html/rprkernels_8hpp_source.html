258static inline int sendFifoMsg(const char* msg, int isBegin) {
259 if (access (FIFO_FILE, F_OK) == -1) {
260 perror("access");
261 return -1;
262 }
263 FILE *fp;
264 fp = fopen(FIFO_FILE, "w");
265 if (fp == NULL) {
266 perror("fopen");
267 return -1;
268 }
269 if (isBegin == 1) {
270 char rKernelName[strlen(msg)+5];
271 memset(rKernelName, 0, sizeof rKernelName);
272 strcat(rKernelName,"B:\0");
273 strcat(rKernelName, msg);
274 strcat(rKernelName, ";\0");
275 if (fputs(rKernelName, fp) == EOF) {
276 perror("fputs");
277 return -1;
278 }
279 } else {
280 if (fputs(msg, fp) == EOF) {
281 perror("fputs");
282 return -1;
283 }
284 }
285 fclose(fp);
286 return 0;
287}

289static double getSourceResults(const Measurement& measurement) {
290 double result = 0.0;
291
292 // we have just one single kernel instance
293 Measurement::KernelSourceMap::const_iterator kernelSourceIt = measurement.kernelSourceMap().begin();
294 Measurement::SourceContainer::const_iterator resultsIt = kernelSourceIt->second.begin();
295 const Measurement::SourceMap& sourceMap = *resultsIt;
296 Measurement::SourceMap::const_iterator sourceIt = sourceMap.begin();
297 for (; sourceIt != sourceMap.end(); ++sourceIt) {
298 const Measurement::DataMap& dataMap = sourceIt->second;
299 Measurement::DataMap::const_iterator dataIt = dataMap.begin();
300 for (; dataIt != dataMap.end(); ++dataIt) {
301 // we are interested in only for the Energy field
302 if (dataIt->first == SourceCapability::Energy) {
303 result +=dataIt->second;
304 break;
305 }
306 }
307 }
308 return result;
309}

549 inline size_t getDeviceID(const std::string &cmd, Kernel_Target_t &target) {
550 if (cmd == "") return 0;
551 const std::string kid = "kernel_"+std::to_string(kernel_id);
552 const char *semicolon = ";";
553 size_t n = cmd.rfind(kid);
554 assert(n != std::string::npos);
555 n = cmd.find_first_of(semicolon, n);
556 assert(n != std::string::npos);
557 size_t m = cmd.find_first_of(semicolon, n+1);
558 assert(m != std::string::npos);
559 const std::string &device = cmd.substr(n+1, m-n-1);
560 target = String2Target[device];
561 return kernel2node[target];
562 }
563 inline size_t getScheduleID(const std::string &cmd) {
564 if (cmd == "") return 0;
565 size_t n = cmd.find_first_of(";");
566 assert(n != std::string::npos);
567 return std::stol(cmd.substr(0,n));

570 // called once at the very beginning
571 int nodeInit() {

575 assert(c);
576 (*c)[0].resize(1);

586 OUT_t *svc(IN_t *in) {
587 bool oneshot = false;
588 size_t selectedDevice = 0;
589 size_t problem_size = 0;
590 Kernel_Target_t target;
591 if (in == nullptr) { // non-streaming kernel execution
592 assert(Task != nullptr);
593 oneshot = true;
594 problem_size = problemSize_F(*Task);
595#if !defined(NO_DPE_SCHEDULE)
596 const std::string &cmd =schedule_kernel(kernel_id, problem_size);
597 Task->cmd = cmd;
598#endif
599 Task->kernel_id = kernel_id;
600 selectedDevice = getDeviceID(Task->cmd, target);
601 in = Task;
602 } else {
603 assert(Task == nullptr);
604 problem_size = problemSize_F(*in);
605 selectedDevice = getDeviceID(in->cmd, target);
606 in->kernel_id = kernel_id;
607 }
608 selector::selectNode(selectedDevice);
609
610 // this is the node that is going to be executed
611 ff_node *node = selector::getNode(selectedDevice);
612
613 ff_node::rpr_measure_t measure;
614 size_t stop, start, pstop,pstart;
615 measure.schedule_id = getScheduleID(in->cmd);
616 measure.problemSize = problem_size;
617 memory_Stats(statusStr.c_str(), start, pstart);
618#if defined(ENERGY_MEASUREMENT)
619 bool measureEnergy = node->rpr_get_measure_energy();
620 if (measureEnergy) EnergyMeasure_Begin("kernel");
621#endif
622 measure.time_before = getusec();
623 OUT_t *out = selector::svc(in); // executes the kernel!
624 measure.time_after = getusec();
625#if defined(ENERGY_MEASUREMENT)
626 if (measureEnergy) EnergyMeasure_End("kernel",measure.energy);
627 else measure.energy = 0.0;
628#endif
629 memory_Stats(statusStr.c_str(), stop, pstop);
630 // gets input/output size and virtual memory
631 measure.bytesIn = node->rpr_get_sizeIn();
632 measure.bytesOut = node->rpr_get_sizeOut();
633 measure.vmSize = (stop>start)?(stop-start):0;
634 measure.vmPeak = (pstop>pstart)?(pstop-pstart):0;
635
636 // stores the collected measures
637 RPR_measures_vector *pw_vector = pw.load();
638 (*pw_vector)[0][0][selectedDevice].first = target;
639 (*pw_vector)[0][0][selectedDevice].second.push_back(measure);
640
641 return oneshot?selector::EOS:out;
642 }
643
644private:
645 const int kernel_id;
646 IN_t *Task;
647 std::string statusStr;
648 std::function<size_t(const IN_t&)> problemSize_F;
649 std::map<Kernel_Target_t, size_t> kernel2node; // multimap ???
650 std::atomic<RPR_measures_vector *> pw;
651};

658class Pipe_Kernel: public ff_Pipe<> {
659 using pipe = ff_Pipe<>;
660
661public:
662 template<typename... STAGES>
663 explicit Pipe_Kernel(const int kernel_id, STAGES &&...stages):
664 ff_Pipe<>(stages...), kernel_id(kernel_id) {
665 //sets bounded queues
666 ff_Pipe<>::setFixedSize(true);
667
668 // Energy is measured for the entire pipeline and not for the single kernel
669 const svector<ff_node*> &nodes = getStages();
670 for(size_t i=0;i<nodes.size();++i)
671 nodes[i]->rpr_set_measure_energy(false);
672 }
673 virtual ~Pipe_Kernel() {}
674
675 int run_and_wait_end() {
676 int queue_size = register_pipeline(*this, kernel_id);
677 ff_Pipe<>::setXNodeInputQueueLength(queue_size);
678 ff_Pipe<>::setXNodeOutputQueueLength(queue_size);
679
680#if defined(ENERGY_MEASUREMENT)
681 EnergyMeasure_Begin("kernel");
682#endif
683 if (pipe::run_and_wait_end()<0) {
684 deregister_pipeline(kernel_id);
685 return -1;
686 }
687#if defined(ENERGY_MEASUREMENT)
688 EnergyMeasure_End("kernel", energy);
689#else
690 energy = 0.0;
691#endif
692 deregister_pipeline(kernel_id);
693 return 0;
694 }

700 RPR_measures_vector rpr_get_measures() {
701 const svector<ff_node*> &nodes = getStages();
702 RPR_measures_vector measures;
703 measures.reserve(nodes.size()+1);
704
705 // energy is stored for the first stage (the Generator stage) which is
706 // always a sequential stage running on the CPU
707 std::vector<RPR_devices_measure> tmp(1);
708 tmp[0].resize(1);
709 tmp[0][CPU].second.resize(1);
710 (tmp[0][CPU].second)[0].energy = energy;
711
712 (tmp[0][CPU].second)[0].time_before= getusec(ff_Pipe<>::startTime()); // stores pipeline starting time
713 (tmp[0][CPU].second)[0].time_after = getusec(); // stores current time
714
715 measures.push_back(tmp);
716
717 for(size_t i=1;i<nodes.size();++i) {
718 const RPR_measures_vector &V = nodes[i]->rpr_get_measures();
719 measures.insert(measures.end(), V.begin(), V.end());
720 }
721 return measures;
722 }
723protected:
724 const int kernel_id;
725 double energy = 0.0;

775 RPR_measures_vector rpr_get_measures() {
