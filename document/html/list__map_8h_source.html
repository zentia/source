687 template <typename Key, typename T, typename Compare, typename Allocator>
688 bool list_map<Key, T, Compare, Allocator>::push_front(const value_type& value)
689 {
690 internal_value_type tempValue(value);
691 typename base_type::insert_return_type baseReturn = base_type::insert(tempValue);
692
693 // Did the insert succeed?
694 if (baseReturn.second)
695 {
696 internal_value_type* pNode = &(*baseReturn.first);
697
698 pNode->mpNext = mNode.mpNext;
699 pNode->mpPrev = &mNode;
700
701 mNode.mpNext->mpPrev = pNode;
702 mNode.mpNext = pNode;
703
704 return true;
705 }
706 else
707 {
708 return false;
709 }
710 }

712 template <typename Key, typename T, typename Compare, typename Allocator>
713 bool list_map<Key, T, Compare, Allocator>::push_back(const value_type& value)
714 {
715 internal_value_type tempValue(value);
716 typename base_type::insert_return_type baseReturn = base_type::insert(tempValue);
717
718 // Did the insert succeed?
719 if (baseReturn.second)
720 {
721 internal_value_type* pNode = &(*baseReturn.first);
722
723 pNode->mpPrev = mNode.mpPrev;
724 pNode->mpNext = &mNode;
725
726 mNode.mpPrev->mpNext = pNode;
727 mNode.mpPrev = pNode;
728
729 return true;
730 }
731 else
732 {
733 return false;
734 }

897 {
898 if (!base_type::validate())
899 {
900 return false;
901 }
902
903 size_type nodeCount(0);
904 list_map_data_base* node = mNode.mpNext;
905 while (node != &mNode)
906 {
907 internal_value_type* data = static_cast<internal_value_type*>(node);
908 if (base_type::find(data->mValue.first) == base_type::end())
909 {
910 return false;
911 }
912 node = node->mpNext;
913 ++nodeCount;
914 }
915 if (nodeCount != size())
916 {
917 return false;
918 }
919 nodeCount = 0;
920 node = mNode.mpPrev;
921 while (node != &mNode)
922 {
923 internal_value_type* data = static_cast<internal_value_type*>(node);
924 if (base_type::find(data->mValue.first) == base_type::end())
925 {
926 return false;
927 }
928 node = node->mpPrev;
929 ++nodeCount;
930 }
931 if (nodeCount != size())
932 {
933 return false;
934 }
935
936 return true;
