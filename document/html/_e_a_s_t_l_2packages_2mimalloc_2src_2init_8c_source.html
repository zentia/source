291mi_decl_thread mi_tld_t* thread_tld = &tld_empty;

293// Allocate fresh tld
294static mi_tld_t* mi_tld_alloc(void) {
295 mi_atomic_increment_relaxed(&thread_count);
296 if (_mi_is_main_thread()) {
297 return &tld_main;
298 }
299 else {
300 // allocate tld meta-data
301 // note: we need to be careful to not access the tld from `_mi_meta_zalloc`
302 // (and in turn from `_mi_arena_alloc_aligned` and `_mi_os_alloc_aligned`).
303 mi_memid_t memid;
304 mi_tld_t* tld = (mi_tld_t*)_mi_meta_zalloc(sizeof(mi_tld_t), &memid);
305 if (tld==NULL) {
306 _mi_error_message(ENOMEM, "unable to allocate memory for thread local data\n");
307 return NULL;
308 }
309 tld->memid = memid;
310 tld->heap_backing = NULL;
311 tld->heaps = NULL;
312 tld->subproc = &subproc_main;
313 tld->numa_node = _mi_os_numa_node();
314 tld->thread_id = _mi_prim_thread_id();
315 tld->thread_seq = mi_atomic_add_acq_rel(&thread_total_count, 1);
316 tld->is_in_threadpool = _mi_prim_thread_is_in_threadpool();
317 return tld;
318 }
319}

337static mi_tld_t* mi_tld(void) {
338 mi_tld_t* tld = thread_tld;
339 if (tld == MI_TLD_INVALID) {
340 _mi_error_message(EFAULT, "internal error: tld is accessed after the thread terminated\n");
341 thread_tld = &tld_empty;
342 }
343 if (tld==&tld_empty) {
344 thread_tld = tld = mi_tld_alloc();
345 }
346 return tld;

402void mi_subproc_delete(mi_subproc_id_t subproc_id) {
403 if (subproc_id == NULL) return;
404 mi_subproc_t* subproc = _mi_subproc_from_id(subproc_id);
405 // check if there are os pages still..
406 bool safe_to_delete = false;

413

415 _mi_stats_merge_from(&_mi_subproc_main()->stats, &subproc->stats);

437// Initialize the thread local default heap, called from `mi_thread_init`
438static bool _mi_thread_heap_init(void) {
439 if (mi_heap_is_initialized(mi_prim_get_default_heap())) return true;
440 if (_mi_is_main_thread()) {
441 // mi_assert_internal(heap_main.thread_id != 0); // can happen on freeBSD where alloc is called before any initialization
442 // the main heap is statically allocated
443 mi_heap_main_init();
444 _mi_heap_set_default_direct(&heap_main);
445 //mi_assert_internal(_mi_heap_default->tld->heap_backing == mi_prim_get_default_heap());
446 }
447 else {
448 // allocates tld data
449 // note: we cannot access thread-locals yet as that can cause (recursive) allocation
450 // (on macOS <= 14 for example where the loader allocates thread-local data on demand).
451 mi_tld_t* tld = mi_tld_alloc();
452
453 // allocate and initialize the heap
454 mi_heap_t* heap = _mi_heap_create(0 /* default tag */, false /* allow destroy? */, _mi_arena_id_none(), tld);
455
456 // associate the heap with this thread
457 // (this is safe, on macOS for example, the heap is set in a dedicated TLS slot and thus does not cause recursive allocation)
458 _mi_heap_set_default_direct(heap);
459
460 // now that the heap is set for this thread, we can set the thread-local tld.
461 thread_tld = tld;
462 }
463 return false;
464}

467// Free the thread local default heap (called from `mi_thread_done`)
468static bool _mi_thread_heap_done(mi_heap_t* heap) {
469 if (!mi_heap_is_initialized(heap)) return true;
470
471 // reset default heap
472 _mi_heap_set_default_direct(_mi_is_main_thread() ? &heap_main : (mi_heap_t*)&_mi_heap_empty);
473
474 // switch to backing heap
475 heap = heap->tld->heap_backing;
476 if (!mi_heap_is_initialized(heap)) return false;
477
478 // delete all non-backing heaps in this thread
479 mi_heap_t* curr = heap->tld->heaps;
480 while (curr != NULL) {
481 mi_heap_t* next = curr->next; // save `next` as `curr` will be freed
482 if (curr != heap) {
483 mi_assert_internal(!mi_heap_is_backing(curr));
484 mi_heap_delete(curr);
485 }
486 curr = next;
487 }
488 mi_assert_internal(heap->tld->heaps == heap && heap->next == NULL);
489 mi_assert_internal(mi_heap_is_backing(heap));
490
491 // collect if not the main thread
492 if (heap != &heap_main) {
493 _mi_heap_collect_abandon(heap);
494 }
495
496 // free heap meta data
497 _mi_meta_free(heap, sizeof(mi_heap_t), heap->memid);
498
499 if (heap == &heap_main) {
500 #if 0
501 // never free the main thread even in debug mode; if a dll is linked statically with mimalloc,
502 // there may still be delete/free calls after the mi_fls_done is called. Issue #207
503 _mi_heap_destroy_pages(heap);
504 mi_assert_internal(heap->tld->heap_backing == &heap_main);
505 #endif
506 }
507
508 return false;
509}

513// --------------------------------------------------------
514// Try to run `mi_thread_done()` automatically so any memory
515// owned by the thread but not yet released can be abandoned
516// and re-owned by another thread.
517//
518// 1. windows dynamic library:
519// call from DllMain on DLL_THREAD_DETACH
520// 2. windows static library:
521// use special linker section to call a destructor when the thread is done
522// 3. unix, pthreads:
523// use a pthread key to call a destructor when a pthread is done
524//
525// In the last two cases we also need to call `mi_process_init`
526// to set up the thread local keys.
527// --------------------------------------------------------
528
529// Set up handlers so `mi_thread_done` is called automatically
530static void mi_process_setup_auto_thread_done(void) {
531 static bool tls_initialized = false; // fine if it races
532 if (tls_initialized) return;
533 tls_initialized = true;
534 _mi_prim_thread_init_auto_done();
535 _mi_heap_set_default_direct(&heap_main);
536}

544// This is called from the `mi_malloc_generic`
545void mi_thread_init(void) mi_attr_noexcept
546{
547 // ensure our process has started already
548 mi_process_init();
549
550 // initialize the thread local default heap
551 // (this will call `_mi_heap_set_default_direct` and thus set the
552 // fiber/pthread key to a non-zero value, ensuring `_mi_thread_done` is called)
553 if (_mi_thread_heap_init()) return; // returns true if already initialized
554
555 mi_subproc_stat_increase(_mi_subproc_main(), threads, 1);
556 //_mi_verbose_message("thread init: 0x%zx\n", _mi_thread_id());
557}

563void _mi_thread_done(mi_heap_t* heap)
564{
565 // calling with NULL implies using the default heap
566 if (heap == NULL) {
567 heap = mi_prim_get_default_heap();
568 if (heap == NULL) return;
569 }
570
571 // prevent re-entrancy through heap_done/heap_set_default_direct (issue #699)
572 if (!mi_heap_is_initialized(heap)) {
573 return;
574 }
575
576 // adjust stats
577 mi_subproc_stat_decrease(_mi_subproc_main(), threads, 1);
578
579 // check thread-id as on Windows shutdown with FLS the main (exit) thread may call this on thread-local heaps...
580 if (heap->tld->thread_id != _mi_prim_thread_id()) return;
581
582 // abandon the thread local heap
583 // note: we store the tld as we should avoid reading `thread_tld` at this point (to avoid reinitializing the thread local storage)

679static void mi_detect_cpu_features(void) {
680 // FSRM for fast short rep movsb/stosb support (AMD Zen3+ (~2020) or Intel Ice Lake+ (~2017))
681 // EMRS for fast enhanced rep movsb/stosb support
682 uint32_t cpu_info[4];
683 if (mi_cpuid(cpu_info, 7)) {
684 _mi_cpu_has_fsrm = ((cpu_info[3] & (1 << 4)) != 0); // bit 4 of EDX : see <https://en.wikipedia.org/wiki/CPUID#EAX=7,_ECX=0:_Extended_Features>
685 _mi_cpu_has_erms = ((cpu_info[1] & (1 << 9)) != 0); // bit 9 of EBX : see <https://en.wikipedia.org/wiki/CPUID#EAX=7,_ECX=0:_Extended_Features>
686 }
687 if (mi_cpuid(cpu_info, 1)) {
688 _mi_cpu_has_popcnt = ((cpu_info[2] & (1 << 23)) != 0); // bit 23 of ECX : see <https://en.wikipedia.org/wiki/CPUID#EAX=1:_Processor_Info_and_Feature_Bits>
689 }
690}
691
692#else
693static void mi_detect_cpu_features(void) {
694 #if MI_ARCH_ARM64
695 _mi_cpu_has_popcnt = true;
696 #endif
697}
698#endif

701// Initialize the process; called by thread_init or the process loader
702void mi_process_init(void) mi_attr_noexcept {
703 // ensure we are called once
704 static mi_atomic_once_t process_init;
705 #if _MSC_VER < 1920
706 mi_heap_main_init(); // vs2017 can dynamically re-initialize heap_main
707 #endif
708 if (!mi_atomic_once(&process_init)) return;
709 _mi_process_is_initialized = true;
710 _mi_verbose_message("process init: 0x%zx\n", _mi_thread_id());
711
712 mi_detect_cpu_features();
713 _mi_os_init();
714 _mi_page_map_init();
715 mi_heap_main_init();
716 mi_tld_main_init();
717 // the following two can potentially allocate (on freeBSD for locks and thread keys)
718 mi_subproc_main_init();
719 mi_process_setup_auto_thread_done();
720 mi_thread_init();
721
722 #if defined(_WIN32) && defined(MI_WIN_USE_FLS)
723 // On windows, when building as a static lib the FLS cleanup happens to early for the main thread.
724 // To avoid this, set the FLS value for the main thread to NULL so the fls cleanup
725 // will not call _mi_thread_done on the (still executing) main thread. See issue #508.
726 _mi_prim_thread_associate_default_heap(NULL);
727 #endif
728
729 mi_stats_reset(); // only call stat reset *after* thread init (or the heap tld == NULL)
730 mi_track_init();
731
732 if (mi_option_is_enabled(mi_option_reserve_huge_os_pages)) {
733 size_t pages = mi_option_get_clamp(mi_option_reserve_huge_os_pages, 0, 128*1024);
734 long reserve_at = mi_option_get(mi_option_reserve_huge_os_pages_at);
735 if (reserve_at != -1) {
736 mi_reserve_huge_os_pages_at(pages, reserve_at, pages*500);
737 } else {
738 mi_reserve_huge_os_pages_interleave(pages, 0, pages*500);
739 }
740 }
741 if (mi_option_is_enabled(mi_option_reserve_os_memory)) {
742 long ksize = mi_option_get(mi_option_reserve_os_memory);
743 if (ksize > 0) {
744 mi_reserve_os_memory((size_t)ksize*MI_KiB, true, true);
745 }
746 }
747}

749// Called when the process is done (through `at_exit`)
750void mi_cdecl _mi_process_done(void) {
751 // only shutdown if we were initialized
752 if (!_mi_process_is_initialized) return;
753 // ensure we are called once
754 static bool process_done = false;
755 if (process_done) return;
756 process_done = true;
757
758 // get the default heap so we don't need to acces thread locals anymore
759 mi_heap_t* heap = mi_prim_get_default_heap(); // use prim to not initialize any heap
760 mi_assert_internal(heap != NULL);
761
762 // release any thread specific resources and ensure _mi_thread_done is called on all but the main thread
763 _mi_prim_thread_done_auto_done();
764
765
766 #ifndef MI_SKIP_COLLECT_ON_EXIT
767 #if (MI_DEBUG || !defined(MI_SHARED_LIB))
768 // free all memory if possible on process exit. This is not needed for a stand-alone process
769 // but should be done if mimalloc is statically linked into another shared library which
770 // is repeatedly loaded/unloaded, see issue #281.
771 mi_heap_collect(heap, true /* force */ );
772 #endif
773 #endif
774
775 // Forcefully release all retained memory; this can be dangerous in general if overriding regular malloc/free
776 // since after process_done there might still be other code running that calls `free` (like at_exit routines,
777 // or C-runtime termination code.
778 if (mi_option_is_enabled(mi_option_destroy_on_exit)) {
779 mi_heap_collect(heap, true /* force */);
780 _mi_heap_unsafe_destroy_all(heap); // forcefully release all memory held by all heaps (of this thread only!)
781 _mi_arenas_unsafe_destroy_all(heap->tld);
782 }
783
784 if (mi_option_is_enabled(mi_option_show_stats) || mi_option_is_enabled(mi_option_verbose)) {
785 mi_stats_print(NULL);
786 }
787 _mi_allocator_done();
788 _mi_verbose_message("process done: 0x%zx\n", tld_main.thread_id);
789 os_preloading = true; // don't call the C runtime anymore
790}
