235 inline void* allocator::allocate(size_t n, size_t alignment, size_t offset, int flags)
236 {
237 #if EASTL_DLL
238 // We currently have no support for implementing flags when
239 // using the C runtime library operator new function. The user
240 // can use SetDefaultAllocator to override the default allocator.
241 EA_UNUSED(offset); EA_UNUSED(flags);
242
243 size_t adjustedAlignment = (alignment > EA_PLATFORM_PTR_SIZE) ? alignment : EA_PLATFORM_PTR_SIZE;
244
245 void* p = new char[n + adjustedAlignment + EA_PLATFORM_PTR_SIZE];
246 void* pPlusPointerSize = (void*)((uintptr_t)p + EA_PLATFORM_PTR_SIZE);
247 void* pAligned = (void*)(((uintptr_t)pPlusPointerSize + adjustedAlignment - 1) & ~(adjustedAlignment - 1));
248
249 void** pStoredPtr = (void**)pAligned - 1;
250 EASTL_ASSERT(pStoredPtr >= p);
251 *(pStoredPtr) = p;
252
253 EASTL_ASSERT(((size_t)pAligned & ~(alignment - 1)) == (size_t)pAligned);
254
255 return pAligned;
256 #elif (EASTL_DEBUGPARAMS_LEVEL <= 0)
257 return ::new(alignment, offset, (char*)0, flags, 0, (char*)0, 0) char[n];
258 #elif (EASTL_DEBUGPARAMS_LEVEL == 1)
259 return ::new(alignment, offset, pName, flags, 0, (char*)0, 0) char[n];
260 #else
261 return ::new(alignment, offset, pName, flags, 0, __FILE__, __LINE__) char[n];
262 #endif
263
264 #undef pName // See above for the definition of this.
265 }

341 template <typename Allocator>
342 inline void* allocate_memory(Allocator& a, size_t n, size_t alignment, size_t alignmentOffset)
343 {
344 void *result;
345 if (alignment <= EASTL_ALLOCATOR_MIN_ALIGNMENT)
346 {
347 result = EASTLAlloc(a, n);
348 // Ensure the result is correctly aligned. An assertion likely indicates a mismatch between EASTL_ALLOCATOR_MIN_ALIGNMENT and the minimum alignment
349 // of EASTLAlloc. If there is a mismatch it may be necessary to define EASTL_ALLOCATOR_MIN_ALIGNMENT to be the minimum alignment of EASTLAlloc, or
350 // to increase the alignment of EASTLAlloc to match EASTL_ALLOCATOR_MIN_ALIGNMENT.
351 EASTL_ASSERT((reinterpret_cast<size_t>(result)& ~(alignment - 1)) == reinterpret_cast<size_t>(result));
352 }
353 else
354 {
355 result = EASTLAllocAligned(a, n, alignment, alignmentOffset);
356 // Ensure the result is correctly aligned. An assertion here may indicate a bug in the allocator.
357 auto resultMinusOffset = (char*)result - alignmentOffset;
358 EA_UNUSED(resultMinusOffset);
359 EASTL_ASSERT((reinterpret_cast<size_t>(resultMinusOffset)& ~(alignment - 1)) == reinterpret_cast<size_t>(resultMinusOffset));
360 }
361 return result;
362 }
