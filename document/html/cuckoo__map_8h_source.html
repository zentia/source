28 struct node_type: public intrusive::cuckoo::node<probeset_type, store_hash_count>
29 {
30 value_type m_val;

34 : m_val( std::make_pair( key_type(key), mapped_type()))
35 {}

37 template <typename K, typename Q>

39 : m_val( std::make_pair( key_type(key), mapped_type(v)))
40 {}
41
42 template <typename K, typename... Args>
43 node_type( K&& key, Args&&... args )
44 : m_val( std::forward<K>(key), std::move( mapped_type(std::forward<Args>(args)...)))
45 {}
46 };
