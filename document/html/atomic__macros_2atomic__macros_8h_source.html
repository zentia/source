145// We write some of our variables in inline assembly, which MSAN
146// doesn't understand. This macro forces initialization of those
147// variables when MSAN is enabled and doesn't pay the initialization
148// cost when it's not enabled.
149#if EA_MSAN_ENABLED
150 #define EASTL_ATOMIC_DEFAULT_INIT(type, var) type var{}
151#else
152 #define EASTL_ATOMIC_DEFAULT_INIT(type, var) type var
153#endif // EA_MSAN_ENABLED
