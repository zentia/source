240 static const size_t nSize = 20;
241 typedef typename List::value_type value_type;
242 value_type arr[ nSize ];
243
244 for ( size_t i = 0; i < nSize; ++i ) {
245 arr[i].nKey = static_cast<int>( i );
246 arr[i].nVal = arr[i].nKey * 10;
247 }
248 shuffle( arr, arr + nSize );
249
250 ASSERT_TRUE( l.empty());
251 ASSERT_CONTAINER_SIZE( l, 0 );
252
253 // insert / find
254 for ( auto& i : arr ) {
255 EXPECT_FALSE( l.contains( i.nKey ));
256 EXPECT_FALSE( l.contains( other_item( i.nKey ), other_less()));
257 EXPECT_FALSE( l.find( i.nKey, []( value_type& item, int ) { ++item.s.nFindCall; } ));
258 EXPECT_EQ( i.s.nFindCall, 0 );
259 EXPECT_FALSE( l.find_with( other_item( i.nKey ), other_less(), []( value_type& item, other_item const& ) { ++item.s.nFindCall; } ));
260 EXPECT_EQ( i.s.nFindCall, 0 );
261
262 if ( i.nKey & 1 )
263 EXPECT_TRUE( l.insert( i ));
264 else {
265 EXPECT_EQ( i.s.nInsertCall, 0 );
266 EXPECT_TRUE( l.insert( i, []( value_type& v ) { ++v.s.nInsertCall; } ));
267 EXPECT_EQ( i.s.nInsertCall, 1 );
268 }
269
270 EXPECT_TRUE( l.contains( i.nKey ));
271 EXPECT_TRUE( l.contains( i ));
272 EXPECT_TRUE( l.contains( other_item( i.nKey ), other_less()));
273 EXPECT_TRUE( l.find( i.nKey, []( value_type& item, int ) { ++item.s.nFindCall; } ));
274 EXPECT_EQ( i.s.nFindCall, 1 );
275 EXPECT_TRUE( l.find( i, []( value_type& item, value_type const& ) { ++item.s.nFindCall; } ));
276 EXPECT_EQ( i.s.nFindCall, 2 );
277 EXPECT_TRUE( l.find_with( other_item( i.nKey ), other_less(), []( value_type& item, other_item const& ) { ++item.s.nFindCall; } ));
278 EXPECT_EQ( i.s.nFindCall, 3 );
279
280 EXPECT_FALSE( l.insert( i ));
281 ASSERT_FALSE( l.empty());
282 }
283 ASSERT_CONTAINER_SIZE( l, nSize );
284
285 // check all items
286 for ( auto const& i : arr ) {
287 EXPECT_TRUE( l.contains( i.nKey ));
288 EXPECT_TRUE( l.contains( i ));
289 EXPECT_TRUE( l.contains( other_item( i.nKey ), other_less()));
290 EXPECT_TRUE( l.find( i.nKey, []( value_type& item, int ) { ++item.s.nFindCall; } ));
291 EXPECT_EQ( i.s.nFindCall, 4 );
292 EXPECT_TRUE( l.find( i, []( value_type& item, value_type const& ) { ++item.s.nFindCall; } ));
293 EXPECT_EQ( i.s.nFindCall, 5 );
294 EXPECT_TRUE( l.find_with( other_item( i.nKey ), other_less(), []( value_type& item, other_item const& ) { ++item.s.nFindCall; } ));
295 EXPECT_EQ( i.s.nFindCall, 6 );
296 }
297 ASSERT_FALSE( l.empty());
298 ASSERT_CONTAINER_SIZE( l, nSize );
299
300 // update existing test
301 for ( auto& i : arr ) {
302 EXPECT_EQ( i.s.nUpdateExistsCall, 0 );
303 std::pair<bool, bool> ret = l.update( i, update_functor());
304 EXPECT_TRUE( ret.first );
305 EXPECT_FALSE( ret.second );
306 EXPECT_EQ( i.s.nUpdateExistsCall, 1 );
307
308 ret = l.update( i, []( bool bNew, value_type& v, value_type& arg ) {
309 EXPECT_FALSE( bNew );
310 EXPECT_EQ( v.s.nUpdateExistsCall, 1 );
311 EXPECT_TRUE( &v == &arg );
312 ++v.s.nUpdateExistsCall;
313 });
314 EXPECT_TRUE( ret.first );
315 EXPECT_FALSE( ret.second );
316 EXPECT_EQ( i.s.nUpdateExistsCall, 2 );
317 }
318
319 // erase test
320 for ( auto const& i : arr ) {
321 if ( i.nKey & 1 )
322 EXPECT_TRUE( l.erase( i.nKey ));
323 else
324 EXPECT_TRUE( l.erase_with( other_item( i.nKey ), other_less()));
325
326 EXPECT_FALSE( l.contains( i ));
327 }
328 EXPECT_TRUE( l.empty());
329 EXPECT_CONTAINER_SIZE( l, 0 );
330
331 // Apply retired pointer to clean links
332 List::gc::force_dispose();
333
334 for ( auto const& i : arr )
335 EXPECT_EQ( i.s.nDisposeCount, 1 );
336
337 // erase with functor
338 for ( auto& i : arr ) {
339 std::pair<bool, bool> ret = l.update( i, update_functor(), false );
340 EXPECT_FALSE( ret.first );
341 EXPECT_FALSE( ret.second );
342
343 ret = l.update( i, update_functor(), true );
344 EXPECT_TRUE( ret.first );
345 EXPECT_TRUE( ret.second );
346 EXPECT_EQ( i.s.nUpdateNewCall, 1 );
347 }
348 EXPECT_FALSE( l.empty());
349 EXPECT_CONTAINER_SIZE( l, nSize );

352 EXPECT_EQ( i.s.nEraseCall, 0 );

358 EXPECT_TRUE( l.erase( i.nKey, []( value_type& item) { ++item.s.nEraseCall; } ));
