49template<class R, class F, class... Args>
50 requires std::is_invocable_r_v<R, F, Args...>
51constexpr R invoke_r(F &&f, Args &&...args)// freestanding
52 noexcept(std::is_nothrow_invocable_r_v<R, F, Args...>) {
53 if constexpr (std::is_void_v<R>)
54 return std::invoke(std::forward<F>(f), std::forward<Args>(args)...);
55 else
56 return std::invoke(std::forward<F>(f), std::forward<Args>(args)...);
57}

334struct _move_only_pointer {
335 union value_type {
336 void *p_ = nullptr;
337 void const *cp_;
338 void (*fp_)();
339 } val;
340
341 _move_only_pointer() = default;
342 _move_only_pointer(_move_only_pointer const &) = delete;
343 _move_only_pointer &operator=(_move_only_pointer const &) = delete;
344
345 constexpr _move_only_pointer(_move_only_pointer &&other) noexcept
346 : val(std::exchange(other.val, {})) {}

348 template<class T>
349 requires std::is_object_v<T>
350 constexpr explicit _move_only_pointer(T *p) noexcept : val{.p_ = p} {}
351
352 template<class T>
353 requires std::is_object_v<T>
354 constexpr explicit _move_only_pointer(T const *p) noexcept : val{.cp_ = p} {}
355
356 template<class T>
357 requires std::is_function_v<T>
358 constexpr explicit _move_only_pointer(T *p) noexcept
359 : val{.fp_ = reinterpret_cast<decltype(val.fp_)>(p)} {}
360
361 template<class T>
362 requires std::is_object_v<T>
363 constexpr _move_only_pointer &operator=(T *p) noexcept {
364 val.p_ = p;
365 return *this;
366 }
367
368 template<class T>
369 requires std::is_object_v<T>
370 constexpr _move_only_pointer &operator=(T const *p) noexcept {
371 val.cp_ = p;
372 return *this;
373 }
374
375 template<class T>
376 requires std::is_function_v<T>
377 constexpr _move_only_pointer &operator=(T *p) noexcept {
378 val.fp_ = reinterpret_cast<decltype(val.fp_)>(p);
379 return *this;
380 }
381
382 constexpr _move_only_pointer &operator=(_move_only_pointer &&other) noexcept {
383 val = std::exchange(other.val, {});
384 return *this;

388template<bool noex, class R, class... Args>
389struct _callable_trait {
390 using handle = _move_only_pointer::value_type;
391
392 typedef auto call_t(handle, Args...) noexcept(noex) -> R;
393 typedef void destroy_t(handle) noexcept;
394
395 struct vtable {
396 call_t *call = 0;
397 destroy_t *destroy = [](handle) noexcept {};
398 };
399
400 static inline constinit vtable const abstract_base;
401
402 template<class T>
403 constexpr static auto get(handle val) {
404 if constexpr (std::is_const_v<T>)
405 return static_cast<T *>(val.cp_);
406 else if constexpr (std::is_object_v<T>)
407 return static_cast<T *>(val.p_);
408 else
409 return reinterpret_cast<T *>(val.fp_);
410 }
411
412 // See also: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=71954
413 template<class T, template<class> class quals>
414 static inline constinit vtable const callable_target{
415 .call = [](handle this_, Args... args) noexcept(noex) -> R {
416 if constexpr (std::is_lvalue_reference_v<T> or std::is_pointer_v<T>) {
417 using Tp = std::remove_reference_t<std::remove_pointer_t<T>>;
418 return luisa::move_only_function_detail::invoke_r<R>(*get<Tp>(this_),
419 static_cast<Args>(args)...);
420 } else {
421 using Fp = typename quals<T>::type;
422 return luisa::move_only_function_detail::invoke_r<R>(static_cast<Fp>(*get<T>(this_)),
423 static_cast<Args>(args)...);
424 }
425 },
426 .destroy =
427 [](handle this_) noexcept {
428 if constexpr (not std::is_lvalue_reference_v<T> and
429 not std::is_pointer_v<T>)
430 delete get<T>(this_);
431 },
432 };
433
434 template<auto f>
435 static inline constinit vtable const unbound_callable_target{
436 .call = [](handle, Args... args) noexcept(noex) -> R { return luisa::move_only_function_detail::invoke_r<R>(f, static_cast<Args>(args)...); },
437 };
438
439 template<auto f, class T, template<class> class quals>
440 static inline constinit vtable const bound_callable_target{
441 .call = [](handle this_, Args... args) noexcept(noex) -> R {
442 if constexpr (std::is_pointer_v<T>) {
443 using Tp = std::remove_pointer_t<T>;
444 return luisa::move_only_function_detail::invoke_r<R>(f, get<Tp>(this_),
445 static_cast<Args>(args)...);
446 } else if constexpr (std::is_lvalue_reference_v<T>) {
447 using Tp = std::remove_reference_t<T>;
448 return luisa::move_only_function_detail::invoke_r<R>(f, *get<Tp>(this_),
449 static_cast<Args>(args)...);
450 } else {
451 using Fp = typename quals<T>::type;
452 return luisa::move_only_function_detail::invoke_r<R>(f, static_cast<Fp>(*get<T>(this_)),
453 static_cast<Args>(args)...);
454 }
455 },
456 .destroy =
457 [](handle this_) noexcept {
458 if constexpr (not std::is_lvalue_reference_v<T> and
459 not std::is_pointer_v<T>)
460 delete get<T>(this_);
461 },
462 };
463
464 template<auto f, class T>
465 static inline constinit vtable const boxed_callable_target{
466 .call = [](handle this_, Args... args) noexcept(noex) -> R {
467 return luisa::move_only_function_detail::invoke_r<R>(f, get<T>(this_),
468 static_cast<Args>(args)...);
469 },
470 .destroy =
471 [](handle this_) noexcept {
472 using D = typename std::unique_ptr<T>::deleter_type;
473 static_assert(std::is_trivially_default_constructible_v<D>);
474 if (auto p = get<T>(this_))
475 D()(p);
476 },
477 };
478};

493template<class S, class R, class... Args>
494class move_only_function<S, R(Args...)> {
495 using signature = _full_fn_sig<S>;

497 template<class T>
498 using cv = typename signature::template cv<T>;
499 template<class T>
500 using ref = typename signature::template ref<T>;
501
502 static constexpr bool noex = signature::is_noexcept;
503 static constexpr bool is_const = std::is_same_v<cv<void>, void const>;
504 static constexpr bool is_lvalue_only = std::is_same_v<ref<int>, int &>;
505 static constexpr bool is_rvalue_only = std::is_same_v<ref<int>, int &&>;

507 template<class T>
508 using cvref = ref<cv<T>>;
509 template<class T>
510 struct inv_quals_f
511 : std::conditional<is_lvalue_only or is_rvalue_only, cvref<T>, cv<T> &> {};
512 template<class T>
513 using inv_quals = typename inv_quals_f<T>::type;
514
515 template<class... T>
516 static constexpr bool is_invocable_using =
517 std::conditional_t<noex, std::is_nothrow_invocable_r<R, T..., Args...>,
518 std::is_invocable_r<R, T..., Args...>>::value;
519
520 template<class VT>
521 static constexpr bool is_callable_from =
522 is_invocable_using<cvref<VT>> and is_invocable_using<inv_quals<VT>>;
523
524 template<auto f, class VT>
525 static constexpr bool is_callable_as_if_from =
526 is_invocable_using<decltype(f), inv_quals<VT>>;

528 using trait = _callable_trait<noex, R, _param_t<Args>...>;
529 using vtable = typename trait::vtable;
530
531 std::reference_wrapper<vtable const> vtbl_ = trait::abstract_base;
532 _move_only_pointer obj_;
533
534public:
535 using result_type = R;

538 move_only_function(nullptr_t) noexcept : move_only_function() {}

540 template<class F, class VT = std::decay_t<F>>
541 move_only_function(F &&f) noexcept(
542 std::is_nothrow_invocable_v<decltype(_take_reference), F>)
543 requires _is_not_self<F, move_only_function> and
544 _does_not_specialize<F, in_place_type_t> and
545 is_callable_from<VT> and std::is_constructible_v<VT, F>

553 inv_quals_f>;
554 obj_ = _take_reference(std::forward<F>(f));
555 }
