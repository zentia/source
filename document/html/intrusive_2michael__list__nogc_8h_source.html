134 {
135 assert( pNode != nullptr );
136 link_checker::is_empty( pNode );
137
138 pNode->m_pNext.store( pos.pCur, memory_model::memory_order_relaxed );
139 if ( cds_likely( pos.pPrev->compare_exchange_strong( pos.pCur, pNode, memory_model::memory_order_release, atomics::memory_order_relaxed )))

556 template <typename Func>
557 std::pair<iterator, bool> update_at_( atomic_node_ptr& refHead, value_type& val, Func func, bool bAllowInsert )
558 {
559 position pos;
560
561 while ( true ) {
562 if ( search( refHead, val, key_comparator(), pos )) {
563 assert( key_comparator()( val, *node_traits::to_value_ptr( *pos.pCur )) == 0 );
564
565 func( false, *node_traits::to_value_ptr( *pos.pCur ) , val );
566 m_Stat.onUpdateExisting();
567 return std::make_pair( iterator( pos.pCur ), false );
568 }
569 else {
570 if ( !bAllowInsert ) {
571 m_Stat.onUpdateFailed();
572 return std::make_pair( end(), false );
573 }
574
575 if ( link_node( node_traits::to_node_ptr( val ), pos )) {
576 ++m_ItemCounter;
577 func( true, val , val );
578 m_Stat.onUpdateNew();
579 return std::make_pair( iterator( node_traits::to_node_ptr( val )), true );
580 }
581 }
582
583 m_Stat.onUpdateRetry();
584 }

643 template <typename Q, typename Compare >
644 bool search( atomic_node_ptr& refHead, const Q& val, Compare cmp, position& pos )
645 {
646 atomic_node_ptr * pPrev;
647 node_type * pNext;
648 node_type * pCur;
649
650 back_off bkoff;
651
652 try_again:
653 pPrev = &refHead;
654 pCur = pPrev->load(memory_model::memory_order_acquire);
655 pNext = nullptr;
656
657 while ( true ) {
658 if ( !pCur ) {
659 pos.pPrev = pPrev;
660 pos.pCur = pCur;
661 pos.pNext = pNext;
662 return false;
663 }
664
665 pNext = pCur->m_pNext.load(memory_model::memory_order_relaxed);
666 if ( cds_unlikely( pCur->m_pNext.load(memory_model::memory_order_acquire) != pNext )) {
667 bkoff();
668 goto try_again;
669 }
670
671 if ( cds_unlikely( pPrev->load(memory_model::memory_order_acquire) != pCur )) {
672 bkoff();
673 goto try_again;
674 }
675
676 assert( pCur != nullptr );
677 int nCmp = cmp( *node_traits::to_value_ptr( *pCur ), val );
678 if ( nCmp >= 0 ) {
679 pos.pPrev = pPrev;
680 pos.pCur = pCur;
681 pos.pNext = pNext;
682 return nCmp == 0;
683 }
684 pPrev = &( pCur->m_pNext );
685 pCur = pNext;
686 }
