407class CpuCallable {

409public:

416 template<class F>

419 template<class A>
420 requires std::is_same_v<expr_value_t<A>, T>
421 [[nodiscard]] Var<T> operator()(A arg) const noexcept {
422 auto fb = detail::FunctionBuilder::current();
423 auto wrapper = [func = _func](T *ptr) noexcept {
424 (*func)(*ptr);
425 };
426 using wrapper_t = decltype(wrapper);
427 auto wrapper2 = [](void *wrapper, void *arg) noexcept {
428 (*reinterpret_cast<wrapper_t *>(wrapper))(reinterpret_cast<T *>(arg));
429 };

431 return def<T>(fb->call(
432 Type::of<T>(), wrapper2, [](void *wrapper_ptr) {
433 luisa::delete_with_allocator(reinterpret_cast<wrapper_t *>(wrapper_ptr));
434 },
435 wrapper_ptr, arg.expression()));
436 }
