102// Can't do template classes with Godot's object model :(
103class SpineObjectWrapper : public REFCOUNTED {
104 GDCLASS(SpineObjectWrapper, REFCOUNTED)
105
106 Object *spine_owner;
107 void *spine_object;
108
109protected:
110 static void _bind_methods() {
111 ClassDB::bind_method(D_METHOD("_internal_spine_objects_invalidated"), &SpineObjectWrapper::spine_objects_invalidated);
112 }
113
114 void spine_objects_invalidated() {
115 spine_object = nullptr;
116#if VERSION_MAJOR > 3
117 spine_owner->disconnect(SNAME("_internal_spine_objects_invalidated"), callable_mp(this, &SpineObjectWrapper::spine_objects_invalidated));
118#else
119 spine_owner->disconnect(SNAME("_internal_spine_objects_invalidated"), this, SNAME("_internal_spine_objects_invalidated"));
120#endif
121 }
122
123 SpineObjectWrapper() : spine_owner(nullptr), spine_object(nullptr) {
124 }
125
126 template<typename OWNER, typename OBJECT>
127 void _set_spine_object_internal(const OWNER *_owner, OBJECT *_object) {
128 if (spine_owner) {
129 ERR_PRINT("Owner already set.");
130 return;
131 }
132 if (spine_object) {
133 ERR_PRINT("Object already set.");
134 return;
135 }
136 if (!_owner) {
137 ERR_PRINT("Owner must not be null.");
138 return;
139 }
140
141 spine_owner = (Object *) _owner;
142 spine_object = _object;
143#if VERSION_MAJOR > 3
144 spine_owner->connect(SNAME("_internal_spine_objects_invalidated"), callable_mp(this, &SpineObjectWrapper::spine_objects_invalidated));
145#else
146 spine_owner->connect(SNAME("_internal_spine_objects_invalidated"), this, SNAME("_internal_spine_objects_invalidated"));
147#endif
148 }
149
150 void *_get_spine_object_internal() { return spine_object; }
151 void *_get_spine_owner_internal() { return spine_owner; }
152};
