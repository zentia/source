93static inline ticks ticks_wait(ticks nticks) {
94#if defined(__linux__) && defined(FF_ESAVER)
95 waitSleep(nticks);
96 return 0;
97#else
98 ticks delta;
99 ticks t0 = getticks();
100 do { delta = (getticks()) - t0; } while (delta < nticks);
101 return delta-nticks;
102#endif

137static inline int memory_Stats(const std::string &status, size_t &vm, size_t &vmp) {
138#if defined(__linux__)
139 std::string line;
140 std::ifstream f;
141 f.open(status.c_str());
142 if (!f.is_open()) return -1;
143 vm = vmp = 0;
144 while (!vm || !vmp) {
145 getline(f, line);
146 if (line.compare(0,7,"VmPeak:") == 0) {
147 /* get rid of " kB" */
148 const std::string &s = line.substr(7);
149 vmp = std::stol(s.substr(0, s.length()-3));
150 } else
151 if (line.compare(0,7,"VmSize:") == 0) {
152 /* get rid of " kB"*/
153 const std::string &s = line.substr(7);
154 vm = std::stol(s.substr(0, s.length()-3));
155 }
156 }
157 f.close();
158 return 0;
159#else
160 error("memory_Stats, not implemented for this platform\n");
161 return -1;
162#endif

175static inline double diffmsec(const struct timeval & a,
176 const struct timeval & b) {
177 long sec = (a.tv_sec - b.tv_sec);
178 long usec = (a.tv_usec - b.tv_usec);
179
180 if(usec < 0) {
181 --sec;
182 usec += 1000000;
183 }
184 return ((double)(sec*1000)+ ((double)usec)/1000.0);
185}

231static inline double ffTime(int tag/*, bool lock=false*/) {
232 static struct timeval tv_start = {0,0};
233 static struct timeval tv_stop = {0,0};
234 // needed to protect gettimeofday
235 // if multiple threads call ffTime
236#if (__cplusplus >= 201103L) || (defined __GXX_EXPERIMENTAL_CXX0X__) || (defined(HAS_CXX11_VARIADIC_TEMPLATES))
237 static lock_t L;
238#else
239 static lock_t L = {0};
240#endif
241
242 double res=0.0;
243 switch(tag) {
244 case START_TIME:{
245 spin_lock(L);
246 gettimeofday(&tv_start,NULL);
247 spin_unlock(L);
248 } break;
249 case STOP_TIME:{
250 spin_lock(L);
251 gettimeofday(&tv_stop,NULL);
252 spin_unlock(L);
253 res = diffmsec(tv_stop,tv_start);
254 } break;
255 case GET_TIME: {
256 res = diffmsec(tv_stop,tv_start);
257 } break;
258 default:
259 res=0;
260 }
261 return res;
262}
