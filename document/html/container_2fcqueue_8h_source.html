321 void fc_apply( fc_record * pRec )
322 {
323 assert( pRec );
324
325 switch ( pRec->op()) {
326 case op_enq:
327 assert( pRec->pValEnq );
328 m_Queue.push( *(pRec->pValEnq ));
329 break;
330 case op_enq_move:
331 assert( pRec->pValEnq );
332 m_Queue.push( std::move( *(pRec->pValEnq )));
333 break;
334 case op_deq:
335 assert( pRec->pValDeq );
336 pRec->bEmpty = m_Queue.empty();
337 if ( !pRec->bEmpty ) {
338 *(pRec->pValDeq) = std::move( m_Queue.front());
339 m_Queue.pop();
340 }
341 break;
342 case op_clear:
343 while ( !m_Queue.empty())
344 m_Queue.pop();
345 break;
346 default:
347 assert(false);
348 break;
349 }
350 }

353 void fc_process( typename fc_kernel::iterator itBegin, typename fc_kernel::iterator itEnd )
354 {
355 typedef typename fc_kernel::iterator fc_iterator;
356
357 for ( fc_iterator it = itBegin, itPrev = itEnd; it != itEnd; ++it ) {
358 switch ( it->op( atomics::memory_order_acquire )) {
359 case op_enq:
360 case op_enq_move:
361 case op_deq:
362 if ( m_Queue.empty()) {
363 if ( itPrev != itEnd && collide( *itPrev, *it ))
364 itPrev = itEnd;
365 else
366 itPrev = it;
367 }
368 break;
369 }
370 }
371 }
372 //@endcond
373
374 private:
375 //@cond
376 bool collide( fc_record& rec1, fc_record& rec2 )
377 {
378 switch ( rec1.op()) {
379 case op_enq:
380 if ( rec2.op() == op_deq ) {
381 assert(rec1.pValEnq);
382 assert(rec2.pValDeq);
383 *rec2.pValDeq = *rec1.pValEnq;
384 rec2.bEmpty = false;
385 goto collided;
386 }
387 break;
388 case op_enq_move:
389 if ( rec2.op() == op_deq ) {
390 assert(rec1.pValEnq);
391 assert(rec2.pValDeq);
392 *rec2.pValDeq = std::move( *rec1.pValEnq );
393 rec2.bEmpty = false;
394 goto collided;
395 }
396 break;
397 case op_deq:
398 switch ( rec2.op()) {
399 case op_enq:
400 case op_enq_move:
401 return collide( rec2, rec1 );
402 }
403 }
404 return false;
405
406 collided:
407 m_FlatCombining.operation_done( rec1 );
408 m_FlatCombining.operation_done( rec2 );
409 m_FlatCombining.internal_statistics().onCollide();
410 return true;
411 }
412 //@endcond
