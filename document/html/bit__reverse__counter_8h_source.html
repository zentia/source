32 counter_type inc()
33 {
34 ++m_nCounter;
35 int nBit;
36 for ( nBit = m_nHighBit - 1; nBit >= 0; --nBit ) {
37 if ( !cds::bitop::complement( m_nReversed, nBit ))
38 break;
39 }
40 if ( nBit < 0 ) {
41 m_nReversed = m_nCounter;
42 ++m_nHighBit;
43 }
44 return m_nReversed;
45 }

47 counter_type dec()
48 {
49 counter_type ret = m_nReversed;
50 --m_nCounter;
51 int nBit;
52 for ( nBit = m_nHighBit - 1; nBit >= 0; --nBit ) {
53 if ( cds::bitop::complement( m_nReversed, nBit ))
54 break;
55 }
56 if ( nBit < 0 ) {
57 m_nReversed = m_nCounter;
58 --m_nHighBit;
59 }
60 return ret;
61 }
