166/*
167 * Converts a UTF-8 string to UTF-16. Acts like strdup. The string must be freed afterwards.
168 * This version allows keeping the output length.
169 */
170static wchar_t* XSUM_widenString(const char* str, int* lenOut)
171{
172 int const len = MultiByteToWideChar(CP_UTF8, 0, str, -1, NULL, 0);
173 if (lenOut != NULL) *lenOut = len;
174 if (len == 0) return NULL;
175 { wchar_t* buf = (wchar_t*)malloc((size_t)len * sizeof(wchar_t));
176 if (buf != NULL) {
177 if (MultiByteToWideChar(CP_UTF8, 0, str, -1, buf, len) == 0) {
178 free(buf);
179 return NULL;
180 } }
181 return buf;
182 }
183}

185/*
186 * Converts a UTF-16 string to UTF-8. Acts like strdup. The string must be freed afterwards.
187 * This version allows keeping the output length.
188 */
189static char* XSUM_narrowString(const wchar_t *str, int *lenOut)
190{
191 int len = WideCharToMultiByte(CP_UTF8, 0, str, -1, NULL, 0, NULL, NULL);
192 if (lenOut != NULL) *lenOut = len;
193 if (len == 0) return NULL;
194 { char* const buf = (char*)malloc((size_t)len * sizeof(char));
195 if (buf != NULL) {
196 if (WideCharToMultiByte(CP_UTF8, 0, str, -1, buf, len, NULL, NULL) == 0) {
197 free(buf);
198 return NULL;
199 } }
200 return buf;
201 }

209/*
210 * fopen wrapper that supports UTF-8
211 *
212 * fopen will only accept ANSI filenames, which means that we can't open Unicode filenames.
213 *
214 * In order to open a Unicode filename, we need to convert filenames to UTF-16 and use _wfopen.
215 */
216XSUM_API FILE* XSUM_fopen(const char* filename, const char* mode)
217{
218 FILE* f = NULL;
219 wchar_t* const wide_filename = XSUM_widenString(filename, NULL);
220 if (wide_filename != NULL) {
221 wchar_t* const wide_mode = XSUM_widenString(mode, NULL);
222 if (wide_mode != NULL) {
223 f = _wfopen(wide_filename, wide_mode);
224 free(wide_mode);
225 }
226 free(wide_filename);
227 }
228 return f;
229}

257XSUM_ATTRIBUTE((__format__(__printf__, 2, 0)))
258static int XSUM_vasprintf(char** strp, const char* format, va_list ap)
259{
260 int size;
261 va_list copy;
262 /*
263 * To be safe, make a va_copy.
264 *
265 * Note that Microsoft doesn't use va_copy in its sample code:
266 * https://docs.microsoft.com/en-us/cpp/c-runtime-library/reference/vsprintf-vsprintf-l-vswprintf-vswprintf-l-vswprintf-l?view=vs-2019
267 */
268 XSUM_va_copy(copy, ap);
269 /* Calculate how many characters we need */
270 size = _vscprintf(format, ap);
271 va_end(copy);
272
273 if (size < 0) {
274 *strp = NULL;
275 return size;
276 } else {
277 int ret;
278 *strp = (char*) malloc((size_t)size + 1);
279 if (*strp == NULL) {
280 return -1;
281 }
282 /* vsprintf into the new buffer */
283 ret = vsprintf(*strp, format, ap);
284 if (ret < 0) {
285 free(*strp);
286 *strp = NULL;
287 }
288 return ret;
289 }
290}

292/*
293 * fprintf wrapper that supports UTF-8.
294 *
295 * fprintf doesn't properly handle Unicode on Windows.
296 *
297 * Additionally, it is codepage sensitive on console and may crash the program.
298 *
299 * Instead, we use vsnprintf, and either print with fwrite or convert to UTF-16
300 * for console output and use the codepage-independent WriteConsoleW.
301 *
302 * Credit to t-mat: https://github.com/t-mat/xxHash/commit/5691423
303 */
304XSUM_ATTRIBUTE((__format__(__printf__, 2, 0)))
305XSUM_API int XSUM_vfprintf(FILE *stream, const char *format, va_list ap)
306{
307 int result;
308 char* u8_str = NULL;
309
310 /*
311 * Generate the UTF-8 output string with vasprintf.
312 */
313 result = XSUM_vasprintf(&u8_str, format, ap);
314
315 if (result >= 0) {
316 const size_t nchar = (size_t)result + 1;
317
318 /*
319 * Check if we are outputting to a console. Don't use XSUM_isConsole
320 * directly -- we don't need to call _get_osfhandle twice.
321 */
322 int fileNb = _fileno(stream);
323 intptr_t handle_raw = _get_osfhandle(fileNb);
324 HANDLE handle = (HANDLE)handle_raw;
325 DWORD dwTemp;
326
327 if (handle_raw < 0) {
328 result = -1;
329 } else if (_isatty(fileNb) && GetConsoleMode(handle, &dwTemp)) {
330 /*
331 * Convert to UTF-16 and output with WriteConsoleW.
332 *
333 * This is codepage independent and works on Windows XP's default
334 * msvcrt.dll.
335 */
336 int len;
337 wchar_t* const u16_buf = XSUM_widenString(u8_str, &len);
338 if (u16_buf == NULL) {
339 result = -1;
340 } else {
341 if (WriteConsoleW(handle, u16_buf, (DWORD)len - 1, &dwTemp, NULL)) {
342 result = (int)dwTemp;
343 } else {
344 result = -1;
345 }
346 free(u16_buf);
347 }
348 } else {
349 /* fwrite the UTF-8 string if we are printing to a file */
350 result = (int)fwrite(u8_str, 1, nchar - 1, stream);
351 if (result == 0) {
352 result = -1;
353 }
354 }
355 free(u8_str);
356 }
357 return result;
358}

365/* Converts a UTF-16 argv to UTF-8. */
366static char** XSUM_convertArgv(int argc, wchar_t* utf16_argv[])
367{
368 char** const utf8_argv = (char**)malloc((size_t)(argc + 1) * sizeof(char*));
369 if (utf8_argv != NULL) {
370 int i;
371 for (i = 0; i < argc; i++) {
372 utf8_argv[i] = XSUM_narrowString(utf16_argv[i], NULL);
373 if (utf8_argv[i] == NULL) {
374 /* Out of memory, whoops. */
375 while (i-- > 0) {
376 free(utf8_argv[i]);
377 }
378 free(utf8_argv);
379 return NULL;
380 }
381 }
382 utf8_argv[argc] = NULL;
383 }
384 return utf8_argv;
385}
386/* Frees arguments returned by XSUM_convertArgv */
387static void XSUM_freeArgv(int argc, char** argv)
388{
389 int i;
390 if (argv == NULL) {
391 return;
392 }
393 for (i = 0; i < argc; i++) {
394 free(argv[i]);
395 }
396 free(argv);
397}

399static int XSUM_wmain(int argc, wchar_t* utf16_argv[])
400{
401 /* Convert the UTF-16 arguments to UTF-8. */
402 char** utf8_argv = XSUM_convertArgv(argc, utf16_argv);
403
404 if (utf8_argv == NULL) {
405 /* An unfortunate but incredibly unlikely error. */
406 fprintf(stderr, "xxhsum: error converting command line arguments!\n");
407 abort();
408 } else {
409 int ret;
410
411 /*
412 * MinGW's terminal uses full block buffering for stderr.
413 *
414 * This is nonstandard behavior and causes text to not display until
415 * the buffer fills.
416 *
417 * `setvbuf()` can easily correct this to make text display instantly.
418 */
419 setvbuf(stderr, NULL, _IONBF, 0);
420
421 /* Call our real main function */
422 ret = XSUM_main(argc, (void*)utf8_argv);
423
424 /* Cleanup */
425 XSUM_freeArgv(argc, utf8_argv);
426 return ret;
427 }
428}

445/*
446 * Wrap `XSUM_wmain()` using `main()` and `__wgetmainargs()` on MinGW without
447 * Unicode support.
448 *
449 * `__wgetmainargs()` is used in the CRT startup to retrieve the arguments for
450 * `wmain()`, so we use it on MinGW to emulate `wmain()`.
451 *
452 * It is an internal function and not declared in any public headers, so we
453 * have to declare it manually.
454 *
455 * An alternative that doesn't mess with internal APIs is `GetCommandLineW()`
456 * with `CommandLineToArgvW()`, but the former doesn't expand wildcards and the
457 * latter requires linking to Shell32.dll and its numerous dependencies.
458 *
459 * This method keeps our dependencies to kernel32.dll and the CRT.
460 *
461 * https://docs.microsoft.com/en-us/cpp/c-runtime-library/getmainargs-wgetmainargs?view=vs-2019
462 */

464 int newmode;
465} _startupinfo;

468extern "C"
469#endif
470int __cdecl __wgetmainargs(
471 int* Argc,
472 wchar_t*** Argv,
473 wchar_t*** Env,
474 int DoWildCard,
475 _startupinfo* StartInfo

480 int utf16_argc;
