46template<typename Value, typename Node, typename Allocator>
47class node_handle_base {
48public:
49 typedef Allocator allocator_type;
50protected:
51 typedef Node node;
52 typedef tbb::internal::allocator_traits<allocator_type> traits_type;
53public:
54
55 node_handle_base() : my_node(NULL), my_allocator() {}
56 node_handle_base(node_handle_base&& nh) : my_node(nh.my_node),
57 my_allocator(std::move(nh.my_allocator)) {
58 nh.my_node = NULL;
59 }
60
61 bool empty() const { return my_node == NULL; }
62 explicit operator bool() const { return my_node != NULL; }
63
64 ~node_handle_base() { internal_destroy(); }
65
66 node_handle_base& operator=(node_handle_base&& nh) {
67 internal_destroy();
68 my_node = nh.my_node;
69 typedef typename traits_type::propagate_on_container_move_assignment pocma_type;
70 tbb::internal::allocator_move_assignment(my_allocator, nh.my_allocator, pocma_type());
71 nh.deactivate();
72 return *this;
73 }
74
75 void swap(node_handle_base& nh) {
76 std::swap(my_node, nh.my_node);
77 typedef typename traits_type::propagate_on_container_swap pocs_type;
78 tbb::internal::allocator_swap(my_allocator, nh.my_allocator, pocs_type());
79 }
80
81 allocator_type get_allocator() const {
82 return my_allocator;
83 }
84
85protected:
86 node_handle_base(node* n) : my_node(n) {}
87
88 void internal_destroy() {
89 if(my_node) {
90 traits_type::destroy(my_allocator, my_node->storage());
91 typename tbb::internal::allocator_rebind<allocator_type, node>::type node_allocator;
92 node_allocator.deallocate(my_node, 1);
93 }
94 }
95
96 void deactivate() { my_node = NULL; }
97
98 node* my_node;
99 allocator_type my_allocator;
100};

103template<typename Key, typename Value, typename Node, typename Allocator>
104class node_handle : public node_handle_base<Value, Node, Allocator> {
105 typedef node_handle_base<Value, Node, Allocator> base_type;
106public:
107 typedef Key key_type;
108 typedef typename Value::second_type mapped_type;
109 typedef typename base_type::allocator_type allocator_type;
110
111 node_handle() : base_type() {}
112
113 key_type& key() const {
114 __TBB_ASSERT(!this->empty(), "Cannot get key from the empty node_type object");
115 return *const_cast<key_type*>(&(this->my_node->value().first));
116 }
117
118 mapped_type& mapped() const {
119 __TBB_ASSERT(!this->empty(), "Cannot get mapped value from the empty node_type object");
120 return this->my_node->value().second;
121 }
122
123private:
124 template<typename T, typename A>
125 friend class tbb::interface5::internal::split_ordered_list;
126
127 template<typename Traits>
128 friend class tbb::interface5::internal::concurrent_unordered_base;
129
130 template<typename Traits>
131 friend class tbb::interface10::internal::concurrent_skip_list;
132
133 node_handle(typename base_type::node* n) : base_type(n) {}
134};
