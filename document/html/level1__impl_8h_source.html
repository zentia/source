30{

93 *s = 0;
94 r = 0;
95 z = 0;
96 }
97 else

99 r = sqrt(a*a + b*b);
100 Scalar amax = aa>ab ? a : b;
101 r = amax>0 ? r : -r;
102 *c = a/r;
103 *s = b/r;
104 z = 1;
105 if (aa > ab) z = *s;
106 if (ab > aa && *c!=RealScalar(0))
107 z = Scalar(1)/ *c;

110 *pb = z;
111 #else

117 *s = Scalar(1);
118 a = b;
119 }
120 else
121 {
122 scale = abs(a) + abs(b);
123 norm = scale*sqrt((numext::abs2(a/scale)) + (numext::abs2(b/scale)));
124 alpha = a/abs(a);
125 *c = abs(a)/norm;
126 *s = alpha*numext::conj(b)/norm;
