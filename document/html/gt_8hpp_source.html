197 virtual ssize_t gather_task(void ** task) {
198 unsigned int cnt;
199 do {
200 cnt=0;
201 do {
202 nextr = selectworker();
203 //assert(offline[nextr]==false);
204 if (workers[nextr]->get(task)) {
205 return nextr;
206 }
207 else if (++cnt == nattempts()) break;
208 } while(1);
209 if (blocking_in) {
210 struct timespec tv;
211 timedwait_timeout(tv);
212 pthread_mutex_lock(cons_m);
213 pthread_cond_timedwait(cons_c, cons_m, &tv);
214 pthread_mutex_unlock(cons_m);
215 } else losetime_in();
216 } while(1);
217 return -1;
218 }

225 inline bool push(void * task, unsigned long retry=((unsigned long)-1), unsigned long ticks=(TICKS2WAIT)) {
226 if (blocking_out) {
227 if (!filter) {
228 bool empty=buffer->empty();
229 while(!buffer->push(task)) {
230 empty = false;
231 struct timespec tv;
232 timedwait_timeout(tv);
233 pthread_mutex_lock(prod_m);
234 pthread_cond_timedwait(prod_c,prod_m, &tv);
235 pthread_mutex_unlock(prod_m);
236 }
237 if (empty) pthread_cond_signal(p_cons_c);
238 } else {
239 bool empty=filter->get_out_buffer()->empty();
240 while(!filter->push(task)) {
241 empty=false;
242 struct timespec tv;
243 timedwait_timeout(tv);
244 pthread_mutex_lock(prod_m);
245 pthread_cond_timedwait(prod_c,prod_m,&tv);
246 pthread_mutex_unlock(prod_m);
247 }
248 if (empty) pthread_cond_signal(p_cons_c);
249 }
250 return true;
251 }
252 if (!filter) {
253 for(unsigned long i=0;i<retry;++i) {
254 if (buffer->push(task)) return true;
255 losetime_out(ticks);
256 }
257 } else
258 for(unsigned long i=0;i<retry;++i) {
259 if (filter->push(task)) return true;
260 losetime_out();
261 }
262 return false;
263 }

563 virtual void * svc(void *) {
564 void * ret = FF_EOS;
565 void * task = NULL;
566 bool filter_outpresent = false;
567 bool outpresent = (buffer != NULL);
568 bool skipfirstpop = skip1pop;
569
570 // the following case is possible when the there is a filter that is a composition
571 if ( filter &&
572 ( (filter->get_out_buffer()!=NULL) || filter->isMultiOutput() ) ) {
573
574 filter_outpresent=true;
575 //set_out_buffer(filter->get_in_buffer());
576 }
577
578
579 // it is possible that a standard node has been automatically (and transparently)
580 // transformed into a multi-input node (see mi_transformer).
581 // In this case we want to call notifyeos only when we have received EOS from all
582 // input channel.
583 bool notify_each_eos = filter ? (filter->neos==1): false;
584
585 // TODO: skipallpop missing!
586
587 gettimeofday(&wtstart,NULL);
588 do {
589 task = NULL;
590 if (!skipfirstpop)
591 nextr = gather_task(&task);
592 else skipfirstpop=false;
593
594 if (task == FF_GO_ON) continue;
595 channelid = (nextr-feedbackid);
596 frominput=true;
597 if (feedbackid>0) { // there are feedback channels
598 if (nextr<feedbackid) {
599 frominput=false;
600 channelid=nextr;
601 }
602 }
603
604 if ((task == FF_EOS) || (task == FF_EOSW)) {
605 if (filter && notify_each_eos)
606 filter->eosnotify(channelid); //workers[nextr]->get_my_id());
607 offline[nextr]=true;
608 ++neos;
609 ret=task;
610 } else if (task == FF_EOS_NOFREEZE) {
611 if (filter && notify_each_eos)
612 filter->eosnotify(channelid); //workers[nextr]->get_my_id());
613 offline[nextr]=true;
614 ++neosnofreeze;
615 ret = task;
616 } else {
617 FFTRACE(++taskcnt);
618 if (filter) {
619 FFTRACE(ticks t0 = getticks());
620
621#if defined(FF_TASK_CALLBACK)
622 if (filter) callbackIn(this);
623#endif
624 task = filter->svc(task);
625
626#if defined(TRACE_FASTFLOW)
627 ticks diff=(getticks()-t0);
628 tickstot +=diff;
629 ticksmin=(std::min)(ticksmin,diff);
630 ticksmax=(std::max)(ticksmax,diff);
631#endif
632 }
633
634 // if the filter returns EOS or GO_OUT we exit immediatly
635 if (task == FF_GO_ON) continue;
636 if ((task == FF_GO_OUT) || (task == FF_EOS_NOFREEZE) || (task == FF_EOSW) ) {
637 ret = task;
638 break; // exiting from the loop without sending the task
639 }
640 if (!task || (task == FF_EOS)) {
641 ret = FF_EOS;
642 break;
643 }
644
645 if (filter_outpresent) filter->ff_send_out(task);
646 else if (outpresent) push(task);
647#if defined(FF_TASK_CALLBACK)
648 else
649 if (filter) callbackOut(this);
650#endif
651 }
652 } while((neos<(size_t)running) && (neosnofreeze<(size_t)running));
653
654 // GO_OUT, EOS_NOFREEZE and EOSW are not propagated !
655 if (ret == FF_EOS) {
656 // we notify the filter only when we have received all EOSs
657 if (!notify_each_eos && filter) filter->eosnotify();
658
659 // push EOS
660 task = ret;
661 if (filter_outpresent) {
662 if (filter->isMultiOutput()) filter->propagateEOS();
663 else filter->ff_send_out(FF_EOS);
664 } else
665 if (outpresent) push(FF_EOS);
666 }
667 if (ret == FF_EOSW) ret = FF_EOS; // EOSW is like an EOS but it is not propagated
668
669 gettimeofday(&wtstop,NULL);
670 wttime+=diffmsec(wtstop,wtstart);
671 if (neos>=(size_t)running) neos=0;
672 if (neosnofreeze>=(size_t)running) neosnofreeze=0;
673
674 return ret;
675 }

744 virtual int all_gather(void *task, void **V) {
745 if (ag_callback) return ag_callback(task,V,ag_callback_arg);
746
747 V[channelid]=task;
748 size_t nw=getnworkers();
749 svector<ff_node*> _workers(nw);
750 for(ssize_t i=0;i<running;++i) {
751 if (!offline[i]) _workers.push_back(workers[i]);
752 else _workers.push_back(nullptr);
753 }
754 svector<size_t> retry(nw);
755
756 for(ssize_t i=0;i<running;++i) {
757 if(i != channelid) {
758 if (_workers[i]) {
759 if (!_workers[i]->get(&V[i])) retry.push_back(i);
760 }
761 }
762 }
763 while(retry.size()) {
764 channelid = retry.back();
765 if(_workers[channelid]->get(&V[channelid])) {
766 retry.pop_back();
767 }
768 else {
769 if (blocking_in) {
770 struct timespec tv;
771 timedwait_timeout(tv);
772 pthread_mutex_lock(cons_m);
773 pthread_cond_timedwait(cons_c, cons_m, &tv);
774 pthread_mutex_unlock(cons_m);
775 } else losetime_in();
776 }
777 }
778 bool eos=false;
779 for(ssize_t i=0;i<running;++i) {
780 if (V[i] == FF_EOS) {
781 eos=true;
782 ++neos;
783 V[i]=nullptr;
784 FFTRACE(taskcnt--);
785 } else
786 if (V[i] == FF_EOS_NOFREEZE) {
787 eos=true;
788 ++neosnofreeze;
789 V[i]=nullptr;
790 FFTRACE(taskcnt--);
791 }
792 }
793 FFTRACE(taskcnt+=nw-1);
794 return eos?-1:0;
795 }
