247// The page flags are put in the bottom 2 bits of the thread_id (for a fast test in `mi_free`)
248// `has_aligned` is true if the page has pointers at an offset in a block (so we unalign before free-ing)
249// `in_full_queue` is true if the page is full and resides in the full queue (so we move it to a regular queue on free-ing)
250#define MI_PAGE_IN_FULL_QUEUE MI_ZU(0x01)

255// There are two special threadid's: 0 for abandoned threads, and 4 for abandoned & mapped threads --
256// abandoned-mapped pages are abandoned but also mapped in an arena so can be quickly found for reuse.
257#define MI_THREADID_ABANDONED MI_ZU(0)
258#define MI_THREADID_ABANDONED_MAPPED (MI_PAGE_FLAG_MASK + 1)

260// Thread free list.
261// Points to a list of blocks that are freed by other threads.
262// The least-bit is set if the page is owned by the current thread. (`mi_page_is_owned`).
263// Ownership is required before we can read any non-atomic fields in the page.
264// This way we can push a block on the thread free list and try to claim ownership atomically in `free.c:mi_free_block_mt`.
265typedef uintptr_t mi_thread_free_t;
266
267// A heap can serve only specific objects signified by its heap tag (e.g. various object types in CPython)
268typedef uint8_t mi_heaptag_t;

270// A page contains blocks of one specific size (`block_size`).
271// Each page has three list of free blocks:
272// `free` for blocks that can be allocated,
273// `local_free` for freed blocks that are not yet available to `mi_malloc`
274// `thread_free` for freed blocks by other threads
275// The `local_free` and `thread_free` lists are migrated to the `free` list
276// when it is exhausted. The separate `local_free` list is necessary to
277// implement a monotonic heartbeat. The `thread_free` list is needed for
278// avoiding atomic operations when allocating from the owning thread.
279//
280// `used - |thread_free|` == actual blocks that are in use (alive)
281// `used - |thread_free| + |free| + |local_free| == capacity`
282//
283// We don't count "freed" (as |free|) but use only the `used` field to reduce
284// the number of memory accesses in the `mi_page_all_free` function(s).
285// Use `_mi_page_free_collect` to collect the thread_free list and update the `used` count.
286//
287// Notes:
288// - Non-atomic fields can only be accessed if having _ownership_ (low bit of `xthread_free` is 1).
289// Combining the `thread_free` list with an ownership bit allows a concurrent `free` to atomically
290// free an object and (re)claim ownership if the page was abandoned.
291// - If a page is not part of a heap it is called "abandoned" (`heap==NULL`) -- in
292// that case the `xthreadid` is 0 or 4 (4 is for abandoned pages that
293// are in the abandoned page lists of an arena, these are called "mapped" abandoned pages).
294// - page flags are in the bottom 3 bits of `xthread_id` for the fast path in `mi_free`.
295// - The layout is optimized for `free.c:mi_free` and `alloc.c:mi_page_alloc`
296// - Using `uint16_t` does not seem to slow things down
297
298typedef struct mi_page_s {
299 _Atomic(mi_threadid_t) xthread_id; // thread this page belongs to. (= `heap->thread_id (or 0 or 4 if abandoned) | page_flags`)
300
301 mi_block_t* free; // list of available free blocks (`malloc` allocates from this list)
302 uint16_t used; // number of blocks in use (including blocks in `thread_free`)
303 uint16_t capacity; // number of blocks committed
304 uint16_t reserved; // number of blocks reserved in memory
305 uint8_t block_size_shift; // if not zero, then `(1 << block_size_shift) == block_size` (only used for fast path in `free.c:_mi_page_ptr_unalign`)
306 uint8_t retire_expire; // expiration count for retired blocks
307
308 mi_block_t* local_free; // list of deferred free blocks by this thread (migrates to `free`)
309 _Atomic(mi_thread_free_t) xthread_free; // list of deferred free blocks freed by other threads (= `mi_block_t* | (1 if owned)`)
310
311 size_t block_size; // size available in each block (always `>0`)
312 uint8_t* page_start; // start of the blocks
313 mi_heaptag_t heap_tag; // tag of the owning heap, used to separate heaps by object type
314 bool free_is_zero; // `true` if the blocks in the free list are zero initialized
315 // padding
316 #if (MI_ENCODE_FREELIST || MI_PADDING)
317 uintptr_t keys[2]; // two random keys to encode the free lists (see `_mi_block_next`) or padding canary
318 #endif
319
320 mi_heap_t* heap; // the heap owning this page (or NULL for abandoned pages)
321 struct mi_page_s* next; // next page owned by the heap with the same `block_size`
322 struct mi_page_s* prev; // previous page owned by the heap with the same `block_size`
323 size_t slice_committed; // committed size relative to the first arena slice of the page data (or 0 if the page is fully committed already)
324 mi_memid_t memid; // provenance of the page memory
325} mi_page_t;
