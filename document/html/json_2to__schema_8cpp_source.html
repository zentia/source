42std::string numeric_type_to_string(const parsing::schema::Type& _type) {
43 const auto handle_variant = [](const auto& _t) -> std::string {
44 using T = std::remove_cvref_t<decltype(_t)>;
45 using Type = parsing::schema::Type;
46 if constexpr (std::is_same<T, Type::Int32>() ||
47 std::is_same<T, Type::Int64>() ||
48 std::is_same<T, Type::UInt32>() ||
49 std::is_same<T, Type::UInt64>() ||
50 std::is_same<T, Type::Integer>()) {
51 return schema::Type::Integer{}.type.str();
52 } else {
53 return schema::Type::Number{}.type.str();
54 }
55 };
56 return rfl::visit(handle_variant, _type.variant_);
57}

59schema::Type handle_validation_type(
60 const parsing::schema::Type& _type,
61 const parsing::schema::ValidationType& _validation_type,
62 const bool _no_required) {
63 auto handle_variant = [&](const auto& _v) -> schema::Type {
64 using T = std::remove_cvref_t<decltype(_v)>;
65 using ValidationType = parsing::schema::ValidationType;
66 if constexpr (std::is_same<T, ValidationType::AllOf>()) {
67 auto all_of = std::vector<schema::Type>();
68 for (const auto& t : _v.types_) {
69 all_of.emplace_back(handle_validation_type(_type, t, _no_required));
70 }
71 return schema::Type{.value = schema::Type::AllOf{.allOf = all_of}};
72
73 } else if constexpr (std::is_same<T, ValidationType::AnyOf>()) {
74 auto any_of = std::vector<schema::Type>();
75 for (const auto& t : _v.types_) {
76 any_of.emplace_back(handle_validation_type(_type, t, _no_required));
77 }
78 return schema::Type{.value = schema::Type::AnyOf{.anyOf = any_of}};
79
80 } else if constexpr (std::is_same<T, ValidationType::OneOf>()) {
81 auto one_of = std::vector<schema::Type>();
82 for (const auto& t : _v.types_) {
83 one_of.emplace_back(handle_validation_type(_type, t, _no_required));
84 }
85 return schema::Type{.value = schema::Type::OneOf{.oneOf = one_of}};
86
87 } else if constexpr (std::is_same<T, ValidationType::Regex>()) {
88 return schema::Type{.value = schema::Type::Regex{.pattern = _v.pattern_}};
89
90 } else if constexpr (std::is_same<T, ValidationType::Size>()) {
91 auto t = type_to_json_schema_type(_type, _no_required);
92 const auto to_size = [](const auto _v) {
93 return static_cast<size_t>(_v);
94 };
95 auto handle_size_variant = [&](auto& _t, const auto& _size_limit) {
96 using U = std::remove_cvref_t<decltype(_t)>;
97 using V = std::remove_cvref_t<decltype(_size_limit)>;
98 if constexpr (std::is_same<U, schema::Type::TypedArray>() ||
99 std::is_same<U, schema::Type::String>()) {
100 if constexpr (std::is_same<V, ValidationType::Minimum>()) {
101 _t.minSize = _size_limit.value_.visit(to_size);
102 return t;
103
104 } else if constexpr (std::is_same<V, ValidationType::Maximum>()) {
105 _t.maxSize = _size_limit.value_.visit(to_size);
106 return t;
107
108 } else if constexpr (std::is_same<V, ValidationType::EqualTo>()) {
109 _t.minSize = _size_limit.value_.visit(to_size);
110 _t.maxSize = _size_limit.value_.visit(to_size);
111 return t;
112
113 } else if constexpr (std::is_same<V, ValidationType::AnyOf>() ||
114 std::is_same<V, ValidationType::AllOf>()) {
115 V v;
116 for (const auto& limiter : _size_limit.types_) {
117 v.types_.push_back(ValidationType{ValidationType::Size{
118 .size_limit_ = rfl::Ref<ValidationType>::make(limiter)}});
119 }
120 return handle_validation_type(_type, ValidationType{.variant_ = v},
121 _no_required);
122 }
123 }
124 return t;
125 };
126
127 return rfl::visit(handle_size_variant, t.value, _v.size_limit_->variant_);
128
129 } else if constexpr (std::is_same<T, ValidationType::ExclusiveMaximum>()) {
130 return schema::Type{.value = schema::Type::ExclusiveMaximum{
131 .exclusiveMaximum = _v.value_,
132 .type = numeric_type_to_string(_type)}};
133
134 } else if constexpr (std::is_same<T, ValidationType::ExclusiveMinimum>()) {
135 return schema::Type{.value = schema::Type::ExclusiveMinimum{
136 .exclusiveMinimum = _v.value_,
137 .type = numeric_type_to_string(_type)}};
138
139 } else if constexpr (std::is_same<T, ValidationType::Maximum>()) {
140 return schema::Type{
141 .value = schema::Type::Maximum{
142 .maximum = _v.value_, .type = numeric_type_to_string(_type)}};
143
144 } else if constexpr (std::is_same<T, ValidationType::Minimum>()) {
145 return schema::Type{
146 .value = schema::Type::Minimum{
147 .minimum = _v.value_, .type = numeric_type_to_string(_type)}};
148
149 } else if constexpr (std::is_same<T, ValidationType::EqualTo>()) {
150 const auto maximum = schema::Type{
151 .value = schema::Type::Maximum{
152 .maximum = _v.value_, .type = numeric_type_to_string(_type)}};
153 const auto minimum = schema::Type{
154 .value = schema::Type::Minimum{
155 .minimum = _v.value_, .type = numeric_type_to_string(_type)}};
156 return schema::Type{.value =
157 schema::Type::AllOf{.allOf = {maximum, minimum}}};
158
159 } else if constexpr (std::is_same<T, ValidationType::NotEqualTo>()) {
160 const auto excl_maximum =
161 schema::Type{.value = schema::Type::ExclusiveMaximum{
162 .exclusiveMaximum = _v.value_,
163 .type = numeric_type_to_string(_type)}};
164 const auto excl_minimum =
165 schema::Type{.value = schema::Type::ExclusiveMinimum{
166 .exclusiveMinimum = _v.value_,
167 .type = numeric_type_to_string(_type)}};
168 return schema::Type{
169 .value = schema::Type::AnyOf{.anyOf = {excl_maximum, excl_minimum}}};
170
171 } else {
172 static_assert(rfl::always_false_v<T>, "Not all cases were covered.");
173 }
174 };
175
176 return rfl::visit(handle_variant, _validation_type.variant_);
177}

179schema::Type type_to_json_schema_type(const parsing::schema::Type& _type,
180 const bool _no_required) {
181 auto handle_variant = [&](const auto& _t) -> schema::Type {
182 using T = std::remove_cvref_t<decltype(_t)>;
183 using Type = parsing::schema::Type;
184 if constexpr (std::is_same<T, Type::Boolean>()) {
185 return schema::Type{.value = schema::Type::Boolean{}};
186
187 } else if constexpr (std::is_same<T, Type::Int32>() ||
188 std::is_same<T, Type::Int64>() ||
189 std::is_same<T, Type::UInt32>() ||
190 std::is_same<T, Type::UInt64>() ||
191 std::is_same<T, Type::Integer>()) {
192 return schema::Type{.value = schema::Type::Integer{}};
193
194 } else if constexpr (std::is_same<T, Type::Float>() ||
195 std::is_same<T, Type::Double>()) {
196 return schema::Type{.value = schema::Type::Number{}};
197
198 } else if constexpr (std::is_same<T, Type::String>() ||
199 std::is_same<T, Type::Bytestring>()) {
200 return schema::Type{.value = schema::Type::String{}};
201
202 } else if constexpr (std::is_same<T, Type::AnyOf>()) {
203 auto any_of = std::vector<schema::Type>();
204 for (const auto& t : _t.types_) {
205 any_of.emplace_back(type_to_json_schema_type(t, _no_required));
206 }
207 return schema::Type{.value = schema::Type::AnyOf{.anyOf = any_of}};
208
209 } else if constexpr (std::is_same<T, Type::Description>()) {
210 auto res = type_to_json_schema_type(*_t.type_, _no_required);
211 const auto update_prediction = [&](auto _v) -> schema::Type {
212 _v.description = _t.description_;
213 return schema::Type{_v};
214 };
215 return rfl::visit(update_prediction, res.value);
216
217 } else if constexpr (std::is_same<T, Type::FixedSizeTypedArray>()) {
218 return schema::Type{
219 .value = schema::Type::FixedSizeTypedArray{
220 .items = Ref<schema::Type>::make(
221 type_to_json_schema_type(*_t.type_, _no_required)),
222 .minItems = _t.size_,
223 .maxItems = _t.size_}};
224
225 } else if constexpr (std::is_same<T, Type::Literal>()) {
226 return schema::Type{.value =
227 schema::Type::StringEnum{.values = _t.values_}};
228
229 } else if constexpr (std::is_same<T, Type::Object>()) {
230 auto properties = rfl::Object<schema::Type>();
231 auto required = std::vector<std::string>();
232 for (const auto& [k, v] : _t.types_) {
233 properties[k] = type_to_json_schema_type(v, _no_required);
234 if (!is_optional(v) && !_no_required) {
235 required.push_back(k);
236 }
237 }
238 auto additional_properties =
239 _t.additional_properties_
240 ? std::make_shared<schema::Type>(type_to_json_schema_type(
241 *_t.additional_properties_, _no_required))
242 : std::shared_ptr<schema::Type>();
243 return schema::Type{.value = schema::Type::Object{
244 .properties = properties,
245 .required = required,
246 .additionalProperties = additional_properties}};
247
248 } else if constexpr (std::is_same<T, Type::Optional>()) {
249 return schema::Type{
250 .value = schema::Type::AnyOf{
251 .anyOf = {type_to_json_schema_type(*_t.type_, _no_required),
252 schema::Type{schema::Type::Null{}}}}};
253
254 } else if constexpr (std::is_same<T, Type::Reference>()) {
255 return schema::Type{
256 .value = schema::Type::Reference{.ref = "#/definitions/" + _t.name_}};
257
258 } else if constexpr (std::is_same<T, Type::StringMap>()) {
259 return schema::Type{
260 .value = schema::Type::StringMap{
261 .additionalProperties = Ref<schema::Type>::make(
262 type_to_json_schema_type(*_t.value_type_, _no_required))}};
263
264 } else if constexpr (std::is_same<T, Type::Tuple>()) {
265 auto items = std::vector<schema::Type>();
266 for (const auto& t : _t.types_) {
267 items.emplace_back(type_to_json_schema_type(t, _no_required));
268 }
269 return schema::Type{.value = schema::Type::Tuple{.prefixItems = items}};
270
271 } else if constexpr (std::is_same<T, Type::TypedArray>()) {
272 return schema::Type{
273 .value = schema::Type::TypedArray{
274 .items = Ref<schema::Type>::make(
275 type_to_json_schema_type(*_t.type_, _no_required))}};
276
277 } else if constexpr (std::is_same<T, Type::Validated>()) {
278 return handle_validation_type(*_t.type_, _t.validation_, _no_required);
279
280 } else {
281 static_assert(rfl::always_false_v<T>, "Not all cases were covered.");
282 }
283 };
284
285 return rfl::visit(handle_variant, _type.variant_);
286}

288std::string to_schema_internal_schema(
289 const parsing::schema::Definition& internal_schema,
290 const yyjson_write_flag _flag, const bool _no_required) {
291 auto definitions = std::map<std::string, schema::Type>();
292 for (const auto& [k, v] : internal_schema.definitions_) {
293 definitions[k] = type_to_json_schema_type(v, _no_required);
294 }
295 using JSONSchemaType =
296 typename TypeHelper<schema::Type::ReflectionType>::JSONSchemaType;
297 const auto to_schema = [&](auto&& _root) -> JSONSchemaType {
298 using U = std::decay_t<decltype(_root)>;
299 return schema::JSONSchema<U>{.root = std::move(_root),
300 .definitions = definitions};
301 };
302 auto root = type_to_json_schema_type(internal_schema.root_, _no_required);
303 const auto json_schema = rfl::visit(to_schema, std::move(root.value));
304 return write(json_schema, _flag);
305}
