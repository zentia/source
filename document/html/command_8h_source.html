537 struct alignas(16) Modification {
538
539 // flags
540 static constexpr auto flag_primitive = 1u << 0u;
541 static constexpr auto flag_transform = 1u << 1u;
542 static constexpr auto flag_opaque_on = 1u << 2u;
543 static constexpr auto flag_opaque_off = 1u << 3u;
544 static constexpr auto flag_opaque = flag_opaque_on | flag_opaque_off;
545 static constexpr auto flag_visibility = 1u << 4u;
546 static constexpr auto flag_user_id = 1u << 5u;
547
548 // members
549 uint index{};
550 uint user_id{};
551 uint flags{};
552 uint vis_mask{};
553 float affine[12]{};
554 uint64_t primitive{};
555
556 // ctor
557 Modification() noexcept = default;
558 explicit Modification(uint index) noexcept : index{index} {}
559 // encode interfaces
560 void set_transform(float4x4 m) noexcept {
561 affine[0] = m[0][0];
562 affine[1] = m[1][0];
563 affine[2] = m[2][0];
564 affine[3] = m[3][0];
565 affine[4] = m[0][1];
566 affine[5] = m[1][1];
567 affine[6] = m[2][1];
568 affine[7] = m[3][1];
569 affine[8] = m[0][2];
570 affine[9] = m[1][2];
571 affine[10] = m[2][2];
572 affine[11] = m[3][2];
573 flags |= flag_transform;
574 }
575 void set_transform_data(const float affine_data[12]) noexcept {
576 for (auto i = 0u; i < 12u; i++) { affine[i] = affine_data[i]; }
577 flags |= flag_transform;
578 }
579 void set_visibility(uint8_t mask) noexcept {
580 vis_mask = mask;
581 flags |= flag_visibility;
582 }
583 void set_opaque(bool opaque) noexcept {
584 flags &= ~flag_opaque;// clear old visibility flags
585 flags |= opaque ? flag_opaque_on : flag_opaque_off;
586 }
587 void set_primitive(uint64_t handle) noexcept {
588 primitive = handle;
589 flags |= flag_primitive;
590 }
591 void set_user_id(uint id) noexcept {
592 user_id = id;
593 flags |= flag_user_id;
594 }
595 };
596
597private:
