6// This class implements the C++11 shared_ptr template. A shared_ptr is like
7// the C++ Standard Library unique_ptr except that it allows sharing of pointers
8// between instances via reference counting. shared_ptr objects can safely be
9// copied and can safely be used in containers such as vector or list.
10//
11// Notes regarding safe usage of shared_ptr:
12// - http://www.boost.org/doc/libs/1_53_0/libs/smart_ptr/shared_ptr.htm#ThreadSafety
13// - If you construct a shared_ptr with a raw pointer, you cannot construct another shared_ptr
14// with just that raw pointer. Insted you need to construct additional shared_ptrs with
15// the originally created shared_ptr. Otherwise you will get a crash.
16// - Usage of shared_ptr is thread-safe, but what it points to isn't automatically thread safe.
17// Multiple shared_ptrs that refer to the same object can be used arbitrarily by multiple threads.
18// - You can use a single shared_ptr between multiple threads in all ways except one: assigment
19// to that shared_ptr. The following is not thread-safe, and needs to be guarded by a mutex
20// or the shared_ptr atomic functions:
21// shared_ptr<Foo> pFoo;
22// // Thread 1:
23// shared_ptr<Foo> pFoo2 = pFoo;
24// // Thread 2:
25// pFoo = make_shared<Foo>();
26//
27// Compatibility note:
28// This version of shared_ptr updates the previous version to have full C++11
29// compatibility. However, in order to add C++11 compatibility there needed to
30// be a few breaking changes which may affect some users. It's likely that most
31// or all breaking changes can be rectified by doing the same thing in a slightly
32// different way. Here's a list of the primary signficant breaking changes:
33// - shared_ptr now takes just one template parameter instead of three.
34// (allocator and deleter). You now specify the allocator and deleter
35// as part of the shared_ptr constructor at runtime.
36// - shared_ptr has thread safety, which

154 inline void ref_count_sp::addref() EA_NOEXCEPT
155 {
156 mRefCount.fetch_add(1, memory_order_relaxed);
157 mWeakRefCount.fetch_add(1, memory_order_relaxed);
158 }

160 inline void ref_count_sp::release()
161 {
162 EASTL_ASSERT((mRefCount.load(memory_order_relaxed) > 0));
163 if(mRefCount.fetch_sub(1, memory_order_release) == 1)
164 {
165 atomic_thread_fence(memory_order_acquire);
166 free_value();
167 }
168
169 weak_release();
170 }
171
172 inline void ref_count_sp::weak_addref() EA_NOEXCEPT
173 {
174 mWeakRefCount.fetch_add(1, memory_order_relaxed);
175 }
176
177 inline void ref_count_sp::weak_release()
178 {
179 EASTL_ASSERT(mWeakRefCount.load(memory_order_relaxed) > 0);
180 if(mWeakRefCount.fetch_sub(1, memory_order_release) == 1)
181 {
182 atomic_thread_fence(memory_order_acquire);
183 free_ref_count_sp();
184 }
185 }
186
187 inline ref_count_sp* ref_count_sp::lock() EA_NOEXCEPT
188 {
189 for(int32_t refCountTemp = mRefCount.load(memory_order_relaxed); refCountTemp != 0; )
190 {
191 if(mRefCount.compare_exchange_weak(refCountTemp, refCountTemp + 1, memory_order_relaxed))
192 {
193 mWeakRefCount.fetch_add(1, memory_order_relaxed);
194 return this;
195 }
196 }
197
198 return nullptr;
199 }

1239 //
1240 // These functions allow shared_ptr to act like other C++11 atomic operations.
1241 // So the same way you can use atomic_load on a raw pointer, you can also
1242 // use it on a shared_ptr. This allows for transparent use of shared_ptr in
1243 // place of raw pointers (e.g. in templates). You do not need to use these
1244 // functions for regular thread-safe direct usage of shared_ptr construction
1245 // and copying, as it's intrinsically thread-safe for that already.
1246 //
1247 // That being said, the following is not thread-safe and needs to be guarded by
1248 // a mutex or the following atomic functions, as it's assigning the *same*
1249 // shared_ptr object from multiple threads as opposed to different shared_ptr
1250 // objects underlying object:
1251 // shared_ptr<Foo> pFoo;
1252 // // Thread 1:
1253 // shared_ptr<Foo> pFoo2 = pFoo;
1254 // // Thread 2:
1255 // pFoo = make_shared<Foo>();

1260 {
1261 // Return true if atomic access to the provided shared_ptr instance is lock-free, false otherwise.
1262 // For this to be lock-free, we would have to be able to copy shared_ptr objects in an atomic way
1263 // as opposed to wrapping it with a mutex like we do below. Given the nature of shared_ptr, it's
1264 // probably not feasible to implement these operations without a mutex. atomic_is_lock_free exists
1265 // in the C++11 Standard because it also applies to other types such as built-in types which can
1266 // be lock-free in their access.
1267 return false;
1268 }
