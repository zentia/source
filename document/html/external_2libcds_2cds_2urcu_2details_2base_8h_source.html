300 public:
301 typedef thread_data<RCUtag> thread_record;
302 typedef cds::details::Allocator< thread_record, Alloc > allocator_type;
303
304 private:
305 atomics::atomic<thread_record *> m_pHead;
306
307 public:
308 thread_list()
309 : m_pHead( nullptr )
310 {}
311
312 ~thread_list()
313 {
314 destroy();
315 }

317 thread_record * alloc()
318 {
319 thread_record * pRec;
320 cds::OS::ThreadId const nullThreadId = cds::OS::c_NullThreadId;
321 cds::OS::ThreadId const curThreadId = cds::OS::get_current_thread_id();
322
323 // First, try to reuse a retired (non-active) HP record
324 for ( pRec = m_pHead.load( atomics::memory_order_acquire ); pRec; pRec = pRec->m_list.next_ ) {
325 cds::OS::ThreadId thId = nullThreadId;
326 if ( !pRec->m_list.thread_id_.compare_exchange_strong( thId, curThreadId, atomics::memory_order_acquire, atomics::memory_order_relaxed ))
327 continue;
328 return pRec;
329 }
330
331 // No records available for reuse
332 // Allocate and push a new record
333 pRec = allocator_type().New( curThreadId );
334
335 thread_record * pOldHead = m_pHead.load( atomics::memory_order_acquire );
336 do {
337 // Compiler barriers: assignment MUST BE inside the loop
338 CDS_COMPILER_RW_BARRIER;
339 pRec->m_list.next_ = pOldHead;
340 CDS_COMPILER_RW_BARRIER;
341 } while ( !m_pHead.compare_exchange_weak( pOldHead, pRec, atomics::memory_order_release, atomics::memory_order_acquire ));
342
343 return pRec;
344 }
345
346 void retire( thread_record * pRec )
347 {
348 assert( pRec != nullptr );
349 pRec->m_list.thread_id_.store( cds::OS::c_NullThreadId, atomics::memory_order_release );
350 }
351
352 void detach_all()
353 {
354 thread_record * pNext = nullptr;
355 cds::OS::ThreadId const nullThreadId = cds::OS::c_NullThreadId;
356
357 for ( thread_record * pRec = m_pHead.load( atomics::memory_order_acquire ); pRec; pRec = pNext ) {
358 pNext = pRec->m_list.next_;
359 if ( pRec->m_list.thread_id_.load( atomics::memory_order_acquire ) != nullThreadId ) {
360 retire( pRec );
361 }
362 }
363 }
364
365 thread_record * head( atomics::memory_order mo ) const
366 {
367 return m_pHead.load( mo );
368 }
369
370 private:
371 void destroy()
372 {
373 allocator_type al;
374 CDS_DEBUG_ONLY( cds::OS::ThreadId const nullThreadId = cds::OS::c_NullThreadId; )
375 CDS_DEBUG_ONLY( cds::OS::ThreadId const mainThreadId = cds::OS::get_current_thread_id() ;)
376
377 thread_record * p = m_pHead.exchange( nullptr, atomics::memory_order_acquire );
378 while ( p ) {
379 thread_record * pNext = p->m_list.next_;
380
381 assert( p->m_list.thread_id_.load( atomics::memory_order_relaxed ) == nullThreadId
382 || p->m_list.thread_id_.load( atomics::memory_order_relaxed ) == mainThreadId
383 );
384
385 al.Delete( p );
386 p = pNext;
387 }
388 }
389 };
