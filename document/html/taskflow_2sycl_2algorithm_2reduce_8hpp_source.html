12template<unsigned nt, typename T>
13struct syclBlockReduce {
14
15 static const unsigned group_size = std::min(nt, SYCL_WARP_SIZE);
16 static const unsigned shm_size = std::max(nt, 2* group_size);
17 static const unsigned num_passes = log2(group_size);
18 static const unsigned num_items = nt / group_size;
19
20 static_assert(
21 nt && (0 == nt % SYCL_WARP_SIZE),
22 "syclBlockReduce requires num threads to be a multiple of warp_size (32)"
23 );

26 T, 1, sycl::access::mode::read_write, sycl::access::target::local
27 >;
28
29 template<typename op_t>
30 T operator()(
31 sycl::nd_item<1>&, T, const shm_t&, unsigned, op_t, bool = true
32 ) const;
33};

35// function: reduce to be called from a block
36template<unsigned nt, typename T>
37template<typename op_t>
38T syclBlockReduce<nt, T>::operator ()(
39 sycl::nd_item<1>& item,
40 T x,
41 const shm_t& shm,
42 unsigned count,
43 op_t op,
44 bool ret
45) const {
46
47 auto tid = item.get_local_id(0);
48
49 // Store your data into shared memory.
50 shm[tid] = x;
51 item.barrier(sycl::access::fence_space::local_space);
52
53 if(tid < group_size) {
54 // Each thread scans within its lane.
55 sycl_strided_iterate<group_size, num_items>([&](auto i, auto j) {
56 if(i > 0) {
57 x = op(x, shm[j]);
58 }
59 }, tid, count);
60 shm[tid] = x;
61 }
62 item.barrier(sycl::access::fence_space::local_space);
63
64 auto count2 = count < group_size ? count : group_size;
65 auto first = (1 & num_passes) ? group_size : 0;
66 if(tid < group_size) {
67 shm[first + tid] = x;
68 }
69 item.barrier(sycl::access::fence_space::local_space);
70
71 sycl_iterate<num_passes>([&](auto pass) {
72 if(tid < group_size) {
73 if(auto offset = 1 << pass; tid + offset < count2) {
74 x = op(x, shm[first + offset + tid]);
75 }
76 first = group_size - first;
77 shm[first + tid] = x;
78 }
79 item.barrier(sycl::access::fence_space::local_space);
80 });
81
82 if(ret) {
83 x = shm[0];
84 item.barrier(sycl::access::fence_space::local_space);
85 }
86 return x;
87}

90template <typename P, typename I, typename T, typename O>
91sycl::event sycl_reduce_loop(
92 P&& p,
93 I input,
94 unsigned count,
95 T* res,
96 O op,
97 bool incl,
98 void* ptr,
99 std::vector<sycl::event> evs
100) {
101
102 using E = std::decay_t<P>;
103 using R = syclBlockReduce<E::nt, T>;
104
105 auto buf = static_cast<T*>(ptr);
106 auto B = (count + E::nv - 1) / E::nv;
107
108 auto e = p.queue().submit([=, evs=std::move(evs)](sycl::handler& h) {
109
110 h.depends_on(evs);
111
112 // create a shared memory
113 typename R::shm_t shm(sycl::range<1>(R::shm_size), h);
114
115 h.parallel_for(
116 sycl::nd_range<1>{sycl::range<1>(B*E::nt), sycl::range<1>(E::nt)},
117 [=](sycl::nd_item<1> item) {
118
119 auto tid = item.get_local_id(0);
120 auto bid = item.get_group(0);
121
122 // get the tile of this group
123 auto tile = sycl_get_tile(bid, E::nv, count);
124
125 // load data from input to register
126 auto x = sycl_mem_to_reg_strided<E::nt, E::vt>(
127 input + tile.begin, tid, tile.count()
128 );
129 // reduce multiple values per thread into a scalar.
130 T s;
131 sycl_strided_iterate<E::nt, E::vt>(
132 [&] (auto i, auto) { s = i ? op(s, x[i]) : x[0]; }, tid, tile.count()
133 );
134 // reduce to a scalar per block.
135 s = R()(
136 item, s, shm, (tile.count()<E::nt ? tile.count() : E::nt), op, false
137 );
138 if(!tid) {
139 (1 == B) ? *res = (incl ? op(*res, s) : s) : buf[bid] = s;
140 }
141 }
142 );
143 });
144
145 if(B > 1) {
146 return sycl_reduce_loop(p, buf, B, res, op, incl, buf+B, {e});
147 }
148 else {
149 return e;
150 }
151}

168template <typename P, typename T>
169unsigned sycl_reduce_buffer_size(unsigned count) {
170 using E = std::decay_t<P>;
171 unsigned B = (count + E::nv - 1) / E::nv;
172 unsigned n = 0;
173 for(auto b=B; b>1; n += (b=(b+E::nv-1)/E::nv));
174 return n*sizeof(T);
175}

257template<typename P, typename I, typename T, typename O>
258void sycl_reduce(P&& p, I first, I last, T* res, O op) {
259
260 unsigned count = std::distance(first, last);
261
262 if(count == 0) {
263 return;
264 }
265
266 // allocate temporary buffer
267 auto tmp = sycl::malloc_device(
268 sycl_reduce_buffer_size<P, T>(count), p.queue()
269 );
270
271 // reduction loop
272 detail::sycl_reduce_loop(p, first, count, res, op, true, tmp, {}).wait();
273
274 // deallocate the temporary buffer
275 sycl::free(tmp, p.queue());
276}
