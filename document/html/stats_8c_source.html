22static void mi_stat_update_mt(mi_stat_count_t* stat, int64_t amount) {
23 if (amount == 0) return;
24 // add atomically
25 int64_t current = mi_atomic_addi64_relaxed(&stat->current, amount);
26 mi_atomic_maxi64_relaxed(&stat->peak, current + amount);
27 if (amount > 0) {
28 mi_atomic_addi64_relaxed(&stat->total, amount);
29 }
30}
31
32static void mi_stat_update(mi_stat_count_t* stat, int64_t amount) {
33 if (amount == 0) return;
34 // add thread local
35 stat->current += amount;
36 if (stat->current > stat->peak) { stat->peak = stat->current; }
37 if (amount > 0) { stat->total += amount; }
38}

64// Adjust stats to compensate; for example before committing a range,
65// first adjust downwards with parts that were already committed so
66// we avoid double counting.
67static void mi_stat_adjust_mt(mi_stat_count_t* stat, int64_t amount) {
68 if (amount == 0) return;
69 // adjust atomically
70 mi_atomic_addi64_relaxed(&stat->current, amount);
71 mi_atomic_addi64_relaxed(&stat->total, amount);
72}

137// unit > 0 : size in binary bytes
138// unit == 0: count as decimal
139// unit < 0 : count in binary
140static void mi_printf_amount(int64_t n, int64_t unit, mi_output_fun* out, void* arg, const char* fmt) {
141 char buf[32]; _mi_memzero_var(buf);
142 int len = 32;
143 const char* suffix = (unit <= 0 ? " " : "B");
144 const int64_t base = (unit == 0 ? 1000 : 1024);
145 if (unit>0) n *= unit;
146
147 const int64_t pos = (n < 0 ? -n : n);
148 if (pos < base) {
149 if (n!=1 || suffix[0] != 'B') { // skip printing 1 B for the unit column
150 _mi_snprintf(buf, len, "%lld %-3s", (long long)n, (n==0 ? "" : suffix));
151 }
152 }
153 else {
154 int64_t divider = base;
155 const char* magnitude = "K";
156 if (pos >= divider*base) { divider *= base; magnitude = "M"; }
157 if (pos >= divider*base) { divider *= base; magnitude = "G"; }
158 const int64_t tens = (n / (divider/10));
159 const long whole = (long)(tens/10);
160 const long frac1 = (long)(tens%10);
161 char unitdesc[8];
162 _mi_snprintf(unitdesc, 8, "%s%s%s", magnitude, (base==1024 ? "i" : ""), suffix);
163 _mi_snprintf(buf, len, "%ld.%ld %-3s", whole, (frac1 < 0 ? -frac1 : frac1), unitdesc);
164 }
165 _mi_fprintf(out, arg, (fmt==NULL ? "%12s" : fmt), buf);
166}

179 _mi_fprintf(out, arg,"%10s:", msg);

187 mi_print_count(stat->total, unit, out, arg);

201 }
202 if (stat->current != 0) {
203 _mi_fprintf(out, arg, " ");
204 _mi_fprintf(out, arg, (notok == NULL ? "not all freed" : notok));
205 _mi_fprintf(out, arg, "\n");
206 }
207 else {
208 _mi_fprintf(out, arg, " ok\n");
209 }
210 }
211 else {
212 mi_print_amount(stat->peak, 1, out, arg);
213 mi_print_amount(stat->total, 1, out, arg);
214 _mi_fprintf(out, arg, "%11s", " "); // no freed
215 mi_print_amount(stat->current, 1, out, arg);
216 _mi_fprintf(out, arg, "\n");
217 }

270//------------------------------------------------------------
271// Use an output wrapper for line-buffered output
272// (which is nice when using loggers etc.)
273//------------------------------------------------------------
274typedef struct buffered_s {
275 mi_output_fun* out; // original output function
276 void* arg; // and state
277 char* buf; // local buffer of at least size `count+1`
278 size_t used; // currently used chars `used <= count`
279 size_t count; // total chars available for output
280} buffered_t;
281
282static void mi_buffered_flush(buffered_t* buf) {
283 buf->buf[buf->used] = 0;
284 _mi_fputs(buf->out, buf->arg, NULL, buf->buf);
285 buf->used = 0;
286}
287
288static void mi_cdecl mi_buffered_out(const char* msg, void* arg) {
289 buffered_t* buf = (buffered_t*)arg;
290 if (msg==NULL || buf==NULL) return;
291 for (const char* src = msg; *src != 0; src++) {
292 char c = *src;
293 if (buf->used >= buf->count) mi_buffered_flush(buf);
294 mi_assert_internal(buf->used < buf->count);
295 buf->buf[buf->used++] = c;
296 if (c == '\n') mi_buffered_flush(buf);
297 }
298}

429mi_msecs_t _mi_clock_now(void) {
430 return _mi_prim_clock_now();
431}
432
433mi_msecs_t _mi_clock_start(void) {
434 if (mi_clock_diff == 0.0) {
435 mi_msecs_t t0 = _mi_clock_now();
436 mi_clock_diff = _mi_clock_now() - t0;
437 }
438 return _mi_clock_now();
439}

441mi_msecs_t _mi_clock_end(mi_msecs_t start) {
442 mi_msecs_t end = _mi_clock_now();
443 return (end - start - mi_clock_diff);
444}

451mi_decl_export void mi_process_info(size_t* elapsed_msecs, size_t* user_msecs, size_t* system_msecs, size_t* current_rss, size_t* peak_rss, size_t* current_commit, size_t* peak_commit, size_t* page_faults) mi_attr_noexcept
452{
453 mi_subproc_t* subproc = _mi_subproc();
454 mi_process_info_t pinfo;
455 _mi_memzero_var(pinfo);
456 pinfo.elapsed = _mi_clock_end(mi_process_start);
457 pinfo.current_commit = (size_t)(mi_atomic_loadi64_relaxed((_Atomic(int64_t)*)(&subproc->stats.committed.current)));
458 pinfo.peak_commit = (size_t)(mi_atomic_loadi64_relaxed((_Atomic(int64_t)*)(&subproc->stats.committed.peak)));
459 pinfo.current_rss = pinfo.current_commit;
460 pinfo.peak_rss = pinfo.peak_commit;
461 pinfo.utime = 0;
462 pinfo.stime = 0;
463 pinfo.page_faults = 0;
464
465 _mi_prim_process_info(&pinfo);
466
467 if (elapsed_msecs!=NULL) *elapsed_msecs = (pinfo.elapsed < 0 ? 0 : (pinfo.elapsed < (mi_msecs_t)PTRDIFF_MAX ? (size_t)pinfo.elapsed : PTRDIFF_MAX));
468 if (user_msecs!=NULL) *user_msecs = (pinfo.utime < 0 ? 0 : (pinfo.utime < (mi_msecs_t)PTRDIFF_MAX ? (size_t)pinfo.utime : PTRDIFF_MAX));
469 if (system_msecs!=NULL) *system_msecs = (pinfo.stime < 0 ? 0 : (pinfo.stime < (mi_msecs_t)PTRDIFF_MAX ? (size_t)pinfo.stime : PTRDIFF_MAX));
470 if (current_rss!=NULL) *current_rss = pinfo.current_rss;
471 if (peak_rss!=NULL) *peak_rss = pinfo.peak_rss;
472 if (current_commit!=NULL) *current_commit = pinfo.current_commit;
473 if (peak_commit!=NULL) *peak_commit = pinfo.peak_commit;
474 if (page_faults!=NULL) *page_faults = pinfo.page_faults;
475}

495typedef struct mi_heap_buf_s {
496 char* buf;
497 size_t size;
498 size_t used;
499 bool can_realloc;
500} mi_heap_buf_t;
501
502static bool mi_heap_buf_expand(mi_heap_buf_t* hbuf) {
503 if (hbuf==NULL) return false;
504 if (hbuf->buf != NULL && hbuf->size>0) {
505 hbuf->buf[hbuf->size-1] = 0;
506 }
507 if (hbuf->size > SIZE_MAX/2 || !hbuf->can_realloc) return false;
508 const size_t newsize = (hbuf->size == 0 ? 2*MI_KiB : 2*hbuf->size);
509 char* const newbuf = (char*)mi_rezalloc(hbuf->buf, newsize);
510 if (newbuf == NULL) return false;
511 hbuf->buf = newbuf;
512 hbuf->size = newsize;
513 return true;
514}
515
516static void mi_heap_buf_print(mi_heap_buf_t* hbuf, const char* msg) {
517 if (msg==NULL || hbuf==NULL) return;
518 if (hbuf->used + 1 >= hbuf->size && !hbuf->can_realloc) return;
519 for (const char* src = msg; *src != 0; src++) {
520 char c = *src;
521 if (hbuf->used + 1 >= hbuf->size) {
522 if (!mi_heap_buf_expand(hbuf)) return;
523 }
524 mi_assert_internal(hbuf->used < hbuf->size);
525 hbuf->buf[hbuf->used++] = c;
526 }
527 mi_assert_internal(hbuf->used < hbuf->size);
