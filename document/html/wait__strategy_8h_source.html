186 template <int Milliseconds = 2>
187 class single_mutex_single_condvar
188 {
189 //@cond
190 std::mutex m_mutex;
191 std::condition_variable m_condvar;
192 bool m_wakeup;
193
194 typedef std::unique_lock< std::mutex > unique_lock;
195 //@endcond
196
197 public:
198 enum {
199 c_nWaitMilliseconds = Milliseconds < 1 ? 1 : Milliseconds
200 };
201
203 template <typename PublicationRecord>
204 struct make_publication_record {
205 typedef PublicationRecord type;
206 };
207
209 single_mutex_single_condvar()
210 : m_wakeup( false )
211 {}
212
214 template <typename PublicationRecord>
215 void prepare( PublicationRecord& /*rec*/ )
216 {}
217
219 template <typename FCKernel, typename PublicationRecord>
220 bool wait( FCKernel& fc, PublicationRecord& rec )
221 {
222 if ( fc.get_operation( rec ) >= req_Operation ) {
223 unique_lock lock( m_mutex );
224 if ( fc.get_operation( rec ) >= req_Operation ) {
225 if ( m_wakeup ) {
226 m_wakeup = false;
227 return true;
228 }
229
230 bool ret = m_condvar.wait_for( lock, std::chrono::milliseconds( c_nWaitMilliseconds )) == std::cv_status::no_timeout;
231 m_wakeup = false;
232 return ret;
233 }
234 }
235 return false;
236 }
237
239 template <typename FCKernel, typename PublicationRecord>
240 void notify( FCKernel& fc, PublicationRecord& /*rec*/ )
241 {
242 wakeup( fc );
243 }
244
246 template <typename FCKernel>
247 void wakeup( FCKernel& /*fc*/ )
248 {
249 unique_lock lock( m_mutex );
250 m_wakeup = true;
251 m_condvar.notify_all();
252 }
253 };
