71 image_buffer generatePNGImage() {
72 image_buffer image;
73 image.width = 1024;
74 image.height = 1024;
75
76 image.buffer = std::make_shared< std::vector<unsigned char> >(image.width * image.height * 4);
77 std::vector<unsigned char>& buffer = *image.buffer;
78
79 const int widthOffset = 4 * image.width;
80 for (unsigned y = 0; y < image.height; y++) {
81 for (unsigned x = 0; x < image.width; x++) {
82 const int pixelOffset = widthOffset * y + 4 * x;
83 buffer[pixelOffset] = 200 * !(x & y);
84 buffer[pixelOffset + 1] = x ^ y;
85 buffer[pixelOffset + 2] = x | y;
86 buffer[pixelOffset + 3] = 255;
87 }
88 }
89 return image;
90 }
