543 template <typename T>
544 requires(eastl::is_default_constructible_v<T>)
545 inline typename eastl::enable_if<eastl::is_unbounded_array<T>::value, eastl::unique_ptr<T>>::type make_unique(
546 size_t n)
547 {
548 typedef typename eastl::remove_extent<T>::type TBase;
549 size_t header_size = max<size_t>(sizeof(size_t), alignof(TBase));
550 size_t alloc_size = header_size + n * sizeof(TBase);
551 auto header_ptr = reinterpret_cast<size_t*>(eastl::GetDefaultAllocator()->allocate(alloc_size));
552 *header_ptr = n;
553 auto ptr = reinterpret_cast<TBase*>(reinterpret_cast<size_t>(header_ptr) + header_size);
554 if constexpr (!std::is_trivially_constructible_v<TBase>)
555 {
556 auto end = ptr + n;
557 for (auto i = ptr; i != end; ++i)
558 {
559 new (i) TBase{};
560 }
561 }
562 return unique_ptr<T>(ptr);
563 }
