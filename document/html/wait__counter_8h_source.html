39 void wait() {
40 int k = --my_count;
41 __TBB_ASSERT( k>=0, "counter underflow" );
42 if( k>0 ) {
43 thread_monitor::cookie c;
44 my_monitor.prepare_wait(c);
45 if( my_count )
46 my_monitor.commit_wait(c);
47 else
48 my_monitor.cancel_wait();
49 }
50 while( n_transients>0 )
51 __TBB_Yield();
52 }
53 void operator++() {
54 ++my_count;
55 }
56 void operator--() {
57 ++n_transients;
58 int k = --my_count;
59 __TBB_ASSERT( k>=0, "counter underflow" );
60 if( k==0 )
61 my_monitor.notify();
62 --n_transients;
63 }
