25 struct link_node
26 {
27 typedef Node node_type;
28 typedef T mapped_type;
29 typedef uint32_t version_type;

31 enum

41 atomics::atomic<node_type *> m_pParent;
42 atomics::atomic<node_type *> m_pLeft;

47 public:

51 , m_pParent( nullptr )
52 , m_pLeft( nullptr )
53 , m_pRight( nullptr )

56 }

61 , m_pParent( pParent )
62 , m_pLeft( pLeft )
63 , m_pRight( pRight )
64 {
65 m_pValue.store( nullptr, atomics::memory_order_release );
66 }

73 void parent( node_type * p, atomics::memory_order order )

75 m_pParent.store( p, order );
76 }
77
78 node_type * child( int nDirection, atomics::memory_order order ) const
79 {
80 assert( nDirection != 0 );
81 return nDirection < 0 ? m_pLeft.load( order ) : m_pRight.load( order );
82 }
83
84 void child( node_type * pChild, int nDirection, atomics::memory_order order )
85 {
86 assert( nDirection != 0 );

88 m_pLeft.store( pChild, order );

90 m_pRight.store( pChild, order );
91 }

118 template <typename BackOff>
119 void wait_until_shrink_completed( atomics::memory_order order ) const
120 {
121 BackOff bkoff;

124 }

127 {
128 return m_nVersion.load( order ) == unlinked;
