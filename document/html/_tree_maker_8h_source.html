31 class SubTreeCreationTask: public tbb::task {
32 TreeNode*& my_root;
33 bool is_continuation;
34 typedef TreeMaker<use_tbbmalloc> MyTreeMaker;
35
36 public:
37 SubTreeCreationTask( TreeNode*& root, long number_of_nodes ) : my_root(root), is_continuation(false) {
38 my_root = MyTreeMaker::allocate_node();
39 my_root->node_count = number_of_nodes;
40 my_root->value = Value(Pi*number_of_nodes);
41 }
42
43 tbb::task* execute() /*override*/ {
44 tbb::task* next = NULL;
45 if( !is_continuation ) {
46 long subtree_size = my_root->node_count - 1;
47 if( subtree_size<1000 ) { /* grainsize */
48 my_root->left = MyTreeMaker::do_in_one_thread(subtree_size/2);
49 my_root->right = MyTreeMaker::do_in_one_thread(subtree_size - subtree_size/2);
50 } else {
51 // Create tasks before spawning any of them.
52 tbb::task* a = new( allocate_child() ) SubTreeCreationTask(my_root->left,subtree_size/2);
53 tbb::task* b = new( allocate_child() ) SubTreeCreationTask(my_root->right,subtree_size - subtree_size/2);
54 recycle_as_continuation();
55 is_continuation = true;
56 set_ref_count(2);
57 spawn(*b);
58 next = a;
59 }
60 }
61 return next;
62 }
63 };

65public:
66 static TreeNode* allocate_node() {
67 return use_tbbmalloc? tbb::scalable_allocator<TreeNode>().allocate(1) : new TreeNode;
68 }
69
70 static TreeNode* do_in_one_thread( long number_of_nodes ) {
71 if( number_of_nodes==0 ) {
72 return NULL;
73 } else {
74 TreeNode* n = allocate_node();
75 n->node_count = number_of_nodes;
76 n->value = Value(Pi*number_of_nodes);
77 --number_of_nodes;
78 n->left = do_in_one_thread( number_of_nodes/2 );
79 n->right = do_in_one_thread( number_of_nodes - number_of_nodes/2 );
80 return n;
81 }
82 }
83
84 static TreeNode* do_in_parallel( long number_of_nodes ) {
85 TreeNode* root_node;
86 SubTreeCreationTask& a = *new(tbb::task::allocate_root()) SubTreeCreationTask(root_node, number_of_nodes);
87 tbb::task::spawn_root_and_wait(a);
88 return root_node;
89 }
90
91 static TreeNode* create_and_time( long number_of_nodes, bool silent=false ) {
92 tbb::tick_count t0, t1;
93 TreeNode* root = allocate_node();
94 root->node_count = number_of_nodes;
95 root->value = Value(Pi*number_of_nodes);
96 --number_of_nodes;
97
98 t0 = tbb::tick_count::now();
99 root->left = do_in_one_thread( number_of_nodes/2 );
100 t1 = tbb::tick_count::now();
101 if ( !silent ) printf ("%24s: time = %.1f msec\n", "half created serially", (t1-t0).seconds()*1000);
102
103 t0 = tbb::tick_count::now();
104 root->right = do_in_parallel( number_of_nodes - number_of_nodes/2 );
105 t1 = tbb::tick_count::now();
106 if ( !silent ) printf ("%24s: time = %.1f msec\n", "half done in parallel", (t1-t0).seconds()*1000);
107
108 return root;
109 }
110};
