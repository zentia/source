81 message_t *svc(message_t* task) {
82 MPI_Status status;
83 while(neos < input_channels){
84
85 int headersLen;
86 MPI_Probe(MPI_ANY_SOURCE, DFF_HEADER_TAG, MPI_COMM_WORLD, &status);
87 MPI_Get_count(&status, MPI_LONG, &headersLen);
88 long headers[headersLen];
89
90 if (MPI_Recv(headers, headersLen, MPI_LONG, status.MPI_SOURCE, DFF_HEADER_TAG, MPI_COMM_WORLD, &status) != MPI_SUCCESS)
91 error("Error on Recv Receiver primo in alto\n");
92 bool feedback = ChannelType::FBK == rank2ChannelType[status.MPI_SOURCE];
93 assert(headers[0]*3+1 == headersLen);
94 if (headers[0] == 1){
95 size_t sz = headers[3];
96
97 if (sz == 0){
98 if (headers[2] == -2){
99 registerLogicalEOS(headers[1]);
100 continue;
101 }
102 registerEOS(status.MPI_SOURCE);
103 continue;
104 }
105 char* buff = new char[sz];
106 if (MPI_Recv(buff,sz,MPI_BYTE, status.MPI_SOURCE, DFF_TASK_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE) != MPI_SUCCESS)
107 error("Error on Recv Receiver Payload\n");
108
109 message_t* out = new message_t(buff, sz, true);
110 out->sender = headers[1];
111 out->chid = headers[2];
112 out->feedback = feedback;
113
114 this->forward(out, status.MPI_SOURCE);
115 } else {
116 int size;
117 MPI_Status localStatus;
118 MPI_Probe(status.MPI_SOURCE, DFF_TASK_TAG, MPI_COMM_WORLD, &localStatus);
119 MPI_Get_count(&localStatus, MPI_BYTE, &size);
120 char* buff = new char[size]; // this can be reused!!
121 MPI_Recv(buff, size, MPI_BYTE, localStatus.MPI_SOURCE, DFF_TASK_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
122 size_t head = 0;
123 for (size_t i = 0; i < (size_t)headers[0]; i++){
124 size_t sz = headers[3*i+3];
125 if (sz == 0){
126 if (headers[3*i+2] == -2){
127 registerLogicalEOS(headers[3*i+1]);
128 continue;
129 }
130 registerEOS(status.MPI_SOURCE);
131 assert(i+1 == (size_t)headers[0]);
132 break;
133 }
134 char* outBuff = new char[sz];
135 memcpy(outBuff, buff+head, sz);
136 head += sz;
137 message_t* out = new message_t(outBuff, sz, true);
138 out->sender = headers[3*i+1];
139 out->chid = headers[3*i+2];
140 out->feedback = feedback;
141
142 this->forward(out, status.MPI_SOURCE);
143 }
144 delete [] buff;
145 }
146
147 }
148
149 return this->EOS;
