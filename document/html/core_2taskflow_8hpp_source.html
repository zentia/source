444

456
457 os << "];\n";
458
459 for(size_t s=0; s<node->_num_successors; ++s) {
460 if(node->_is_conditioner()) {
461 // case edge is dashed
462 os << 'p' << node << " -> p" << node->_edges[s]
463 << " [style=dashed label=\"" << s << "\"];\n";
464 } else {
465 os << 'p' << node << " -> p" << node->_edges[s] << ";\n";
466 }

470 if(node->_parent && node->_parent->_handle.index() == Node::SUBFLOW &&
471 node->_num_successors == 0

474 }
