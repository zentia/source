284 template <typename Q>
285 std::pair<iterator, bool> update( const Q& val, bool bInsert = true )
286 {
287 scoped_node_ptr sp( node_allocator().New( base_class::random_level(), val ));
288 node_type * pNode;
289 std::pair<bool, bool> bRes = base_class::update( *sp, [&pNode](bool, node_type& item, node_type&) { pNode = &item; }, bInsert );
290 if ( bRes.first && bRes.second )
291 sp.release();
292 else if ( !bRes.first )
293 return std::make_pair( end(), false );
294 assert( pNode );
295 return std::make_pair( node_to_iterator( pNode ), bRes.second );
296 }
