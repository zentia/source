59 class BlockAllocator : public SpineObject {

63 public:
64 BlockAllocator(int initialBlockSize) : initialBlockSize(initialBlockSize) {

68 ~BlockAllocator() {

74 template<typename T>
75 T *allocate(size_t num) {
76 return (T *) _allocate((int) (sizeof(T) * num));
77 }

79 void compress() {
80 if (blocks.size() == 1) return;
81 int totalSize = 0;
82 for (int i = 0, n = (int)blocks.size(); i < n; i++) {
83 totalSize += blocks[i].size;
84 SpineExtension::free(blocks[i].memory, __FILE__, __LINE__);
85 }
86 blocks.clear();
87 blocks.add(newBlock(totalSize));
88 }
89
90 private:
91 void *_allocate(int numBytes) {
92 // 16-byte align allocations
93 int alignedNumBytes = numBytes + (numBytes % 16 != 0 ? 16 - (numBytes % 16) : 0);
94 Block *block = &blocks[blocks.size() - 1];
95 if (!block->canFit(alignedNumBytes)) {
96 blocks.add(newBlock(MathUtil::max(initialBlockSize, alignedNumBytes)));
97 block = &blocks[blocks.size() - 1];
98 }
99 return block->allocate(alignedNumBytes);
100 }
101
102 Block newBlock(int numBytes) {
103 Block block = {MathUtil::max(initialBlockSize, numBytes), 0, nullptr};
104 block.memory = SpineExtension::alloc<uint8_t>(block.size, __FILE__, __LINE__);
105 return block;
106 }
