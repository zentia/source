26 // extent
27 //
28 // extent<T, I>::value is an integral type representing the number of
29 // elements in the Ith dimension of array type T.
30 //
31 // For a given array type T[N], extent<T[N]>::value == N.
32 // For a given multi-dimensional array type T[M][N], extent<T[M][N], 0>::value == N.
33 // For a given multi-dimensional array type T[M][N], extent<T[M][N], 1>::value == M.
34 // For a given array type T and a given dimension I where I >= rank<T>::value, extent<T, I>::value == 0.
35 // For a given array type of unknown extent T[], extent<T[], 0>::value == 0.
36 // For a given non-array type T and an arbitrary dimension I, extent<T, I>::value == 0.
37 //
39
40 #define EASTL_TYPE_TRAIT_extent_CONFORMANCE 1 // extent is conforming.
41

43 struct extent_help : public eastl::integral_constant<size_t, 0> {};
44

46 struct extent_help<T[I], 0> : public eastl::integral_constant<size_t, I> {};
47
48 template<typename T, unsigned N, unsigned I>
49 struct extent_help<T[I], N> : public eastl::extent_help<T, N - 1> { };
50
51 template<typename T, unsigned N>
52 struct extent_help<T[], N> : public eastl::extent_help<T, N - 1> {};

236 // is_convertible
237 //
238 // Given two (possible identical) types From and To, is_convertible<From, To>::value == true
239 // if and only if an lvalue of type From can be implicitly converted to type To,
240 // or is_void<To>::value == true
241 //
242 // An instance of the type predicate holds true if the expression To to = from;, where from is an object of type From, is well-formed.
243 //
244 // is_convertible may only be applied to complete types.
245 // Type To may not be an abstract type.
246 // If the conversion is ambiguous, the program is ill-formed.
247 // If either or both of From and To are class types, and the conversion would invoke
248 // non-public member functions of either From or To (such as a private constructor of To,
249 // or a private conversion operator of From), the program is ill-formed.
250 //
251 // Note that without compiler help, both is_convertible and is_base
252 // can produce compiler errors if the conversion is ambiguous.
253 // Example:
254 // struct A {};
255 // struct B : A {};
256 // struct C : A {};
257 // struct D : B, C {};
258 // is_convertible<D*, A*>::value; // Generates compiler error.
260
261 #if EASTL_COMPILER_INTRINSIC_TYPE_TRAITS_AVAILABLE && (defined(_MSC_VER) || (defined(__clang__) && EA_COMPILER_HAS_FEATURE(is_convertible_to)))
262 #define EASTL_TYPE_TRAIT_is_convertible_CONFORMANCE 1 // is_convertible is conforming.
263
264 // Problem: VC++ reports that int is convertible to short, yet if you construct a short from an int then VC++ generates a warning:
265 // warning C4242: 'initializing' : conversion from 'int' to 'short', possible loss of data. We can deal with this by making
266 // is_convertible be false for conversions that could result in loss of data. Or we could make another trait called is_lossless_convertible
267 // and use that appropriately in our code. Or we could put the onus on the user to work around such warnings.
268 template <typename From, typename To>
269 struct is_convertible : public integral_constant<bool, __is_convertible_to(From, To)>{};
270
271 #else
272 #define EASTL_TYPE_TRAIT_is_convertible_CONFORMANCE 1
273
274 template<typename From, typename To, bool = eastl::is_void<From>::value || eastl::is_function<To>::value || eastl::is_array<To>::value >
275 struct is_convertible_helper // Anything is convertible to void. Nothing is convertible to a function or an array.
276 { static const bool value = eastl::is_void<To>::value; };
277
278 template<typename From, typename To>
279 class is_convertible_helper<From, To, false>
280 {
281 template<typename To1>
282 static void ToFunction(To1); // We try to call this function with an instance of From. It is valid if From can be converted to To.
283
284 template<typename /*From1*/, typename /*To1*/>
285 static eastl::no_type is(...);
286
287 template<typename From1, typename To1>
288 static decltype(ToFunction<To1>(eastl::declval<From1>()), eastl::yes_type()) is(int);
289
290 public:
291 static const bool value = sizeof(is<From, To>(0)) == 1;
292 };
293
294 template<typename From, typename To>
295 struct is_convertible
296 : public integral_constant<bool, is_convertible_helper<From, To>::value> {};
297
298 #endif
299
300 #if !defined(EA_COMPILER_NO_TEMPLATE_ALIASES)
301 template<typename From, typename To>
302 EA_CONSTEXPR bool is_convertible_v = is_convertible<From, To>::value;
303 #endif

414 //
415 // is_polymorphic<T>::value == true if and only if T is a class or struct
416 // that declares or inherits a virtual function. is_polymorphic may only
417 // be applied to complete types.
418 //
420
421 #if EASTL_COMPILER_INTRINSIC_TYPE_TRAITS_AVAILABLE && (defined(_MSC_VER) || defined(EA_COMPILER_GNUC) || (defined(__clang__) && EA_COMPILER_HAS_FEATURE(is_polymorphic)))
422 #define EASTL_TYPE_TRAIT_is_polymorphic_CONFORMANCE 1 // is_polymorphic is conforming.
423
424 template <typename T>
425 struct is_polymorphic : public integral_constant<bool, __is_polymorphic(T)>{};
426 #else
427 #define EASTL_TYPE_TRAIT_is_polymorphic_CONFORMANCE 1 // is_polymorphic is conforming.
428
429 template <typename T>
430 struct is_polymorphic_imp1
431 {
432 typedef typename remove_cv<T>::type t;
433
434 struct helper_1 : public t
435 {
436 helper_1();
437 ~helper_1() throw();
438 char pad[64];
439 };
440
441 struct helper_2 : public t
442 {
443 helper_2();
444 virtual ~helper_2() throw();
445 #ifndef _MSC_VER
446 virtual void foo();
447 #endif
448 char pad[64];
449 };
450
451 static const bool value = (sizeof(helper_1) == sizeof(helper_2));
452 };
453
454 template <typename T>
455 struct is_polymorphic_imp2{ static const bool value = false; };
456
457 template <bool is_class>
458 struct is_polymorphic_selector{ template <typename T> struct rebind{ typedef is_polymorphic_imp2<T> type; }; };
459
460 template <>
461 struct is_polymorphic_selector<true>{ template <typename T> struct rebind{ typedef is_polymorphic_imp1<T> type; }; };
462
463 template <typename T>
464 struct is_polymorphic_value{
465 typedef is_polymorphic_selector<is_class<T>::value> selector;
466 typedef typename selector::template rebind<T> binder;
467 typedef typename binder::type imp_type;
468 static const bool value = imp_type::value;
469 };

471 template <typename T>
472 struct is_polymorphic : public integral_constant<bool, is_polymorphic_value<T>::value>{};
473 #endif

569 // decay
570 //
571 // Converts the type T to its decayed equivalent. That means doing
572 // lvalue to rvalue, array to pointer, function to pointer conversions,
573 // and removal of const and volatile.
574 // This is the type conversion silently applied by the compiler to
575 // all function arguments when passed by value.
576
577 #define EASTL_TYPE_TRAIT_decay_CONFORMANCE 1 // decay is conforming.
578
579 template<typename T>
580 struct decay
581 {
582 typedef typename eastl::remove_reference<T>::type U;
583
584 typedef typename eastl::conditional<
585 eastl::is_array<U>::value,
586 typename eastl::remove_extent<U>::type*,
587 typename eastl::conditional<
588 eastl::is_function<U>::value,
589 typename eastl::add_pointer<U>::type,
590 typename eastl::remove_cv<U>::type
591 >::type
592 >::type type;
593 };
594
595
596 // decay_t is the C++14 using typedef for typename decay<T>::type, though
597 // it requires only C++11 compiler functionality to implement.
598 // We provide a backwards-compatible means to access it through a macro for pre-C++11 compilers.
599 #if defined(EA_COMPILER_NO_TEMPLATE_ALIASES)
600 #define EASTL_DECAY_T(T) typename decay<T>::type
601 #else
602 template<typename T>
603 using decay_t = typename decay<T>::type;
604 #define EASTL_DECAY_T(T) decay_t<T>
605 #endif

609 // common_type
610 //
611 // Determines the common type among all types T..., that is the type all T...
612 // can be implicitly converted to.
613 //
614 // It is intended that this be specialized by the user for cases where it
615 // is useful to do so. Example specialization:
616 // template <typename Class1, typename Class2>
617 // struct common_type<MyClass1, MyClass2>{ typedef MyBaseClassB type; };
618 //
619 // The member typedef type shall be defined as set out in 20.9.7.6,p3. All types in
620 // the parameter pack T shall be complete or (possibly cv) void. A program may
621 // specialize this trait if at least one template parameter in the specialization
622 // is a user-defined type. Note: Such specializations are needed when only
623 // explicit conversions are desired among the template arguments.
625
626 #define EASTL_TYPE_TRAIT_common_type_CONFORMANCE 1 // common_type is conforming.
627
628 template<typename... T>
629 struct common_type;
630
631 template<typename T>
632 struct common_type<T>
633 { typedef decay_t<T> type; }; // Question: Should we use T or decay_t<T> here? The C++11 Standard specifically (20.9.7.6,p3) specifies that it be without decay, but libc++ uses decay.
634
635 template<typename T, typename U>
636 struct common_type<T, U>
637 {
638 typedef decay_t<decltype(true ? declval<T>() : declval<U>())> type; // The type of a tertiary expression is set by the compiler to be the common type of the two result types.
639 };
640
641 template<typename T, typename U, typename... V>
642 struct common_type<T, U, V...>
643 { typedef typename common_type<typename common_type<T, U>::type, V...>::type type; };
644
645
646 // common_type_t is the C++14 using typedef for typename common_type<T...>::type.
647 // We provide a backwards-compatible means to access it through a macro for pre-C++11 compilers.
648 #if defined(EA_COMPILER_NO_TEMPLATE_ALIASES)
649 #define EASTL_COMMON_TYPE_T(...) typename common_type<__VA_ARGS__>::type
650 #else
651 template <typename... T>
652 using common_type_t = typename common_type<T...>::type;
653 #define EASTL_COMMON_TYPE_T(...) common_type_t<__VA_ARGS__>
654 #endif
