112 /*
113 MSVC does not support inlining of thread_local static data members in class for DLL.
114 Example:
115 // .h
116 class storage
117 {
118 public:
119 record& get()
120 {
121 return tls_;
122 }
123
124 private:
125 static thread_local record tls_;
126 };
127
128 // .cpp (DLL)
129 thread_local record storage::tls_
130 */
131#define CDS_DISABLE_CLASS_TLS_INLINE

142//if constexpr support (C++17)

144 // Standard way to check if the compiler supports "if constexpr"
145 // Of course, MS VC doesn't support any standard way
146# if defined __cpp_if_constexpr
147# if __cpp_if_constexpr >= 201606
148# define constexpr_if if constexpr
149# endif
150# elif CDS_COMPILER_VERSION >= CDS_COMPILER_MSVC14_1_3 && _MSVC_LANG > CDS_CPLUSPLUS_14
151 // MS-specific WTF.
152 // Don't work in /std:c++17 because /std:c++17 key defines _MSVC_LANG=201402 (c++14) in VC 15.3
153# define constexpr_if if constexpr
154# endif
