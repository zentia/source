120 pNext = res.guards.protect( 1, h->m_pNext, []( node_type * p ) -> value_type * { return node_traits::to_value_ptr( p );});
121
122 if ( pNext == nullptr ) {
123 base_class::m_Stat.onEmptyDequeue();

127 if ( base_class::m_pHead.compare_exchange_strong( h, pNext, memory_model::memory_order_release, atomics::memory_order_relaxed )) {
128 node_type * t = base_class::m_pTail.load(memory_model::memory_order_acquire);
129 if ( h == t )

132 }

134 base_class::m_Stat.onDequeueRace();
135 bkoff();
136 }
