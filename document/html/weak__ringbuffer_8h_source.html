196 template <typename Q, typename CopyFunc>
197 bool push( Q* arr, size_t count, CopyFunc copy )
198 {
199 assert( count < capacity());
200 counter_type back = back_.load( memory_model::memory_order_relaxed );
201
202 assert( static_cast<size_t>( back - pfront_ ) <= capacity());
203
204 if ( static_cast<size_t>( pfront_ + capacity() - back ) < count ) {
205 pfront_ = front_.load( memory_model::memory_order_acquire );
206
207 if ( static_cast<size_t>( pfront_ + capacity() - back ) < count ) {
208 // not enough space
209 return false;
210 }
211 }
212
213 // copy data
214 for ( size_t i = 0; i < count; ++i, ++back )
215 copy( buffer_[buffer_.mod( back )], arr[i] );
216
217 back_.store( back, memory_model::memory_order_release );
218
219 return true;
220 }

249 typename std::enable_if< std::is_constructible<value_type, Args...>::value, bool>::type
250 emplace( Args&&... args )
251 {

255

352 template <typename Q, typename CopyFunc>
353 bool pop( Q* arr, size_t count, CopyFunc copy )
354 {
355 assert( count < capacity());
356
357 counter_type front = front_.load( memory_model::memory_order_relaxed );
358 assert( static_cast<size_t>( cback_ - front ) < capacity());
359
360 if ( static_cast<size_t>( cback_ - front ) < count ) {
361 cback_ = back_.load( memory_model::memory_order_acquire );
362 if ( static_cast<size_t>( cback_ - front ) < count )
363 return false;
364 }
365
366 // copy data
367 value_cleaner cleaner;
368 for ( size_t i = 0; i < count; ++i, ++front ) {
369 value_type& val = buffer_[buffer_.mod( front )];
370 copy( arr[i], val );
371 cleaner( val );
372 }
373
374 front_.store( front, memory_model::memory_order_release );
375 return true;
376 }

682 void* back( size_t size )
683 {
684 assert( size > 0 );
685
686 // Any data is rounded to 8-byte boundary
687 size_t real_size = calc_real_size( size );
688
689 // check if we can reserve real_size bytes
690 assert( real_size < capacity());
691 counter_type back = back_.load( memory_model::memory_order_relaxed );
692
693 assert( static_cast<size_t>( back - pfront_ ) <= capacity());
694
695 if ( static_cast<size_t>( pfront_ + capacity() - back ) < real_size ) {
696 pfront_ = front_.load( memory_model::memory_order_acquire );
697
698 if ( static_cast<size_t>( pfront_ + capacity() - back ) < real_size ) {
699 // not enough space
700 return nullptr;
701 }
702 }
703
704 uint8_t* reserved = buffer_.buffer() + buffer_.mod( back );
705
706 // Check if the buffer free space is enough for storing real_size bytes
707 size_t tail_size = capacity() - static_cast<size_t>( buffer_.mod( back ));
708 if ( tail_size < real_size ) {
709 // make unused tail
710 assert( tail_size >= sizeof( size_t ));
711 assert( !is_tail( tail_size ));
712
713 *reinterpret_cast<size_t*>( reserved ) = make_tail( tail_size - sizeof(size_t));
714 back += tail_size;
715
716 // We must be in beginning of buffer
717 assert( buffer_.mod( back ) == 0 );
718
719 if ( static_cast<size_t>( pfront_ + capacity() - back ) < real_size ) {
720 pfront_ = front_.load( memory_model::memory_order_acquire );
721
722 if ( static_cast<size_t>( pfront_ + capacity() - back ) < real_size ) {
723 // not enough space
724 return nullptr;
725 }
726 }
727
728 back_.store( back, memory_model::memory_order_release );
729 reserved = buffer_.buffer();
730 }
731
732 // reserve and store size
733 *reinterpret_cast<size_t*>( reserved ) = size;
734
735 return reinterpret_cast<void*>( reserved + sizeof( size_t ));
736 }

805 std::pair<void*, size_t> front()
806 {
807 counter_type front = front_.load( memory_model::memory_order_relaxed );
808 assert( static_cast<size_t>( cback_ - front ) < capacity());
809
810 if ( cback_ - front < sizeof( size_t )) {
811 cback_ = back_.load( memory_model::memory_order_acquire );
812 if ( cback_ - front < sizeof( size_t ))
813 return std::make_pair( nullptr, 0u );
814 }
815
816 uint8_t * buf = buffer_.buffer() + buffer_.mod( front );
817
818 // check alignment
819 assert( ( reinterpret_cast<uintptr_t>( buf ) & ( sizeof( uintptr_t ) - 1 )) == 0 );
820
821 size_t size = *reinterpret_cast<size_t*>( buf );
822 if ( is_tail( size )) {
823 // unused tail, skip
824 CDS_VERIFY( pop_front());
825
826 front = front_.load( memory_model::memory_order_relaxed );
827
828 if ( cback_ - front < sizeof( size_t )) {
829 cback_ = back_.load( memory_model::memory_order_acquire );
830 if ( cback_ - front < sizeof( size_t ))
831 return std::make_pair( nullptr, 0u );
832 }
833
834 buf = buffer_.buffer() + buffer_.mod( front );
835 size = *reinterpret_cast<size_t*>( buf );
836
837 assert( !is_tail( size ));
838 assert( buf == buffer_.buffer());
839 }
840
841#ifdef _DEBUG
842 size_t real_size = calc_real_size( size );
843 if ( static_cast<size_t>( cback_ - front ) < real_size ) {
844 cback_ = back_.load( memory_model::memory_order_acquire );
845 assert( static_cast<size_t>( cback_ - front ) >= real_size );
846 }
847#endif
848
849 return std::make_pair( reinterpret_cast<void*>( buf + sizeof( size_t )), size );
850 }

881 assert( static_cast<size_t>( cback_ - front ) <= capacity());
