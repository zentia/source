184 template <class Set>
185 void test( Set& s )
186 {
187 // Precondition: set is empty
188 // Postcondition: set is empty
189
190 ASSERT_TRUE( s.empty());
191 ASSERT_CONTAINER_SIZE( s, 0 );
192 size_t const nSetSize = kSize;
193
194 typedef typename Set::value_type value_type;
195
196 std::vector< value_type > data;
197 std::vector< size_t> indices;
198 data.reserve( kSize );
199 indices.reserve( kSize );
200 for ( size_t key = 0; key < kSize; ++key ) {
201 data.push_back( value_type( static_cast<int>( key )));
202 indices.push_back( key );
203 }
204 shuffle( indices.begin(), indices.end());
205
206 // insert/find
207 for ( auto idx : indices ) {
208 auto& i = data[ idx ];
209
210 ASSERT_FALSE( s.contains( i.nKey ));
211 ASSERT_FALSE( s.contains( i ));
212 ASSERT_FALSE( s.contains( other_item( i.key()), other_less()));
213 ASSERT_FALSE( s.find( i.nKey, []( value_type&, int ) {} ));
214 ASSERT_FALSE( s.find_with( other_item( i.key()), other_less(), []( value_type&, other_item const& ) {} ));
215 ASSERT_TRUE( s.find( i.nKey ) == s.end());
216 ASSERT_TRUE( s.find_with( other_item( i.key()), other_less()) == s.end());
217
218 std::pair<bool, bool> updResult;
219
220 updResult = s.update( i, []( value_type&, value_type* )
221 {
222 ASSERT_TRUE( false );
223 }, false );
224 EXPECT_FALSE( updResult.first );
225 EXPECT_FALSE( updResult.second );
226
227 updResult = s.upsert( i, false );
228 EXPECT_FALSE( updResult.first );
229 EXPECT_FALSE( updResult.second );
230
231 switch ( i.key() % 4 ) {
232 case 0:
233 ASSERT_TRUE( s.insert( i ));
234 ASSERT_FALSE( s.insert( i ));
235 updResult = s.update( i, []( value_type& val, value_type* arg)
236 {
237 ASSERT_TRUE( arg != nullptr );
238 EXPECT_EQ( val.key(), arg->key());
239 }, false );
240 EXPECT_TRUE( updResult.first );
241 EXPECT_FALSE( updResult.second );
242 break;
243 case 1:
244 EXPECT_EQ( i.nUpdateNewCount, 0u );
245 ASSERT_TRUE( s.insert( i, []( value_type& v ) { ++v.nUpdateNewCount;} ));
246 EXPECT_EQ( i.nUpdateNewCount, 1u );
247 ASSERT_FALSE( s.insert( i, []( value_type& v ) { ++v.nUpdateNewCount;} ));
248 EXPECT_EQ( i.nUpdateNewCount, 1u );
249 i.nUpdateNewCount = 0;
250 break;
251 case 2:
252 updResult = s.update( i, []( value_type& /*val*/, value_type* arg )
253 {
254 EXPECT_TRUE( arg == nullptr );
255 });
256 EXPECT_TRUE( updResult.first );
257 EXPECT_TRUE( updResult.second );
258 break;
259 case 3:
260 updResult = s.upsert( i );
261 EXPECT_TRUE( updResult.first );
262 EXPECT_TRUE( updResult.second );
263 break;
264 }
265
266 ASSERT_TRUE( s.contains( i.nKey ));
267 ASSERT_TRUE( s.contains( i ));
268 ASSERT_TRUE( s.contains( other_item( i.key()), other_less()));
269 EXPECT_EQ( i.nFindCount, 0u );
270 ASSERT_TRUE( s.find( i.nKey, []( value_type& v, int ) { ++v.nFindCount; } ));
271 EXPECT_EQ( i.nFindCount, 1u );
272 ASSERT_TRUE( s.find_with( other_item( i.key()), other_less(), []( value_type& v, other_item const& ) { ++v.nFindCount; } ));
273 EXPECT_EQ( i.nFindCount, 2u );
274 ASSERT_TRUE( s.find( i.nKey ) != s.end());
275 ASSERT_TRUE( s.find_with( other_item( i.key()), other_less()) != s.end());
276 EXPECT_EQ( s.find( i.nKey )->nKey, i.key());
277 EXPECT_EQ( s.find_with( other_item( i.key()), other_less())->nKey, i.key());
278 }
279 ASSERT_FALSE( s.empty());
280 ASSERT_CONTAINER_SIZE( s, nSetSize );
281
282 std::for_each( data.begin(), data.end(), []( value_type& v ) { v.clear_stat(); });
283
284 // erase
285 shuffle( indices.begin(), indices.end());
286 for ( auto idx : indices ) {
287 auto& i = data[ idx ];
288
289 ASSERT_TRUE( s.contains( i.nKey ));
290 ASSERT_TRUE( s.contains( i ));
291 ASSERT_TRUE( s.contains( other_item( i.key()), other_less()));
292 EXPECT_EQ( i.nFindCount, 0u );
293 ASSERT_TRUE( s.find( i.nKey, []( value_type& v, int ) { ++v.nFindCount; } ));
294 EXPECT_EQ( i.nFindCount, 1u );
295 ASSERT_TRUE( s.find_with( other_item( i.key()), other_less(), []( value_type& v, other_item const& ) { ++v.nFindCount; } ));
296 EXPECT_EQ( i.nFindCount, 2u );
297 ASSERT_TRUE( s.find( i.nKey ) != s.end());
298 ASSERT_TRUE( s.find_with( other_item( i.key()), other_less()) != s.end());
299 EXPECT_EQ( s.find( i.nKey )->nKey, i.key());
300 EXPECT_EQ( s.find_with( other_item( i.key()), other_less())->nKey, i.key());
301
302
303 value_type v( i );
304 switch ( i.key() % 6 ) {
305 case 0:
306 ASSERT_FALSE( s.unlink( v ));
307 ASSERT_TRUE( s.unlink( i ));
308 ASSERT_FALSE( s.unlink( i ));
309 break;
310 case 1:
311 ASSERT_TRUE( s.erase( i.key()));
312 ASSERT_FALSE( s.erase( i.key()));
313 break;
314 case 2:
315 ASSERT_TRUE( s.erase( v ));
316 ASSERT_FALSE( s.erase( v ));
317 break;
318 case 3:
319 ASSERT_TRUE( s.erase_with( other_item( i.key()), other_less()));
320 ASSERT_FALSE( s.erase_with( other_item( i.key()), other_less()));
321 break;
322 case 4:
323 EXPECT_EQ( i.nEraseCount, 0u );
324 ASSERT_TRUE( s.erase( v, []( value_type& val ) { ++val.nEraseCount; } ));
325 EXPECT_EQ( i.nEraseCount, 1u );
326 ASSERT_FALSE( s.erase( v, []( value_type& val ) { ++val.nEraseCount; } ));
327 EXPECT_EQ( i.nEraseCount, 1u );
328 break;
329 case 5:
330 EXPECT_EQ( i.nEraseCount, 0u );
331 ASSERT_TRUE( s.erase_with( other_item( i.key()), other_less(), []( value_type& val ) { ++val.nEraseCount; } ));
332 EXPECT_EQ( i.nEraseCount, 1u );
333 ASSERT_FALSE( s.erase_with( other_item( i.key()), other_less(), []( value_type& val ) { ++val.nEraseCount; } ));
334 EXPECT_EQ( i.nEraseCount, 1u );
335 break;
336 }
337
338 ASSERT_FALSE( s.contains( i.nKey ));
339 ASSERT_FALSE( s.contains( i ));
340 ASSERT_FALSE( s.contains( other_item( i.key()), other_less()));
341 ASSERT_FALSE( s.find( i.nKey, []( value_type&, int ) {} ));
342 ASSERT_FALSE( s.find_with( other_item( i.key()), other_less(), []( value_type&, other_item const& ) {} ));
343 ASSERT_TRUE( s.find( i.nKey ) == s.end());
344 ASSERT_TRUE( s.find_with( other_item( i.key()), other_less()) == s.end());
345 }
346 ASSERT_TRUE( s.empty());
347 ASSERT_CONTAINER_SIZE( s, 0u );
348
349 // Force retiring cycle
350 Set::gc::force_dispose();
351 for ( auto& i : data ) {
352 EXPECT_EQ( i.nDisposeCount, 1u );
353 }
354
355 // clear
356 for ( auto& i : data ) {
357 i.clear_stat();
358 ASSERT_TRUE( s.insert( i ));
359 }
360 ASSERT_FALSE( s.empty());
361 ASSERT_CONTAINER_SIZE( s, nSetSize );
362
363 // Iterator test
364 for ( auto it = s.begin(); it != s.end(); ++it ) {
365 ++it->nFindCount;
366 }
367 for ( auto it = s.cbegin(); it != s.cend(); ++it ) {
368 EXPECT_EQ( it->nFindCount, 1u );
369 }
370 for ( auto& i : data ) {
371 EXPECT_EQ( i.nFindCount, 1u );
372 }
373
374 // erase_at() test
375 for ( auto it = s.begin(); it != s.end(); ++it ) {
376 EXPECT_TRUE( s.erase_at( it ));
377 EXPECT_FALSE( s.erase_at( it ));
378 }
379
380 ASSERT_TRUE( s.empty());
381 ASSERT_CONTAINER_SIZE( s, 0u );
382 ASSERT_TRUE( s.begin() == s.end());
383 ASSERT_TRUE( s.cbegin() == s.cend());
384
385 // Force retiring cycle
386 Set::gc::force_dispose();
387 for ( auto& i : data ) {
388 EXPECT_EQ( i.nDisposeCount, 1u );
389 }
390
391 }
392 };
