16// defer() is a macro to defer execution of a statement until the surrounding
17// scope is closed and is typically used to perform cleanup logic once a
18// function returns.
19//
20// Note: Unlike golang's defer(), the defer statement is executed when the
21// surrounding *scope* is closed, not necessarily the function.
22//
23// Example usage:
24//
25// void sayHelloWorld()
26// {
27// defer(printf("world\n"));
28// printf("hello ");
29// }
30//

194void parallel(Iter begin, Iter end, size_t batch, F f, size_t inplace_batch_threahold = 1ull) {
195 auto n = _distance(begin, end);
196 size_t batchCount = (n + batch - 1) / batch;
197 if (batchCount < inplace_batch_threahold) {
198 for (size_t i = 0; i < batchCount; ++i) {
199 auto toAdvance = std::min((size_t)n, batch);
200 auto l = begin;
201 auto r = begin;
202 _advance(r, toAdvance);
203 n -= toAdvance;
204 f(l, r);
205 begin = r;
206 }
207 } else {
208 auto thread_count = std::clamp<size_t>(batchCount, 1u, std::thread::hardware_concurrency());

215 f(begin + begin_idx, begin + end_idx);

217 counter.done();
