73template <typename T>
74constexpr size_t floor_log2(T n) {
75
76 static_assert(std::is_unsigned_v<T>, "log2 only supports unsigned integer types");
77
78#if defined(_MSC_VER)
79 unsigned long index;
80 if constexpr (sizeof(T) == 8) {
81 _BitScanReverse64(&index, n);
82 } else {
83 _BitScanReverse(&index, static_cast<unsigned long>(n));
84 }
85 return static_cast<size_t>(index);
86#elif defined(__GNUC__) || defined(__clang__)
87 if constexpr (sizeof(T) == 8) {
88 return 63 - __builtin_clzll(n);
89 } else {
90 return 31 - __builtin_clz(n);
91 }
92#else
93 // Portable fallback: Uses bit shifts to count leading zeros manually
94 size_t log = 0;
95 while (n >>= 1) {
96 ++log;
97 }
98 return log;
99#endif
100}

343constexpr size_t coprime(size_t N) {
344 if(N < 3) {
345 return 1;
346 }
347 for (size_t x = N; --x > 0;) {
348 if (std::gcd(x, N) == 1) {
349 return x;
350 }
351 }
352 return 1;
353}
