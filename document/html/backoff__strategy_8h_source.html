230 template <typename Traits = exponential_const_traits >
231 class exponential
232 {
233 public:
234 typedef Traits traits;
235
236 typedef typename traits::fast_path_backoff spin_backoff ;
237 typedef typename traits::slow_path_backoff yield_backoff ;
238
239 protected:
240 size_t m_nExpCur ;
241
242 spin_backoff m_bkSpin ;
243 yield_backoff m_bkYield ;
244
245 public:
247 exponential() noexcept
248 : m_nExpCur( traits::lower_bound )
249 {}
250
251 //@cond
252 void operator ()() noexcept(noexcept(std::declval<spin_backoff>()()) && noexcept(std::declval<yield_backoff>()()))
253 {
254 if ( m_nExpCur <= traits::upper_bound ) {
255 for ( size_t n = 0; n < m_nExpCur; ++n )
256 m_bkSpin();
257 m_nExpCur *= 2;
258 }
259 else
260 m_bkYield();
261 }
262
263 template <typename Predicate>
264 bool operator()( Predicate pr ) noexcept( noexcept(std::declval<Predicate>()()) && noexcept(std::declval<spin_backoff>()()) && noexcept(std::declval<yield_backoff>()()))
265 {
266 if ( m_nExpCur <= traits::upper_bound ) {
267 for ( size_t n = 0; n < m_nExpCur; ++n ) {
268 if ( m_bkSpin(pr))
269 return true;
270 }
271 m_nExpCur *= 2;
272 }
273 else
274 return m_bkYield(pr);
275 return false;
276 }
277
278 void reset() noexcept( noexcept( std::declval<spin_backoff>().reset()) && noexcept( std::declval<yield_backoff>().reset()))
279 {
280 m_nExpCur = traits::lower_bound;
281 m_bkSpin.reset();
282 m_bkYield.reset();
283 }
284 //@endcond
285 };
