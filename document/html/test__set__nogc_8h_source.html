213 std::vector< value_type > data;

216 indices.reserve( kSize );
217 for ( size_t key = 0; key < kSize; ++key ) {

219 indices.push_back( key );
220 }
221 shuffle( indices.begin(), indices.end());
222
223 // insert/find
224 for ( auto idx : indices ) {
225 auto& i = data[idx];
226
227 ASSERT_FALSE( s.contains( i.nKey ) != s.end());
228 ASSERT_FALSE( s.contains( i ) != s.end());
229 ASSERT_FALSE( s.contains( other_item( i.key()), other_less()) != s.end());
230
231 std::pair<typename Set::iterator, bool> updResult;
232
233 std::string str;
234 updResult = s.update( i.key(), false );
235 EXPECT_TRUE( updResult.first == s.end());
236 EXPECT_FALSE( updResult.second );
237
238 typename Set::iterator it;
239 switch ( idx % 6 ) {
240 case 0:
241 it = s.insert( i );
242 ASSERT_TRUE( it != s.end());
243 it->nFindCount = it->nKey;
244 ASSERT_TRUE( s.insert( i ) == s.end());
245 updResult = s.update( i, false );
246 EXPECT_TRUE( updResult.first == it );
247 EXPECT_FALSE( updResult.second );
248 break;
249 case 1:
250 it = s.insert( i.key());
251 ASSERT_TRUE( it != s.end());
252 ASSERT_TRUE( s.insert( i.key()) == s.end());
253 it->nFindCount = it->nKey;
254 updResult = s.update( i.key(), false );
255 EXPECT_TRUE( updResult.first == it );
256 EXPECT_FALSE( updResult.second );
257 break;
258 case 2:
259 updResult = s.update( i );
260 EXPECT_TRUE( updResult.first != s.end());
261 EXPECT_TRUE( updResult.second );
262 updResult.first->nFindCount = updResult.first->nKey;
263 break;
264 case 3:
265 updResult = s.update( i.key());
266 EXPECT_TRUE( updResult.first != s.end());
267 EXPECT_TRUE( updResult.second );
268 updResult.first->nFindCount = updResult.first->nKey;
269 break;
270 case 4:
271 it = s.emplace( i.key());
272 ASSERT_TRUE( it != s.end());

274 ASSERT_FALSE( s.emplace( i.key()) != s.end());
275 break;
