241 template <typename Q, typename Func>
242 std::pair<bool, bool> update( const Q& val, Func func, bool bAllowInsert = true )
243 {
244 scoped_node_ptr sp( cxx_leaf_node_allocator().New( val ));
245 std::pair<bool, bool> bRes = base_class::update( *sp,
246 [&func, &val](bool bNew, leaf_node& node, leaf_node&){ func( bNew, node.m_Value, val ); },
247 bAllowInsert );
248 if ( bRes.first && bRes.second )
249 sp.release();
250 return bRes;
251 }
