29 template <class GC, typename Tag = opt::none>
30 class node
31 {
32 public:
33 typedef GC gc;
34 typedef Tag tag;
35
36 typedef cds::details::marked_ptr<node, 1> marked_ptr;
37 typedef typename gc::template atomic_marked_ptr< marked_ptr> atomic_marked_ptr;
38 //@cond
39 typedef atomic_marked_ptr tower_item_type;
40
41 //@endcond
42
43 protected:
44 //@cond
45 atomic_marked_ptr m_pNext;
46 unsigned int m_nHeight;
47 atomic_marked_ptr * m_arrNext;
48 atomics::atomic<unsigned int> m_nUnlink;
49 //@endcond
50
51 public:
52 node()
53 : m_pNext( nullptr )
54 , m_nHeight( 1 )
55 , m_arrNext( nullptr )
56 {
57 m_nUnlink.store( 1, atomics::memory_order_release );
58 }
59
60
62 void make_tower( unsigned int nHeight, atomic_marked_ptr * nextTower )
63 {
64 assert( nHeight > 0 );
65 assert( (nHeight == 1 && nextTower == nullptr) // bottom-list node
66 || (nHeight > 1 && nextTower != nullptr) // node at level of more than 0
67 );
68
69 m_arrNext = nextTower;
70 m_nHeight = nHeight;
71 m_nUnlink.store( nHeight, atomics::memory_order_release );
72 }
73
74 //@cond
75 atomic_marked_ptr * release_tower()
76 {
77 atomic_marked_ptr * pTower = m_arrNext;
78 m_arrNext = nullptr;
79 m_nHeight = 1;
80 return pTower;
81 }
82
83 atomic_marked_ptr * get_tower() const
84 {
85 return m_arrNext;
86 }
87
88 bool has_tower() const
89 {
90 return m_nHeight > 1;
91 }
92 //@endcond
93
95 atomic_marked_ptr& next( unsigned int nLevel )
96 {
97 assert( nLevel < height());
98 assert( nLevel == 0 || (nLevel > 0 && m_arrNext != nullptr));
99
100 if ( nLevel ) {
101 // TSan: data race between m_arrNext[ nLevel - 1 ] and make_tower()
102 // In fact, m_arrNext is a const array that is never changed
103 CDS_TSAN_ANNOTATE_HAPPENS_BEFORE( &m_arrNext[ nLevel - 1 ] );
104 return m_arrNext[nLevel - 1];
105 }
106 return m_pNext;
107 }
108
110 atomic_marked_ptr const& next( unsigned int nLevel ) const
111 {
112 assert( nLevel < height());
113 assert( nLevel == 0 || nLevel > 0 && m_arrNext != nullptr );
114
115 if ( nLevel ) {
116 CDS_TSAN_ANNOTATE_HAPPENS_BEFORE( &m_arrNext[nLevel - 1] );
117 return m_arrNext[nLevel - 1];
118 }
119 return m_pNext;
120 }
121
123 atomic_marked_ptr& operator[]( unsigned int nLevel )
124 {
125 return next( nLevel );
126 }
127
129 atomic_marked_ptr const& operator[]( unsigned int nLevel ) const
130 {
131 return next( nLevel );
132 }
133
135 unsigned int height() const
136 {
137 return m_nHeight;
138 }
139
141 void clear()
142 {
143 assert( m_arrNext == nullptr );
144 m_pNext.store( marked_ptr(), atomics::memory_order_release );
145 }
146
147 //@cond
148 bool is_cleared() const
149 {
150 return m_pNext == atomic_marked_ptr()
151 && m_arrNext == nullptr
152 && m_nHeight <= 1;
153 }
154
155 bool level_unlinked( unsigned nCount = 1 )
156 {
157 return m_nUnlink.fetch_sub( nCount, atomics::memory_order_relaxed ) == 1;
158 }
159
160 bool is_upper_level( unsigned nLevel ) const
161 {
162 return m_nUnlink.load( atomics::memory_order_relaxed ) == nLevel + 1;
163 }
164 //@endcond
165 };

283 class xor_shift {
284 //@cond
285 atomics::atomic<unsigned int> m_nSeed;
286
287 static_assert( MaxHeight > 1, "MaxHeight" );
288 static_assert( MaxHeight <= c_nHeightLimit, "MaxHeight is too large" );
289 static unsigned int const c_nBitMask = (1u << ( MaxHeight - 1 )) - 1;
290 //@endcond
291
292 public:
294 static unsigned int const c_nUpperBound = MaxHeight;
295
297 xor_shift()
298 {
299 m_nSeed.store( (unsigned int) cds::OS::Timer::random_seed(), atomics::memory_order_relaxed );
300 }
301
303 unsigned int operator()()
304 {
305 /* ConcurrentSkipListMap.java
306 private int randomLevel() {
307 int x = randomSeed;
308 x ^= x << 13;
309 x ^= x >>> 17;
310 randomSeed = x ^= x << 5;
311 if ((x & 0x80000001) != 0) // test highest and lowest bits
312 return 0;
313 int level = 1;
314 while (((x >>>= 1) & 1) != 0) ++level;
315 return level;
316 }
317 */
318 unsigned int x = m_nSeed.load( atomics::memory_order_relaxed );
319 x ^= x << 13;
320 x ^= x >> 17;
321 x ^= x << 5;
322 m_nSeed.store( x, atomics::memory_order_relaxed );
323 unsigned int nLevel = ((x & 0x00000001) != 0) ? 0 : cds::bitop::LSB( (~(x >> 1)) & c_nBitMask );
324
325 assert( nLevel < c_nUpperBound );
326 return nLevel;
327 }
328 };

353 class turbo
354 {
355 //@cond
356 atomics::atomic<unsigned int> m_nSeed;
357
358 static_assert( MaxHeight > 1, "MaxHeight" );
359 static_assert( MaxHeight <= c_nHeightLimit, "MaxHeight is too large" );
360 static unsigned int const c_nBitMask = (1u << ( MaxHeight - 1 )) - 1;
361 //@endcond
362 public:
364 static unsigned int const c_nUpperBound = MaxHeight;
365
367 turbo()
368 {
369 m_nSeed.store( (unsigned int) cds::OS::Timer::random_seed(), atomics::memory_order_relaxed );
370 }
371
373 unsigned int operator()()
374 {
375 /*
376 private int randomLevel() {
377 int level = 0;
378 int r = randomSeed;
379 randomSeed = r * 134775813 + 1;
380 if (r < 0) {
381 while ((r <<= 1) > 0)
382 ++level;
383 }
384 return level;
385 }
386 */
387 /*
388 The low bits are apparently not very random (the original used only
389 upper 16 bits) so we traverse from highest bit down (i.e., test
390 sign), thus hardly ever use lower bits.
391 */
392 unsigned int x = m_nSeed.load( atomics::memory_order_relaxed ) * 134775813 + 1;
393 m_nSeed.store( x, atomics::memory_order_relaxed );
394 unsigned int nLevel = ( x & 0x80000000 ) ? ( c_nUpperBound - 1 - cds::bitop::MSBnz( (x & c_nBitMask ) | 1 )) : 0;
395
396 assert( nLevel < c_nUpperBound );
397 return nLevel;
398 }
399 };
