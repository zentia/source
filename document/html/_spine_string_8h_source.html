182 bool startsWith(const String &needle) const {
183 if (needle.length() > length()) return false;
184 for (int i = 0; i < (int)needle.length(); i++) {
185 if (buffer()[i] != needle.buffer()[i]) return false;
186 }
187 return true;

198 if (startIndex < 0 || startIndex >= (int)_length || length < 0 || startIndex + length > (int)_length) {
199 return String();

202 memcpy(subStr, _buffer + startIndex, length);
