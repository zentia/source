8/* ----------------------------------------------------------------------------
9 We have a special "mini" allocator just for allocation of meta-data like
10 the heap (`mi_heap_t`) or thread-local data (`mi_tld_t`).
11
12 We reuse the bitmap of the arena's for allocation of 64b blocks inside
13 an arena slice (64KiB).
14 We always ensure that meta data is zero'd (we zero on `free`)
15-----------------------------------------------------------------------------*/
16

25#define MI_META_PAGE_SIZE MI_ARENA_SLICE_SIZE
26#define MI_META_PAGE_ALIGN MI_ARENA_SLICE_ALIGN
27
28#define MI_META_BLOCK_SIZE (128) // large enough such that META_MAX_SIZE >= 4k (even on 32-bit)
29#define MI_META_BLOCK_ALIGN MI_META_BLOCK_SIZE
30#define MI_META_BLOCKS_PER_PAGE (MI_META_PAGE_SIZE / MI_META_BLOCK_SIZE) // 512
31#define MI_META_MAX_SIZE (MI_BCHUNK_SIZE * MI_META_BLOCK_SIZE)
32
33typedef struct mi_meta_page_s {
34 _Atomic(struct mi_meta_page_s*) next; // a linked list of meta-data pages (never released)
35 mi_memid_t memid; // provenance of the meta-page memory itself
36 mi_bbitmap_t blocks_free; // a small bitmap with 1 bit per block.
37} mi_meta_page_t;

39static mi_decl_cache_align _Atomic(mi_meta_page_t*) mi_meta_pages = MI_ATOMIC_VAR_INIT(NULL);

42#if MI_DEBUG > 1
43static mi_meta_page_t* mi_meta_page_of_ptr(void* p, size_t* block_idx) {
44 mi_meta_page_t* mpage = (mi_meta_page_t*)((uint8_t*)mi_align_down_ptr(p,MI_META_PAGE_ALIGN) + _mi_os_secure_guard_page_size());
45 if (block_idx != NULL) {
46 *block_idx = ((uint8_t*)p - (uint8_t*)mpage) / MI_META_BLOCK_SIZE;
47 }
48 return mpage;
49}

52static mi_meta_page_t* mi_meta_page_next( mi_meta_page_t* mpage ) {
53 return mi_atomic_load_ptr_acquire(mi_meta_page_t, &mpage->next);
54}

56static void* mi_meta_block_start( mi_meta_page_t* mpage, size_t block_idx ) {
57 mi_assert_internal(_mi_is_aligned((uint8_t*)mpage - _mi_os_secure_guard_page_size(), MI_META_PAGE_ALIGN));
58 mi_assert_internal(block_idx < MI_META_BLOCKS_PER_PAGE);
59 void* p = ((uint8_t*)mpage - _mi_os_secure_guard_page_size() + (block_idx * MI_META_BLOCK_SIZE));
60 mi_assert_internal(mpage == mi_meta_page_of_ptr(p,NULL));
61 return p;
62}

64// allocate a fresh meta page and add it to the global list.
65static mi_meta_page_t* mi_meta_page_zalloc(void) {
66 // allocate a fresh arena slice
67 // note: careful with _mi_subproc as it may recurse into mi_tld and meta_page_zalloc again.. (same with _mi_os_numa_node()...)
68 mi_memid_t memid;
69 uint8_t* base = (uint8_t*)_mi_arenas_alloc_aligned(_mi_subproc(), MI_META_PAGE_SIZE, MI_META_PAGE_ALIGN, 0,
70 true /* commit*/, (MI_SECURE==0) /* allow large? */,
71 NULL /* req arena */, 0 /* thread_seq */, -1 /* numa node */, &memid);
72 if (base == NULL) return NULL;
73 mi_assert_internal(_mi_is_aligned(base,MI_META_PAGE_ALIGN));
74 if (!memid.initially_zero) {
75 _mi_memzero_aligned(base, MI_ARENA_SLICE_SIZE);
76 }
77
78 // guard pages
79 #if MI_SECURE >= 1
80 _mi_os_secure_guard_page_set_at(base, memid.is_pinned);
81 _mi_os_secure_guard_page_set_before(base + MI_META_PAGE_SIZE, memid.is_pinned);
82 #endif
83
84 // initialize the page and free block bitmap
85 mi_meta_page_t* mpage = (mi_meta_page_t*)(base + _mi_os_secure_guard_page_size());
86 mpage->memid = memid;
87 mi_bbitmap_init(&mpage->blocks_free, MI_META_BLOCKS_PER_PAGE, true /* already_zero */);
88 const size_t mpage_size = offsetof(mi_meta_page_t,blocks_free) + mi_bbitmap_size(MI_META_BLOCKS_PER_PAGE, NULL);
89 const size_t info_blocks = _mi_divide_up(mpage_size,MI_META_BLOCK_SIZE);
90 const size_t guard_blocks = _mi_divide_up(_mi_os_secure_guard_page_size(), MI_META_BLOCK_SIZE);
91 mi_assert_internal(info_blocks + 2*guard_blocks < MI_META_BLOCKS_PER_PAGE);
92 mi_bbitmap_unsafe_setN(&mpage->blocks_free, info_blocks + guard_blocks, MI_META_BLOCKS_PER_PAGE - info_blocks - 2*guard_blocks);
93
94 // push atomically in front of the meta page list
95 // (note: there is no ABA issue since we never free meta-pages)
96 mi_meta_page_t* old = mi_atomic_load_ptr_acquire(mi_meta_page_t,&mi_meta_pages);
97 do {
98 mi_atomic_store_ptr_release(mi_meta_page_t, &mpage->next, old);
99 } while(!mi_atomic_cas_ptr_weak_acq_rel(mi_meta_page_t,&mi_meta_pages,&old,mpage));
100 return mpage;
101}

104// allocate meta-data
105mi_decl_noinline void* _mi_meta_zalloc( size_t size, mi_memid_t* pmemid )
106{
107 mi_assert_internal(pmemid != NULL);
108 size = _mi_align_up(size,MI_META_BLOCK_SIZE);
109 if (size == 0 || size > MI_META_MAX_SIZE) return NULL;
110 const size_t block_count = _mi_divide_up(size,MI_META_BLOCK_SIZE);
111 mi_assert_internal(block_count > 0 && block_count < MI_BCHUNK_BITS);
112 mi_meta_page_t* mpage0 = mi_atomic_load_ptr_acquire(mi_meta_page_t,&mi_meta_pages);
113 mi_meta_page_t* mpage = mpage0;
114 while (mpage != NULL) {
115 size_t block_idx;
116 if (mi_bbitmap_try_find_and_clearN(&mpage->blocks_free, block_count, 0, &block_idx)) {
117 // found and claimed `block_count` blocks
118 *pmemid = _mi_memid_create_meta(mpage, block_idx, block_count);
119 return mi_meta_block_start(mpage,block_idx);
120 }
121 else {
122 mpage = mi_meta_page_next(mpage);
123 }
124 }
125 // failed to find space in existing pages
126 if (mi_atomic_load_ptr_acquire(mi_meta_page_t,&mi_meta_pages) != mpage0) {
127 // the page list was updated by another thread in the meantime, retry
128 return _mi_meta_zalloc(size,pmemid);
129 }
130 // otherwise, allocate a fresh metapage and try once more
131 mpage = mi_meta_page_zalloc();
132 if (mpage != NULL) {
133 size_t block_idx;
134 if (mi_bbitmap_try_find_and_clearN(&mpage->blocks_free, block_count, 0, &block_idx)) {
135 // found and claimed `block_count` blocks
136 *pmemid = _mi_memid_create_meta(mpage, block_idx, block_count);
137 return mi_meta_block_start(mpage,block_idx);
138 }
139 }
140 // if all this failed, allocate from the OS
141 return _mi_os_alloc(size, pmemid);
142}

144// free meta-data
145mi_decl_noinline void _mi_meta_free(void* p, size_t size, mi_memid_t memid) {
146 if (p==NULL) return;
147 if (memid.memkind == MI_MEM_META) {
148 mi_assert_internal(_mi_divide_up(size, MI_META_BLOCK_SIZE) == memid.mem.meta.block_count);
149 const size_t block_count = memid.mem.meta.block_count;
150 const size_t block_idx = memid.mem.meta.block_index;
151 mi_meta_page_t* mpage = (mi_meta_page_t*)memid.mem.meta.meta_page;
152 mi_assert_internal(mi_meta_page_of_ptr(p,NULL) == mpage);
153 mi_assert_internal(block_idx + block_count <= MI_META_BLOCKS_PER_PAGE);
154 mi_assert_internal(mi_bbitmap_is_clearN(&mpage->blocks_free, block_idx, block_count));
155 // we zero on free (and on the initial page allocation) so we don't need a "dirty" map
156 _mi_memzero_aligned(mi_meta_block_start(mpage, block_idx), block_count*MI_META_BLOCK_SIZE);
157 mi_bbitmap_setN(&mpage->blocks_free, block_idx, block_count);
158 }
159 else {
160 _mi_arenas_free(p,size,memid);
161 }
162}

164// used for debug output
165bool _mi_meta_is_meta_page(void* p)
166{
167 mi_meta_page_t* mpage0 = mi_atomic_load_ptr_acquire(mi_meta_page_t, &mi_meta_pages);
168 mi_meta_page_t* mpage = mpage0;
169 while (mpage != NULL) {
170 if ((void*)mpage == p) return true;
171 mpage = mi_meta_page_next(mpage);
172 }
173 return false;
174}
