194inline void eastl::safe_object::clear_references()
195{
196 while(mpSafePtrList != NULL)
197 {
198 safe_ptr_base* const pNext = mpSafePtrList->mpNext;
199 mpSafePtrList->mpNext = NULL;
200 mpSafePtrList->mpObject = NULL;
201 mpSafePtrList = pNext;
202 }
203}

206inline eastl::safe_object::~safe_object()
207{
208 safe_ptr_base* pIter = mpSafePtrList;
209
210 while(pIter)
211 {
212 safe_ptr_base* const pNext = pIter->mpNext;
213 pIter->mpNext = NULL;
214 pIter->mpObject = NULL;
215 pIter = pNext;
216 }
217}

227inline void eastl::safe_object::remove(safe_ptr_base* pBase) const
228{
229 // We have a singly-linked list (starting with mpSafePtrList) and need to
230 // remove an element from within it.
231 if(pBase == mpSafePtrList)
232 mpSafePtrList = mpSafePtrList->mpNext;
233 else
234 {
235 for(safe_ptr_base *pPrev = mpSafePtrList, *pCurrent = mpSafePtrList->mpNext;
236 pCurrent;
237 pPrev = pCurrent, pCurrent = pCurrent->mpNext)
238 {
239 if(pCurrent == pBase)
240 {
241 pPrev->mpNext = pCurrent->mpNext;
242 break;
243 }
244 }
245 }
246}
