6// This file implements a doubly-linked list, much like the C++ std::list class.
7// The primary distinctions between this list and std::list are:
8// - list doesn't implement some of the less-frequently used functions
9// of std::list. Any required functions can be added at a later time.
10// - list has a couple extension functions that increase performance.
11// - list can contain objects with alignment requirements. std::list cannot
12// do so without a bit of tedious non-portable effort.
13// - list has optimizations that don't exist in the STL implementations
14// supplied by library vendors for our targeted platforms.
15// - list supports debug memory naming natively.
16// - list::size() by default is not a constant time function, like the list::size
17// in some std implementations such as STLPort and SGI STL but unlike the
18// list in Dinkumware and Metrowerks. The EASTL_LIST_SIZE_CACHE option can change this.
19// - list provides a guaranteed portable node definition that allows users
20// to write custom fixed size node allocators that are portable.
21// - list is easier to read, debug, and visualize.
22// - list is savvy to an environment that doesn't have exception handling,
23// as is sometimes the case with console or embedded environments.
24// - list has less deeply nested function calls and allows the user to
25// enable forced inlining in debug builds in order to reduce bloat.
26// - list doesn't keep a member size variable. This means that list is
27// smaller than std::list (depends on std::list) and that for most operations
28// it is faster than std::list. However, the list::size function is slower.
29// - list::size_type is defined as eastl_size_t instead of size_t in order to
30// save memory and run faster on 64 bit systems.

455 // Swaps the nodes a and b in the lists to which they belong. This is similar to
456 // splicing a into b's list and b into a's list at the same time.
457 // Works by swapping the members of a and b, and fixes up the lists that a and b
458 // were part of to point to the new members.
459 inline void ListNodeBase::swap(ListNodeBase& a, ListNodeBase& b) EA_NOEXCEPT
460 {
461 const ListNodeBase temp(a);
462 a = b;
463 b = temp;
464
465 if(a.mpNext == &b)
466 a.mpNext = a.mpPrev = &a;
467 else
468 a.mpNext->mpPrev = a.mpPrev->mpNext = &a;
469
470 if(b.mpNext == &a)
471 b.mpNext = b.mpPrev = &b;
472 else
473 b.mpNext->mpPrev = b.mpPrev->mpNext = &b;
474 }

477 // splices the [first,last) range from its current list into our list before this node.
478 inline void ListNodeBase::splice(ListNodeBase* first, ListNodeBase* last) EA_NOEXCEPT
479 {
480 // We assume that [first, last] are not within our list.
481 last->mpPrev->mpNext = this;
482 first->mpPrev->mpNext = last;
483 this->mpPrev->mpNext = first;
484
485 ListNodeBase* const pTemp = this->mpPrev;
486 this->mpPrev = last->mpPrev;
487 last->mpPrev = first->mpPrev;
488 first->mpPrev = pTemp;
489 }

492 inline void ListNodeBase::reverse() EA_NOEXCEPT
493 {
494 ListNodeBase* pNode = this;
495 do
496 {
497 EA_ANALYSIS_ASSUME(pNode != NULL);
498 ListNodeBase* const pTemp = pNode->mpNext;
499 pNode->mpNext = pNode->mpPrev;
500 pNode->mpPrev = pTemp;
501 pNode = pNode->mpPrev;
502 }
503 while(pNode != this);
504 }

1051 {

1053 {

1055 // allocator that's unequal to x's allocator, we need to reallocate our elements with
1056 // our current allocator and reallocate it with x's allocator. If the allocators are
1057 // equal then we can use a more optimal algorithm that doesn't reallocate our elements
1058 // but instead can copy them in place.

1068 clear();

1070 #if EASTL_ALLOCATOR_COPY_ENABLED
1071 internalAllocator() = x.internalAllocator();
1072 #endif
1073 }
1074
1075 DoAssign(x.begin(), x.end(), eastl::false_type());
1076 }

1467 template <typename T, typename Allocator>
1468 inline void list<T, Allocator>::splice(const_iterator position, this_type& x)
1469 {
1470 // Splicing operations cannot succeed if the two containers use unequal allocators.
1471 // This issue is not addressed in the C++ 1998 standard but is discussed in the
1472 // LWG defect reports, such as #431. There is no simple solution to this problem.
1473 // One option is to throw an exception. Another option which probably captures the
1474 // user intent most of the time is to copy the range from the source to the dest and
1475 // remove it from the source.
1476
1477 if(internalAllocator() == x.internalAllocator())
1478 {
1479 #if EASTL_LIST_SIZE_CACHE
1480 if(x.mSize)
1481 {
1482 (position.mpNode)->splice(x.internalNode().mpNext, &x.internalNode());
1483 mSize += x.mSize;
1484 x.mSize = 0;
1485 }
1486 #else
1487 if(!x.empty())
1488 (position.mpNode)->splice(x.internalNode().mpNext, &x.internalNode());
1489 #endif
1490 }
1491 else
1492 {
1493 insert(position, x.begin(), x.end());
1494 x.clear();
1495 }
1496 }

1720 template <typename T, typename Allocator>
1721 template <typename Compare>
1722 typename list<T, Allocator>::iterator
1723 list<T, Allocator>::DoSort(iterator i1, iterator end2, size_type n, Compare& compare)
1724 {
1725 // A previous version of this function did this by creating temporary lists,
1726 // but that was incompatible with fixed_list because the sizes could be too big.
1727 // We sort subsegments by recursive descent. Then merge as we ascend.
1728 // Return an iterator to the beginning of the sorted subsegment.
1729 // Start with a special case for small node counts.
1730 switch (n)
1731 {
1732 case 0:
1733 case 1:
1734 return i1;
1735
1736 case 2:
1737 // Potentialy swap these two nodes and return the resulting first of them.
1738 if(compare(*--end2, *i1))
1739 {
1740 end2.mpNode->remove();
1741 end2.mpNode->insert(i1.mpNode);
1742 return end2;
1743 }
1744 return i1;
1745
1746 case 3:
1747 {
1748 // We do a list insertion sort. Measurements showed this improved performance 3-12%.
1749 iterator lowest = i1;
1750
1751 for(iterator current = i1.next(); current != end2; ++current)
1752 {
1753 if(compare(*current, *lowest))
1754 lowest = current;
1755 }
1756
1757 if(lowest == i1)
1758 ++i1;
1759 else
1760 {
1761 lowest.mpNode->remove();
1762 lowest.mpNode->insert(i1.mpNode);
1763 }
1764
1765 if(compare(*--end2, *i1)) // At this point, i1 refers to the second element in this three element segment.
1766 {
1767 end2.mpNode->remove();
1768 end2.mpNode->insert(i1.mpNode);
1769 }
1770
1771 return lowest;
1772 }
1773 }
1774
1775 // Divide the range into two parts are recursively sort each part. Upon return we will have
1776 // two halves that are each sorted but we'll need to merge the two together before returning.
1777 iterator result;
1778 size_type nMid = (n / 2);
1779 iterator end1 = eastl::next(i1, (difference_type)nMid);
1780 i1 = DoSort(i1, end1, nMid, compare); // Return the new beginning of the first sorted sub-range.
1781 iterator i2 = DoSort(end1, end2, n - nMid, compare); // Return the new beginning of the second sorted sub-range.
1782
1783 // If the start of the second list is before the start of the first list, insert the first list
1784 // into the second at an appropriate starting place.
1785 if(compare(*i2, *i1))
1786 {
1787 // Find the position to insert the first list into the second list.
1788 iterator ix = i2.next();
1789 while((ix != end2) && compare(*ix, *i1))
1790 ++ix;
1791
1792 // Cut out the initial segment of the second list and move it to be in front of the first list.
1793 ListNodeBase* i2Cut = i2.mpNode;
1794 ListNodeBase* i2CutLast = ix.mpNode->mpPrev;
1795 result = i2;
1796 end1 = i2 = ix;
1797 ListNodeBase::remove_range(i2Cut, i2CutLast);
1798 i1.mpNode->insert_range(i2Cut, i2CutLast);
1799 }
1800 else
1801 {
1802 result = i1;
1803 end1 = i2;
1804 }
1805
1806 // Merge the two segments. We do this by merging the second sub-segment into the first, by walking forward in each of the two sub-segments.
1807 for(++i1; (i1 != end1) && (i2 != end2); ++i1) // while still working on either segment...
1808 {
1809 if(compare(*i2, *i1)) // If i2 is less than i1 and it needs to be merged in front of i1...
1810 {
1811 // Find the position to insert the i2 list into the i1 list.
1812 iterator ix = i2.next();
1813 while((ix != end2) && compare(*ix, *i1))
1814 ++ix;
1815
1816 // Cut this section of the i2 sub-segment out and merge into the appropriate place in the i1 list.
1817 ListNodeBase* i2Cut = i2.mpNode;
1818 ListNodeBase* i2CutLast = ix.mpNode->mpPrev;
1819 if(end1 == i2)
1820 end1 = ix;
1821 i2 = ix;
1822 ListNodeBase::remove_range(i2Cut, i2CutLast);
1823 i1.mpNode->insert_range(i2Cut, i2CutLast);
1824 }
1825 }
1826
1827 return result;
1828 }
