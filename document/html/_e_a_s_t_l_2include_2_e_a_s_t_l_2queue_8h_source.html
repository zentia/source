172 template <typename T, typename Container>
173 inline queue<T, Container>::queue(std::initializer_list<value_type> ilist)
174 : c() // We could alternatively use c(ilist) here, but that would require c to have an ilist constructor.
175 {
176 // Better solution but requires an insert function.
177 // c.insert(ilist.begin(), ilist.end());
178
179 // Possibly slower solution but doesn't require an insert function.
180 for(typename std::initializer_list<value_type>::iterator it = ilist.begin(); it != ilist.end(); ++it)
181 {
182 const value_type& value = *it;
183 c.push_back(value);
184 }
185 }
