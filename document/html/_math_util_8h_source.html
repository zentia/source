104 struct SP_API Interpolation {
105 virtual float apply(float a) = 0;
106
107 virtual float interpolate(float start, float end, float a) {
108 return start + (end - start) * apply(a);
109 }
110
111 virtual ~Interpolation() {};
112 };

114 struct SP_API PowInterpolation : public Interpolation {
115 PowInterpolation(int power) : power(power) {
116 }
117
118 float apply(float a) {
119 if (a <= 0.5f) return MathUtil::pow(a * 2.0f, (float) power) / 2.0f;
120 return MathUtil::pow((a - 1.0f) * 2.0f, (float) power) / (power % 2 == 0 ? -2.0f : 2.0f) + 1.0f;
121 }
122
123 int power;
124 };
125
126 struct SP_API PowOutInterpolation : public Interpolation {
127 PowOutInterpolation(int power) : power(power) {
128 }
129
130 float apply(float a) {
131 return MathUtil::pow(a - 1, (float) power) * (power % 2 == 0 ? -1.0f : 1.0f) + 1.0f;
132 }
133
134 int power;
135 };
