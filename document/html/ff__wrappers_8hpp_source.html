65 if (this->n->isMultiInput()) {
66 ff_minode* mi = reinterpret_cast<ff_minode*>(this->n);
67 mi->set_running(get_num_feedbackchannels()+1);
68 }
69 return n->svc_init();

72 void * svc(void* in) {
73 // with feedback channels it might be null
74 if (in == nullptr) return n->svc(nullptr);
75 message_t* msg = (message_t*)in;
76
77
78 if (this->n->isMultiInput()) {
79 int channelid = msg->sender;
80 ff_minode* mi = reinterpret_cast<ff_minode*>(this->n);
81 mi->set_input_channelid(channelid, fromInput());
82 if (!this->fromInput()) return n->svc(in);
83 }
84
85 // received a logical EOS
86 if (msg->data.getLen() == 0){
87 this->n->eosnotify(msg->sender);
88 delete msg;
89 return GO_ON;
90 }
91
92 bool datacopied=true;
93 void* inputData = this->n->deserializeF(msg->data, datacopied);
94 if (!datacopied) msg->data.doNotCleanup();
95 delete msg;
96 return n->svc(inputData);
97 }
98
99 void svc_end(){this->n->svc_end();}
100
101 ff_node* getOriginal(){ return this->n; }
102};

109class WrapperOUT: public internal_mo_transformer {
110private:
111 int outchannels; // number of output channels the wrapped node is supposed to have
112 int defaultDestination;
113 int myID;
114 int feedbackChannels, localFeedbacks, remoteFeedbacks;
115public:
116
117 WrapperOUT(ff_node* n, int id, int outchannels=-1, int remoteFeedbacks = 0, bool cleanup=false, int defaultDestination = -1): internal_mo_transformer(this, false), outchannels(outchannels), defaultDestination(defaultDestination), myID(id), remoteFeedbacks(remoteFeedbacks){
118 this->n = n;
119 this->cleanup= cleanup;
120 registerCallback(ff_send_out_to_cbk, this);
121 }
122
123 bool serialize(void* in, int id) {
124 if (localFeedbacks){
125 if (id < localFeedbacks) {
126 if (id == -1) return ff_send_out(in);
127 return ff_send_out_to(in, id);
128 }
129 // from 0 to feedbackChannels-1 are feedback channels
130 // from feedbackChannels to outchannels-1 are forward channels
131 id -= localFeedbacks;
132 }
133
134 message_t* msg = new message_t;
135 msg->feedback = (id < remoteFeedbacks && remoteFeedbacks);
136 if (!msg->feedback) id -= remoteFeedbacks;
137
138 bool datacopied = this->n->serializeF(in, msg->data);
139 msg->sender = myID;
140 msg->chid = id;
141 if (!datacopied) msg->data.freetaskF = this->n->freetaskF;
142 if (localFeedbacks) {
143 // all forward channels are multiplexed in feedbackChannels which coicide with the sender node!
144 ff_send_out_to(msg, localFeedbacks);
145 } else
146 ff_send_out(msg);
147 if (datacopied) this->n->freetaskF(in);
148 return true;
149 }
150
151 int svc_init() {
152 // save the channel id fo the sender, useful for when there are feedbacks in the application
153
154 // these are local feedback channels
155 localFeedbacks = internal_mo_transformer::get_num_feedbackchannels();
156 feedbackChannels = localFeedbacks + remoteFeedbacks;
157
158 if (this->n->isMultiOutput()) {
159 ff_monode* mo = reinterpret_cast<ff_monode*>(this->n);
160 mo->set_virtual_outchannels(outchannels);
161 mo->set_virtual_feedbackchannels(feedbackChannels);
162 }
163
164 return n->svc_init();
165 }
166
167 void * svc(void* in) {
168 void* out = n->svc(in);
169 if (out > FF_TAG_MIN) return out;
170 serialize(out, defaultDestination);
171 return GO_ON;
172 }
173
174 void svc_end(){n->svc_end();}
175
176 int run(bool skip_init=false) {
177 return internal_mo_transformer::run(skip_init);
178 }
179
180
182 size_t get_num_outchannels() const { return outchannels; }
183 size_t get_num_feedbackchannels() const { return feedbackChannels; }
184
185
186 ff::ff_node* getOriginal(){return this->n;}
187
188 static inline bool ff_send_out_to_cbk(void* task, int id,
189 unsigned long retry,
190 unsigned long ticks, void* obj) {
191 return ((WrapperOUT*)obj)->serialize(task, id);
192 }
193};
