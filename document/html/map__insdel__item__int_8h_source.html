172 template <class Map>
173 void do_test( Map& testMap )
174 {
175 typedef Inserter<Map> inserter;
176 typedef Deleter<Map> deleter;
177
178 // Fill the map
179 {
180 std::vector<key_type> v;
181 v.reserve( s_nMapSize );
182 for ( size_t i = 0; i < s_nMapSize; ++i )
183 v.push_back( i );
184 shuffle( v.begin(), v.end());
185 for ( auto i: v )
186 EXPECT_TRUE( testMap.insert( i, i ));
187 }
188
189 cds_test::thread_pool& pool = get_pool();
190 pool.add( new inserter( pool, testMap ), s_nInsertThreadCount );
191 pool.add( new deleter( pool, testMap ), s_nDeleteThreadCount );
192
193 propout() << std::make_pair( "insert_thread_count", s_nInsertThreadCount )
194 << std::make_pair( "delete_thread_count", s_nDeleteThreadCount )
195 << std::make_pair( "map_size", s_nMapSize )
196 << std::make_pair( "goal_item", s_nGoalItem )
197 << std::make_pair( "attempt_count", s_nAttemptCount );
198
199
200 std::chrono::milliseconds duration = pool.run();
201
202 propout() << std::make_pair( "duration", duration );
203
204 size_t nInsertSuccess = 0;
205 size_t nInsertFailed = 0;
206 size_t nDeleteSuccess = 0;
207 size_t nDeleteFailed = 0;
208
209 for ( size_t i = 0; i < pool.size(); ++i ) {
210 cds_test::thread& thr = pool.get( i );
211 switch ( thr.type()) {
212 case insert_thread:
213 {
214 inserter& t = static_cast<inserter&>(thr);
215 EXPECT_EQ( t.m_nInsertSuccess, s_nAttemptCount ) << "thread=" << t.id();
216 nInsertSuccess += t.m_nInsertSuccess;
217 nInsertFailed += t.m_nInsertFailed;
218 }
219 break;
220 case delete_thread:
221 {
222 deleter& t = static_cast<deleter&>(thr);
223 EXPECT_EQ( t.m_nDeleteSuccess, s_nAttemptCount ) << "thread=" << t.id();
224 nDeleteSuccess += t.m_nDeleteSuccess;
225 nDeleteFailed += t.m_nDeleteFailed;
226 }
227 break;
228 default:
229 assert( false );
230 }
231 }
232
233 EXPECT_EQ( nInsertSuccess, nDeleteSuccess );
234 EXPECT_TRUE( testMap.contains( s_nGoalItem ));
235 EXPECT_CONTAINER_SIZE( testMap, s_nMapSize );
236
237 propout()
238 << std::make_pair( "insert_success", nInsertSuccess )
239 << std::make_pair( "insert_failed", nInsertFailed )
240 << std::make_pair( "delete_success", nDeleteSuccess )
241 << std::make_pair( "delete_failed", nDeleteFailed );
242
243 // Check if the map contains all items
244 for ( size_t i = 0; i < s_nMapSize; ++i )
245 EXPECT_TRUE( testMap.contains( i )) << "key=" << i;
246
247 check_before_cleanup( testMap );
248
249 testMap.clear();
250 additional_check( testMap );
251 print_stat( propout(), testMap );
252 additional_cleanup( testMap );
253 }
