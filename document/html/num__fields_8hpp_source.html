4/*
5We infer the number of fields using by figuring out how many fields
6we need to construct it. This is done by implementing the constructible
7concept, see below.
8
9However, there is a problem with C arrays. Suppose you have a struct
10like this:
11
12struct A{
13 int arr[3];
14};
15
16Then, the struct can be initialized like this:
17
18const auto a = A{1, 2, 3};
19
20This is a problem, because a naive logic would believe that A
21has three fields, when in fact it has only one.
22
23That is why we use the constructible concept to get the maximum
24possible number of fields and then try to subdivide them into arrays
25in order to figure out which of these fields is in fact an array.
26
27Basically, for every field there is, we try to squeeze as many variables into
28the potential array as we can without missing variables in subsequent fields.
29This is the purpose of get_nested_array_size().
30*/

110 template <std::size_t index, std::size_t size, std::size_t rest>
111 static consteval std::size_t get_nested_array_size() {
112 if constexpr (size < 1) {
113 return 1;
114 } else if constexpr (constructible_with_nested<index, size, rest>() &&
115 !constructible_with_nested<index, size, rest + 1>()) {
116 return size;
117 } else {
118 return get_nested_array_size<index, size - 1, rest + 1>();

122 template <std::size_t max_args, std::size_t index = 0>
123 static consteval std::size_t find_the_sole_non_empty_base_index() {
124 static_assert(index < max_args);
125 constexpr auto check = []<std::size_t... l, std::size_t... r>(
126 std::index_sequence<l...>,
127 std::index_sequence<r...>) {

129 T{any_empty_base<T>(l)..., any_base<T>(0), any_empty_base<T>(r)...};

134 std::make_index_sequence<max_args - index - 1>())) {
135 return index;
136 } else {
137 return find_the_sole_non_empty_base_index<max_args, index + 1>();
138 }

170 template <std::size_t index, std::size_t max>
171 static consteval std::size_t constructible_no_brace_elision() {
172 static_assert(index <= max);
173 if constexpr (index == max) {
174 return 0;
175 } else {
176 return 1 +
177 constructible_no_brace_elision<
178 index + get_nested_array_size<index, max - index, 0>(), max>();

182 static consteval std::size_t count_fields() {
183 constexpr std::size_t max_agg_args = count_max_args_in_agg_init();
184#if defined(REFLECT_CPP_C_ARRAYS_OR_INHERITANCE) || \
185 defined(REFLECTCPP_C_ARRAYS_OR_INHERITANCE)
186 constexpr std::size_t no_brace_ellison_args =
187 constructible_no_brace_elision<0, max_agg_args>();
188 constexpr std::size_t base_args = base_param_num<no_brace_ellison_args>();
189 if constexpr (no_brace_ellison_args == 0 && base_args == 0) {
190 // Empty struct
191 return 0;
192 } else if constexpr (base_args == no_brace_ellison_args) {
193 // Special case when the derived class is empty.
194 // In such cases the filed number is the fields in base class.
195 // Note that there should be only one base class in this case.
196 return get_nested_base_field_count<
197 find_the_sole_non_empty_base_index<max_agg_args>()>();
198 } else {
199 return no_brace_ellison_args - base_args;
200 }
201#else
202 return max_agg_args;
203#endif
204 }
