450 template <typename Q, typename Compare >
451 bool find_position( Q const& val, position& pos, Compare cmp, bool bStopIfFound, bool bStrictSearch ) const
452 {
453 node_type * pPred;
454 node_type * pSucc;
455 node_type * pCur = nullptr;
456
457 int nCmp = 1;
458
459 unsigned int nHeight = c_nMaxHeight;
460 retry:
461 if ( !bStrictSearch )
462 nHeight = m_nHeight.load( memory_model::memory_order_relaxed );
463 pPred = m_Head.head();
464
465 for ( int nLevel = (int) nHeight - 1; nLevel >= 0; --nLevel ) {
466 while ( true ) {
467 pCur = pPred->next( nLevel ).load( memory_model::memory_order_relaxed );
468
469 if ( !pCur ) {
470 // end of the list at level nLevel - goto next level
471 break;
472 }
473
474 pSucc = pCur->next( nLevel ).load( memory_model::memory_order_relaxed );
475
476 if ( pPred->next( nLevel ).load( memory_model::memory_order_acquire ) != pCur
477 || pCur->next( nLevel ).load( memory_model::memory_order_acquire ) != pSucc )
478 {
479 goto retry;
480 }
481
482 nCmp = cmp( *node_traits::to_value_ptr( pCur ), val );
483 if ( nCmp < 0 )
484 pPred = pCur;
485 else if ( nCmp == 0 && bStopIfFound )
486 goto found;
487 else
488 break;
489 }
490
491 pos.pPrev[ nLevel ] = pPred;
492 pos.pSucc[ nLevel ] = pCur;
493 }
494
495 if ( nCmp != 0 )
496 return false;
497
498 found:
499 pos.pCur = pCur;
500 return pCur && nCmp == 0;
501 }

503 template <typename Func>
504 bool insert_at_position( value_type& val, node_type * pNode, position& pos, Func f )
505 {
506 unsigned int nHeight = pNode->height();
507
508 for ( unsigned int nLevel = 1; nLevel < nHeight; ++nLevel )
509 pNode->next( nLevel ).store( nullptr, memory_model::memory_order_relaxed );
510
511 {
512 node_type * p = pos.pSucc[0];
513 pNode->next( 0 ).store( pos.pSucc[ 0 ], memory_model::memory_order_release );
514 if ( !pos.pPrev[0]->next(0).compare_exchange_strong( p, pNode, memory_model::memory_order_release, memory_model::memory_order_relaxed )) {
515 return false;
516 }
517 f( val );
518 }
519
520 for ( unsigned int nLevel = 1; nLevel < nHeight; ++nLevel ) {
521 node_type * p = nullptr;
522 while ( true ) {
523 node_type * q = pos.pSucc[ nLevel ];
524
525 if ( pNode->next( nLevel ).compare_exchange_strong( p, q, memory_model::memory_order_release, memory_model::memory_order_relaxed )) {
526 p = q;
527 if ( pos.pPrev[nLevel]->next(nLevel).compare_exchange_strong( q, pNode, memory_model::memory_order_release, memory_model::memory_order_relaxed ))
528 break;
529 }
530
531 // Renew insert position
532 find_position( val, pos, key_comparator(), false, true );
533 }
534 }
535 return true;
536 }

650 bool insert( value_type& val )
651 {
652 node_type * pNode = node_traits::to_node_ptr( val );
653 scoped_node_ptr scp( pNode );
654 unsigned int nHeight = pNode->height();
655 bool bTowerOk = nHeight > 1 && pNode->get_tower() != nullptr;
656 bool bTowerMade = false;
657
658 position pos;
659 while ( true )
660 {
661 bool bFound = find_position( val, pos, key_comparator(), true, true );
662 if ( bFound ) {
663 // scoped_node_ptr deletes the node tower if we create it
664 if ( !bTowerMade )
665 scp.release();
666
667 m_Stat.onInsertFailed();
668 return false;
669 }
670
671 if ( !bTowerOk ) {
672 build_node( pNode );
673 nHeight = pNode->height();
674 bTowerMade =
675 bTowerOk = true;
676 }
677
678 if ( !insert_at_position( val, pNode, pos, []( value_type& ) {} )) {
679 m_Stat.onInsertRetry();
680 continue;
681 }
682
683 increase_height( nHeight );
684 ++m_ItemCounter;
685 m_Stat.onAddNode( nHeight );
686 m_Stat.onInsertSuccess();
687 scp.release();
688 return true;
689 }

719 template <typename Func>
720 std::pair<bool, bool> update( value_type& val, Func func, bool bInsert = true )
721 {
722 node_type * pNode = node_traits::to_node_ptr( val );
723 scoped_node_ptr scp( pNode );
724 unsigned int nHeight = pNode->height();
725 bool bTowerOk = nHeight > 1 && pNode->get_tower() != nullptr;
726 bool bTowerMade = false;
727
728 position pos;
729 while ( true )
730 {
731 bool bFound = find_position( val, pos, key_comparator(), true, true );
732 if ( bFound ) {
733 // scoped_node_ptr deletes the node tower if we create it before
734 if ( !bTowerMade )
735 scp.release();
736
737 func( false, *node_traits::to_value_ptr(pos.pCur), val );
738 m_Stat.onUpdateExist();
739 return std::make_pair( true, false );
740 }
741
742 if ( !bInsert ) {
743 scp.release();
744 return std::make_pair( false, false );
745 }
746
747 if ( !bTowerOk ) {
748 build_node( pNode );
749 nHeight = pNode->height();
750 bTowerMade =
751 bTowerOk = true;
752 }
753
754 if ( !insert_at_position( val, pNode, pos, [&func]( value_type& item ) { func( true, item, item ); })) {
755 m_Stat.onInsertRetry();
756 continue;
757 }
758
759 increase_height( nHeight );
760 ++m_ItemCounter;
761 scp.release();
762 m_Stat.onAddNode( nHeight );
763 m_Stat.onUpdateNew();
764 return std::make_pair( true, true );
765 }
