217 int setMappingList(const char* str) {
218 rrcnt = 0; // reset rrcnt
219
220 if (str == NULL) return -1; // use the previous mapping list
221 char* _str = const_cast<char*>(str), *_str_end;
222 svector<int> List(64);
223 do {
224 while (*_str == ' ' || *_str == '\t' || *_str == ',')
225 ++_str;
226 unsigned cpuid = strtoul(_str, &_str_end, 0);
227 if (_str == _str_end) {
228 error("setMapping, invalid mapping string\n");
229 return -1;
230 }
231 if (cpuid > (num_cores - 1)) {
232 error("setMapping, invalid cpu id in the mapping string\n");
233 return -1;
234 }
235 _str = _str_end;
236 List.push_back(cpuid);
237
238 if (*_str == '\0')
239 break;
240 } while (1);
241
242 unsigned int size = (unsigned int) List.size();
243 int ret = size;
244 if (!isPowerOf2(size)) {
245 size = nextPowerOf2(size);
246 List.reserve(size);
247 }
248 mask = size - 1;
249 for (size_t i = List.size(), j = 0; i < size; ++i, j++)
250 List.push_back(List[j]);
251 CList = List;
252 return ret;
253 }

255 void setMappingList(const std::vector<size_t> &mapping) {
256 rrcnt = 0; // reset rrcnt
257
258 if ((mapping.size() > (mask+1)) || (mapping.size()==0)) {
259 error("Invalid pinng vector: ignoring it\n");
260 return; // use the previous mapping list
261 }
262 svector<int> List(mask + 1);
263 for (size_t i=0; i<mapping.size(); ++i) {
264 auto cpuid = mapping[i];
265 if (cpuid > (num_cores - 1)) {
266 error("setMapping, invalid cpu id in the mapping string\n");
267 return;
268 }
269 List.push_back(cpuid);
270 }
271
272 unsigned int size = (unsigned int) List.size();
273 if (!isPowerOf2(size)) {
274 size = nextPowerOf2(size);
275 List.reserve(size);
276 }
277 mask = size - 1;
278 for (size_t i = List.size(), j = 0; i < size; ++i, j++)
279 List.push_back(List[j]);
280 CList = List;
281 }
