57 template <class T>
58 rfl::Result<T> to_basic_type(const InputVarType& _var) const noexcept {
59 const auto type = _var.val_.getType();
60 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
61 if (type != capnp::DynamicValue::TEXT) {
62 return error("Could not cast to string.");
63 }
64 return std::string(_var.val_.as<capnp::Text>().cStr());
65
66 } else if constexpr (std::is_same<std::remove_cvref_t<T>,
67 rfl::Bytestring>()) {
68 if (type != capnp::DynamicValue::DATA) {
69 return error("Could not cast to bytestring.");
70 }
71 const auto data = _var.val_.as<capnp::Data>();
72 const auto begin = internal::ptr_cast<const std::byte*>(data.begin());
73 return rfl::Bytestring(begin, begin + data.size());
74
75 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
76 if (type != capnp::DynamicValue::BOOL) {
77 return error("Could not cast to boolean.");
78 }
79 return _var.val_.as<bool>();
80
81 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
82 if (type != capnp::DynamicValue::FLOAT) {
83 return error("Could not cast to float.");
84 }
85 return static_cast<T>(_var.val_.as<double>());
86
87 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
88 switch (type) {
89 case capnp::DynamicValue::INT:
90 return static_cast<T>(_var.val_.as<int64_t>());
91
92 case capnp::DynamicValue::UINT:
93 return static_cast<T>(_var.val_.as<uint64_t>());
94
95 default:
96 return error("Could not cast to an integer.");
97 }
98
99 } else if constexpr (internal::is_literal_v<T>) {
100 if (type != capnp::DynamicValue::ENUM) {
101 return error("Could not cast to an enum.");
102 }
103 return T::from_value(_var.val_.as<capnp::DynamicEnum>().getRaw());
104
105 } else {
106 static_assert(rfl::always_false_v<T>, "Unsupported type.");
107 }
108 }

153 template <class MapReader>
154 std::optional<Error> read_map(const MapReader& _map_reader,
155 const InputMapType& _map) const noexcept {
156 try {
157 const auto entries = _map.val_.get("entries").as<capnp::DynamicList>();
158 for (auto entry : entries) {
159 auto s = entry.template as<capnp::DynamicStruct>();
160 const char* key = s.get("key").as<capnp::Text>().cStr();
161 _map_reader.read(std::string_view(key), InputVarType{s.get("value")});
162 }
163 return std::nullopt;
164 } catch (std::exception& e) {
