47 static Result<T> read(const R& _r, const InputVarType& _var) noexcept {
48 if constexpr (internal::has_read_reflector<T>) {
49 const auto wrap_in_t = [](auto _named_tuple) -> Result<T> {
50 try {
51 return Reflector<T>::to(_named_tuple);
52 } catch (std::exception& e) {
53 return error(e.what());
54 }
55 };
56 return Parser<R, W, typename Reflector<T>::ReflType,
57 ProcessorsType>::read(_r, _var)
58 .and_then(wrap_in_t);
59
60 } else if constexpr (schemaful::IsSchemafulReader<R> &&
61 internal::is_literal_v<T>) {
62 return _r.template to_basic_type<T>(_var);
63
64 } else if constexpr (R::template has_custom_constructor<T>) {
65 return _r.template use_custom_constructor<T>(_var);
66
67 } else {
68 if constexpr (internal::has_reflection_type_v<T>) {
69 using ReflectionType = std::remove_cvref_t<typename T::ReflectionType>;
70 const auto wrap_in_t = [](auto _named_tuple) -> Result<T> {
71 try {
72 return T{std::move(_named_tuple)};
73 } catch (std::exception& e) {
74 return error(e.what());
75 }
76 };
77 return Parser<R, W, ReflectionType, ProcessorsType>::read(_r, _var)
78 .and_then(wrap_in_t);
79
80 } else if constexpr (std::is_class_v<T> && std::is_aggregate_v<T>) {
81 if constexpr (ProcessorsType::default_if_missing_) {
82 return read_struct_with_default(_r, _var);
83 } else {
84 return read_struct(_r, _var);
85 }
86
87 } else if constexpr (std::is_enum_v<T>) {
88 if constexpr (ProcessorsType::underlying_enums_ ||
89 schemaful::IsSchemafulReader<R>) {
90 return static_cast<T>(
91 *_r.template to_basic_type<std::underlying_type_t<T>>(_var));
92 } else {
93 using StringConverter = internal::enums::StringConverter<T>;
94 return _r.template to_basic_type<std::string>(_var).and_then(
95 StringConverter::string_to_enum);
96 }
97
98 } else {
99 return _r.template to_basic_type<std::remove_cvref_t<T>>(_var);
100 }
101 }
102 }

104 template <class P>
105 static void write(const W& _w, const T& _var, const P& _parent) noexcept {
106 if constexpr (internal::has_write_reflector<T>) {
107 Parser<R, W, typename Reflector<T>::ReflType, ProcessorsType>::write(
108 _w, Reflector<T>::from(_var), _parent);
109
110 } else if constexpr (schemaful::IsSchemafulWriter<W> &&
111 internal::is_literal_v<T>) {
112 ParentType::add_value(_w, _var, _parent);
113
114 } else if constexpr (internal::has_reflection_type_v<T>) {
115 using ReflectionType = std::remove_cvref_t<typename T::ReflectionType>;
116 if constexpr (internal::has_reflection_method_v<T>) {
117 Parser<R, W, ReflectionType, ProcessorsType>::write(
118 _w, _var.reflection(), _parent);
119 } else {
120 const auto& [r] = _var;
121 Parser<R, W, ReflectionType, ProcessorsType>::write(_w, r, _parent);
122 }
123
124 } else if constexpr (std::is_class_v<T> && std::is_aggregate_v<T>) {
125 const auto ptr_named_tuple = ProcessorsType::template process<T>(
126 internal::to_ptr_named_tuple(_var));
127 using PtrNamedTupleType = std::remove_cvref_t<decltype(ptr_named_tuple)>;
128 Parser<R, W, PtrNamedTupleType, ProcessorsType>::write(
129 _w, ptr_named_tuple, _parent);
130
131 } else if constexpr (std::is_enum_v<T>) {
132 if constexpr (ProcessorsType::underlying_enums_ ||
133 schemaful::IsSchemafulWriter<W>) {
134 const auto val = static_cast<std::underlying_type_t<T>>(_var);
135 ParentType::add_value(_w, val, _parent);
136 } else {
137 using StringConverter = internal::enums::StringConverter<T>;
138 const auto str = StringConverter::enum_to_string(_var);
139 ParentType::add_value(_w, str, _parent);
140 }
141
142 } else {
143 ParentType::add_value(_w, _var, _parent);
144 }

148 static schema::Type to_schema(
149 std::map<std::string, schema::Type>* _definitions) {
150 using U = std::remove_cvref_t<T>;
151 using Type = schema::Type;
152 if constexpr (std::is_same<U, bool>()) {
153 return Type{Type::Boolean{}};
154
155 } else if constexpr (std::is_same<U, rfl::Bytestring>()) {
156 return Type{Type::Bytestring{}};
157
158 } else if constexpr (std::is_same<U, std::int32_t>()) {
159 return Type{Type::Int32{}};
160
161 } else if constexpr (std::is_same<U, std::int64_t>()) {
162 return Type{Type::Int64{}};
163
164 } else if constexpr (std::is_same<U, std::uint32_t>()) {
165 return Type{Type::UInt32{}};
166
167 } else if constexpr (std::is_same<U, std::uint64_t>()) {
168 return Type{Type::UInt64{}};
169
170 } else if constexpr (std::is_integral<U>()) {
171 return Type{Type::Integer{}};
172
173 } else if constexpr (std::is_same<U, float>()) {
174 return Type{Type::Float{}};
175
176 } else if constexpr (std::is_floating_point_v<U>) {
177 return Type{Type::Double{}};
178
179 } else if constexpr (std::is_same<U, std::string>()) {
180 return Type{Type::String{}};
181
182 } else if constexpr (rfl::internal::is_description_v<U>) {
183 return make_description<U>(_definitions);
184
185 } else if constexpr (std::is_enum_v<U>) {
186 return make_enum<U>(_definitions);
187
188 } else if constexpr (std::is_class_v<U> && std::is_aggregate_v<U>) {
189 return make_reference<U>(_definitions);
190
191 } else if constexpr (internal::is_literal_v<U>) {
192 return Type{Type::Literal{.values_ = U::strings()}};
193
194 } else if constexpr (internal::is_validator_v<U>) {
195 return make_validated<U>(_definitions);
196
197 } else if constexpr (internal::has_reflection_type_v<U>) {
198 return make_reference<U>(_definitions);
199
200 } else {
201 static_assert(rfl::always_false_v<U>, "Unsupported type.");
202 }

233 template <class U>
234 static schema::Type make_reference(
235 std::map<std::string, schema::Type>* _definitions) {
236 using Type = schema::Type;
237 const auto name = make_type_name<U>();
238 if (_definitions->find(name) == _definitions->end()) {
239 (*_definitions)[name] =
240 Type{Type::Integer{}}; // Placeholder to avoid infinite loop.
241 if constexpr (internal::has_reflection_type_v<U>) {
242 (*_definitions)[name] =
243 Parser<R, W, typename U::ReflectionType, ProcessorsType>::to_schema(
244 _definitions);
245 } else {
246 using NamedTupleType = internal::processed_t<U, ProcessorsType>;
247 (*_definitions)[name] =
248 Parser<R, W, NamedTupleType, ProcessorsType>::to_schema(
249 _definitions);
250 }
251 }
252 return Type{Type::Reference{name}};

272 static Result<T> read_struct(const R& _r, const InputVarType& _var) {
273 alignas(T) unsigned char buf[sizeof(T)]{};
274 auto ptr = internal::ptr_cast<T*>(&buf);
275 auto view = ProcessorsType::template process<T>(to_view(*ptr));
276 using ViewType = std::remove_cvref_t<decltype(view)>;
277 const auto [set, err] =
278 Parser<R, W, ViewType, ProcessorsType>::read_view(_r, _var, &view);
279 if (err) [[unlikely]] {
280 call_destructors_where_necessary(set, &view);
281 return error(err->what());
282 }
283 auto res = Result<T>(std::move(*ptr));
284 call_destructors_where_necessary(set, &view);
285 return res;
286 }

292 static Result<T> read_struct_with_default(const R& _r,
293 const InputVarType& _var) {
294 auto t = T{};

296 using ViewType = decltype(view);
297 const auto err =
298 Parser<R, W, ViewType, ProcessorsType>::read_view_with_default(_r, _var,
299 &view);
300 if (err) [[unlikely]] {
301 return error(*err);
302 }
303 return t;
304 }
