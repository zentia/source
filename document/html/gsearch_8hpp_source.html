112 Emitter(const std::vector<ff_node*> &W, const long &foundflag, const bool all=false):
113 foundflag(foundflag),W(W),start(NULL),counter(0),all(all) {}
114
115 void setStart(T*const n) { start = n;}
116 void setNodeToSearch(T*const n, bool a=false) {
117 all = a;
118 for(auto w: W) ((Worker*)w)->setNodeToSearch(n,all);

121 inline int svc_init() {
122 mask.reset();
123 counter=0;
124 return 0;
125 }
126 inline void *svc(void *t) {
127 if (t == NULL) {
128 if (start==NULL) return NULL;
129 mask.set(start->getId());
130 ff_send_out((void*)start);
131 ++counter;
132 std::deque<T*> outNodes;
133 start->out_nodes(outNodes);
134 for(T *n: outNodes) {
135 ++counter;
136 mask.set(n->getId());
137 ff_send_out((void*)n);
138 }
139 return GO_ON;
140 }
141 --counter;
142 if (!all && foundflag>0) return NULL;
143 const T &node = *(static_cast<T*>(t));
144
145 std::deque<T*> outNodes;
146 node.out_nodes(outNodes);
147 auto k =0;
148 for(T *n: outNodes) {
149 if (!all && (++k == CHECK_FOUND_N)) {
150 if (foundflag>0) return NULL;
151 k=0;
152 }
153 if (!mask.test(n->getId())) {
154 ++counter;
155 mask.set(n->getId());
156 ff_send_out((void*)n);
157 }
158 }
159 if (counter == 0) return NULL;
160 return ((!all && foundflag>0) ? NULL : GO_ON);
161 }
162 protected:
163 const long &foundflag;
164 const std::vector<ff_node*> &W;
165 T *start;
166 unsigned long counter;
167 bool all;
168 std::bitset<N> mask;
169 };

182 ff_graphsearch(const int nw=ff_numCores(), const bool all=false):found(0),start(NULL),all(all) {
183 for(int i=0;i<nw;++i) W.push_back(new Worker(&found,all));
184 farm = new ff_farm(false, 524288*nw, 524800*nw, false, nw, true);
185 farm->add_workers(W);
186 scheduler = new Emitter(W,found);
187 farm->add_emitter(scheduler);
188 farm->wrap_around();
189 if (farm->run_then_freeze()<0) {
190 error("running farm ff_graphsearch\n");
191 } else farm->wait_freezing();
192 }

194 ~ff_graphsearch() {
195 found = 1;
196 if (farm) farm->run_and_wait_end();
197 while(W.size()){
198 Worker * w = static_cast<Worker*>(W.back());
199 delete w;
200 W.pop_back();
201 }
202 if (scheduler) delete scheduler;
203 if (farm) delete farm;
204 }

209 // One shot search. It returns just one result (valid only if the return value is true)
210 inline bool search(T *const st, T* const search, T *&result, const int nw=-1) {
211 if (nw > (int)W.size()) {
212 error("ff_graphsearch:search: nw too big, using nw=%d\n", W.size());
213 }
214 found = 0;
215 scheduler->setStart(st);
216 scheduler->setNodeToSearch(search);
217 resetqueues();
218 farm->run_then_freeze(nw);
219 farm->wait_freezing();
220 if (found > 0) {
221 for(auto w : W) {
222 const std::deque<T*> &r = ((Worker*)w)->Found();
223 if (r.size()) {
224 result = r.back();
225 return true;
226 }
227 }
228 }
229 return false;
230 }

236 }
237 found = 0;

239 scheduler->setNodeToSearch(search,true);
240 resetqueues();

242 farm->wait_freezing();
243 if (found > 0) {
244 for(auto w : W) {

246 if (r.size()) {

248 }
249 }

251 }
252 return false;
