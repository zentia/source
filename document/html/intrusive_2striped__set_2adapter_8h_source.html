225 std::pair<bool, bool> update( value_type& val, Func f, bool bAllowInsert )
226 {
227 if ( bAllowInsert ) {
228 std::pair<iterator, bool> res = m_Set.insert( val );
229 f( res.second, *res.first, val );
230 return std::make_pair( true, res.second );
231 }

233 auto it = m_Set.find( val, key_comparator());
234 if ( it == m_Set.end())
