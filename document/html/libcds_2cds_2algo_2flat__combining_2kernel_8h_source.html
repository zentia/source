291 publication_record_type * acquire_record()
292 {
293 publication_record_type * pRec = m_pThreadRec.get();
294 if ( !pRec ) {
295 // Allocate new publication record
296 pRec = cxx11_allocator().New();
297 m_pThreadRec.reset( pRec );
298 m_Stat.onCreatePubRecord();
299
300 // Insert in allocated list
301 assert( m_pAllocatedHead != nullptr );
302 publication_record* p = m_pAllocatedHead->pNextAllocated.load( memory_model::memory_order_relaxed );
303 do {
304 pRec->pNextAllocated.store( p, memory_model::memory_order_release );
305 } while ( !m_pAllocatedHead->pNextAllocated.compare_exchange_weak( p, pRec, memory_model::memory_order_release, atomics::memory_order_acquire ));
306
307 publish( pRec );
308 }
309 else if ( pRec->nState.load( memory_model::memory_order_acquire ) != active )
310 publish( pRec );
311
312 assert( pRec->op() == req_EmptyRecord );
313
314 return pRec;

571 void publish( publication_record_type* pRec )
572 {
573 assert( pRec->nState.load( memory_model::memory_order_relaxed ) == inactive );
574
575 pRec->nAge.store( m_nCount.load(memory_model::memory_order_relaxed), memory_model::memory_order_relaxed );
576 pRec->nState.store( active, memory_model::memory_order_relaxed );
577
578 // Insert record to publication list
579 if ( m_pHead != static_cast<publication_record *>(pRec)) {
580 publication_record * p = m_pHead->pNext.load( memory_model::memory_order_relaxed );
581 if ( p != static_cast<publication_record *>( pRec )) {
582 do {
583 pRec->pNext.store( p, memory_model::memory_order_release );
584 // Failed CAS changes p
585 } while ( !m_pHead->pNext.compare_exchange_weak( p, static_cast<publication_record *>(pRec),
586 memory_model::memory_order_release, atomics::memory_order_acquire ));
587 m_Stat.onActivatePubRecord();
588 }
589 }

600 template <class Container>
601 void try_combining( Container& owner, publication_record_type* pRec )
602 {
603 if ( m_Mutex.try_lock()) {
604 // The thread becomes a combiner
605 lock_guard l( m_Mutex, std::adopt_lock_t());
606
607 // The record pRec can be excluded from publication list. Re-publish it
608 republish( pRec );
609
610 combining( owner );
611 assert( pRec->op( memory_model::memory_order_relaxed ) == req_Response );
612 }
613 else {
614 // There is another combiner, wait while it executes our request
615 if ( !wait_for_combining( pRec )) {
616 // The thread becomes a combiner
617 lock_guard l( m_Mutex, std::adopt_lock_t());
618
619 // The record pRec can be excluded from publication list. Re-publish it
620 republish( pRec );
621
622 combining( owner );
623 assert( pRec->op( memory_model::memory_order_relaxed ) == req_Response );
624 }
625 }

656 template <class Container>
657 void combining( Container& owner )
658 {
659 // The thread is a combiner
660 assert( !m_Mutex.try_lock());
661
662 unsigned int const nCurAge = m_nCount.fetch_add( 1, memory_model::memory_order_relaxed ) + 1;
663
664 unsigned int nEmptyPassCount = 0;
665 unsigned int nUsefulPassCount = 0;
666 for ( unsigned int nPass = 0; nPass < m_nCombinePassCount; ++nPass ) {
667 if ( combining_pass( owner, nCurAge ))
668 ++nUsefulPassCount;
669 else if ( ++nEmptyPassCount > nUsefulPassCount )
670 break;
671 }
672
673 m_Stat.onCombining();
674 if ( ( nCurAge & m_nCompactFactor ) == 0 )
675 compact_list( nCurAge );
676 }

678 template <class Container>
679 bool combining_pass( Container& owner, unsigned int nCurAge )
680 {
681 publication_record* p = m_pHead;
682 bool bOpDone = false;
683 while ( p ) {
684 switch ( p->nState.load( memory_model::memory_order_acquire )) {
685 case active:
686 if ( p->op( memory_model::memory_order_acquire ) >= req_Operation ) {
687 p->nAge.store( nCurAge, memory_model::memory_order_relaxed );
688 owner.fc_apply( static_cast<publication_record_type*>( p ));
689 operation_done( *p );
690 bOpDone = true;
691 }
692 break;
693 case inactive:
694 break;
695 case removed:
696 // Such record will be removed on compacting phase
697 break;
698 default:
700 assert( false );
701 }
702 p = p->pNext.load( memory_model::memory_order_acquire );
703 }
704 return bOpDone;
705 }

707 template <class Container>
708 void batch_combining( Container& owner )
709 {
710 // The thread is a combiner
711 assert( !m_Mutex.try_lock());
712
713 unsigned int const nCurAge = m_nCount.fetch_add( 1, memory_model::memory_order_relaxed ) + 1;
714
715 for ( unsigned int nPass = 0; nPass < m_nCombinePassCount; ++nPass )
716 owner.fc_process( begin(), end());
717
718 combining_pass( owner, nCurAge );
719 m_Stat.onCombining();
720 if ( ( nCurAge & m_nCompactFactor ) == 0 )
721 compact_list( nCurAge );
722 }
723
724 bool wait_for_combining( publication_record_type* pRec )
725 {
726 m_waitStrategy.prepare( *pRec );
727 m_Stat.onPassiveWait();
728
729 while ( pRec->op( memory_model::memory_order_acquire ) != req_Response ) {
730 // The record can be excluded from publication list. Reinsert it
731 republish( pRec );
732
733 m_Stat.onPassiveWaitIteration();
734
735 // Wait while operation processing
736 if ( m_waitStrategy.wait( *this, *pRec ))
737 m_Stat.onWakeupByNotifying();
738
739 if ( m_Mutex.try_lock()) {
740 if ( pRec->op( memory_model::memory_order_acquire ) == req_Response ) {
741 // Operation is done
742 m_Mutex.unlock();
743
744 // Wake up a pending threads
745 m_waitStrategy.wakeup( *this );
746 m_Stat.onPassiveWaitWakeup();
747
748 break;
749 }
750 // The thread becomes a combiner
751 m_Stat.onPassiveToCombiner();
752 return false;
753 }
754 }
755 return true;
756 }

758 void compact_list( unsigned int nCurAge )
759 {
760 // Compacts publication list
761 // This function is called only by combiner thread
762
763 try_again:
764 publication_record * pPrev = m_pHead;
765 for ( publication_record * p = pPrev->pNext.load( memory_model::memory_order_acquire ); p; ) {
766 switch ( p->nState.load( memory_model::memory_order_relaxed )) {
767 case active:
768 if ( p->nAge.load( memory_model::memory_order_relaxed ) + m_nCompactFactor < nCurAge )
769 {
770 publication_record * pNext = p->pNext.load( memory_model::memory_order_relaxed );
771 if ( pPrev->pNext.compare_exchange_strong( p, pNext,
772 memory_model::memory_order_acquire, atomics::memory_order_relaxed ))
773 {
774 p->nState.store( inactive, memory_model::memory_order_release );
775 p = pNext;
776 m_Stat.onDeactivatePubRecord();
777 continue;
778 }
779 }
780 break;
781
782 case removed:
783 publication_record * pNext = p->pNext.load( memory_model::memory_order_acquire );
784 if ( cds_likely( pPrev->pNext.compare_exchange_strong( p, pNext, memory_model::memory_order_acquire, atomics::memory_order_relaxed ))) {
785 p = pNext;
786 continue;
787 }
788 else {
789 // CAS can be failed only in beginning of list
790 assert( pPrev == m_pHead );
791 goto try_again;
792 }
793 }
794 pPrev = p;
795 p = p->pNext.load( memory_model::memory_order_acquire );
796 }
797
798 // Iterate over allocated list to find removed records
799 pPrev = m_pAllocatedHead;
800 for ( publication_record * p = pPrev->pNextAllocated.load( memory_model::memory_order_acquire ); p; ) {
801 if ( p->nState.load( memory_model::memory_order_relaxed ) == removed ) {
802 publication_record * pNext = p->pNextAllocated.load( memory_model::memory_order_relaxed );
803 if ( pPrev->pNextAllocated.compare_exchange_strong( p, pNext, memory_model::memory_order_acquire, atomics::memory_order_relaxed )) {
804 free_publication_record( static_cast<publication_record_type *>( p ));
805 p = pNext;
806 continue;
807 }
808 }
809
810 pPrev = p;
811 p = p->pNextAllocated.load( memory_model::memory_order_relaxed );
812 }
813
814 m_Stat.onCompactPublicationList();
815 }
