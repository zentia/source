241 inline bool push(void * const data) { /* modify only pwrite pointer */
242 assert(data != NULL);
243
244 if (available()) {
253 WMB();
254 //std::atomic_thread_fence(std::memory_order_release);
255 buf[pwrite] = data;
256 pwrite = pwrite + ((pwrite+1 >= size) ? (1-size): 1); // circular buffer
257 return true;
258 }
259 return false;
260 }

267 inline bool multipush(void * const data[], int len) {
268 if ((unsigned)len>=size) return false;
269 unsigned long last = pwrite + ((pwrite+ --len >= size) ? (len-size): len);
270 unsigned long r = len-(last+1), l=last;
271 unsigned long i;
272
273 if (buf[last]==NULL) {
274
275 if (last < pwrite) {
276 for(i=len;i>r;--i,--l)
277 buf[l] = data[i];
278 for(i=(size-1);i>=pwrite;--i,--r)
279 buf[i] = data[r];
280
281 } else
282 for(int i=len;i>=0;--i)
283 buf[pwrite+i] = data[i];
284
285 WMB();
286 pwrite = pwrite + ((last+1 >= size) ? 0 : (last+1));
287#if defined(SWSR_MULTIPUSH)
288 mcnt = 0; // reset mpush counter
289#endif
290 return true;
291 }
292 return false;
293 }

486 inline bool push(void * const data) {
487 assert(data);
488
489 const unsigned long next = pwrite + ((pwrite+1>=size)?(1-size):1);
490 if (next != pread) {
491 buf[pwrite] = data;
492 /* We have to ensure that all writes have been committed
493 * in memory before we change the value of the pwrite
494 * reference otherwise the reader can read stale data.
495 */
496 WMB();
497 pwrite =next;
498 return true;
499 }
500 return false;
501 }
502
506 inline bool pop(void ** data) {
507 assert(data);
508
509 if (empty()) return false;
510 *data = buf[pread];
511 pread = pread + ((pread+1 >= size) ? (1-size): 1);
512 return true;
513 }
514
518 inline void reset() {
519 pread=0;
520 pwrite=0;
521 if (size<=512) for(unsigned long i=0;i<size;++i) buf[i]=0;
522 else memset(buf,0,size*sizeof(void*));
523 }

529 // long len = pwrite-pread;
530 // if (len>=0) return len;
531 //return size+len;
532 long tpread=pread, tpwrite=pwrite;
