115[[nodiscard]] constexpr auto max_mip_levels(uint3 size, uint requested_levels) noexcept {
116 if (luisa::any(size == 0u)) { error_image_size_contains_zero(size); }
117 auto max_size = (std::max)({size.x, size.y, size.z});
118 auto max_levels = 0u;
119 while (max_size != 0u) {
120 max_size >>= 1u;
121 max_levels++;
122 }
123 return requested_levels == 0u ? max_levels : (std::min)(requested_levels, max_levels);
