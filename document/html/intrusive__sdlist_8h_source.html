6// intrusive_sdlist is a special kind of intrusive list which we say is
7// "singly-doubly" linked. Instead of having a typical intrusive list node
8// which looks like this:
9//
10// struct intrusive_sdlist_node {
11// intrusive_sdlist_node *mpNext;
12// intrusive_sdlist_node *mpPrev;
13// };
14//
15// We instead have one that looks like this:
16//
17// struct intrusive_sdlist_node {
18// intrusive_sdlist_node* mpNext;
19// intrusive_sdlist_node** mppPrevNext;
20// };
21//
22// This may seem to be suboptimal, but it has one specific advantage: it allows
23// the intrusive_sdlist class to be the size of only one pointer instead of two.
24// This may seem like a minor optimization, but some users have wanted to create
25// thousands of empty instances of these.
26// This is because while an intrusive_list class looks like this:
27//
28// class intrusive_list {
29// intrusive_list_node mBaseNode;
30// };
31//
32// an intrusive_sdlist class looks like this:
33//
34// class intrusive_sdlist {
35// intrusive_sdlist_node* mpNext;
36// };
37//
38// So here we make a list of plusses and minuses of intrusive sdlists
39// compared to intrusive_lists and intrusive_slists:
40//
41// | list | slist | sdlist
42// ---------------------------------------------------------
43// min size | 8 | 4 | 4
44// node size | 8 | 4 | 8
45// anonymous erase | yes | no | yes
46// reverse iteration | yes | no | no
47//

410 template <typename T>
411 inline void intrusive_sdlist<T>::push_back(value_type& value)
412 {
413 intrusive_sdlist_node* pNext = mpNext;
414 intrusive_sdlist_node** ppPrevNext = &mpNext;
415
416 while(pNext)
417 {
418 ppPrevNext = &pNext->mpNext;
419 pNext = pNext->mpNext;
420 }
421
422 *ppPrevNext = &value;
423 value.mppPrevNext = ppPrevNext;
424 value.mpNext = NULL;
425 }

613 template <typename T>
614 bool operator==(const intrusive_sdlist<T>& a, const intrusive_sdlist<T>& b)
615 {
616 // If we store an mSize member for intrusive_sdlist, we want to take advantage of it here.
617 typename intrusive_sdlist<T>::const_iterator ia = a.begin();
618 typename intrusive_sdlist<T>::const_iterator ib = b.begin();
619 typename intrusive_sdlist<T>::const_iterator enda = a.end();
620 typename intrusive_sdlist<T>::const_iterator endb = b.end();
621
622 while((ia != enda) && (ib != endb) && (*ia == *ib))
623 {
624 ++ia;
625 ++ib;
626 }
627 return (ia == enda) && (ib == endb);
628 }
