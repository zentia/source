382[[nodiscard]] constexpr auto is_atomic_operation(CallOp op) noexcept {
383 auto op_value = luisa::to_underlying(op);
384 return op_value >= luisa::to_underlying(CallOp::ATOMIC_EXCHANGE) && op_value <= luisa::to_underlying(CallOp::ATOMIC_FETCH_MAX);
385}
386
387[[nodiscard]] constexpr auto is_autodiff_operation(CallOp op) noexcept {
388 auto op_value = luisa::to_underlying(op);
389 return op_value >= luisa::to_underlying(CallOp::REQUIRES_GRADIENT) && op_value <= luisa::to_underlying(CallOp::DETACH);
390}
391
392[[nodiscard]] constexpr auto is_vector_maker(CallOp op) noexcept {
393 auto op_value = luisa::to_underlying(op);
394 return op_value >= luisa::to_underlying(CallOp::MAKE_BOOL2) && op_value <= luisa::to_underlying(CallOp::MAKE_FLOAT4);
395}

397[[nodiscard]] constexpr auto is_matrix_maker(CallOp op) noexcept {
398 return op == CallOp::MAKE_FLOAT2X2 ||
399 op == CallOp::MAKE_FLOAT3X3 ||
400 op == CallOp::MAKE_FLOAT4X4;
