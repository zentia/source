104 template <class T>
105 rfl::Result<T> to_basic_type(const InputVarType _var) const noexcept {
106 if constexpr (std::is_same<std::remove_cvref_t<T>, std::string>()) {
107 const auto r = yyjson_get_str(_var.val_);
108 if (r == NULL) {
109 return error("Could not cast to string.");
110 }
111 return std::string(r);
112
113 } else if constexpr (std::is_same<std::remove_cvref_t<T>, bool>()) {
114 if (!yyjson_is_bool(_var.val_)) {
115 return error("Could not cast to boolean.");
116 }
117 return yyjson_get_bool(_var.val_);
118
119 } else if constexpr (std::is_floating_point<std::remove_cvref_t<T>>()) {
120 if (!yyjson_is_num(_var.val_)) {
121 return error("Could not cast to double.");
122 }
123 return static_cast<T>(yyjson_get_num(_var.val_));
124
125 } else if constexpr (std::is_unsigned<std::remove_cvref_t<T>>()) {
126 if (!yyjson_is_int(_var.val_)) {
127 return error("Could not cast to int.");
128 }
129 return static_cast<T>(yyjson_get_uint(_var.val_));
130
131 } else if constexpr (std::is_integral<std::remove_cvref_t<T>>()) {
132 if (!yyjson_is_int(_var.val_)) {
133 return error("Could not cast to int.");
134 }
135 return static_cast<T>(yyjson_get_sint(_var.val_));
136
137 } else {
138 static_assert(rfl::always_false_v<T>, "Unsupported type.");
139 }
140 }
