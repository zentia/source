40template<BinaryOp op, typename Lhs, typename Rhs>
41constexpr auto// (ret, lhs_cast, rhs_cast)
42dsl_binary_op_return_type_helper() noexcept {
43
44 static_assert(!any_dsl_v<Lhs, Rhs>);
45
46 constexpr auto lhs_is_scalar = luisa::is_scalar_v<Lhs>;
47 constexpr auto rhs_is_scalar = luisa::is_scalar_v<Rhs>;
48 constexpr auto lhs_is_vector = luisa::is_vector_v<Lhs>;
49 constexpr auto rhs_is_vector = luisa::is_vector_v<Rhs>;
50 constexpr auto lhs_is_matrix = luisa::is_matrix_v<Lhs>;
51 constexpr auto rhs_is_matrix = luisa::is_matrix_v<Rhs>;
52 constexpr auto lhs_is_integral = luisa::is_integral_or_vector_v<Lhs>;
53 constexpr auto rhs_is_integral = luisa::is_integral_or_vector_v<Rhs>;
54 constexpr auto lhs_is_boolean = luisa::is_boolean_or_vector_v<Lhs>;
55 constexpr auto rhs_is_boolean = luisa::is_boolean_or_vector_v<Rhs>;
56 constexpr auto lhs_is_fp = is_floating_point_or_vector_v<Lhs> || is_matrix_v<Lhs>;
57 constexpr auto rhs_is_fp = is_floating_point_or_vector_v<Rhs> || is_matrix_v<Rhs>;
58
59 using lhs_elem = std::conditional_t<lhs_is_matrix, float, vector_element_t<Lhs>>;
60 using rhs_elem = std::conditional_t<rhs_is_matrix, float, vector_element_t<Rhs>>;
61 // we only allow implicit conversion between
62 // - scalars and vectors with the same element type
63 // - integral and floating-point scalars
64 // - equally sized integral scalars
65 // - for SHL/SHR, rhs can be any integral scalar and will be cast to the same element type of lhs
66 // to avoid unexpected (and typically expensive) behaviors
67 static_assert(
68 // no conversion; or scalars and vectors with the same element type
69 std::is_same_v<lhs_elem, rhs_elem> ||
70 // integral and floating-point scalars
71 (lhs_is_scalar && rhs_is_scalar &&
72 ((lhs_is_integral && rhs_is_fp) ||
73 (lhs_is_fp && rhs_is_integral) ||
74 (lhs_is_integral && rhs_is_integral &&
75 sizeof(lhs_elem) == sizeof(rhs_elem))) ||
76 // for SHL/SHR, the rhs might be any unsigned integer
77 ((op == BinaryOp::SHL || op == BinaryOp::SHR) &&
78 lhs_is_integral && rhs_is_scalar && rhs_is_integral)),
79 "Binary operator requires operands "
80 "of the same element type.");
81
82 constexpr auto lhs = expr_value_t<Lhs>{};
83 constexpr auto rhs = expr_value_t<Rhs>{};
84
85 if constexpr (op == BinaryOp::ADD ||
86 op == BinaryOp::SUB ||
87 op == BinaryOp::MUL ||
88 op == BinaryOp::DIV) {
89 static_assert((lhs_is_integral || lhs_is_fp) &&
90 (rhs_is_integral || rhs_is_fp),
91 "Arithmetic operator requires integral "
92 "or floating-point operands.");
93 if constexpr (lhs_is_scalar && rhs_is_scalar) {
94 if constexpr (lhs_is_integral && rhs_is_fp) {
95 dsl_binary_op_fp_integral_implicit_conversion_detected<Lhs, Rhs>();
96 return std::make_tuple(rhs, rhs, rhs);
97 } else if constexpr (lhs_is_fp && rhs_is_integral) {
98 dsl_binary_op_fp_integral_implicit_conversion_detected<Lhs, Rhs>();
99 return std::make_tuple(lhs, lhs, lhs);
100 } else {
101 auto ret = decltype(lhs * rhs){};
102 return std::make_tuple(ret, ret, ret);
103 }
104 } else {
105 auto ret = decltype(lhs * rhs){};
106 return std::make_tuple(ret, lhs, rhs);
107 }
108 } else if constexpr (op == BinaryOp::MOD) {
109 static_assert(lhs_is_integral && rhs_is_integral,
110 "Modulo operator requires integral operands.");
111 return std::make_tuple(decltype(lhs % rhs){}, lhs, rhs);
112 } else if constexpr (op == BinaryOp::EQUAL ||
113 op == BinaryOp::NOT_EQUAL ||
114 op == BinaryOp::LESS ||
115 op == BinaryOp::LESS_EQUAL ||
116 op == BinaryOp::GREATER ||
117 op == BinaryOp::GREATER_EQUAL) {
118 if constexpr (lhs_is_scalar && rhs_is_scalar) {
119 if constexpr (lhs_is_integral && rhs_is_fp) {
120 dsl_binary_op_fp_integral_implicit_conversion_detected<Lhs, Rhs>();
121 return std::make_tuple(bool{}, rhs, rhs);
122 } else if constexpr (lhs_is_fp && rhs_is_integral) {
123 dsl_binary_op_fp_integral_implicit_conversion_detected<Lhs, Rhs>();
124 return std::make_tuple(bool{}, lhs, lhs);
125 } else {
126 return std::make_tuple(bool{}, lhs, rhs);
127 }
128 } else {
129 auto ret = decltype(lhs == rhs){};
130 return std::make_tuple(ret, lhs, rhs);
131 }
132 } else if constexpr (op == BinaryOp::BIT_AND ||
133 op == BinaryOp::BIT_OR ||
134 op == BinaryOp::BIT_XOR) {
135 static_assert((lhs_is_integral && rhs_is_integral) ||
136 (lhs_is_scalar && lhs_is_boolean &&
137 rhs_is_scalar && rhs_is_boolean),
138 "Bitwise operator requires integral or boolean operands.");
139 if constexpr (lhs_is_integral) {
140 return std::make_tuple(decltype(lhs & rhs){}, lhs, rhs);
141 } else {
142 return std::make_tuple(decltype(lhs && rhs){}, lhs, rhs);
143 }
144 } else if constexpr (op == BinaryOp::AND ||
145 op == BinaryOp::OR) {
146 static_assert(lhs_is_vector && rhs_is_vector &&
147 lhs_is_boolean && rhs_is_boolean,
148 "Logical operator requires boolean vector operands. To perform "
149 "logical operations on scalars, please use the bitwise operators "
150 "and note that the short-circuit evaluation is not supported.");
151 return std::make_tuple(decltype(lhs && rhs){}, lhs, rhs);
152 } else if constexpr (op == BinaryOp::SHL ||
153 op == BinaryOp::SHR) {
154 static_assert(lhs_is_integral && rhs_is_integral,
155 "Shift operator requires integral operands.");
156 if constexpr (rhs_is_scalar) {
157 auto rhs_cast = lhs_elem{};
158 auto ret = decltype(lhs << rhs_cast){};
159 return std::make_tuple(ret, lhs, rhs_cast);
160 } else {
161 auto ret = decltype(lhs << rhs){};
162 return std::make_tuple(ret, lhs, rhs);
163 }
164 } else {
165 static_assert(always_false_v<Lhs>);
166 }
167}

172#define LUISA_MAKE_GLOBAL_DSL_BINARY_OP(op, op_tag_name) \
173 template<typename LhsT, typename RhsT> \
174 requires luisa::compute::any_dsl_v<LhsT, RhsT> && \
175 luisa::is_basic_v<luisa::compute::expr_value_t<LhsT>> && \
176 luisa::is_basic_v<luisa::compute::expr_value_t<RhsT>> \
177 [[nodiscard]] inline auto operator op(LhsT &&lhs, RhsT &&rhs) noexcept { \
178 using Lhs = luisa::compute::expr_value_t<LhsT>; \
179 using Rhs = luisa::compute::expr_value_t<RhsT>; \
180 auto [ret, lhs_cast, rhs_cast] = \
181 luisa::compute::detail::dsl_binary_op_return_type_helper< \
182 luisa::compute::BinaryOp::op_tag_name, Lhs, Rhs>(); \
183 using Ret = decltype(ret); \
184 using LhsCast = decltype(lhs_cast); \
185 using RhsCast = decltype(rhs_cast); \
186 auto lhs_expr = luisa::compute::detail::extract_expression(std::forward<LhsT>(lhs)); \
187 auto rhs_expr = luisa::compute::detail::extract_expression(std::forward<RhsT>(rhs)); \
188 auto fb = luisa::compute::detail::FunctionBuilder::current(); \
189 if constexpr (!std::is_same_v<LhsCast, Lhs>) { \
190 lhs_expr = fb->cast(luisa::compute::Type::of<LhsCast>(), \
191 luisa::compute::CastOp::STATIC, lhs_expr); \
192 } \
193 if constexpr (!std::is_same_v<RhsCast, Rhs>) { \
194 rhs_expr = fb->cast(luisa::compute::Type::of<RhsCast>(), \
195 luisa::compute::CastOp::STATIC, rhs_expr); \
196 } \
197 return luisa::compute::dsl::def<Ret>(fb->binary( \
198 luisa::compute::Type::of<Ret>(), \
199 luisa::compute::BinaryOp::op_tag_name, \
200 lhs_expr, rhs_expr)); \
201 }
