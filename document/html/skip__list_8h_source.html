21 template <class GC, typename NodeTraits, typename BackOff, bool IsConst>
22 class iterator {
23 public:
24 typedef GC gc;
25 typedef NodeTraits node_traits;
26 typedef BackOff back_off;
27 typedef typename node_traits::node_type node_type;
28 typedef typename node_traits::value_type value_type;
29 static constexpr bool const c_isConst = IsConst;
30
31 typedef typename std::conditional< c_isConst, value_type const&, value_type&>::type value_ref;
32
33 protected:
34 typedef typename node_type::marked_ptr marked_ptr;
35 typedef typename node_type::atomic_marked_ptr atomic_marked_ptr;
36
37 typename gc::Guard m_guard;
38 node_type * m_pNode;
39
40 protected:
41 static value_type * gc_protect( marked_ptr p )
42 {
43 return node_traits::to_value_ptr( p.ptr());
44 }
45
46 void next()
47 {
48 typename gc::Guard g;
49 g.copy( m_guard );
50 back_off bkoff;
51
52 for (;;) {
53 if ( m_pNode->next( m_pNode->height() - 1 ).load( atomics::memory_order_acquire ).bits()) {
54 // Current node is marked as deleted. So, its next pointer can point to anything
55 // In this case we interrupt our iteration and returns end() iterator.
56 *this = iterator();
57 return;
58 }
59
60 marked_ptr p = m_guard.protect( (*m_pNode)[0], gc_protect );
61 node_type * pp = p.ptr();
62 if ( p.bits()) {
63 // p is marked as deleted. Spin waiting for physical removal
64 bkoff();
65 continue;
66 }
67 else if ( pp && pp->next( pp->height() - 1 ).load( atomics::memory_order_relaxed ).bits()) {
68 // p is marked as deleted. Spin waiting for physical removal
69 bkoff();
70 continue;
71 }
72
73 m_pNode = pp;
74 break;
75 }
76 }
77
78 public: // for internal use only!!!
79 iterator( node_type& refHead )
80 : m_pNode( nullptr )
81 {
82 back_off bkoff;
83
84 for (;;) {
85 marked_ptr p = m_guard.protect( refHead[0], gc_protect );
86 if ( !p.ptr()) {
87 // empty skip-list
88 m_guard.clear();
89 break;
90 }
91
92 node_type * pp = p.ptr();
93 // Logically deleted node is marked from highest level
94 if ( !pp->next( pp->height() - 1 ).load( atomics::memory_order_acquire ).bits()) {
95 m_pNode = pp;
96 break;
97 }
98
99 bkoff();
100 }
101 }
102
103 public:
104 iterator()
105 : m_pNode( nullptr )
106 {}
107
108 iterator( iterator const& s)
109 : m_pNode( s.m_pNode )
110 {
111 m_guard.assign( node_traits::to_value_ptr(m_pNode));
112 }
113
114 value_type * operator ->() const
115 {
116 assert( m_pNode != nullptr );
117 assert( node_traits::to_value_ptr( m_pNode ) != nullptr );
118
119 return node_traits::to_value_ptr( m_pNode );
120 }
121
122 value_ref operator *() const
123 {
124 assert( m_pNode != nullptr );
125 assert( node_traits::to_value_ptr( m_pNode ) != nullptr );
126
127 return *node_traits::to_value_ptr( m_pNode );
128 }
129
131 iterator& operator ++()
132 {
133 next();
134 return *this;
135 }
136
137 iterator& operator =(const iterator& src)
138 {
139 m_pNode = src.m_pNode;
140 m_guard.copy( src.m_guard );
141 return *this;
142 }
143
144 template <typename Bkoff, bool C>
145 bool operator ==(iterator<gc, node_traits, Bkoff, C> const& i ) const
146 {
147 return m_pNode == i.m_pNode;
148 }
149 template <typename Bkoff, bool C>
150 bool operator !=(iterator<gc, node_traits, Bkoff, C> const& i ) const
151 {
152 return !( *this == i );
153 }

539 template <typename Func>
540 bool insert( value_type& val, Func f )
541 {
542 typename gc::Guard gNew;
543 gNew.assign( &val );
544
545 node_type * pNode = node_traits::to_node_ptr( val );
546 scoped_node_ptr scp( pNode );
547 unsigned int nHeight = pNode->height();
548 bool bTowerOk = pNode->has_tower(); // nHeight > 1 && pNode->get_tower() != nullptr;
549 bool bTowerMade = false;
550
551 position pos;
552 while ( true )
553 {
554 if ( find_position( val, pos, key_comparator(), true )) {
555 // scoped_node_ptr deletes the node tower if we create it
556 if ( !bTowerMade )
557 scp.release();
558
559 m_Stat.onInsertFailed();
560 return false;
561 }
562
563 if ( !bTowerOk ) {
564 build_node( pNode );
565 nHeight = pNode->height();
566 bTowerMade = pNode->has_tower();
567 bTowerOk = true;
568 }
569
570 if ( !insert_at_position( val, pNode, pos, f )) {
571 m_Stat.onInsertRetry();
572 continue;
573 }
574
575 increase_height( nHeight );
576 ++m_ItemCounter;
577 m_Stat.onAddNode( nHeight );
578 m_Stat.onInsertSuccess();
579 scp.release();
580 return true;
581 }
582 }

609 template <typename Func>
610 std::pair<bool, bool> update( value_type& val, Func func, bool bInsert = true )
611 {
612 typename gc::Guard gNew;
613 gNew.assign( &val );
614
615 node_type * pNode = node_traits::to_node_ptr( val );
616 scoped_node_ptr scp( pNode );
617 unsigned int nHeight = pNode->height();
618 bool bTowerOk = pNode->has_tower();
619 bool bTowerMade = false;
620
621 position pos;
622 while ( true )
623 {
624 bool bFound = find_position( val, pos, key_comparator(), true );
625 if ( bFound ) {
626 // scoped_node_ptr deletes the node tower if we create it before
627 if ( !bTowerMade )
628 scp.release();
629
630 func( false, *node_traits::to_value_ptr(pos.pCur), val );
631 m_Stat.onUpdateExist();
632 return std::make_pair( true, false );
633 }
634
635 if ( !bInsert ) {
636 scp.release();
637 return std::make_pair( false, false );
638 }
639
640 if ( !bTowerOk ) {
641 build_node( pNode );
642 nHeight = pNode->height();
643 bTowerMade = pNode->has_tower();
644 bTowerOk = true;
645 }
646
647 if ( !insert_at_position( val, pNode, pos, [&func]( value_type& item ) { func( true, item, item ); })) {
648 m_Stat.onInsertRetry();
649 continue;
650 }
651
652 increase_height( nHeight );
653 ++m_ItemCounter;
654 scp.release();
655 m_Stat.onAddNode( nHeight );
656 m_Stat.onUpdateNew();
657 return std::make_pair( true, true );
658 }
659 }

684 bool unlink( value_type& val )
685 {
686 position pos;
687
688 if ( !find_position( val, pos, key_comparator(), false )) {
689 m_Stat.onUnlinkFailed();
690 return false;
691 }
692
693 node_type * pDel = pos.pCur;
694 assert( key_comparator()( *node_traits::to_value_ptr( pDel ), val ) == 0 );
695
696 unsigned int nHeight = pDel->height();
697 typename gc::Guard gDel;
698 gDel.assign( node_traits::to_value_ptr(pDel));
699
700 if ( node_traits::to_value_ptr( pDel ) == &val && try_remove_at( pDel, pos, [](value_type const&) {} )) {
701 --m_ItemCounter;
702 m_Stat.onRemoveNode( nHeight );
703 m_Stat.onUnlinkSuccess();
704 return true;
705 }
706
707 m_Stat.onUnlinkFailed();
708 return false;
709 }

1127 void help_remove( int nLevel, node_type* pPred, marked_node_ptr pCur )
1128 {
1129 if ( pCur->is_upper_level( nLevel )) {
1130 marked_node_ptr p( pCur.ptr());

1146 template <typename Q, typename Compare >
1147 bool find_position( Q const& val, position& pos, Compare cmp, bool bStopIfFound )
1148 {
1149 node_type * pPred;
1150 marked_node_ptr pSucc;
1151 marked_node_ptr pCur;
1152
1153 // Hazard pointer array:
1154 // pPred: [nLevel * 2]
1155 // pSucc: [nLevel * 2 + 1]
1156
1157 retry:
1158 pPred = m_Head.head();
1159 int nCmp = 1;
1160
1161 for ( int nLevel = static_cast<int>( c_nMaxHeight - 1 ); nLevel >= 0; --nLevel ) {
1162 pos.guards.assign( nLevel * 2, node_traits::to_value_ptr( pPred ));
1163 while ( true ) {
1164 pCur = pos.guards.protect( nLevel * 2 + 1, pPred->next( nLevel ), gc_protect );
1165 if ( pCur.bits()) {
1166 // pCur.bits() means that pPred is logically deleted
1167 goto retry;
1168 }
1169
1170 if ( pCur.ptr() == nullptr ) {
1171 // end of list at level nLevel - goto next level
1172 break;
1173 }
1174
1175 // pSucc contains deletion mark for pCur
1176 pSucc = pCur->next( nLevel ).load( memory_model::memory_order_acquire );
1177
1178 if ( pPred->next( nLevel ).load( memory_model::memory_order_acquire ).all() != pCur.ptr())
1179 goto retry;
1180
1181 if ( pSucc.bits()) {
1182 // pCur is marked, i.e. logically deleted
1183 // try to help deleting pCur
1184 help_remove( nLevel, pPred, pCur );
1185 goto retry;
1186 }
1187 else {
1188 nCmp = cmp( *node_traits::to_value_ptr( pCur.ptr()), val );
1189 if ( nCmp < 0 ) {
1190 pPred = pCur.ptr();
1191 pos.guards.copy( nLevel * 2, nLevel * 2 + 1 ); // pPrev guard := cur guard
1192 }
1193 else if ( nCmp == 0 && bStopIfFound )
1194 goto found;
1195 else
1196 break;
1197 }
1198 }
1199
1200 // Next level
1201 pos.pPrev[nLevel] = pPred;
1202 pos.pSucc[nLevel] = pCur.ptr();
1203 }
1204
1205 if ( nCmp != 0 )
1206 return false;
1207
1208 found:
1209 pos.pCur = pCur.ptr();
1210 return pCur.ptr() && nCmp == 0;
1211 }

1213 bool find_min_position( position& pos )
1214 {
1215 node_type * pPred;
1216 marked_node_ptr pSucc;
1217 marked_node_ptr pCur;
1218
1219 // Hazard pointer array:
1220 // pPred: [nLevel * 2]
1221 // pSucc: [nLevel * 2 + 1]
1222
1223 retry:
1224 pPred = m_Head.head();
1225
1226 for ( int nLevel = static_cast<int>( c_nMaxHeight - 1 ); nLevel >= 0; --nLevel ) {
1227 pos.guards.assign( nLevel * 2, node_traits::to_value_ptr( pPred ));
1228 pCur = pos.guards.protect( nLevel * 2 + 1, pPred->next( nLevel ), gc_protect );
1229
1230 // pCur.bits() means that pPred is logically deleted
1231 // head cannot be deleted
1232 assert( pCur.bits() == 0 );
1233
1234 if ( pCur.ptr()) {
1235
1236 // pSucc contains deletion mark for pCur
1237 pSucc = pCur->next( nLevel ).load( memory_model::memory_order_acquire );
1238
1239 if ( pPred->next( nLevel ).load( memory_model::memory_order_acquire ).all() != pCur.ptr())
1240 goto retry;
1241
1242 if ( pSucc.bits()) {
1243 // pCur is marked, i.e. logically deleted.
1244 // try to help deleting pCur
1245 help_remove( nLevel, pPred, pCur );
1246 goto retry;
1247 }
1248 }
1249
1250 // Next level
1251 pos.pPrev[nLevel] = pPred;
1252 pos.pSucc[nLevel] = pCur.ptr();
1253 }
1254
1255 return ( pos.pCur = pCur.ptr()) != nullptr;

1297 else {
1298 if ( !pSucc.ptr())
1299 break;
1300
1301 pPred = pCur.ptr();
1302 pos.guards.copy( nLevel * 2, nLevel * 2 + 1 );
1303 }
1304 }
1305
1306 // Next level
1307 pos.pPrev[nLevel] = pPred;
1308 pos.pSucc[nLevel] = pCur.ptr();
1309 }
1310
1311 return ( pos.pCur = pCur.ptr()) != nullptr;
1312 }

1314 bool renew_insert_position( value_type& val, node_type * pNode, position& pos )
1315 {
1316 node_type * pPred;
1317 marked_node_ptr pSucc;
1318 marked_node_ptr pCur;
1319 key_comparator cmp;
1320
1321 // Hazard pointer array:
1322 // pPred: [nLevel * 2]
1323 // pSucc: [nLevel * 2 + 1]
1324
1325 retry:
1326 pPred = m_Head.head();
1327 int nCmp = 1;
1328
1329 for ( int nLevel = static_cast<int>( c_nMaxHeight - 1 ); nLevel >= 0; --nLevel ) {
1330 pos.guards.assign( nLevel * 2, node_traits::to_value_ptr( pPred ));
1331 while ( true ) {
1332 pCur = pos.guards.protect( nLevel * 2 + 1, pPred->next( nLevel ), gc_protect );
1333 if ( pCur.bits()) {
1334 // pCur.bits() means that pPred is logically deleted
1335 goto retry;
1336 }
1337
1338 if ( pCur.ptr() == nullptr ) {
1339 // end of list at level nLevel - goto next level
1340 break;
1341 }
1342
1343 // pSucc contains deletion mark for pCur
1344 pSucc = pCur->next( nLevel ).load( memory_model::memory_order_acquire );
1345
1346 if ( pPred->next( nLevel ).load( memory_model::memory_order_acquire ).all() != pCur.ptr())
1347 goto retry;
1348
1349 if ( pSucc.bits()) {
1350 // pCur is marked, i.e. logically deleted
1351 if ( pCur.ptr() == pNode ) {
1352 // Node is removing while we are inserting it
1353 return false;
1354 }
1355 // try to help deleting pCur
1356 help_remove( nLevel, pPred, pCur );
1357 goto retry;
1358 }
1359 else {
1360 nCmp = cmp( *node_traits::to_value_ptr( pCur.ptr()), val );
1361 if ( nCmp < 0 ) {
1362 pPred = pCur.ptr();
1363 pos.guards.copy( nLevel * 2, nLevel * 2 + 1 ); // pPrev guard := cur guard
1364 }
1365 else
1366 break;
1367 }
1368 }
1369
1370 // Next level
1371 pos.pPrev[nLevel] = pPred;
1372 pos.pSucc[nLevel] = pCur.ptr();
1373 }
1374
1375 return nCmp == 0;
1376 }

1378 template <typename Func>
1379 bool insert_at_position( value_type& val, node_type * pNode, position& pos, Func f )
1380 {
1381 unsigned int const nHeight = pNode->height();
1382
1383 for ( unsigned int nLevel = 1; nLevel < nHeight; ++nLevel )
1384 pNode->next( nLevel ).store( marked_node_ptr(), memory_model::memory_order_relaxed );
1385
1386 // Insert at level 0
1387 {
1388 marked_node_ptr p( pos.pSucc[0] );
1389 pNode->next( 0 ).store( p, memory_model::memory_order_release );
1390 if ( !pos.pPrev[0]->next( 0 ).compare_exchange_strong( p, marked_node_ptr( pNode ), memory_model::memory_order_release, atomics::memory_order_relaxed ))
1391 return false;
1392
1393 f( val );
1394 }
1395
1396 // Insert at level 1..max
1397 for ( unsigned int nLevel = 1; nLevel < nHeight; ++nLevel ) {
1398 marked_node_ptr p;
1399 while ( true ) {
1400 marked_node_ptr pSucc( pos.pSucc[nLevel] );
1401
1402 // Set pNode->next
1403 // pNode->next can have "logical deleted" flag if another thread is removing pNode right now
1404 if ( !pNode->next( nLevel ).compare_exchange_strong( p, pSucc,
1405 memory_model::memory_order_release, atomics::memory_order_acquire ))
1406 {
1407 // pNode has been marked as removed while we are inserting it
1408 // Stop inserting
1409 assert( p.bits() != 0 );
1410
1411 // Here pNode is linked at least level 0 so level_unlinked() cannot returns true
1412 CDS_VERIFY_FALSE( pNode->level_unlinked( nHeight - nLevel ));
1413
1414 // pNode is linked up to nLevel - 1
1415 // Remove it via find_position()
1416 find_position( val, pos, key_comparator(), false );
1417
1418 m_Stat.onLogicDeleteWhileInsert();
1419 return true;
1420 }
1421 p = pSucc;
1422
1423 // Link pNode into the list at nLevel
1424 if ( pos.pPrev[nLevel]->next( nLevel ).compare_exchange_strong( pSucc, marked_node_ptr( pNode ),
1425 memory_model::memory_order_release, atomics::memory_order_relaxed ))
1426 {
1427 // go to next level
1428 break;
1429 }
1430
1431 // Renew insert position
1432 m_Stat.onRenewInsertPosition();
1433
1434 if ( !renew_insert_position( val, pNode, pos )) {
1435 // The node has been deleted while we are inserting it
1436 // Update current height for concurent removing
1437 CDS_VERIFY_FALSE( pNode->level_unlinked( nHeight - nLevel ));
1438
1439 m_Stat.onRemoveWhileInsert();
1440
1441 // help to removing val
1442 find_position( val, pos, key_comparator(), false );
1443 return true;
1444 }
1445 }
1446 }
1447 return true;
1448 }

1450 template <typename Func>
1451 bool try_remove_at( node_type * pDel, position& pos, Func f )
1452 {
1453 assert( pDel != nullptr );
1454
1455 marked_node_ptr pSucc;
1456 back_off bkoff;
1457
1458 // logical deletion (marking)
1459 for ( unsigned int nLevel = pDel->height() - 1; nLevel > 0; --nLevel ) {
1460 pSucc = pDel->next( nLevel ).load( memory_model::memory_order_relaxed );
1461 if ( pSucc.bits() == 0 ) {
1462 bkoff.reset();
1463 while ( !( pDel->next( nLevel ).compare_exchange_weak( pSucc, pSucc | 1,
1464 memory_model::memory_order_release, atomics::memory_order_acquire )
1465 || pSucc.bits() != 0 ))
1466 {
1467 bkoff();
1468 m_Stat.onMarkFailed();
1469 }
1470 }
1471 }
1472
1473 marked_node_ptr p( pDel->next( 0 ).load( memory_model::memory_order_relaxed ).ptr());
1474 while ( true ) {
1475 if ( pDel->next( 0 ).compare_exchange_strong( p, p | 1, memory_model::memory_order_release, atomics::memory_order_acquire ))
1476 {
1477 f( *node_traits::to_value_ptr( pDel ));
1478
1479 // Physical deletion
1480 // try fast erase
1481 p = pDel;
1482
1483 for ( int nLevel = static_cast<int>( pDel->height() - 1 ); nLevel >= 0; --nLevel ) {
1484
1485 pSucc = pDel->next( nLevel ).load( memory_model::memory_order_acquire );
1486 if ( pos.pPrev[nLevel]->next( nLevel ).compare_exchange_strong( p, marked_node_ptr( pSucc.ptr()),
1487 memory_model::memory_order_acq_rel, atomics::memory_order_relaxed ))
1488 {
1489 pDel->level_unlinked();
1490 }
1491 else {
1492 // Make slow erase
1493# ifdef CDS_DEBUG
1494 if ( find_position( *node_traits::to_value_ptr( pDel ), pos, key_comparator(), false ))
1495 assert( pDel != pos.pCur );
1496# else
1497 find_position( *node_traits::to_value_ptr( pDel ), pos, key_comparator(), false );
1498# endif
1499 m_Stat.onSlowErase();
1500 return true;
1501 }
1502 }
1503
1504 // Fast erasing success
1505 gc::retire( node_traits::to_value_ptr( pDel ), dispose_node );
1506 m_Stat.onFastErase();
1507 return true;
1508 }
1509 else if ( p.bits()) {
1510 // Another thread is deleting pDel right now
1511 m_Stat.onEraseContention();
1512 return false;
1513 }
1514 m_Stat.onEraseRetry();
1515 bkoff();
1516 }
1517 }

1524 template <typename Q, typename Compare, typename Func>
1525 finsd_fastpath_result find_fastpath( Q& val, Compare cmp, Func f )
1526 {
1527 node_type * pPred;
1528 marked_node_ptr pCur;
1529 marked_node_ptr pNull;
1530
1531 // guard array:
1532 // 0 - pPred on level N
1533 // 1 - pCur on level N
1534 typename gc::template GuardArray<2> guards;
1535 back_off bkoff;
1536 unsigned attempt = 0;
1537
1538 try_again:
1539 pPred = m_Head.head();
1540 for ( int nLevel = static_cast<int>( m_nHeight.load( memory_model::memory_order_relaxed ) - 1 ); nLevel >= 0; --nLevel ) {
1541 pCur = guards.protect( 1, pPred->next( nLevel ), gc_protect );
1542
1543 while ( pCur != pNull ) {
1544 if ( pCur.bits()) {
1545 // pPred is being removed
1546 if ( ++attempt < 4 ) {
1547 bkoff();
1548 goto try_again;
1549 }
1550
1551 return find_fastpath_abort;
1552 }
1553
1554 if ( pCur.ptr()) {
1555 int nCmp = cmp( *node_traits::to_value_ptr( pCur.ptr()), val );
1556 if ( nCmp < 0 ) {
1557 guards.copy( 0, 1 );
1558 pPred = pCur.ptr();
1559 pCur = guards.protect( 1, pCur->next( nLevel ), gc_protect );
1560 }
1561 else if ( nCmp == 0 ) {
1562 // found
1563 f( *node_traits::to_value_ptr( pCur.ptr()), val );
1564 return find_fastpath_found;
1565 }
1566 else {
1567 // pCur > val - go down
1568 break;
1569 }
1570 }
1571 }
1572 }
1573
1574 return find_fastpath_not_found;
1575 }
