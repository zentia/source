178inline void syclGraph::dump(
179 std::ostream& os, const void* root, const std::string& root_name
180) const {
181
182 // recursive dump with stack
183 std::stack<std::tuple<const syclGraph*, const syclNode*, int>> stack;
184 stack.push(std::make_tuple(this, nullptr, 1));
185
186 int pl = 0;
187
188 while(!stack.empty()) {
189
190 auto [graph, parent, l] = stack.top();
191 stack.pop();
192
193 for(int i=0; i<pl-l+1; i++) {
194 os << "}\n";
195 }
196
197 if(parent == nullptr) {
198 if(root) {
199 os << "subgraph cluster_p" << root << " {\nlabel=\"syclFlow: ";
200 if(root_name.empty()) os << 'p' << root;
201 else os << root_name;
202 os << "\";\n" << "color=\"red\"\n";
203 }
204 else {
205 os << "digraph syclFlow {\n";
206 }
207 }
208 else {
209 os << "subgraph cluster_p" << parent << " {\nlabel=\"syclSubflow: ";
210 if(parent->_name.empty()) os << 'p' << parent;
211 else os << parent->_name;
212 os << "\";\n" << "color=\"purple\"\n";
213 }
214
215 for(auto& v : graph->_nodes) {
216
217 os << 'p' << v.get() << "[label=\"";
218 if(v->_name.empty()) {
219 os << 'p' << v.get() << "\"";
220 }
221 else {
222 os << v->_name << "\"";
223 }
224 os << "];\n";
225
226 for(const auto s : v->_successors) {
227 os << 'p' << v.get() << " -> " << 'p' << s << ";\n";
228 }
