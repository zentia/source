237 void thread_routine() {
238 threadid = ff_getThreadID();
239#if defined(FF_INITIAL_BARRIER)
240 if (barrier) {
241 barrier->doBarrier(tid);
242 }
243 /* else {
244 * printf("THREAD %ld skip barrier\n", threadid);
245 * }
246 */
247#endif
248 void * ret;
249 do {
250 init_error=false;
251 if (svc_init()<0) {
252 error("ff_thread, svc_init failed, thread exit!!!\n");
253 init_error=true;
254 break;
255 } else {
256 ret = svc(NULL);
257 }
258 svc_end();
259
260 if (disable_cancelability()) {
261 error("ff_thread, thread_routine, could not change thread cancelability");
262 return;
263 }
264
265 // acquire lock. While freezing is true,
266 // freeze and wait.
267 pthread_mutex_lock(&mutex);
268 if (ret != FF_EOS_NOFREEZE && !stp) {
269 if ((freezing == 0) && (ret == FF_EOS)) stp = true;
270 while(freezing==1) { // NOTE: freezing can change to 2
271 frozen=true;
272 pthread_cond_signal(&cond_frozen);
273 pthread_cond_wait(&cond,&mutex);
274 }
275 }
276
277 //thawed=true;
278 //pthread_cond_signal(&cond);
279 //frozen=false;
280 if (freezing != 0) freezing = 1; // freeze again next time
281 pthread_mutex_unlock(&mutex);
282
283 if (enable_cancelability()) {
284 error("ff_thread, thread_routine, could not change thread cancelability");
285 return;
286 }
287 } while(!stp);
288
289 if (freezing) {
290 pthread_mutex_lock(&mutex);
291 frozen=true;
292 pthread_cond_signal(&cond_frozen);
293 pthread_mutex_unlock(&mutex);
294 }
295 isdone = true;
296 }

334 virtual int spawn(int cpuId=-1) {
335 if (spawned) return -1;
336
337 if ((attr = (pthread_attr_t*)malloc(sizeof(pthread_attr_t))) == NULL) {
338 error("spawn: pthread can not be created, malloc failed\n");
339 return -1;
340 }
341 if (pthread_attr_init(attr)) {
342 perror("pthread_attr_init: pthread can not be created.");
343 return -1;
344 }
345
346 int CPUId = -1;
347 if (default_mapping)
348 init_thread_affinity(attr, cpuId);
349 if (CPUId==-2) return -2;
350
351 if (barrier)
352 tid= internal_threadCounter.fetch_add(1);
353 else
354 tid= internal_threadCounter_noBarrier.fetch_add(1);
355 int r=0;
356 if ((r=pthread_create(&th_handle, attr,
357 proxy_thread_routine, this)) != 0) {
358 errno=r;
359 perror("pthread_create: pthread creation failed.");
360 barrier?--internal_threadCounter:--internal_threadCounter_noBarrier;
361 return -2;
362 }
363 spawned = true;
364 return CPUId;
365 }

367 virtual int wait() {
368 int r=0;
369 stp=true;
370 if (isfrozen()) {
371 wait_freezing();
372 thaw();
373 }
374 if (spawned) {
375 pthread_join(th_handle, NULL);
376 barrier ? --internal_threadCounter: --internal_threadCounter_noBarrier;
377 }
378 if (attr) {
379 if (pthread_attr_destroy(attr)) {
380 error("ERROR: ff_thread.wait: pthread_attr_destroy fails!");
381 r=-1;
382 }
383 free(attr);
384 attr = NULL;
385 }
386 spawned=false;
387 return r;
388 }

592 struct timespec tv;
593 timedwait_timeout(tv);

595 pthread_cond_timedwait(cons_c, cons_m,&tv);

1422 void * ret = FF_EOS;
1423 bool inpresent = (filter->get_in_buffer() != NULL);
1424 bool outpresent = (filter->get_out_buffer() != NULL);
1425 bool skipfirstpop = filter->skipfirstpop();
1426 bool exit=false;
1427 bool filter_outpresent = false;
1428 size_t neos=input_neos;
1429
1430
1431 // if the node is a combine where the last stage is a multi-output
1432 if ( filter && ( !outpresent && filter->isMultiOutput() ) ) {
1433 filter_outpresent=true;
1434 }
1435 gettimeofday(&filter->wtstart,NULL);
1436 do {
1437#ifdef DFF_ENABLED
1438 if (!filter->skipallpop() && inpresent){
1439#else
1440 if (inpresent) {
1441#endif
1442 if (!skipfirstpop) pop(&task);
1443 else skipfirstpop=false;
1444 if ((task == FF_EOS) || (task == FF_EOSW) ||
1445 (task == FF_EOS_NOFREEZE)) {
1446 ret = task;
1447
1448 if (--neos > 0) continue;
1449 filter->eosnotify();
1450
1451 // only EOS and EOSW are propagated
1452 if ( (task == FF_EOS) || (task == FF_EOSW) ) {
1453 if (outpresent) push(task);
1454 if (filter_outpresent) filter->propagateEOS();
1455 }
1456 break;
1457 }
1458 if (task == FF_GO_OUT) break;
1459 }
1460 FFTRACE(++filter->taskcnt);
1461 FFTRACE(ticks t0 = getticks());
1462
1463#if defined(FF_TASK_CALLBACK)
1464 if (filter) callbackIn();
1465#endif
1466
1467 ret = filter->svc(task);
1468
1469#if defined(TRACE_FASTFLOW)
1470 ticks diff=(getticks()-t0);
1471 filter->tickstot +=diff;
1472 filter->ticksmin=(std::min)(filter->ticksmin,diff); // (std::min) for win portability)
1473 filter->ticksmax=(std::max)(filter->ticksmax,diff);
1474#endif
1475
1476 if (ret == FF_GO_OUT) break;
1477 if (!ret || (ret >= FF_EOSW)) { // EOS or EOS_NOFREEZE or EOSW
1478 // NOTE: The EOS is gonna be produced in the output queue
1479 // and the thread exits even if there might be some tasks
1480 // in the input queue !!!
1481 if (!ret) ret = FF_EOS;
1482 exit=true;
1483 }
1484 if ( outpresent && ((ret != FF_GO_ON) && (ret != FF_EOS_NOFREEZE)) ) {
1485 push(ret);
1486#if defined(FF_TASK_CALLBACK)
1487 if (filter) callbackOut();
1488#endif
1489 }
1490 } while(!exit);
1491
1492 gettimeofday(&filter->wtstop,NULL);
1493 filter->wttime+=diffmsec(filter->wtstop,filter->wtstart);
1494
1495 return ret;
1496 }
