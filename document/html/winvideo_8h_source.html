64bool WinInit(HINSTANCE hInstance, int nCmdShow, WNDCLASSEX *uwc, const char *title, bool fixedsize)
65{
66 WNDCLASSEX wndclass; // Our app's windows class
67 if(uwc) {
68 memcpy(&wndclass, uwc, sizeof(wndclass));
69 g_pUserProc = uwc->lpfnWndProc;
70 } else {
71 memset(&wndclass, 0, sizeof(wndclass));
72 wndclass.hCursor = LoadCursor(NULL, IDC_ARROW);
73 wndclass.lpszClassName = title;
74 }
75 wndclass.cbSize = sizeof(wndclass);
76 wndclass.hInstance = hInstance;
77 wndclass.lpfnWndProc = InternalWndProc;
78 wndclass.style |= CS_HREDRAW | CS_VREDRAW;
79 wndclass.hbrBackground = CreateSolidBrush(RGBKEY);
80
81 if( !RegisterClassExA(&wndclass) ) return false;
82 int xaddend = GetSystemMetrics(fixedsize?SM_CXFIXEDFRAME:SM_CXFRAME)*2;
83 int yaddend = GetSystemMetrics(fixedsize?SM_CYFIXEDFRAME:SM_CYFRAME)*2 + GetSystemMetrics(SM_CYCAPTION);
84 if(wndclass.lpszMenuName) yaddend += GetSystemMetrics(SM_CYMENU);
85
86 // Setup the new window's physical parameters - and tell Windows to create it
87 g_hAppWnd = CreateWindowA(wndclass.lpszClassName, // Window class name
88 title, // Window caption
89 !fixedsize ? WS_OVERLAPPEDWINDOW : // Window style
90 WS_OVERLAPPED|WS_CAPTION|WS_SYSMENU|WS_MINIMIZEBOX,
91 CW_USEDEFAULT, // Initial x pos: use default placement
92 0, // Initial y pos: not used here
93 g_sizex+xaddend,// Initial x size
94 g_sizey+yaddend,// Initial y size
95 NULL, // parent window handle
96 NULL, // window menu handle
97 hInstance, // program instance handle
98 NULL); // Creation parameters
99 return g_hAppWnd != NULL;
100}

103static bool RedirectIOToConsole(void)
104{
105 int hConHandle; size_t lStdHandle;
106 CONSOLE_SCREEN_BUFFER_INFO coninfo;
107 FILE *fp;
108 // allocate a console for this app
109 AllocConsole();
110
111 // set the screen buffer to be big enough to let us scroll text
112 GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &coninfo);
113 coninfo.dwSize.Y = MAX_CONSOLE_LINES;
114 SetConsoleScreenBufferSize(GetStdHandle(STD_OUTPUT_HANDLE), coninfo.dwSize);
115
116 // redirect unbuffered STDOUT to the console
117 lStdHandle = (size_t)GetStdHandle(STD_OUTPUT_HANDLE);
118 hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
119 if(hConHandle <= 0) return false;
120 fp = _fdopen( hConHandle, "w" );
121 *stdout = *fp;
122 setvbuf( stdout, NULL, _IONBF, 0 );
123
124 // redirect unbuffered STDERR to the console
125 lStdHandle = (size_t)GetStdHandle(STD_ERROR_HANDLE);
126 hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
127 if(hConHandle > 0) {
128 fp = _fdopen( hConHandle, "w" );
129 *stderr = *fp;
130 setvbuf( stderr, NULL, _IONBF, 0 );
131 }
132
133 // redirect unbuffered STDIN to the console
134 lStdHandle = (size_t)GetStdHandle(STD_INPUT_HANDLE);
135 hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
136 if(hConHandle > 0) {
137 fp = _fdopen( hConHandle, "r" );
138 *stdin = *fp;
139 setvbuf( stdin, NULL, _IONBF, 0 );
140 }
141
142 // make cout, wcout, cin, wcin, wcerr, cerr, wclog and clog
143 // point to console as well
144 std::ios::sync_with_stdio();
145 return true;
146}

191static bool loop_once(video *v)
192{
193 // screen update notify
194 if(int updates = g_updates) {
195 g_updates = 0;
196 if(g_video->updating) { g_skips += updates-1; g_fps++; }
197 else g_skips += updates;
198 UpdateWindow(g_hAppWnd);
199 }
200 // update fps
201 DWORD msec = GetTickCount();
202 if(v->calc_fps && msec >= g_msec+1000) {
203 double sec = (msec - g_msec)/1000.0;
204 char buffer[256], n = _snprintf(buffer, 128, "%s: %d fps", v->title, int(double(g_fps + g_skips)/sec));
205 if(g_skips) _snprintf(buffer+n, 128, " - %d skipped = %d updates", int(g_skips/sec), int(g_fps/sec));
206 SetWindowTextA(g_hAppWnd, buffer);
207 g_msec = msec; g_skips = g_fps = 0;
208 }
209 // event processing, including painting
210 MSG msg;
211 if(PeekMessage(&msg, NULL, 0, 0, PM_REMOVE)){
212 if( msg.message == WM_QUIT ) { v->running = false; return false; }
213 if( !hAccelTable || !TranslateAccelerator(msg.hwnd, hAccelTable, &msg) ){
214 TranslateMessage(&msg);
215 DispatchMessage(&msg);
216 }
217 return true; // try again
218 }
219 return false;
220}

223void video::main_loop()
224{
225 // let Windows draw and unroll the window
226 InvalidateRect(g_hAppWnd, 0, false);
227 g_msec = GetTickCount(); // let's stay for 0,5 sec
228 while(g_msec + 500 > GetTickCount()) { loop_once(this); Sleep(1); }
229 g_msec = GetTickCount();
230 // now, start main process
231 if(threaded) {
232 g_handles[0] = CreateThread (
233 NULL, // LPSECURITY_ATTRIBUTES security_attrs
234 0, // SIZE_T stacksize
235 (LPTHREAD_START_ROUTINE) thread_video,
236 this, // argument
237 0, 0);
238 if(!g_handles[0]) { DisplayError("Can't create thread"); return; }
239 else // harmless race is possible here
240 g_handles[1] = CreateEvent(NULL, false, false, NULL);
241 while(running) {
242 while(loop_once(this));
243 YIELD_TO_THREAD(); // give time for processing when running on single CPU
244 DWORD r = MsgWaitForMultipleObjects(2, g_handles, false, INFINITE, QS_ALLINPUT^QS_MOUSEMOVE);
245 if(r == WAIT_OBJECT_0) break; // thread terminated
246 }
247 running = false;
248 if(WaitForSingleObject(g_handles[0], 3000) == WAIT_TIMEOUT){
249 // there was not enough time for graceful shutdown, killing the example with code 1.
250 exit(1);
251 }
252 if(g_handles[0]) CloseHandle(g_handles[0]);
253 if(g_handles[1]) CloseHandle(g_handles[1]);
254 g_handles[0] = g_handles[1] = 0;
255 }
256 else on_process();
257}

260bool video::next_frame()
261{
262 if(!running) return false;
263 g_updates++; // Fast but inaccurate counter. The data race here is benign.
264 if(!threaded) while(loop_once(this));
265 else if(g_handles[1]) {
266 SetEvent(g_handles[1]);
267 YIELD_TO_THREAD();
268 }
269 return true;
270}
