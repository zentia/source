623// Function: reset
624template <typename... Ps>
625void Pipeline<Ps...>::reset() {
626
627 _num_tokens = 0;
628
629 for(size_t l = 0; l<num_lines(); l++) {
630 _pipeflows[l]._pipe = 0;
631 _pipeflows[l]._line = l;
632
633 _pipeflows[l]._num_deferrals = 0;
634 _pipeflows[l]._dependents.clear();
635 }
636
637 assert(_ready_tokens.empty() == true);
638 _token_dependencies.clear();
639 _deferred_tokens.clear();
640
641 _lines[0][0].join_counter.store(0, std::memory_order_relaxed);
642
643 for(size_t l=1; l<num_lines(); l++) {
644 for(size_t f=1; f<num_pipes(); f++) {
645 _lines[l][f].join_counter.store(
646 static_cast<size_t>(_meta[f].type), std::memory_order_relaxed
647 );
648 }
649 }
650
651 for(size_t f=1; f<num_pipes(); f++) {
652 _lines[0][f].join_counter.store(1, std::memory_order_relaxed);
653 }
654
655 for(size_t l=1; l<num_lines(); l++) {
656 _lines[l][0].join_counter.store(
657 static_cast<size_t>(_meta[0].type) - 1, std::memory_order_relaxed
658 );
659 }

692 // valid (e.g., 12.defer(16))
693 if (*it >= _num_tokens) {
694 _token_dependencies[*it].push_back(pf._token);

744// Procedure: _resolve_token_dependencies
745// Resolve dependencies for tokens that defer to current token
746//
747// For example,
748// 12.defer(16);
749// 13.defer(16);
750// _token_dependencies will have the entry
751// {key: 16, value: std::vector{12, 13}}
752//
753// When 16 finishes, we need to remove 16 from 12's and 13's
754// individual_dependents
755template <typename... Ps>
756void Pipeline<Ps...>::_resolve_token_dependencies(Pipeflow& pf) {
757
758 if (auto it = _token_dependencies.find(pf._token);
759 it != _token_dependencies.end()) {
760
761 // iterate tokens that defer to pf._token
762 // (e.g., 12 and 13)
763 for(size_t target : it->second) {
764
765 auto dpf = _deferred_tokens.find(target);
766
767 assert(dpf != _deferred_tokens.end());
768
769 // erase pf._token from target's _dependents
770 // (e.g., remove 16 from 12's dependents)
771 dpf->second._dependents.erase(pf._token);
772 // dpf->second._dependent_satellites[pf._token]
773 //);
774
775 // target has no dependents
776 if (dpf->second._dependents.empty()) {
777
778 // push target into _ready_tokens queue
779 _ready_tokens.emplace(dpf->second._token, dpf->second._num_deferrals);
780 //_ready_tokens.push(
781 // std::make_pair(dpf->second._token, dpf->second._num_deferrals)
782 //);
783
784 // erase target from _deferred_tokens
785 _deferred_tokens.erase(dpf);
786 }
787 }
788
789 // remove pf._token from _token_dependencies
790 // (e.g., remove the entry
791 // {key: 16, value: std::vector{12, 13}} from _token_dependencies)
792 _token_dependencies.erase(it);
793 }

796// Procedure: _build
797template <typename... Ps>
798void Pipeline<Ps...>::_build() {
799
800 using namespace std::literals::string_literals;
801
802 FlowBuilder fb(_graph);
803
804 // init task
805 _tasks[0] = fb.emplace([this]() {
806 return static_cast<int>(_num_tokens % num_lines());
807 }).name("cond");
808
809 // line task
810 for(size_t l = 0; l < num_lines(); l++) {
811
812 _tasks[l + 1] = fb.emplace([this, l] (tf::Runtime& rt) mutable {
813
814 auto pf = &_pipeflows[l];
815
816 pipeline:
817
818 _lines[pf->_line][pf->_pipe].join_counter.store(
819 static_cast<size_t>(_meta[pf->_pipe].type), std::memory_order_relaxed
820 );
821
822 // First pipe does all jobs of initialization and token dependencies
823 if (pf->_pipe == 0) {
824 // _ready_tokens queue is not empty
825 // substitute pf with the token at the front of the queue
826 if (!_ready_tokens.empty()) {
827 pf->_token = _ready_tokens.front().first;
828 pf->_num_deferrals = _ready_tokens.front().second;
829 _ready_tokens.pop();
830 }
831 else {
832 pf->_token = _num_tokens;
833 pf->_num_deferrals = 0;
834 }
835
836 handle_token_dependency:
837
838 if (pf->_stop = false, _on_pipe(*pf, rt); pf->_stop == true) {
839 // here, the pipeline is not stopped yet because other
840 // lines of tasks may still be running their last stages
841 return;
842 }
843
844 if (_num_tokens == pf->_token) {
845 ++_num_tokens;
846 }
847
848 if (pf->_dependents.empty() == false){
849 // check if the pf->_dependents have valid dependents
850 _check_dependents(*pf);
851
852 // tokens in pf->_dependents are all valid dependents
853 if (pf->_dependents.size()) {
854
855 // construct a data structure for pf in _deferred_tokens
856 _construct_deferred_tokens(*pf);
857 goto pipeline;
858 }
859
860 // tokens in pf->_dependents are invalid dependents
861 // directly goto on_pipe on the same line
862 else {
863 goto handle_token_dependency;
864 }
865 }
866
867 // Every token within the deferral range needs to check
868 // if it can resolve dependencies on other tokens.
869 if (pf->_token <= _longest_deferral) {
870 _resolve_token_dependencies(*pf);
871 }
872 }
873 else {
874 _on_pipe(*pf, rt);
875 }
876
877 size_t c_f = pf->_pipe;
878 size_t n_f = (pf->_pipe + 1) % num_pipes();
879 size_t n_l = (pf->_line + 1) % num_lines();
880
881 pf->_pipe = n_f;
882
883 // ---- scheduling starts here ----
884 // Notice that the shared variable f must not be changed after this
885 // point because it can result in data race due to the following
886 // condition:
887 //
888 // a -> b
889 // | |
890 // v v
891 // c -> d
892 //
893 // d will be spawned by either c or b, so if c changes f but b spawns d
894 // then data race on f will happen
895
896 std::array<int, 2> retval;
897 size_t n = 0;
898
899 // downward dependency
900 if(_meta[c_f].type == PipeType::SERIAL &&
901 _lines[n_l][c_f].join_counter.fetch_sub(
902 1, std::memory_order_acq_rel) == 1
903 ) {
904 retval[n++] = 1;
905 }
906
907 // forward dependency
908 if(_lines[pf->_line][n_f].join_counter.fetch_sub(
909 1, std::memory_order_acq_rel) == 1
910 ) {
911 retval[n++] = 0;
912 }
913
914 // notice that the task index starts from 1
915 switch(n) {
916 case 2: {
917 rt.schedule(_tasks[n_l+1]);
918 goto pipeline;
919 }
920 case 1: {
921 // downward dependency
922 if (retval[0] == 1) {
923 pf = &_pipeflows[n_l];
924 }
925 // forward dependency
926 goto pipeline;
927 }
928 }
929 }).name("rt-"s + std::to_string(l));
930
931 _tasks[0].precede(_tasks[l+1]);
932 }
933}

1527void ScalablePipeline<P>::_check_dependents(Pipeflow& pf) {
1528 ++pf._num_deferrals;
1529
1530 for (auto it = pf._dependents.begin(); it != pf._dependents.end();) {
1531
1532 // valid (e.g., 12.defer(16))
1533 if (*it >= _num_tokens) {
1534 _token_dependencies[*it].push_back(pf._token);
1535 _longest_deferral = std::max(_longest_deferral, *it);
1536 ++it;
1537 }
1538 // valid or invalid (e.g., 12.defer(7))
1539 else {

1541
1542 // valid (e.g., 7 is deferred)

1544 _token_dependencies[*it].push_back(pf._token);
1545 ++it;

1551 }

1570// Procedure: _resolve_token_dependencies
1571// Resolve dependencies for tokens that defer to current token
1572template <typename P>
1573void ScalablePipeline<P>::_resolve_token_dependencies(Pipeflow& pf) {
1574
1575 if (auto it = _token_dependencies.find(pf._token);
1576 it != _token_dependencies.end()) {
1577
1578 // iterate tokens that defer to pf._token
1579 for(size_t target : it->second) {
1580
1581 auto dpf = _deferred_tokens.find(target);
1582
1583 assert(dpf != _deferred_tokens.end());
1584
1585 // erase pf._token from target's _dependents
1586 dpf->second._dependents.erase(pf._token);
1587
1588 // target has no dependents
1589 if (dpf->second._dependents.empty()) {
1590 _ready_tokens.emplace(dpf->second._token, dpf->second._num_deferrals);
1591 _deferred_tokens.erase(dpf);
1592 }
1593 }
1594
1595 _token_dependencies.erase(it);
1596 }

1599// Procedure: _build
1600template <typename P>
1601void ScalablePipeline<P>::_build() {
1602
1603 using namespace std::literals::string_literals;
1604
1605 FlowBuilder fb(_graph);
1606
1607 // init task
1608 _tasks[0] = fb.emplace([this]() {
1609 return static_cast<int>(_num_tokens % num_lines());
1610 }).name("cond");
1611
1612 // line task
1613 for(size_t l = 0; l < num_lines(); l++) {
1614
1615 _tasks[l + 1] = fb.emplace([this, l] (tf::Runtime& rt) mutable {
1616
1617 auto pf = &_pipeflows[l];
1618
1619 pipeline:
1620
1621 _line(pf->_line, pf->_pipe).join_counter.store(
1622 static_cast<size_t>(_pipes[pf->_pipe]->type()), std::memory_order_relaxed
1623 );
1624
1625 // First pipe does all jobs of initialization and token dependencies
1626 if (pf->_pipe == 0) {
1627 // _ready_tokens queue is not empty
1628 // substitute pf with the token at the front of the queue
1629 if (!_ready_tokens.empty()) {
1630 pf->_token = _ready_tokens.front().first;
1631 pf->_num_deferrals = _ready_tokens.front().second;
1632 _ready_tokens.pop();
1633 }
1634 else {
1635 pf->_token = _num_tokens;
1636 pf->_num_deferrals = 0;
1637 }
1638
1639 handle_token_dependency:
1640
1641 if (pf->_stop = false, _on_pipe(*pf, rt); pf->_stop == true) {
1642 // here, the pipeline is not stopped yet because other
1643 // lines of tasks may still be running their last stages
1644 return;
1645 }
1646
1647 if (_num_tokens == pf->_token) {
1648 ++_num_tokens;
1649 }
1650
1651 if (pf->_dependents.empty() == false){
1652 // check if the pf->_dependents have valid dependents
1653 _check_dependents(*pf);
1654
1655 // tokens in pf->_dependents are all valid dependents
1656 if (pf->_dependents.size()) {
1657
1658 // construct a data structure for pf in _deferred_tokens
1659 _construct_deferred_tokens(*pf);
1660 goto pipeline;
1661 }
1662
1663 // tokens in pf->_dependents are invalid dependents
1664 // directly goto on_pipe on the same line
1665 else {
1666 goto handle_token_dependency;
1667 }
1668 }
1669
1670 // Every token within the deferral range needs to check
1671 // if it can resolve dependencies on other tokens.
1672 if (pf->_token <= _longest_deferral) {
1673 _resolve_token_dependencies(*pf);
1674 }
1675 }
1676 else {
1677 _on_pipe(*pf, rt);
1678 }
1679
1680 size_t c_f = pf->_pipe;
1681 size_t n_f = (pf->_pipe + 1) % num_pipes();
1682 size_t n_l = (pf->_line + 1) % num_lines();
1683
1684 pf->_pipe = n_f;
1685
1686 // ---- scheduling starts here ----
1687 // Notice that the shared variable f must not be changed after this
1688 // point because it can result in data race due to the following
1689 // condition:
1690 //
1691 // a -> b
1692 // | |
1693 // v v
1694 // c -> d
1695 //
1696 // d will be spawned by either c or b, so if c changes f but b spawns d
1697 // then data race on f will happen
1698
1699 std::array<int, 2> retval;
1700 size_t n = 0;
1701
1702 // downward dependency
1703 if(_pipes[c_f]->type() == PipeType::SERIAL &&
1704 _line(n_l, c_f).join_counter.fetch_sub(
1705 1, std::memory_order_acq_rel) == 1
1706 ) {
1707 retval[n++] = 1;
1708 }
1709
1710 // forward dependency
1711 if(_line(pf->_line, n_f).join_counter.fetch_sub(
1712 1, std::memory_order_acq_rel) == 1
1713 ) {
1714 retval[n++] = 0;
1715 }
1716
1717 // notice that the task index starts from 1
1718 switch(n) {
1719 case 2: {
1720 rt.schedule(_tasks[n_l+1]);
1721 goto pipeline;
1722 }
1723 case 1: {
1724 if (retval[0] == 1) {
1725 pf = &_pipeflows[n_l];
1726 }
1727 goto pipeline;
1728 }
1729 }
1730 }).name("rt-"s + std::to_string(l));
1731
1732 _tasks[0].precede(_tasks[l+1]);
1733 }
1734}
