258 void link_node( node_type * pNode, node_type * pPred, node_type * pCur )
259 {
260 assert( pPred->m_pNext.load(memory_model::memory_order_relaxed).ptr() == pCur );
261 link_checker::is_empty( pNode );
262
263 pNode->m_pNext.store( marked_node_ptr(pCur), memory_model::memory_order_release );
264 pPred->m_pNext.store( marked_node_ptr(pNode), memory_model::memory_order_release );
265 }

267 void unlink_node( node_type * pPred, node_type * pCur, node_type * pHead )
268 {
269 assert( pPred->m_pNext.load(memory_model::memory_order_relaxed).ptr() == pCur );
270
271 node_type * pNext = pCur->m_pNext.load(memory_model::memory_order_relaxed).ptr();
272 pCur->m_pNext.store( marked_node_ptr( pHead, 1 ), memory_model::memory_order_release ); // logical removal + back-link for search
273 pPred->m_pNext.store( marked_node_ptr( pNext ), memory_model::memory_order_release); // physically deleting
274 }

294 void next()
295 {
296 assert( m_pNode != nullptr );
297
298 if ( m_pNode ) {
299 typename gc::Guard g;
300 node_type * pCur = node_traits::to_node_ptr( m_pNode );
301 if ( pCur->m_pNext.load( memory_model::memory_order_relaxed ).ptr() != nullptr ) { // if pCur is not tail node
302 node_type * pNext;
303 do {
304 pNext = pCur->m_pNext.load(memory_model::memory_order_relaxed).ptr();
305 g.assign( node_traits::to_value_ptr( pNext ));
306 } while ( pNext != pCur->m_pNext.load(memory_model::memory_order_relaxed).ptr());
307
308 m_pNode = m_Guard.assign( g.template get<value_type>());
309 }
310 }
311 }
312
313 void skip_deleted()
314 {
315 if ( m_pNode != nullptr ) {
316 typename gc::Guard g;
317 node_type * pNode = node_traits::to_node_ptr( m_pNode );
318
319 // Dummy tail node could not be marked
320 while ( pNode->is_marked()) {
321 node_type * p = pNode->m_pNext.load(memory_model::memory_order_relaxed).ptr();
322 g.assign( node_traits::to_value_ptr( p ));
323 if ( p == pNode->m_pNext.load(memory_model::memory_order_relaxed).ptr())
324 pNode = p;
325 }
326 if ( pNode != node_traits::to_node_ptr( m_pNode ))
327 m_pNode = m_Guard.assign( g.template get<value_type>());
328 }
329 }

331 iterator_type( node_type * pNode )
332 {
333 m_pNode = m_Guard.assign( node_traits::to_value_ptr( pNode ));
334 skip_deleted();

861 void clear()
862 {
863 typename gc::Guard guard;
864 marked_node_ptr h;
865 while ( !empty()) {
866 h = m_Head.m_pNext.load( memory_model::memory_order_relaxed );
867 guard.assign( node_traits::to_value_ptr( h.ptr()));
868 if ( m_Head.m_pNext.load(memory_model::memory_order_acquire) == h ) {
869 m_Head.m_Lock.lock();
870 h->m_Lock.lock();
871
872 unlink_node( &m_Head, h.ptr(), &m_Head );
873 --m_ItemCounter;
874
875 h->m_Lock.unlock();
876 m_Head.m_Lock.unlock();
877
878 retire_node( h.ptr()) ; // free node
879 }
880 }
881 }

927 bool insert_at( node_type * pHead, value_type& val )
928 {
929 position pos;
930 key_comparator cmp;
931
932 while ( true ) {
933 search( pHead, val, pos, key_comparator());
934 {
935 scoped_position_lock alp( pos );
936 if ( validate( pos.pPred, pos.pCur )) {
937 if ( pos.pCur != &m_Tail && cmp( *node_traits::to_value_ptr( *pos.pCur ), val ) == 0 ) {
938 // failed: key already in list
939 m_Stat.onInsertFailed();
940 return false;
941 }
942 else {
943 link_node( node_traits::to_node_ptr( val ), pos.pPred, pos.pCur );
944 break;
945 }
946 }
947 }
948
949 m_Stat.onInsertRetry();
950 }
951
952 ++m_ItemCounter;
953 m_Stat.onInsertSuccess();
954 return true;
955 }

989 template <typename Func>
990 std::pair<bool, bool> update_at( node_type * pHead, value_type& val, Func func, bool bAllowInsert )
991 {
992 position pos;
993 key_comparator cmp;
994
995 while ( true ) {
996 search( pHead, val, pos, key_comparator());
997 {
998 scoped_position_lock alp( pos );
999 if ( validate( pos.pPred, pos.pCur )) {
1000 if ( pos.pCur != &m_Tail && cmp( *node_traits::to_value_ptr( *pos.pCur ), val ) == 0 ) {
1001 // key already in the list
1002
1003 func( false, *node_traits::to_value_ptr( *pos.pCur ) , val );
1004 m_Stat.onUpdateExisting();
1005 return std::make_pair( true, false );
1006 }
1007 else {
1008 // new key
1009 if ( !bAllowInsert ) {
1010 m_Stat.onUpdateFailed();
1011 return std::make_pair( false, false );
1012 }
1013
1014 link_node( node_traits::to_node_ptr( val ), pos.pPred, pos.pCur );
1015 func( true, val, val );
1016 break;
1017 }
1018 }
1019 }
1020
1021 m_Stat.onUpdateRetry();
1022 }
1023
1024 ++m_ItemCounter;
1025 m_Stat.onUpdateNew();
1026 return std::make_pair( true, true );
1027 }

1029 bool unlink_at( node_type * pHead, value_type& val )
1030 {
1031 position pos;
1032 key_comparator cmp;
1033
1034 while ( true ) {
1035 search( pHead, val, pos, key_comparator());
1036 {
1037 int nResult = 0;
1038 {
1039 scoped_position_lock alp( pos );
1040 if ( validate( pos.pPred, pos.pCur )) {
1041 if ( pos.pCur != &m_Tail
1042 && cmp( *node_traits::to_value_ptr( *pos.pCur ), val ) == 0
1043 && node_traits::to_value_ptr( pos.pCur ) == &val )
1044 {
1045 // item found
1046 unlink_node( pos.pPred, pos.pCur, pHead );
1047 nResult = 1;
1048 }
1049 else
1050 nResult = -1;
1051 }
1052 }
1053
1054 if ( nResult ) {
1055 if ( nResult > 0 ) {
1056 --m_ItemCounter;
1057 retire_node( pos.pCur );
1058 m_Stat.onEraseSuccess();
1059 return true;
1060 }
1061
1062 m_Stat.onEraseFailed();
1063 return false;
1064 }
1065 }
1066
1067 m_Stat.onEraseRetry();
1068 }
