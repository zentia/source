28// Fast allocation in a page: just pop from the free list.
29// Fall back to generic allocation only if the list is empty.
30// Note: in release mode the (inlined) routine is about 7 instructions with a single test.
31extern inline void* _mi_page_malloc_zero(mi_heap_t* heap, mi_page_t* page, size_t size, bool zero) mi_attr_noexcept
32{
33 if (page->block_size != 0) { // not the empty heap
34 mi_assert_internal(mi_page_block_size(page) >= size);
35 mi_assert_internal(_mi_is_aligned(page, MI_PAGE_ALIGN));
36 mi_assert_internal(_mi_ptr_page(page)==page);
37 }
38
39 // check the free list
40 mi_block_t* const block = page->free;
41 if mi_unlikely(block == NULL) {
42 return _mi_malloc_generic(heap, size, zero, 0);
43 }
44 mi_assert_internal(block != NULL && _mi_ptr_page(block) == page);
45
46 // pop from the free list
47 page->free = mi_block_next(page, block);
48 page->used++;
49 mi_assert_internal(page->free == NULL || _mi_ptr_page(page->free) == page);
50 mi_assert_internal(page->block_size < MI_MAX_ALIGN_SIZE || _mi_is_aligned(block, MI_MAX_ALIGN_SIZE));
51
52 #if MI_DEBUG>3
53 if (page->free_is_zero && size > sizeof(*block)) {
54 mi_assert_expensive(mi_mem_is_zero(block+1,size - sizeof(*block)));
55 }
56 #endif
57
58 // allow use of the block internally
59 // note: when tracking we need to avoid ever touching the MI_PADDING since
60 // that is tracked by valgrind etc. as non-accessible (through the red-zone, see `mimalloc/track.h`)
61 mi_track_mem_undefined(block, mi_page_usable_block_size(page));
62
63 // zero the block? note: we need to zero the full block size (issue #63)
64 if mi_unlikely(zero) {
65 mi_assert_internal(page->block_size != 0); // do not call with zero'ing for huge blocks (see _mi_malloc_generic)
66 mi_assert_internal(!mi_page_is_huge(page));
67 #if MI_PADDING
68 mi_assert_internal(page->block_size >= MI_PADDING_SIZE);
69 #endif
70 if (page->free_is_zero) {
71 block->next = 0;
72 mi_track_mem_defined(block, page->block_size - MI_PADDING_SIZE);
73 }
74 else {
75 _mi_memzero_aligned(block, page->block_size - MI_PADDING_SIZE);
76 }
77 }
78
79 #if (MI_DEBUG>0) && !MI_TRACK_ENABLED && !MI_TSAN
80 if (!zero && !mi_page_is_huge(page)) {
81 memset(block, MI_DEBUG_UNINIT, mi_page_usable_block_size(page));
82 }
83 #elif (MI_SECURE!=0)
84 if (!zero) { block->next = 0; } // don't leak internal data
85 #endif
86
87 #if (MI_STAT>0)
88 const size_t bsize = mi_page_usable_block_size(page);
89 if (bsize <= MI_LARGE_MAX_OBJ_SIZE) {
90 mi_heap_stat_increase(heap, malloc_normal, bsize);
91 mi_heap_stat_counter_increase(heap, malloc_normal_count, 1);
92 #if (MI_STAT>1)
93 const size_t bin = _mi_bin(bsize);
94 mi_heap_stat_increase(heap, malloc_bins[bin], 1);
95 #endif
96 }
97 #endif
98
99 #if MI_PADDING // && !MI_TRACK_ENABLED
100 mi_padding_t* const padding = (mi_padding_t*)((uint8_t*)block + mi_page_usable_block_size(page));
101 ptrdiff_t delta = ((uint8_t*)padding - (uint8_t*)block - (size - MI_PADDING_SIZE));
102 #if (MI_DEBUG>=2)
103 mi_assert_internal(delta >= 0 && mi_page_usable_block_size(page) >= (size - MI_PADDING_SIZE + delta));
104 #endif
105 mi_track_mem_defined(padding,sizeof(mi_padding_t)); // note: re-enable since mi_page_usable_block_size may set noaccess
106 padding->canary = mi_ptr_encode_canary(page,block,page->keys);
107 padding->delta = (uint32_t)(delta);
108 #if MI_PADDING_CHECK
109 if (!mi_page_is_huge(page)) {
110 uint8_t* fill = (uint8_t*)padding - delta;
111 const size_t maxpad = (delta > MI_MAX_ALIGN_SIZE ? MI_MAX_ALIGN_SIZE : delta); // set at most N initial padding bytes
112 for (size_t i = 0; i < maxpad; i++) { fill[i] = MI_DEBUG_PADDING; }
113 }
114 #endif
115 #endif
116
117 return block;
118}

132static inline mi_decl_restrict void* mi_heap_malloc_small_zero(mi_heap_t* heap, size_t size, bool zero) mi_attr_noexcept {
133 mi_assert(heap != NULL);
134 mi_assert(size <= MI_SMALL_SIZE_MAX);
135 #if MI_DEBUG
136 const uintptr_t tid = _mi_thread_id();
137 mi_assert(heap->tld->thread_id == 0 || heap->tld->thread_id == tid); // heaps are thread local
138 #endif
139 #if (MI_PADDING || MI_GUARDED)
140 if (size == 0) { size = sizeof(void*); }
141 #endif
142 #if MI_GUARDED
143 if (mi_heap_malloc_use_guarded(heap,size)) {
144 return _mi_heap_malloc_guarded(heap, size, zero);
145 }
146 #endif
147
148 // get page in constant time, and allocate from it
149 mi_page_t* page = _mi_heap_get_free_small_page(heap, size + MI_PADDING_SIZE);
150 void* const p = _mi_page_malloc_zero(heap, page, size + MI_PADDING_SIZE, zero);
151 mi_track_malloc(p,size,zero);
152
153 #if MI_STAT>1
154 if (p != NULL) {
155 if (!mi_heap_is_initialized(heap)) { heap = mi_prim_get_default_heap(); }
156 mi_heap_stat_increase(heap, malloc_requested, mi_usable_size(p));
157 }
158 #endif
159 #if MI_DEBUG>3
160 if (p != NULL && zero) {
161 mi_assert_expensive(mi_mem_is_zero(p, size));
162 }
163 #endif
164 return p;
165}

176// The main allocation function
177extern inline void* _mi_heap_malloc_zero_ex(mi_heap_t* heap, size_t size, bool zero, size_t huge_alignment) mi_attr_noexcept {
178 // fast path for small objects
179 if mi_likely(size <= MI_SMALL_SIZE_MAX) {
180 mi_assert_internal(huge_alignment == 0);
181 return mi_heap_malloc_small_zero(heap, size, zero);
182 }
183 #if MI_GUARDED
184 else if (huge_alignment==0 && mi_heap_malloc_use_guarded(heap,size)) {
185 return _mi_heap_malloc_guarded(heap, size, zero);
186 }
187 #endif
188 else {
189 // regular allocation
190 mi_assert(heap!=NULL);
191 mi_assert(heap->tld->thread_id == 0 || heap->tld->thread_id == _mi_thread_id()); // heaps are thread local
192 void* const p = _mi_malloc_generic(heap, size + MI_PADDING_SIZE, zero, huge_alignment); // note: size can overflow but it is detected in malloc_generic
193 mi_track_malloc(p,size,zero);
194
195 #if MI_STAT>1
196 if (p != NULL) {
197 if (!mi_heap_is_initialized(heap)) { heap = mi_prim_get_default_heap(); }
198 mi_heap_stat_increase(heap, malloc_requested, mi_usable_size(p));
199 }
200 #endif
201 #if MI_DEBUG>3
202 if (p != NULL && zero) {
203 mi_assert_expensive(mi_mem_is_zero(p, size));
204 }
205 #endif
206 return p;
207 }

271void* _mi_heap_realloc_zero(mi_heap_t* heap, void* p, size_t newsize, bool zero) mi_attr_noexcept {
272 // if p == NULL then behave as malloc.
273 // else if size == 0 then reallocate to a zero-sized block (and don't return NULL, just as mi_malloc(0)).
274 // (this means that returning NULL always indicates an error, and `p` will not have been freed in that case.)
275 const size_t size = (p==NULL ? 0 : _mi_usable_size(p,"mi_realloc"));
276 if mi_unlikely(newsize <= size && newsize >= (size / 2) && newsize > 0) { // note: newsize must be > 0 or otherwise we return NULL for realloc(NULL,0)
277 mi_assert_internal(p!=NULL);
278 // todo: do not track as the usable size is still the same in the free; adjust potential padding?
279 // mi_track_resize(p,size,newsize)
280 // if (newsize < size) { mi_track_mem_noaccess((uint8_t*)p + newsize, size - newsize); }
281 return p; // reallocation still fits and not more than 50% waste
282 }
283 void* newp = mi_heap_malloc(heap,newsize);
284 if mi_likely(newp != NULL) {
285 if (zero && newsize > size) {
286 // also set last word in the previous allocation to zero to ensure any padding is zero-initialized
287 const size_t start = (size >= sizeof(intptr_t) ? size - sizeof(intptr_t) : 0);
288 _mi_memzero((uint8_t*)newp + start, newsize - start);
289 }
290 else if (newsize == 0) {
291 ((uint8_t*)newp)[0] = 0; // work around for applications that expect zero-reallocation to be zero initialized (issue #725)
292 }
293 if mi_likely(p != NULL) {
294 const size_t copysize = (newsize > size ? size : newsize);
295 mi_track_mem_defined(p,copysize); // _mi_useable_size may be too large for byte precise memory tracking..
296 _mi_memcpy(newp, p, copysize);
297 mi_free(p); // only free the original pointer if successful
298 }
299 }
300 return newp;
301}

460/*-------------------------------------------------------
461C++ new and new_aligned
462The standard requires calling into `get_new_handler` and
463throwing the bad_alloc exception on failure. If we compile
464with a C++ compiler we can implement this precisely. If we
465use a C compiler we cannot throw a `bad_alloc` exception
466but we call `exit` instead (i.e. not returning).
467-------------------------------------------------------*/
468
469#ifdef __cplusplus
470#include <new>
471static bool mi_try_new_handler(bool nothrow) {
472 #if defined(_MSC_VER) || (__cplusplus >= 201103L)
473 std::new_handler h = std::get_new_handler();
474 #else
475 std::new_handler h = std::set_new_handler();
476 std::set_new_handler(h);
477 #endif
478 if (h==NULL) {
479 _mi_error_message(ENOMEM, "out of memory in 'new'");
480 #if defined(_CPPUNWIND) || defined(__cpp_exceptions) // exceptions are not always enabled
481 if (!nothrow) {
482 throw std::bad_alloc();
483 }
484 #else
485 MI_UNUSED(nothrow);
486 #endif
487 return false;
488 }
489 else {
490 h();
491 return true;
492 }
493}
494#else
495typedef void (*std_new_handler_t)(void);
496
497#if (defined(__GNUC__) || (defined(__clang__) && !defined(_MSC_VER))) // exclude clang-cl, see issue #631
498std_new_handler_t __attribute__((weak)) _ZSt15get_new_handlerv(void) {
499 return NULL;
500}
501static std_new_handler_t mi_get_new_handler(void) {
502 return _ZSt15get_new_handlerv();
503}
504#else
505// note: on windows we could dynamically link to `?get_new_handler@std@@YAP6AXXZXZ`.
506static std_new_handler_t mi_get_new_handler() {
507 return NULL;
508}
509#endif
510
511static bool mi_try_new_handler(bool nothrow) {
512 std_new_handler_t h = mi_get_new_handler();
513 if (h==NULL) {
514 _mi_error_message(ENOMEM, "out of memory in 'new'");
515 if (!nothrow) {
516 abort(); // cannot throw in plain C, use abort
517 }
518 return false;
519 }
520 else {
521 h();
522 return true;
523 }
524}
525#endif

527mi_decl_export mi_decl_noinline void* mi_heap_try_new(mi_heap_t* heap, size_t size, bool nothrow ) {
528 void* p = NULL;
529 while(p == NULL && mi_try_new_handler(nothrow)) {
530 p = mi_heap_malloc(heap,size);
531 }
532 return p;
533}

610#if MI_GUARDED
611// We always allocate a guarded allocation at an offset (`mi_page_has_aligned` will be true).
612// We then set the first word of the block to `0` for regular offset aligned allocations (in `alloc-aligned.c`)
613// and the first word to `~0` for guarded allocations to have a correct `mi_usable_size`
614
615static void* mi_block_ptr_set_guarded(mi_block_t* block, size_t obj_size) {
616 // TODO: we can still make padding work by moving it out of the guard page area
617 mi_page_t* const page = _mi_ptr_page(block);
618 mi_page_set_has_aligned(page, true);
619 block->next = MI_BLOCK_TAG_GUARDED;
620
621 // set guard page at the end of the block
622 const size_t block_size = mi_page_block_size(page); // must use `block_size` to match `mi_free_local`
623 const size_t os_page_size = _mi_os_page_size();
624 mi_assert_internal(block_size >= obj_size + os_page_size + sizeof(mi_block_t));
625 if (block_size < obj_size + os_page_size + sizeof(mi_block_t)) {
626 // should never happen
627 mi_free(block);
628 return NULL;
629 }
630 uint8_t* guard_page = (uint8_t*)block + block_size - os_page_size;
631 // note: the alignment of the guard page relies on blocks being os_page_size aligned which
632 // is ensured in `mi_arena_page_alloc_fresh`.
633 mi_assert_internal(_mi_is_aligned(block, os_page_size));
634 mi_assert_internal(_mi_is_aligned(guard_page, os_page_size));
635 if (!page->memid.is_pinned && _mi_is_aligned(guard_page, os_page_size)) {
636 _mi_os_protect(guard_page, os_page_size);
637 }
638 else {
639 _mi_warning_message("unable to set a guard page behind an object due to pinned memory (large OS pages?) (object %p of size %zu)\n", block, block_size);
640 }
641
642 // align pointer just in front of the guard page
643 size_t offset = block_size - os_page_size - obj_size;
644 mi_assert_internal(offset > sizeof(mi_block_t));
645 if (offset > MI_PAGE_MAX_OVERALLOC_ALIGN) {
646 // give up to place it right in front of the guard page if the offset is too large for unalignment
647 offset = MI_PAGE_MAX_OVERALLOC_ALIGN;
648 }
649 void* p = (uint8_t*)block + offset;
650 mi_track_align(block, p, offset, obj_size);
651 mi_track_mem_defined(block, sizeof(mi_block_t));
652 return p;
653}

655mi_decl_restrict void* _mi_heap_malloc_guarded(mi_heap_t* heap, size_t size, bool zero) mi_attr_noexcept
656{
657 #if defined(MI_PADDING_SIZE)
658 mi_assert(MI_PADDING_SIZE==0);
659 #endif
660 // allocate multiple of page size ending in a guard page
661 // ensure minimal alignment requirement?
662 const size_t os_page_size = _mi_os_page_size();
663 const size_t obj_size = (mi_option_is_enabled(mi_option_guarded_precise) ? size : _mi_align_up(size, MI_MAX_ALIGN_SIZE));
664 const size_t bsize = _mi_align_up(_mi_align_up(obj_size, MI_MAX_ALIGN_SIZE) + sizeof(mi_block_t), MI_MAX_ALIGN_SIZE);
665 const size_t req_size = _mi_align_up(bsize + os_page_size, os_page_size);
666 mi_block_t* const block = (mi_block_t*)_mi_malloc_generic(heap, req_size, zero, 0 /* huge_alignment */);
667 if (block==NULL) return NULL;
668 void* const p = mi_block_ptr_set_guarded(block, obj_size);
669
670 // stats
671 mi_track_malloc(p, size, zero);
672 if (p != NULL) {
673 if (!mi_heap_is_initialized(heap)) { heap = mi_prim_get_default_heap(); }
674 #if MI_STAT>1
675 mi_heap_stat_increase(heap, malloc_requested, mi_usable_size(p));
676 #endif
677 mi_heap_stat_counter_increase(heap, malloc_guarded_count, 1);
678 }
679 #if MI_DEBUG>3
680 if (p != NULL && zero) {
681 mi_assert_expensive(mi_mem_is_zero(p, size));
682 }
683 #endif
684 return p;
685}
