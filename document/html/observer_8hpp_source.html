535 // task summary

804// Procedure: summary
805inline void TFProfObserver::summary(std::ostream& os) const {
806
807 using namespace std::chrono;
808
809 Summary summary;
810 std::optional<observer_stamp_t> view_beg, view_end;
811
812 // find the first non-empty worker
813 size_t first;
814 for(first = 0; first<_timeline.segments.size(); ++first) {
815 if(_timeline.segments[first].size() > 0) {
816 break;
817 }
818 }
819
820 // not timeline data to dump
821 if(first == _timeline.segments.size()) {
822 goto end_of_summary;
823 }
824
825 for(size_t w=first; w<_timeline.segments.size(); w++) {
826 for(size_t l=0; l<_timeline.segments[w].size(); l++) {
827
828 if(_timeline.segments[w][l].empty()) {
829 continue;
830 }
831
832 // worker w at level l
833 WorkerSummary ws;
834 ws.id = w;
835 ws.level = l;
836 ws.count = _timeline.segments[w][l].size();
837
838 // scan all tasks at level l
839 for(size_t i=0; i<_timeline.segments[w][l].size(); ++i) {
840
841 // update the entire span
842 auto& s = _timeline.segments[w][l][i];
843 view_beg = view_beg ? (std::min)(*view_beg, s.beg) : s.beg;
844 view_end = view_end ? (std::max)(*view_end, s.end) : s.end;
845
846 // update the task summary
847 size_t t = duration_cast<microseconds>(s.end - s.beg).count();
848
849 auto& x = summary.tsum[static_cast<int>(s.type)];
850 x.count += 1;
851 x.total_span += t;
852 x.min_span = (x.count == 1) ? t : (std::min)(t, x.min_span);
853 x.max_span = (x.count == 1) ? t : (std::max)(t, x.max_span);
854
855 // update the worker summary
856 ws.total_span += t;
857 ws.min_span = (i == 0) ? t : (std::min)(t, ws.min_span);
858 ws.max_span = (i == 0) ? t : (std::max)(t, ws.max_span);
859
860 auto&y = ws.tsum[static_cast<int>(s.type)];
861 y.count += 1;
862 y.total_span += t;
863 y.min_span = (y.count == 1) ? t : (std::min)(t, y.min_span);
864 y.max_span = (y.count == 1) ? t : (std::max)(t, y.max_span);
865
866 // update the delay
867 //if(i) {
868 // size_t d = duration_cast<nanoseconds>(
869 // s.beg - _timeline.segments[w][l][i-1].end
870 // ).count();
871 // ws.total_delay += d;
872 // ws.min_delay = (i == 1) ? d : std::min(ws.min_delay, d);
873 // ws.max_delay = (i == 1) ? d : std::max(ws.max_delay, d);
874 //}
875 }
876 summary.wsum.push_back(ws);
877 }
878 }
879
880 end_of_summary:
881
882 size_t view = 0;
883 if(view_beg && view_end) {
884 view = duration_cast<microseconds>(*view_end - *view_beg).count();
885 }
886
887 os << "==Observer " << _timeline.uid << ": "
888 << num_workers() << " workers completed "
889 << num_tasks() << " tasks in "
890 << view << " us\n";
891
892 summary.dump(os);
893}
