26 //
27 // Linearly searches a typelist using compile-time recursion to inspect each T in
28 // the typelist and returns its index, if the type is found. If the T isn't found
29 // in the typelist -1 is returned.

39 static EA_CONSTEXPR_OR_CONST int value = is_same_v<T, Head> ? I : get_type_index<I + 1, T, Types...>::value;

120 // type_list
121 //
122 // type_list is a simple struct that allows us to pass template parameter packs
123 // around in a single struct, and deduce parameter packs from function arguments
124 // like so:
125 //
126 // template <typename... Ts> void foo(type_list<Ts...>);
127 // foo(type_list<A, B, C>); // deduces Ts... as A, B, C
128 //
129 template <typename...> struct type_list {};
130

133 // unique_type_list
134 //
135 // unique_type_list is a meta-function which takes a parameter pack as its
136 // argument, and returns a type_list with duplicate types removed, like so:
137 //
138 // unique_type_list<int, int, string>::type; // type = type_list<int, string>
139 // unique_type_list<int, string, string>::type; // type = type_list<int, string>
140 //
141 // To use unique_type_list, specialize a variadic class template for a single
142 // type parameter, which is type_list<Ts...>:
143 //
144 // template <typename... Ts> struct foo {};
145 // template <typename... Ts> struct foo<type_list<Ts...>> {};
146 //
147 // Then instantiate the template with unique_type_list_t<Ts...> as its parameter:
148 //
149 // template <typename... Ts> struct bar : public foo<unique_type_list_t<Ts...>> {}
150 //
151 // See overload_set below for examples.
152 template <typename T, typename... Ts>
153 struct unique_type_list : public unique_type_list<Ts...>
154 {
155 template <typename... Args>
156 static enable_if_t<!disjunction_v<is_same<T, Args>...>, type_list<T, Args...>>
157 types(type_list<Args...>);
158
159 template <typename... Args>
160 static enable_if_t<disjunction_v<is_same<T, Args>...>, type_list<Args...>>
161 types(type_list<Args...>);
162
163 typedef decltype(types(declval<typename unique_type_list<Ts...>::type>())) type;
164 };
165
166 template <typename T>
167 struct unique_type_list<T>
168 {
169 using type = type_list<T>;
170 };
171
172 template <typename... Ts>
173 using unique_type_list_t = typename unique_type_list<Ts...>::type;

177 // overload_resolution_t
178 //
179 // Given an input type and a typelist (which is a stand-in for alternative
180 // function overloads) this traits will return the same type chosen as if
181 // overload_resolution has selected a function to run.

184 // a single overload of an individual type
185 template <typename T>
186 struct overload
187 {
188 // Overload is implicitly convertible to the surrogated function
189 // call for pointer to member functions (pmf). This gets around
190 // variadic pack expansion in a class using statement being a C++17
191 // language feature. It is the core mechanism of aggregating all the
192 // individual overloads into the overload_set structure.
193 using F = T (*)(T);

195 };
