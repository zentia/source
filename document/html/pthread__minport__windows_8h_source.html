240// Win XP hasn't Condition variables!
241//
242// Douglas C. Schmidt and Irfan Pyarali
243// Strategies for Implementing POSIX Condition Variables on Win32
244// http://www.cs.wustl.edu/~schmidt/win32-cv-1.html
245
246typedef struct
247{
248 int waiters_count_;
249 // Count of the number of waiters.
250
251 CRITICAL_SECTION waiters_count_lock_;
252 // Serialize access to <waiters_count_>.
253
254 int release_count_;
255 // Number of threads to release via a <pthread_cond_broadcast> or a
256 // <pthread_cond_signal>.
257
258 int wait_generation_count_;
259 // Keeps track of the current "generation" so that we don't allow
260 // one thread to steal all the "releases" from the broadcast.
261
262 HANDLE event_;
263 // A manual-reset event that's used to block and release waiting
264 // threads.
265} pthread_cond_t;
266
267INLINE int pthread_cond_init(pthread_cond_t RESTRICT * cv,
268 const pthread_condattr_t RESTRICT * attr) {
269 cv->waiters_count_ = 0;
270 cv->wait_generation_count_ = 0;
271 cv->release_count_ = 0;
272
273 // Create a manual-reset event.
274 cv->event_ = CreateEvent (NULL, // no security
275 TRUE, // manual-reset
276 FALSE, // non-signaled initially
277 NULL); // unnamed
278
279 pthread_mutex_init(&cv->waiters_count_lock_,NULL);
280 return 0;
281}
282
283INLINE int pthread_cond_wait(pthread_cond_t RESTRICT * cv,
284 pthread_mutex_t RESTRICT * external_mutex) {
285 // Avoid race conditions.
286 EnterCriticalSection (&cv->waiters_count_lock_);
287
288 // Increment count of waiters.
289 cv->waiters_count_++;
290
291 // Store current generation in our activation record.
292 int my_generation = cv->wait_generation_count_;
293
294 LeaveCriticalSection (&cv->waiters_count_lock_);
295 LeaveCriticalSection (external_mutex);
296
297 for (;;) {
298 // Wait until the event is signaled.
299 WaitForSingleObject (cv->event_, INFINITE);
300
301 EnterCriticalSection (&cv->waiters_count_lock_);
302 // Exit the loop when the <cv->event_> is signaled and
303 // there are still waiting threads from this <wait_generation>
304 // that haven't been released from this wait yet.
305 int wait_done = cv->release_count_ > 0
306 && cv->wait_generation_count_ != my_generation;
307 LeaveCriticalSection (&cv->waiters_count_lock_);
308
309 if (wait_done)
310 break;
311 }
312
313 EnterCriticalSection (external_mutex);
314 EnterCriticalSection (&cv->waiters_count_lock_);
315 cv->waiters_count_--;
316 cv->release_count_--;
317 int last_waiter = cv->release_count_ == 0;
318 LeaveCriticalSection (&cv->waiters_count_lock_);
319
320 if (last_waiter)
321 // We're the last waiter to be notified, so reset the manual event.
322 ResetEvent (cv->event_);
323 return 0;
324}
325
326INLINE int pthread_cond_signal(pthread_cond_t *cv) {
327 EnterCriticalSection (&cv->waiters_count_lock_);
328 if (cv->waiters_count_ > cv->release_count_) {
329 SetEvent (cv->event_); // Signal the manual-reset event.
330 cv->release_count_++;
331 cv->wait_generation_count_++;
332 }
333 LeaveCriticalSection (&cv->waiters_count_lock_);
334 return 0;
335}
336
337INLINE int pthread_cond_broadcast(pthread_cond_t *cv) {
338 EnterCriticalSection (&cv->waiters_count_lock_);
339 if (cv->waiters_count_ > 0) {
340 SetEvent (cv->event_);
341 // Release all the threads in this generation.
342 cv->release_count_ = cv->waiters_count_;
343
344 // Start a new generation.
345 cv->wait_generation_count_++;
346 }
347 LeaveCriticalSection (&cv->waiters_count_lock_);
348 return 0;
349}
