462 template <typename T, unsigned int Padding >
463 struct apply_padding
464 {
465 private:
466 enum { padding = Padding & ~padding_flags };
467
468 public:
469 static constexpr const size_t c_nPadding =
470 static_cast<unsigned int>(padding) == static_cast<unsigned int>(cache_line_padding) ? cds::c_nCacheLineSize :
471 static_cast<unsigned int>(padding) == static_cast<unsigned int>(no_special_padding) ? 0 : padding;
472
473 static_assert( (c_nPadding & (c_nPadding - 1)) == 0, "Padding must be a power-of-two number" );
474
475 typedef apply_padding_helper< T,
476 c_nPadding,
477 c_nPadding == 0,
478 sizeof( T ) < c_nPadding ? padding_datasize_less : sizeof( T ) == c_nPadding ? padding_datasize_equal : padding_datasize_greater,
479 (Padding & padding_tiny_data_only) != 0
480 > result;
481
482 typedef typename result::type type;
483
484 typedef typename std::conditional<
485 std::is_same< typename result::padding_type, void >::value,
486 unsigned int,
487 typename result::padding_type
488 >::type padding_type;
489 };
