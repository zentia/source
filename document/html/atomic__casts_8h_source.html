120template <typename Pun, typename T, eastl::enable_if_t<!eastl::is_same_v<Pun, T>, int> = 0>
121EASTL_FORCE_INLINE Pun AtomicTypePunCast(const T& fromType) EA_NOEXCEPT
122{
123 static_assert(sizeof(Pun) == sizeof(T), "eastl::atomic<T> : Pun and T must be the same size for type punning!");
124
129 typename eastl::aligned_storage<sizeof(Pun), alignof(Pun)>::type ret;
130 memcpy(eastl::addressof(ret), eastl::addressof(fromType), sizeof(Pun));
131 return reinterpret_cast<Pun&>(ret);
132}

141template <typename T>
142EASTL_FORCE_INLINE T AtomicNegateOperand(T val) EA_NOEXCEPT
143{
144 static_assert(eastl::is_integral<T>::value, "eastl::atomic<T> : Integral Negation must be an Integral type!");
145 static_assert(!eastl::is_volatile<T>::value, "eastl::atomic<T> : T must not be volatile!");
146
147 return static_cast<T>(0U - static_cast<eastl::make_unsigned_t<T>>(val));
148}
