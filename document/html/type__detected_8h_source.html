52 // is_detected
53 //
54 // Checks if some supplied arguments (Args) respect a constraint (Op).
55 // is_detected expands to true_type if the arguments respect the constraint, false_type otherwise.
56 // This helper is convenient to use for compile time introspection.
57 //
58 // https://en.cppreference.com/w/cpp/experimental/is_detected
59 //
60 // Example:
61 // template <class T, class U>
62 // using detect_can_use_addition_operator = decltype(declval<T>() + declval<U>());
63 //
64 // template <class T, class U>
65 // void sum(const T& t, const U& u)
66 // {
67 // static_assert(is_detected<detect_can_use_addition_operator, T, U>::value, "Supplied types cannot be summedtogether.");
68 // // or...
69 // static_assert(is_detected_v<detect_can_use_addition_operator, T, U>, "Supplied types cannot be summedtogether.");
70 // return t + u;
71 // }
72 //
74 template <template <class...> class Op, class... Args>
75 using is_detected = typename internal::detector<nonesuch, void, Op, Args...>::value_t;
76
77#if EASTL_VARIABLE_TEMPLATES_ENABLED
78 template <template <class...> class Op, class... Args>

80#endif

83 // detected_t
84 //
85 // Check which type we obtain after expanding some arguments (Args) over a constraint (Op).
86 // If the constraint cannot be applied, the result type will be nonesuch.
87 //
88 // https://en.cppreference.com/w/cpp/experimental/is_detected
89 //
90 // Example:
91 // template <class T, class U>
92 // using detect_can_use_addition_operator = decltype(declval<T>() + declval<U>());
93 //
94 // using result_type = detected_t<detect_can_use_addition_operator, int, int>;
95 // // result_type == int
96 // using failed_result_type = detected_t<detect_can_use_addition_operator, int, string>;
97 // // failed_result_type == nonesuch
98 //
100 template <template <class...> class Op, class... Args>
101 using detected_t = typename internal::detector<nonesuch, void, Op, Args...>::type;
102
104 // detected_or
105 //
106 // Checks if some supplied arguments (Args) respect a constraint (Op).
107 // Expand to a struct that contains two type aliases:
108 // - type: the type we obtain after expanding some arguments (Args) over a constraint (Op).
109 // If the constraint cannot be applied, the result type will be the suplied Default type.
110 // - value_t: true_type if the arguments respect the constraint, false_type otherwise.
111 //
112 // https://en.cppreference.com/w/cpp/experimental/is_detected
113 //
114 // Example:
115 // template <class T, class U>
116 // using detected_calling_foo = decltype(declval<T>().foo());
117 //
118 // using result = detected_or<bool, detected_calling_foo, std::string>; // std::string doesn't have foo member.
119 // function.
120 // // result::type == bool
121 // // result::value_t == false_type
122 //
124 template <class Default, template <class...> class Op, class... Args>
125 using detected_or = internal::detector<Default, void, Op, Args...>;
126
128 // detected_or_t
129 //
130 // Equivalent to detected_or<Default, Op, Args...>::type.
131 //
133 template <class Default, template <class...> class Op, class... Args>
134 using detected_or_t = typename detected_or<Default, Op, Args...>::type;
135
137 // is_detected_exact
138 //
139 // Check that the type we obtain after expanding some arguments (Args) over a constraint (Op) is equivalent to
140 // Expected.
141 //
142 // template <class T, class U>
143 // using detected_calling_size = decltype(declval<T>().size());
144 //
145 // using result = is_detected_exact<int, detected_calling_size, std::string>;
146 // result == false_type // std::string::size returns eastl_size_t which is not the same as int.
147 //
149 template <class Expected, template <class...> class Op, class... Args>
150 using is_detected_exact = is_same<Expected, detected_t<Op, Args...>>;
151
152#if EASTL_VARIABLE_TEMPLATES_ENABLED
153 template <class Expected, template <class...> class Op, class... Args>
154 EA_CONSTEXPR bool is_detected_exact_v = is_detected_exact<Expected, Op, Args...>::value;
155#endif

158 // is_detected_convertible
159 //
160 // Check that the type we obtain after expanding some arguments (Args) over a constraint (Op) is convertible to
161 // Expected.
162 //
163 // template <class T, class U>
164 // using detected_calling_size = decltype(declval<T>().size());
165 //
166 // using result = is_detected_convertible<int, detected_calling_size, std::string>;
167 // result == true_type // std::string::size returns eastl_size_t which is convertible to int.
168 //
170 template <class To, template <class...> class Op, class... Args>
171 using is_detected_convertible = is_convertible<detected_t<Op, Args...>, To>;
172

174 template <class To, template <class...> class Op, class... Args>
175 EA_CONSTEXPR bool is_detected_convertible_v = is_detected_convertible<To, Op, Args...>::value;
176#endif
