146 clDeviceInUse.push_back(false);

151 }
152 delete [] deviceIds;

154 //std::cerr << "OpenCL platform detection - end \n";

157 for(std::vector<cl_device_id>::iterator iter=clDevices.begin(); iter < clDevices.end(); ++iter) {

168 }

204 std::vector<ssize_t> coAllocateGPUDeviceRR(size_t n=1, ssize_t preferred_dev=-1, bool exclusive=false, bool identical=false) {
205 cl_device_type dt;
206 size_t count = n;
207 std::vector<ssize_t> ret;
208 pthread_mutex_lock(&instanceMutex);
209 //start from either the user-defined preferred_dev or the last RR-allocated device
210 size_t dev = (preferred_dev>=0)? (preferred_dev%clDevices.size()): lastAssigned;
211 //perform multiple passes over the device list,
212 //stop if no allocation happens in one pass
213 size_t count_pre = count;
214 while (true) {
215 count_pre = count;
216 for (size_t i = 0; i < clDevices.size(); i++) {
217 clGetDeviceInfo(clDevices[dev], CL_DEVICE_TYPE,
218 sizeof(cl_device_type), &(dt), NULL);
219 if ((!clDeviceInUse[dev] | !exclusive) //dev is free or not exclusive mode
220 && ((dt) & CL_DEVICE_TYPE_GPU)) { //dev is a GPU
221 ret.push_back(dev);
222 if (--count == 0)
223 break;
224 }
225 ++dev;
226 dev %= clDevices.size();
227 } //end pass
228 if(!count) { // commit
229 // TODO check if identical
230 for (size_t i=0; i<ret.size();++i)
231 clDeviceInUse[ret[i]]=true;
232 lastAssigned=dev;
233 break;
234 }
235 if(count_pre == count) { // roll back
236 //std::cerr << "Not enough GPUs: aborting\n";
237 ret.clear();
238 break;
239 }
240 //continue to next pass
241 }
242 pthread_mutex_unlock(&instanceMutex);
243 return ret;
244 }
