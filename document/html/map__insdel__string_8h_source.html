157 template <class Map>
158 void do_test( Map& testMap )
159 {
160 typedef Inserter<Map> inserter;
161 typedef Deleter<Map> deleter;
162
163 cds_test::thread_pool& pool = get_pool();
164 pool.add( new inserter( pool, testMap ), s_nInsertThreadCount );
165 pool.add( new deleter( pool, testMap ), s_nDeleteThreadCount );
166
167 propout() << std::make_pair( "insert_thread_count", s_nInsertThreadCount )
168 << std::make_pair( "delete_thread_count", s_nDeleteThreadCount )
169 << std::make_pair( "pass_count", s_nThreadPassCount )
170 << std::make_pair( "map_size", s_nMapSize );
171
172 std::chrono::milliseconds duration = pool.run();
173
174 propout() << std::make_pair( "duration", duration );
175
176 size_t nInsertSuccess = 0;
177 size_t nInsertFailed = 0;
178 size_t nDeleteSuccess = 0;
179 size_t nDeleteFailed = 0;
180
181 for ( size_t i = 0; i < pool.size(); ++i ) {
182 cds_test::thread& thr = pool.get( i );
183 switch ( thr.type()) {
184 case insert_thread:
185 {
186 inserter& t = static_cast<inserter&>(thr);
187 nInsertSuccess += t.m_nInsertSuccess;
188 nInsertFailed += t.m_nInsertFailed;
189 }
190 break;
191 case delete_thread:
192 {
193 deleter& t = static_cast<deleter&>(thr);
194 nDeleteSuccess += t.m_nDeleteSuccess;
195 nDeleteFailed += t.m_nDeleteFailed;
196 }
197 break;
198 default:
199 assert( false );
200 }
201 }
202
203 propout()
204 << std::make_pair( "insert_success", nInsertSuccess )
205 << std::make_pair( "insert_failed", nInsertFailed )
206 << std::make_pair( "delete_success", nDeleteSuccess )
207 << std::make_pair( "delete_failed", nDeleteFailed )
208 << std::make_pair( "finish_map_size", testMap.size());
209
210 check_before_cleanup( testMap );
211
212 //testMap.clear();
213 for ( auto const& str: s_arrKeys )
214 testMap.erase( str );
215 EXPECT_TRUE( testMap.empty());
216 EXPECT_EQ( testMap.size(), 0u );
217
218 additional_check( testMap );
219 print_stat( propout(), testMap );
220 additional_cleanup( testMap );
221 }
