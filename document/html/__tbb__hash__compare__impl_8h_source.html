69template<typename E, typename S, typename A>
70__TBB_DEPRECATED_MSG("tbb::tbb_hasher is deprecated, use std::hash") inline size_t tbb_hasher( const std::basic_string<E,S,A>& s ) {
71 size_t h = 0;
72 for( const E* c = s.c_str(); *c; ++c )
73 h = static_cast<size_t>(*c) ^ (h * internal::hash_multiplier);
74 return h;
75}
