579 size_t nHash = base_class::hashing( val );
580 bucket_type * pBucket;

582 scoped_cell_lock sl( base_class::m_MutexPolicy, nHash );
583 pBucket = base_class::bucket( nHash );

642 template <typename Q, typename Func>
643 std::pair<bool, bool> update( Q const& val, Func func, bool bAllowInsert = true )
644 {
645 std::pair<bool, bool> result;
646 bool bResize = false;
647 size_t nHash = base_class::hashing( val );
648 bucket_type * pBucket;
649 {
650 scoped_cell_lock sl( base_class::m_MutexPolicy, nHash );
651 pBucket = base_class::bucket( nHash );
652
653 result = pBucket->update( val, func, bAllowInsert );
654 if ( result.first && result.second )
655 bResize = base_class::m_ResizingPolicy( ++base_class::m_ItemCounter, *this, *pBucket );
656 }
657
658 if ( bResize )
659 base_class::resize();
660 return result;
661 }
