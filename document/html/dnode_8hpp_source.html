158 inline bool internal_push(void * ptr, CI& comm) {
159 // gets the peers involved in one single communication
160 const int peers=comm.getDescriptor()->getPeers();
161 if (ptr == (void*)FF_EOS) {
162 //std::cerr << "DNODE prepare to send FF_DEOS to " << peers <<" peers\n";
163 for(int i=0;i<peers;++i) {
164 msg_t msg;
165 msg.init(FF_DEOS,msg_t::HEADER_LENGTH);
166 if (!comm.put(msg,i)) return false;
167 //std::cerr << "DNODE sends FF_DEOS to " << i <<"\n";
168 }
169 return true;
170 }
171
172 if (CI::MULTIPUT) {
173 svector<iovec> v;
174 for(int i=0;i<peers;++i) {
175 v.clear();
176 callbackArg.resize(0);
177 prepare(v, ptr, i);
178
179 msg_t hdr(new uint32_t(v.size()), msg_t::HEADER_LENGTH, freeHdr);
180 comm.putmore(hdr,i);
181 callbackArg.resize(v.size());
182 for(size_t j=0;j<v.size()-1;++j) {
183 msg_t msg(v[j].iov_base, v[j].iov_len,freeMsg,callbackArg[j]);
184 comm.putmore(msg,i);
185 }
186 msg_t msg(v[v.size()-1].iov_base, v[v.size()-1].iov_len,freeMsg,callbackArg[v.size()-1]);
187 if (!comm.put(msg,i)) return false;
188 }
189 } else {
190 svector<iovec> v;
191 callbackArg.resize(0);
192 prepare(v, ptr);
193
194 msg_t hdr(new uint32_t(v.size()), msg_t::HEADER_LENGTH, freeHdr);
195 comm.putmore(hdr);
196 callbackArg.resize(v.size());
197 for(size_t j=0;j<v.size()-1;++j) {
198 msg_t msg(v[j].iov_base, v[j].iov_len,freeMsg,callbackArg[j]);
199 comm.putmore(msg);
200 }
201 msg_t msg(v[v.size()-1].iov_base, v[v.size()-1].iov_len,freeMsg,callbackArg[v.size()-1]);
202 if (!comm.put(msg)) return false;
203 }
204 return true;
205 }

229 virtual inline bool pop(void ** ptr) {
230 if (skipdnode || P) return ff_node::pop(ptr);
231
232 // gets the peers involved in one single communication
233 const int sendingPeers=com.getToWait();
234#ifdef _WIN32
235 svector<msg_t*> ** v = (svector<msg_t*> **) malloc(sendingPeers*sizeof(svector<msg_t*> *));
236#else // C99
237 svector<msg_t*>* v[sendingPeers];
238#endif
239 for(int i=0;i<sendingPeers;++i) {
240 msg_t hdr;
241 int sender=-1;
242 if (!com.gethdr(hdr, sender)) {
243 error("dnode:pop: ERROR: receiving header from peer");
244 // free (v); // Win
245 return false;
246 }
247 if (isEos(static_cast<char *>(hdr.getData()))) {
248 //std::cerr << "DNODE gets DEOS\n";
249 if (++neos==com.getDescriptor()->getPeers()) {
250 com.done();
251 *ptr = (void*)FF_EOS;
252 neos=0;
253 // free (v) // Win
254 return true;
255 }
256 if (sendingPeers==1) i=-1; // reset for index
257 continue;
258 }
259 uint32_t len = *static_cast<uint32_t*>(hdr.getData());
260 int ventry = (sendingPeers==1)?0:sender;
261 prepare(v[ventry], len, sender);
262 assert(v[ventry]->size() == len);
263
264 for(size_t j=0;j<len;++j)
265 if (!com.get(*(v[ventry]->operator[](j)),sender)) {
266 error("dnode:pop: ERROR: receiving data from peer");
267 // free (v) // Win32
268 return false;
269 }
270 }
271 com.done();
272
273 unmarshalling(v, sendingPeers, *ptr);
274 // free (v) // Win32
275 return true;
276 }

362 virtual void prepare(svector<msg_t*>*& v, size_t len, const int sender=-1) {
363 svector<msg_t*> * v2 = new svector<msg_t*>(len);

366 msg_t * m = new msg_t;
