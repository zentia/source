241 template <typename T>
242 bool operator==(const intrusive_slist<T>& a, const intrusive_slist<T>& b)
243 {
244 // If we store an mSize member for intrusive_slist, we want to take advantage of it here.
245 typename intrusive_slist<T>::const_iterator ia = a.begin();
246 typename intrusive_slist<T>::const_iterator ib = b.begin();
247 typename intrusive_slist<T>::const_iterator enda = a.end();
248 typename intrusive_slist<T>::const_iterator endb = b.end();
249
250 while((ia != enda) && (ib != endb) && (*ia == *ib))
251 {
252 ++ia;
253 ++ib;
254 }
255 return (ia == enda) && (ib == endb);
256 }
