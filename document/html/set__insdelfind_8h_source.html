78 Set& rSet = m_Set;
79 Set_InsDelFind& fixture = pool().template fixture<Set_InsDelFind>();
80
81 unsigned int i = 0;
82 size_t const nNormalize = size_t(-1) / ( fixture.s_nSetSize * 2);
83
84 size_t nRand = 0;
85 while ( !time_elapsed()) {
86 nRand = cds::bitop::RandXorShift(nRand);
87 size_t n = nRand / nNormalize;
88 switch ( s_arrShuffle[i] ) {
89 case do_find:
90 if ( rSet.contains( n ))
91 ++m_nFindSuccess;
92 else
93 ++m_nFindFailed;
94 break;
95 case do_insert:
96 if ( rSet.insert( n ))
97 ++m_nInsertSuccess;
98 else
99 ++m_nInsertFailed;
100 break;
101 case do_delete:
102 if ( rSet.erase( n ))
103 ++m_nDeleteSuccess;
104 else
105 ++m_nDeleteFailed;
106 break;
107 }
108
109 if ( ++i >= c_nShuffleSize )
110 i = 0;
111 }
112 }

116 template <class Set>
117 void do_test( Set& testSet )
118 {
119 typedef Worker<Set> work_thread;
120
121 // fill map - only odd number
122 {
123 size_t * pInitArr = new size_t[ s_nSetSize ];
124 size_t * pEnd = pInitArr + s_nSetSize;
125 for ( size_t i = 0; i < s_nSetSize; ++i )
126 pInitArr[i] = i * 2 + 1;
127 shuffle( pInitArr, pEnd );
128 for ( size_t * p = pInitArr; p < pEnd; ++p )
129 testSet.insert( typename Set::value_type( *p, *p ));
130 delete [] pInitArr;
131 }
132
133 cds_test::thread_pool& pool = get_pool();
134 pool.add( new work_thread( pool, testSet ), s_nThreadCount );
135
136 propout() << std::make_pair( "thread_count", s_nThreadCount )
137 << std::make_pair( "set_size", s_nSetSize )
138 << std::make_pair( "insert_percentage", s_nInsertPercentage )
139 << std::make_pair( "delete_percentage", s_nDeletePercentage )
140 << std::make_pair( "total_duration", s_nDuration );
141
142 std::chrono::milliseconds duration = pool.run( std::chrono::seconds( s_nDuration ));
143
144 propout() << std::make_pair( "duration", duration );
145
146 size_t nInsertSuccess = 0;
147 size_t nInsertFailed = 0;
148 size_t nDeleteSuccess = 0;
149 size_t nDeleteFailed = 0;
150 size_t nFindSuccess = 0;
151 size_t nFindFailed = 0;
152 for ( size_t i = 0; i < pool.size(); ++i ) {
153 work_thread& thr = static_cast<work_thread&>( pool.get( i ));
154 nInsertSuccess += thr.m_nInsertSuccess;
155 nInsertFailed += thr.m_nInsertFailed;
156 nDeleteSuccess += thr.m_nDeleteSuccess;
157 nDeleteFailed += thr.m_nDeleteFailed;
158 nFindSuccess += thr.m_nFindSuccess;
159 nFindFailed += thr.m_nFindFailed;
160 }
161
162 propout()
163 << std::make_pair( "insert_success", nInsertSuccess )
164 << std::make_pair( "insert_failed", nInsertFailed )
165 << std::make_pair( "delete_success", nDeleteSuccess )
166 << std::make_pair( "delete_failed", nDeleteFailed )
167 << std::make_pair( "find_success", nFindSuccess )
168 << std::make_pair( "find_failed", nFindFailed );
169
170 {
171 ASSERT_TRUE( std::chrono::duration_cast<std::chrono::seconds>(duration).count() > 0 );
172 size_t nTotalOps = nInsertSuccess + nInsertFailed + nDeleteSuccess + nDeleteFailed + nFindSuccess + nFindFailed;
173 propout() << std::make_pair( "avg_speed", nTotalOps / std::chrono::duration_cast<std::chrono::seconds>(duration).count());
174 }
