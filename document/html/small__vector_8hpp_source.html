55class SmallVectorBase {
56protected:

61 : BeginX(FirstEl), EndX(FirstEl), CapacityX((char*)FirstEl+Size) {}

65 void grow_pod(void *FirstEl, size_t MinSizeInBytes, size_t TSize){
66 size_t CurSizeBytes = size_in_bytes();
67 size_t NewCapacityInBytes = 2 * capacity_in_bytes() + TSize; // Always grow.
68 if (NewCapacityInBytes < MinSizeInBytes) {
69 NewCapacityInBytes = MinSizeInBytes;
70 }
71
72 void *NewElts;
73 if (BeginX == FirstEl) {
74 NewElts = std::malloc(NewCapacityInBytes);
75
76 // Copy the elements over. No need to run dtors on PODs.
77 memcpy(NewElts, this->BeginX, CurSizeBytes);
78 } else {
79 // If this wasn't grown from the inline copy, grow the allocated space.
80 NewElts = realloc(this->BeginX, NewCapacityInBytes);
81 }
82 //assert(NewElts && "Out of memory");
83
84 this->EndX = (char*)NewElts+CurSizeBytes;
85 this->BeginX = NewElts;
86 this->CapacityX = (char*)this->BeginX + NewCapacityInBytes;
87 }
88
89public:
91 size_t size_in_bytes() const {
92 return size_t((char*)EndX - (char*)BeginX);
93 }
94
96 size_t capacity_in_bytes() const {
97 return size_t((char*)CapacityX - (char*)BeginX);
98 }
99
100 bool empty() const { return BeginX == EndX; }
101};

295template <typename T, bool isPodLike>
296void SmallVectorTemplateBase<T, isPodLike>::grow(size_t MinSize) {
297 size_t CurCapacity = this->capacity();
298 size_t CurSize = this->size();
299 // Always grow, even from zero.
300 size_t NewCapacity = size_t(tf::detail::NextCapacity(CurCapacity+2));
301 if (NewCapacity < MinSize)
302 NewCapacity = MinSize;
303 T *NewElts = static_cast<T*>(std::malloc(NewCapacity*sizeof(T)));
304
305 // Move the elements over.
306 this->uninitialized_move(this->begin(), this->end(), NewElts);
307
308 // Destroy the original elements.
309 destroy_range(this->begin(), this->end());
310
311 // If this wasn't grown from the inline copy, deallocate the old space.
312 if (!this->isSmall())
313 std::free(this->begin());
314
315 this->setEnd(NewElts+CurSize);
316 this->BeginX = NewElts;
317 this->CapacityX = this->begin()+NewCapacity;
318}

528 iterator insert(iterator I, T &&Elt) {
529 if (I == this->end()) { // Important special case for empty vector.
530 this->push_back(::std::move(Elt));
531 return this->end()-1;
532 }
533
534 //assert(I >= this->begin() && "Insertion iterator is out of bounds.");
535 //assert(I <= this->end() && "Inserting past the end of the vector.");
536
537 if (this->EndX >= this->CapacityX) {
538 size_t EltNo = I-this->begin();
539 this->grow();
540 I = this->begin()+EltNo;
541 }
542
543 ::new ((void*) this->end()) T(::std::move(this->back()));
544 // Push everything else over.
545 std::move_backward(I, this->end()-1, this->end());
546 this->setEnd(this->end()+1);
547
548 // If we just moved the element we're inserting, be sure to update
549 // the reference.
550 T *EltPtr = &Elt;
551 if (I <= EltPtr && EltPtr < this->EndX)
552 ++EltPtr;
553
554 *I = ::std::move(*EltPtr);
555 return I;

587 iterator insert(iterator I, size_type NumToInsert, const T &Elt) {
588 // Convert iterator to elt# to avoid invalidating iterator when we reserve()
589 size_t InsertElt = I - this->begin();
590
591 if (I == this->end()) { // Important special case for empty vector.
592 append(NumToInsert, Elt);
593 return this->begin()+InsertElt;
594 }
595
596 //assert(I >= this->begin() && "Insertion iterator is out of bounds.");
597 //assert(I <= this->end() && "Inserting past the end of the vector.");
598
599 // Ensure there is enough space.
600 reserve(this->size() + NumToInsert);
601
602 // Uninvalidate the iterator.
603 I = this->begin()+InsertElt;
604
605 // If there are more elements between the insertion point and the end of the
606 // range than there are being inserted, we can use a simple approach to
607 // insertion. Since we already reserved space, we know that this won't
608 // reallocate the vector.
609 if (size_t(this->end()-I) >= NumToInsert) {
610 T *OldEnd = this->end();
611 append(std::move_iterator<iterator>(this->end() - NumToInsert),
612 std::move_iterator<iterator>(this->end()));
613
614 // Copy the existing elements that get replaced.
615 std::move_backward(I, OldEnd-NumToInsert, OldEnd);
616
617 std::fill_n(I, NumToInsert, Elt);
618 return I;
619 }
620
621 // Otherwise, we're inserting more elements than exist already, and we're
622 // not inserting at the end.
623
624 // Move over the elements that we're about to overwrite.
625 T *OldEnd = this->end();
626 this->setEnd(this->end() + NumToInsert);
627 size_t NumOverwritten = OldEnd-I;
628 this->uninitialized_move(I, OldEnd, this->end()-NumOverwritten);
629
630 // Replace the overwritten part.
631 std::fill_n(I, NumOverwritten, Elt);
632
633 // Insert the non-overwritten middle part.
634 std::uninitialized_fill_n(OldEnd, NumToInsert-NumOverwritten, Elt);
635 return I;
636 }

638 template<typename ItTy>
639 iterator insert(iterator I, ItTy From, ItTy To) {
640 // Convert iterator to elt# to avoid invalidating iterator when we reserve()
641 size_t InsertElt = I - this->begin();
642
643 if (I == this->end()) { // Important special case for empty vector.
644 append(From, To);
645 return this->begin()+InsertElt;
646 }
647
648 //assert(I >= this->begin() && "Insertion iterator is out of bounds.");
649 //assert(I <= this->end() && "Inserting past the end of the vector.");
650
651 size_t NumToInsert = std::distance(From, To);
652
653 // Ensure there is enough space.
654 reserve(this->size() + NumToInsert);
655
656 // Uninvalidate the iterator.
657 I = this->begin()+InsertElt;
658
659 // If there are more elements between the insertion point and the end of the
660 // range than there are being inserted, we can use a simple approach to
661 // insertion. Since we already reserved space, we know that this won't
662 // reallocate the vector.
663 if (size_t(this->end()-I) >= NumToInsert) {
664 T *OldEnd = this->end();
665 append(std::move_iterator<iterator>(this->end() - NumToInsert),
666 std::move_iterator<iterator>(this->end()));
667
668 // Copy the existing elements that get replaced.
669 std::move_backward(I, OldEnd-NumToInsert, OldEnd);
670
671 std::copy(From, To, I);
672 return I;
673 }
674
675 // Otherwise, we're inserting more elements than exist already, and we're
676 // not inserting at the end.
677
678 // Move over the elements that we're about to overwrite.
679 T *OldEnd = this->end();
680 this->setEnd(this->end() + NumToInsert);
681 size_t NumOverwritten = OldEnd-I;
682 this->uninitialized_move(I, OldEnd, this->end()-NumOverwritten);
683
684 // Replace the overwritten part.
685 for (T *J = I; NumOverwritten > 0; --NumOverwritten) {
686 *J = *From;
687 ++J; ++From;
688 }
689
690 // Insert the non-overwritten middle part.
691 this->uninitialized_copy(From, To, OldEnd);
692 return I;
693 }

739template <typename T>
740void SmallVectorImpl<T>::swap(SmallVectorImpl<T> &RHS) {
741 if (this == &RHS) return;
742
743 // We can only avoid copying elements if neither vector is small.
744 if (!this->isSmall() && !RHS.isSmall()) {
745 std::swap(this->BeginX, RHS.BeginX);
746 std::swap(this->EndX, RHS.EndX);
747 std::swap(this->CapacityX, RHS.CapacityX);
748 return;
749 }
750 if (RHS.size() > this->capacity())
751 this->grow(RHS.size());
752 if (this->size() > RHS.capacity())
753 RHS.grow(this->size());
754
755 // Swap the shared elements.
756 size_t NumShared = this->size();
757 if (NumShared > RHS.size()) NumShared = RHS.size();
758 for (size_type i = 0; i != NumShared; ++i)
759 std::swap((*this)[i], RHS[i]);
760
761 // Copy over the extra elts.
762 if (this->size() > RHS.size()) {
763 size_t EltDiff = this->size() - RHS.size();
764 this->uninitialized_copy(this->begin()+NumShared, this->end(), RHS.end());
765 RHS.setEnd(RHS.end()+EltDiff);
766 this->destroy_range(this->begin()+NumShared, this->end());
767 this->setEnd(this->begin()+NumShared);
768 } else if (RHS.size() > this->size()) {
769 size_t EltDiff = RHS.size() - this->size();
770 this->uninitialized_copy(RHS.begin()+NumShared, RHS.end(), this->end());
771 this->setEnd(this->end() + EltDiff);
772 this->destroy_range(RHS.begin()+NumShared, RHS.end());
773 RHS.setEnd(RHS.begin()+NumShared);
774 }
775}

777template <typename T>
778SmallVectorImpl<T> &SmallVectorImpl<T>::
779 operator=(const SmallVectorImpl<T> &RHS) {
780 // Avoid self-assignment.
781 if (this == &RHS) return *this;
782
783 // If we already have sufficient space, assign the common elements, then
784 // destroy any excess.
785 size_t RHSSize = RHS.size();
786 size_t CurSize = this->size();
787 if (CurSize >= RHSSize) {
788 // Assign common elements.
789 iterator NewEnd;
790 if (RHSSize)

827SmallVectorImpl<T> &SmallVectorImpl<T>::operator=(SmallVectorImpl<T> &&RHS) {
828 // Avoid self-assignment.
829 if (this == &RHS) return *this;
830
831 // If the RHS isn't small, clear this vector and then steal its buffer.
832 if (!RHS.isSmall()) {
833 this->destroy_range(this->begin(), this->end());
834 if (!this->isSmall()) std::free(this->begin());
835 this->BeginX = RHS.BeginX;
836 this->EndX = RHS.EndX;
837 this->CapacityX = RHS.CapacityX;
838 RHS.resetToSmall();
839 return *this;
840 }
841
842 // If we already have sufficient space, assign the common elements, then
843 // destroy any excess.
844 size_t RHSSize = RHS.size();
845 size_t CurSize = this->size();
846 if (CurSize >= RHSSize) {
847 // Assign common elements.
848 iterator NewEnd = this->begin();
849 if (RHSSize)
850 NewEnd = std::move(RHS.begin(), RHS.end(), NewEnd);
851
852 // Destroy excess elements and trim the bounds.
853 this->destroy_range(NewEnd, this->end());
854 this->setEnd(NewEnd);
855
856 // Clear the RHS.
857 RHS.clear();
858
859 return *this;
860 }
861
862 // If we have to grow to have enough elements, destroy the current elements.
863 // This allows us to avoid copying them during the grow.
864 // FIXME: this may not actually make any sense if we can efficiently move
865 // elements.
866 if (this->capacity() < RHSSize) {
867 // Destroy current elements.
868 this->destroy_range(this->begin(), this->end());
869 this->setEnd(this->begin());
870 CurSize = 0;
871 this->grow(RHSSize);
872 } else if (CurSize) {
873 // Otherwise, use assignment for the already-constructed elements.
874 std::move(RHS.begin(), RHS.begin()+CurSize, this->begin());
875 }
876
877 // Move-construct the new elements in place.
878 this->uninitialized_move(RHS.begin()+CurSize, RHS.end(),
879 this->begin()+CurSize);
880
881 // Set end.
882 this->setEnd(this->begin()+RHSSize);
883
884 RHS.clear();
885 return *this;
886}
