42 : base_class( cfg.s_nMapSize, cfg.s_nLoadFactor )
43 {}
44
45 template <typename Iterator>
46 typename std::enable_if< std::is_same< Iterator, typename base_class::iterator>::value, Iterator>::type
47 get_begin()
48 {
49 return base_class::begin();
50 }
51
52 template <typename Iterator>
53 typename std::enable_if< std::is_same< Iterator, typename base_class::iterator>::value, Iterator>::type
54 get_end()
55 {
56 return base_class::end();
57 }
58
59 // for testing
60 static constexpr bool const c_bExtractSupported = true;
61 static constexpr bool const c_bLoadFactorDepended = true;
62 static constexpr bool const c_bEraseExactKey = false;
