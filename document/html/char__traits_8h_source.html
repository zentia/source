307 template <typename T>
308 const T* CharTypeStringRSearch(const T* p1Begin, const T* p1End,
309 const T* p2Begin, const T* p2End)
310 {
311 // Test for zero length strings, in which case we have a match or a failure,
312 // but the return value is the same either way.
313 if((p1Begin == p1End) || (p2Begin == p2End))
314 return p1Begin;
315
316 // Test for a pattern of length 1.
317 if((p2Begin + 1) == p2End)
318 return CharTypeStringFindEnd(p1Begin, p1End, *p2Begin);
319
320 // Test for search string length being longer than string length.
321 if((p2End - p2Begin) > (p1End - p1Begin))
322 return p1End;
323
324 // General case.
325 const T* pSearchEnd = (p1End - (p2End - p2Begin) + 1);
326 const T* pCurrent1;
327 const T* pCurrent2;
328
329 while(pSearchEnd != p1Begin)
330 {
331 // Search for the last occurrence of *p2Begin.
332 pCurrent1 = CharTypeStringFindEnd(p1Begin, pSearchEnd, *p2Begin);
333 if(pCurrent1 == pSearchEnd) // If the first char of p2 wasn't found,
334 return p1End; // then we immediately have failure.
335
336 // In this case, *pTemp == *p2Begin. So compare the rest.
337 pCurrent2 = p2Begin;
338 while(*pCurrent1++ == *pCurrent2++)
339 {
340 if(pCurrent2 == p2End)
341 return (pCurrent1 - (p2End - p2Begin));
342 }
343
344 // A smarter algorithm might know to subtract more than just one,
345 // but in most cases it won't make much difference anyway.
346 --pSearchEnd;
347 }
348
349 return p1End;
350 }
