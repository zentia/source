35 static Result<TupleType> read(const R& _r,
36 const InputVarType& _var) noexcept {
37 if constexpr (schemaful::IsSchemafulReader<R>) {
38 using NamedTupleType = schemaful::tuple_to_named_tuple_t<TupleType>;
39 const auto to_tuple = [](auto&& _named_tuple) {
40 return [&]<int... _is>(std::integer_sequence<int, _is...>) {
41 return TupleType(std::move(rfl::get<_is>(_named_tuple))...);
42 }(std::make_integer_sequence<int, NamedTupleType::size()>());
43 };
44 return Parser<R, W, NamedTupleType, ProcessorsType>::read(_r, _var)
45 .transform(to_tuple);
46
47 } else {
48 const auto parse = [&](const InputArrayType& _arr) -> Result<TupleType> {
49 alignas(TupleType) unsigned char buf[sizeof(TupleType)]{};
50 auto ptr = internal::ptr_cast<TupleType*>(&buf);
51 const auto tuple_reader =
52 TupleReader<R, W, TupleType, _ignore_empty_containers,
53 _all_required, ProcessorsType>(&_r, ptr);
54 auto err = _r.read_array(tuple_reader, _arr);
55 if (err) {
56 call_destructors_on_tuple_where_necessary(tuple_reader.num_set(),
57 ptr);
58 return error(*err);
59 }
60 err = tuple_reader.handle_missing_fields();
61 if (err) {
62 call_destructors_on_tuple_where_necessary(tuple_reader.num_set(),
63 ptr);
64 return error(*err);
65 }
66 auto res = Result<TupleType>(std::move(*ptr));
67 call_destructors_on_tuple_where_necessary(tuple_reader.num_set(), ptr);
68 return res;
69 };
70
71 return _r.to_array(_var).and_then(parse);
72 }
