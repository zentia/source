102 // Check factor to avoid asking for number of workers while there might be no arena.
103 size_t new_size = factor ? factor * g_threadNums.local() : 0;
104 if (new_size != my_size) {
105 if (my_array) {
106 NFS_Free(my_array);
107 // Following two assignments must be done here for sake of exception safety.
108 my_array = NULL;
109 my_size = 0;
110 }
111 if (new_size) {
112 my_array = static_cast<affinity_id*>(NFS_Allocate(new_size, sizeof(affinity_id), NULL ));
113 memset(my_array, 0, sizeof(affinity_id) * new_size);
114 my_size = new_size;
115 }

125template<typename Range, typename Body, typename Partitioner>
126class start_for : public fake_task {
127 Range my_range;
128 Body my_body;
129 typename Partitioner::task_partition_type my_partition;
130 size_t m_executedBegin, m_executedEnd;
131 bool m_firstTimeRun;
132 size_t m_joinedBegin, m_joinedEnd;
133 test_partitioner_utils::BinaryTree* m_tree;
134public:
135 start_for( const Range& range, const Body& body, Partitioner& partitioner,
136 test_partitioner_utils::BinaryTree* tree ) :
137 my_range(range), my_body(body), my_partition(partitioner),
138 m_executedBegin(0), m_executedEnd(0), m_firstTimeRun(true),
139 m_joinedBegin(/* grows left */ range.end()), m_joinedEnd(range.end()), m_tree(tree)
140 {
141 if (m_tree) {
142 m_tree->push_node( test_partitioner_utils::make_node(my_range.begin(), my_range.end(), affinity()) );
143 }
144 }
145
146
147 start_for( start_for& parent_, typename Partitioner::split_type& split_obj) :
148 my_range(parent_.my_range, split_obj),
149 my_body(parent_.my_body),
150 my_partition(parent_.my_partition, split_obj),
151 m_executedBegin(0), m_executedEnd(0), m_firstTimeRun(true),
152 m_joinedBegin(/* grows left */ my_range.end()), m_joinedEnd(my_range.end()),
153 m_tree(parent_.m_tree)
154 {
155 set_parent(parent_.parent());
156 my_partition.set_affinity(*this);
157
158 if (m_tree) {
159 // collecting splitting statistics
160 m_tree->push_node( test_partitioner_utils::make_node(my_range.begin(),
161 my_range.end(),
162 affinity()) );
163 m_tree->push_node( test_partitioner_utils::make_node(parent_.my_range.begin(),
164 parent_.my_range.end(),
165 parent_.affinity()) );
166 }
167 }

169
170 start_for( start_for& parent_, const Range& r, depth_t d ) :
171 my_range(r),
172 my_body(parent_.my_body),
173 my_partition(parent_.my_partition, tbb::split()),
174 m_executedBegin(0), m_executedEnd(0), m_firstTimeRun(true),
175 m_joinedBegin(/* grows left */ r.end()), m_joinedEnd(r.end()),
176 m_tree(parent_.m_tree)
177 {
178 set_parent(parent_.parent());
179 my_partition.set_affinity(*this);
180 my_partition.align_depth( d );
181 }
182 fake_task* execute() __TBB_override {
183 my_partition.check_being_stolen( *this );
184 size_t origBegin = my_range.begin();
185 size_t origEnd = my_range.end();
186
187 my_partition.execute(*this, my_range);
188
189 ASSERT(m_executedEnd == m_joinedBegin, "Non-continuous execution");
190 m_executedEnd = m_joinedEnd;
191
192 ASSERT(origBegin == m_executedBegin && origEnd == m_executedEnd,
193 "Not all iterations were processed");
194 return NULL;
195 }
196
197 void run_body( Range &r ) {
198 if( r.is_ensure_non_emptiness() )
199 ASSERT( !r.empty(), "Empty ranges are not allowed" );
200 my_body(r);
201 if (m_firstTimeRun) {
202 m_firstTimeRun = false;
203 m_executedBegin = m_executedEnd = r.begin();
204 }
205 ASSERT(m_executedBegin <= r.begin() && m_executedEnd <= r.end(),
206 "Non-continuous execution");
207 m_executedEnd = r.end();
208 }
209
210 void offer_work(typename Partitioner::split_type& split_obj) {
211 start_for sibling(*this, split_obj);
212 sibling.execute();
213 join(sibling.m_executedBegin, sibling.m_executedEnd);
214 }
215
216 void offer_work(const Range& r, depth_t d = 0) {
217 start_for sibling(*this, r, d);
218 sibling.execute();
219 join(sibling.m_executedBegin, sibling.m_executedEnd);
220 }
221 void join(size_t siblingExecutedBegin, size_t siblingExecutedEnd) {
222 ASSERT(siblingExecutedEnd == m_joinedBegin, "?");
223 m_joinedBegin = siblingExecutedBegin;
224 }

289 struct use_case_settings_t {
290 size_t thread_num; // number of threads used during current use case
291 unsigned factors_array_len; // size of 'factors' array
292 size_t range_begin; // beginning of range iterations
293 bool provide_feedback; // 'true' if range should give feedback
294 bool ensure_non_empty_size; // don't allow empty size ranges
295
296 size_t above_threads_size_tolerance; // allowed value for number of created ranges
297 // when initial size of the range was greater or
298 // equal to number of threads
299
300 size_t below_threads_size_tolerance; // allowed value for number of created ranges
301 // when initial size of the range was less than
302 // number of threads
303
304 size_t between_min_max_ranges_tolerance; // allowed value for difference of iterations
305 // between bigger and lesser ranges
306
307 CheckerFuncType checker; // checker function for a particular test case
308 };

315 static void uniform_distribution_checker(const char* rangeName, size_t rangeSize, const use_case_settings_t* settings,
316 const RangeStatisticData& stat)
317 {
318 // Checking that all threads were given a task
319 if (rangeSize >= settings->thread_num) {
320 uint64_t disparity =
321 max(stat.m_rangeNum, settings->thread_num) - min(stat.m_rangeNum, settings->thread_num);
322 if (disparity > settings->above_threads_size_tolerance) {
323 REPORT("ERROR: '%s (f=%d|e=%d)': |#ranges(%llu)-#threads(%llu)|=%llu > %llu=tolerance\n",
324 rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size), stat.m_rangeNum,
325 settings->thread_num, disparity, uint64_t(settings->above_threads_size_tolerance));
326 ASSERT(disparity <= settings->above_threads_size_tolerance, "Incorrect number of range "
327 "objects was created before work balancing phase started");
328 }
329 } else if (settings->ensure_non_empty_size && rangeSize != 0) {
330 uint64_t disparity = max(stat.m_rangeNum, rangeSize) - min(stat.m_rangeNum, rangeSize);
331 if (disparity > settings->below_threads_size_tolerance ) {
332 REPORT("ERROR: '%s (f=%d|e=%d)': |#ranges-range size|=%llu > %llu=tolerance\n",
333 rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
334 disparity, uint64_t(settings->below_threads_size_tolerance));
335 ASSERT(disparity <= settings->below_threads_size_tolerance, "Incorrect number of range objects"
336 " was created before work balancing phase started");
337 }
338 }
339 // Checking difference between min and max number of range iterations
340 size_t diff = stat.m_maxRangeSize - stat.m_minRangeSize;
341 if (diff > settings->between_min_max_ranges_tolerance) {
342 REPORT("ERROR: '%s (f=%d|e=%d)': range size difference=%llu > %llu=tolerance\n",
343 rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
344 uint64_t(diff), uint64_t(settings->between_min_max_ranges_tolerance));
345 ASSERT(diff <= settings->between_min_max_ranges_tolerance, "Uniform iteration distribution error");
346 }
347 }
348 // Checker for test cases where ranges don't provide feedback during proportional split to
349 // partitioner and differ from tbb::blocked_range implementation in their splitting algorithm
350 static void nonuniform_distribution_checker(const char* rangeName, size_t rangeSize, const use_case_settings_t* settings,
351 const RangeStatisticData& stat)
352 {
353 if (stat.m_rangeNum > settings->thread_num) {
354 REPORT("ERROR: '%s (f=%d|e=%d)': %llu=#ranges > #threads=%llu\n",
355 rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
356 uint64_t(stat.m_rangeNum), uint64_t(settings->thread_num));
357 ASSERT(stat.m_rangeNum <= settings->thread_num,
358 "Incorrect number of range objects was created before work balancing phase started");
359 }
360 // Checking difference between min and max number of range iterations
361 size_t diff = stat.m_maxRangeSize - stat.m_minRangeSize;
362 if (diff > rangeSize) {
363 REPORT("ERROR: '%s (f=%d|e=%d)': range size difference=%llu > %llu=initial range size\n",
364 rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
365 uint64_t(diff), uint64_t(rangeSize));
366 ASSERT(diff <= rangeSize, "Iteration distribution error");
367 }
368 }
369
370protected:
371 size_t m_parallel_group_thread_starting_index; // starting index of thread
372
373 template <typename Range, typename Partitioner, typename T>
374 void test(use_case_settings_t& settings, T factors[], size_t (*rsgFunc)(T*, unsigned, size_t)
375 = &default_range_size_generator<T>) const
376 {
377 for (unsigned i = 0; i < settings.factors_array_len; ++i) {
378 size_t range_end = rsgFunc(factors, i, settings.thread_num);
379 RangeStatisticData stat = { /*range num=*/ 0, /*minimal size of range=*/ 0,
380 /*maximal size of range=*/ 0, /*minimal size of range was not rewritten yet=*/ false };
381 Range range = Range(settings.range_begin, range_end, &stat, settings.provide_feedback,
382 settings.ensure_non_empty_size);
383 Partitioner my_partitioner;
384 test_case(range, SimpleBody(), my_partitioner, NULL);
385 size_t range_size = range_end - settings.range_begin;
386 const char* rangeName = typeid(range).name();
387 settings.checker(rangeName, range_size, &settings, stat);
388 }
389 }

393void test() {
394 size_t hw_threads_num = tbb::tbb_thread::hardware_concurrency();
395 size_t threadsToRunOn = std::min<size_t>(max_simulated_threads, hw_threads_num);
396
397 size_t parallel_group_thread_starting_index = 1;
398 while( parallel_group_thread_starting_index <= max_simulated_threads - threadsToRunOn ) {
399 NativeParallelFor(threadsToRunOn, ParallelTestBody(parallel_group_thread_starting_index));
400 parallel_group_thread_starting_index += threadsToRunOn;
401 }
402 NativeParallelFor(max_simulated_threads - parallel_group_thread_starting_index,
403 ParallelTestBody(parallel_group_thread_starting_index));
404}

420template<>
421void check_tree<tbb::static_partitioner>(const test_partitioner_utils::BinaryTree& tree) {
422 std::vector<test_partitioner_utils::TreeNode* > tree_leafs;
423 tree.fill_leafs(tree_leafs);
424 typedef std::vector<size_t> Slots;
425 Slots affinity_slots(tree_leafs.size() + 1, 0);
426
427 for (std::vector<test_partitioner_utils::TreeNode*>::iterator i = tree_leafs.begin(); i != tree_leafs.end(); ++i) {
428 affinity_slots[(*i)->m_affinity]++;
429 if ((*i)->m_affinity == 0)
430 ASSERT((*i)->m_range_begin == task_affinity_whitebox::range_begin,
431 "Task with affinity 0 was executed with wrong range");
432 }
433
434 typedef std::iterator_traits<Slots::iterator>::difference_type slots_difference_type;
435 ASSERT(std::count(affinity_slots.begin(), affinity_slots.end(), size_t(0)) == slots_difference_type(1),
436 "static_partitioner incorrectly distributed tasks by threads");
437 ASSERT(std::count(affinity_slots.begin(), affinity_slots.end(), size_t(1)) == slots_difference_type(g_threadNums.local()),
438 "static_partitioner incorrectly distributed tasks by threads");
439 ASSERT(affinity_slots[tbb::this_task_arena::current_thread_index() + 1] == 0,
440 "static_partitioner incorrectly assigns task with 0 affinity");
441 ASSERT(std::accumulate(affinity_slots.begin(), affinity_slots.end(), size_t(0)) == g_threadNums.local(),
442 "static_partitioner has created more tasks than the number of threads");
443}
