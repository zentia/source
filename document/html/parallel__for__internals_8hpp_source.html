85 /* -------------------- Parallel For/Reduce Macros -------------------- */
86 /* Usage example:
87 * // loop parallelization using 3 workers
88 * // and a minimum task grain of 2
89 * wthread = 3;
90 * grain = 2;
91 * for(int i=0;i<N;++i) FF_PARFOR_BEGIN(for,i,0,N,1,grain,wthread) {
92 * A[i]=f(i) ----> A[i]=f(i);
93 * } FF_PARFOR_END(for);
94 *
95 * parallel for + reduction:
96 *
97 * s=4;
98 * for(int i=0;i<N;++i) FF_PARFORREDUCE_BEGIN(for,s,0,i,0,N,1,grain,wthread) {
99 * s*=f(i) ----> s*=f(i);
100 * } FF_PARFORREDUCE_END(for,s,*);
101 *
102 *
103 * FF_PARFOR_INIT(pf,maxwthread);
104 * ....
105 * while(k<nTime) { while(k<nTime) {
106 * for(int i=0;i<N;++i) FF_PARFORREDUCE_START(pf,s,0,i,0,N,1,grain,wthread) {
107 * s*=f(i,k); ----> s*=f(i,k);
108 * } } FF_PARFORREDUCE_STOP(pf,s,*);
109 * }
110 * ....
111 *
112 * FF_PARFOR_DONE(pf);
113 *
114 *
115 * NOTE: inside the body of the PARFOR/PARFORREDUCE, it is possible to use the
116 * '_ff_thread_id' const integer variable to identify the thread id
117 * running the sequential portion of the loop.
118 */

415// parallel for task, it represents a range (start,end( of indexes
416struct forall_task_t {
417 forall_task_t() : end(0) {
418 start.store(0); // MA: consistency of store to be checked
419 }
420 forall_task_t(const forall_task_t &t):end(t.end) {
421 start.store(t.start.load(std::memory_order_relaxed)); // MA: consistency of store to be checked
422 }
423 forall_task_t & operator=(const forall_task_t &t) {
424 start=t.start.load(std::memory_order_relaxed), end=t.end;
425 return *this;
426 }
427 void set(long s, long e) { start=s,end=e; }
428
429 std::atomic_long start;
430 long end;
431};

468 // initialize the data vector
469 virtual inline size_t init_data(ssize_t start, ssize_t stop) {
470 static_scheduling = false; // enable work stealing in the nextTaskConcurrent
471 const long numtasks = std::lrint(std::ceil((stop-start)/(double)_step));
472 long totalnumtasks = std::lrint(std::ceil(numtasks/(double)_chunk));
473 long tt = totalnumtasks;
474 size_t ntxw = totalnumtasks / _nw;
475 size_t r = totalnumtasks % _nw;
476
477 // try to keep the n. of tasks per worker as smaller as possible
478 if (ntxw == 0 && r>=1) { ntxw = 1, r = 0; }
479
480 data.resize(_nw); eossent.resize(_nw);
481 taskv.resize(8*_nw); // 8 is the maximum n. of jumps, see the heuristic below
482 skip1=false,jump=0,maxid=-1;
483
484 ssize_t end, t=0, e;
485 for(size_t i=0;i<_nw && totalnumtasks>0;++i, totalnumtasks-=t) {
486 t = ntxw + ( (r>1 && (i<r)) ? 1 : 0 );
487 e = start + (t*_chunk - 1)*_step + 1;
488 end = (e<stop) ? e : stop;
489 data[i].ntask=t;
490 data[i].task.set(start,end);
491 start = (end-1)+_step;
492 }
493
494 if (totalnumtasks) {
495 assert(totalnumtasks==1);
496 // try to keep the n. of tasks per worker as smaller as possible
497 if (ntxw > 1) data[_nw-1].ntask += totalnumtasks;
498 else { --tt, _chunk*=2; }
499 data[_nw-1].task.end = stop;
500 }
501 // printf("init_data\n");
502 // for(size_t i=0;i<_nw;++i) {
503 // printf("W=%ld %ld <%ld,%ld>\n", i, data[i].ntask.load(), data[i].task.start.load(), data[i].task.end);
504 // }
505 // printf("totaltasks=%ld\n", tt);
506 return tt;
507 }
508 // initialize the data vector
509 virtual inline size_t init_data_static(long start, long stop) {
510 assert(_chunk <= 0);
511 static_scheduling = true; // this forces static scheduling in the nextTaskConcurrent
512 skip1=false,jump=0,maxid=-1;
513
514 if (_chunk == 0) {
515 // default static scheduling, i.e. the iteration space is almost equally divided
516 // in contiguous chunks among threads
517 const long numtasks = std::lrint(std::ceil((stop-start)/(double)_step));
518 long totalnumtasks = (long)_nw;
519 size_t r = numtasks % _nw;
520 _chunk = numtasks / long(_nw);
521
522 data.resize(_nw); taskv.resize(_nw);eossent.resize(_nw);
523
524 long end, e;
525 for(size_t i=0; totalnumtasks>0; ++i,--totalnumtasks) {
526 e = start + (_chunk - 1)*_step + 1 + ((i<r) ? _step : 0 );
527 end = (e<stop) ? e : stop;
528 data[i].ntask=1;
529 data[i].task.set(start,end);
530 start = (end-1)+_step;
531 }
532 if (r) ++_chunk;
533 return _nw;
534 }
535 // fill out the table with only the first task just to start the worker threads
536 long chunk = -_chunk;
537 _chunk = stop; // needed because sendTask has to send the range (begin, stop(
538 const long numtasks = std::lrint(std::ceil((stop-start)/(double)_step));
539 const long totalnumtasks = std::lrint(std::ceil(numtasks/(double)chunk));
540 const size_t ntxw = (std::min)(_nw, (size_t)totalnumtasks);
541
542 for(size_t i=0;i<ntxw;++i) {
543 data[i].ntask = 1;
544 data[i].task.set(start+long(i)*chunk,stop);
545 }

551 // }
552 // printf("total task=%ld\n", ntxw);
553
554 return ntxw;
555 }

584 inline bool sendTask(const bool skipmore=false) {
585#ifdef FF_PARFOR_PASSIVE_NOSTEALING
586 if(canUseNoStealing()){
587 // Just start the workers and die.
588 for(size_t wid=0;wid<_nw;++wid) {
589 lb->ff_send_out_to((void*) &dummyTask, (int) wid);
590 }
591 return true;
592 }
593#endif
594 size_t remaining = totaltasks;
595 const long endchunk = (_chunk-1)*_step + 1;
596
597 more:
598 for(size_t wid=0;wid<_nw;++wid) {
599 if (data[wid].ntask >0) {
600 long start = data[wid].task.start;
601 long end = (std::min)(start+endchunk, data[wid].task.end);
602 taskv[wid+jump].set(start, end);
603 lb->ff_send_out_to(&taskv[wid+jump], (int) wid);
604 --remaining, --data[wid].ntask;
605 (data[wid].task).start = (end-1)+_step;
606 eossent[wid]=false;
607 } else skip1=true; //skip2=skip3=true;
608 }
609 // January 2014 (massimo): this heuristic maight not be the best option in presence
610 // of very high load imbalance between iterations.
611 // Update: removed skip2 and skip3 so that it is less aggressive !
612
613 jump+=long(_nw);
614 assert((jump / _nw) <= 8);
615 // heuristic: try to assign more task at the very beginning
616 if (!skipmore && !skip1 && totaltasks>=4*_nw) { skip1=true; goto more;}
617 //if (!skip2 && totaltasks>=64*_nw) { skip1=false; skip2=true; goto moretask;}
618 //if (!skip3 && totaltasks>=1024*_nw){ skip1=false; skip2=false; skip3=true; goto moretask;}
619 return (remaining>0);
620 }

623 for(size_t id=0;id<_nw;++id) {
624 taskv[id].set(0,0);
625 lb->ff_send_out_to(&taskv[id], int(id));

628 inline bool nextTaskConcurrentNoStealing(forall_task_t *task, const int wid) {
629#ifdef FF_PARFOR_PASSIVE_NOSTEALING
630 long r = _nextIteration.fetch_add(_step);
631 if(r >= _stop){return false;}
632 task->set(r, r + _step);
633 return true;
634#else
635 FF_IGNORE_UNUSED(task);
636 FF_IGNORE_UNUSED(wid);
637 error("To use nextTaskConcurrentNoStealing you need to define macro FF_PARFOR_PASSIVE_NOSTEALING\n");
638 return false;

642 // this method is accessed concurrently by all worker threads
643 inline bool nextTaskConcurrent(forall_task_t *task, const int wid) {
644#ifdef FF_PARFOR_PASSIVE_NOSTEALING
645 if(canUseNoStealing()){
646 return nextTaskConcurrentNoStealing(task, wid);
647 }
648#endif
649 const long endchunk = (_chunk-1)*_step + 1; // next end-point
650 auto id = wid;
651 L1:
652 if (data[id].ntask.load(std::memory_order_acquire)>0) {
653 auto oldstart = data[id].task.start.load(std::memory_order_relaxed);
654 auto end = (std::min)(oldstart+endchunk, data[id].task.end);
655 auto newstart = (end-1)+_step;
656
657 if (!data[id].task.start.compare_exchange_weak(oldstart, newstart,
658 std::memory_order_release,
659 std::memory_order_relaxed)) {
660 workerlosetime_in(_nw <= lb->getnworkers());
661 goto L1; // restart the sequence from the beginning
662 }
663
664 // after fetch_sub ntask may be less than 0
665 data[id].ntask.fetch_sub(1,std::memory_order_release);
666 if (oldstart<end) { // it might be possible that oldstart == end
667 task->set(oldstart, end);
668 return true;
669 }
670 }
671
672 // no available task for the current thread
673 if (static_scheduling) return false; // <------------------------------------
674
675#if !defined(PARFOR_MULTIPLE_TASKS_STEALING)
676 // the following scheduling policy for the tasks focuses mostly to load-balancing
677 long _maxid = 0, ntask = 0;
678 if (maxid.load(std::memory_order_acquire)<0)
679 _maxid = (long) (std::max_element(data.begin(),data.end(),data_cmp) - data.begin());
680 else _maxid = maxid;
681 ntask = data[_maxid].ntask.load(std::memory_order_relaxed);
682 if (ntask>0) {
683 if (_maxid != maxid) maxid.store(_maxid, std::memory_order_release);
684 id = _maxid;
685 goto L1;
686 }
687 // no more tasks, exit
688
689#else
690 // the following scheduling policy for the tasks is a little bit more
691 // complex and costly. It tries to find a trade-off between
692 // task-to-thread localy and load-balancing by moving a bunch of tasks
693 // from one thread to another one
694 long _maxid = 0, ntask = 0;
695 if (maxid.load(std::memory_order_acquire)<0)
696 _maxid = (std::max_element(data.begin(),data.end(),data_cmp) - data.begin());
697 else _maxid = maxid;
698 L2:
699 ntask = data[_maxid].ntask.load(std::memory_order_relaxed);
700 if (ntask>0) {
701 if (_maxid != maxid) maxid.store(_maxid, std::memory_order_release);
702 if (ntask<=3) { id = _maxid; goto L1; }
703
704 // try to steal half of the tasks remaining to _maxid
705
706 auto oldstart = data[_maxid].task.start.load(std::memory_order_relaxed);
707 auto q = ((data[_maxid].task.end-oldstart)/_chunk) >> 1;
708 if (q<=3) { id = _maxid; goto L1; }
709 auto newstart = oldstart + (q*_chunk-1)*_step +1;
710 if (!data[_maxid].task.start.compare_exchange_weak(oldstart, newstart,
711 std::memory_order_release,
712 std::memory_order_relaxed)) {
713 workerlosetime_in(_nw <= lb->getnworkers());
714 goto L2; // restart the sequence from the beginning
715 }
716 assert(newstart <= data[_maxid].task.end);
717
718 data[_maxid].ntask.fetch_sub(q, std::memory_order_release);
719 data[wid].task.start.store(oldstart, std::memory_order_relaxed);
720 data[wid].task.end = newstart;
721 data[wid].ntask.store(q, std::memory_order_release);
722 id = wid;
723 goto L1;
724 }
725#endif
726 return false;
727 }

729 inline bool nextTask(forall_task_t *task, const int wid) {
730#ifdef FF_PARFOR_PASSIVE_NOSTEALING
731 if(canUseNoStealing()){
732 return nextTaskConcurrentNoStealing(task, wid);
733 }
734#endif
735 const long endchunk = (_chunk-1)*_step + 1;
736 int id = wid;
737 if (data[id].ntask) {
738 L1:
739 long start = data[id].task.start;
740 long end = (std::min)(start+endchunk, data[id].task.end);
741 --data[id].ntask, (data[id].task).start = (end-1)+_step;
742 task->set(start, end);
743 return true;
744 }
745 // no available task for the current thread
746#if !defined(PARFOR_MULTIPLE_TASKS_STEALING)
747 // the following scheduling policy for the tasks focuses mostly to load-balancing
748 if (maxid<0) { //check if maxid has been set
749 L2:
750 maxid = (long) (std::max_element(data.begin(),data.end(),data_cmp) - data.begin());
751 if (data[maxid].ntask > 0) {
752 id=maxid;
753 goto L1;
754 }
755 // no more tasks, exit
756 } else {
757 if (data[maxid].ntask > 0) {
758 id=maxid;
759 goto L1;
760 }
761 goto L2;
762 }
763#else
764 auto flag=false;
765 if (maxid<0) {
766 L2:
767 maxid = (std::max_element(data.begin(),data.end(),data_cmp) - data.begin());
768 flag=true;
769 }
770 id = maxid;
771 if (data[id].ntask>0) {
772 if (data[id].ntask<=3) goto L1;
773
774 // steal half of the tasks
775 auto q = data[id].ntask >> 1, r = data[id].ntask & 0x1;
776 data[id].ntask = q;
777 data[wid].ntask = q+r;
778 data[wid].task.end = data[id].task.end;
779 data[id].task.end = data[id].task.start + (q*_chunk-1)*_step +1;
780 data[wid].task.start = data[id].task.end;
781 id = wid;
782 goto L1;
783 } else if (!flag) goto L2;
784#endif
785 return false;
786 }
