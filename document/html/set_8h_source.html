363 template <typename Key, typename Compare, typename Allocator>
364 inline eastl::pair<typename set<Key, Compare, Allocator>::iterator,
365 typename set<Key, Compare, Allocator>::iterator>
366 set<Key, Compare, Allocator>::equal_range(const Key& k)
367 {
368 // The resulting range will either be empty or have one element,
369 // so instead of doing two tree searches (one for lower_bound and
370 // one for upper_bound), we do just lower_bound and see if the
371 // result is a range of size zero or one.
372 const iterator itLower(lower_bound(k));
373
374 if((itLower == end()) || compare(k, *itLower)) // If at the end or if (k is < itLower)...
375 return eastl::pair<iterator, iterator>(itLower, itLower);
376
377 iterator itUpper(itLower);
378 return eastl::pair<iterator, iterator>(itLower, ++itUpper);
379 }

401 // https://en.cppreference.com/w/cpp/container/set/erase_if
403 template <class Key, class Compare, class Allocator, class Predicate>
404 typename set<Key, Compare, Allocator>::size_type erase_if(set<Key, Compare, Allocator>& c, Predicate predicate)
405 {
406 auto oldSize = c.size();
407 for (auto i = c.begin(), last = c.end(); i != last;)
408 {
409 if (predicate(*i))
410 {
411 i = c.erase(i);
412 }
413 else
414 {
415 ++i;
416 }
417 }
418 return oldSize - c.size();
419 }

554 template <typename Key, typename Compare, typename Allocator>
555 inline eastl::pair<typename multiset<Key, Compare, Allocator>::iterator,
556 typename multiset<Key, Compare, Allocator>::iterator>
557 multiset<Key, Compare, Allocator>::equal_range(const Key& k)
558 {
559 // There are multiple ways to implement equal_range. The implementation mentioned
560 // in the C++ standard and which is used by most (all?) commercial STL implementations
561 // is this:
562 // return eastl::pair<iterator, iterator>(lower_bound(k), upper_bound(k));
563 //
564 // This does two tree searches -- one for the lower bound and one for the
565 // upper bound. This works well for the case whereby you have a large container
566 // and there are lots of duplicated values. We provide an alternative version
567 // of equal_range called equal_range_small for cases where the user is confident
568 // that the number of duplicated items is only a few.
569
570 return eastl::pair<iterator, iterator>(lower_bound(k), upper_bound(k));
571 }

584 template <typename Key, typename Compare, typename Allocator>
585 inline eastl::pair<typename multiset<Key, Compare, Allocator>::iterator,
586 typename multiset<Key, Compare, Allocator>::iterator>
587 multiset<Key, Compare, Allocator>::equal_range_small(const Key& k)
588 {
589 // We provide alternative version of equal_range here which works faster
590 // for the case where there are at most small number of potential duplicated keys.
591 const iterator itLower(lower_bound(k));
592 iterator itUpper(itLower);
593
594 while((itUpper != end()) && !compare(k, *itUpper))
595 ++itUpper;
596
597 return eastl::pair<iterator, iterator>(itLower, itUpper);
598 }

623 template <class Key, class Compare, class Allocator, class Predicate>
624 typename multiset<Key, Compare, Allocator>::size_type erase_if(multiset<Key, Compare, Allocator>& c, Predicate predicate)
625 {
626 auto oldSize = c.size();
627 // Erases all elements that satisfy the predicate pred from the container.
628 for (auto i = c.begin(), last = c.end(); i != last;)
629 {
630 if (predicate(*i))
631 {
632 i = c.erase(i);
633 }
634 else
635 {
636 ++i;
637 }
638 }
639 return oldSize - c.size();
