148void print_call_stack() {
149 Harness::internal::print_stack_mutex.lock();
150 fflush(stdout); fflush(stderr);
151 #if BACKTRACE_FUNCTION_AVAILABLE
152 const int sz = 100; // max number of frames to capture
153 void *buff[sz];
154 int n = backtrace(buff, sz);
155 REPORT("Call stack info (%d):\n", n);
156 backtrace_symbols_fd(buff, n, fileno(stdout));
157 #elif __SUNPRO_CC
158 REPORT("Call stack info:\n");
159 printstack(fileno(stdout));
160 #elif _WIN32_WINNT > 0x0501 && _MSC_VER>=1500 && !__TBB_WIN8UI_SUPPORT
161 const int sz = 62; // XP limitation for number of frames
162 void *buff[sz];
163 int n = CaptureStackBackTrace(0, sz, buff, NULL);
164 REPORT("Call stack info (%d):\n", n);
165 static LONG once = 0;
166 if( !InterlockedExchange(&once, 1) )
167 SymInitialize(GetCurrentProcess(), NULL, TRUE);
168 const int len = 255; // just some reasonable string buffer size
169 union { SYMBOL_INFO sym; char pad[sizeof(SYMBOL_INFO)+len]; };
170 sym.MaxNameLen = len;
171 sym.SizeOfStruct = sizeof( SYMBOL_INFO );
172 DWORD64 offset;
173 for(int i = 1; i < n; i++) { // skip current frame
174 if(!SymFromAddr( GetCurrentProcess(), DWORD64(buff[i]), &offset, &sym )) {
175 sym.Address = ULONG64(buff[i]); offset = 0; sym.Name[0] = 0;
176 }
177 REPORT("[%d] %016I64X+%04I64X: %s\n", i, sym.Address, offset, sym.Name); //TODO: print module name
178 }
179 #endif /*BACKTRACE_FUNCTION_AVAILABLE*/
180 Harness::internal::print_stack_mutex.unlock();
181}

311inline void ParseCommandLine( int argc, char* argv[] ) {
312 if( !argc ) REPORT("Command line with 0 arguments\n");
313 int i = 1;
314 if( i<argc ) {
315 if( strncmp( argv[i], "-v", 2 )==0 ) {
316 Verbose = true;
317 ++i;
318 }
319 }
320 if( i<argc ) {
321 char* endptr;
322 MinThread = strtol( argv[i], &endptr, 0 );
323 if( *endptr==':' )
324 MaxThread = strtol( endptr+1, &endptr, 0 );
325 else if( *endptr=='\0' )
326 MaxThread = MinThread;
327 if( *endptr!='\0' ) {
328 REPORT_FATAL_ERROR("garbled nthread range\n");
329 exit(1);
330 }
331 if( MinThread<0 ) {
332 REPORT_FATAL_ERROR("nthread must be nonnegative\n");
333 exit(1);
334 }
335 if( MaxThread<MinThread ) {
336 REPORT_FATAL_ERROR("nthread range is backwards\n");
337 exit(1);
338 }
339 ++i;
340 }
341#if __TBB_STDARGS_BROKEN
342 if ( !argc )
343 argc = 1;
344 else {
345 while ( i < argc && argv[i][0] == 0 )
346 ++i;
347 }
348#endif /* __TBB_STDARGS_BROKEN */
349 if( i!=argc ) {
350 REPORT_FATAL_ERROR("Usage: %s [-v] [nthread|minthread:maxthread]\n", argv[0] );
351 exit(1);
352 }
353}
354#endif /* HARNESS_NO_PARSE_COMMAND_LINE */

529template<typename Index, typename Body>
530class NativeParallelForTask: NoCopy {
531public:
532 NativeParallelForTask( Index index_, const Body& body_ ) :
533 index(index_),
534 body(body_)
535 {}
536
538 void start() {
539#if _WIN32||_WIN64
540 unsigned thread_id;
541#if __TBB_WIN8UI_SUPPORT
542 std::thread* thread_tmp=new std::thread(thread_function, this);
543 thread_handle = thread_tmp->native_handle();
544 thread_id = 0;
545#else
546 unsigned stack_size = 0;
547#if HARNESS_THREAD_STACK_SIZE
548 stack_size = HARNESS_THREAD_STACK_SIZE;
549#endif
550 thread_handle = (HANDLE)_beginthreadex( NULL, stack_size, thread_function, this, 0, &thread_id );
551#endif
552 ASSERT( thread_handle!=0, "NativeParallelFor: _beginthreadex failed" );
553#else
554#if __ICC==1100
555 #pragma warning (push)
556 #pragma warning (disable: 2193)
557#endif /* __ICC==1100 */
558 // Some machines may have very large hard stack limit. When the test is
559 // launched by make, the default stack size is set to the hard limit, and
560 // calls to pthread_create fail with out-of-memory error.
561 // Therefore we set the stack size explicitly (as for TBB worker threads).
562#if !defined(HARNESS_THREAD_STACK_SIZE)
563#if __i386__||__i386||__arm__
564 const size_t stack_size = 1*MByte;
565#elif __x86_64__
566 const size_t stack_size = 2*MByte;
567#else
568 const size_t stack_size = 4*MByte;
569#endif
570#else
571 const size_t stack_size = HARNESS_THREAD_STACK_SIZE;
572#endif /* HARNESS_THREAD_STACK_SIZE */
573 pthread_attr_t attr_stack;
574 int status = pthread_attr_init(&attr_stack);
575 ASSERT(0==status, "NativeParallelFor: pthread_attr_init failed");
576 status = pthread_attr_setstacksize( &attr_stack, stack_size );
577 ASSERT(0==status, "NativeParallelFor: pthread_attr_setstacksize failed");
578 status = pthread_create(&thread_id, &attr_stack, thread_function, this);
579 ASSERT(0==status, "NativeParallelFor: pthread_create failed");
580 pthread_attr_destroy(&attr_stack);
581#if __ICC==1100
582 #pragma warning (pop)
583#endif
584#endif /* _WIN32||_WIN64 */
585 }
586
588 void wait_to_finish() {
589#if _WIN32||_WIN64
590 DWORD status = WaitForSingleObjectEx( thread_handle, INFINITE, FALSE );
591 ASSERT( status!=WAIT_FAILED, "WaitForSingleObject failed" );
592 CloseHandle( thread_handle );
593#else
594 int status = pthread_join( thread_id, NULL );
595 ASSERT( !status, "pthread_join failed" );
596#endif
597#if HARNESS_NO_ASSERT
598 (void)status;
599#endif
600 }
601
602private:
603#if _WIN32||_WIN64
604 HANDLE thread_handle;
605#else
606 pthread_t thread_id;
607#endif
608
610 const Index index;
611
613 const Body body;
614
615#if _WIN32||_WIN64
616 static unsigned __stdcall thread_function( void* object )
617#else
618 static void* thread_function(void* object)
619#endif
620 {
621 NativeParallelForTask& self = *static_cast<NativeParallelForTask*>(object);
622 (self.body)(self.index);
623#if HARNESS_TBBMALLOC_THREAD_SHUTDOWN && __TBB_SOURCE_DIRECTLY_INCLUDED && (_WIN32||_WIN64)
624 // in those cases can't release per-thread cache automatically,
625 // so do it manually
626 // TODO: investigate less-intrusive way to do it, for example via FLS keys
627 __TBB_mallocThreadShutdownNotification();
628#endif
629 return 0;
630 }
631};

635template<typename Index, typename Body>
636void NativeParallelFor( Index n, const Body& body ) {
637 typedef NativeParallelForTask<Index,Body> task;
638
639 if( n>0 ) {
640 // Allocate array to hold the tasks
641 task* array = static_cast<task*>(operator new( n*sizeof(task) ));
642
643 // Construct the tasks
644 for( Index i=0; i!=n; ++i )
645 new( &array[i] ) task(i,body);
646
647 // Start the tasks
648 for( Index i=0; i!=n; ++i )
649 array[i].start();
650
651 // Wait for the tasks to finish and destroy each one.
652 for( Index i=n; i; --i ) {
653 array[i-1].wait_to_finish();
654 array[i-1].~task();
655 }
656
657 // Deallocate the task array
658 operator delete(array);
659 }
660}
