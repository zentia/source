542 // string hashes
543 //
544 // Note that our string hashes here intentionally are slow for long strings.
545 // The reasoning for this is so:
546 // - The large majority of hashed strings are only a few bytes long.
547 // - The hash function is significantly more efficient if it can make this assumption.
548 // - The user is welcome to make a custom hash for those uncommon cases where
549 // long strings need to be hashed. Indeed, the user can probably make a
550 // special hash customized for such strings that's better than what we provide.
552
553 template <> struct hash<char*>
554 {
555 size_t operator()(const char* p) const
556 {
557 uint32_t c, result = 2166136261U; // FNV1 hash. Perhaps the best string hash. Intentionally uint32_t instead of size_t, so the behavior is the same regardless of size.
558 while((c = (uint8_t)*p++) != 0) // Using '!=' disables compiler warnings.
559 result = (result * 16777619) ^ c;
560 return (size_t)result;
561 }
562 };

570 result = (result * 16777619) ^ c;

675 template <typename String>
676 struct string_hash
677 {
678 typedef String string_type;
679 typedef typename String::value_type value_type;
680 typedef typename eastl::make_unsigned<value_type>::type unsigned_value_type;
681
682 size_t operator()(const string_type& s) const
683 {
684 const unsigned_value_type* p = (const unsigned_value_type*)s.c_str();
685 uint32_t c, result = 2166136261U; // Intentionally uint32_t instead of size_t, so the behavior is the same regardless of size.
686 while((c = *p++) != 0)
687 result = (result * 16777619) ^ c;
688 return (size_t)result;
689 }
