60class MyJob: public ::rml::job {
61public:
63 enum state_t {
65 unallocated,
67 idle,
69 busy,
71 clean
72 };
73 tbb::atomic<int> state;
74 tbb::atomic<int> processing_count;
75 void update( state_t new_state, state_t old_state ) {
76 int o = state.compare_and_swap(new_state,old_state);
77 ASSERT( o==old_state, "illegal transition" );
78 }
79 void update_from_either( state_t new_state, state_t old_state1, state_t old_state2 ) {
80 int snapshot;
81 do {
82 snapshot = state;
83 ASSERT( snapshot==old_state1||snapshot==old_state2, "illegal transition" );
84 } while( state.compare_and_swap(new_state,snapshot)!=snapshot );
85 }
86 MyJob() {
87 state=unallocated;
88 processing_count=0;
89 }
90 ~MyJob() {
91 // Overwrite so that accidental use after destruction can be detected.
92 memset(static_cast<void*>(this),-1,sizeof(*this));
93 }
94};

155 void acknowledge_close_connection() __TBB_override {
156 ASSERT( expect_close_connection, NULL );
157 for( size_t k=next_job_index; k>0; ) {
158 --k;
159 ASSERT( job_array[k].state==MyJob::clean, NULL );
160 }
161 delete[] job_array;
162 job_array = NULL;
163 ASSERT( my_server, NULL );
164 update( destroyed, live );
165 delete this;
166 }

168 void cleanup( job& j_ ) __TBB_override {
169 REMARK("client %d: cleanup(%p) called\n",client_id(),&j_);
170 ASSERT( state==live, NULL );
171 MyJob& j = static_cast<MyJob&>(j_);
172 while( j.state==MyJob::busy )
173 my_server->yield();
174 j.update(MyJob::clean,MyJob::idle);
175 REMARK("client %d: cleanup(%p) returns\n",client_id(),&j_);
176 }

178 job* create_one_job();
179
180protected:
181 void do_process( job* j_ ) {
182 ASSERT( state==live, NULL );
183 MyJob& j = static_cast<MyJob&>(*j_);
184 ASSERT( j_, NULL );
185 j.update(MyJob::busy,MyJob::idle);
186 // use of the plain addition (not the atomic increment) is intentonial
187 j.processing_count = j.processing_count + 1;
188 ASSERT( my_stack_size>OverheadStackSize, NULL );
189#ifdef __ia64__
190 // Half of the stack is reserved for RSE, so test only remaining half.
191 UseStackSpace( (my_stack_size-OverheadStackSize)/2 );
192#else
193 UseStackSpace( my_stack_size-OverheadStackSize );
194#endif
195 j.update(MyJob::idle,MyJob::busy);
196 my_server->yield();
197 }
198public:

202 }

227template<typename Client>
228typename Client::job* ClientBase<Client>::create_one_job() {
229 REMARK("client %d: create_one_job() called\n",client_id());
230 size_t k = next_job_index++;
231 ASSERT( state==live, NULL );
232 // Following assertion depends on assumption that implementation does not destroy jobs until
233 // the connection is closed. If the implementation is changed to destroy jobs sooner, the
234 // test logic in this header will have to be reworked.
235 ASSERT( k<my_max_job_count, "RML allocated more than max_job_count jobs simultaneously" );
236 ASSERT( k<JobArraySize, "JobArraySize not big enough (problem is in test, not RML)" );
237 MyJob& j = job_array[k];
238 j.update(MyJob::idle,MyJob::unallocated);
239 REMARK("client %d: create_one_job() for k=%d returns %p\n",client_id(),int(k),&j);
240 return &j;
241}

262void Checker::check_number_of_threads_delivered( int n_delivered, int n_requested, int n_extra ) const {
263 ASSERT( default_concurrency>=0, NULL );
264 if( tracker.all_set() ) return;
265 // Check that number of threads delivered is reasonable.
266 int n_avail = default_concurrency;
267 if( n_extra>0 )
268 n_avail-=n_extra;
269 if( n_avail<0 )
270 n_avail=0;
271 if( n_requested>default_concurrency )
272 n_avail += n_requested-default_concurrency;
273 int n_expected = n_requested;
274 if( n_expected>n_avail )
275 n_expected=n_avail;
276 const char* msg = NULL;
277 if( n_delivered>n_avail ) {
278 if( ++tracker.n_more_than_available>1 )
279 return;
280 msg = "server delivered more threads than were theoretically available";
281 } else if( n_delivered>n_expected ) {
282 if( ++tracker.n_too_many_threads>1 )
283 return;
284 msg = "server delivered more threads than expected";
285 } else if( n_delivered<n_expected ) {
286 if( ++tracker.n_system_overload>1 )
287 return;
288 msg = "server delivered fewer threads than ideal; or, the system is overloaded?";
289 }
290 if( msg ) {
291 REPORT("Warning: %s (n_delivered=%d n_avail=%d n_requested=%d n_extra=%d default_concurrency=%d)\n",
292 msg, n_delivered, n_avail, n_requested, n_extra, default_concurrency );
293 }

319template<typename Factory,typename Client>
320void DoOneConnection<Factory,Client>::operator()( size_t i ) const {
321 ASSERT( i<MaxClient, NULL );
322 Client* client = new Client;
323 client->initialize( Client::is_omp ? JobArraySize : n_thread, nesting, ClientStackSize[i] );
324 Factory factory;
325 memset( &factory, 0, sizeof(factory) );
326 typename Factory::status_type status = factory.open();
327 ASSERT( status==Factory::st_success, NULL );
328
329 typename Factory::server_type* server;
330 status = factory.make_server( server, *client );
331 ASSERT( status==Factory::st_success, NULL );
332 Harness::ConcurrencyTracker ct;
333 REMARK("client %d: opened server n_thread=%d nesting=(%d,%d)\n",
334 client->client_id(), n_thread, nesting.level, nesting.limit);
335 client->set_server( server );
336 Checker checker( *server );
337
338 FireUpJobs( *server, *client, n_thread, n_extra, check_delivered && !client->is_strict() ? &checker : NULL );
339
340 // Close the connection
341 client->expect_close_connection = true;
342 REMARK("client %d: calling request_close_connection\n", client->client_id());
343#if !RML_USE_WCRM
344 int default_concurrency = server->default_concurrency();
345#endif
346 server->request_close_connection();
347 // Client deletes itself when it sees call to acknowledge_close_connection from server.
348 factory.close();
349#if !RML_USE_WCRM
350 if( TestSingleConnection )
351 __TBB_ASSERT_EX( uintptr_t(factory.scratch_ptr)==uintptr_t(default_concurrency), "under/over subscription?" );
352#endif
353}

356template<typename Factory, typename Client>
357void SimpleTest() {
358 Harness::ConcurrencyTracker::Reset();
359 TestSingleConnection = true;
360 N_TestConnections = 1;
361 for( int n_thread=MinThread; n_thread<=MaxThread; ++n_thread ) {
362 // Test a single connection, no nesting, no extra threads
363 DoOneConnection<Factory,Client> doc(n_thread,Nesting(0,0),0,false);
364 doc(0);
365 }
366#if !TRIVIAL
367 TestSingleConnection = false;
368 for( int n_thread=MinThread; n_thread<=MaxThread; ++n_thread ) {
369 // Test parallel connections
370 for( int n_client=1; n_client<=int(MaxClient); ++n_client ) {
371 N_TestConnections = n_client;
372 REMARK("SimpleTest: n_thread=%d n_client=%d\n",n_thread,n_client);
373 NativeParallelFor( n_client, DoOneConnection<Factory,Client>(n_thread,Nesting(0,0),0,false) );
374 }
375 // Test server::independent_thread_number_changed
376 N_TestConnections = 1;
377 for( int n_extra=-4; n_extra<=32; n_extra=n_extra+1+n_extra/5 ) {
378 DoOneConnection<Factory,Client> doc(n_thread,Nesting(0,0),n_extra,true);
379 doc(0);
380 }
381#if !RML_USE_WCRM
382 // Test nested connections
383 DoOneConnection<Factory,Client> doc(n_thread,Nesting(0,2),0,false);
384 doc(0);
385#endif
386 }
387 ASSERT( Harness::ConcurrencyTracker::PeakParallelism()>1 || server_concurrency==0, "No multiple connections exercised?" );
388#endif /* !TRIVIAL */
389 // Let RML catch up.
390 while( ClientConstructions!=ClientDestructions )
391 Harness::Sleep(1);
392}
