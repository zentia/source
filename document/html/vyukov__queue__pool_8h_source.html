89 template <typename T, typename Traits = vyukov_queue_pool_traits >
90 class vyukov_queue_pool
91 {
92 public:
93 typedef cds::intrusive::VyukovMPMCCycleQueue< T, Traits > queue_type ;
94
95 public:
96 typedef T value_type ;
97 typedef Traits traits;
98 typedef typename std::allocator_traits<typename traits::allocator>::template rebind_alloc<value_type> allocator_type;
99 typedef typename traits::back_off back_off;
100
101 protected:
102 //@cond
103 typedef cds::details::Allocator< value_type, allocator_type > cxx_allocator;
104 typedef typename cxx_allocator::allocator_type std_allocator;
105
106 queue_type m_Queue;
107 value_type * m_pFirst;
108 value_type * m_pLast;
109 //@endcond
110
111 protected:
112 //@cond
113 void preallocate_pool()
114 {
115 m_pFirst = std_allocator().allocate( m_Queue.capacity());
116 m_pLast = m_pFirst + m_Queue.capacity();
117
118 for ( value_type * p = m_pFirst; p < m_pLast; ++p ) {
119 CDS_VERIFY( m_Queue.push( *p )) ; // must be true
120 }
121 }
122
123 bool from_pool( value_type * p ) const
124 {
125 return m_pFirst <= p && p < m_pLast;
126 }
127 //@endcond
128
129 public:
131
136 vyukov_queue_pool( size_t nCapacity = 0 )
137 : m_Queue( nCapacity )
138 {
139 preallocate_pool();
140 }
141
143 ~vyukov_queue_pool()
144 {
145 m_Queue.clear();
146 std_allocator().deallocate( m_pFirst, m_Queue.capacity());
147 }
148
150
157 value_type * allocate( size_t n )
158 {
159 assert( n == 1 );
160 CDS_UNUSED(n);
161
162 value_type * p = m_Queue.pop();
163 if ( p ) {
164 assert( from_pool(p));
165 return new( p ) value_type;
166 }
167 // The pool is empty - allocate new from the heap
168 return cxx_allocator().New();
169 }
170
172
179 void deallocate( value_type * p, size_t n )
180 {
181 assert( n == 1 );
182 CDS_UNUSED(n);
183
184 if ( p ) {
185 if ( from_pool(p)) {
186 p->~value_type();
187 // The queue can notify about false fullness state
188 // so we push in loop
189 back_off bkoff;
190 while ( !m_Queue.push( *p ))
191 bkoff();
192 }
193 else
194 cxx_allocator().Delete( p );
195 }
196 }
197 };

429 void preallocate_pool()
430 {
431 size_t const nCount = m_Queue.capacity();
432 m_pFirst = std_allocator().allocate( nCount );
433 m_pLast = m_pFirst + nCount;
434

437 }

442 }

452 bounded_vyukov_queue_pool( size_t nCapacity = 0 )
453 : m_Queue( nCapacity )
454 {
455 preallocate_pool();
456 }
457
459 ~bounded_vyukov_queue_pool()
460 {
461 m_Queue.clear();
462 std_allocator().deallocate( m_pFirst, m_Queue.capacity());
463 }
464
466
473 value_type * allocate( size_t n )
474 {
475 assert( n == 1 );
476 CDS_UNUSED( n );
477
478 value_type * p = m_Queue.pop();
479
480 if ( !p ) {
481 back_off bkoff;
482 while ( m_Queue.size()) {
483 p = m_Queue.pop();
484 if ( p )
485 goto ok;
486 bkoff();
487 }
488
489 // The pool is empty
490 CDS_THROW_EXCEPTION( std::bad_alloc());
491 }
492
493 ok:
494 assert( from_pool(p));
495 return p;
496 }
497
499
505 void deallocate( value_type * p, size_t n )
506 {
507 assert( n == 1 );
508 CDS_UNUSED( n );
509
510 if ( p ) {
511 assert( from_pool( p ));
512 back_off bkoff;
513 // The queue can notify it is full but that is false fullness state
514 // So, we push in loop
515 while ( !m_Queue.push(*p))
516 bkoff();
517 }
518 }
