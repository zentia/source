56 split_bitstring( bitstring const& h, size_t nBitOffset )
57 : cur_( reinterpret_cast<uint8_t const*>( &h ) + nBitOffset / c_nBitPerByte )
58 , offset_( nBitOffset % c_nBitPerByte )
59 , first_( reinterpret_cast<uint8_t const*>( &h ))
60 , last_( first_ + c_bitstring_size )
61 {}

72 return cur_ >= last_;
73 }

80 uint_type cut( unsigned count )
81 {
82 assert( !eos());
83
84 uint_type result = 0;
85# if defined( CDS_ARCH_LITTLE_ENDIAN )
86 for ( unsigned done = 0; done < count; ) {
87 assert( cur_ < last_ );
88 unsigned bits = count - done;
89 if ( bits > c_nBitPerByte - offset_ )
90 bits = c_nBitPerByte - offset_;
91
92 result |= static_cast<uint_type>(( *cur_ >> offset_ ) & (( 1 << bits ) - 1 )) << done;
93
94 offset_ += bits;
95 assert( offset_ <= c_nBitPerByte );
96 if ( offset_ == c_nBitPerByte ) {
97 offset_ = 0;
98 ++cur_;
99 }
100 done += bits;
101 }
102# else
103 while ( count ) {
104 assert( cur_ < last_ );
105
106 unsigned bits = count <= ( c_nBitPerByte - offset_ ) ? count : c_nBitPerByte - offset_;
107
108 result = ( result << bits ) | (( *cur_ >> offset_ ) & ( ( 1 << bits ) - 1 ));
109
110 offset_ += bits;
111 assert( offset_ <= c_nBitPerByte );
112 if ( offset_ == c_nBitPerByte ) {
113 offset_ = 0;
114 ++cur_;
115 }
116 count -= bits;
117 }
118# endif
119
120 return result;
121 }
122
124
129 uint_type safe_cut( unsigned count )
130 {
131 if ( eos())
132 return 0;
133
134 unsigned const rest = static_cast<unsigned>( last_ - cur_ - 1 ) * c_nBitPerByte + ( c_nBitPerByte - offset_ );
135 if ( rest < count )
136 count = rest;
137 return count ? cut( count ) : 0;
138 }
139
141 void reset() noexcept
142 {
143 cur_ = first_;
144 offset_ = 0;
145 }

149 {
150 return reinterpret_cast<bitstring const *>( first_ );
151 }

156 return offset_ + (cur_ - first_) * c_nBitPerByte;
157 }
158

161 {
162 return c_bitstring_size * c_nBitPerByte - bit_offset();
163 }
164
166 static constexpr bool is_correct( unsigned /*count*/ )

168 return true;
169 }

177 //@endcond

234

237 result |= static_cast<uint_type>( *cur_ ) << i;
238 ++cur_;
239 }
240# else
