71 fftree(ff_node *ffnode_, fftype nodetype_) :
72 nodetype(nodetype_) {
73 do_comp = (nodetype == WORKER || nodetype == OCL_WORKER || nodetype == TPC_WORKER);
74 ffnode.ffnode = ffnode_;
75 isroot = ispattern();
76 hasocl = (nodetype == OCL_WORKER);

115 ~fftree() {
116 for (size_t i = 0; i < children.size(); ++i)
117 if (children[i].first && !children[i].second)
118 delete children[i].first;
119 if (isroot) {
120 treeLock.lock();
121 //pthread_mutex_lock(&treeLock);
122#if defined(DEBUG_FFTREE)
123 std::set<fftree *>::iterator it = roots.find(this);
124 assert(it != roots.end());
125 roots.erase(it);
126#else
127 roots.erase(this);
128#endif
129 treeLock.unlock();
130 //pthread_mutex_unlock(&treeLock);
131 }
132 }

158 std::string fftype_tostr(fftype t) {
159 switch (t) {
160 case FARM:
161 return "FARM";
162 case PIPE:
163 return "PIPE";
164 case EMITTER:
165 return "EMITTER";
166 case WORKER:
167 return "WORKER";
168 case OCL_WORKER:
169 return "OPENCL WORKER";
170 case TPC_WORKER:
171 return "TPC WORKER";
172 case COLLECTOR:
173 return "COLLECTOR";
174 }
175 return std::string("puppa"); //never reached
176 }

182 bool hasOpenCLNode() const {
183 if (hasocl) return true;
184 for (size_t i = 0; i < children.size(); ++i)
185 if (children[i].first && children[i].first->hasOpenCLNode()) return true;
186 return false;
187 }

189 bool hasTPCNode() const {
190 if (hastpc) return true;

192 if (children[i].first && children[i].first->hasTPCNode()) return true;
193 return false;

197 void print(std::ostream &os) {
198 os << (ispattern() ? "[" : "(") << fftype_tostr(nodetype);
199 if (do_comp)
200 os << " docomp";
201 for (size_t i = 0; i < children.size(); ++i) {
202 if (children[i].first) {
203 os << " ";
204 children[i].first->print(os);
205 }
206 }
207 os << (ispattern() ? "]" : ")");
208 }
209
210 size_t threadcount(threadcount_t *tc) const {
211 size_t cnt = !ispattern();
212 if (cnt) {
213 tc->n_emitter += nodetype == EMITTER;
214 tc->n_collector += nodetype == COLLECTOR;
215 tc->n_workers += (nodetype == WORKER || nodetype == OCL_WORKER);
216 tc->n_docomp += do_comp;
217 }
218 for (size_t i = 0; i < children.size(); ++i)
219 if (children[i].first)
220 cnt += children[i].first->threadcount(tc);
221 return cnt;
222 }

225static inline void print_fftrees(std::ostream &os) {
226 size_t t = 0, total = 0;
227 for (std::set<fftree *>::const_iterator it = roots.begin();
228 it != roots.end(); ++it) {
229 threadcount_t tc;
230 os << "> ff tree @" << *it << "\n";
231 (*it)->print(os);
232 total += (t = (*it)->threadcount(&tc));
233 os << "\nthread count = " << t << "\n";
234 }
235 os << "total thread count = " << total << "\n";
236}
