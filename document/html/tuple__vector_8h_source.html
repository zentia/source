6// tuple_vector is a data container that is designed to abstract and simplify
7// the handling of a "structure of arrays" layout of data in memory. In
8// particular, it mimics the interface of vector, including functionality to do
9// inserts, erases, push_backs, and random-access. It also provides a
10// RandomAccessIterator and corresponding functionality, making it compatible
11// with most STL (and STL-esque) algorithms such as ranged-for loops, find_if,
12// remove_if, or sort.
13
14// When used or applied properly, this container can improve performance of
15// some algorithms through cache-coherent data accesses or allowing for
16// sensible SIMD programming, while keeping the structure of a single
17// container, to permit a developer to continue to use existing algorithms in
18// STL and the like.
19//
20// Consult doc/Bonus/tuple_vector_readme.md for more information.

209 template<typename Allocator, size_type I, typename Indices, typename... VecTypes>
210 static pair<void*, size_type> DoAllocate(TupleVecImpl<Allocator, Indices, VecTypes...> &vec, void** ppNewLeaf, size_type capacity, size_type offset)
211 {
212 size_type allocationOffset = CalculatAllocationOffset(offset);
213 size_type allocationSize = CalculateAllocationSize(offset, capacity);
214 pair<void*, size_type> allocation = TupleRecurser<Ts...>::template DoAllocate<Allocator, I + 1, Indices, VecTypes...>(
215 vec, ppNewLeaf, capacity, allocationSize);
216 ppNewLeaf[I] = (void*)((uintptr_t)(allocation.first) + allocationOffset);
217 return allocation;
218 }

243 void DoUninitializedMoveAndDestruct(const size_type begin, const size_type end, T* pDest)
244 {
245 T* pBegin = mpData + begin;
246 T* pEnd = mpData + end;
247 eastl::uninitialized_move_ptr_if_noexcept(pBegin, pEnd, pDest);
248 eastl::destruct(pBegin, pEnd);
249 }
250
251 void DoInsertAndFill(size_type pos, size_type n, size_type numElements, const T& arg)
252 {
253 T* pDest = mpData + pos;
254 T* pDataEnd = mpData + numElements;
255 const T temp = arg;
256 const size_type nExtra = (numElements - pos);
257 if (n < nExtra) // If the inserted values are entirely within initialized memory (i.e. are before mpEnd)...
258 {
259 eastl::uninitialized_move_ptr(pDataEnd - n, pDataEnd, pDataEnd);
260 eastl::move_backward(pDest, pDataEnd - n, pDataEnd); // We need move_backward because of potential overlap issues.
261 eastl::fill(pDest, pDest + n, temp);
262 }
263 else
264 {
265 eastl::uninitialized_fill_n_ptr(pDataEnd, n - nExtra, temp);
266 eastl::uninitialized_move_ptr(pDest, pDataEnd, pDataEnd + n - nExtra);
267 eastl::fill(pDest, pDataEnd, temp);
268 }
269 }
270
271 void DoInsertRange(T* pSrcBegin, T* pSrcEnd, T* pDestBegin, size_type numDataElements)
272 {
273 size_type pos = static_cast<size_type>(pDestBegin - mpData);
274 size_type n = static_cast<size_type>(pSrcEnd - pSrcBegin);
275 T* pDataEnd = mpData + numDataElements;
276 const size_type nExtra = numDataElements - pos;
277 if (n < nExtra) // If the inserted values are entirely within initialized memory (i.e. are before mpEnd)...
278 {
279 eastl::uninitialized_move_ptr(pDataEnd - n, pDataEnd, pDataEnd);
280 eastl::move_backward(pDestBegin, pDataEnd - n, pDataEnd); // We need move_backward because of potential overlap issues.
281 eastl::copy(pSrcBegin, pSrcEnd, pDestBegin);
282 }
283 else
284 {
285 eastl::uninitialized_copy(pSrcEnd - (n - nExtra), pSrcEnd, pDataEnd);
286 eastl::uninitialized_move_ptr(pDestBegin, pDataEnd, pDataEnd + n - nExtra);
287 eastl::copy(pSrcBegin, pSrcEnd - (n - nExtra), pDestBegin);
288 }
289 }
290
291 void DoInsertValue(size_type pos, size_type numElements, T&& arg)
292 {
293 T* pDest = mpData + pos;
294 T* pDataEnd = mpData + numElements;
295
296 eastl::uninitialized_move_ptr(pDataEnd - 1, pDataEnd, pDataEnd);
297 eastl::move_backward(pDest, pDataEnd - 1, pDataEnd); // We need move_backward because of potential overlap issues.
298 eastl::destruct(pDest);
299 ::new (pDest) T(eastl::forward<T>(arg));
300 }

302 T* mpData = nullptr;
303};

583 void assign(size_type n, const Ts&... args)
584 {
585 if (n > mNumCapacity)
586 {
587 this_type temp(n, args..., get_allocator()); // We have little choice but to reallocate with new memory.
588 swap(temp);
589 }
590 else if (n > mNumElements) // If n > mNumElements ...
591 {
592 size_type oldNumElements = mNumElements;
593 swallow((eastl::fill(TupleVecLeaf<Indices, Ts>::mpData, TupleVecLeaf<Indices, Ts>::mpData + oldNumElements, args), 0)...);
594 swallow((eastl::uninitialized_fill_ptr(TupleVecLeaf<Indices, Ts>::mpData + oldNumElements,
595 TupleVecLeaf<Indices, Ts>::mpData + n, args), 0)...);
596 mNumElements = n;
597 }
598 else // else 0 <= n <= mNumElements
599 {
600 swallow((eastl::fill(TupleVecLeaf<Indices, Ts>::mpData, TupleVecLeaf<Indices, Ts>::mpData + n, args), 0)...);
601 erase(begin() + n, end());
602 }

605 void assign(const_iterator first, const_iterator last)
606 {
607#if EASTL_ASSERT_ENABLED
608 if (EASTL_UNLIKELY(!validate_iterator_pair(first, last)))
609 EASTL_FAIL_MSG("tuple_vector::assign -- invalid iterator pair");
610#endif
611 size_type newNumElements = last - first;
612 if (newNumElements > mNumCapacity)
613 {
614 this_type temp(first, last, get_allocator());
615 swap(temp);
616 }
617 else
618 {
619 const void* ppOtherData[sizeof...(Ts)] = {first.mpData[Indices]...};
620 size_type firstIdx = first.mIndex;
621 size_type lastIdx = last.mIndex;
622 if (newNumElements > mNumElements) // If n > mNumElements ...
623 {
624 size_type oldNumElements = mNumElements;
625 swallow((eastl::copy((Ts*)(ppOtherData[Indices]) + firstIdx,
626 (Ts*)(ppOtherData[Indices]) + firstIdx + oldNumElements,
