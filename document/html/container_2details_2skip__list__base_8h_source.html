159 template <typename Node, typename Traits>
160 class node_allocator
161 {
162 protected:
163 typedef Node node_type;
164 typedef Traits traits;
165
166 typedef typename node_type::tower_item_type node_tower_item;
167
168 typedef typename std::allocator_traits<typename traits::allocator>::template rebind_alloc<unsigned char> tower_allocator_type;
169 typedef typename std::allocator_traits<typename traits::allocator>::template rebind_alloc<node_type> node_allocator_type;
170
171 static size_t const c_nTowerItemSize = sizeof(node_tower_item);
172 static size_t const c_nNodePadding = sizeof(node_type) % c_nTowerItemSize;
173 static size_t const c_nNodeSize = sizeof(node_type) + (c_nNodePadding ? (c_nTowerItemSize - c_nNodePadding) : 0);
174
175 static constexpr size_t node_size( unsigned int nHeight ) noexcept
176 {
177 return c_nNodeSize + (nHeight - 1) * c_nTowerItemSize;
178 }
179
180 static unsigned char * alloc_space( unsigned int nHeight )
181 {
182 unsigned char * pMem;
183 size_t const sz = node_size( nHeight );
184
185 if ( nHeight > 1 ) {
186 pMem = tower_allocator_type().allocate( sz );
187
188 // check proper alignments
189 assert( (((uintptr_t) pMem) & (alignof(node_type) - 1)) == 0 );
190 assert( (((uintptr_t) (pMem + c_nNodeSize)) & (alignof(node_tower_item) - 1)) == 0 );
191 return pMem;
192 }
193 else
194 pMem = reinterpret_cast<unsigned char *>( node_allocator_type().allocate( 1 ));
195
196 return pMem;
197 }
198
199 static void free_space( unsigned char * p, unsigned int nHeight )
200 {
201 assert( p != nullptr );
202
203 if ( nHeight == 1 )
204 node_allocator_type().deallocate( reinterpret_cast<node_type *>(p), 1 );
205 else
206 tower_allocator_type().deallocate( p, node_size(nHeight));
207 }
208
209 public:
210 template <typename Q>
211 node_type * New( unsigned int nHeight, Q const& v )
212 {
213 unsigned char * pMem = alloc_space( nHeight );
214 node_type * p = new( pMem )
215 node_type( nHeight, nHeight > 1 ? reinterpret_cast<node_tower_item *>(pMem + c_nNodeSize) : nullptr, v );
216 return p;
217 }
218
219 template <typename... Args>
220 node_type * New( unsigned int nHeight, Args&&... args )
221 {
222 unsigned char * pMem = alloc_space( nHeight );
223 node_type * p = new( pMem )
224 node_type( nHeight, nHeight > 1 ? reinterpret_cast<node_tower_item *>(pMem + c_nNodeSize) : nullptr,
225 std::forward<Args>(args)... );
226 return p;
227 }
228
229 void Delete( node_type * p )
230 {
231 assert( p != nullptr );
232
233 unsigned int nHeight = p->height();
234 node_allocator_type a;
235 std::allocator_traits<node_allocator_type>::destroy( a, p );
236 free_space( reinterpret_cast<unsigned char *>(p), nHeight );
237 }
238 };
