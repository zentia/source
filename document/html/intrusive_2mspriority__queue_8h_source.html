255 bool push( value_type& val )
256 {
257 tag_type const curId = cds::OS::get_current_thread_id();
258
259 // Insert new item at bottom of the heap
260 m_Lock.lock();
261 if ( m_ItemCounter.value() >= capacity()) {
262 // the heap is full
263 m_Lock.unlock();
264 m_Stat.onPushFailed();
265 return false;
266 }
267
268 counter_type i = m_ItemCounter.inc();
269 assert( i < m_Heap.capacity());
270
271 node& refNode = m_Heap[i];
272 refNode.lock();
273 m_Lock.unlock();
274 assert( refNode.m_nTag == tag_type( Empty ));
275 assert( refNode.m_pVal == nullptr );
276 refNode.m_pVal = &val;
277 refNode.m_nTag = curId;
278 refNode.unlock();
279
280 // Move item towards top of heap while it has a higher priority than its parent
281 heapify_after_push( i, curId );
282
283 m_Stat.onPushSuccess();
284 return true;
285 }
286
288
292 value_type * pop()
293 {
294 node& refTop = m_Heap[1];
295
296 m_Lock.lock();
297 if ( m_ItemCounter.value() == 0 ) {
298 // the heap is empty
299 m_Lock.unlock();
300 m_Stat.onPopFailed();
301 return nullptr;
302 }
303 counter_type nBottom = m_ItemCounter.dec();
304 assert( nBottom < m_Heap.capacity());
305 assert( nBottom > 0 );
306
307 refTop.lock();
308 if ( nBottom == 1 ) {
309 refTop.m_nTag = tag_type( Empty );
310 value_type * pVal = refTop.m_pVal;
311 refTop.m_pVal = nullptr;
312 refTop.unlock();
313 m_Lock.unlock();
314 m_Stat.onPopSuccess();
315 return pVal;
316 }
317
318 node& refBottom = m_Heap[nBottom];
319 refBottom.lock();
320 m_Lock.unlock();
321 refBottom.m_nTag = tag_type(Empty);
322 value_type * pVal = refBottom.m_pVal;
323 refBottom.m_pVal = nullptr;
324 refBottom.unlock();
325
326 if ( refTop.m_nTag == tag_type(Empty)) {
327 // nBottom == nTop
328 refTop.unlock();
329 m_Stat.onPopSuccess();
330 return pVal;
331 }
332
333 std::swap( refTop.m_pVal, pVal );
334 refTop.m_nTag = tag_type( Available );
335
336 // refTop will be unlocked inside heapify_after_pop
337 heapify_after_pop( &refTop );
338
339 m_Stat.onPopSuccess();
340 return pVal;
341 }

409 void heapify_after_push( counter_type i, tag_type curId )
410 {
411 key_comparator cmp;
412 back_off bkoff;
413
414 // Move item towards top of the heap while it has higher priority than parent
415 while ( i > 1 ) {
416 bool bProgress = true;
417 counter_type nParent = i / 2;
418 node& refParent = m_Heap[nParent];
419 refParent.lock();
420 node& refItem = m_Heap[i];
421 refItem.lock();
422
423 if ( refParent.m_nTag == tag_type(Available) && refItem.m_nTag == curId ) {
424 if ( cmp( *refItem.m_pVal, *refParent.m_pVal ) > 0 ) {
425 std::swap( refItem.m_nTag, refParent.m_nTag );
426 std::swap( refItem.m_pVal, refParent.m_pVal );
427 m_Stat.onPushHeapifySwap();
428 i = nParent;
429 }
430 else {
431 refItem.m_nTag = tag_type(Available);
432 i = 0;
433 }
434 }
435 else if ( refParent.m_nTag == tag_type( Empty )) {
436 m_Stat.onItemMovedTop();
437 i = 0;
438 }
439 else if ( refItem.m_nTag != curId ) {
440 m_Stat.onItemMovedUp();
441 i = nParent;
442 }
443 else {
444 m_Stat.onPushEmptyPass();
445 bProgress = false;
446 }
447
448 refItem.unlock();
449 refParent.unlock();
450
451 if ( !bProgress )
452 bkoff();
453 else
454 bkoff.reset();
455 }
456
457 if ( i == 1 ) {
458 node& refItem = m_Heap[i];
459 refItem.lock();
460 if ( refItem.m_nTag == curId )
461 refItem.m_nTag = tag_type(Available);
462 refItem.unlock();
463 }
464 }
465
466 void heapify_after_pop( node * pParent )
467 {
468 key_comparator cmp;
469 counter_type const nCapacity = m_Heap.capacity();
470
471 counter_type nParent = 1;
472 for ( counter_type nChild = nParent * 2; nChild < nCapacity; nChild *= 2 ) {
473 node* pChild = &m_Heap[ nChild ];
474 pChild->lock();
475
476 if ( pChild->m_nTag == tag_type( Empty )) {
477 pChild->unlock();
478 break;
479 }
480
481 counter_type const nRight = nChild + 1;
482 if ( nRight < nCapacity ) {
483 node& refRight = m_Heap[nRight];
484 refRight.lock();
485
486 if ( refRight.m_nTag != tag_type( Empty ) && cmp( *refRight.m_pVal, *pChild->m_pVal ) > 0 ) {
487 // get right child
488 pChild->unlock();
489 nChild = nRight;
490 pChild = &refRight;
491 }
492 else
493 refRight.unlock();
494 }
495
496 // If child has higher priority than parent then swap
497 // Otherwise stop
498 if ( cmp( *pChild->m_pVal, *pParent->m_pVal ) > 0 ) {
499 std::swap( pParent->m_nTag, pChild->m_nTag );
500 std::swap( pParent->m_pVal, pChild->m_pVal );
501 pParent->unlock();
502 m_Stat.onPopHeapifySwap();
503 nParent = nChild;
504 pParent = pChild;
505 }
506 else {
507 pChild->unlock();
508 break;
509 }
510 }
511 pParent->unlock();
512 }
513 //@endcond
