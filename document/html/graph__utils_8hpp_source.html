100// returns:
101// - the innermost BB that contains the node 'n' passed as argument
102// - null if the node 'n' is not found
103// - the node 'n' if the starting node is 'n'
104static inline ff_node* getBB(ff_node* startnode, ff_node* n) {
105 if (startnode == n) return n;
106
107 if (startnode->isPipe()) {
108 ff_pipeline* pipe = reinterpret_cast<ff_pipeline*>(startnode);
109 svector<ff_node*> Vn = pipe->getStages();
110 for(size_t i=0;i<Vn.size();++i) {
111 ff_node* r = getBB(Vn[i], n);
112 if (r) return ((r==n)?pipe:r);
113 }
114 return nullptr;
115 }
116 if (startnode->isAll2All()) {
117 ff_a2a* a2a = reinterpret_cast<ff_a2a*>(startnode);
118 svector<ff_node*> L = a2a->getFirstSet();
119 svector<ff_node*> R = a2a->getSecondSet();
120 for(size_t i=0;i<L.size();++i) {
121 ff_node* r = getBB(L[i], n);
122 if (r) return ((r==n)?a2a:r);
123 }
124 for(size_t i=0;i<R.size();++i) {
125 ff_node* r = getBB(R[i], n);
126 if (r) return ((r==n)?a2a:r);
127 }
128 return nullptr;
129 }
130 if (startnode->isOFarm()) abort(); // TODO: ofarm
131 if (startnode->isFarm()) {
132 ff_farm* farm = reinterpret_cast<ff_farm*>(startnode);
133 if (farm->getEmitter() == n) return farm;
134 if (farm->getCollector() == n) return farm;
135 svector<ff_node*> W = farm->getWorkers();
136 for(size_t i=0;i<W.size();++i) {
137 ff_node* r = getBB(W[i], n);
138 if (r) return ((r==n)?farm:r);
139 }
140 return nullptr;
141 }
142 if (startnode->isComp()) {
143 ff_comb* comb = reinterpret_cast<ff_comb*>(startnode);
144 ff_node* cl = comb->getLeft();
145 ff_node* cr = comb->getRight();
146 if (cl->isComp()) {
147 ff_node* r = getBB(cl,n);
148 if (r) return ((r==n)?cl:r);
149 } else {
150 if (comb->getFirst() == n) return comb;
151 }
152 if (cr->isComp()) {
153 ff_node* r = getBB(cr,n);
154 if (r) return ((r==n)?cr:r);
155 } else {
156 if (comb->getLast() == n) return comb;
157 }
158 return nullptr;
159 }
160 return nullptr;
161}
