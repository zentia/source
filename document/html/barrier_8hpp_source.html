169class spinBarrier: public ffBarrier {
170public:
171
172 spinBarrier(const size_t _maxNThreads=MAX_NUM_THREADS):maxNThreads(_maxNThreads), _barrier(0) {
173 barArray=new bool[maxNThreads];
174 assert(barArray!=NULL);
175 }
176
177 ~spinBarrier() {
178 if (barArray != NULL) delete [] barArray;
179 barArray=NULL;
180 }
181
182 inline int barrierSetup(size_t init) {
183 assert(init>0);
184 if (init == _barrier) return -1;
185 for(size_t i=0; i<maxNThreads; ++i) barArray[i]=false;
186 B[0]=0; B[1]=0;
187 _barrier = init;
188 return 0;
189 }
190
191 inline void doBarrier(size_t tid) {
192 assert(tid<maxNThreads);
193 const int whichBar = (barArray[tid] ^= true); // computes % 2
194 long c = ++B[whichBar];
195 if ((size_t)c == _barrier) {
196 B[whichBar]=0;
197 return;
198 }
199 // spin-wait
200 while(c) {
201 c = B[whichBar];
202 PAUSE(); // TODO: define a spin policy !
203 }
204 }
205private:
206 const size_t maxNThreads;
207 size_t _barrier;
208 bool* barArray;
209 std::atomic<long> B[2];
210};
