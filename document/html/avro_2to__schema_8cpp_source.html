35inline bool is_named_type(const parsing::schema::Type& _type) {
36 return _type.variant_.visit([&](const auto& _v) -> bool {
37 using T = std::remove_cvref_t<decltype(_v)>;
38 return std::is_same<T, parsing::schema::Type::Object>() ||
39 std::is_same<T, parsing::schema::Type::Literal>();
40 });
41}

43schema::Type type_to_avro_schema_type(
44 const parsing::schema::Type& _type,
45 const std::map<std::string, parsing::schema::Type>& _definitions,
46 std::set<std::string>* _already_known, size_t* _num_unnamed) {
47 auto handle_variant = [&](const auto& _t) -> schema::Type {
48 using T = std::remove_cvref_t<decltype(_t)>;
49 using Type = parsing::schema::Type;
50 if constexpr (std::is_same<T, Type::Boolean>()) {
51 return schema::Type{.value = schema::Type::Boolean{}};
52
53 } else if constexpr (std::is_same<T, Type::Bytestring>()) {
54 return schema::Type{.value = schema::Type::Bytes{}};
55
56 } else if constexpr (std::is_same<T, Type::Int32>() ||
57 std::is_same<T, Type::Int64>() ||
58 std::is_same<T, Type::UInt32>() ||
59 std::is_same<T, Type::UInt64>() ||
60 std::is_same<T, Type::Integer>()) {
61 return schema::Type{.value = schema::Type::Int{}};
62
63 } else if constexpr (std::is_same<T, Type::Int64>() ||
64 std::is_same<T, Type::UInt32>() ||
65 std::is_same<T, Type::UInt64>()) {
66 return schema::Type{.value = schema::Type::Long{}};
67
68 } else if constexpr (std::is_same<T, Type::Float>()) {
69 return schema::Type{.value = schema::Type::Float{}};
70
71 } else if constexpr (std::is_same<T, Type::Double>()) {
72 return schema::Type{.value = schema::Type::Double{}};
73
74 } else if constexpr (std::is_same<T, Type::String>()) {
75 return schema::Type{.value = schema::Type::String{}};
76
77 } else if constexpr (std::is_same<T, Type::AnyOf>()) {
78 auto any_of = std::vector<schema::Type>();
79 for (const auto& t : _t.types_) {
80 any_of.emplace_back(type_to_avro_schema_type(
81 t, _definitions, _already_known, _num_unnamed));
82 }
83 return schema::Type{.value = any_of};
84
85 } else if constexpr (std::is_same<T, Type::Description>()) {
86 // TODO: Return descriptions
87 return type_to_avro_schema_type(*_t.type_, _definitions, _already_known,
88 _num_unnamed);
89
90 } else if constexpr (std::is_same<T, Type::FixedSizeTypedArray>()) {
91 return schema::Type{
92 .value = schema::Type::Array{
93 .items = Ref<schema::Type>::make(type_to_avro_schema_type(
94 *_t.type_, _definitions, _already_known, _num_unnamed))}};
95
96 } else if constexpr (std::is_same<T, Type::Literal>()) {
97 return schema::Type{
98 .value = schema::Type::Enum{.name = std::string("unnamed_") +
99 std::to_string(++(*_num_unnamed)),
100 .symbols = _t.values_}};
101
102 } else if constexpr (std::is_same<T, Type::Object>()) {
103 auto record = schema::Type::Record{
104 .name = std::string("unnamed_") + std::to_string(++(*_num_unnamed))};
105 for (const auto& [k, v] : _t.types_) {
106 record.fields.push_back(schema::Type::RecordField{
107 .name = k,
108 .type = Ref<schema::Type>::make(type_to_avro_schema_type(
109 v, _definitions, _already_known, _num_unnamed))});
110 }
111 return schema::Type{.value = record};
112
113 } else if constexpr (std::is_same<T, Type::Optional>()) {
114 return schema::Type{
115 .value = std::vector<schema::Type>(
116 {type_to_avro_schema_type(*_t.type_, _definitions, _already_known,
117 _num_unnamed),
118 schema::Type{schema::Type::Null{}}})};
119
120 } else if constexpr (std::is_same<T, Type::Reference>()) {
121 if (_definitions.find(_t.name_) != _definitions.end() &&
122 !is_named_type(_definitions.at(_t.name_))) {
123 return type_to_avro_schema_type(_definitions.at(_t.name_), _definitions,
124 _already_known, _num_unnamed);
125
126 } else if (_already_known->find(_t.name_) != _already_known->end() ||
127 _definitions.find(_t.name_) == _definitions.end()) {
128 return schema::Type{.value = schema::Type::Reference{.type = _t.name_}};
129
130 } else {
131 _already_known->insert(_t.name_);
132 return type_to_avro_schema_type(_definitions.at(_t.name_), _definitions,
133 _already_known, _num_unnamed)
134 .with_name(_t.name_);
135 }
136
137 } else if constexpr (std::is_same<T, Type::StringMap>()) {
138 return schema::Type{
139 .value = schema::Type::Map{
140 .values = Ref<schema::Type>::make(
141 type_to_avro_schema_type(*_t.value_type_, _definitions,
142 _already_known, _num_unnamed))}};
143
144 } else if constexpr (std::is_same<T, Type::Tuple>()) {
145 return type_to_avro_schema_type(
146 Type{parsing::schemaful::tuple_to_object(_t)}, _definitions,
147 _already_known, _num_unnamed);
148
149 } else if constexpr (std::is_same<T, Type::TypedArray>()) {
150 return schema::Type{
151 .value = schema::Type::Array{
152 .items = Ref<schema::Type>::make(type_to_avro_schema_type(
153 *_t.type_, _definitions, _already_known, _num_unnamed))}};
154
155 } else if constexpr (std::is_same<T, Type::Validated>()) {
156 // Avro knows no validation.
157 return type_to_avro_schema_type(*_t.type_, _definitions, _already_known,
158 _num_unnamed);
159
160 } else {
161 static_assert(rfl::always_false_v<T>, "Not all cases were covered.");
162 }
163 };
164
165 return rfl::visit(handle_variant, _type.variant_);
166}

168std::string to_json_representation(
169 const parsing::schema::Definition& _internal_schema) {
170 std::set<std::string> already_known;
171 size_t num_unnamed = 0;
172 const auto avro_schema = type_to_avro_schema_type(
173 _internal_schema.root_, _internal_schema.definitions_, &already_known,
174 &num_unnamed);
175 return rfl::json::write(avro_schema);
176}
