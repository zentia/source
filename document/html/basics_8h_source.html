172class pulse {
173 class clock_body {
174 size_t& ms;
175 int& reps;
176 signal_t val;
177 public:
178 clock_body(size_t& _ms, int& _reps) : ms(_ms), reps(_reps), val(low) {}
179 bool operator()(signal_t& out) {
180 rt_sleep((int)ms);
181 if (reps>0) --reps;
182 if (val==low) val = high;
183 else val = low;
184 out = val;
185 return reps>0 || reps == -1;
186 }
187 };
188 graph& my_graph;
189 size_t ms, init_ms;
190 int reps, init_reps;
191 input_node<signal_t> clock_node;
192
193public:
194 pulse(graph& g, size_t _ms=1000, int _reps=-1) :
195 my_graph(g), ms(_ms), init_ms(_ms), reps(_reps), init_reps(_reps),
196 clock_node(g, clock_body(ms, reps))
197 {}
198 pulse(const pulse& src) :
199 my_graph(src.my_graph), ms(src.init_ms), init_ms(src.init_ms),
200 reps(src.init_reps), init_reps(src.init_reps),
201 clock_node(src.my_graph, clock_body(ms, reps))
202 {}
203 ~pulse() {}
204 // Assignment changes the behavior of LHS to that of the RHS, but doesn't change owning graph
205 pulse& operator=(const pulse& src) {
206 ms = src.ms; init_ms = src.init_ms; reps = src.reps; init_reps = src.init_reps;
207 return *this;
208 }
209 input_node<signal_t>& get_out() { return clock_node; }
210 void activate() { clock_node.activate(); }
211 void reset() { reps = init_reps; }

303template <int N>
304class and_gate : public gate<N> {
305 using gate<N>::my_graph;
306 typedef typename gate<N>::ports_type ports_type;
307 typedef typename gate<N>::input_port_t::output_type from_input;
308 class and_body {
309 signal_t *ports;
310 signal_t state;
311 bool touched;
312 public:
313 and_body() : state(undefined), touched(false) {
314 ports = new signal_t[N];
315 for (int i=0; i<N; ++i) ports[i] = undefined;
316 }
317 void operator()(const from_input& v, ports_type& p) {
318 ports[v.tag()] = cast_to<signal_t>(v);
319 signal_t new_state=high;
320 size_t i=0;
321 while (i<N) {
322 if (ports[i] == low) { new_state = low; break; }
323 else if (ports[i] == undefined && new_state != low) { new_state = undefined; }
324 ++i;
325 }
326 if (!touched || state != new_state) {
327 state = new_state;
328 tbb::flow::get<0>(p).try_put(state);
329 touched = true;
330 }
331 }
332 };
333 public:
334 and_gate(graph& g) : gate<N>(g, and_body()) {}
335 and_gate(const and_gate<N>& src) : gate<N>(src.my_graph, and_body()) {}
336 ~and_gate() {}

506 typedef gate<4>::ports_type ports_type;

508 class digit_body {
509 signal_t ports[4];

523 if (ports[2] == high) new_state += 4;
524 if (ports[3] == high) new_state += 8;

530 }
531 }
532 };
533 string label;
534 unsigned int state;
535 bool report_changes;
536 public:
537 digit(graph& g, string l, bool rc=false) :

542 label(src.label), state(0), report_changes(src.report_changes) {}

545 // state is reset as in constructors
