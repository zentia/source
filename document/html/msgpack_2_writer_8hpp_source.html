95 OutputVarType new_value(const T& _var) const noexcept {
96 using Type = std::remove_cvref_t<T>;
97 if constexpr (std::is_same<Type, std::string>()) {
98 msgpack_pack_str(pk_, _var.size());
99 msgpack_pack_str_body(pk_, _var.c_str(), _var.size());
100 } else if constexpr (std::is_same<Type, rfl::Bytestring>()) {
101 msgpack_pack_bin(pk_, _var.size());
102 msgpack_pack_bin_body(pk_, _var.c_str(), _var.size());
103 } else if constexpr (std::is_same<Type, bool>()) {
104 if (_var) {
105 msgpack_pack_true(pk_);
106 } else {
107 msgpack_pack_false(pk_);
108 }
109 } else if constexpr (std::is_floating_point<Type>()) {
110 msgpack_pack_double(pk_, static_cast<double>(_var));
111 } else if constexpr (std::is_integral<Type>()) {
112 msgpack_pack_int64(pk_, static_cast<std::int64_t>(_var));
113 } else {
114 static_assert(rfl::always_false_v<T>, "Unsupported type.");
115 }
116 return OutputVarType{};
117 }
