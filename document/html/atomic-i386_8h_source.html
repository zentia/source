194static __inline__ unsigned long atomic_add_return(unsigned long i, atomic_t *v)
195{
196 int __i;
197#ifdef CONFIG_M386
198 unsigned long flags;
199 if(unlikely(boot_cpu_data.x86==3))
200 goto no_xadd;
201#endif
202 /* Modern 486+ processor */
203 __i = i;
204 __asm__ __volatile__(
205 LOCK_PREFIX "xaddl %0, %1;"
206 :"=r"(i)
207 :"m"(v->counter), "0"(i));
208 return i + __i;
209
210#ifdef CONFIG_M386
211no_xadd: /* Legacy 386 processor */
212 local_irq_save(flags);
213 __i = atomic_read(v);
214 atomic_set(v, i + __i);
215 local_irq_restore(flags);
216 return i + __i;
217#endif
218}

253static inline unsigned long atomic_add_unless(atomic_t *v, unsigned long a, unsigned long u)
254{
255 unsigned long c, old;
256 c = atomic_read(v);
257 for (;;) {
258 if (unlikely(c == (u)))
259 break;
260 old = atomic_cmpxchg((v), c, c + (a));
261 if (likely(old == c))
262 break;
263 c = old;
264 }
265 return c != (u);
266}
